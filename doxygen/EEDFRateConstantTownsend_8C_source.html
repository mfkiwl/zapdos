49EEDFRateConstantTownsend::EEDFRateConstantTownsend(const InputParameters & parameters)
50 : Material(parameters),
51 _townsend_coefficient(declareProperty<Real>("alpha" + getParam<std::string>("number") + "_" +
52 getParam<std::string>("reaction"))),
53 _d_alpha_d_en(declareProperty<Real>("d_alpha" + getParam<std::string>("number") + "_d_en_" +
54 getParam<std::string>("reaction"))),
55 _d_alpha_d_var_id(declareProperty<unsigned int>("d_alpha" + getParam<std::string>("number") +
56 "_d_var_id_" +
57 getParam<std::string>("reaction"))),
58 _em(coupledValue("electrons")),
59 _mean_en(coupledValue("mean_energy"))
60{
61 std::vector<Real> temp_x;
62 std::vector<Real> temp_y;
63 std::vector<Real> actual_mean_energy;
64 std::vector<Real> rate_coefficient;
65 std::string file_name =
66 getParam<std::string>("file_location") + "/" + getParam<FileName>("property_file");
67 MooseUtils::checkFileReadable(file_name);
68 const char * charPath = file_name.c_str();
69 std::ifstream myfile(charPath);
70 Real value;
71
72 if (myfile.is_open())
73 {
74 while (myfile >> value)
75 {
76 temp_x.push_back(value);
77 myfile >> value;
78 temp_y.push_back(value);
79
80 actual_mean_energy.push_back(1);
81 rate_coefficient.push_back(1);
82 }
83 myfile.close();
84 }
85 else
86 mooseError("Unable to open file");
87
88 // Ensure that arrays are sorted (should be done externally or by Bolsig+ wrapper; this is not
89 // permanent)
90 std::vector<size_t> idx(actual_mean_energy.size());
91 std::iota(idx.begin(), idx.end(), 0);
92 std::sort(
93 idx.begin(), idx.end(), [&temp_x](size_t i1, size_t i2) { return temp_x[i1] < temp_x[i2]; });
94 for (MooseIndex(idx) i = 0; i < idx.size(); ++i)
95 {
96 actual_mean_energy[i] = temp_x[idx[i]];
97 rate_coefficient[i] = temp_y[idx[i]];
98 }
99 _coefficient_interpolation.setData(actual_mean_energy, rate_coefficient);
100}

103EEDFRateConstantTownsend::computeQpProperties()
104{
105 Real actual_mean_energy = std::exp(_mean_en[_qp] - _em[_qp]);
106
107 _townsend_coefficient[_qp] = _coefficient_interpolation.sample(actual_mean_energy);
108
109 _d_alpha_d_en[_qp] = _coefficient_interpolation.sampleDerivative(actual_mean_energy);
110
111 if (_townsend_coefficient[_qp] < 0)
112 {
113 _townsend_coefficient[_qp] = 0;
114 _d_alpha_d_en[_qp] = 0;
115 }
116}
