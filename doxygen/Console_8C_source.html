203 const Action * common = actions.empty() ? nullptr : *actions.begin();
204
205 if (!parameters.isParamSetByUser("execute_on"))
206 {
207 // Honor the 'print_linear_residuals' option, only if 'linear' has not been set in 'execute_on'

214 _execute_on.push_back("nonlinear");
215 else
216 _execute_on.erase("nonlinear");
217 }
218
219 if (!_pars.isParamSetByUser("perf_log") && common && common->getParam<bool>("print_perf_log"))
220 {
221 _perf_log = true;
222 _solve_log = true;
223 }
224
225 // Append the common 'execute_on' to the setting for this object
226 // This is unique to the Console object, all other objects inherit from the common options
227 if (common)
228 {
229 const ExecFlagEnum & common_execute_on = common->getParam<ExecFlagEnum>("execute_on");
230 for (auto & mme : common_execute_on)
231 _execute_on.push_back(mme);
232 }
233
234 // If --show-outputs is used, enable it
235 if (_app.getParam<bool>("show_outputs"))
236 _system_info_flags.push_back("output");
237}

253void
254Console::initialSetup()
255{
256 // Only allow the main app to change the perf_log settings.
257 if (_app.name() == "main")
258 {
259 // Disable libMesh log
260 if (!_libmesh_log)
261 libMesh::perflog.disable_logging();
262 }
263
264 // system info flag can be changed only before console initial setup
265 _allow_changing_sysinfo_flag = false;
266
267 // If execute_on = 'initial' perform the output
268 if (wantOutput("system_information", EXEC_INITIAL))
269 outputSystemInformation();
270
271 // Call the base class method
272 TableOutput::initialSetup();
273
274 // If file output is desired, wipe out the existing file if not recovering
275 if (!_app.isRecovering())
276 writeStreamToFile(false);
277
278 // Enable verbose output if Executioner has it enabled
279 if (_app.getExecutioner()->isParamValid("verbose") &&
280 _app.getExecutioner()->getParam<bool>("verbose"))
281 _verbose = true;
282
283 // If the user adds "final" to the execute on, append this to the postprocessors, scalars, etc.,
284 // but only
285 // if the parameter (e.g., postprocessor_execute_on) has not been modified by the user.
286 if (_execute_on.contains("final"))
287 {
288 if (!_pars.isParamSetByUser("postprocessor_execute_on"))
289 _advanced_execute_on["postprocessors"].push_back("final");
290 if (!_pars.isParamSetByUser("scalars_execute_on"))
291 _advanced_execute_on["scalars"].push_back("final");
292 if (!_pars.isParamSetByUser("vector_postprocessor_execute_on"))
293 _advanced_execute_on["vector_postprocessors"].push_back("final");
294 }

309void
310Console::output()
311{
312 const auto & type = _current_execute_flag;
313
314 // Return if the current output is not on the desired interval
315 if (type != EXEC_FINAL && !onInterval())
316 return;
317
318 // Output the system information first; this forces this to be the first item to write by default
319 // However, 'output_system_information_on' still operates correctly, so it may be changed by the
320 // user
321 if (wantOutput("system_information", type) && !(type == EXEC_INITIAL))
322 outputSystemInformation();
323
324 // Write the input
325 if (wantOutput("input", type))
326 outputInput();
327
328 // Write the timestep information ("Time Step 0 ..."), this is controlled with "execute_on"
329 // We only write the initial and final here. All of the intermediate outputs will be written
330 // through timestepSetup.
331 if (type == EXEC_INITIAL && _execute_on.contains(EXEC_INITIAL))
332 writeTimestepInformation(/*output_dt = */ false);
333 else if (type == EXEC_FINAL && _execute_on.contains(EXEC_FINAL))
334 {
335 if (wantOutput("postprocessors", type) || wantOutput("scalars", type))
336 _console << "\nFINAL:\n";
337 }
338
339 // Print Non-linear Residual (control with "execute_on")
340 if (type == EXEC_NONLINEAR && _execute_on.contains(EXEC_NONLINEAR))
341 {
342 if (_nonlinear_iter == 0)
343 _old_nonlinear_norm = std::numeric_limits<Real>::max();
344
345 _console << std::right << std::setw(2) << _nonlinear_iter
346 << " Nonlinear |R| = " << outputNorm(_old_nonlinear_norm, _norm) << '\n';
347
348 _old_nonlinear_norm = _norm;
349 }
350
351 // Print Linear Residual (control with "execute_on")
352 else if (type == EXEC_LINEAR && _execute_on.contains(EXEC_LINEAR))
353 {
354 if (_linear_iter == 0)
355 _old_linear_norm = std::numeric_limits<Real>::max();
356
357 _console << std::right << std::setw(7) << _linear_iter
358 << " Linear |R| = " << outputNorm(_old_linear_norm, _norm) << '\n';
359
360 _old_linear_norm = _norm;
361 }
362
363 // Write variable norms
364 else if (type == EXEC_TIMESTEP_END)
365 {
366 if (_perf_log_interval && _t_step % _perf_log_interval == 0)
367 write(Moose::perf_log.get_perf_info(), false);
368 writeVariableNorms();
369 }
370
371 if (wantOutput("postprocessors", type))
372 outputPostprocessors();
373
374 if (wantOutput("scalars", type))
375 outputScalarVariables();
376
377 if (wantOutput("reporters", type))
378 outputReporters();
379
380 // Write the file
381 writeStreamToFile();
382
383 _console << std::flush;
384}

386void
387Console::writeStreamToFile(bool append)
388{
389 if (!_write_file)
390 return;
391
392 // Create the stream
393 std::ofstream output;
394
395 // Open the file
396 if (append)
397 output.open(filename().c_str(), std::ios::app | std::ios::out);
398 else
399 output.open(filename().c_str(), std::ios::trunc);
400
401 if (output.fail())
402 mooseError("Unable to open file ", filename());
403
404 std::string s = _file_output_stream.str();
405 // Write contents of file output stream and close the file
406 output << MooseUtils::removeColor(s);
407 output.close();
408
409 // Clear the file output stream
410 _file_output_stream.str("");
411}

413void
414Console::writeTimestepInformation(bool output_dt)
415{
416 // Stream to build the time step information
417 std::stringstream oss;
418
419 // Write timestep data for transient executioners
420 if (_transient)
421 {
422 // Write time step and time information
423 oss << "\nTime Step " << timeStep();
424 unsigned int time_step_digits = oss.str().length() - 11;
425
426 // Print the time
427 oss << ", time = " << formatTime(getOutputTime());
428
429 if (output_dt)
430 {
431 if (!_verbose)
432 // Show the time delta information
433 oss << ", dt = " << std::left << formatTime(dt());
434
435 // Show old time information, if desired on separate lines
436 else
437 {
438 unsigned int fillsize = 19 + time_step_digits;
439 oss << '\n'
440 << std::right << std::setw(fillsize) << std::setfill(' ') << "old time = " << std::left
441 << formatTime(timeOld()) << '\n';
442
443 // Show the time delta information
444 oss << std::right << std::setw(fillsize) << std::setfill(' ') << "dt = " << std::left
445 << formatTime(dt()) << '\n';
446
447 // Show the old time delta information, if desired
448 if (_verbose)
449 oss << std::right << std::setw(fillsize) << std::setfill(' ')
450 << "old dt = " << formatTime(_dt_old) << '\n';
451 }
452 }
453
454 oss << '\n';
455
456 // Output to the screen
457 _console << oss.str() << std::flush;
458 }

461std::string
462Console::formatTime(const Real t) const
463{
464 std::ostringstream oss;
465 if (_time_format != TimeFormatEnum::DTIME)
466 {
467 if (_precision > 0)
468 oss << std::setw(_precision) << std::setprecision(_precision) << std::setfill('0')
469 << std::showpoint;
470 if (_scientific_time)
471 oss << std::scientific;
472
473 if (_time_format == TimeFormatEnum::PLAIN)
474 oss << t;
475 else if (_time_format == TimeFormatEnum::SECOND)
476 oss << t << "s";
477 else if (_time_format == TimeFormatEnum::MINUTE)
478 oss << t / 60 << "m";
479 else if (_time_format == TimeFormatEnum::HOUR)
480 oss << t / 3600 << "h";
481 else if (_time_format == TimeFormatEnum::DAY)
482 oss << t / 86400 << "d";
483 }
484 else
485 {
486 Real abst = t;
487 if (t < 0)
488 {
489 oss << "-";
490 abst = -t;
491 }
492 int days = std::floor(abst / 24 / 3600);
493 int hours = std::floor(abst / 3600 - days * 24);
494 int mins = std::floor(abst / 60 - days * 24 * 60 - hours * 60);
495 Real second = abst - days * 24 * 3600 - hours * 3600 - mins * 60;
496
497 if (days != 0)
498 oss << days << "d";
499 if (hours != 0 || mins != 0 || second != 0)
500 {
501 if (days != 0)
502 oss << " ";
503 oss << std::setfill('0') << std::setw(2) << hours << ":" << std::setfill('0') << std::setw(2)
504 << mins << ":";
505
506 if (second < 10)
507 oss << "0";
508 if (_precision > 0)
509 oss << std::setw(_precision) << std::setprecision(_precision) << std::setfill('0')
510 << std::showpoint;
511 if (_scientific_time)
512 oss << std::scientific;
513 oss << second;
514 }
515 }
516 return oss.str();
517}

519void
520Console::writeVariableNorms()
521{
522 // If all_variable_norms is true, then so should outlier printing
523 if (_all_variable_norms)
524 _outlier_variable_norms = true;
525
526 // if we are not priting anything, let's not waste time computing the norms below and just exit
527 // this call
528 if ((_all_variable_norms == false) && (_outlier_variable_norms == false))
529 return;
530
531 // if it is an eigenvalue prolblem, we do not know to define RHS,
532 // and then we do not know how to compute variable norms
533 if (dynamic_cast<EigenProblem *>(_problem_ptr) != nullptr)
534 return;
535
536 // Flag set when header prints
537 bool header = false;
538
539 // String stream for variable norm information
540 std::ostringstream oss;
541
542 for (const auto i : make_range(_problem_ptr->numNonlinearSystems()))
543 {
544 // Get a references to the NonlinearSystem and libMesh system
545 NonlinearSystemBase & nl = _problem_ptr->getNonlinearSystemBase(i);
546 System & sys = nl.system();
547
548 // Storage for norm outputs
549 std::map<std::string, Real> other;
550 std::map<std::string, Real> outlier;
551
552 // Average norm
553 unsigned int n_vars = sys.n_vars();
554 Real avg_norm = (nl.nonlinearNorm() * nl.nonlinearNorm()) / n_vars;
555
556 // Compute the norms for each of the variables
557 for (unsigned int i = 0; i < n_vars; i++)
558 {
559 // Compute the norm and extract the variable name
560 Real var_norm = sys.calculate_norm(nl.RHS(), i, DISCRETE_L2);
561 var_norm *= var_norm; // use the norm squared
562 std::string var_name = sys.variable_name(i);
563
564 // Outlier if the variable norm is greater than twice (default) of the average norm
565 if (_outlier_variable_norms && (var_norm > _outlier_multiplier[1] * avg_norm))
566 {
567 // Print the header
568 if (!header)
569 {
570 oss << "\nOutlier Variable Residual Norms:\n";
571 header = true;
572 }
573
574 // Set the color, RED if the variable norm is 0.8 (default) of the total norm
575 std::string color = COLOR_YELLOW;
576 if (_outlier_variable_norms && (var_norm > _outlier_multiplier[0] * avg_norm * n_vars))
577 color = COLOR_RED;
578
579 // Display the residual
580 oss << " " << var_name << ": " << std::scientific << color << std::sqrt(var_norm)
581 << COLOR_DEFAULT << '\n';
582 }
583
584 // GREEN
585 else if (_all_variable_norms)
586 {
587 // Print the header if it doesn't already exist
588 if (!header)
589 {
590 oss << "\nVariable Residual Norms:\n";
591 header = true;
592 }
593 oss << " " << var_name << ": " << std::scientific << COLOR_GREEN << std::sqrt(var_norm)
594 << COLOR_DEFAULT << '\n';
595 }
596 }
597 }
598
599 // Update the output streams
600 _console << oss.str() << std::flush;
601}

603// Quick helper to output the norm in color
604std::string
605Console::outputNorm(const Real & old_norm, const Real & norm, const unsigned int precision)
606{
607 std::string color = COLOR_GREEN;
608
609 // Red if the residual went up... or if the norm is nan
610 if (norm != norm || norm > old_norm)
611 color = COLOR_RED;
612 // Yellow if change is less than 5%
613 else if ((old_norm - norm) / old_norm <= 0.05)
614 color = COLOR_YELLOW;
615
616 std::stringstream oss;
617 oss << std::scientific << std::setprecision(precision) << color << norm << COLOR_DEFAULT;
618
619 return oss.str();
620}

683void
684Console::outputSystemInformation()
685{
686 // skip system information output for sub-apps other than the zero-th of a MultiApp
687 // because they are using the same inputs and are most likely having the same information.
688 if (_app.multiAppNumber() > 0)
689 return;
690
691 if (_system_info_flags.contains("framework"))
692 _console << ConsoleUtils::outputFrameworkInformation(_app);
693
694 if (_system_info_flags.contains("mesh"))
695 _console << ConsoleUtils::outputMeshInformation(*_problem_ptr);
696
697 if (_system_info_flags.contains("nonlinear"))
698 {
699 for (const auto i : make_range(_problem_ptr->numNonlinearSystems()))
700 {

762void
763Console::write(std::string message, bool indent /*=true*/)
764{
765 // Do nothing if the message is empty, writing empty strings messes with multiapp indenting
766 if (message.empty())
767 return;
768
769 // Write the message to file
770 if (_write_file)
771 _file_output_stream << message;
772
773 // The empty case gets the right behavior, even though the boolean is technically wrong
774 bool this_message_ends_in_newline = message.empty() ? true : (message.back() == '\n');
775 bool this_message_starts_with_newline = message.empty() ? true : (message.front() == '\n');
776
777 // Apply MultiApp indenting
778 if ((this_message_starts_with_newline || _last_message_ended_in_newline) && indent &&
779 _app.multiAppLevel() > 0)
780 MooseUtils::indentMessage(_app.name(), message);
781
782 // Write message to the screen
783 if (_write_screen)
784 Moose::out << message;
785
786 _last_message_ended_in_newline = this_message_ends_in_newline;
787}
