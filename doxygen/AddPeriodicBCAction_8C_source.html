59void
60AddPeriodicBCAction::setPeriodicVars(PeriodicBoundaryBase & p,
61 const std::vector<VariableName> & var_names)
62{
63 NonlinearSystemBase & nl = _problem->getNonlinearSystemBase(/*nl_sys_num=*/0);
64 const std::vector<VariableName> * var_names_ptr;
65
66 // If var_names is empty - then apply this periodic condition to all variables in the system
67 if (var_names.empty())
68 var_names_ptr = &nl.getVariableNames();
69 else
70 var_names_ptr = &var_names;
71
72 for (const auto & var_name : *var_names_ptr)
73 {
74 if (!nl.hasScalarVariable(var_name))
75 {
76 unsigned int var_num = nl.getVariable(0, var_name).number();
77 p.set_variable(var_num);
78 _mesh->addPeriodicVariable(var_num, p.myboundary, p.pairedboundary);
79 }
80 }
81}

83bool
84AddPeriodicBCAction::autoTranslationBoundaries()
85{
86 auto displaced_problem = _problem->getDisplacedProblem();
87
88 if (isParamValid("auto_direction"))
89 {
90 // If we are working with a parallel mesh then we're going to ghost all the boundaries
91 // everywhere because we don't know what we need...
92 if (_mesh->isDistributedMesh())
93 {
94 bool is_orthogonal_mesh = _mesh->detectOrthogonalDimRanges();
95
96 // If we can't detect the orthogonal dimension ranges for this
97 // Mesh, then auto_direction periodicity isn't going to work.
98 if (!is_orthogonal_mesh)
99 mooseError("Could not detect orthogonal dimension ranges for DistributedMesh.");
100 }
101
102 std::vector<std::string> auto_dirs = getParam<std::vector<std::string>>("auto_direction");
103
104 int dim_offset = _mesh->dimension() - 2;
105 for (const auto & dir : auto_dirs)
106 {
107 int component = -1;
108 if (dir == "X" || dir == "x")
109 component = 0;
110 else if (dir == "Y" || dir == "y")
111 {
112 if (dim_offset < 0)
113 mooseError("Cannot wrap 'Y' direction when using a 1D mesh");
114 component = 1;
115 }
116 else if (dir == "Z" || dir == "z")
117 {
118 if (dim_offset <= 0)
119 mooseError("Cannot wrap 'Z' direction when using a 1D or 2D mesh");
120 component = 2;
121 }
122
123 if (component >= 0)
124 {
125 const std::pair<BoundaryID, BoundaryID> * boundary_ids =
126 _mesh->getPairedBoundaryMapping(component);
127 RealVectorValue v;
128 v(component) = _mesh->dimensionWidth(component);
129 PeriodicBoundary p(v);
130
131 if (boundary_ids == NULL)
132 mooseError(
133 "Couldn't auto-detect a paired boundary for use with periodic boundary conditions");
134
135 p.myboundary = boundary_ids->first;
136 p.pairedboundary = boundary_ids->second;
137 setPeriodicVars(p, getParam<std::vector<VariableName>>("variable"));
138 auto & eq = _problem->es();
139 for (const auto i : make_range(eq.n_systems()))
140 eq.get_system(i).get_dof_map().add_periodic_boundary(p);
141 if (displaced_problem)
142 {
143 auto & deq = displaced_problem->es();
144 for (const auto i : make_range(deq.n_systems()))
145 deq.get_system(i).get_dof_map().add_periodic_boundary(p);
146 }
147 }
148 }
149 return true;
150 }
151 return false;
152}

154void
155AddPeriodicBCAction::act()
156{
157 if (_current_task == "add_geometric_rm")
158 // Tell the mesh to hold off on deleting remote elements because we need to wait for our
159 // periodic boundaries to be added
160 Action::_mesh->allowRemoteElementRemoval(false);
161
162 if (_current_task == "add_algebraic_rm")
163 {
164 auto rm_params = _factory.getValidParams("ElementSideNeighborLayers");
165
166 rm_params.set<std::string>("for_whom") = "PeriodicBCs";
167 if (!_mesh)
168 mooseError("We should have added periodic boundaries and consequently we should have set the "
169 "_mesh by now");
170
171 rm_params.set<MooseMesh *>("mesh") = _mesh;
172 // The default GhostPointNeighbors ghosting functor in libMesh handles the geometric ghosting of
173 // periodic boundaries for us, so we only need to handle the algebraic ghosting here
174 rm_params.set<Moose::RelationshipManagerType>("rm_type") =
175 Moose::RelationshipManagerType::ALGEBRAIC;
176
177 if (rm_params.areAllRequiredParamsValid())
178 {
179 auto rm_obj = _factory.create<RelationshipManager>(
180 "ElementSideNeighborLayers", "periodic_bc_ghosting_" + name(), rm_params);
181
182 if (!_app.addRelationshipManager(rm_obj))
183 _factory.releaseSharedObjects(*rm_obj);
184 }
185 else
186 mooseError("Invalid initialization of ElementSideNeighborLayers");
187 }
188
189 if (_current_task == "add_periodic_bc")
190 {
191 auto & nl = _problem->getNonlinearSystemBase(/*nl_sys_num=*/0);
192 _mesh = &_problem->mesh();
193 auto displaced_problem = _problem->getDisplacedProblem();
194
195 if (!autoTranslationBoundaries())
196 {
197 if (_pars.isParamValid("translation"))
198 {
199 RealVectorValue translation = getParam<RealVectorValue>("translation");
200
201 PeriodicBoundary p(translation);
202 p.myboundary = _mesh->getBoundaryID(getParam<BoundaryName>("primary"));
203 p.pairedboundary = _mesh->getBoundaryID(getParam<BoundaryName>("secondary"));
204 setPeriodicVars(p, getParam<std::vector<VariableName>>("variable"));
205
206 auto & eq = _problem->es();
207 for (const auto i : make_range(eq.n_systems()))
208 eq.get_system(i).get_dof_map().add_periodic_boundary(p);
209 if (displaced_problem)
210 {
211 auto & deq = displaced_problem->es();
212 for (const auto i : make_range(deq.n_systems()))
213 deq.get_system(i).get_dof_map().add_periodic_boundary(p);
214 }
215 }
216 else if (getParam<std::vector<std::string>>("transform_func") != std::vector<std::string>())
217 {
218 std::vector<std::string> inv_fn_names =
219 getParam<std::vector<std::string>>("inv_transform_func");
220 std::vector<std::string> fn_names = getParam<std::vector<std::string>>("transform_func");
221
222 // If the user provided a forward transformation, they must also provide an inverse -- we
223 // can't form the inverse of an arbitrary function automatically...
224 if (inv_fn_names == std::vector<std::string>())
225 mooseError("You must provide an inv_transform_func for FunctionPeriodicBoundary!");
226
227 FunctionPeriodicBoundary pb(*_problem, fn_names);
228 pb.myboundary = _mesh->getBoundaryID(getParam<BoundaryName>("primary"));
229 pb.pairedboundary = _mesh->getBoundaryID(getParam<BoundaryName>("secondary"));
230 setPeriodicVars(pb, getParam<std::vector<VariableName>>("variable"));
231
232 FunctionPeriodicBoundary ipb(*_problem, inv_fn_names);
233 ipb.myboundary =
234 _mesh->getBoundaryID(getParam<BoundaryName>("secondary")); // these are swapped
235 ipb.pairedboundary =
236 _mesh->getBoundaryID(getParam<BoundaryName>("primary")); // these are swapped
237 setPeriodicVars(ipb, getParam<std::vector<VariableName>>("variable"));
238
239 // Add the pair of periodic boundaries to the dof map
240 auto & eq = _problem->es();
241 for (const auto i : make_range(eq.n_systems()))
242 eq.get_system(i).get_dof_map().add_periodic_boundary(pb, ipb);
243 if (displaced_problem)
244 {
245 auto & deq = displaced_problem->es();
246 for (const auto i : make_range(deq.n_systems()))
247 deq.get_system(i).get_dof_map().add_periodic_boundary(pb, ipb);
248 }
249 }
250 else
251 {
252 mooseError(
253 "You have to specify either 'auto_direction', 'translation' or 'trans_func' in your "
254 "period boundary section '" +
255 _name + "'");
256 }
257 }
258
259 // Now make sure that the mesh default ghosting functor has its periodic bcs set
260 _mesh->getMesh().default_ghosting().set_periodic_boundaries(
261 nl.dofMap().get_periodic_boundaries());
262 if (displaced_problem)
263 displaced_problem->mesh().getMesh().default_ghosting().set_periodic_boundaries(
264 displaced_problem->nlSys(/*nl_sys_num=*/0).dofMap().get_periodic_boundaries());
265 }
266}
