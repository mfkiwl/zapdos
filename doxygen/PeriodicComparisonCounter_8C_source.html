66void
67PeriodicComparisonCounter::finalize()
68{
69 Real remainder = std::remainder(_feproblem.time(), _period);
70 Real max = std::numeric_limits<double>::epsilon();
71 Real min = -std::numeric_limits<double>::epsilon();
72
73 if ((remainder >= min) && (remainder <= max))
74 {
75 _period_counter++;
76
77 switch (_comparison_type)
78 {
79 case ComparisonType::EQUALS:
80 if (_value1 == _value2)
81 _counter++;
82 else
83 _counter = 0.0;
84 break;
85 case ComparisonType::GREATER_THAN_EQUALS:
86 if (_value1 >= _value2)
87 _counter++;
88 else
89 _counter = 0.0;
90 break;
91 case ComparisonType::LESS_THAN_EQUALS:
92 if (_value1 <= _value2)
93 _counter++;
94 else
95 _counter = 0.0;
96 break;
97 case ComparisonType::GREATER_THAN:
98 if (_value1 > _value2)
99 _counter++;
100 else
101 _counter = 0.0;
102 break;
103 case ComparisonType::LESS_THAN:
104 if (_value1 < _value2)
105 _counter++;
106 else
107 _counter = 0.0;
108 break;
109 default:
110 mooseError("Invalid comparison type.");
111 }
112 }
