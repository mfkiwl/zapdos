43bool
44SolutionInvalidity::solutionInvalid() const
45{
46 libmesh_parallel_only(comm());
47
48 bool is_invalid = false;
49 for (auto & entry : _counts)
50 {
51 if (entry.counts)
52 {
53 is_invalid = true;
54 break;
55 }
56 }
57
58 comm().max(is_invalid);
59 return is_invalid > 0;

102void
103SolutionInvalidity::sync()
104{
105 std::map<processor_id_type,
106 std::vector<std::tuple<std::string, std::string, unsigned int, unsigned int>>>
107 data_to_send;
108
109 if (processor_id() != 0)
110 {
111 for (const auto id : index_range(_counts))
112 {
113 const auto & entry = _counts[id];
114 if (entry.counts)
115 {
116 const auto & info = _solution_invalidity_registry.item(id);
117 data_to_send[0].emplace_back(
118 info.object_type, info.message, entry.counts, entry.timestep_counts);
119 _counts[id].total_counts = 0;
120 }
121 }
122 }
123
124 const auto receive_data = [this](const processor_id_type libmesh_dbg_var(pid), const auto & data)
125 {
126 mooseAssert(pid != 0, "Should not be used except processor 0");
127
128 for (const auto & [object_type, message, counts, timestep_counts] : data)
129 {
130 InvalidSolutionID main_id = 0;
131 const moose::internal::SolutionInvalidityName name(object_type, message);
132 if (_solution_invalidity_registry.keyExists(name))
133 main_id = _solution_invalidity_registry.id(name);
134 else
135 main_id = moose::internal::getSolutionInvalidityRegistry().registerInvalidity(object_type,
136 message);
137 if (_counts.size() <= main_id)
138 _counts.resize(main_id + 1);
139
140 _counts[main_id].counts += counts;
141 _counts[main_id].timestep_counts += timestep_counts;
142 _counts[main_id].total_counts += timestep_counts;
143 }
144 };
145
146 TIMPI::push_parallel_vector_data(comm(), data_to_send, receive_data);
147}

156SolutionInvalidity::FullTable
157SolutionInvalidity::summaryTable() const
158{
159 FullTable vtable({"Object", "Converged", "Timestep", "Total", "Message"}, 4);
160
161 vtable.setColumnFormat({
162 VariadicTableColumnFormat::AUTO, // Object Type
163 VariadicTableColumnFormat::AUTO, // Converged Iteration Warnings
164 VariadicTableColumnFormat::AUTO, // Latest Time Step Warnings
165 VariadicTableColumnFormat::AUTO, // Total Simulation Warnings
166 VariadicTableColumnFormat::AUTO, // Message
167 });
168
169 vtable.setColumnPrecision({
170 1, // Object Name
171 0, // Converged Iteration Warnings
172 0, // Latest Time Step Warnings
173 0, // Total Simulation Warnings
174 1, // Message
175 });
176
177 if (processor_id() == 0)
178 {
179 for (const auto id : index_range(_counts))
180 {
181 const auto & entry = _counts[id];
182 if (entry.counts > 0)
183 {
184 const auto & info = _solution_invalidity_registry.item(id);
185 vtable.addRow(info.object_type, // Object Type
186 entry.counts, // Converged Iteration Warnings
187 entry.timestep_counts, // Latest Time Step Warnings
188 entry.total_counts, // Total Iteration Warnings
189 info.message // Message
190 );
191 }
192 }
193 }
194
195 return vtable;
196}

198void
199dataStore(std::ostream & stream, SolutionInvalidity & solution_invalidity, void * context)
200{
201 if (solution_invalidity.processor_id() != 0)
202 return;
203
204 // Build data structure for store
205 auto size = solution_invalidity._counts.size();
206 dataStore(stream, size, context);
207
208 for (const auto id : index_range(solution_invalidity._counts))
209 {
210 const auto & entry = solution_invalidity._counts[id];
211 const auto & info = solution_invalidity._solution_invalidity_registry.item(id);
212 std::string type = info.object_type;
213 std::string message = info.message;
214 dataStore(stream, type, context);
215 dataStore(stream, message, context);
216 dataStore(stream, entry.counts, context);
217 dataStore(stream, entry.timestep_counts, context);
218 dataStore(stream, entry.total_counts, context);
219 }
220}

222void
223dataLoad(std::istream & stream, SolutionInvalidity & solution_invalidity, void * context)
224{
225 if (solution_invalidity.processor_id() != 0)
226 return;
227
228 std::size_t num_counts;
229 // load data block size
230 dataLoad(stream, num_counts, context);
231
232 std::string object_type, message;
233 InvalidSolutionID id;
234
235 // loop over and load stored data
236 for (size_t i = 0; i < num_counts; i++)
237 {
238 dataLoad(stream, object_type, context);
239 dataLoad(stream, message, context);
240
241 const moose::internal::SolutionInvalidityName name(object_type, message);
242 if (solution_invalidity._solution_invalidity_registry.keyExists(name))
243 id = solution_invalidity._solution_invalidity_registry.id(name);
244 else
245 id =
246 moose::internal::getSolutionInvalidityRegistry().registerInvalidity(object_type, message);
247
248 if (solution_invalidity._counts.size() <= id)
249 solution_invalidity._counts.resize(id + 1);
250
251 const auto & entry = solution_invalidity._counts[id];
252 dataLoad(stream, entry.counts, context);
253 dataLoad(stream, entry.timestep_counts, context);
254 dataLoad(stream, entry.total_counts, context);
255 }
