28InputParameters
29PatchMeshGenerator::validParams()
30{
31 InputParameters params = MeshGenerator::validParams();
32 MooseEnum dims("2=2 3", "2");
33 params.addParam<MooseEnum>("dim",
34 dims,
35 "The dimension of the mesh to be generated. Patch meshes are only "
36 "valid in 2 or 3 dimensions.");
37
38 MooseEnum elem_types("QUAD4 QUAD8 HEX8 HEX20", "QUAD4");
39 params.addParam<MooseEnum>("elem_type",
40 elem_types,
41 "The type of element from libMesh to "
42 "generate (default: linear element for "
43 "requested dimension)");
44 params.addParam<Real>("x_length", 0.24, "Length of the domain in the x direction.");
45 params.addParam<Real>("y_length", 0.12, "Length of the domain in the y direction.");
46 params.addParam<Real>("z_length", 0.0, "Length of the domain in the z direction.");
47 params.addParam<Real>("x_offset", 0.0, "Offset of the Cartesian origin in the x direction.");
48 params.addParam<Real>("y_offset", 0.0, "Offset of the Cartesian origin in the y direction.");
49 params.addParam<Real>("z_offset", 0.0, "Offset of the Cartesian origin in the z direction.");
50 params.addClassDescription("Creates 2D or 3D patch meshes.");
51 return params;

54PatchMeshGenerator::PatchMeshGenerator(const InputParameters & parameters)
55 : MeshGenerator(parameters),
56 _dim(getParam<MooseEnum>("dim")),
57 _elem_type(getParam<MooseEnum>("elem_type")),
58 _xlength(getParam<Real>("x_length")),
59 _ylength(getParam<Real>("y_length")),
60 _zlength(getParam<Real>("z_length")),
61 _xoffset(getParam<Real>("x_offset")),
62 _yoffset(getParam<Real>("y_offset")),
63 _zoffset(getParam<Real>("z_offset"))
64{
65 if (_xlength <= 0.0)
66 paramError("x_length", "Must be greater than zero");
67
68 if (_ylength <= 0.0)
69 paramError("y_length", "Must be greater than zero");
70
71 if (_dim == 2 && (_elem_type == "HEX8" || _elem_type == "HEX20"))
72 paramError("elem_type", "Must be QUAD4 or QUAD8 for 2-dimensional meshes.");
73
74 if (_dim == 2 && _zlength != 0.0)
75 paramError("z_length", "Must be zero for 2-dimensional meshes");
76
77 if (_dim == 3 && !parameters.isParamSetByUser("elem_type"))
78 _elem_type = "HEX8";
79
80 if (_dim == 3 && (_elem_type == "QUAD4" || _elem_type == "QUAD8"))
81 paramError("elem_type", "Must be HEX8 or HEX20 for 3-dimensional meshes.");
82
83 // If domain dimensions are not set by user for 3-dimensions a unit cube is used
84 // as default as per MacNeal and Harder (1985)
85 if (_dim == 3)
86 {
87 if (!parameters.isParamSetByUser("x_length"))
88 _xlength = 1.0;
89 if (!parameters.isParamSetByUser("y_length"))
90 _ylength = 1.0;
91 if (!parameters.isParamSetByUser("z_length"))
92 _zlength = 1.0;
93 }
94
95 if (_dim == 3 && _zlength <= 0.0)
96 paramError("z_length", "Must be greater than zero for 3-dimensional meshes");
97}

99std::unique_ptr<MeshBase>
100PatchMeshGenerator::generate()
101{
102 auto mesh = buildMeshBaseObject();
103 BoundaryInfo & boundary_info = mesh->get_boundary_info();
104
105 unsigned num_nodes = 0;
106 if (_elem_type == "QUAD4")
107 num_nodes = 8;
108 else if (_elem_type == "QUAD8")
109 num_nodes = 20;
110 else if (_elem_type == "HEX8")
111 num_nodes = 16;
112 else if (_elem_type == "HEX20")
113 num_nodes = 48;
114
115 std::vector<Node *> nodes(num_nodes);
116
117 const Real xmax = _xoffset + _xlength;
118 const Real ymax = _yoffset + _ylength;
119 const Real zmax = _zoffset + _zlength;
120
121 if (_dim == 2)
122 {
123 const std::vector<Real> x_interior_fractions{1.0 / 6.0, 3.0 / 4.0, 2.0 / 3.0, 1.0 / 3.0};
124 const std::vector<Real> y_interior_fractions{1.0 / 6.0, 1.0 / 4.0, 2.0 / 3.0, 2.0 / 3.0};
125
126 // Exterior node positions
127 std::vector<Point> node_positions{{_xoffset, _yoffset, _zoffset},
128 {xmax, _yoffset, _zoffset},
129 {xmax, ymax, _zoffset},
130 {_xoffset, ymax, _zoffset}};
131 // Interior node positions
132 for (unsigned i = 0; i < x_interior_fractions.size(); ++i)
133 node_positions.push_back({_xoffset + x_interior_fractions[i] * _xlength,
134 _yoffset + y_interior_fractions[i] * _ylength,
135 _zoffset});
136
137 if (_elem_type == "QUAD8")
138 {
139 // Exterior midside node positions
140 for (unsigned i = 0; i < 4; ++i)
141 node_positions.push_back(0.5 * (node_positions[i] + node_positions[(i + 1) % 4]));
142
143 // Exterior to interior midside node positions
144 for (unsigned i = 0; i < 4; ++i)
145 node_positions.push_back(0.5 * (node_positions[i] + node_positions[i + 4]));
146
147 // Interior midside node positions
148 for (unsigned i = 4; i < 8; ++i)
149 node_positions.push_back(0.5 * (node_positions[i] + node_positions[(i + 1) % 4 + 4]));
150 }
151
152 for (unsigned node_id = 0; node_id < num_nodes; ++node_id)
153 nodes[node_id] = mesh->add_point(node_positions[node_id], node_id);
154
155 if (_elem_type == "QUAD4")
156 makeQuad4Elems(*mesh, nodes);
157 else
158 makeQuad8Elems(*mesh, nodes);
159
160 boundary_info.sideset_name(1) = "bottom";
161 boundary_info.sideset_name(2) = "right";
162 boundary_info.sideset_name(3) = "top";
163 boundary_info.sideset_name(4) = "left";
164
165 boundary_info.nodeset_name(100) = "bottom_left";
166 boundary_info.nodeset_name(101) = "bottom_right";
167 boundary_info.nodeset_name(102) = "top_right";
168 boundary_info.nodeset_name(103) = "top_left";
169 }
170 else
171 {
172 const std::vector<Real> x_interior_fractions{
173 0.249, 0.826, 0.850, 0.273, 0.320, 0.677, 0.788, 0.165};
174 const std::vector<Real> y_interior_fractions{
175 0.342, 0.288, 0.649, 0.750, 0.186, 0.305, 0.693, 0.745};
176 const std::vector<Real> z_interior_fractions{
177 0.192, 0.288, 0.263, 0.230, 0.643, 0.683, 0.644, 0.702};
178
179 // Exterior node positions
180 std::vector<Point> node_positions{{_xoffset, _yoffset, _zoffset},
181 {xmax, _yoffset, _zoffset},
182 {xmax, ymax, _zoffset},
183 {_xoffset, ymax, _zoffset},
184 {_xoffset, _yoffset, zmax},
185 {xmax, _yoffset, zmax},
186 {xmax, ymax, zmax},
187 {_xoffset, ymax, zmax}};
188
189 // Interior node positions
190 for (unsigned i = 0; i < x_interior_fractions.size(); ++i)
191 node_positions.push_back({_xoffset + x_interior_fractions[i] * _xlength,
192 _yoffset + y_interior_fractions[i] * _ylength,
193 _zoffset + z_interior_fractions[i] * _zlength});
194
195 if (_elem_type == "HEX20")
196 {
197 // Midside Nodes
198
199 // Four on back face
200 for (unsigned i = 0; i < 4; ++i)
201 node_positions.push_back(0.5 * (node_positions[i] + node_positions[(i + 1) % 4]));
202
203 // Four on front face
204 for (unsigned i = 4; i < 8; ++i)
205 node_positions.push_back(0.5 * (node_positions[i] + node_positions[(i + 1) % 4 + 4]));
206
207 // Four on remaining exterior edges
208 for (unsigned i = 0; i < 4; ++i)
209 node_positions.push_back(0.5 * (node_positions[i] + node_positions[i + 4]));
210
211 // Four on interior hex back face
212 for (unsigned i = 8; i < 12; ++i)
213 node_positions.push_back(0.5 * (node_positions[i] + node_positions[(i + 1) % 4 + 8]));
214
215 // Four on interior hex front face
216 for (unsigned i = 12; i < 16; ++i)
217 node_positions.push_back(0.5 * (node_positions[i] + node_positions[(i + 1) % 4 + 12]));
218
219 // Four on remaining interior hex edges
220 for (unsigned i = 8; i < 12; ++i)
221 node_positions.push_back(0.5 * (node_positions[i] + node_positions[i + 4]));
222
223 // Eight on remaining interior edges
224 for (unsigned i = 0; i < 8; ++i)
225 node_positions.push_back(0.5 * (node_positions[i] + node_positions[i + 8]));
226 }
227
228 for (unsigned node_id = 0; node_id < num_nodes; ++node_id)
229 nodes[node_id] = mesh->add_point(node_positions[node_id], node_id);
230
231 if (_elem_type == "HEX8")
232 makeHex8Elems(*mesh, nodes);
233 else
234 makeHex20Elems(*mesh, nodes);
235
236 boundary_info.sideset_name(1) = "front";
237 boundary_info.sideset_name(2) = "bottom";
238 boundary_info.sideset_name(3) = "left";
239 boundary_info.sideset_name(4) = "right";
240 boundary_info.sideset_name(5) = "top";
241 boundary_info.sideset_name(6) = "back";
242
243 boundary_info.nodeset_name(100) = "bottom_back_left";
244 boundary_info.nodeset_name(101) = "bottom_back_right";
245 boundary_info.nodeset_name(102) = "top_back_right";
246 boundary_info.nodeset_name(103) = "top_back_left";
247 boundary_info.nodeset_name(104) = "bottom_front_left";
248 boundary_info.nodeset_name(105) = "bottom_front_right";
249 boundary_info.nodeset_name(106) = "top_front_right";
250 boundary_info.nodeset_name(107) = "top_front_left";
251 }
252
253 mesh->prepare_for_use();
254
255 return dynamic_pointer_cast<MeshBase>(mesh);
256}

258void
259PatchMeshGenerator::makeQuad4Elems(MeshBase & mesh, const std::vector<Node *> & nodes)
260{
261 BoundaryInfo & boundary_info = mesh.get_boundary_info();
262 std::vector<std::vector<int>> element_connectivity{
263 {0, 1, 5, 4}, {1, 2, 6, 5}, {2, 3, 7, 6}, {3, 0, 4, 7}, {4, 5, 6, 7}};
264 for (unsigned i = 0; i < 5; ++i)
265 {
266 Elem * elem = mesh.add_elem(new Quad4);
267 for (unsigned j = 0; j < 4; ++j)
268 elem->set_node(j) = nodes[element_connectivity[i][j]];
269
270 elem->subdomain_id() = i + 1;
271
272 if (i < 4)
273 {
274 boundary_info.add_node(i, i + 100);
275 boundary_info.add_side(elem, 0, i + 1);
276 }
277 }
278}
