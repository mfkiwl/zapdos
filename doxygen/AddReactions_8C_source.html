95void
96AddReactions::act()
97{
98 std::vector<int> other_index;
99 std::vector<int> reactant_indices;
100 std::vector<std::string> other_variables;
101 other_variables.resize(3);
102 other_variables[0] = "v";
103 other_variables[1] = "w";
104 other_variables[2] = "x";
105 std::vector<bool> include_species;
106 std::string product_kernel_name;
107 std::string reactant_kernel_name;
108 std::string energy_kernel_name;
109 // std::vector<NonlinearVariableName> variables =
110 // getParam<std::vector<NonlinearVariableName>>("species");
111 // std::vector<VariableName> electron_energy =
112 // getParam<std::vector<VariableName>>("electron_energy");
113 // std::string electron_density = getParam<std::string>("electron_density");
114
115 if (_current_task == "add_aux_variable")
116 {
117 for (unsigned int i = 0; i < _num_reactions; ++i)
118 {
119 auto var_params = _factory.getValidParams("MooseVariableConstMonomial");
120 //_problem->addAuxVariable(_aux_var_name[i], FIRST);
121 _problem->addAuxVariable("MooseVariableConstMonomial", _aux_var_name[i], var_params);
122 }
123 }
124
125 if (_current_task == "add_aux_kernel")
126 {
127 // This boolean is only for EEDF reactions. It is always set to false for every other reaction
128 // type.
129 if (_track_rates)
130 {
131 std::string kernel_name;
132
133 // For AuxKernels function and constant-based reactions are identical.
134 // They are added separately just by convention.
135 // getKernelName(_num_reactants, is_energy, is_aux)
136 for (unsigned int i = 0; i < _num_function_reactions; ++i)
137 {
138 kernel_name = getReactionKernelName(_reactants[_function_reaction_number[i]].size(), true);
139 addAuxRate(kernel_name, _function_reaction_number[i]);
140 }
141
142 for (unsigned int i = 0; i < _num_constant_reactions; ++i)
143 {
144 kernel_name =
145 getReactionKernelName(_reactants[_constant_reaction_number[i]].size(), true);
146 addAuxRate(kernel_name, _constant_reaction_number[i]);
147 }
148 }
149 }
150
151 if (_current_task == "add_material")
152 {
153 for (unsigned int i = 0; i < _num_reactions; ++i)
154 {
155 if (_rate_type[i] == "Constant")
156 {
157 addConstantCoefficient(i);
158 }
159 else if (_rate_type[i] == "Equation")
160 {
161 /*
162 * THIS WILL NEED TO BE MODIFIED.
163 * ADD COUPLED NONLINEAR VARIABLES THAT NEED
164 * DERIVATIVES. NO AD CAPABILITY YET SO THIS
165 * WILL NEED TO BE DONE MANUALLY.
166 */
167 addFunctionCoefficient(i);
168 }
169 else if (_superelastic_reaction[i] == true)
170 {
171 mooseError("Superelastic reactions not supported yet!");
172 // addSuperelasticCoefficient(i);
173 }
174 }
175 }
176
177 // Add appropriate kernels to each reactant and product.
178 if (_current_task == "add_kernel")
179 {
180 // Initialize the kernel name
181 std::string kernel_name;
182
183 /*
184 *
185 * FUNCTION REACTIONS
186 *
187 * (Note that functions will be added as normal kernels, not AD.
188 * No AD functionality exists for parsed materials.)
189 */
190 for (unsigned int i = 0; i < _num_function_reactions; ++i)
191 {
192 for (unsigned int j = 0; j < _species.size(); ++j)
193 {
194 kernel_name = getReactionKernelName(_reactants[_function_reaction_number[i]].size(), false);
195 if (_species_count[_function_reaction_number[i]][j] != 0)
196 {
197 addFunctionReaction(_function_reaction_number[i], j, kernel_name);
198 }
199 else
200 continue;
201 }
202 }
203
204 /*
205 *
206 * CONSTANT REACTIONS
207 *
208 */
209 for (unsigned int i = 0; i < _num_constant_reactions; ++i)
210 {
211 for (unsigned int j = 0; j < _species.size(); ++j)
212 {
213 kernel_name = getReactionKernelName(_reactants[_constant_reaction_number[i]].size(), false);
214 if (_species_count[_constant_reaction_number[i]][j] != 0)
215 {
216 addConstantReaction(_constant_reaction_number[i], j, kernel_name);
217 }
218 else
219 continue;
220 }
221 }
222 }

260void
261AddReactions::addConstantReaction(const unsigned & reaction_num,
262 const unsigned & species_num,
263 const std::string & kernel_name)
264{
265 std::string kernel_identifier;
266
267 InputParameters params = _factory.getValidParams(kernel_name);
268 params.set<std::string>("reaction") = _reaction[reaction_num];
269 params.set<std::string>("number") = Moose::stringify(reaction_num);
270 params.set<std::vector<SubdomainName>>("block") = getParam<std::vector<SubdomainName>>("block");
271
272 params.set<NonlinearVariableName>("variable") = _species[species_num];
273 for (unsigned int k = 0; k < _reactants[reaction_num].size(); ++k)
274 {
275 params.set<std::vector<VariableName>>(_reactant_names[k]) = {_reactants[reaction_num][k]};
276 if (_species[species_num] == _reactants[reaction_num][k])
277 {
278 params.set<bool>("_" + _reactant_names[k] + "_eq_u") = true;
279 }
280 }
281 params.set<Real>("coefficient") = _species_count[reaction_num][species_num];
282 kernel_identifier = "kernel_constant_" + getParam<std::vector<SubdomainName>>("block")[0] +
283 std::to_string(reaction_num) + "_" + std::to_string(species_num);
284
285 params.set<std::vector<SubdomainName>>("block") = getParam<std::vector<SubdomainName>>("block");
286 _problem->addKernel(kernel_name, kernel_identifier + "_" + _name, params);
287}

289void
290AddReactions::addFunctionReaction(const unsigned & reaction_num,
291 const unsigned & species_num,
292 const std::string & kernel_name)
293{
294 // Source and sink terms that are dictated by a parsed function rate coefficient are little tricky
295 // to handle, so they require a separate kernel function. As of the writing of this code no
296 // automatic differentiation capabilities exist for parsed materials, so the jacobians need to be
297 // included manually. Luckily parsed rate coefficients almost always only exist in terms of
298 // electron or gas temperature, but in its current form this still requires computing the jacobian
299 // of the parsed function with respect to AT LEAST electrons and _mean_en.
300 std::string kernel_identifier;
301
302 InputParameters params = _factory.getValidParams(kernel_name);
303 params.set<std::string>("reaction") = _reaction[reaction_num];
304 params.set<std::string>("number") = Moose::stringify(reaction_num);
305 params.set<std::vector<SubdomainName>>("block") = getParam<std::vector<SubdomainName>>("block");
306
307 // define the coupled variables in the function to compute jacobian contribution
308 // params.set<std::vector<VariableName>>("args") =
309 // getParam<std::vector<VariableName>>("equation_variables");
310
311 // TO DO: (1) INCLUDE VARIABLES FOR JACOBIAN CONTRIBUTION OF RATE COEFFICIENT
312 // (2) ALLOW FOR BOTH ELECTRON AND GAS TEMPERATURE
313
314 params.set<NonlinearVariableName>("variable") = _species[species_num];
315 for (unsigned int k = 0; k < _reactants[reaction_num].size(); ++k)
316 {
317 params.set<std::vector<VariableName>>(_reactant_names[k]) = {_reactants[reaction_num][k]};
318 if (_species[species_num] == _reactants[reaction_num][k])
319 {
320 params.set<bool>("_" + _reactant_names[k] + "_eq_u") = true;
321 }
322 }
323 params.set<Real>("coefficient") = _species_count[reaction_num][species_num];
324 kernel_identifier = "kernel_function_" + getParam<std::vector<SubdomainName>>("block")[0] +
325 std::to_string(reaction_num) + "_" + std::to_string(species_num);
326
327 _problem->addKernel(kernel_name, kernel_identifier + "_" + _name, params);
328}

330std::string
331AddReactions::getReactionKernelName(const unsigned & num_reactants, const bool & is_aux)
332{
333 /*
334 * This function only gets kernel names for non-eedf kernels.
335 * There are three types of kernels (First-, Second-, and Third-order reactions)
336 * Each has variants for (a) log formulation (b) linear formulation.
337 */
338 std::string name = "Reaction";
339 if (is_aux)
340 name += "Rate";
341
342 if (num_reactants == 1)
343 name += "FirstOrder";
344 if (num_reactants == 2)
345 name += "SecondOrder";
346 if (num_reactants == 3)
347 name += "ThirdOrder";
348
349 return (name + _log_append);
350}
