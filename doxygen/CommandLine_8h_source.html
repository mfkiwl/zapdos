174template <typename T>
175bool
176CommandLine::search(const std::string & option_name, T & argument)
177{
178 std::map<std::string, Option>::iterator pos = _cli_options.find(option_name);
179 if (pos != _cli_options.end())
180 {
181 for (unsigned int i = 0; i < pos->second.cli_switch.size(); ++i)
182 {
183 for (size_t j = 0; j < _args.size(); j++)
184 {
185 auto arg = _args[j];
186
187 if (arg == pos->second.cli_switch[i])
188 {
189 // "Flag" CLI options are added as Boolean types, when we see them
190 // we set the Boolean argument to true
191 if (pos->second.argument_type == NONE)
192 argument = true;
193 else if (j + 1 < _args.size())
194 {
195 std::stringstream ss;
196 ss << _args[j + 1];
197
198 setArgument(ss, argument);
199 }
200 return true;
201 }
202 }
203 }
204
205 if (pos->second.required)
206 {
207 Moose::err << "Required parameter: " << option_name << " missing\n";
208 printUsage();
209 }
210 return false;
211 }
212 mooseError("Unrecognized option name");
213}

215template <typename T>
216bool
217CommandLine::search(const std::string & option_name, std::vector<T> & argument)
218{
219 std::map<std::string, Option>::iterator pos = _cli_options.find(option_name);
220 if (pos != _cli_options.end())
221 {
222 for (unsigned int i = 0; i < pos->second.cli_switch.size(); ++i)
223 {
224 for (size_t j = 0; j < _argv.size(); j++)
225 {
226 auto arg = _argv[j];
227
228 if (arg == pos->second.cli_switch[i])
229 {
230 // "Flag" CLI options added vector of Boolean types may apprear multiple times on the
231 // command line (like a repeated verbosity flag to increase verbosity), when we see them
232 // we append a true value to the vector.
233 if (pos->second.argument_type == NONE)
234 argument.push_back(T());
235 else
236 while (j + 1 < _argv.size() && _argv[j + 1][0] != '-' &&
237 _argv[j + 1].find("=") == std::string::npos)
238 {
239 std::stringstream ss;
240 ss << _argv[j + 1];
241
242 T item;
243 setArgument(ss, item);
244 argument.push_back(item);
245 ++j;
246 }
247 }
248 }
249 }
250
251 if (pos->second.required && argument.empty())
252 {
253 Moose::err << "Required parameter: " << option_name << " missing\n";
254 printUsage();
255 }
256 return false;
257 }
258 mooseError("Unrecognized option name");
