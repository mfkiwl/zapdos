35AuxiliarySystem::AuxiliarySystem(FEProblemBase & subproblem, const std::string & name)
36 : SystemBase(subproblem, name, Moose::VAR_AUXILIARY),
37 PerfGraphInterface(subproblem.getMooseApp().perfGraph(), "AuxiliarySystem"),
38 _fe_problem(subproblem),
39 _sys(subproblem.es().add_system<System>(name)),
40 _current_solution(_sys.current_local_solution.get()),
41 _u_dot(NULL),
42 _u_dotdot(NULL),
43 _u_dot_old(NULL),
44 _u_dotdot_old(NULL),
45 _aux_scalar_storage(_app.getExecuteOnEnum()),
46 _nodal_aux_storage(_app.getExecuteOnEnum()),
47 _mortar_nodal_aux_storage(_app.getExecuteOnEnum()),
48 _elemental_aux_storage(_app.getExecuteOnEnum()),
49 _nodal_vec_aux_storage(_app.getExecuteOnEnum()),
50 _elemental_vec_aux_storage(_app.getExecuteOnEnum()),

218void
219AuxiliarySystem::addVariable(const std::string & var_type,
220 const std::string & name,
221 InputParameters & parameters)
222{
223 SystemBase::addVariable(var_type, name, parameters);
224
225 auto fe_type = FEType(Utility::string_to_enum<Order>(parameters.get<MooseEnum>("order")),
226 Utility::string_to_enum<FEFamily>(parameters.get<MooseEnum>("family")));
227
228 if (var_type == "MooseVariableScalar")
229 return;
230
231 for (THREAD_ID tid = 0; tid < libMesh::n_threads(); tid++)
232 {
233 if (fe_type.family == LAGRANGE_VEC || fe_type.family == NEDELEC_ONE ||
234 fe_type.family == MONOMIAL_VEC || fe_type.family == RAVIART_THOMAS)
235 {
236 auto * var = _vars[tid].getActualFieldVariable<RealVectorValue>(name);
237 if (var)
238 {
239 if (var->feType().family == LAGRANGE_VEC)
240 _nodal_vars[tid].push_back(var);
241 else
242 _elem_vars[tid].push_back(var);
243 }
244 }
245
246 else
247 {
248 MooseVariableBase * var_base = _vars[tid].getVariable(name);
249
250 auto * const var = dynamic_cast<MooseVariableField<Real> *>(var_base);
251
252 if (var)
253 {
254 if (var->feType().family == LAGRANGE)
255 _nodal_vars[tid].push_back(var);
256 else
257 _elem_vars[tid].push_back(var);
258 }
259
260 auto * const avar = dynamic_cast<MooseVariableField<RealEigenVector> *>(var_base);
261
262 if (avar)
263 {
264 if (avar->feType().family == LAGRANGE)
265 _nodal_vars[tid].push_back(avar);
266 else
267 _elem_vars[tid].push_back(avar);
268 }
269 }
270 }

283void
284AuxiliarySystem::addKernel(const std::string & kernel_name,
285 const std::string & name,
286 InputParameters & parameters)
287{
288 for (THREAD_ID tid = 0; tid < libMesh::n_threads(); tid++)
289 {
290 if (parameters.get<std::string>("_moose_base") == "AuxKernel")
291 {
292 std::shared_ptr<AuxKernel> kernel =
293 _factory.create<AuxKernel>(kernel_name, name, parameters, tid);
294 if (kernel->isNodal())
295 {
296 if (kernel->isMortar())
297 _mortar_nodal_aux_storage.addObject(kernel, tid);
298 else
299 _nodal_aux_storage.addObject(kernel, tid);
300 }
301 else
302 _elemental_aux_storage.addObject(kernel, tid);
303 }
304
305 else if (parameters.get<std::string>("_moose_base") == "VectorAuxKernel")
306 {
307 std::shared_ptr<VectorAuxKernel> kernel =
308 _factory.create<VectorAuxKernel>(kernel_name, name, parameters, tid);
309 if (kernel->isNodal())
310 {
311 if (kernel->isMortar())
312 mooseError("Vector mortar aux kernels not yet implemented");
313 _nodal_vec_aux_storage.addObject(kernel, tid);
314 }
315 else
316 _elemental_vec_aux_storage.addObject(kernel, tid);
317 }
318
319 else if (parameters.get<std::string>("_moose_base") == "ArrayAuxKernel")
320 {
321 std::shared_ptr<ArrayAuxKernel> kernel =
322 _factory.create<ArrayAuxKernel>(kernel_name, name, parameters, tid);
323 if (kernel->isNodal())
324 {
325 if (kernel->isMortar())
326 mooseError("Vector mortar aux kernels not yet implemented");
327 _nodal_array_aux_storage.addObject(kernel, tid);
328 }
329 else
330 _elemental_array_aux_storage.addObject(kernel, tid);
331 }
332 }

406void
407AuxiliarySystem::compute(ExecFlagType type)
408{
409 // avoid division by dt which might be zero.
410 if (_fe_problem.dt() > 0. && _time_integrator)
411 _time_integrator->preStep();
412
413 // We need to compute time derivatives every time each kind of the variables is finished, because:
414 //
415 // a) the user might want to use the aux variable value somewhere, thus we need to provide the
416 // up-to-date value
417 // b) time integration system works with the whole vectors of solutions, thus we cannot update
418 // only a part of the vector
419 //
420
421 if (_vars[0].scalars().size() > 0)
422 {
423 computeScalarVars(type);
424 // compute time derivatives of scalar aux variables _after_ the values were updated
425 if (_fe_problem.dt() > 0. && _time_integrator)
426 _time_integrator->computeTimeDerivatives();
427 }
428
429 if (_vars[0].fieldVariables().size() > 0)
430 {
431 computeNodalArrayVars(type);
432 computeNodalVecVars(type);
433 computeNodalVars(type);
434 computeMortarNodalVars(type);
435 computeElementalArrayVars(type);
436 computeElementalVecVars(type);
437 computeElementalVars(type);
438
439 // compute time derivatives of nodal aux variables _after_ the values were updated
440 if (_fe_problem.dt() > 0. && _time_integrator)
441 _time_integrator->computeTimeDerivatives();
442 }
443
444 if (_serialized_solution.get())
445 serializeSolution();
446}

617void
618AuxiliarySystem::setScalarVariableCoupleableTags(ExecFlagType type)
619{
620 const MooseObjectWarehouse<AuxScalarKernel> & storage = _aux_scalar_storage[type];
621 const std::vector<std::shared_ptr<AuxScalarKernel>> & objects = storage.getActiveObjects(0);
622
623 std::set<TagID> needed_sc_var_matrix_tags;
624 std::set<TagID> needed_sc_var_vector_tags;
625 for (const auto & obj : objects)
626 {
627 auto & sc_var_coup_vtags = obj->getScalarVariableCoupleableVectorTags();
628 needed_sc_var_vector_tags.insert(sc_var_coup_vtags.begin(), sc_var_coup_vtags.end());
629
630 auto & sc_var_coup_mtags = obj->getScalarVariableCoupleableMatrixTags();
631 needed_sc_var_matrix_tags.insert(sc_var_coup_mtags.begin(), sc_var_coup_mtags.end());
632 }
633
634 _fe_problem.setActiveScalarVariableCoupleableMatrixTags(needed_sc_var_matrix_tags, 0);
635 _fe_problem.setActiveScalarVariableCoupleableVectorTags(needed_sc_var_vector_tags, 0);
636}

645void
646AuxiliarySystem::computeScalarVars(ExecFlagType type)
647{
648 setScalarVariableCoupleableTags(type);
649
650 // Reference to the current storage container
651 const MooseObjectWarehouse<AuxScalarKernel> & storage = _aux_scalar_storage[type];
652
653 if (storage.hasActiveObjects())
654 {
655 TIME_SECTION("computeScalarVars", 1);
656
657 PARALLEL_TRY
658 {
659 // FIXME: run multi-threaded
660 THREAD_ID tid = 0;
661 if (storage.hasActiveObjects())
662 {
663 _fe_problem.reinitScalars(tid);
664
665 const std::vector<std::shared_ptr<AuxScalarKernel>> & objects =
666 storage.getActiveObjects(tid);
667
668 // Call compute() method on all active AuxScalarKernel objects
669 for (const auto & obj : objects)
670 obj->compute();
671
672 const std::vector<MooseVariableScalar *> & scalar_vars = getScalarVariables(tid);
673 for (const auto & var : scalar_vars)
674 var->insert(solution());
675 }
676 }
677 PARALLEL_CATCH;
678
679 solution().close();
680 _sys.update();
681 }
682
683 clearScalarVariableCoupleableTags();
684}

794{
