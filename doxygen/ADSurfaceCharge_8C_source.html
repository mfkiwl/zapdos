44ADSurfaceCharge::ADSurfaceCharge(const InputParameters & parameters)
45 : ADMaterial(parameters),
46 _sigma(declareADProperty<Real>("surface_charge")),
47 _sigma_old(getMaterialPropertyOld<Real>("surface_charge")),
48 _plasma_current(declareProperty<Real>("plasma_current")),
49
50 _r_units(1. / getParam<Real>("position_units")),
51
52 _potential_units(getParam<std::string>("potential_units")),
53 _grad_potential(adCoupledGradient("potential"))
54{
55 // Define voltage scaling parameter
56 if (_potential_units.compare("V") == 0)
57 _voltage_scaling = 1.;
58 else if (_potential_units.compare("kV") == 0)
59 _voltage_scaling = 1000;
60
61 _num_species = coupledComponents("species");
62
63 // Resize the vectors to store _num_species pointers
64 _species.resize(_num_species);
65 _grad_species.resize(_num_species);
66 _mu.resize(_num_species);
67 _diff.resize(_num_species);
68 _sgn.resize(_num_species);
69
70 for (unsigned int i = 0; i < _num_species; ++i)
71 {
72 _species[i] = &adCoupledValue("species", i);
73 _grad_species[i] = &adCoupledGradient("species", i);
74 _mu[i] = &getADMaterialProperty<Real>("mu" + (*getVar("species", i)).name());
75 _diff[i] = &getADMaterialProperty<Real>("diff" + (*getVar("species", i)).name());
76 _sgn[i] = &getMaterialProperty<Real>("sgn" + (*getVar("species", i)).name());
77 }
78
79 // Precalculate constant values
80 _q_times_NA = 1.602e-19 * 6.022e23 / _voltage_scaling;
81}

89void
90ADSurfaceCharge::computeQpProperties()
91{
92 if (_material_data_type == Moose::FACE_MATERIAL_DATA || boundaryRestricted())
93 {
94
95 // COMPUTE CHARGED FLUX
96 _charge_flux = 0.0;
97 for (unsigned int i = 0; i < _num_species; ++i)
98 {
99 _charge_flux += (*_sgn[i])[_qp] * std::exp((*_species[i])[_qp]) *
100 ((*_sgn[i])[_qp] * (*_mu[i])[_qp] * -_grad_potential[_qp] * _r_units -
101 (*_diff[i])[_qp] * (*_grad_species[i])[_qp] * _r_units) *
102 _normals[_qp];
103 }
104
105 _plasma_current[_qp] = MetaPhysicL::raw_value(_charge_flux) * 1.602e-19 * 6.022e23;
106
107 _sigma[_qp] = _sigma_old[_qp] + _charge_flux * _q_times_NA * _dt;
108 }
109 else
110 _sigma[_qp] = 0.;
111}
