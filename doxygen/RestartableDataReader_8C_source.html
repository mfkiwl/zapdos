68std::vector<std::unordered_map<std::string, RestartableDataReader::HeaderEntry>>
69RestartableDataReader::readHeader(InputStream & header_input) const
70{
71 std::vector<std::unordered_map<std::string, RestartableDataReader::HeaderEntry>> header;
72
73 auto stream_ptr = header_input.get();
74 auto & stream = *stream_ptr;
75
76 stream.seekg(0);
77
78 const auto error = [&header_input](auto... args)
79 {
80 std::stringstream err_prefix;
81 err_prefix << "While reading restartable data in ";
82 const auto filename = header_input.getFilename();
83 if (filename)
84 err_prefix << std::filesystem::absolute(filename->parent_path());
85 else
86 err_prefix << "memory";
87 err_prefix << ":\n\n";
88
89 mooseError(err_prefix.str(), args...);
90 };
91
92 // ID
93 char this_id[2];
94 stream.read(this_id, 2);
95 if (this_id[0] != 'R' || this_id[1] != 'D')
96 error("The data is invalid or corrupted (unexpected header)");
97
98 // File version
99 std::remove_const<decltype(CURRENT_BACKUP_FILE_VERSION)>::type this_file_version;
100 dataLoad(stream, this_file_version, nullptr);
101 if (this_file_version != CURRENT_BACKUP_FILE_VERSION)
102 error("There is a mismatch in the backup version\n\n",
103 "Current backup version: ",
104 CURRENT_BACKUP_FILE_VERSION,
105 "\nLoaded backup version: ",
106 this_file_version);
107
108 // Type id for a basic type
109 std::size_t this_compare_hash_code;
110 dataLoad(stream, this_compare_hash_code, nullptr);
111#ifndef RESTARTABLE_SKIP_CHECK_HASH_CODE
112 if (this_compare_hash_code != typeid(COMPARE_HASH_CODE_TYPE).hash_code())
113 error("The backup is not compatible\n\nThe hash code check for a basic type (",
114 MooseUtils::prettyCppType<COMPARE_HASH_CODE_TYPE>(),
115 ") failed.\nIt is likely that this backup was stored with a different architecture or "
116 "operating system");
117#endif
118
119 // Number of procs
120 decltype(n_processors()) this_n_procs = 0;
121 dataLoad(stream, this_n_procs, nullptr);
122 if (_error_on_different_number_of_processors && this_n_procs != n_processors())
123 error("The number of MPI ranks is not consistent\n\nCurrent MPI ranks: ",
124 n_processors(),
125 "\nLoaded MPI ranks: ",
126 this_n_procs);
127
128 // Number of data
129 decltype(dataSize()) this_num_data = 0;
130 dataLoad(stream, this_num_data, nullptr);
131 if (this_num_data != dataSize())
132 error("The number of threads is not consistent\n\nCurrent threads: ",
133 dataSize(),
134 "\nLoaded threads: ",
135 this_num_data);
136
137 header.resize(dataSize());
138
139 // Size of data for each thread
140 std::vector<std::size_t> tid_n_data(dataSize());
141 for (const auto tid : make_range(dataSize()))
142 dataLoad(stream, tid_n_data[tid], nullptr);
143
144 // The position of the current data that we're loading
145 std::size_t current_data_position = 0;
146
147 // Load the data header for each thread
148 for (const auto tid : make_range(dataSize()))
149 for (const auto i : make_range(tid_n_data[tid]))
150 {
151 std::ignore = i;
152
153 std::string name;
154 dataLoad(stream, name, nullptr);
155 mooseAssert(name.size(), "Empty name");
156
157 mooseAssert(!header[tid].count(name), "Data '" + name + "' is already inserted");
158 auto & entry = header[tid][name];
159
160 dataLoad(stream, entry.size, nullptr);
161 dataLoad(stream, entry.type_hash_code, nullptr);
162 dataLoad(stream, entry.type, nullptr);
163 dataLoad(stream, entry.has_context, nullptr);
164 entry.position = current_data_position;
165
166 current_data_position += entry.size;
167 }
168
169 stream.seekg(0);
170
171 return header;
172}

174void
175RestartableDataReader::restore(const DataNames & filter_names /* = {} */)
176{
177 if (!_streams.header || !_streams.data)
178 mooseError("RestartableDataReader::restore(): Cannot restore because an input was not set");
179
180 _is_restoring = true;
181
182 // Set everything as not loaded
183 for (const auto tid : make_range(dataSize()))
184 for (auto & value : currentData(tid))
185 value.setNotLoaded({});
186
187 // Read the header
188 _header = readHeader(*_streams.header);
189
190 for (const auto tid : index_range(_header))
191 {
192 auto & data = currentData(tid);
193 const auto & header = _header[tid];
194
195 // TODO: Think about what to do with missing data
196 // Load the data in the order that it was requested
197 for (auto & value : data)
198 {
199 const auto & name = value.name();
200
201 auto find_header = header.find(name);
202 if (find_header == header.end())
203 continue;
204
205 auto & header_entry = find_header->second;
206
207 // Only restore values if we're either recovering or the data isn't filtered out
208 const auto is_data_in_filter = filter_names.find(name) != filter_names.end();
209 if (!is_data_in_filter)
210 deserializeValue(*_streams.data, value, header_entry);
211 }
212 }
213}

247void
248RestartableDataReader::deserializeValue(
249 InputStream & data_input,
250 RestartableDataValue & value,
251 const RestartableDataReader::HeaderEntry & header_entry) const
252{
253 mooseAssert(!value.loaded(), value.name() + " is already loaded");
254
255 auto error = [&data_input, &value](auto... args)
256 {
257 std::stringstream err;
258 err << "While loading restartable data\n\n";
259 err << "From: ";
260 const auto filename = data_input.getFilename();
261 if (filename)
262 err << std::filesystem::absolute(filename->parent_path());
263 else
264 err << "memory";
265 err << "\nData name: " << value.name();
266 err << "\nData type: " << value.type();
267 err << "\n\n";
268 mooseError(err.str(), args...);
269 };
270
271 if (!isSameType(header_entry, value.typeId()))
272 error("The stored type of '", header_entry.type, "' does not match");
273
274 auto stream_ptr = data_input.get();
275 auto & stream = *stream_ptr;
276
277 stream.seekg(header_entry.position);
278 value.load(stream);
279
280 if (stream.tellg() == -1)
281 error("An error was encountered when reading from the stream");
282
283 const std::size_t loaded_size = stream.tellg() - header_entry.position;
284 if (loaded_size != header_entry.size)
285 error("The data read does not match the data stored\n\n",
286 "Stored size: ",
287 header_entry.size,
288 "\nLoaded size: ",
289 loaded_size);
290}

292bool
293RestartableDataReader::isAvailable(const std::filesystem::path & folder_base)
294{
295 const auto header_path = restartableDataFile(folder_base);
296 const auto data_path = restartableDataFile(folder_base);
297
298 const auto available = [](const auto & filename)
299 { return MooseUtils::pathExists(filename) && MooseUtils::checkFileReadable(filename); };
300
301 const bool header_available = available(header_path);
302 const bool data_available = available(data_path);
303
304 if (header_available != data_available)
305 mooseError("The restart ",
306 header_available ? "header" : "data",
307 " is available but the corresponding ",
308 header_available ? "data" : "header",
309 " is not available\n\n",
310 "Header (",
311 header_available ? "available" : "missing",
312 "): ",
313 std::filesystem::absolute(header_path),
314 "\nData (",
315 data_available ? "available" : "missing",
316 "): ",
317 std::filesystem::absolute(header_path));
318
319 return header_available && data_available;
320}

344RestartableDataReader::restoreData(const std::string & data_name,
345 std::unique_ptr<RestartableDataValue> value,
346 const THREAD_ID tid)
347{
348 auto & data_map = currentData(tid);
349 if (data_map.hasData(data_name))
350 mooseError("RestartableDataReader::restoreData(): Cannot declare restartable data '",
351 data_name,
352 "' because it has already been declared");
353
354 const auto & header = getHeader(data_name, tid);
355 auto & added_value = currentData(tid).addData(std::move(value));
356 deserializeValue(*_streams.data, added_value, header);
357 return added_value;
