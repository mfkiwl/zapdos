21void
22Factory::reg(std::shared_ptr<RegistryEntryBase> obj)
23{
24 const std::string obj_name = obj->name();
25 const std::string & label = obj->_label;
26 const std::string & deprecated_time = obj->_deprecated_time;
27 const std::string & replacement_name = obj->_replaced_by;
28 const std::string & file = obj->_file;
29 const int line = obj->_line;
30
31 // do nothing if we have already added this exact object before
32 auto key = std::make_pair(label, obj_name);
33 if (_objects_by_label.find(key) != _objects_by_label.end())
34 return;
35
36 /*
37 * If _registerable_objects has been set the user has requested that we only register some
38 * subset
39 * of the objects for a dynamically loaded application. The objects listed in *this*
40 * application's
41 * registerObjects() method will have already been registered before that member was set.
42 *
43 * If _registerable_objects is empty, the factory is unrestricted
44 */
45 if (_registerable_objects.empty() ||
46 _registerable_objects.find(obj_name) != _registerable_objects.end())
47 {
48 if (_name_to_object.find(obj_name) != _name_to_object.end())
49 mooseError("Object '" + obj_name + "' registered from multiple files: ",
50 file,
51 " and ",
52 _name_to_line.getInfo(obj_name).file());
53 _name_to_object[obj_name] = obj;
54 _objects_by_label.insert(key);
55 }
56 _name_to_line.addInfo(obj_name, file, line);
57
58 if (!replacement_name.empty())
59 _deprecated_name[obj_name] = replacement_name;
60 if (!deprecated_time.empty())
61 _deprecated_time[obj_name] = parseTime(deprecated_time);
62
63 // TODO: Possibly store and print information about objects that are skipped here?
64}

66InputParameters
67Factory::getValidParams(const std::string & obj_name)
68{
69 const auto it = _name_to_object.find(obj_name);
70
71 // Check if the object is registered
72 if (it == _name_to_object.end())
73 reportUnregisteredError(obj_name);
74
75 // Print out deprecated message, if it exists
76 deprecatedMessage(obj_name);
77
78 // Return the parameters
79 auto params = it->second->buildParameters();
80 params.addPrivateParam("_moose_app", &_app);
81
82 return params;
83}

85std::shared_ptr<MooseObject>
86Factory::create(const std::string & obj_name,
87 const std::string & name,
88 const InputParameters & parameters,
89 THREAD_ID tid /* =0 */,
90 bool print_deprecated /* =true */)
91{
92 if (print_deprecated)
93 mooseDeprecated("Factory::create() is deprecated, please use Factory::create<T>() instead");
94
95 // Pointer to the object constructor
96 const auto it = _name_to_object.find(obj_name);
97
98 // Check if the object is registered
99 if (it == _name_to_object.end())
100 reportUnregisteredError(obj_name);
101
102 // Print out deprecated message, if it exists
103 deprecatedMessage(obj_name);
104
105 // Create the actual parameters object that the object will reference
106 InputParameters & params =
107 _app.getInputParameterWarehouse().addInputParameters(name, parameters, tid);
108
109 // Set the _type parameter
110 params.set<std::string>("_type") = obj_name;
111
112 // Check to make sure that all required parameters are supplied
113 params.checkParams(name);
114
115 // register type name as constructed
116 _constructed_types.insert(obj_name);
117
118 // add FEProblem pointers to object's params object
119 if (_app.actionWarehouse().problemBase())
120 _app.actionWarehouse().problemBase()->setInputParametersFEProblem(params);
121
122 // call the function pointer to build the object
123 auto obj = it->second->build(params);
124
125 auto fep = std::dynamic_pointer_cast<FEProblemBase>(obj);
126 if (fep)
127 _app.actionWarehouse().problemBase() = fep;
128
129 // Make sure no unexpected parameters were added by the object's constructor or by the action
130 // initiating this create call. All parameters modified by the constructor must have already
131 // been specified in the object's validParams function.
132 InputParameters orig_params = getValidParams(obj_name);
133 if (orig_params.n_parameters() != parameters.n_parameters())
134 {
135 std::set<std::string> orig, populated;
136 for (const auto & it : orig_params)
137 orig.emplace(it.first);
138 for (const auto & it : parameters)
139 populated.emplace(it.first);
140
141 std::set<std::string> diff;
142 std::set_difference(populated.begin(),
143 populated.end(),
144 orig.begin(),
145 orig.end(),
146 std::inserter(diff, diff.begin()));
147
148 if (!diff.empty())
149 {
150 std::stringstream ss;
151 for (const auto & name : diff)
152 ss << ", " << name;
153 mooseError("attempted to set unregistered parameter(s) for ",
154 obj_name,
155 " object:\n ",
156 ss.str().substr(2));
157 }
158 }
159
160 return obj;
161}

175std::time_t
176Factory::parseTime(const std::string t_str)
177{
178 // The string must be a certain length to be valid
179 if (t_str.size() != 16)
180 mooseError("The deprecated time not formatted correctly; it must be given as mm/dd/yyyy HH:MM");
181
182 // Store the time, the time must be specified as: mm/dd/yyyy HH:MM
183 std::time_t t_end;
184 struct tm * t_end_info;
185 time(&t_end);
186 t_end_info = localtime(&t_end);
187 t_end_info->tm_mon = std::atoi(t_str.substr(0, 2).c_str()) - 1;
188 t_end_info->tm_mday = std::atoi(t_str.substr(3, 2).c_str());
189 t_end_info->tm_year = std::atoi(t_str.substr(6, 4).c_str()) - 1900;
190 t_end_info->tm_hour = std::atoi(t_str.substr(11, 2).c_str()) + 1;
191 t_end_info->tm_min = std::atoi(t_str.substr(14, 2).c_str());
192 t_end_info->tm_sec = 0;
193 t_end = mktime(t_end_info);
194 return t_end;
195}

197void
198Factory::deprecatedMessage(const std::string obj_name)
199{
200 const auto time_it = _deprecated_time.find(obj_name);
201
202 // If the object is not deprecated return
203 if (time_it == _deprecated_time.end())
204 return;
205
206 // If the message has already been printed, return
207 if (_deprecated_types.count(obj_name))
208 return;
209 _deprecated_types.emplace(obj_name);
210
211 // We dont need a backtrace on this, this is user-facing
212 const auto current_show_trace = Moose::show_trace;
213 Moose::show_trace = false;
214
215 // Get the current time
216 std::time_t now;
217 time(&now);
218
219 // Get the stop time
220 std::time_t t_end = time_it->second;
221
222 // Message storage
223 std::ostringstream msg;
224
225 const auto name_it = _deprecated_name.find(obj_name);
226
227 // Expired object
228 if (now > t_end)
229 {
230 msg << "***** Invalid Object: " << obj_name << " *****\n";
231 msg << "Expired on " << ctime(&t_end);
232
233 // Append replacement object, if it exists
234 if (name_it != _deprecated_name.end())
235 msg << "Update your application using the '" << name_it->second << "' object";
236
237 // Produce the error message
238 mooseDeprecationExpired(msg.str());
239 }
240
241 // Expiring object
242 else
243 {
244 // Build the basic message
245 msg << "Deprecated Object: " << obj_name << "\n";
246 msg << "This object will be removed on " << ctime(&t_end);
247
248 // Append replacement object, if it exists
249 if (name_it != _deprecated_name.end())
250 msg << "Replace " << obj_name << " with " << name_it->second;
251
252 // Produce the error message
253 mooseDeprecated(msg.str());
254 }
255 Moose::show_trace = current_show_trace;
256}

258void
259Factory::reportUnregisteredError(const std::string & obj_name) const
260{
261 std::ostringstream oss;
262 std::set<std::string> paths = _app.getLoadedLibraryPaths();
263
264 oss << "A '" + obj_name + "' is not a registered object.\n";
265
266 if (!paths.empty())
267 {
268 oss << "\nWe loaded objects from the following libraries and still couldn't find your "
269 "object:\n\t";
270 std::copy(paths.begin(), paths.end(), infix_ostream_iterator<std::string>(oss, "\n\t"));
271 oss << '\n';
272 }
273
274 oss << "\nIf you are trying to find this object in a dynamically loaded library, make sure that\n"
275 "the library can be found either in your \"Problem/library_path\" parameter or in the\n"
276 "MOOSE_LIBRARY_PATH environment variable.";
277
278 mooseError(oss.str());
279}
