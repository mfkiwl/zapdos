16template <bool is_ad>
17FunctionMaterialPropertyDescriptor<is_ad>::FunctionMaterialPropertyDescriptor(
18 const std::string & expression, MooseObject * parent, bool required)
19 : _dependent_symbols(), _derivative_symbols(), _parent(parent), _required(required)
20{
21 auto define = expression.find_last_of(":=");
22
23 // expression contains a ':='
24 if (define != std::string::npos)
25 {
26 // section before ':=' is the name used in the function expression
27 _fparser_name = expression.substr(0, define - 1);
28
29 // parse right hand side
30 parseDerivative(expression.substr(define + 1));
31 }
32 else
33 {
34 // parse entire expression and use natural material property base name
35 // for D(x(t),t,t) this would simply be 'x'!
36 parseDerivative(expression);
37 _fparser_name = _base_name;
38 }
39
40 _value = nullptr;
41 updatePropertyName();
42}

118template <bool is_ad>
119void
120FunctionMaterialPropertyDescriptor<is_ad>::parseDerivative(const std::string & expression)
121{
122 auto open = expression.find_first_of("[");
123 auto close = expression.find_last_of("]");
124
125 if (open == std::string::npos && close == std::string::npos)
126 {
127 // no derivative requested
128 parseDependentSymbols(expression);
129
130 return;
131 }
132 else if (open != std::string::npos && close != std::string::npos &&
133 expression.substr(0, open) == "D")
134 {
135 // tokenize splits the arguments in d2h2:=D[h2(eta1,eta2),eta1] into 'h2(eta1' 'eta2)' 'eta1'
136 // DAMN!!
137 auto arguments = expression.substr(open + 1, close - open - 1);
138 auto close2 = arguments.find_last_of(")");
139
140 if (close2 == std::string::npos)
141 {
142 // rest of argument list 0 is the function and 1,.. are the variable to take the derivative
143 // w.r.t.
144 MooseUtils::tokenize(arguments, _derivative_symbols, 0, ",");
145
146 // check for empty [] brackets
147 if (_derivative_symbols.size() > 0)
148 {
149 // parse argument zero of D[] as the function material property
150 parseDependentSymbols(_derivative_symbols[0]);
151
152 // remove function from the _derivative_symbols vector
153 _derivative_symbols.erase(_derivative_symbols.begin());
154 updatePropertyName();
155
156 return;
157 }
158 }
159 else
160 {
161 parseDependentSymbols(arguments.substr(0, close2 + 1));
162 MooseUtils::tokenize(arguments.substr(close2 + 2), _derivative_symbols, 0, ",");
163 updatePropertyName();
164 return;
165 }
166 }
167
168 mooseError("Malformed material_properties expression '", expression, "'");
169}

171template <bool is_ad>
172void
173FunctionMaterialPropertyDescriptor<is_ad>::parseDependentSymbols(const std::string & expression)
174{
175 auto open = expression.find_first_of("(");
176 auto close = expression.find_last_of(")");
177
178 if (open == std::string::npos && close == std::string::npos)
179 {
180 // material property name without arguments
181 _base_name = expression;
182 }
183 else if (open != std::string::npos && close != std::string::npos)
184 {
185 // take material property name before bracket
186 _base_name = expression.substr(0, open);
187
188 // parse argument list
189 MooseUtils::tokenize(expression.substr(open + 1, close - open - 1), _dependent_symbols, 0, ",");
190
191 // cremove duplicates from dependent variable list
192 std::sort(_dependent_symbols.begin(), _dependent_symbols.end());
193 _dependent_symbols.erase(std::unique(_dependent_symbols.begin(), _dependent_symbols.end()),
194 _dependent_symbols.end());
195 }
196 else
197 mooseError("Malformed material_properties expression '", expression, "'");
198}

213template <bool is_ad>
214const GenericMaterialProperty<Real, is_ad> &
215FunctionMaterialPropertyDescriptor<is_ad>::value() const
216{
217 if (_value == nullptr)
218 {
219 DerivativeMaterialInterface<Material> * _material_parent =
220 dynamic_cast<DerivativeMaterialInterface<Material> *>(_parent);
221 DerivativeMaterialInterface<Kernel> * _kernel_parent =
222 dynamic_cast<DerivativeMaterialInterface<Kernel> *>(_parent);
223
224 // property name
225 auto name = derivativePropertyName(_base_name, _derivative_symbols);
226
227 // get the material property reference
228 if (_material_parent)
229 _value = _required ? &(_material_parent->getGenericMaterialProperty<Real, is_ad>(name))
230 : &(_material_parent->getGenericZeroMaterialProperty<Real, is_ad>(name));
231 else if (_kernel_parent)
232 _value = _required ? &(_kernel_parent->getGenericMaterialProperty<Real, is_ad>(name))
233 : &(_kernel_parent->getGenericZeroMaterialProperty<Real, is_ad>(name));
234 else
235 mooseError("A FunctionMaterialPropertyDescriptor must be owned by either a Material or a "
236 "Kernel object.");
237 }
238
239 return *_value;
240}
