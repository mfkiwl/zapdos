622 int new_index = _num_reactions - 1;

624 // We also need to resize rate_coefficient and threshold_energy to account
625 // for the new reaction(s)
626 _superelastic_index.resize(_num_reactions + superelastic_reactions, 0);
627 _superelastic_reaction.resize(_num_reactions + superelastic_reactions, false);
628 _rate_coefficient.resize(_num_reactions + superelastic_reactions);
629 _threshold_energy.resize(_num_reactions + superelastic_reactions);
630 _rate_equation.resize(_num_reactions + superelastic_reactions);
631 _species_count.resize(_num_reactions + superelastic_reactions,
632 std::vector<Real>(_species.size()));
633 _reactants.resize(_reactants.size() + superelastic_reactions);
634 _products.resize(_products.size() + superelastic_reactions);
635 _aux_var_name.resize(_num_reactions + superelastic_reactions);
636 _energy_change.resize(_num_reactions + superelastic_reactions);
637 _reaction_coefficient_name.resize(_num_reactions + superelastic_reactions);
638 if (superelastic_reactions > 0)
639 {
640 for (unsigned int i = 0; i < _num_reactions; ++i)
641 {
642 if (_reversible_reaction[i] == true)
643 {
644 // _reaction.resize(_num_reactions + 1);
645 new_index += 1;
646 // This index refers to the ORIGINAL reaction. Example:
647 // If reaction #2 out of 5 (index = 1 from [0,4]) is reversible, then a
648 // 6th reaction (index = 5) will be added. This is the superelastic reaction.
649 // _superelastic_index is intended to refer back to the original reversible
650 // reaction later in the code, so we can reverse any energy change and refer
651 // to the forward reaction rate if necessary. Thus in this particular case,
652 // _superelastic_index[5] = 1.
653 _superelastic_index[new_index] = i;
654 _superelastic_reaction[new_index] = true;
655 _rate_coefficient[new_index] = NAN;
656 _threshold_energy[new_index] = -_threshold_energy[i];
657 _aux_var_name[new_index] = "rate_constant" + std::to_string(new_index);
658 _reaction_coefficient_name[new_index] = "rate_constant" + std::to_string(new_index);
659 if (_rate_equation[i] == true)
660 {
661 _rate_equation[new_index] = true;
662 }
663 else
664 {
665 _rate_equation[new_index] = false;
666 }
667
668 if (_energy_change[i])
669 _energy_change[new_index] = true;
670 else
671 _energy_change[new_index] = false;
672
673 // Here we reverse the products and reactants to build superelastic reactions.
674 for (MooseIndex(_num_products[i]) j = 0; j < _num_products[i]; ++j)
675 {
676 new_reaction.append(_products[i][j]);
677 _reactants[new_index].push_back(_products[i][j]);
678 if (j == _num_products[i] - 1)
679 {
680 break;
681 }
682 else
683 {
684 new_reaction.append(" + ");
685 }
686 }
687 new_reaction.append(" -> ");
688 for (unsigned int j = 0; j < _reactants[i].size(); ++j)
689 {
690 new_reaction.append(_reactants[i][j]);
691 _products[new_index].push_back(_reactants[i][j]);
692 if (j == _reactants[i].size() - 1)
693 {
694 break;
695 }
696 else
697 {
698 new_reaction.append(" + ");
699 }
700 }
701 _reaction.push_back(new_reaction);
702 }
703
704 // Calculate coefficients
705 for (unsigned int j = 0; j < _species.size(); ++j)
706 {
707 for (unsigned int k = 0; k < _reactants[new_index].size(); ++k)
708 {
709 if (_reactants[new_index][k] == _species[j])
710 {
711 _species_count[new_index][j] -= 1;
712 }
713 if (getParam<bool>("include_electrons") == true)
714 {
715 if (_reactants[new_index][k] == getParam<std::string>("electron_density"))
716 {
717 _electron_index[new_index] = k;
718 }
719 }
720 }
721 for (unsigned int k = 0; k < _products[new_index].size(); ++k)
722 {
723 if (_products[new_index][k] == _species[j])
724 {
725 _species_count[new_index][j] += 1;
726 }
727 }
728 }
729 }
730 }
731
732 _num_reactions += superelastic_reactions;
733 _reaction_coefficient_name.resize(_num_reactions);
734 // Find the unique species across all reaction pathways
735 // Note that this also accounts for species that are not tracked in case
736 // some of the species are considered to be uniform background gases or
737 // arbitrary source/sink terms.
738 sort(_all_participants.begin(), _all_participants.end());
739 std::vector<std::string>::iterator it;
740 it = std::unique(_all_participants.begin(), _all_participants.end());
741 _all_participants.resize(std::distance(_all_participants.begin(), it));
742
743 // Find the stoichometric coefficient for each participant.
744 _stoichiometric_coeff.resize(_reaction.size());
745
746 for (unsigned int i = 0; i < _reaction.size(); ++i)
747 {
748 _stoichiometric_coeff[i].resize(_all_participants.size(), 0);
749
750 for (unsigned int j = 0; j < _all_participants.size(); ++j)
751 {
752 for (unsigned int k = 0; k < _reactants[i].size(); ++k)
753 {
754 if (_reactants[i][k] == _all_participants[j])
755 {
756 _stoichiometric_coeff[i][j] -= 1;
757 }
758 }
759
760 for (unsigned int k = 0; k < _products[i].size(); ++k)
761 {
762 if (_products[i][k] == _all_participants[j])
763 {
764 _stoichiometric_coeff[i][j] += 1;
765 }
766 }
767 }
768 }
769 _reaction_participants.resize(_num_reactions);
770 _reaction_stoichiometric_coeff.resize(_num_reactions);
771 // Now we find which index of _all_participants is associated with _species
772 // so they can be accurately referred to later if necessary.
773
774 _species_index.resize(
775 _species.size()); // Stores vector of indices relating _all_participants to _species.
776 std::vector<std::string>::iterator iter;
777
778 for (unsigned int i = 0; i < _species.size(); ++i)
779 {
780 iter = std::find(_all_participants.begin(), _all_participants.end(), _species[i]);
781 _species_index[i] = std::distance(_all_participants.begin(), iter);
782 }
783
784 // Finally, we reduce _all_participants to find just the relevant participants
785 // (and stoichiometric coefficients) for each individual reaction.
786
787 for (unsigned int i = 0; i < _num_reactions; ++i)
788 {
789 std::vector<std::string> species_temp(
790 _reactants[i]); // Copy reactants into new temporary vector
791 species_temp.insert(species_temp.end(),
792 _products[i].begin(),
793 _products[i].end()); // Append products to new temp vector (so it now stores
794 // all reactants and products)
795
796 // Separate out the unique values from species_temp
797 sort(species_temp.begin(), species_temp.end());
798 std::vector<std::string>::iterator it;
799 it = std::unique(species_temp.begin(), species_temp.end());
800 species_temp.resize(std::distance(species_temp.begin(), it));
801 // _reaction_participants[i].resize(species_temp.size());
802
803 // Copy over the species from species_temp to the permanent _reaction_participants vector
804 // Now each unique participant per reaction is stored in _reaction_participants.
805 // Note that reaction_participants only stores the TRACKED species.
806 for (unsigned int j = 0; j < species_temp.size(); ++j)
807 {
808 if (std::find(_species.begin(), _species.end(), species_temp[j]) != _species.end())
809 _reaction_participants[i].push_back(species_temp[j]);
810 }
811
812 _reaction_stoichiometric_coeff[i].resize(_reaction_participants[i].size(), 0);
813
814 for (unsigned int j = 0; j < _reaction_participants[i].size(); ++j)
815 {
816 for (unsigned int k = 0; k < _reactants[i].size(); ++k)
817 {
818 if (_reactants[i][k] == _reaction_participants[i][j])
819 {
820 _reaction_stoichiometric_coeff[i][j] -= 1;
821 }
822 }
823
824 for (unsigned int k = 0; k < _products[i].size(); ++k)
825 {
826 if (_products[i][k] == _reaction_participants[i][j])
827 {
828 _reaction_stoichiometric_coeff[i][j] += 1;
829 }
830 }
831 }
832
833 if (_energy_change[i])
834 {
835 if (_electron_energy.size() == 0 && _gas_energy.size() == 0)
836 mooseError("Reactions have energy changes, but no electron or gas temperature variable is "
837 "included!");
838 }
839 }
840 if (_electron_energy.size() > 0)
841 {
842 _electron_energy_term.push_back(true);
843 _energy_variable.push_back(_electron_energy[0]);
844 }
845 if (_gas_energy.size() > 0)
846 {
847 _electron_energy_term.push_back(false);
848 _energy_variable.push_back(_gas_energy[0]);
849 }
850
851 // With all reactions set up, now we check to make sure all equations are balanced. This requires
852 // the parameter balance_check = true and a vector of values corresponding to the number of
853 // particles associated with each species. TO DO: check charge balance as well
854 if (getParam<bool>("balance_check"))
855 {
856
857 int index; // stores index of species in the reactant/product arrays
858 // std::vector<std::string>::iterator iter;
859 Real r_sum;
860 Real p_sum;
861 std::vector<std::string> faulty_reaction;
862 bool unbalanced = false;
863 // bool electron_reaction;
864
865 // charge balance is not yet implemented
866 // bool charge_balance = getParam<bool>("charge_balance_check");
867
868 for (unsigned int i = 0; i < _num_reactions; ++i)
869 {
870 // electron_reaction = false;
871 r_sum = 0;
872 p_sum = 0;
873 for (unsigned int j = 0; j < _reactants[i].size(); ++j)
874 {
875 auto iter = std::find(_species.begin(), _species.end(), _reactants[i][j]);
876 index = std::distance(_species.begin(), iter);
877 if (_species[index] == getParam<std::string>("electron_density"))
878 continue;
879 else
880 r_sum += num_particles[index];
881 // if (_species[index] == getParam<std::string>("electron_density"))
882 // electron_reaction = true;
883 }
884 for (unsigned int j = 0; j < _products[i].size(); ++j)
885 {
886 auto iter = std::find(_species.begin(), _species.end(), _products[i][j]);
887 index = std::distance(_species.begin(), iter);
888 if (_species[index] == getParam<std::string>("electron_density"))
889 continue;
890 else
891 p_sum += num_particles[index];
892 // if (_species[index] == getParam<std::string>("electron_density"))
893 // electron_reaction = true;
894 }
895 // if (r_sum != p_sum && !electron_reaction)
896 if (r_sum != p_sum)
897 {
898 unbalanced = true;
899 faulty_reaction.push_back(_reaction[i]);
900 }
901 }
902 if (unbalanced)
903 {
904 string error_str;
905 for (unsigned int i = 0; i < faulty_reaction.size(); ++i)
906 {
907 error_str.append(" ");
908 error_str.append(faulty_reaction[i]);
909 error_str.append("\n");
910 }
911
912 mooseError("The following equations are unbalanced:\n",
913 error_str,
914 "Fix unbalanced reactions or particle conservation will not be enforced.");
915 }
916 }
917
918 // Here we check to make sure that there are no aux_variables in the species list.
919 // Only nonlinear variables should be included.
920 for (unsigned int i = 0; i < _species.size(); ++i)
921 {
922 for (unsigned int j = 0; j < _aux_species.size(); ++j)
923 {
924 if (_species[i] == _aux_species[j])
925 mooseError("Species " + Moose::stringify(_species[i]) +
926 " is included as both a species and aux_species!\nA species can only be either "
927 "a nonlinear variable or an auxiliary variable, not both. Note that any species "
928 "included as an aux_species will be treated as an auxiliary variable and will "
929 "not have any source or sink terms applied to it (though it will be included as "
930 "a reactant in the source/sink terms of other nonlinear variables.)");
931 }
932 }
933
934 // Last we check for file names. This will be REQUIRED in future versions.
935 // If a file name does not exist, an error is thrown.
936 // txt, csv, and dat files are checked.
937 for (unsigned int i = 0; i < _num_eedf_reactions; ++i)
938 {
939 if (_is_identified[i])
940 {
941 std::string fileloc = getParam<std::string>("file_location") + "/" + _reaction_identifier[i];
942 if (file_exists(fileloc))
943 {
944 continue;
945 }
946
947 if (file_exists(fileloc + ".txt"))
948 {
949 _reaction_identifier[i] += ".txt";
950 continue;
951 }
952
953 if (file_exists(fileloc + ".csv"))
954 {
955 _reaction_identifier[i] += ".csv";
956 continue;
957 }
958
959 if (file_exists(fileloc + ".dat"))
960 {
961 _reaction_identifier[i] += ".dat";
962 continue;
963 }
964
965 // If we've come this far, the file does not exist. Error time.
966 mooseError("File " + fileloc +
967 " does not exist. \nMake sure the rate coefficient file exists and is spelled "
968 "correctly in the directory denoted by file_location.\nThe program "
969 "automatically checks for txt, csv, and dat files.\n(Note that if no "
970 "file_location parameter is added, the current directory is used.)");
971 }
