271template <typename T>
272void
273MooseObjectWarehouseBase<T>::addObject(std::shared_ptr<T> object,
274 THREAD_ID tid /*= 0*/,
275 bool /* recurse = true */)
276{
277 checkThreadID(tid);
278
279 // Stores object in list of all objects
280 _all_objects[tid].push_back(object);
281
282 // If enabled, store object in a list of all active
283 bool enabled = object->enabled();
284 if (enabled)
285 _active_objects[tid].push_back(object);
286
287 // Perform casts to the Block/BoundaryRestrictable
288 std::shared_ptr<BoundaryRestrictable> bnd =
289 std::dynamic_pointer_cast<BoundaryRestrictable>(object);
290 std::shared_ptr<BlockRestrictable> blk = std::dynamic_pointer_cast<BlockRestrictable>(object);
291
292 // Boundary Restricted
293 if (bnd && bnd->boundaryRestricted())
294 {
295 const std::set<BoundaryID> & ids = bnd->boundaryIDs();
296 for (std::set<BoundaryID>::const_iterator it = ids.begin(); it != ids.end(); ++it)
297 {
298 _all_boundary_objects[tid][*it].push_back(object);
299 if (enabled)
300 _active_boundary_objects[tid][*it].push_back(object);
301 }
302 }
303
304 // Block Restricted
305 else if (blk)
306 {
307 const std::set<SubdomainID> & ids =
308 blk->blockRestricted() ? blk->blockIDs() : blk->meshBlockIDs();
309 for (std::set<SubdomainID>::const_iterator it = ids.begin(); it != ids.end(); ++it)
310 {
311 _all_block_objects[tid][*it].push_back(object);
312 if (enabled)
313 _active_block_objects[tid][*it].push_back(object);
314 }
315
316 // Check variables
317 std::shared_ptr<Coupleable> c_ptr = std::dynamic_pointer_cast<Coupleable>(object);
318 if (c_ptr)
319 for (MooseVariableFieldBase * var : c_ptr->getCoupledMooseVars())
320 blk->checkVariable(*var);
321
322 const InputParameters & parameters = object->parameters();
323
324 SubProblem & problem = *parameters.get<SubProblem *>("_subproblem");
325
326 THREAD_ID tid = parameters.get<THREAD_ID>("_tid");
327
328 if (parameters.isParamValid("variable"))
329 {
330 // Try the scalar version first
331 std::string variable_name = parameters.getMooseType("variable");
332 if (variable_name == "")
333 // When using vector variables, we are only going to use the first one in the list at the
334 // interface level...
335 variable_name = parameters.getVecMooseType("variable")[0];
336
337 blk->checkVariable(problem.getVariable(
338 tid, variable_name, Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY));
339 }
340 }
341}

573void
574MooseObjectWarehouseBase<T>::sort(THREAD_ID tid /* = 0*/)
575{
576 checkThreadID(tid);
577

741template <typename T>

743MooseObjectWarehouseBase<T>::subdomainsCovered(std::set<SubdomainID> & subdomains_covered,
744 std::set<std::string> & unique_variables,
745 THREAD_ID tid /*=0*/) const
746{
747 for (const auto & object : _active_objects[tid])
748 unique_variables.insert(object->variable().name());
