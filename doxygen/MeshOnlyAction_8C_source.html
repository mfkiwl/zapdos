34void
35MeshOnlyAction::act()
36{
37 std::string mesh_file = _app.parameters().get<std::string>("mesh_only");
38 auto & mesh_ptr = _app.actionWarehouse().mesh();
39
40 // Print information about the mesh
41 _console << mesh_ptr->getMesh().get_info(/* verbosity = */ 2) << std::endl;
42
43 bool should_generate = false;
44
45 // If no argument specified or if the argument following --mesh-only starts
46 // with a dash, try to build an output filename based on the input mesh filename.
47 if (mesh_file.empty() || (mesh_file[0] == '-'))
48 should_generate = true;
49 // There's something following the --mesh-only flag, let's make an attempt to validate it.
50 // If we don't find a '.' or we DO find an equals sign, chances are this is not a file!
51 else if ((mesh_file.find('.') == std::string::npos || mesh_file.find('=') != std::string::npos))
52 {
53 mooseWarning("The --mesh-only option should be followed by a file name. Move it to the end of "
54 "your CLI args or follow it by another \"-\" argument.");
55 should_generate = true;
56 }
57
58 if (should_generate)
59 {
60 mesh_file = _app.parser().getPrimaryFileName();
61 size_t pos = mesh_file.find_last_of('.');
62
63 // Default to writing out an ExodusII mesh base on the input filename.
64 mesh_file = mesh_file.substr(0, pos) + "_in.e";
65 }
66
72 if (mesh_file.find(".e") + 2 == mesh_file.size())
73 {
74 TIME_SECTION("act", 1, "Writing Exodus");
75
76 auto & output_mesh = mesh_ptr->getMesh();
77 ExodusII_IO exio(output_mesh);
78
79 Exodus::setOutputDimensionInExodusWriter(exio, *mesh_ptr);
80
81 // Default to non-HDF5 output for wider compatibility
82 exio.set_hdf5_writing(false);
83
84 exio.write(mesh_file);
85
86 // Check if extra element integers should be outputted to Exodus file
87 unsigned int n_eeid = output_mesh.n_elem_integers();
88
89 // Iterate through all actions and see if `Outputs/output_extra_element_ids = true` in input
90 // file
91 const auto & output_actions = _app.actionWarehouse().getActionListByName("add_output");
92 // Truth of whether to output extra element ids is initially determined by whether
93 // there are extra element ids defined on the mesh
94 bool output_extra_ids = (n_eeid > 0);
95 bool restrict_element_id_names = false;
96 std::vector<std::string> element_id_names;
97 for (const auto & act : output_actions)
98 {
99 // Extract the Output action
100 AddOutputAction * action = dynamic_cast<AddOutputAction *>(act);
101 if (!action)
102 continue;
103
104 InputParameters & params = action->getObjectParams();
105 if (params.isParamSetByUser("output_extra_element_ids"))
106 {
107 // User has set output_extra_element_ids, truth of output_extra_ids determined by value of
108 // parameter
109 output_extra_ids = params.get<bool>("output_extra_element_ids");
110 if (output_extra_ids)
111 {
112 // `Outputs/extra_element_ids_to_output` sets a subset of extra element ids that should
113 // be outputted to Exodus
114 restrict_element_id_names = params.isParamValid("extra_element_ids_to_output");
115 if (restrict_element_id_names)
116 {
117 element_id_names = params.get<std::vector<std::string>>("extra_element_ids_to_output");
118 // Check which element id names actually are defined on the mesh, remove from
119 // element_id_names if they don't belong
120 for (auto it = element_id_names.begin(); it != element_id_names.end();)
121 {
122 // Erase contents of iterator and return iterator if element integer does not exist
123 if (!output_mesh.has_elem_integer(*it))
124 {
125 it = element_id_names.erase(it);
126 mooseWarning("Extra element id ",
127 *it,
128 " defined in Outputs/extra_element_ids_to_output "
129 "is not defined on the mesh and will be ignored.");
130 }
131 // Increment iterator if element integer exists
132 else
133 ++it;
134 }
135 }
136 }
137 break;
138 }
139 }
140
141 if (output_extra_ids)
142 {
143 // Retrieve extra element id names and associated data
144 const auto n_elem = output_mesh.n_elem();
145 std::vector<std::string> eeid_vars;
146 const auto n_eeid_to_output = restrict_element_id_names ? element_id_names.size() : n_eeid;
147 std::vector<Number> eeid_soln(n_elem * n_eeid_to_output);
148 unsigned int soln_index = 0;
149 for (unsigned int i = 0; i < n_eeid; i++)
150 {
151 const auto eeid_name = output_mesh.get_elem_integer_name(i);
152 // If `Outputs/extra_element_ids_to_output` is not set, output all ids to Exodus
153 // Otherwise only output if the extra id name is contained within
154 // `Outputs/extra_element_ids_to_output`
155 if (!restrict_element_id_names ||
156 (std::find(element_id_names.begin(), element_id_names.end(), eeid_name) !=
157 element_id_names.end()))
158 {
159 eeid_vars.push_back(output_mesh.get_elem_integer_name(i));
160 for (const auto & elem : output_mesh.element_ptr_range())
161 {
162 eeid_soln[soln_index] = (int)elem->get_extra_integer(i);
163 ++soln_index;
164 }
165 }
166 }
167
168 // Check size of output variables just in case none of the variables in
169 // `Outpus/extra_element_ids_to_output` are specified on the actual mesh
170 if (eeid_vars.size() > 0)
171 {
172 // Invoke ExodusII_IO_Helper to output extra element ids to Exodus file
173 auto & exio_helper = exio.get_exio_helper();
174
175 // Output empty timestep to Exodus file
176 int empty_timestep = 1;
177 Real default_time = 1.0;
178 exio_helper.write_timestep(empty_timestep, default_time);
179
180 // Write extra element id data to Exodus file
181 std::vector<std::set<subdomain_id_type>> vars_active_subdomains;
182 vars_active_subdomains.resize(n_eeid_to_output);
183 exio_helper.initialize_element_variables(eeid_vars, vars_active_subdomains);
184 exio_helper.write_element_values(
185 output_mesh, eeid_soln, empty_timestep, vars_active_subdomains);
186 }
187 else
188 mooseWarning(
189 "Outputs/output_extra_element_ids is set to true but no extra element ids are being "
190 "outputted. Please check extra element ids are properly defined on the mesh and any "
191 "variables specified in Outputs/extra_element_ids_to_output are spelled correctly.");
192 }
193 }
194
195 else if (mesh_file.find(".cpr") + 4 == mesh_file.size())
196 {
197 TIME_SECTION("act", 1, "Writing Checkpoint");
198
199 CheckpointIO io(mesh_ptr->getMesh(), true);
200 io.write(mesh_file);
201
202 // Write mesh metadata
203 if (processor_id() == 0)
204 {
205 const auto filenames = _app.writeRestartableMetaData(MooseApp::MESH_META_DATA, mesh_file);
206 Moose::out << "Mesh meta data written into "
207 << std::filesystem::absolute(filenames[0].parent_path()) << "." << std::endl;
208 }
209 }
210 else
211 {
212 // Just write the file using the name requested by the user.
213 mesh_ptr->getMesh().write(mesh_file);
214 }
215}
