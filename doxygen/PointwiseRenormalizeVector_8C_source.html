30PointwiseRenormalizeVector::PointwiseRenormalizeVector(const InputParameters & parameters)
31 : GeneralUserObject(parameters),
32 _mesh(_fe_problem.mesh()),
33 _var_names(getParam<std::vector<VariableName>>("v")),
34 _target_norm(getParam<Real>("norm"))
35{
36 const MooseVariableFieldBase * first_var = nullptr;
37 for (const auto & var_name : _var_names)
38 {
39 auto & var = _fe_problem.getVariable(0, var_name);
40 if (!first_var)
41 first_var = &var;
42 else
43 {
44 // check order and family for consistency
45 if (first_var->feType() != var.feType())
46 paramError("v", "All supplied variables must be of the same order and family.");
47 // check block restriction for consistency
48 if (!first_var->hasBlocks(var.blockIDs()) || !var.hasBlocks(first_var->blockIDs()))
49 paramError("v", "All supplied variables must have the same block restriction.");
50 }
51
52 if (_sys.number() != var.sys().system().number())
53 paramError("v", "Variables must be all in the non-linear system.");
54
55 if (var.isArray())
56 {
57 const auto & array_var = _fe_problem.getArrayVariable(0, var_name);
58 for (unsigned int p = 0; p < var.count(); ++p)
59 _var_numbers.push_back(_sys.system().variable_number(array_var.componentName(p)));
60 }
61 else
62 _var_numbers.push_back(_sys.system().variable_number(var_name));
63 }
64}

73void
74PointwiseRenormalizeVector::execute()
75{
76 auto & dof_map = _sys.system().get_dof_map();
77 const auto local_dof_begin = dof_map.first_dof();
78 const auto local_dof_end = dof_map.end_dof();
79
80 std::vector<std::vector<dof_id_type>> dof_indices(_var_numbers.size());
81 std::vector<Real> cache(_var_numbers.size());
82
83 for (const auto & elem : *_mesh.getActiveLocalElementRange())
84 {
85 // prepare variable dofs
86 for (const auto i : index_range(_var_numbers))
87 {
88 dof_map.dof_indices(elem, dof_indices[i], _var_numbers[i]);
89
90 // check that all vars have the same number of dofs
91 mooseAssert(dof_indices[i].size() == dof_indices[0].size(),
92 "All specified variables should have the same number of DOFs");
93 }
94
95 // iterate over current, old, and older solutions
96 for (const auto s : make_range(3))
97 if (_sys.hasSolutionState(s))
98 {
99 auto & solution = _sys.solutionState(s);
100
101 // loop over all DOFs
102 for (const auto j : index_range(dof_indices[0]))
103 {
104 // check if the first variable's DOFs are local (if they are all other variables should
105 // have local DOFS as well)
106 if (dof_indices[0][j] > local_dof_end || dof_indices[0][j] < local_dof_begin)
107 continue;
108
109 // compute current norm
110 Real norm = 0.0;
111 for (const auto i : index_range(_var_numbers))
112 norm += Utility::pow<2>(solution(dof_indices[i][j]));
113
114 if (norm == 0.0)
115 continue;
116 norm = std::sqrt(norm);
117
118 // renormalize
119 for (const auto i : index_range(_var_numbers))
120 solution.set(dof_indices[i][j], solution(dof_indices[i][j]) / norm * _target_norm);
121 }
122 }
123 }
124}
