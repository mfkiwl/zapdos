39ScalarSplineInterpolation::ScalarSplineInterpolation(const InputParameters & parameters)
40 : AuxScalarKernel(parameters),
41 _sampler_var(coupledScalarValue("sampler")),
42 _sampler_const(getParam<Real>("const_sampler")),
43 _sampling_format(getParam<std::string>("sampling_format")),
44 _use_time(getParam<bool>("use_time")),
45 _use_log(getParam<bool>("use_log")),
46 _scale_factor(getParam<Real>("scale_factor"))
47{
48 std::vector<Real> x_val;
49 std::vector<Real> y_val;
50 std::string file_name =
51 getParam<std::string>("file_location") + "/" + getParam<FileName>("property_file");
52 MooseUtils::checkFileReadable(file_name);
53 const char * charPath = file_name.c_str();
54 std::ifstream myfile(charPath);
55 Real value;
56
57 if (myfile.is_open())
58 {
59 while (myfile >> value)
60 {
61 x_val.push_back(value);
62 myfile >> value;
63 y_val.push_back(value);
64 }
65 myfile.close();
66 }
67 else
68 mooseError("Unable to open file");
69
70 _coefficient_interpolation.setData(x_val, y_val);
71}

73Real
74ScalarSplineInterpolation::computeValue()
75{
76 Real val;
77 if (isCoupledScalar("sampler"))
78 val = _coefficient_interpolation.sample(_sampler_var[_i]);
79 else if (!isCoupledScalar("sampler") && _use_time)
80 // val = _coefficient_interpolation_linear.sample(_t);
81 val = _coefficient_interpolation.sample(_t);
82 else
83 val = _coefficient_interpolation.sample(_sampler_const);
84
85 // Ensure positivity
86 if (val < 0.0)
87 {
88 val = 0.0;
89 }
90 val = val * _scale_factor;
91
92 if (_use_log && val > 0)
93 return log(val);
94 else if (_use_log && val <= 0)
95 return 1e-4;
96 else
97 return val;
98}
