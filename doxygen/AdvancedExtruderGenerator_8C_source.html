44
45 params.addRequiredParam<MeshGeneratorName>("input", "The mesh to extrude");
46
47 params.addClassDescription(
48 "Extrudes a 1D mesh into 2D, or a 2D mesh into 3D, can have a variable height for each "
49 "elevation, variable number of layers within each elevation, variable growth factors of "
50 "axial element sizes within each elevation and remap subdomain_ids, boundary_ids and element "
51 "extra integers within each elevation as well as interface boundaries between neighboring "
52 "elevation layers.");
53
54 params.addRequiredParam<std::vector<Real>>("heights", "The height of each elevation");
55
56 params.addRangeCheckedParam<std::vector<Real>>(
57 "biases", "biases>0.0", "The axial growth factor used for mesh biasing for each elevation.");
58

60 "num_layers", "The number of layers for each elevation - must be num_elevations in length!");

125AdvancedExtruderGenerator::AdvancedExtruderGenerator(const InputParameters & parameters)
126 : MeshGenerator(parameters),
127 _input(getMesh("input")),
128 _heights(getParam<std::vector<Real>>("heights")),
129 _biases(isParamValid("biases") ? getParam<std::vector<Real>>("biases")
130 : std::vector<Real>(_heights.size(), 1.0)),
131 _num_layers(getParam<std::vector<unsigned int>>("num_layers")),
132 _subdomain_swaps(getParam<std::vector<std::vector<subdomain_id_type>>>("subdomain_swaps")),
133 _boundary_swaps(getParam<std::vector<std::vector<boundary_id_type>>>("boundary_swaps")),
134 _elem_integer_names_to_swap(getParam<std::vector<std::string>>("elem_integer_names_to_swap")),
135 _elem_integers_swaps(
136 getParam<std::vector<std::vector<std::vector<dof_id_type>>>>("elem_integers_swaps")),
137 _direction(getParam<Point>("direction")),
138 _has_top_boundary(isParamValid("top_boundary")),
139 _top_boundary(isParamValid("top_boundary") ? getParam<boundary_id_type>("top_boundary") : 0),
140 _has_bottom_boundary(isParamValid("bottom_boundary")),
141 _bottom_boundary(isParamValid("bottom_boundary") ? getParam<boundary_id_type>("bottom_boundary")
142 : 0),
143 _upward_boundary_source_blocks(
144 isParamValid("upward_boundary_source_blocks")
145 ? getParam<std::vector<std::vector<subdomain_id_type>>>("upward_boundary_source_blocks")
146 : std::vector<std::vector<subdomain_id_type>>(_heights.size(),
147 std::vector<subdomain_id_type>())),
148 _upward_boundary_ids(
149 isParamValid("upward_boundary_ids")
150 ? getParam<std::vector<std::vector<boundary_id_type>>>("upward_boundary_ids")
151 : std::vector<std::vector<boundary_id_type>>(_heights.size(),
152 std::vector<boundary_id_type>())),
153 _downward_boundary_source_blocks(isParamValid("downward_boundary_source_blocks")
154 ? getParam<std::vector<std::vector<subdomain_id_type>>>(
155 "downward_boundary_source_blocks")
156 : std::vector<std::vector<subdomain_id_type>>(
157 _heights.size(), std::vector<subdomain_id_type>())),
158 _downward_boundary_ids(
159 isParamValid("downward_boundary_ids")
160 ? getParam<std::vector<std::vector<boundary_id_type>>>("downward_boundary_ids")
161 : std::vector<std::vector<boundary_id_type>>(_heights.size(),
162 std::vector<boundary_id_type>())),
163 _twist_pitch(getParam<Real>("twist_pitch"))
164{
165 if (!_direction.norm())
166 paramError("direction", "Must have some length!");
167
168 // Normalize it
169 _direction /= _direction.norm();
170
171 const auto num_elevations = _heights.size();
172
173 if (_num_layers.size() != num_elevations)
174 paramError("heights", "The length of 'heights' and 'num_layers' must be the same in ", name());
175
176 if (_subdomain_swaps.size() && (_subdomain_swaps.size() != num_elevations))
177 paramError("subdomain_swaps",
178 "If specified, 'subdomain_swaps' must be the same length as 'heights' in ",
179 name());
180
181 _subdomain_swap_pairs.resize(_subdomain_swaps.size());
182
183 // Reprocess the subdomain swaps to make pairs out of them so they are easier to use
184 for (unsigned int i = 0; i < _subdomain_swaps.size(); i++)
185 {
186 const auto & elevation_swaps = _subdomain_swaps[i];
187 auto & elevation_swap_pairs = _subdomain_swap_pairs[i];
188
189 if (elevation_swaps.size() % 2)
190 paramError("subdomain_swaps",
191 "Row ",
192 i + 1,
193 " of subdomain_swaps in ",
194 name(),
195 " does not contain an even number of entries! Num entries: ",
196 elevation_swaps.size());
197
198 for (unsigned int j = 0; j < elevation_swaps.size(); j += 2)
199 elevation_swap_pairs[elevation_swaps[j]] = elevation_swaps[j + 1];
200 }
201
202 if (_boundary_swaps.size() && (_boundary_swaps.size() != num_elevations))
203 paramError("boundary_swaps",
204 "If specified, 'boundary_swaps' must be the same length as 'heights' in ",
205 name());
206
207 _boundary_swap_pairs.resize(_boundary_swaps.size());
208
209 // Reprocess the boundary swaps to make pairs out of them so they are easier to use
210 for (unsigned int i = 0; i < _boundary_swaps.size(); i++)
211 {
212 const auto & elevation_bdry_swaps = _boundary_swaps[i];
213 auto & elevation_bdry_swap_pairs = _boundary_swap_pairs[i];
214
215 if (elevation_bdry_swaps.size() % 2)
216 paramError("boundary_swaps",
217 "Row ",
218 i + 1,
219 " of boundary_swaps in ",
220 name(),
221 " does not contain an even number of entries! Num entries: ",
222 elevation_bdry_swaps.size());
223
224 for (unsigned int j = 0; j < elevation_bdry_swaps.size(); j += 2)
225 elevation_bdry_swap_pairs[elevation_bdry_swaps[j]] = elevation_bdry_swaps[j + 1];
226 }
227
228 if (_elem_integers_swaps.size() &&
229 _elem_integers_swaps.size() != _elem_integer_names_to_swap.size())
230 paramError("elem_integers_swaps",
231 "If specified, 'elem_integers_swaps' must have the same length as the length of "
232 "'elem_integer_names_to_swap'.");
233
234 for (const auto & unit_elem_integers_swaps : _elem_integers_swaps)
235 if (unit_elem_integers_swaps.size() != num_elevations)
236 paramError("elem_integers_swaps",
237 "If specified, each element of 'elem_integers_swaps' must have the same length as "
238 "the length of 'heights'.");
239
240 _elem_integers_swap_pairs.resize(num_elevations * _elem_integer_names_to_swap.size());
241 // Reprocess the elem_integers_swaps to make pairs out of them so they are easier to use
242 for (unsigned int i = 0; i < _elem_integer_names_to_swap.size(); i++)
243 {
244 for (unsigned int j = 0; j < num_elevations; j++)
245 {
246 const auto & elevation_extra_swaps = _elem_integers_swaps[i][j];
247 auto & elevation_extra_swap_pairs = _elem_integers_swap_pairs[i * num_elevations + j];
248
249 if (elevation_extra_swaps.size() % 2)
250 paramError("elem_integers_swaps",
251 "Row ",
252 i * num_elevations + j + 1,
253 " of elem_integers_swaps in ",
254 name(),
255 " does not contain an even number of entries! Num entries: ",
256 elevation_extra_swaps.size());
257 for (unsigned int k = 0; k < elevation_extra_swaps.size(); k += 2)
258 elevation_extra_swap_pairs[elevation_extra_swaps[k]] = elevation_extra_swaps[k + 1];
259 }
260 }
261
262 bool has_negative_entry = false;
263 bool has_positive_entry = false;
264 for (const auto & h : _heights)
265 {
266 if (h > 0.0)
267 has_positive_entry = true;
268 else
269 has_negative_entry = true;
270 }
271
272 if (has_negative_entry && has_positive_entry)
273 paramError("heights", "Cannot have both positive and negative heights!");
274 if (_biases.size() != _heights.size())
275 paramError("biases", "Size of this parameter, if provided, must be the same as heights.");
276
277 if (_upward_boundary_source_blocks.size() != _upward_boundary_ids.size() ||
278 _upward_boundary_ids.size() != _heights.size())
279 paramError(
280 "upward_boundary_ids",
281 "This parameter must have the same length as upward_boundary_source_blocks and heights.");
282 for (unsigned int i = 0; i < _upward_boundary_source_blocks.size(); i++)
283 if (_upward_boundary_source_blocks[i].size() != _upward_boundary_ids[i].size())
284 paramError("upward_boundary_ids",
285 "Every element of this parameter must have the same length as the corrresponding "
286 "element of upward_boundary_source_blocks.");
287
288 if (_downward_boundary_source_blocks.size() != _downward_boundary_ids.size() ||
289 _downward_boundary_ids.size() != _heights.size())
290 paramError(
291 "downward_boundary_ids",
292 "This parameter must have the same length as downward_boundary_source_blocks and heights.");
293 for (unsigned int i = 0; i < _downward_boundary_source_blocks.size(); i++)
294 if (_downward_boundary_source_blocks[i].size() != _downward_boundary_ids[i].size())
295 paramError("downward_boundary_ids",
296 "Every element of this parameter must have the same length as the corrresponding "
297 "element of downward_boundary_source_blocks.");
298}

300std::unique_ptr<MeshBase>
301AdvancedExtruderGenerator::generate()
302{
303 // Note: bulk of this code originally from libmesh mesh_modification.C
304 // Original copyright: Copyright (C) 2002-2019 Benjamin S. Kirk, John W. Peterson, Roy H. Stogner
305 // Original license is LGPL so it can be used here.
306
307 auto mesh = buildMeshBaseObject();
308 mesh->set_mesh_dimension(_input->mesh_dimension() + 1);
309
310 // Check if the element integer names are existent in the input mesh.
311 for (unsigned int i = 0; i < _elem_integer_names_to_swap.size(); i++)
312 if (_input->has_elem_integer(_elem_integer_names_to_swap[i]))
313 _elem_integer_indices_to_swap.push_back(
314 _input->get_elem_integer_index(_elem_integer_names_to_swap[i]));
315 else
316 paramError("elem_integer_names_to_swap",
317 "Element ",
318 i + 1,
319 " of 'elem_integer_names_to_swap' in is not a valid extra element integer of the "
320 "input mesh.");
321
322 // prepare for transferring extra element integers from orignal mesh to the extruded mesh.
323 const unsigned int num_extra_elem_integers = _input->n_elem_integers();
324 std::vector<std::string> id_names;
325
326 for (unsigned int i = 0; i < num_extra_elem_integers; i++)
327 {
328 id_names.push_back(_input->get_elem_integer_name(i));
329 if (!mesh->has_elem_integer(id_names[i]))
330 mesh->add_elem_integer(id_names[i]);
331 }
332
333 // retreive subdomain/sideset/nodeset name maps
334 const auto & input_subdomain_map = _input->get_subdomain_name_map();
335 const auto & input_sideset_map = _input->get_boundary_info().get_sideset_name_map();
336 const auto & input_nodeset_map = _input->get_boundary_info().get_nodeset_name_map();
337
338 std::unique_ptr<MeshBase> input = std::move(_input);
339
340 // If we're using a distributed mesh... then make sure we don't have any remote elements hanging
341 // around
342 if (!input->is_serial())
343 mesh->delete_remote_elements();
344
345 unsigned int total_num_layers = std::accumulate(_num_layers.begin(), _num_layers.end(), 0);
346
347 auto total_num_elevations = _heights.size();
348
349 dof_id_type orig_elem = input->n_elem();
350 dof_id_type orig_nodes = input->n_nodes();
351
352#ifdef LIBMESH_ENABLE_UNIQUE_ID
353 unique_id_type orig_unique_ids = input->parallel_max_unique_id();
354#endif
355
356 unsigned int order = 1;
357
358 BoundaryInfo & boundary_info = mesh->get_boundary_info();
359 const BoundaryInfo & input_boundary_info = input->get_boundary_info();
360
361 // We know a priori how many elements we'll need
362 mesh->reserve_elem(total_num_layers * orig_elem);
363
364 // For straightforward meshes we need one or two additional layers per
365 // element.
366 if (input->elements_begin() != input->elements_end() &&
367 (*input->elements_begin())->default_order() == SECOND)
368 order = 2;
369 mesh->comm().max(order);
370
371 mesh->reserve_nodes((order * total_num_layers + 1) * orig_nodes);
372
373 // Container to catch the boundary IDs handed back by the BoundaryInfo object
374 std::vector<boundary_id_type> ids_to_copy;
375
376 Point old_distance;
377 Point current_distance;
378
379 for (const auto & node : input->node_ptr_range())
380 {
381 unsigned int current_node_layer = 0;
382
383 old_distance.zero();
384
385 // e is the elevation layer ordering
386 for (unsigned int e = 0; e < total_num_elevations; e++)
387 {
388 auto num_layers = _num_layers[e];
389
390 auto height = _heights[e];
391
392 auto bias = _biases[e];
393
394 // k is the element layer ordering within each elevation layer
395 for (unsigned int k = 0; k < order * num_layers + (e == 0 ? 1 : 0); ++k)
396 {
397 // For the first layer we don't need to move
398 if (e == 0 && k == 0)
399 current_distance.zero();
400 else
401 {
402 // Shift the previous position by a certain fraction of 'height' along the extrusion
403 // direction to get the new position.
404 auto layer_index = (k - (e == 0 ? 1 : 0)) / order + 1;
405
406 const auto step_size = MooseUtils::absoluteFuzzyEqual(bias, 1.0)
407 ? height / (Real)num_layers / (Real)order
408 : height * std::pow(bias, (Real)(layer_index - 1)) *
409 (1.0 - bias) /
410 (1.0 - std::pow(bias, (Real)(num_layers))) / (Real)order;
411
412 current_distance = old_distance + _direction * step_size;
413
414 // Handle helicoidal extrusion
415 if (!MooseUtils::absoluteFuzzyEqual(_twist_pitch, 0.))
416 {
417 // twist 1 should be 'normal' to the extruded shape
418 RealVectorValue twist1 = _direction.cross(*node);
419 // This happens for any node on the helicoidal extrusion axis
420 if (!MooseUtils::absoluteFuzzyEqual(twist1.norm(), .0))
421 twist1 /= twist1.norm();
422 const RealVectorValue twist2 = twist1.cross(_direction);
423
424 auto twist = (cos(2. * libMesh::pi * layer_index * step_size / _twist_pitch) -
425 cos(2. * libMesh::pi * (layer_index - 1) * step_size / _twist_pitch)) *
426 twist2 +
427 (sin(2. * libMesh::pi * layer_index * step_size / _twist_pitch) -
428 sin(2. * libMesh::pi * (layer_index - 1) * step_size / _twist_pitch)) *
429 twist1;
430 twist *= std::sqrt(node->norm_sq() + libMesh::Utility::pow<2>(_direction * (*node)));
431 current_distance += twist;
432 }
433 }
434
435 Node * new_node = mesh->add_point(*node + current_distance,
436 node->id() + (current_node_layer * orig_nodes),
437 node->processor_id());
438
439#ifdef LIBMESH_ENABLE_UNIQUE_ID
440 // Let's give the base of the extruded mesh the same
441 // unique_ids as the source mesh, in case anyone finds that
442 // a useful map to preserve.
443 const unique_id_type uid = (current_node_layer == 0)
444 ? node->unique_id()
445 : orig_unique_ids +
446 (current_node_layer - 1) * (orig_nodes + orig_elem) +
447 node->id();
448
449 new_node->set_unique_id(uid);
450#endif
451
452 input_boundary_info.boundary_ids(node, ids_to_copy);
453 if (_boundary_swap_pairs.empty())
454 boundary_info.add_node(new_node, ids_to_copy);
455 else
456 for (const auto & id_to_copy : ids_to_copy)
457 boundary_info.add_node(new_node,
458 _boundary_swap_pairs[e].count(id_to_copy)
459 ? _boundary_swap_pairs[e][id_to_copy]
460 : id_to_copy);
461
462 old_distance = current_distance;
463 current_node_layer++;
464 }
465 }
466 }
467
468 const std::set<boundary_id_type> & side_ids = input_boundary_info.get_side_boundary_ids();
469
470 boundary_id_type next_side_id =
471 side_ids.empty() ? 0 : cast_int<boundary_id_type>(*side_ids.rbegin() + 1);
472
473 // side_ids may not include ids from remote elements, in which case
474 // some processors may have underestimated the next_side_id; let's
475 // fix that.
476 input->comm().max(next_side_id);
477
478 for (const auto & elem : input->element_ptr_range())
479 {
480 const ElemType etype = elem->type();
481
482 // build_extrusion currently only works on coarse meshes
483 libmesh_assert(!elem->parent());
484
485 unsigned int current_layer = 0;
486
487 for (unsigned int e = 0; e != total_num_elevations; e++)
488 {
489 auto num_layers = _num_layers[e];
490
491 for (unsigned int k = 0; k != num_layers; ++k)
492 {
493 std::unique_ptr<Elem> new_elem;
494 bool isFlipped(false);
495 switch (etype)
496 {
497 case EDGE2:
498 {
499 new_elem = std::make_unique<Quad4>();
500 new_elem->set_node(0) =
501 mesh->node_ptr(elem->node_ptr(0)->id() + (current_layer * orig_nodes));
502 new_elem->set_node(1) =
503 mesh->node_ptr(elem->node_ptr(1)->id() + (current_layer * orig_nodes));
504 new_elem->set_node(2) =
505 mesh->node_ptr(elem->node_ptr(1)->id() + ((current_layer + 1) * orig_nodes));
506 new_elem->set_node(3) =
507 mesh->node_ptr(elem->node_ptr(0)->id() + ((current_layer + 1) * orig_nodes));
508
509 if (elem->neighbor_ptr(0) == remote_elem)
510 new_elem->set_neighbor(3, const_cast<RemoteElem *>(remote_elem));
511 if (elem->neighbor_ptr(1) == remote_elem)
512 new_elem->set_neighbor(1, const_cast<RemoteElem *>(remote_elem));
513
514 break;
515 }
516 case EDGE3:
517 {
518 new_elem = std::make_unique<Quad9>();
519 new_elem->set_node(0) =
520 mesh->node_ptr(elem->node_ptr(0)->id() + (2 * current_layer * orig_nodes));
521 new_elem->set_node(1) =
522 mesh->node_ptr(elem->node_ptr(1)->id() + (2 * current_layer * orig_nodes));
523 new_elem->set_node(2) =
524 mesh->node_ptr(elem->node_ptr(1)->id() + ((2 * current_layer + 2) * orig_nodes));
525 new_elem->set_node(3) =
526 mesh->node_ptr(elem->node_ptr(0)->id() + ((2 * current_layer + 2) * orig_nodes));
527 new_elem->set_node(4) =
528 mesh->node_ptr(elem->node_ptr(2)->id() + (2 * current_layer * orig_nodes));
529 new_elem->set_node(5) =
530 mesh->node_ptr(elem->node_ptr(1)->id() + ((2 * current_layer + 1) * orig_nodes));
531 new_elem->set_node(6) =
532 mesh->node_ptr(elem->node_ptr(2)->id() + ((2 * current_layer + 2) * orig_nodes));
533 new_elem->set_node(7) =
534 mesh->node_ptr(elem->node_ptr(0)->id() + ((2 * current_layer + 1) * orig_nodes));
535 new_elem->set_node(8) =
536 mesh->node_ptr(elem->node_ptr(2)->id() + ((2 * current_layer + 1) * orig_nodes));
537
538 if (elem->neighbor_ptr(0) == remote_elem)
539 new_elem->set_neighbor(3, const_cast<RemoteElem *>(remote_elem));
540 if (elem->neighbor_ptr(1) == remote_elem)
541 new_elem->set_neighbor(1, const_cast<RemoteElem *>(remote_elem));
542
543 break;
544 }
545 case TRI3:
546 {
547 new_elem = std::make_unique<Prism6>();
548 new_elem->set_node(0) =
549 mesh->node_ptr(elem->node_ptr(0)->id() + (current_layer * orig_nodes));
550 new_elem->set_node(1) =
551 mesh->node_ptr(elem->node_ptr(1)->id() + (current_layer * orig_nodes));
552 new_elem->set_node(2) =
553 mesh->node_ptr(elem->node_ptr(2)->id() + (current_layer * orig_nodes));
554 new_elem->set_node(3) =
555 mesh->node_ptr(elem->node_ptr(0)->id() + ((current_layer + 1) * orig_nodes));
556 new_elem->set_node(4) =
