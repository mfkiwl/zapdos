62template <bool is_ad>
63template <typename T>
64Real
65SideIntegralFunctorPostprocessorTempl<is_ad>::computeLocalContribution(const T & functor_arg) const
66{
67 // It's possible the functor is not defined on that side of the boundary
68 // We wont allow that case, unless explicitly requested by the user,
69 // as this means the sideset is reversed, but we will allow the case where
70 // both sides are defined
71 const bool has_elem = checkFunctorDefinedOnSideBlock();
72
73 if (has_elem)
74 return MetaPhysicL::raw_value(_prefactor(functor_arg, determineState()) *
75 _functor(functor_arg, determineState()));
76 else
77 {
78 if (!_partial_integral)
79 errorFunctorNotDefinedOnSideBlock();
80 return 0;
81 }
82}

92template <bool is_ad>
93bool
94SideIntegralFunctorPostprocessorTempl<is_ad>::checkFunctorDefinedOnSideBlock() const
95{
96 if (_functor.hasBlocks(_current_elem->subdomain_id()) &&
97 _prefactor.hasBlocks(_current_elem->subdomain_id()))
98 return true;
99 else
100 {
101#ifndef NDEBUG
102 const auto neighbor_ptr = _current_elem->neighbor_ptr(_current_side);
103 if (neighbor_ptr)
104 {
105 mooseAssert(_functor.hasBlocks(_current_elem->subdomain_id()) ||
106 _functor.hasBlocks(neighbor_ptr->subdomain_id()),
107 "Functor should be defined on at least one side of the boundary");
108 mooseAssert(_prefactor.hasBlocks(_current_elem->subdomain_id()) ||
109 _prefactor.hasBlocks(neighbor_ptr->subdomain_id()),
110 "Prefactor should be defined on at least one side of the boundary");
111 }
112#endif
113 return false;
114 }
115}
