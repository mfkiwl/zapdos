55ReactionSecondOrderEnergy::computeQpResidual()
56{

117Real
118ReactionSecondOrderEnergy::computeQpOffDiagJacobian(unsigned int jvar)
119{
120 if (_v_eq_u && !_w_eq_u)
121 {
122 if (jvar == _w_id)
123 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * _v[_qp] *
124 _phi[_j][_qp] * _threshold_energy;
125 else
126 return 0.0;
127 }
128 else if (!_v_eq_u && _w_eq_u)
129 {
130 if (jvar == _w_id)
131 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * _w[_qp] *
132 _phi[_j][_qp] * _threshold_energy;
133 else
134 return 0.0;
135 }
136 else if (!_v_eq_u && !_w_eq_u)
137 {
138 if (jvar == _v_id && jvar == _w_id)
139 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * 2.0 * _v[_qp] *
140 _phi[_j][_qp] * _threshold_energy;
141 else if (jvar == _v_id && jvar != _w_id)
142 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * _w[_qp] *
143 _phi[_j][_qp] * _threshold_energy;
144 else if (jvar != _v_id && jvar == _w_id)
145 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * _v[_qp] *
146 _phi[_j][_qp] * _threshold_energy;
147 else
148 return 0.0;
149 }
150 else
151 return 0.0;
