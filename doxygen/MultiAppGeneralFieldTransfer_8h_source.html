624
630 CachedData(const Cache & cache, const FunctionBase<Output> & backup, Real default_value)
631 : _cache(cache), _backup(backup.clone()), _default_value(default_value)
632 {
633 }
634
636 CachedData(const CachedData & primary)
637 : _cache(primary._cache),
638 _backup(primary._backup->clone()),
639 _default_value(primary._default_value)
640 {
641 }
642
643 void init_context(FEMContext &) {}
644
646 Output eval_at_node(const FEMContext &,
647 unsigned int /*i*/,
648 unsigned int /*elem_dim*/,

650 bool /*extra_hanging_dofs*/,
651 const Real /*time*/)
