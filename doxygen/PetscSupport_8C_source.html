137void
138setSolverOptions(const SolverParams & solver_params)
139{
140 // set PETSc options implied by a solve type
141 switch (solver_params._type)
142 {
143 case Moose::ST_PJFNK:
144 setSinglePetscOption("-snes_mf_operator");
145 setSinglePetscOption("-mat_mffd_type", stringify(solver_params._mffd_type));
146 break;
147
148 case Moose::ST_JFNK:
149 setSinglePetscOption("-snes_mf");
150 setSinglePetscOption("-mat_mffd_type", stringify(solver_params._mffd_type));
151 break;
152
153 case Moose::ST_NEWTON:
154 break;
155
156 case Moose::ST_FD:
157 setSinglePetscOption("-snes_fd");
158 break;
159
160 case Moose::ST_LINEAR:
161 setSinglePetscOption("-snes_type", "ksponly");
162 setSinglePetscOption("-snes_monitor_cancel");
163 break;
164 }
165
166 Moose::LineSearchType ls_type = solver_params._line_search;
167 if (ls_type == Moose::LS_NONE)
168 ls_type = Moose::LS_BASIC;
169
170 if (ls_type != Moose::LS_DEFAULT && ls_type != Moose::LS_CONTACT && ls_type != Moose::LS_PROJECT)
171 setSinglePetscOption("-snes_linesearch_type", stringify(ls_type));
172}

174void
175petscSetupDM(NonlinearSystemBase & nl)
176{
177 PetscErrorCode ierr;
178 PetscBool ismoose;
179 DM dm = LIBMESH_PETSC_NULLPTR;
180
181 // Initialize the part of the DM package that's packaged with Moose; in the PETSc source tree this
182 // call would be in DMInitializePackage()
183 ierr = DMMooseRegisterAll();
184 CHKERRABORT(nl.comm().get(), ierr);
185 // Create and set up the DM that will consume the split options and deal with block matrices.
186 PetscNonlinearSolver<Number> * petsc_solver =
187 dynamic_cast<PetscNonlinearSolver<Number> *>(nl.nonlinearSolver());
188 SNES snes = petsc_solver->snes();
189 // if there exists a DMMoose object, not to recreate a new one
190 ierr = SNESGetDM(snes, &dm);
191 CHKERRABORT(nl.comm().get(), ierr);
192 if (dm)
193 {
194 ierr = PetscObjectTypeCompare((PetscObject)dm, DMMOOSE, &ismoose);
195 CHKERRABORT(nl.comm().get(), ierr);
196 if (ismoose)
197 return;
198 }
199 ierr = DMCreateMoose(nl.comm().get(), nl, &dm);
200 CHKERRABORT(nl.comm().get(), ierr);
201 ierr = DMSetFromOptions(dm);
202 CHKERRABORT(nl.comm().get(), ierr);
203 ierr = DMSetUp(dm);
204 CHKERRABORT(nl.comm().get(), ierr);
205 ierr = SNESSetDM(snes, dm);
206 CHKERRABORT(nl.comm().get(), ierr);
207 ierr = DMDestroy(&dm);
208 CHKERRABORT(nl.comm().get(), ierr);
209 // We temporarily comment out this updating function because
210 // we lack an approach to check if the problem
211 // structure has been changed from the last iteration.
212 // The indices will be rebuilt for every timestep.
213 // TODO: figure out a way to check the structure changes of the
214 // matrix
215 // ierr = SNESSetUpdate(snes,SNESUpdateDMMoose);
216 // CHKERRABORT(nl.comm().get(),ierr);
217}

237void
238petscSetOptions(const PetscOptions & po, const SolverParams & solver_params)
239{
240#if PETSC_VERSION_LESS_THAN(3, 7, 0)
241 PetscOptionsClear();
242#else
243 PetscOptionsClear(LIBMESH_PETSC_NULLPTR);
244#endif
245
246 setSolverOptions(solver_params);
247
248 // Add any additional options specified in the input file
249 for (const auto & flag : po.flags)
250 setSinglePetscOption(flag.rawName().c_str());
251
252 // Add option pairs
253 for (auto & option : po.pairs)
254 setSinglePetscOption(option.first, option.second);
255
256 addPetscOptionsFromCommandline();
257}

275PetscErrorCode
276petscNonlinearConverged(SNES snes,
277 PetscInt it,
278 PetscReal xnorm,
279 PetscReal snorm,
280 PetscReal fnorm,
281 SNESConvergedReason * reason,
282 void * ctx)
283{
284 FEProblemBase & problem = *static_cast<FEProblemBase *>(ctx);
285 NonlinearSystemBase & system = problem.currentNonlinearSystem();
286
287 // Let's be nice and always check PETSc error codes.
288 PetscErrorCode ierr = 0;
289
290 // Temporary variables to store SNES tolerances. Usual C-style would be to declare
291 // but not initialize these... but it bothers me to leave anything uninitialized.
292 PetscReal atol = 0.; // absolute convergence tolerance
293 PetscReal rtol = 0.; // relative convergence tolerance
294 PetscReal stol = 0.; // convergence (step) tolerance in terms of the norm of the change in the
295 // solution between steps
296 PetscInt maxit = 0; // maximum number of iterations
297 PetscInt maxf = 0; // maximum number of function evaluations
298
299 // Ask the SNES object about its tolerances.
300 ierr = SNESGetTolerances(snes, &atol, &rtol, &stol, &maxit, &maxf);
301 CHKERRABORT(problem.comm().get(), ierr);
302
303 // Ask the SNES object about its divergence tolerance.
304 PetscReal divtol = 0.; // relative divergence tolerance
305#if !PETSC_VERSION_LESS_THAN(3, 8, 0)
306 ierr = SNESGetDivergenceTolerance(snes, &divtol);
307 CHKERRABORT(problem.comm().get(), ierr);
308#endif
309
310 // Get current number of function evaluations done by SNES.
311 PetscInt nfuncs = 0;
312 ierr = SNESGetNumberFunctionEvals(snes, &nfuncs);
313 CHKERRABORT(problem.comm().get(), ierr);
314
315 // Whether or not to force SNESSolve() take at least one iteration regardless of the initial
316 // residual norm
317#if !PETSC_VERSION_LESS_THAN(3, 8, 4)
318 PetscBool force_iteration = PETSC_FALSE;
319 ierr = SNESGetForceIteration(snes, &force_iteration);
320 CHKERRABORT(problem.comm().get(), ierr);
321
322 if (force_iteration && !(problem.getNonlinearForcedIterations()))
323 problem.setNonlinearForcedIterations(1);
324
325 if (!force_iteration && (problem.getNonlinearForcedIterations()))
326 {
327 ierr = SNESSetForceIteration(snes, PETSC_TRUE);
328 CHKERRABORT(problem.comm().get(), ierr);
329 }
330#endif
331
332 // See if SNESSetFunctionDomainError() has been called. Note:
333 // SNESSetFunctionDomainError() and SNESGetFunctionDomainError()
334 // were added in different releases of PETSc.
335 PetscBool domainerror;
336 ierr = SNESGetFunctionDomainError(snes, &domainerror);
337 CHKERRABORT(problem.comm().get(), ierr);
338 if (domainerror)
339 {
340 *reason = SNES_DIVERGED_FUNCTION_DOMAIN;
341 return 0;
342 }
343
344 // Error message that will be set by the FEProblemBase.
345 std::string msg;
346
347 // xnorm: 2-norm of current iterate
348 // snorm: 2-norm of current step
349 // fnorm: 2-norm of function at current iterate
350 MooseNonlinearConvergenceReason moose_reason =
351 problem.checkNonlinearConvergence(msg,
352 it,
353 xnorm,
354 snorm,
355 fnorm,
356 rtol,
357 divtol,
358 stol,
359 atol,
360 nfuncs,
361 maxf,
362 system._initial_residual_before_preset_bcs,
363 std::numeric_limits<Real>::max());
364
365 if (msg.length() > 0)
366#if !PETSC_VERSION_LESS_THAN(3, 17, 0)
367 PetscInfo(snes, "%s", msg.c_str());
368#else
369 PetscInfo(snes, msg.c_str());
370#endif
371
372 switch (moose_reason)
373 {
374 case MooseNonlinearConvergenceReason::ITERATING:
375 *reason = SNES_CONVERGED_ITERATING;
376 break;
377
378 case MooseNonlinearConvergenceReason::CONVERGED_FNORM_ABS:
379 *reason = SNES_CONVERGED_FNORM_ABS;
380 break;
381
382 case MooseNonlinearConvergenceReason::CONVERGED_FNORM_RELATIVE:
383 *reason = SNES_CONVERGED_FNORM_RELATIVE;
384 break;
385
386 case MooseNonlinearConvergenceReason::DIVERGED_DTOL:
387#if !PETSC_VERSION_LESS_THAN(3, 8, 0) // A new convergence enum in PETSc 3.8
388 *reason = SNES_DIVERGED_DTOL;
389#endif
390 break;
391
392 case MooseNonlinearConvergenceReason::CONVERGED_SNORM_RELATIVE:
393 *reason = SNES_CONVERGED_SNORM_RELATIVE;
394 break;
395
396 case MooseNonlinearConvergenceReason::DIVERGED_FUNCTION_COUNT:
397 *reason = SNES_DIVERGED_FUNCTION_COUNT;
398 break;
399
400 case MooseNonlinearConvergenceReason::DIVERGED_FNORM_NAN:
401 *reason = SNES_DIVERGED_FNORM_NAN;
402 break;
403
404 case MooseNonlinearConvergenceReason::DIVERGED_LINE_SEARCH:
405 *reason = SNES_DIVERGED_LINE_SEARCH;
406 break;
407
408 case MooseNonlinearConvergenceReason::DIVERGED_NL_RESIDUAL_PINGPONG:
409 *reason = SNES_DIVERGED_LOCAL_MIN;
410 break;
411 }
412
413 return 0;
414}

477void
478petscSetKSPDefaults(FEProblemBase & problem, KSP ksp)
479{
480 auto & es = problem.es();
481
482 PetscReal rtol = es.parameters.get<Real>("linear solver tolerance");
483 PetscReal atol = es.parameters.get<Real>("linear solver absolute tolerance");
484
485 // MOOSE defaults this to -1 for some dumb reason
486 if (atol < 0)
487 atol = 1e-50;
488
489 PetscReal maxits = es.parameters.get<unsigned int>("linear solver maximum iterations");
490
491 // 1e100 is because we don't use divtol currently
492 KSPSetTolerances(ksp, rtol, atol, 1e100, maxits);
493
494 petscSetDefaultPCSide(problem, ksp);
495
496 petscSetDefaultKSPNormType(problem, ksp);
497}

499void
500petscSetDefaults(FEProblemBase & problem)
501{
502 for (auto nl_index : make_range(problem.numNonlinearSystems()))
503 {
504 // dig out PETSc solver
505 NonlinearSystemBase & nl = problem.getNonlinearSystemBase(nl_index);
506 PetscNonlinearSolver<Number> * petsc_solver =
507 dynamic_cast<PetscNonlinearSolver<Number> *>(nl.nonlinearSolver());
508 SNES snes = petsc_solver->snes();
509 KSP ksp;
510 SNESGetKSP(snes, &ksp);
511
512 SNESSetMaxLinearSolveFailures(snes, 1000000);
513
514 // In 3.0.0, the context pointer must actually be used, and the
515 // final argument to KSPSetConvergenceTest() is a pointer to a
516 // routine for destroying said private data context. In this case,
517 // we use the default context provided by PETSc in addition to
518 // a few other tests.
519 {
520 auto ierr =
521 SNESSetConvergenceTest(snes, petscNonlinearConverged, &problem, LIBMESH_PETSC_NULLPTR);
522 CHKERRABORT(nl.comm().get(), ierr);
523 }
524
525 petscSetKSPDefaults(problem, ksp);
526 }

529void
530storePetscOptions(FEProblemBase & fe_problem, const InputParameters & params)
531{
532 // Note: Options set in the Preconditioner block will override those set in the Executioner block
533 if (params.isParamValid("solve_type") && !params.isParamValid("_use_eigen_value"))
534 {
535 // Extract the solve type
536 const std::string & solve_type = params.get<MooseEnum>("solve_type");
537 fe_problem.solverParams()._type = Moose::stringToEnum<Moose::SolveType>(solve_type);
538 }
539
540 if (params.isParamValid("line_search"))
541 {
542 MooseEnum line_search = params.get<MooseEnum>("line_search");
543 if (fe_problem.solverParams()._line_search == Moose::LS_INVALID || line_search != "default")
544 {
545 Moose::LineSearchType enum_line_search =
546 Moose::stringToEnum<Moose::LineSearchType>(line_search);
547 fe_problem.solverParams()._line_search = enum_line_search;
548 if (enum_line_search == LS_CONTACT || enum_line_search == LS_PROJECT)
549 for (auto nl_index : make_range(fe_problem.numNonlinearSystems()))
550 {
551 NonlinearImplicitSystem * nl_system = dynamic_cast<NonlinearImplicitSystem *>(
552 &fe_problem.getNonlinearSystemBase(nl_index).system());
553 if (!nl_system)
554 mooseError("You've requested a line search but you must be solving an EigenProblem. "
555 "These two things are not consistent.");
556 PetscNonlinearSolver<Real> * petsc_nonlinear_solver =
557 dynamic_cast<PetscNonlinearSolver<Real> *>(nl_system->nonlinear_solver.get());
558 if (!petsc_nonlinear_solver)
559 mooseError("Currently the MOOSE line searches all use Petsc, so you "
560 "must use Petsc as your non-linear solver.");
561 petsc_nonlinear_solver->linesearch_object =
562 std::make_unique<ComputeLineSearchObjectWrapper>(fe_problem);
563 }
564 }
565 }
566
567 if (params.isParamValid("mffd_type"))
568 {
569 MooseEnum mffd_type = params.get<MooseEnum>("mffd_type");
570 fe_problem.solverParams()._mffd_type = Moose::stringToEnum<Moose::MffdType>(mffd_type);
571 }
572
573 // The parameters contained in the Action
574 const auto & petsc_options = params.get<MultiMooseEnum>("petsc_options");
575 const auto & petsc_pair_options =
576 params.get<MooseEnumItem, std::string>("petsc_options_iname", "petsc_options_value");
577
578 // A reference to the PetscOptions object that contains the settings that will be used in the
579 // solve
580 Moose::PetscSupport::PetscOptions & po = fe_problem.getPetscOptions();
581
582 // First process the single petsc options/flags
583 processPetscFlags(petsc_options, po);
584
585 // Then process the option-value pairs
586 processPetscPairs(petsc_pair_options, fe_problem.mesh().dimension(), po);
587}

589void
590processPetscFlags(const MultiMooseEnum & petsc_flags, PetscOptions & po)
591{
592 // Update the PETSc single flags
593 for (const auto & option : petsc_flags)
594 {
601 if (option == "-log_summary" || option == "-log_view")
602 mooseError("The PETSc option \"-log_summary\" or \"-log_view\" can only be used on the "
603 "command line. Please "
604 "remove it from the input file");
605
606 // Warn about superseded PETSc options (Note: -snes is not a REAL option, but people used it in
607 // their input files)
608 else
609 {
610 std::string help_string;
611 if (option == "-snes" || option == "-snes_mf" || option == "-snes_mf_operator")
612 help_string = "Please set the solver type through \"solve_type\".";
613 else if (option == "-ksp_monitor")
614 help_string = "Please use \"Outputs/print_linear_residuals=true\"";
615
616 if (help_string != "")
617 mooseWarning("The PETSc option ",
618 std::string(option),
619 " should not be used directly in a MOOSE input file. ",
620 help_string);
621 }
622
623 // Update the stored items, but do not create duplicates
624 if (!po.flags.contains(option))
625 po.flags.push_back(option);
626 }
627}

629void
630processPetscPairs(const std::vector<std::pair<MooseEnumItem, std::string>> & petsc_pair_options,
631 const unsigned int mesh_dimension,
632 PetscOptions & po)
633{
634 // the boolean in these pairs denote whether the user has specified any of the reason flags in the
635 // input file
636 std::array<std::pair<bool, std::string>, 2> reason_flags = {
637 {std::make_pair(false, "-snes_converged_reason"),
638 std::make_pair(false, "-ksp_converged_reason")}};
639
640 for (auto & reason_flag : reason_flags)
641 if (po.flags.contains(reason_flag.second))
642 // We register the reason option as already existing
643 reason_flag.first = true;
644
645 // Setup the name value pairs
646 bool boomeramg_found = false;
647 bool strong_threshold_found = false;
648#if !PETSC_VERSION_LESS_THAN(3, 7, 0)
649 bool superlu_dist_found = false;
650 bool fact_pattern_found = false;
651 bool tiny_pivot_found = false;
652#endif
653 std::string pc_description = "";
654#if !PETSC_VERSION_LESS_THAN(3, 12, 0)
655 // If users use HMG, we would like to set
656 bool hmg_found = false;
657 bool matptap_found = false;
658 bool hmg_strong_threshold_found = false;
659#endif
660#if !PETSC_VERSION_LESS_THAN(3, 19, 2)
661 // Check for if users have set the options_left option
662 bool options_left_set = false;
663#endif
664 std::vector<std::pair<std::string, std::string>> new_options;
665
666 for (const auto & option : petsc_pair_options)
667 {
668 new_options.clear();
669
670 // Do not add duplicate settings
671 if (MooseUtils::findPair(po.pairs, option.first, MooseUtils::Any) == po.pairs.end())
672 {
673#if !PETSC_VERSION_LESS_THAN(3, 9, 0)
674 if (option.first == "-pc_factor_mat_solver_package")
675 new_options.emplace_back("-pc_factor_mat_solver_type", option.second);
676#else
677 if (option.first == "-pc_factor_mat_solver_type")
678 new_options.push_back("-pc_factor_mat_solver_package", option.second);
679#endif
680
681 // Look for a pc description
682 if (option.first == "-pc_type" || option.first == "-pc_sub_type" ||
683 option.first == "-pc_hypre_type")
684 pc_description += option.second + ' ';
685
686#if !PETSC_VERSION_LESS_THAN(3, 12, 0)
687 if (option.first == "-pc_type" && option.second == "hmg")
688 hmg_found = true;
689
690 // MPIAIJ for PETSc 3.12.0: -matptap_via
691 // MAIJ for PETSc 3.12.0: -matmaijptap_via
692 // MPIAIJ for PETSc 3.13 to 3.16: -matptap_via, -matproduct_ptap_via
693 // MAIJ for PETSc 3.13 to 3.16: -matproduct_ptap_via
694 // MPIAIJ for PETSc 3.17 and higher: -matptap_via, -mat_product_algorithm
695 // MAIJ for PETSc 3.17 and higher: -mat_product_algorithm
696#if !PETSC_VERSION_LESS_THAN(3, 17, 0)
697 if (hmg_found && (option.first == "-matptap_via" || option.first == "-matmaijptap_via" ||
698 option.first == "-matproduct_ptap_via"))
699 new_options.emplace_back("-mat_product_algorithm", option.second);
700#elif !PETSC_VERSION_LESS_THAN(3, 13, 0)
701 if (hmg_found && (option.first == "-matptap_via" || option.first == "-matmaijptap_via"))
702 new_options.emplace_back("-matproduct_ptap_via", option.second);
703#else
704 if (hmg_found && (option.first == "-matproduct_ptap_via"))
705 {
706 new_options.emplace_back("-matptap_via", option.second);
707 new_options.emplace_back("-matmaijptap_via", option.second);
708 }
709#endif
710
711 if (option.first == "-matptap_via" || option.first == "-matmaijptap_via" ||
712 option.first == "-matproduct_ptap_via" || option.first == "-mat_product_algorithm")
713 matptap_found = true;
714
715 // For 3D problems, we need to set this 0.7
716 if (option.first == "-hmg_inner_pc_hypre_boomeramg_strong_threshold")
717 hmg_strong_threshold_found = true;
718#endif
719 // This special case is common enough that we'd like to handle it for the user.
720 if (option.first == "-pc_hypre_type" && option.second == "boomeramg")
721 boomeramg_found = true;
722 if (option.first == "-pc_hypre_boomeramg_strong_threshold")
723 strong_threshold_found = true;
724#if !PETSC_VERSION_LESS_THAN(3, 7, 0)
725 if ((option.first == "-pc_factor_mat_solver_package" ||
726 option.first == "-pc_factor_mat_solver_type") &&
727 option.second == "superlu_dist")
728 superlu_dist_found = true;
729 if (option.first == "-mat_superlu_dist_fact")
730 fact_pattern_found = true;
731 if (option.first == "-mat_superlu_dist_replacetinypivot")
732 tiny_pivot_found = true;
733#endif
734
735#if !PETSC_VERSION_LESS_THAN(3, 19, 2)
736 if (option.first == "-options_left")
737 options_left_set = true;
738#endif
739
740 if (!new_options.empty())
741 std::copy(new_options.begin(), new_options.end(), std::back_inserter(po.pairs));
742 else
743 po.pairs.push_back(option);
744 }
745 else
746 {
747 for (unsigned int j = 0; j < po.pairs.size(); j++)
748 if (option.first == po.pairs[j].first)
749 po.pairs[j].second = option.second;
750 }
751 }
752
753#if !PETSC_VERSION_LESS_THAN(3, 14, 0)
754 for (const auto & reason_flag : reason_flags)
755 // Was the option already found in PetscOptions::flags? Or does it exist in PetscOptions::pairs
756 // as an iname already? If not, then we add our flag
757 if (!reason_flag.first && (std::find_if(po.pairs.begin(),
758 po.pairs.end(),
759 [&reason_flag](auto & pair) {
760 return pair.first == reason_flag.second;
761 }) == po.pairs.end()))
762 po.pairs.emplace_back(reason_flag.second, "::failed");
763#endif
764
765 // When running a 3D mesh with boomeramg, it is almost always best to supply a strong threshold
766 // value. We will provide that for the user here if they haven't supplied it themselves.
767 if (boomeramg_found && !strong_threshold_found && mesh_dimension == 3)
768 {
769 po.pairs.emplace_back("-pc_hypre_boomeramg_strong_threshold", "0.7");
770 pc_description += "strong_threshold: 0.7 (auto)";
771 }
772
773#if !PETSC_VERSION_LESS_THAN(3, 12, 0)
774 if (hmg_found && !hmg_strong_threshold_found && mesh_dimension == 3)
775 {
776 po.pairs.emplace_back("-hmg_inner_pc_hypre_boomeramg_strong_threshold", "0.7");
777 pc_description += "strong_threshold: 0.7 (auto)";
778 }
779
780 // Default PETSc PtAP takes too much memory, and it is not quite useful
781 // Let us switch to use new algorithm
782 if (hmg_found && !matptap_found)
783 {
784#if !PETSC_VERSION_LESS_THAN(3, 17, 0)
785 po.pairs.emplace_back("-mat_product_algorithm", "allatonce");
786#elif !PETSC_VERSION_LESS_THAN(3, 13, 0)
787 po.pairs.emplace_back("-matproduct_ptap_via", "allatonce");
788#else
789 po.pairs.emplace_back("-matptap_via", "allatonce");
790 po.pairs.emplace_back("-matmaijptap_via", "allatonce");
791#endif
792 }
793#endif
794
795#if !PETSC_VERSION_LESS_THAN(3, 7, 0)
796 // In PETSc-3.7.{0--4}, there is a bug when using superlu_dist, and we have to use
797 // SamePattern_SameRowPerm, otherwise we use whatever we have in PETSc
798 if (superlu_dist_found && !fact_pattern_found)
799 {
800 po.pairs.emplace_back("-mat_superlu_dist_fact",
801#if PETSC_VERSION_LESS_THAN(3, 7, 5)
802 "SamePattern_SameRowPerm");
803 pc_description += "mat_superlu_dist_fact: SamePattern_SameRowPerm ";
804#else
805 "SamePattern");
806 pc_description += "mat_superlu_dist_fact: SamePattern ";
807#endif
808 }
809
810 // restore this superlu option
811 if (superlu_dist_found && !tiny_pivot_found)
812 {
813 po.pairs.emplace_back("-mat_superlu_dist_replacetinypivot", "1");
814 pc_description += " mat_superlu_dist_replacetinypivot: true ";
815 }
816#endif
817 // Set Preconditioner description
818 po.pc_description = pc_description;
819
820 // Turn off default options_left warnings added in 3.19.3 pre-release for all PETSc builds
821 // (PETSc commit: 59f199a7), unless the user has set a preference.
822#if !PETSC_VERSION_LESS_THAN(3, 19, 2)
823 if (!options_left_set && !po.flags.contains("-options_left"))
824 po.pairs.emplace_back("-options_left", "0");
825#endif
826}

939void
940colorAdjacencyMatrix(PetscScalar * adjacency_matrix,
941 unsigned int size,
942 unsigned int colors,
943 std::vector<unsigned int> & vertex_colors,
944 const char * coloring_algorithm)
945{
946 // Mat A will be a dense matrix from the incoming data structure
947 Mat A;
948 MatCreate(MPI_COMM_SELF, &A);
949 MatSetSizes(A, size, size, size, size);
950 MatSetType(A, MATSEQDENSE);
951 // PETSc requires a non-const data array to populate the matrix
952 MatSeqDenseSetPreallocation(A, adjacency_matrix);
953 MatAssemblyBegin(A, MAT_FINAL_ASSEMBLY);
954 MatAssemblyEnd(A, MAT_FINAL_ASSEMBLY);
955
956 // Convert A to a sparse matrix
957 MatConvert(A,
958 MATAIJ,
959#if PETSC_VERSION_LESS_THAN(3, 7, 0)
960 MAT_REUSE_MATRIX,
961#else
962 MAT_INPLACE_MATRIX,
963#endif
964 &A);
965
966 ISColoring iscoloring;
967 MatColoring mc;
968 MatColoringCreate(A, &mc);
969 MatColoringSetType(mc, coloring_algorithm);
970 MatColoringSetMaxColors(mc, static_cast<PetscInt>(colors));
971
972 // Petsc normally colors by distance two (neighbors of neighbors), we just want one
973 MatColoringSetDistance(mc, 1);
974 MatColoringSetFromOptions(mc);
975 MatColoringApply(mc, &iscoloring);
976
977 PetscInt nn;
978 IS * is;
979#if PETSC_RELEASE_LESS_THAN(3, 12, 0)
980 ISColoringGetIS(iscoloring, &nn, &is);
981#else
982 ISColoringGetIS(iscoloring, PETSC_USE_POINTER, &nn, &is);
983#endif
984
985 if (nn > static_cast<PetscInt>(colors))
986 throw std::runtime_error("Not able to color with designated number of colors");
987
988 for (int i = 0; i < nn; i++)
989 {
990 PetscInt isize;
991 const PetscInt * indices;
992 ISGetLocalSize(is[i], &isize);
993 ISGetIndices(is[i], &indices);
994 for (int j = 0; j < isize; j++)
995 {
996 mooseAssert(indices[j] < static_cast<PetscInt>(vertex_colors.size()), "Index out of bounds");
997 vertex_colors[indices[j]] = i;
998 }
999 ISRestoreIndices(is[i], &indices);
1000 }
1001
1002 MatDestroy(&A);
1003 MatColoringDestroy(&mc);
1004 ISColoringDestroy(&iscoloring);
1005}
