53void
54Steady::execute()
55{
56 if (_app.isRecovering())
57 {
58 _console << "\nCannot recover steady solves!\nExiting...\n" << std::endl;
59 return;
60 }
61
62 _time_step = 0;
63 _time = _time_step;
64 _problem.outputStep(EXEC_INITIAL);
65 _time = _system_time;
66
67 preExecute();
68
69 _problem.advanceState();
70
71 // first step in any steady state solve is always 1 (preserving backwards compatibility)
72 _time_step = 1;
73
74#ifdef LIBMESH_ENABLE_AMR
75
76 // Define the refinement loop
77 unsigned int steps = _problem.adaptivity().getSteps();
78 for (unsigned int r_step = 0; r_step <= steps; r_step++)
79 {
80#endif // LIBMESH_ENABLE_AMR
81 _problem.timestepSetup();
82
83 _last_solve_converged = _fixed_point_solve->solve();
84
85 if (!lastSolveConverged())
86 {
87 _console << "Aborting as solve did not converge" << std::endl;
88 break;
89 }
90
91 _problem.computeIndicators();
92 _problem.computeMarkers();
93
94 // need to keep _time in sync with _time_step to get correct output
95 _time = _time_step;
96 _problem.outputStep(EXEC_TIMESTEP_END);
97 _time = _system_time;
98
99#ifdef LIBMESH_ENABLE_AMR
100 if (r_step != steps)
101 {
102 _problem.adaptMesh();
103 }
104
105 _time_step++;
106 }
107#endif
108
109 {
110 TIME_SECTION("final", 1, "Executing Final Objects")
111 _problem.execMultiApps(EXEC_FINAL);
112 _problem.finalizeMultiApps();
113 _problem.postExecute();
114 _problem.execute(EXEC_FINAL);
115 _time = _time_step;
116 _problem.outputStep(EXEC_FINAL);
117 _time = _system_time;
118 }
119
120 postExecute();
121}
