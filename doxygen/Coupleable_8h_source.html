1709Coupleable::getVarHelper(const std::string & var_name_in, unsigned int comp)

1713
1714 // First check for supplied name
1715 if (!checkVar(var_name, comp, 0))

1718 auto it = _new_to_deprecated_coupled_vars.find(var_name);
1719 if (it == _new_to_deprecated_coupled_vars.end())
1720 return nullptr;
1721 else

1724 if (checkVar(deprecated_name, comp, 0))
1725 name_to_use = deprecated_name;

1728 }

1732

1734 "Trying to get a coupled var " << name_to_use << " that doesn't exist");
1735
1736 if (auto coupled_var = dynamic_cast<T *>(coupled_vars_it->second[comp]))
1737 return coupled_var;
1738 else
1739 {
1740 for (auto & var : _coupled_standard_moose_vars)
1741 if (var->name() == name_to_use)
1742 mooseError("The named variable is a standard variable, try a "
1743 "'coupled[Value/Gradient/Dot/etc]...' function instead");
1744 for (auto & var : _coupled_vector_moose_vars)
1745 if (var->name() == name_to_use)
1746 mooseError("The named variable is a vector variable, try a "
1747 "'coupledVector[Value/Gradient/Dot/etc]...' function instead");

1755 "used does not support. Try using the functor system routines instead.");
