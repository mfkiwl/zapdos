45ADReal
46HagelaarElectronBC::computeQpResidual()
47{
48 if (_normals[_qp] * -1.0 * -_grad_potential[_qp] > 0.0)
49 {
50 _a = 1.0;
51 }
52 else
53 {
54 _a = 0.0;
55 }
56
57 _v_thermal =
58 std::sqrt(8 * _e[_qp] * 2.0 / 3 * std::exp(_mean_en[_qp] - _u[_qp]) / (M_PI * _massem[_qp]));
59
60 return _test[_i][_qp] * _r_units * (1. - _r) / (1. + _r) *
61 (-(2 * _a - 1) * _muem[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_u[_qp]) *
62 _normals[_qp] +
63 0.5 * _v_thermal * std::exp(_u[_qp]));
64}
