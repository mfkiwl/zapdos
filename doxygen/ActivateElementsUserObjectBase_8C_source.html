48void
49ActivateElementsUserObjectBase::setNewBoundayName()
50{
51 mooseAssert(!_expand_boundary_name.empty(), "Expanded boundary name is empty");
52 // add the new boundary and get its boundary id
53 _boundary_ids = _mesh.getBoundaryIDs(_expand_boundary_name, true);
54 mooseAssert(!_boundary_ids.empty(), "Boundary ID is empty.");
55 _mesh.setBoundaryName(_boundary_ids[0], _expand_boundary_name[0]);
56 _mesh.getMesh().get_boundary_info().sideset_name(_boundary_ids[0]) = _expand_boundary_name[0];
57 _mesh.getMesh().get_boundary_info().nodeset_name(_boundary_ids[0]) = _expand_boundary_name[0];
58
59 auto displaced_problem = _fe_problem.getDisplacedProblem();
60 if (displaced_problem)
61 {
62 _disp_boundary_ids = displaced_problem->mesh().getBoundaryIDs(_expand_boundary_name, true);
63 mooseAssert(!_disp_boundary_ids.empty(), "Boundary ID in the displaced mesh is empty");
64
65 displaced_problem->mesh().setBoundaryName(_disp_boundary_ids[0], _expand_boundary_name[0]);
66 displaced_problem->mesh().getMesh().get_boundary_info().sideset_name(_disp_boundary_ids[0]) =
67 _expand_boundary_name[0];
68 displaced_problem->mesh().getMesh().get_boundary_info().nodeset_name(_disp_boundary_ids[0]) =
69 _expand_boundary_name[0];
70 }

73void
74ActivateElementsUserObjectBase::execute()
75{
76 if (isElementActivated() && _current_elem->subdomain_id() != _active_subdomain_id &&
77 _current_elem->subdomain_id() != _inactive_subdomain_id)
78 {
79 /*
80 _current_elem subdomain id is not assignable
81 create a copy of this element from MooseMesh
82 */
83 dof_id_type ele_id = _current_elem->id();
84 Elem * ele = _mesh.elemPtr(ele_id);
85
86 // Add element to the activate subdomain
87 ele->subdomain_id() = _active_subdomain_id;
88
89 // Reassign element in the reference mesh while using a displaced mesh
90 auto displaced_problem = _fe_problem.getDisplacedProblem();
91 if (displaced_problem)
92 {
93 Elem * disp_ele = displaced_problem->mesh().elemPtr(ele_id);
94 disp_ele->subdomain_id() = _active_subdomain_id;
95 }
96
97 // Save the newly activated element id and node for updating boundary info later
98 _newly_activated_elem.insert(ele_id);
99 for (unsigned int i = 0; i < ele->n_nodes(); ++i)
100 _newly_activated_node.insert(ele->node_id(i));
101 }
102}

104void
105ActivateElementsUserObjectBase::finalize()
106{
107 /*
108 Synchronize ghost element subdomain ID
109 Note: this needs to be done before updating boundary info because
110 updating boundary requires the updated element subdomain ids
111 */
112 SyncSubdomainIds sync(_mesh.getMesh());
113 Parallel::sync_dofobject_data_by_id(_mesh.getMesh().comm(),
114 _mesh.getMesh().elements_begin(),
115 _mesh.getMesh().elements_end(),
116 sync);
117 // Update boundary info
118 updateBoundaryInfo(_mesh);
119
120 // Similarly for the displaced mesh
121 auto displaced_problem = _fe_problem.getDisplacedProblem();
122 if (displaced_problem)
123 {
124 SyncSubdomainIds sync_mesh(displaced_problem->mesh().getMesh());
125 Parallel::sync_dofobject_data_by_id(displaced_problem->mesh().getMesh().comm(),
126 displaced_problem->mesh().getMesh().elements_begin(),
127 displaced_problem->mesh().getMesh().elements_end(),
128 sync_mesh);
129 updateBoundaryInfo(displaced_problem->mesh());
130 }
131
132 // Reinit equation systems
133 _fe_problem.meshChanged();
134
135 // Get storage ranges for the newly activated elements and boundary nodes
136 ConstElemRange & elem_range = *this->getNewlyActivatedElementRange();
137 ConstBndNodeRange & bnd_node_range = *this->getNewlyActivatedBndNodeRange();
138
139 // Apply initial condition for the newly activated elements
140 initSolutions(elem_range, bnd_node_range);
141
142 // Initialize stateful material properties for the newly activated elements
143 _fe_problem.initElementStatefulProps(elem_range, false);
144
145 // Clear the list
146 _newly_activated_elem.clear();
147 _newly_activated_node.clear();
148
149 _node_to_remove_from_bnd.clear();
150}

152void
153ActivateElementsUserObjectBase::getNodesToRemoveFromBnd(std::set<dof_id_type> & remove_set,
154 std::set<dof_id_type> & add_set)
155{
156 // get the difference between the remove_set and the add_set,
157 // save the difference in _node_to_remove_from_bnd
158 int sz = remove_set.size() + add_set.size();
159 std::vector<dof_id_type> v(sz);
160 std::vector<dof_id_type>::iterator it = std::set_difference(
161 remove_set.begin(), remove_set.end(), add_set.begin(), add_set.end(), v.begin());
162 v.resize(it - v.begin());
163 _node_to_remove_from_bnd.clear();
164 for (auto id : v)
165 _node_to_remove_from_bnd.insert(id);
166}

177void
178ActivateElementsUserObjectBase::updateBoundaryInfo(MooseMesh & mesh)
179{
180 // save the removed ghost sides and associated nodes to sync across processors
181 std::unordered_map<processor_id_type, std::vector<std::pair<dof_id_type, unsigned int>>>
182 ghost_sides_to_remove;
183 std::unordered_map<processor_id_type, std::vector<dof_id_type>> ghost_nodes_to_remove;
184
185 // save nodes are added and removed
186 std::set<dof_id_type> add_nodes, remove_nodes;
187
188 for (auto ele_id : _newly_activated_elem)
189 {
190 Elem * ele = mesh.elemPtr(ele_id);
191 for (auto s : ele->side_index_range())
192 {
193 Elem * neighbor_ele = ele->neighbor_ptr(s);
194 if (neighbor_ele == nullptr)
195 {
196 // add this side to boundary
197 mesh.getMesh().get_boundary_info().add_side(ele, s, _boundary_ids[0]);
198 insertNodeIdsOnSide(ele, s, add_nodes);
199 }
200 else
201 {
202 if (neighbor_ele->subdomain_id() != _active_subdomain_id &&
203 neighbor_ele->subdomain_id() != _inactive_subdomain_id)
204 {
205 // add this side to boundary
206 mesh.getMesh().get_boundary_info().add_side(ele, s, _boundary_ids[0]);
207 insertNodeIdsOnSide(ele, s, add_nodes);
208 }
209 else
210 {
211 // remove this side from the boundary
212 mesh.getMesh().get_boundary_info().remove_side(ele, s);
213 insertNodeIdsOnSide(ele, s, remove_nodes);
214
215 // remove the neighbor side from the boundary
216 unsigned int neighbor_s = neighbor_ele->which_neighbor_am_i(ele);
217 mesh.getMesh().get_boundary_info().remove_side(neighbor_ele, neighbor_s);
218 insertNodeIdsOnSide(neighbor_ele, neighbor_s, remove_nodes);
219
220 if (neighbor_ele->processor_id() != this->processor_id())
221 ghost_sides_to_remove[neighbor_ele->processor_id()].emplace_back(neighbor_ele->id(),
222 neighbor_s);
223 }
224 }
225 }
226 }
227 // make sure to remove nodes that are not in the add list
228 getNodesToRemoveFromBnd(remove_nodes, add_nodes);
229 for (auto node_id : _node_to_remove_from_bnd)
230 mesh.getMesh().get_boundary_info().remove_node(mesh.nodePtr(node_id), _boundary_ids[0]);
231
232 // synchronize boundary information across processors
233 push_boundary_side_info(mesh, ghost_sides_to_remove);
234 push_boundary_node_info(mesh, ghost_nodes_to_remove);
235 mesh.getMesh().get_boundary_info().parallel_sync_side_ids();
236 mesh.getMesh().get_boundary_info().parallel_sync_node_ids();
237 mesh.update();
238}

240void
241ActivateElementsUserObjectBase::push_boundary_side_info(
242 MooseMesh & mesh,
243 std::unordered_map<processor_id_type, std::vector<std::pair<dof_id_type, unsigned int>>> &
244 elems_to_push)
245{
246 auto elem_action_functor =
247 [&mesh, this](processor_id_type,
248 const std::vector<std::pair<dof_id_type, unsigned int>> & received_elem)
249 {
250 // remove the side
251 for (const auto & pr : received_elem)
252 mesh.getMesh().get_boundary_info().remove_side(
253 mesh.getMesh().elem_ptr(pr.first), pr.second, this->getExpandedBoundaryID());
254 };
255
256 Parallel::push_parallel_vector_data(
257 mesh.getMesh().get_boundary_info().comm(), elems_to_push, elem_action_functor);
258}

280ConstElemRange *
281ActivateElementsUserObjectBase::getNewlyActivatedElementRange()
282{
283 // deletes the object first
284 _activated_elem_range.reset();
285
286 // create a vector of the newly activated elements
287 std::vector<Elem *> elems;
288 for (auto elem_id : _newly_activated_elem)
289 elems.push_back(_mesh.elemPtr(elem_id));
290
291 // Make some fake element iterators defining this vector of
292 // elements
293 Elem * const * elempp = const_cast<Elem * const *>(elems.data());
294 Elem * const * elemend = elempp + elems.size();
295
296 const auto elems_begin =
297 MeshBase::const_element_iterator(elempp, elemend, Predicates::NotNull<Elem * const *>());
298
299 const auto elems_end =
300 MeshBase::const_element_iterator(elemend, elemend, Predicates::NotNull<Elem * const *>());
301 if (!_activated_elem_range)
302 _activated_elem_range = std::make_unique<ConstElemRange>(elems_begin, elems_end);
303
304 return _activated_elem_range.get();
305}

344ConstNodeRange *
345ActivateElementsUserObjectBase::getNewlyActivatedNodeRange()
346{
347 // deletes the object first
348 _activated_node_range.reset();
349
350 // create a vector of the newly activated nodes
351 std::vector<const Node *> nodes;
352 for (auto elem_id : _newly_activated_elem)
353 {
354 const Node * const * elem_nodes = _mesh.elemPtr(elem_id)->get_nodes();
355 unsigned int n_nodes = _mesh.elemPtr(elem_id)->n_nodes();
356 for (unsigned int n = 0; n < n_nodes; ++n)
357 {
358 // check if all the elements connected to this node are newly activated
359 const Node * nd = elem_nodes[n];
360 if (isNewlyActivated(nd))
361 nodes.push_back(nd);
362 }
363 }
364
365 // Make some fake node iterators defining this vector of
366 // nodes
367 Node * const * nodepp = const_cast<Node * const *>(nodes.data());
368 Node * const * nodeend = nodepp + nodes.size();
369
370 const auto nodes_begin =
371 MeshBase::const_node_iterator(nodepp, nodeend, Predicates::NotNull<Node * const *>());
372
373 const auto nodes_end =
374 MeshBase::const_node_iterator(nodeend, nodeend, Predicates::NotNull<Node * const *>());
375
376 if (!_activated_node_range)
377 _activated_node_range = std::make_unique<ConstNodeRange>(nodes_begin, nodes_end);
378
379 return _activated_node_range.get();

382bool
383ActivateElementsUserObjectBase::isNewlyActivated(const Node * nd)
384{
385 const auto & node_to_elem_map = _mesh.nodeToElemMap();
386 auto node_to_elem_pair = node_to_elem_map.find(nd->id());
387 if (node_to_elem_pair != node_to_elem_map.end())
388 {
389 const std::vector<dof_id_type> & connected_ele_ids = node_to_elem_pair->second;
390 for (auto connected_ele_id : connected_ele_ids)
391 {
392 // check the connected elements
393 if (_mesh.elemPtr(connected_ele_id)->subdomain_id() == _inactive_subdomain_id)
394 return false;
395 if (_mesh.elemPtr(connected_ele_id)->subdomain_id() == _active_subdomain_id &&
396 std::find(_newly_activated_elem.begin(), _newly_activated_elem.end(), connected_ele_id) ==
397 _newly_activated_elem.end())
398 return false;
399 }
400 }
401 return true;
402}

404void
405ActivateElementsUserObjectBase::initSolutions(ConstElemRange & elem_range,
406 ConstBndNodeRange & bnd_node_range)
407{
408 // project initial condition to the current solution
409 _fe_problem.projectInitialConditionOnCustomRange(elem_range, bnd_node_range);
410
411 auto & nl = _fe_problem.getNonlinearSystemBase(_sys.number());
412 NumericVector<Number> & current_solution = *nl.system().current_local_solution;
413 NumericVector<Number> & old_solution = nl.solutionOld();
414 NumericVector<Number> & older_solution = nl.solutionOlder();
415
416 NumericVector<Number> & current_aux_solution =
417 *_fe_problem.getAuxiliarySystem().system().current_local_solution;
418 NumericVector<Number> & old_aux_solution = _fe_problem.getAuxiliarySystem().solutionOld();
419 NumericVector<Number> & older_aux_solution = _fe_problem.getAuxiliarySystem().solutionOlder();
420
421 DofMap & dof_map = nl.dofMap();
422 DofMap & dof_map_aux = _fe_problem.getAuxiliarySystem().dofMap();
423
424 std::set<dof_id_type> dofs, dofs_aux;
425 // get dofs for the newly added elements
426 for (auto & elem : elem_range)
427 {
428 std::vector<dof_id_type> di, di_aux;
429 dof_map.dof_indices(elem, di);
430 dof_map_aux.dof_indices(elem, di_aux);
431 for (unsigned int i = 0; i < di.size(); ++i)
432 dofs.insert(di[i]);
433 for (unsigned int i = 0; i < di_aux.size(); ++i)
434 dofs_aux.insert(di_aux[i]);
435
436 di.clear();
437 di_aux.clear();
438 }
439
440 // update solutions
441 for (auto dof : dofs)
442 {
443 old_solution.set(dof, current_solution(dof));
444 older_solution.set(dof, current_solution(dof));
445 }
446 // update aux solutions
447 for (auto dof_aux : dofs_aux)
448 {
449 old_aux_solution.set(dof_aux, current_aux_solution(dof_aux));
450 older_aux_solution.set(dof_aux, current_aux_solution(dof_aux));
451 }
452
453 dofs.clear();
454 dofs_aux.clear();
455
456 current_solution.close();
457 old_solution.close();
458 older_solution.close();
459
460 current_aux_solution.close();
461 old_aux_solution.close();
462 older_aux_solution.close();
463
464 _fe_problem.restoreSolutions();
465}
