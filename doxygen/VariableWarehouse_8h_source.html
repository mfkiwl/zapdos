258template <typename T>
259MooseVariableFV<T> *
260VariableWarehouse::getFVVariable(const std::string & var_name)
261{
262 auto it = _fv_vars_by_name.find(var_name);
263 if (it == _fv_vars_by_name.end())
264 mooseError("Requested variable ",
265 var_name,
266 " doesn't exist as a finite volume variable in the warehouse.");
267
268 return it->second;
269}
