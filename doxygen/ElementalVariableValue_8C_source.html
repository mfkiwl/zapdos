42void
43ElementalVariableValue::execute()
44{
45 _value = 0;
46 if (_element && (_element->processor_id() == processor_id()))
47 {
48 _fe_problem.setCurrentSubdomainID(_element, 0);
49 _subproblem.prepare(_element, _tid);
50 _subproblem.reinitElem(_element, _tid);
51
52 MooseVariableField<Real> & var = static_cast<MooseVariableField<Real> &>(
53 _subproblem.getActualFieldVariable(_tid, _var_name));
54 const VariableValue & u = var.sln();
55 unsigned int n = u.size();
56 for (unsigned int i = 0; i < n; i++)
57 _value += u[i];
58 _value /= n;
59 }
60}
