45ParsedNodeTransformGenerator::ParsedNodeTransformGenerator(const InputParameters & parameters)
46 : MeshGenerator(parameters), FunctionParserUtils<false>(parameters), _input(getMesh("input"))
47{
48 for (const auto i : index_range(_functions))
49 {
50 _functions[i] = std::make_shared<SymFunction>();
51
52 // set FParser internal feature flags
53 setParserFeatureFlags(_functions[i]);
54
55 // add the constant expressions
56 addFParserConstants(_functions[i],
57 getParam<std::vector<std::string>>("constant_names"),
58 getParam<std::vector<std::string>>("constant_expressions"));
59
60 // parse function
61 if (_functions[i]->Parse(getParam<ParsedFunctionExpression>(_func_name[i]), "x,y,z") >= 0)
62 paramError(_func_name[i], "Invalid function: ", _functions[i]->ErrorMsg());
63 }
64
65 _func_params.resize(3);
66}

68std::unique_ptr<MeshBase>
69ParsedNodeTransformGenerator::generate()
70{
71 std::unique_ptr<MeshBase> mesh = std::move(_input);
72
73 for (auto & node : mesh->node_ptr_range())
74 {
75 for (const auto i : make_range(3))
76 _func_params[i] = (*node)(i);
77
78 for (const auto i : make_range(3))
79 (*node)(i) = evaluate(_functions[i], _func_name[i]);
80 }
81
82 return mesh;
