64Real
65EEDFReactionLog::computeQpJacobian()
66{
67 if (_var.number() == _em_id)
68 {
69 Real actual_mean_en = std::exp(_mean_en[_qp] - _em[_qp]);
70 Real d_k_d_em = _d_k_d_actual_mean_en[_qp] * actual_mean_en * -_phi[_j][_qp];
71
72 return -_test[_i][_qp] * std::exp(_em[_qp] + _target[_qp]) * d_k_d_em * _coefficient;
73 }
74 else if (_var.number() == _target_id)
75 {
76 return -_test[_i][_qp] * std::exp(_em[_qp] + _target[_qp]) * _reaction_coeff[_qp] *
77 _coefficient * _phi[_j][_qp];
78 }
79 else
80 return 0;
81}

84EEDFReactionLog::computeQpOffDiagJacobian(unsigned int jvar)
85{
86 // Define multiplication factor (determining if product or reactant)
87
88 Real actual_mean_en = std::exp(_mean_en[_qp] - _em[_qp]);
89
90 Real d_k_d_mean_en = _d_k_d_actual_mean_en[_qp] * actual_mean_en * _phi[_j][_qp];
91 Real d_k_d_em = _d_k_d_actual_mean_en[_qp] * actual_mean_en * -_phi[_j][_qp];
92
93 if (jvar == _mean_en_id)
94 return -_test[_i][_qp] * std::exp(_em[_qp] + _target[_qp]) * d_k_d_mean_en * _coefficient;
95
96 else if (jvar == _em_id)
97 return -_test[_i][_qp] * std::exp(_em[_qp] + _target[_qp]) * _coefficient *
98 (_reaction_coeff[_qp] * _phi[_j][_qp] + d_k_d_em);
99 else if (jvar == _target_id)
100 return -_test[_i][_qp] * std::exp(_em[_qp] + _target[_qp]) * _reaction_coeff[_qp] *
101 _coefficient * _phi[_j][_qp];
