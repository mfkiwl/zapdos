46void
47ComputeNodalKernelsThread::pre()
48{
49 _num_cached = 0;
50
51 if (!_tags.size() || _tags.size() == _fe_problem.numVectorTags(Moose::VECTOR_TAG_RESIDUAL))
52 _nkernel_warehouse = &_nodal_kernels;
53 else if (_tags.size() == 1)
54 _nkernel_warehouse = &(_nodal_kernels.getVectorTagObjectWarehouse(*(_tags.begin()), _tid));
55 else
56 _nkernel_warehouse = &(_nodal_kernels.getVectorTagsObjectWarehouse(_tags, _tid));
57}

59void
60ComputeNodalKernelsThread::onNode(ConstNodeRange::const_iterator & node_it)
61{
62 const Node * node = *node_it;
63
64 std::set<const NodalKernelBase *> nks_executed;
65
66 // prepare variables
67 for (auto * var : _aux_sys._nodal_vars[_tid])
68 var->prepareAux();
69
70 _fe_problem.reinitNode(node, _tid);
71
72 const std::set<SubdomainID> & block_ids = _aux_sys.mesh().getNodeBlockIds(*node);
73 for (const auto & block : block_ids)
74 if (_nkernel_warehouse->hasActiveBlockObjects(block, _tid))
75 {
76 std::set<TagID> needed_fe_var_vector_tags;
77 _nkernel_warehouse->updateBlockFEVariableCoupledVectorTagDependency(
78 block, needed_fe_var_vector_tags, _tid);
79 _fe_problem.setActiveFEVariableCoupleableVectorTags(needed_fe_var_vector_tags, _tid);
80
81 const auto & objects = _nkernel_warehouse->getActiveBlockObjects(block, _tid);
82 for (const auto & nodal_kernel : objects)
83 if (nks_executed.emplace(nodal_kernel.get()).second)
84 nodal_kernel->computeResidual();
85 }
86
87 _num_cached++;
88
89 if (_num_cached == 20) // Cache 20 nodes worth before adding into the residual
90 {
91 _num_cached = 0;
92 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
93 _fe_problem.addCachedResidual(_tid);
94 }
95}

102void
103ComputeNodalKernelsThread::printGeneralExecutionInformation() const
104{
105 if (!_fe_problem.shouldPrintExecution(_tid) || !_nkernel_warehouse->hasActiveObjects())
106 return;
107
108 const auto & console = _fe_problem.console();
109 const auto & execute_on = _fe_problem.getCurrentExecuteOnFlag();
110 console << "[DBG] Computing nodal kernels contribution to residual on nodes on " << execute_on
111 << std::endl;
112 console << "[DBG] Nodes at boundaries between blocks will execute lower block ID first"
113 << std::endl;
114 console << "[DBG] Ordering of the nodal kernels on nodes they are defined on:" << std::endl;
115 console << _nkernel_warehouse->activeObjectsToFormattedString() << std::endl;
116}
