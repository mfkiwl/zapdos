50Real
51Reactant3BodyScalar::computeQpJacobian()
52{
53 if (_v_eq_u && _w_eq_u)
54 {
55 return -_stoichiometric_coeff * _rate_coefficient[_i] * 3.0 * std::pow(_u[_i], 2.0);
56 }
57 else if (_v_eq_u && !_w_eq_u)
58 {
59 return -_stoichiometric_coeff * _rate_coefficient[_i] * 2.0 * _u[_i] * _w[_i];
60 }
61 else if (!_v_eq_u && _w_eq_u)
62 {
63 return -_stoichiometric_coeff * _rate_coefficient[_i] * 2.0 * _u[_i] * _v[_i];
64 }
65 else
66 {
67 return -_stoichiometric_coeff * _rate_coefficient[_i] * _v[_i] * _w[_i];
68 }
69}

71Real
72Reactant3BodyScalar::computeQpOffDiagJacobian(unsigned int jvar)
73{
74 if (isCoupledScalar("v") && isCoupledScalar("w"))
75 {
76 if (!_v_eq_u && !_w_eq_u)
77 {
78 if (_v_var != _w_var)
79 {
80 if (jvar == _v_var)
81 return -_stoichiometric_coeff * _rate_coefficient[_i] * _u[_i] * _w[_i];
82 else if (jvar == _w_var)
83 return -_stoichiometric_coeff * _rate_coefficient[_i] * _u[_i] * _v[_i];
84 else
85 return 0.0;
86 }
87 else
88 {
89 if (jvar == _v_var)
90 return -_stoichiometric_coeff * _rate_coefficient[_i] * _u[_i] * 2.0 * _v[_i];
91 else
92 return 0.0;
93 }
94 }
95 else
96 return 0.0;
97 }
98 else if (isCoupledScalar("v") && !isCoupledScalar("w"))
99 {
100 if (jvar == _v_var && !_v_eq_u)
101 {
102 return -_stoichiometric_coeff * _rate_coefficient[_i] * _u[_i] * _w[_i];
103 }
104 else
105 return 0.0;
106 }
107 else if (!isCoupledScalar("v") && isCoupledScalar("w"))
108 {
109 if (jvar == _w_var && !_w_eq_u)
