67void
68CentralDifference::computeTimeDerivatives()
69{
70 if (!_sys.solutionUDot())
71 mooseError("CentralDifference: Time derivative of solution (`u_dot`) is not stored. Please "
72 "set uDotRequested() to true in FEProblemBase before requesting `u_dot`.");
73
74 if (!_sys.solutionUDotDot())
75 mooseError("CentralDifference: Time derivative of solution (`u_dotdot`) is not stored. Please "
76 "set uDotDotRequested() to true in FEProblemBase before requesting `u_dot`.");
77
78 // Declaring u_dot and u_dotdot
79 auto & u_dot = *_sys.solutionUDot();
80 auto & u_dotdot = *_sys.solutionUDotDot();
81
82 // Computing derivatives
83 computeTimeDerivativeHelper(
84 u_dot, u_dotdot, _solution_old, _solution_older, _solution_old_old_old);
85
86 // make sure _u_dotdot and _u_dot are in good state
87 u_dotdot.close();
88 u_dot.close();
89
90 // used for Jacobian calculations
91 _du_dot_du = 1.0 / (2 * _dt);
92 _du_dotdot_du = 1.0 / (_dt * _dt);
93
94 // Computing udotdot "factor"
95 // u_dotdot_factor = u_dotdot - (u - u_old)/dt^2 = (u - 2* u_old + u_older - u + u_old) / dt^2
96 // u_dotdot_factor = (u_older - u_old)/dt^2
97 if (_sys.hasVector(_u_dotdot_factor_tag)) // so that we don't do this in the aux system
98 {
99 auto & u_dotdot_factor = _sys.getVector(_u_dotdot_factor_tag);
100 u_dotdot_factor = _sys.solutionOlder();
101 u_dotdot_factor -= _sys.solutionOld();
102 u_dotdot_factor *= 1.0 / (_dt * _dt);
103 u_dotdot_factor.close();
104 }
105
106 // Computing udot "factor"
107 // u_dot_factor = u_dot - (u - u_old)/2/dt = (u - u_older)/ 2/ dt - (u - u_old)/2/dt
108 // u_dot_factor = (u_old - u_older)/2/dt
109 if (_sys.hasVector(_u_dot_factor_tag)) // so that we don't do this in the aux system
110 {
111 auto & u_dot_factor = _sys.getVector(_u_dot_factor_tag);
112 u_dot_factor = _sys.solutionOld();
113 u_dot_factor -= _sys.solutionOlder();
114 u_dot_factor *= 1.0 / (2.0 * _dt);
115 u_dot_factor.close();
116 }
117}
