85template <bool is_ad>
86void
87ParsedMaterialHelper<is_ad>::functionParse(
88 const std::string & function_expression,
89 const std::vector<std::string> & constant_names,
90 const std::vector<std::string> & constant_expressions,
91 const std::vector<std::string> & mat_prop_expressions,
92 const std::vector<PostprocessorName> & postprocessor_names,
93 const std::vector<std::string> & tol_names,
94 const std::vector<Real> & tol_values)
95{
96 // build base function object
97 _func_F = std::make_shared<SymFunction>();
98
99 // set FParser internal feature flags
100 setParserFeatureFlags(_func_F);
101
102 // initialize constants
103 addFParserConstants(_func_F, constant_names, constant_expressions);
104
105 // add further constants coming from default value coupling
106 if (_map_mode == VariableNameMappingMode::USE_PARAM_NAMES)
107 for (const auto & acd : _arg_constant_defaults)
108 if (!_func_F->AddConstant(acd, _pars.defaultCoupledValue(acd)))
109 mooseError("Invalid constant name in parsed function object");
110
111 // set variable names based on map_mode
112 switch (_map_mode)
113 {
114 case VariableNameMappingMode::USE_MOOSE_NAMES:
115 for (unsigned int i = 0; i < _nargs; ++i)
116 _symbol_names[i] = _arg_names[i];
117 break;
118
119 case VariableNameMappingMode::USE_PARAM_NAMES:
120 for (unsigned i = 0; i < _nargs; ++i)
121 {
122 if (_arg_param_numbers[i] < 0)
123 _symbol_names[i] = _arg_param_names[i];
124 else
125 _symbol_names[i] = _arg_param_names[i] + std::to_string(_arg_param_numbers[i]);
126 }
127 break;
128
129 default:
130 mooseError("Unknown variable mapping mode.");
131 }
132
133 // tolerance vectors
134 if (tol_names.size() != tol_values.size())
135 mooseError("The parameter vectors tol_names and tol_values must have equal length.");
136
137 // set tolerances
138 _tol.resize(_nargs);
139 for (unsigned int i = 0; i < _nargs; ++i)
140 {
141 _tol[i] = -1.0;
142
143 // for every argument look through the entire tolerance vector to find a match
144 for (MooseIndex(tol_names) j = 0; j < tol_names.size(); ++j)
145 if (_symbol_names[i] == tol_names[j])
146 {
147 _tol[i] = tol_values[j];
148 break;
149 }
150 }
151
152 // get all material properties
153 unsigned int nmat_props = mat_prop_expressions.size();
154 _mat_prop_descriptors.resize(nmat_props);
155 for (unsigned int i = 0; i < nmat_props; ++i)
156 {
157 // parse the material property parameter entry into a FunctionMaterialPropertyDescriptor
158 _mat_prop_descriptors[i] = FunctionMaterialPropertyDescriptor<is_ad>(
159 mat_prop_expressions[i], this, _error_on_missing_material_properties);
160
161 // get the fparser symbol name for the new material property
162 _symbol_names.push_back(_mat_prop_descriptors[i].getSymbolName());
163 }
164
165 // get all coupled postprocessors
166 for (const auto & pp : postprocessor_names)
167 {
168 _postprocessor_values.push_back(&this->getPostprocessorValueByName(pp));
169 _symbol_names.push_back(pp);
170 }
171
172 // build 'variables' argument for fparser
173 std::string variables = Moose::stringify(_symbol_names);
174
175 // build the base function
176 if (_func_F->Parse(function_expression, variables) >= 0)
177 mooseError("Invalid function\n",
178 function_expression,
179 '\n',
180 variables,
181 "\nin ParsedMaterialHelper.\n",
182 _func_F->ErrorMsg());
183
184 // create parameter passing buffer
185 _func_params.resize(_nargs + nmat_props + _postprocessor_values.size());
186
187 // perform next steps (either optimize or take derivatives and then optimize)
188
189 // let rank 0 do the work first to populate caches
190 if (_communicator.rank() != 0)
191 _communicator.barrier();
192
193 functionsPostParse();
194
195 // wait for ranks > 0 to catch up
196 if (_communicator.rank() == 0)
197 _communicator.barrier();
198}

218template <bool is_ad>
219void
220ParsedMaterialHelper<is_ad>::computeQpProperties()
221{
222 // fill the parameter vector, apply tolerances
223 for (unsigned int i = 0; i < _nargs; ++i)
224 {
225 if (_tol[i] < 0.0)
226 _func_params[i] = (*_args[i])[_qp];
227 else
228 {
229 auto a = (*_args[i])[_qp];
230 _func_params[i] = a < _tol[i] ? _tol[i] : (a > 1.0 - _tol[i] ? 1.0 - _tol[i] : a);
231 }
232 }
233
234 // insert material property values
235 auto nmat_props = _mat_prop_descriptors.size();
236 for (MooseIndex(_mat_prop_descriptors) i = 0; i < nmat_props; ++i)
237 _func_params[i + _nargs] = _mat_prop_descriptors[i].value()[_qp];
238
239 // insert material property values
240 auto npps = _postprocessor_values.size();
241 for (MooseIndex(_postprocessor_values) i = 0; i < npps; ++i)
242 _func_params[i + _nargs + nmat_props] = *_postprocessor_values[i];
243
244 // set function value
245 if (_prop_F)
246 (*_prop_F)[_qp] = evaluate(_func_F, _name);
247}
