44template <typename AuxKernelType>
45void
46ComputeElemAuxBcsThread<AuxKernelType>::operator()(const ConstBndElemRange & range)
47{
48 ParallelUniqueId puid;
49 _tid = puid.id;
50
51 // Reference to all boundary restricted AuxKernels for the current thread
52 const auto & boundary_kernels = _storage.getActiveBoundaryObjects(_tid);
53
54 printGeneralExecutionInformation();
55
56 for (const auto & belem : range)
57 {
58 const Elem * elem = belem->_elem;
59 unsigned short int side = belem->_side;
60 BoundaryID boundary_id = belem->_bnd_id;
61 SubdomainID last_did = Elem::invalid_subdomain_id;
62
63 // need to update the boundary ID in assembly
64 _fe_problem.setCurrentBoundaryID(boundary_id, _tid);
65
66 if (elem->processor_id() == _fe_problem.processor_id())
67 {
68 // Locate the AuxKernel objects for the current BoundaryID
69 const auto iter = boundary_kernels.find(boundary_id);
70
71 if (iter != boundary_kernels.end() && !(iter->second.empty()))
72 {
73 printBoundaryExecutionInformation(boundary_id, iter->second);
74 auto did = elem->subdomain_id();
75 if (did != last_did)
76 {
77 _fe_problem.subdomainSetup(did, _tid);
78 last_did = did;
79 }
80 _fe_problem.setCurrentSubdomainID(elem, _tid);
81 _fe_problem.prepare(elem, _tid);
82 _fe_problem.reinitElemFace(elem, side, boundary_id, _tid);
83
84 const Elem * lower_d_elem = _fe_problem.mesh().getLowerDElem(elem, side);
85 if (lower_d_elem)
86 _fe_problem.reinitLowerDElem(lower_d_elem, _tid);
87
88 const Elem * neighbor = elem->neighbor_ptr(side);
89
90 // The last check here is absolutely necessary otherwise we will attempt to evaluate
91 // neighbor materials on neighbor elements that aren't evaluable, e.g. don't have algebraic
92 // ghosting
93 bool compute_interface =
94 neighbor && neighbor->active() &&
95 _fe_problem.getInterfaceMaterialsWarehouse().hasActiveBoundaryObjects(boundary_id,
96 _tid);
97
98 // Set up Sentinel class so that, even if reinitMaterialsFace() throws, we
99 // still remember to swap back during stack unwinding.
100 SwapBackSentinel sentinel(_fe_problem, &FEProblem::swapBackMaterialsFace, _tid);
101 SwapBackSentinel neighbor_sentinel(
102 _fe_problem, &FEProblem::swapBackMaterialsNeighbor, _tid, compute_interface);
103
104 if (_need_materials)
105 {
106 std::set<unsigned int> needed_mat_props;
107 for (const auto & aux : iter->second)
108 {
109 const std::set<unsigned int> & mp_deps = aux->getMatPropDependencies();
110 needed_mat_props.insert(mp_deps.begin(), mp_deps.end());
111 }
112 _fe_problem.setActiveMaterialProperties(needed_mat_props, _tid);
113
114 _fe_problem.reinitMaterialsFace(elem->subdomain_id(), _tid);
115
116 _fe_problem.reinitMaterialsBoundary(boundary_id, _tid);
117
118 if (compute_interface)
119 {
120 _fe_problem.reinitNeighbor(elem, side, _tid);
121 _fe_problem.reinitMaterialsNeighbor(neighbor->subdomain_id(), _tid);
122 _fe_problem.reinitMaterialsInterface(boundary_id, _tid);
123 }
124 }
125
126 for (const auto & aux : iter->second)
127 {
128 aux->compute();
129 aux->variable().insert(_aux_sys.solution());
130 }
131
132 if (_need_materials)
133 _fe_problem.clearActiveMaterialProperties(_tid);
134 }
135 }
136 }
137}

158template <typename AuxKernelType>
159void
160ComputeElemAuxBcsThread<AuxKernelType>::printBoundaryExecutionInformation(
161 unsigned int boundary_id, const std::vector<std::shared_ptr<AuxKernelType>> & kernels) const
162{
163 if (!_fe_problem.shouldPrintExecution(_tid) || !_storage.hasActiveObjects() ||
164 _boundaries_exec_printed.count(boundary_id))
165 return;
166
167 const auto & console = _fe_problem.console();
168 console << "[DBG] Ordering on boundary " << boundary_id << std::endl;
169 std::vector<MooseObject *> objs_ptrs;
170 for (auto & kernel_ptr : kernels)
171 if (kernel_ptr->hasBoundary(boundary_id))
172 objs_ptrs.push_back(dynamic_cast<MooseObject *>(kernel_ptr.get()));
173 std::string list_kernels = ConsoleUtils::mooseObjectVectorToString(objs_ptrs);
174 console << ConsoleUtils::formatString(list_kernels, "[DBG]") << std::endl;
175 _boundaries_exec_printed.insert(boundary_id);
176}
