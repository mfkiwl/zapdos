78void
79Checkpoint::outputStep(const ExecFlagType & type)
80{
81 // Output is not allowed
82 if (!_allow_output && type != EXEC_FORCED)
83 return;
84
85 // If recovering disable output of initial condition, it was already output
86 if (type == EXEC_INITIAL && _app.isRecovering())
87 return;
88
89 // store current simulation time
90 _last_output_time = _time;
91
92 // set current type
93 _current_execute_flag = type;
94
95 // Check whether we should output, then do it.
96 if (shouldOutput())
97 {
98 TIME_SECTION("outputStep", 2, "Outputting Checkpoint");
99 output();
100 }
101
102 _current_execute_flag = EXEC_NONE;
103}

105bool
106Checkpoint::shouldOutput()
107{
108 // Check if the checkpoint should "normally" output, i.e. if it was created
109 // through checkpoint=true
110 bool should_output =
111 (onInterval() || _current_execute_flag == EXEC_FINAL) ? FileOutput::shouldOutput() : false;
112
113 // If this is either a auto-created checkpoint, or if its an existing checkpoint acting
114 // as the autosave and that checkpoint isn't on its interval, then output.
115 if (_is_autosave == SYSTEM_AUTOSAVE || (_is_autosave == MODIFIED_EXISTING && !should_output))
116 {
117 // If this is a pure system-created autosave through AutoCheckpointAction,
118 // then sync across processes and only output one time per signal received.
119 comm().max(Moose::interrupt_signal_number);
120 // Reading checkpoint on time step 0 is not supported
121 should_output = (Moose::interrupt_signal_number != 0) && (timeStep() > 0);
122 if (should_output)
123 {
124 _console << "Unix signal SIGUSR1 detected. Outputting checkpoint file. \n";
125 // Reset signal number since we output
126 Moose::interrupt_signal_number = 0;
127 }
128 }
129 return should_output;
130}

132void
133Checkpoint::output()
134{
135 // Create the output directory
136 const auto cp_dir = directory();
137 Utility::mkdir(cp_dir.c_str());
138
139 // Create the output filename
140 const auto current_file = filename();
141
142 // Create the libMesh Checkpoint_IO object
143 MeshBase & mesh = _es_ptr->get_mesh();
144 CheckpointIO io(mesh, true);
145
146 // Create checkpoint file structure
147 CheckpointFileNames curr_file_struct;
148
149 curr_file_struct.checkpoint = current_file + _app.checkpointSuffix();
150
151 // Write the checkpoint file
152 io.write(curr_file_struct.checkpoint);
153
154 // Write out meta data if there is any (only on processor zero)
155 if (processor_id() == 0)
156 {
157 const auto paths = _app.writeRestartableMetaData(curr_file_struct.checkpoint);
158 curr_file_struct.restart.insert(curr_file_struct.restart.begin(), paths.begin(), paths.end());
159 }
160
161 // Write out the backup
162 const auto paths = _app.backup(_app.restartFolderBase(current_file));
163 curr_file_struct.restart.insert(curr_file_struct.restart.begin(), paths.begin(), paths.end());
164
165 // Remove old checkpoint files
166 updateCheckpointFiles(curr_file_struct);
167}

169void
170Checkpoint::updateCheckpointFiles(CheckpointFileNames file_struct)
171{
172 // Update the list of stored files
173 _file_names.push_back(file_struct);
174
175 // Remove the file and the corresponding directory if it's empty
176 const auto remove_file = [this](const std::filesystem::path & path)
177 {
178 std::error_code err;
179
180 if (!std::filesystem::remove(path, err))
181 mooseWarning("Error during the deletion of checkpoint file\n",
182 std::filesystem::absolute(path),
183 "\n\n",
184 err.message());
185
186 const auto dir = path.parent_path();
187 if (std::filesystem::is_empty(dir))
188 if (!std::filesystem::remove(dir, err))
189 mooseError("Error during the deletion of checkpoint directory\n",
190 std::filesystem::absolute(dir),
191 "\n\n",
192 err.message());
193 };
194
195 // Remove un-wanted files
196 if (_file_names.size() > _num_files)
197 {
198 // Extract the filenames to be removed
199 CheckpointFileNames delete_files = _file_names.front();
200
201 // Remove these filenames from the list
202 _file_names.pop_front();
203
204 // Delete restartable data
205 for (const auto & path : delete_files.restart)
206 remove_file(path);
207
208 // Delete checkpoint files
209 // This file may not exist so don't worry about checking for success
210 if (processor_id() == 0)
211 CheckpointIO::cleanup(delete_files.checkpoint,
212 _problem_ptr->mesh().isDistributedMesh() ? comm().size() : 1);
213 }
214}
