48ReactionFirstOrder::computeQpResidual()

53Real
54ReactionFirstOrder::computeQpJacobian()
55{
56 if (_v_eq_u)
57 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * _phi[_j][_qp];
58 else
59 return 0.0;
60}

62Real
63ReactionFirstOrder::computeQpOffDiagJacobian(unsigned int jvar)
64{
65 if (!_v_eq_u && jvar == _v_id)
66 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * _phi[_j][_qp];
67 else
68 return 0.0;
