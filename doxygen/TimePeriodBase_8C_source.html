32void
33TimePeriodBase::setupTimes()
34{
35 // Check that start/end time are the same length
36 if (_end_time.size() != _start_time.size())
37 mooseError("The end time and start time vectors must be the same length.");
38
39 // Resize the start/end times if only a single value given
40 if (_end_time.size() == 1 && (_disable.size() > 1 || _enable.size() > 1))
41 {
42 unsigned int size = std::max(_disable.size(), _enable.size());
43 _end_time = std::vector<Real>(size, _end_time[0]);
44 _start_time = std::vector<Real>(size, _start_time[0]);
45 }
46 else if (_end_time.size() != _disable.size() && _end_time.size() != _enable.size())
47 mooseError("The start/end time input must be a scalar or the same length as the enable/disable "
48 "lists.");
49
50 // Test that start and end times are in proper order
51 for (unsigned int i = 0; i < _start_time.size(); ++i)
52 if (_start_time[i] >= _end_time[i])
53 mooseError("The start time(s) must be less than the end time(s).");
54}
