20InputParameters
21PseudoTimestep::validParams()
22{
23 InputParameters params = GeneralPostprocessor::validParams();
24
26 MooseEnum method("SER RDM EXP", "SER");
27
28 method.addDocumentation("SER", "Switched evolution relaxation");
29 method.addDocumentation("EXP", "Exponential progression");
30 method.addDocumentation("RDM", "Residual difference method");
31
32 params.addRequiredParam<MooseEnum>(
33 "method", method, "The method used for pseudotimestep timemarching");
34 params.addRequiredRangeCheckedParam<Real>("initial_dt", "initial_dt > 0", "Initial timestep");
35 params.addRequiredRangeCheckedParam<Real>(
36 "alpha", "alpha > 0", "The parameter alpha used in the scaling of the timestep");
37 params.addParam<Real>("max_dt", "The largest timestep allowed");
38 params.addParam<unsigned int>(
39 "iterations_window",
40 "For how many iterations should the residual be tracked (only applies to the SER method)");
41 // Because this post-processor is meant to be used with PostprocessorDT, it
42 // should be executed on initial (not included by default) and timestep end.
43 params.set<ExecFlagEnum>("execute_on") = {EXEC_TIMESTEP_END, EXEC_INITIAL};
44 params.suppressParameter<ExecFlagEnum>("execute_on");
45
46 params.addClassDescription("Computes pseudo-time steps for obtaining steady-state solutions "
47 "through a pseudo transient process.");
48
49 return params;
50}

52PseudoTimestep::PseudoTimestep(const InputParameters & parameters)
53 : GeneralPostprocessor(parameters),
54 _method(getParam<MooseEnum>("method").getEnum<PseudotimeMethod>()),
55 _initial_dt(getParam<Real>("initial_dt")),
56 _alpha(getParam<Real>("alpha")),
57 _bound(isParamValid("max_dt")),
58 _max_dt(_bound ? getParam<Real>("max_dt") : std::numeric_limits<Real>::max()),
59 _residual_norms_sequence(declareRestartableData<std::vector<Real>>("residual_norms_sequence")),
60 _iterations_step_sequence(declareRestartableData<std::vector<Real>>("iterations_step_sequence"))
61{
62 if (!_fe_problem.isTransient())
63 mooseError("This pseudotimestepper can only be used if the steady state is computed via time "
64 "integration.");
65
66 if (_method == PseudotimeMethod::SER)
67 {
68 if (parameters.isParamValid("iterations_window"))
69 _iterations_window = getParam<unsigned int>("iterations_window");
70 else
71 _iterations_window = 1;
72 }
73 else if (parameters.isParamValid("iterations_window"))
74 mooseError("The iterations window can be only provided for the SER method.");
75}

90Real
91PseudoTimestep::currentResidualNorm() const
92{
93 NonlinearSystemBase & nl = _fe_problem.getNonlinearSystemBase(_sys.number());
94 Real res_norm = 0.0;
95 for (const auto var_num : make_range(nl.system().n_vars()))
96 {
97 auto var_res = nl.system().calculate_norm(nl.getResidualNonTimeVector(), var_num, DISCRETE_L2);
98 res_norm = res_norm + std::pow(var_res, 2);
99 }
100 res_norm = std::sqrt(res_norm);
101 return res_norm;
102}

117Real
118PseudoTimestep::timestepSER() const
119{
120 const unsigned int curr_step = _fe_problem.timeStep();
121
122 const unsigned int steps_size = _iterations_step_sequence.size();
123 const unsigned int res_size = _residual_norms_sequence.size();
124 const unsigned int prev_steps = std::min(_iterations_window + 1, curr_step);
125
126 const Real factor = std::pow(_residual_norms_sequence[res_size - prev_steps] /
127 _residual_norms_sequence[res_size - 1],
128 _alpha);
129
130 const Real update_dt = _iterations_step_sequence[steps_size - 1] * factor;
131 return update_dt;
132}

134Real
135PseudoTimestep::timestepRDM() const
136{
137 const unsigned int res_size = _residual_norms_sequence.size();
138 const unsigned int steps_size = _iterations_step_sequence.size();
139
140 Real exponent = 0.0;
141
142 if (_residual_norms_sequence[res_size - 1] < _residual_norms_sequence[res_size - 2])
143 exponent = (_residual_norms_sequence[res_size - 2] - _residual_norms_sequence[res_size - 1]) /
144 _residual_norms_sequence[res_size - 2];
145 const Real update_dt = _iterations_step_sequence[steps_size - 1] * std::pow(_alpha, exponent);
146 return update_dt;
147}

150PseudoTimestep::timestepEXP() const

154 const Real update_dt = _initial_dt * factor;

158void
159PseudoTimestep::execute()
160{
161 Transient * transient = dynamic_cast<Transient *>(_app.getExecutioner());
162
163 // internal parameters needed for computing residuals and setting next timestep
164 Real res_norm;
165 Real curr_dt;
166 Real update_dt;
167
168 if (_current_execute_flag == EXEC_INITIAL)
169 _dt = _initial_dt;
170
171 // at the end of each timestep call the postprocessor to set values for dt
172 if (_current_execute_flag == EXEC_TIMESTEP_END)
173 {
174 res_norm = currentResidualNorm();
175 _residual_norms_sequence.push_back(res_norm);
176
177 curr_dt = transient->getDT();
178 _iterations_step_sequence.push_back(curr_dt);
179
180 _dt = curr_dt;
181
182 // since some of the residual methods require previous residuals the pseudo timesteppers
183 // start at the second step in the computation
184 if (_fe_problem.timeStep() > 1)
185 {
186 update_dt = curr_dt;
187 switch (_method)
188 {
189 case PseudotimeMethod::SER:
190 update_dt = timestepSER();
191 break;
192 case PseudotimeMethod::EXP:
193 update_dt = timestepEXP();
194 break;
195 case PseudotimeMethod::RDM:
196 update_dt = timestepRDM();
197 break;
198 }
199 if (_bound)
200 _dt = std::min(_max_dt, update_dt);
201 else
202 _dt = update_dt;
203 }
204 outputPseudoTimestep(_fe_problem.time());
205 }
206}
