38ADGasElectronMoments::ADGasElectronMoments(const InputParameters & parameters)
39 : ADMaterial(parameters),
40 _potential_units(getParam<std::string>("potential_units")),
41 _time_units(getParam<Real>("time_units")),
42 _use_moles(getParam<bool>("use_moles")),
43 _muem(declareADProperty<Real>("muem")),
44 _diffem(declareADProperty<Real>("diffem")),
45 _mumean_en(declareADProperty<Real>("mumean_en")),
46 _diffmean_en(declareADProperty<Real>("diffmean_en")),
47 _sgnmean_en(declareProperty<Real>("sgnmean_en")),
48 _sgnem(declareProperty<Real>("sgnem")),
49 _em(adCoupledValue("em")),
50 _mean_en(adCoupledValue("mean_en"))
51{
52 if (_potential_units.compare("V") == 0)
53 _voltage_scaling = 1.;
54 else if (_potential_units.compare("kV") == 0)
55 _voltage_scaling = 1000;
56
57 std::vector<Real> actual_mean_energy;
58 std::vector<Real> mu;
59 std::vector<Real> diff;
60
61 std::string file_name = getParam<FileName>("property_tables_file");
62 MooseUtils::checkFileReadable(file_name);
63 const char * charPath = file_name.c_str();
64 std::ifstream myfile(charPath);
65 Real value;
66
67 if (myfile.is_open())
68 {
69 while (myfile >> value)
70 {
71 actual_mean_energy.push_back(value);
72 myfile >> value;
73 mu.push_back(value);
74 myfile >> value;
75 diff.push_back(value);
76 }
77 myfile.close();
78 }
79
80 else
81 mooseError("Unable to open file");
82
83 _mu_interpolation.setData(actual_mean_energy, mu);
84 _diff_interpolation.setData(actual_mean_energy, diff);
85}

87void
88ADGasElectronMoments::computeQpProperties()
89{
90 actual_mean_en = std::exp(_mean_en[_qp].value() - _em[_qp].value());
91
92 _sgnmean_en[_qp] = -1.0;
93 _sgnem[_qp] = -1.0;
94
95 // For interpolated values the derivative propagation needs to be explicitly defined.
96
97 // Electron diffusion coefficient - value and derivatives
98 _diffem[_qp].value() = _diff_interpolation.sample(actual_mean_en) * _time_units;
99
100 _diffem[_qp].derivatives() = _diff_interpolation.sampleDerivative(actual_mean_en) *
101 actual_mean_en *
102 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _time_units;
103
104 // Electron mobility - value and derivatives
105 _muem[_qp].value() = _mu_interpolation.sample(actual_mean_en) * _voltage_scaling * _time_units;
106
107 _muem[_qp].derivatives() = _mu_interpolation.sampleDerivative(actual_mean_en) * actual_mean_en *
108 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) *
109 _voltage_scaling * _time_units;
110
111 // Mean electron energy diffusivity and mobility:
112 _diffmean_en[_qp].value() = 5.0 / 3.0 * _diffem[_qp].value();
113
114 _diffmean_en[_qp].derivatives() = 5.0 / 3.0 * _diffem[_qp].derivatives();
115
116 _mumean_en[_qp].value() = 5.0 / 3.0 * _muem[_qp].value();
117
118 _mumean_en[_qp].derivatives() = 5.0 / 3.0 * _muem[_qp].derivatives();
119}
