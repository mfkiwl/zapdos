405 // Set the TIMPI sync type via --timpi-sync
406 const auto & timpi_sync = parameters.get<std::string>("timpi_sync");
407 const_cast<Parallel::Communicator &>(comm()).sync_type(timpi_sync);
408
409#ifdef HAVE_GPERFTOOLS
410 if (isUltimateMaster())
411 {
412 bool has_cpu_profiling = false;
413 bool has_heap_profiling = false;
414 static std::string cpu_profile_file;
415 static std::string heap_profile_file;
416
417 // For CPU profiling, users need to have environment 'MOOSE_PROFILE_BASE'
418 if (std::getenv("MOOSE_PROFILE_BASE"))
419 {
420 has_cpu_profiling = true;
421 cpu_profile_file =
422 std::getenv("MOOSE_PROFILE_BASE") + std::to_string(_comm->rank()) + ".prof";
423 // create directory if needed
424 auto name = MooseUtils::splitFileName(cpu_profile_file);
425 if (!name.first.empty())
426 {
427 if (processor_id() == 0)
428 MooseUtils::makedirs(name.first.c_str());
429 _comm->barrier();
430 }
431 }
432
433 // For Heap profiling, users need to have 'MOOSE_HEAP_BASE'
434 if (std::getenv("MOOSE_HEAP_BASE"))
435 {
436 has_heap_profiling = true;
437 heap_profile_file = std::getenv("MOOSE_HEAP_BASE") + std::to_string(_comm->rank());
438 // create directory if needed
439 auto name = MooseUtils::splitFileName(heap_profile_file);
440 if (!name.first.empty())
441 {
442 if (processor_id() == 0)
443 MooseUtils::makedirs(name.first.c_str());
444 _comm->barrier();
445 }
446 }
447
448 // turn on profiling only on selected ranks
449 if (isParamValid("gperf_profiler_on"))
450 {
451 auto rankstr = getParam<std::string>("gperf_profiler_on");
452 std::vector<processor_id_type> ranks;
453 bool success = MooseUtils::tokenizeAndConvert(rankstr, ranks, ", ");
454 if (!success)
455 mooseError("Invalid argument for --gperf-profiler-on: '", rankstr, "'");
456 for (auto & rank : ranks)
457 {
458 if (rank >= _comm->size())
459 mooseError("Invalid argument for --gperf-profiler-on: ",
460 rank,
461 " is greater than or equal to ",
462 _comm->size());
463 if (rank == _comm->rank())
464 {
465 _cpu_profiling = has_cpu_profiling;
466 _heap_profiling = has_heap_profiling;
467 }
468 }
469 }
470 else
471 {
472 _cpu_profiling = has_cpu_profiling;
473 _heap_profiling = has_heap_profiling;
474 }
475
476 if (_cpu_profiling)
477 if (!ProfilerStart(cpu_profile_file.c_str()))
478 mooseError("CPU profiler is not started properly");
479
480 if (_heap_profiling)
481 {
482 HeapProfilerStart(heap_profile_file.c_str());
483 if (!IsHeapProfilerRunning())
484 mooseError("Heap profiler is not started properly");
485 }
486 }
487#else
488 if (std::getenv("MOOSE_PROFILE_BASE") || std::getenv("MOOSE_HEAP_BASE"))
489 mooseError("gperftool is not available for CPU or heap profiling");
490#endif
491
492 // If this will be a language server then turn off output until that starts
493 if (isParamValid("language_server"))
494 _output_buffer_cache = Moose::out.rdbuf(nullptr);
495
496 Registry::addKnownLabel(_type);
497 Moose::registerAll(_factory, _action_factory, _syntax);
498
499 _the_warehouse = std::make_unique<TheWarehouse>();
500 _the_warehouse->registerAttribute<AttribMatrixTags>("matrix_tags", 0);
501 _the_warehouse->registerAttribute<AttribVectorTags>("vector_tags", 0);
502 _the_warehouse->registerAttribute<AttribExecOns>("exec_ons", 0);
503 _the_warehouse->registerAttribute<AttribSubdomains>("subdomains", 0);
504 _the_warehouse->registerAttribute<AttribBoundaries>("boundaries", 0);
505 _the_warehouse->registerAttribute<AttribThread>("thread", 0);
506 _the_warehouse->registerAttribute<AttribExecutionOrderGroup>("execution_order_group", 0);
507 _the_warehouse->registerAttribute<AttribPreIC>("pre_ic", 0);
508 _the_warehouse->registerAttribute<AttribPreAux>("pre_aux");
509 _the_warehouse->registerAttribute<AttribPostAux>("post_aux");
510 _the_warehouse->registerAttribute<AttribName>("name", "dummy");
511 _the_warehouse->registerAttribute<AttribSystem>("system", "dummy");
512 _the_warehouse->registerAttribute<AttribVar>("variable", -1);
513 _the_warehouse->registerAttribute<AttribInterfaces>("interfaces", 0);
514 _the_warehouse->registerAttribute<AttribSysNum>("sys_num", libMesh::invalid_uint);
515 _the_warehouse->registerAttribute<AttribResidualObject>("residual_object");
516 _the_warehouse->registerAttribute<AttribSorted>("sorted");
517 _the_warehouse->registerAttribute<AttribDisplaced>("displaced", -1);
518
519 if (isParamValid("_argc") && isParamValid("_argv"))
520 {
521 int argc = getParam<int>("_argc");
522 char ** argv = getParam<char **>("_argv");
523
524 _sys_info = std::make_unique<SystemInfo>(argc, argv);
525 }
526 if (isParamValid("_command_line"))
527 _command_line = getParam<std::shared_ptr<CommandLine>>("_command_line");
528 else
529 mooseError("Valid CommandLine object required");
530
531 if (_check_input && isParamValid("recover"))
532 mooseError("Cannot run --check-input with --recover. Recover files might not exist");
533
534 if (isParamValid("start_in_debugger") && _multiapp_level == 0)
535 {
536 auto command = getParam<std::string>("start_in_debugger");
537
538 Moose::out << "Starting in debugger using: " << command << std::endl;
539
540 auto hostname = MooseUtils::hostname();
541
542 std::stringstream command_stream;
543
544 // This will start XTerm and print out some info first... then run the debugger
545 command_stream << "xterm -e \"echo 'Rank: " << processor_id() << " Hostname: " << hostname
546 << " PID: " << getpid() << "'; echo ''; ";
547
548 // Figure out how to run the debugger
549 if (command.find("lldb") != std::string::npos || command.find("gdb") != std::string::npos)
550 command_stream << command << " -p " << getpid();
551 else
552 mooseError("Unknown debugger: ",
553 command,
554 "\nIf this is truly what you meant then contact moose-users to have a discussion "
555 "about adding your debugger.");
556
557 // Finish up the command
558 command_stream << "\""
559 << " & ";
560
561 std::string command_string = command_stream.str();
562 Moose::out << "Running: " << command_string << std::endl;
563
564 int ret = std::system(command_string.c_str());
565 libmesh_ignore(ret);
566
567 // Sleep to allow time for the debugger to attach
568 std::this_thread::sleep_for(std::chrono::seconds(10));
569 }
570
571 if (!parameters.isParamSetByAddParam("stop_for_debugger"))
572 {
573 Moose::out << "\nStopping for " << getParam<unsigned int>("stop_for_debugger")
574 << " seconds to allow attachment from a debugger.\n";
575
576 Moose::out << "\nAll of the processes you can connect to:\n";
577 Moose::out << "rank - hostname - pid\n";
578
579 auto hostname = MooseUtils::hostname();
580
581 {
582 // The 'false' turns off the serialization warning
583 SerializerGuard sg(_communicator, false); // Guarantees that the processors print in order
584 Moose::err << processor_id() << " - " << hostname << " - " << getpid() << "\n";
585 }
586
587 Moose::out << "\nWaiting...\n" << std::endl;
588
589 // Sleep to allow time for the debugger to attach
590 std::this_thread::sleep_for(std::chrono::seconds(getParam<unsigned int>("stop_for_debugger")));
591 }
592
593 if (_master_mesh && _multiapp_level == 0)
594 mooseError("Mesh can be passed in only for sub-apps");
595
596 if (_master_displaced_mesh && !_master_mesh)
597 mooseError("_master_mesh should have been set when _master_displaced_mesh is set");
598
599 // Data specifically associated with the mesh (meta-data) that will read from the restart
600 // file early during the simulation setup so that they are available to Actions and other objects
601 // that need them during the setup process. Most of the restartable data isn't made available
602 // until all objects have been created and all Actions have been executed (i.e. initialSetup).
603 registerRestartableDataMapName(MooseApp::MESH_META_DATA, MooseApp::MESH_META_DATA_SUFFIX);
604
605 if (parameters.have_parameter<bool>("use_legacy_dirichlet_bc"))
606 mooseDeprecated("The parameter 'use_legacy_dirichlet_bc' is no longer valid.\n\n",
607 "All Dirichlet boundary conditions are preset by default.\n\n",
608 "Remove said parameter in ",
609 name(),
610 " to remove this deprecation warning.");
611
612 Moose::out << std::flush;
613}

616{
617#ifdef HAVE_GPERFTOOLS
618 // CPU profiling stop
619 if (_cpu_profiling)
620 ProfilerStop();
621 // Heap profiling stop
622 if (_heap_profiling)
623 HeapProfilerStop();
624#endif
625 _action_warehouse.clear();

629 // Don't wait for implicit destruction of input parameter storage

632 // This is dirty, but I don't know what else to do. Obviously, others
633 // have had similar problems if you look above. In specific, the

1260std::unique_ptr<Backup>
1261MooseApp::finalizeRestore()
1262{
1263 if (!_rd_reader.isRestoring())
1264 mooseError("MooseApp::finalizeRestore(): Not currently restoring");
1265
1266 // This gives us access to the underlying streams so that we can return it if needed
1267 auto input_streams = _rd_reader.clear();
1268
1269 std::unique_ptr<Backup> backup;
1270
1271 // Give them back a backup if this restore started from a Backup, in which case
1272 // the two streams in the Backup are formed into StringInputStreams
1273 if (auto header_string_input = dynamic_cast<StringInputStream *>(input_streams.header.get()))
1274 {
1275 auto data_string_input = dynamic_cast<StringInputStream *>(input_streams.data.get());
1276 mooseAssert(data_string_input, "Should also be a string input");
1277
1278 auto header_sstream = header_string_input->release();
1279 mooseAssert(header_sstream, "Header not available");
1280
1281 auto data_sstream = data_string_input->release();
1282 mooseAssert(data_sstream, "Data not available");
1283
1284 backup = std::make_unique<Backup>();
1285 backup->header = std::move(header_sstream);
1286 backup->data = std::move(data_sstream);
1287 }
1288
1289 return backup;

1330void
1331MooseApp::recursivelyCreateExecutors(const std::string & current_executor_name,
1332 std::list<std::string> & possible_roots,
1333 std::list<std::string> & current_branch)
1334{

1336 if (_executors.find(current_executor_name) != _executors.end())
1337 return;
1338
1339 // Is this one already on the current branch (i.e. there is a cycle)
1340 if (std::find(current_branch.begin(), current_branch.end(), current_executor_name) !=
1341 current_branch.end())
1342 {
1343 std::stringstream exec_names_string;
1344
1345 auto branch_it = current_branch.begin();
1346
1347 exec_names_string << *branch_it++;
1348
1349 for (; branch_it != current_branch.end(); ++branch_it)
1350 exec_names_string << ", " << *branch_it;
1351
1352 exec_names_string << ", " << current_executor_name;
1353
1354 mooseError("Executor cycle detected: ", exec_names_string.str());
1355 }
1356
1357 current_branch.push_back(current_executor_name);
1358
1359 // Build the dependencies first
1360 const auto & params = *_executor_params[current_executor_name].second;
1361
1362 for (const auto & param : params)
1363 {
1364 if (params.have_parameter<ExecutorName>(param.first))
1365 {
1366 const auto & dependency_name = params.get<ExecutorName>(param.first);
1367
1368 possible_roots.remove(dependency_name);
1369
1370 if (!dependency_name.empty())
1371 recursivelyCreateExecutors(dependency_name, possible_roots, current_branch);
1372 }
1373 }
1374
1375 // Add this Executor
1376 const auto & type = _executor_params[current_executor_name].first;
1377 addExecutor(type, current_executor_name, params);
1378
1379 current_branch.pop_back();
1380}

1382void
1383MooseApp::createExecutors()
1384{
1385 // Do we have any?
1386 if (_executor_params.empty())
1387 return;
1388
1389 // Holds the names of Executors that may be the root executor
1390 std::list<std::string> possibly_root;
1391
1392 // What is already built
1393 std::map<std::string, bool> already_built;
1394
1395 // The Executors that are currently candidates for being roots
1396 std::list<std::string> possible_roots;
1397
1398 // The current line of dependencies - used for finding cycles
1399 std::list<std::string> current_branch;
1400
1401 // Build the NullExecutor
1402 {
1403 auto params = _factory.getValidParams("NullExecutor");
1404 _null_executor = _factory.create<NullExecutor>("NullExecutor", "_null_executor", params);
1405 }
1406
1407 for (const auto & params_entry : _executor_params)
1408 {
1409 const auto & name = params_entry.first;
1410
1411 // Did we already make this one?
1412 if (_executors.find(name) != _executors.end())
1413 continue;
1414
1415 possible_roots.emplace_back(name);
1416
1417 recursivelyCreateExecutors(name, possible_roots, current_branch);
1418 }
1419
1420 // If there is more than one possible root - error
1421 if (possible_roots.size() > 1)
1422 {
1423 auto root_string_it = possible_roots.begin();
1424
1425 std::stringstream roots_string;
1426
1427 roots_string << *root_string_it++;
1428
1429 for (; root_string_it != possible_roots.end(); ++root_string_it)
1430 roots_string << ", " << *root_string_it;
1431
1432 mooseError("Multiple Executor roots found: ", roots_string.str());
1433 }
1434
1435 // Set the root executor
1436 _executor = _executors[possible_roots.front()];
1437}

1998void
1999MooseApp::loadLibraryAndDependencies(const std::string & library_filename,
2000 const Parameters & params,
2001 const bool load_dependencies)
2002{
2003 std::string line;
2004 std::string dl_lib_filename;
2005
2006 // This RE looks for absolute path libtool filenames (i.e. begins with a slash and ends with a
2007 // .la)
2008 pcrecpp::RE re_deps("(/\\S*\\.la)");
2009
2010 std::ifstream la_handle(library_filename.c_str());
2011 if (la_handle.is_open())
2012 {
2013 while (std::getline(la_handle, line))
2014 {
2015 // Look for the system dependent dynamic library filename to open
2016 if (line.find("dlname=") != std::string::npos)
2017 // Magic numbers are computed from length of this string "dlname=' and line minus that
2018 // string plus quotes"
2019 dl_lib_filename = line.substr(8, line.size() - 9);
2020

2028 // Recurse here to load dependent libraries in depth-first order
2029 loadLibraryAndDependencies(depend_library.as_string(), params, load_dependencies);
2030 }
2031
2032 // There's only one line in the .la file containing the dependency libs so break after
2033 // finding it
2034 break;
2035 }
2036 }
2037 la_handle.close();
2038 }
2039
2040 // This should only occur if we have static linkage.
2041 if (dl_lib_filename.empty())
2042 return;

2045

2047 // 1) make sure we haven't already loaded this library
2048 // AND 2) make sure we have a library name (we won't for static linkage)
2049 // Note: Here was are going to assume uniqueness based on the filename alone. This has significant
2050 // implications for applications that have "diamond" inheritance of libraries (usually
2051 // modules). We will only load one of those libraries, versions be damned.
2052 auto dyn_lib_it = _lib_handles.find(file_name);
2053 if (dyn_lib_it == _lib_handles.end())
2054 {
2055 // Assemble the actual filename using the base path of the *.la file and the dl_lib_filename
2056 const auto dl_lib_full_path = dir + '/' + dl_lib_filename;
2057
2058 MooseUtils::checkFileReadable(dl_lib_full_path, false, /*throw_on_unreadable=*/true);
2059
2060#ifdef LIBMESH_HAVE_DLOPEN
2061 void * const lib_handle = dlopen(dl_lib_full_path.c_str(), RTLD_LAZY);
2062#else
2063 void * const lib_handle = nullptr;
2064#endif
2065
2066 if (!lib_handle)
2067 mooseError("The library file \"",
2068 dl_lib_full_path,
2069 "\" exists and has proper permissions, but cannot by dynamically loaded.\nThis "
2070 "generally means that the loader was unable to load one or more of the "
2071 "dependencies listed in the supplied library (see otool or ldd).\n");
2072
2073 DynamicLibraryInfo lib_info;
2074 lib_info.library_handle = lib_handle;
2075 lib_info.full_path = library_filename;
2076
2077 auto insert_ret = _lib_handles.insert(std::make_pair(file_name, lib_info));
2078 mooseAssert(insert_ret.second == true, "Error inserting into lib_handles map");
2079
2080 dyn_lib_it = insert_ret.first;
2081 }
2082
2083 // Library has been loaded, check to see if we've called the requested registration method
2084 const auto registration_method = params.get<std::string>("registration_method");
2085 auto & entry_sym_from_curr_lib = dyn_lib_it->second.entry_symbols;
2086
2087 if (entry_sym_from_curr_lib.find(registration_method) == entry_sym_from_curr_lib.end())
2088 {
2089 // get the pointer to the method in the library. The dlsym()
2090 // function returns a null pointer if the symbol cannot be found,
2091 // we also explicitly set the pointer to NULL if dlsym is not
2092 // available.
2093#ifdef LIBMESH_HAVE_DLOPEN
2094 void * registration_handle =
2095 dlsym(dyn_lib_it->second.library_handle, registration_method.c_str());
2096#else
2097 void * registration_handle = nullptr;
2098#endif

2103 {
2104 case APPLICATION:
2105 {
2106 using register_app_t = void (*)();
2107 register_app_t * const reg_ptr = reinterpret_cast<register_app_t *>(&registration_handle);
2108 (*reg_ptr)();
2109 break;
2110 }
2111 case REGALL:
2112 {
2113 using register_app_t = void (*)(Factory *, ActionFactory *, Syntax *);
2114 register_app_t * const reg_ptr = reinterpret_cast<register_app_t *>(&registration_handle);
2115 (*reg_ptr)(params.get<Factory *>("factory"),
2116 params.get<ActionFactory *>("action_factory"),
2117 params.get<Syntax *>("syntax"));
2118 break;
2119 }
2120 default:
2121 mooseError("Unhandled RegistrationType");
2122 }

2124 entry_sym_from_curr_lib.insert(registration_method);
2125 }
2126 else
2127 {
2128
2129#if defined(DEBUG) && defined(LIBMESH_HAVE_DLOPEN)
2130 // We found a dynamic library that doesn't have a dynamic
2131 // registration method in it. This shouldn't be an error, so
2132 // we'll just move on.
2133 if (!registration_handle)
2134 mooseWarning("Unable to find extern \"C\" method \"",
2135 registration_method,
2136 "\" in library: ",
2137 dyn_lib_it->first,
2138 ".\n",
2139 "This doesn't necessarily indicate an error condition unless you believe that "
2140 "the method should exist in that library.\n",
2141 dlerror());

2444RelationshipManager &
2445MooseApp::createRMFromTemplateAndInit(const RelationshipManager & template_rm,
2446 MeshBase & mesh,
2447 const DofMap * const dof_map)
2448{
2449 auto & mesh_to_clone = _template_to_clones[&template_rm];
2450 auto it = mesh_to_clone.find(&mesh);
2451 if (it != mesh_to_clone.end())
2452 {
2453 // We've already created a clone for this mesh
2454 auto & clone_rm = *it->second;
2455 if (!clone_rm.dofMap() && dof_map)
2456 // We didn't have a DofMap before, but now we do, so we should re-init
2457 clone_rm.init(mesh, dof_map);
2458 else if (clone_rm.dofMap() && dof_map && (clone_rm.dofMap() != dof_map))
2459 mooseError("Attempting to create and initialize an existing clone with a different DofMap. "
2460 "This should not happen.");
2461
2462 return clone_rm;
2463 }
2464
2465 // It's possible that this method is going to get called for multiple different MeshBase
2466 // objects. If that happens, then we *cannot* risk having a MeshBase object with a ghosting
2467 // functor that is init'd with another MeshBase object. So the safe thing to do is to make a
2468 // different RM for every MeshBase object that gets called here. Then the
2469 // RelationshipManagers stored here in MooseApp are serving as a template only
2470 auto pr = mesh_to_clone.emplace(
2471 std::make_pair(&const_cast<const MeshBase &>(mesh),
2472 dynamic_pointer_cast<RelationshipManager>(template_rm.clone())));
2473 mooseAssert(pr.second, "An insertion should have happened");
2474 auto & clone_rm = *pr.first->second;
2475 clone_rm.init(mesh, dof_map);
2476 return clone_rm;

2509void
2510MooseApp::attachRelationshipManagers(Moose::RelationshipManagerType rm_type,
2511 bool attach_geometric_rm_final)
2512{
2513 for (auto & rm : _relationship_managers)
2514 {
2515 if (!rm->isType(rm_type))
2516 continue;
2517
2518 // RM is already attached (this also handles the geometric early case)
2519 if (_attached_relationship_managers[rm_type].count(rm.get()))
2520 continue;
2521
2522 if (rm_type == Moose::RelationshipManagerType::GEOMETRIC)
2523 {
2524 // The problem is not built yet - so the ActionWarehouse currently owns the mesh
2525 MooseMesh * const mesh = _action_warehouse.mesh().get();
2526
2527 // "attach_geometric_rm_final = true" inidicate that it is the last chance to attach
2528 // geometric RMs. Therefore, we need to attach them.
2529 if (!rm->attachGeometricEarly() && !attach_geometric_rm_final)
2530 {
2531 // Will attach them later (during algebraic). But also, we need to tell the mesh that we
2532 // shouldn't be deleting remote elements yet
2533 if (!mesh->getMeshPtr())
2534 mooseError("We should have attached a MeshBase object to the mesh by now");
2535
2536 mesh->allowRemoteElementRemoval(false);
2537 }
2538 else
2539 {
2540 MeshBase & undisp_mesh_base = mesh->getMesh();
2541 const DofMap * const undisp_nl_dof_map =
2542 _executioner ? &feProblem().systemBaseNonlinear(0).dofMap() : nullptr;
2543 undisp_mesh_base.add_ghosting_functor(
2544 createRMFromTemplateAndInit(*rm, undisp_mesh_base, undisp_nl_dof_map));
2545
2546 // In the final stage, if there is a displaced mesh, we need to
2547 // clone ghosting functors for displacedMesh
2548 if (attach_geometric_rm_final && _action_warehouse.displacedMesh())
2549 {
2550 MeshBase & disp_mesh_base = _action_warehouse.displacedMesh()->getMesh();
2551 const DofMap * disp_nl_dof_map = nullptr;
2552 if (_executioner && feProblem().getDisplacedProblem())
2553 disp_nl_dof_map = &feProblem().getDisplacedProblem()->systemBaseNonlinear(0).dofMap();
2554 disp_mesh_base.add_ghosting_functor(
2555 createRMFromTemplateAndInit(*rm, disp_mesh_base, disp_nl_dof_map));
2556 }
2557 else if (_action_warehouse.displacedMesh())
2558 mooseError("The displaced mesh should not yet exist at the time that we are attaching "
2559 "early geometric relationship managers.");
2560
2561 // Mark this RM as attached
2562 mooseAssert(!_attached_relationship_managers[rm_type].count(rm.get()), "Already attached");
2563 _attached_relationship_managers[rm_type].insert(rm.get());
2564 }
2565 }
2566 else // rm_type is algebraic or coupling
2567 {
2568 if (!_executioner && !_executor)
2569 mooseError("We must have an executioner by now or else we do not have to data to add "
2570 "algebraic or coupling functors to in MooseApp::attachRelationshipManagers");
2571
2572 // Now we've built the problem, so we can use it
2573 auto & problem = feProblem();
2574 auto & undisp_nl = problem.systemBaseNonlinear(0);
2575 auto & undisp_nl_dof_map = undisp_nl.dofMap();
2576 auto & undisp_mesh = problem.mesh().getMesh();
2577
2578 if (rm->useDisplacedMesh() && problem.getDisplacedProblem())
2579 {
2580 if (rm_type == Moose::RelationshipManagerType::COUPLING)
2581 // We actually need to add this to the FEProblemBase NonlinearSystemBase's DofMap
2582 // because the DisplacedProblem "nonlinear" DisplacedSystem doesn't have any matrices
2583 // for which to do coupling. It's actually horrifying to me that we are adding a
2584 // coupling functor, that is going to determine its couplings based on a displaced
2585 // MeshBase object, to a System associated with the undisplaced MeshBase object (there
2586 // is only ever one EquationSystems object per MeshBase object and visa versa). So here
2587 // I'm left with the choice of whether to pass in a MeshBase object that is *not* the
2588 // MeshBase object that will actually determine the couplings or to pass in the MeshBase
2589 // object that is inconsistent with the System DofMap that we are adding the coupling
2590 // functor for! Let's err on the side of *libMesh* consistency and pass properly paired
2591 // MeshBase-DofMap
2592 problem.addCouplingGhostingFunctor(
2593 createRMFromTemplateAndInit(*rm, undisp_mesh, &undisp_nl_dof_map),
2594 /*to_mesh = */ false);
2595
2596 else if (rm_type == Moose::RelationshipManagerType::ALGEBRAIC)
2597 {
2598 auto & displaced_problem = *problem.getDisplacedProblem();
2599 MeshBase & disp_mesh = displaced_problem.mesh().getMesh();
2600 const DofMap * const disp_nl_dof_map = &displaced_problem.systemBaseNonlinear(0).dofMap();
2601 displaced_problem.addAlgebraicGhostingFunctor(
2602 createRMFromTemplateAndInit(*rm, disp_mesh, disp_nl_dof_map),
2603 /*to_mesh = */ false);
2604 }
2605 }
2606 else // undisplaced
2607 {
2608 if (rm_type == Moose::RelationshipManagerType::COUPLING)
2609 problem.addCouplingGhostingFunctor(
2610 createRMFromTemplateAndInit(*rm, undisp_mesh, &undisp_nl_dof_map),
2611 /*to_mesh = */ false);
2612
2613 else if (rm_type == Moose::RelationshipManagerType::ALGEBRAIC)
2614 problem.addAlgebraicGhostingFunctor(
2615 createRMFromTemplateAndInit(*rm, undisp_mesh, &undisp_nl_dof_map),
2616 /*to_mesh = */ false);
2617 }
2618
2619 // Mark this RM as attached
2620 mooseAssert(!_attached_relationship_managers[rm_type].count(rm.get()), "Already attached");
