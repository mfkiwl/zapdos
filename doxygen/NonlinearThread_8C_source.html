63void
64NonlinearThread::subdomainChanged()
65{
66 // This should come first to setup the residual objects before we do dependency determination of
67 // material properties and variables
68 determineObjectWarehouses();
69
70 _fe_problem.subdomainSetup(_subdomain, _tid);
71
72 // Update variable Dependencies
73 std::set<MooseVariableFEBase *> needed_moose_vars;
74 _tag_kernels->updateBlockVariableDependency(_subdomain, needed_moose_vars, _tid);
75 _ibc_warehouse->updateBoundaryVariableDependency(needed_moose_vars, _tid);
76 _dg_warehouse->updateBlockVariableDependency(_subdomain, needed_moose_vars, _tid);
77 _ik_warehouse->updateBoundaryVariableDependency(needed_moose_vars, _tid);
78
79 // Update FE variable coupleable vector tags
80 std::set<TagID> needed_fe_var_vector_tags;
81 _tag_kernels->updateBlockFEVariableCoupledVectorTagDependency(
82 _subdomain, needed_fe_var_vector_tags, _tid);
83 _ibc_warehouse->updateBlockFEVariableCoupledVectorTagDependency(
84 _subdomain, needed_fe_var_vector_tags, _tid);
85 _fe_problem.getMaterialWarehouse().updateBlockFEVariableCoupledVectorTagDependency(
86 _subdomain, needed_fe_var_vector_tags, _tid);
87
88 // Update material dependencies
89 std::set<unsigned int> needed_mat_props;
90 _tag_kernels->updateBlockMatPropDependency(_subdomain, needed_mat_props, _tid);
91 _ibc_warehouse->updateBoundaryMatPropDependency(needed_mat_props, _tid);
92 _dg_warehouse->updateBlockMatPropDependency(_subdomain, needed_mat_props, _tid);
93 _ik_warehouse->updateBoundaryMatPropDependency(needed_mat_props, _tid);
94
95 if (_fe_problem.haveFV())
96 {
97 // Re-query the finite volume elemental kernels
98 _fv_kernels.clear();
99 _fe_problem.theWarehouse()
100 .query()
101 .template condition<AttribSysNum>(_nl.number())
102 .template condition<AttribSystem>("FVElementalKernel")
103 .template condition<AttribSubdomains>(_subdomain)
104 .template condition<AttribThread>(_tid)
105 .queryInto(_fv_kernels);
106 for (const auto fv_kernel : _fv_kernels)
107 {
108 const auto & fv_mv_deps = fv_kernel->getMooseVariableDependencies();
109 needed_moose_vars.insert(fv_mv_deps.begin(), fv_mv_deps.end());
110 const auto & fv_mp_deps = fv_kernel->getMatPropDependencies();
111 needed_mat_props.insert(fv_mp_deps.begin(), fv_mp_deps.end());
112 }
113 }
114
115 _fe_problem.setActiveElementalMooseVariables(needed_moose_vars, _tid);
116 _fe_problem.setActiveMaterialProperties(needed_mat_props, _tid);
117 _fe_problem.setActiveFEVariableCoupleableVectorTags(needed_fe_var_vector_tags, _tid);
118 _fe_problem.prepareMaterials(_subdomain, _tid);
119}

121void
122NonlinearThread::onElement(const Elem * elem)
123{
124 _fe_problem.prepare(elem, _tid);
125 _fe_problem.reinitElem(elem, _tid);
126
127 // Set up Sentinel class so that, even if reinitMaterials() throws, we
128 // still remember to swap back during stack unwinding.
129 SwapBackSentinel sentinel(_fe_problem, &FEProblem::swapBackMaterials, _tid);
130
131 _fe_problem.reinitMaterials(_subdomain, _tid);
132
133 if (dynamic_cast<ComputeJacobianThread *>(this))
134 if (_nl.getScalarVariables(_tid).size() > 0)
135 _fe_problem.reinitOffDiagScalars(_tid);
136
137 computeOnElement();
138}

140void
141NonlinearThread::computeOnElement()
142{
143 if (_tag_kernels->hasActiveBlockObjects(_subdomain, _tid))
144 {
145 const auto & kernels = _tag_kernels->getActiveBlockObjects(_subdomain, _tid);
146 for (const auto & kernel : kernels)
147 compute(*kernel);

149
150 if (_fe_problem.haveFV())
151 for (auto kernel : _fv_kernels)
152 compute(*kernel);

155void
156NonlinearThread::onBoundary(const Elem * elem,
157 unsigned int side,
158 BoundaryID bnd_id,
159 const Elem * lower_d_elem /*=nullptr*/)
160{
161 if (_ibc_warehouse->hasActiveBoundaryObjects(bnd_id, _tid))
162 {
163 _fe_problem.reinitElemFace(elem, side, bnd_id, _tid);
164
165 // Needed to use lower-dimensional variables on Materials
166 if (lower_d_elem)
167 _fe_problem.reinitLowerDElem(lower_d_elem, _tid);
168
169 // Set up Sentinel class so that, even if reinitMaterialsFace() throws, we
170 // still remember to swap back during stack unwinding.
171 SwapBackSentinel sentinel(_fe_problem, &FEProblem::swapBackMaterialsFace, _tid);
172
173 _fe_problem.reinitMaterialsFace(elem->subdomain_id(), _tid);
174 _fe_problem.reinitMaterialsBoundary(bnd_id, _tid);
175
176 computeOnBoundary(bnd_id, lower_d_elem);
177
178 if (lower_d_elem)
179 {
180 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
181 accumulateLower();
182 }
183 }

195void
196NonlinearThread::onInterface(const Elem * elem, unsigned int side, BoundaryID bnd_id)
197{
198 if (_ik_warehouse->hasActiveBoundaryObjects(bnd_id, _tid))
199 {
200
201 // Pointer to the neighbor we are currently working on.
202 const Elem * neighbor = elem->neighbor_ptr(side);
203
204 if (neighbor->active())
205 {
206 _fe_problem.reinitNeighbor(elem, side, _tid);
207
208 // Set up Sentinels so that, even if one of the reinitMaterialsXXX() calls throws, we
209 // still remember to swap back during stack unwinding. Note that face, boundary, and interface
210 // all operate with the same MaterialData object
211 SwapBackSentinel face_sentinel(_fe_problem, &FEProblem::swapBackMaterialsFace, _tid);
212 _fe_problem.reinitMaterialsFace(elem->subdomain_id(), _tid);
213 _fe_problem.reinitMaterialsBoundary(bnd_id, _tid);
214
215 SwapBackSentinel neighbor_sentinel(_fe_problem, &FEProblem::swapBackMaterialsNeighbor, _tid);
216 _fe_problem.reinitMaterialsNeighbor(neighbor->subdomain_id(), _tid);
217
218 // Has to happen after face and neighbor properties have been computed. Note that we don't use
219 // a sentinel here because FEProblem::swapBackMaterialsFace is going to handle face materials,
220 // boundary materials, and interface materials (e.g. it queries the boundary material data
221 // with the current element and side
222 _fe_problem.reinitMaterialsInterface(bnd_id, _tid);
223
224 computeOnInterface(bnd_id);
225
226 {
227 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
228 accumulateNeighbor();
229 }
230 }
231 }

242void
243NonlinearThread::onInternalSide(const Elem * elem, unsigned int side)
244{
245 if (_dg_warehouse->hasActiveBlockObjects(_subdomain, _tid))
246 {
247 // Pointer to the neighbor we are currently working on.
248 const Elem * neighbor = elem->neighbor_ptr(side);
249
250 _fe_problem.reinitElemNeighborAndLowerD(elem, side, _tid);
251
252 // Set up Sentinels so that, even if one of the reinitMaterialsXXX() calls throws, we
253 // still remember to swap back during stack unwinding.
254 SwapBackSentinel face_sentinel(_fe_problem, &FEProblem::swapBackMaterialsFace, _tid);
255 _fe_problem.reinitMaterialsFace(elem->subdomain_id(), _tid);
256
257 SwapBackSentinel neighbor_sentinel(_fe_problem, &FEProblem::swapBackMaterialsNeighbor, _tid);
258 _fe_problem.reinitMaterialsNeighbor(neighbor->subdomain_id(), _tid);
259
260 computeOnInternalFace(neighbor);
261
262 {
263 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
264 accumulateNeighborLower();
265 }
266 }

341void
342NonlinearThread::printBlockExecutionInformation() const
343{
344 // Number of objects executing is approximated by size of warehouses
345 const int num_objects = _kernels.size() + _fv_kernels.size() + _integrated_bcs.size() +
346 _dg_kernels.size() + _interface_kernels.size();
347 const auto & console = _fe_problem.console();
348 const auto block_name = _mesh.getSubdomainName(_subdomain);
349
350 if (_fe_problem.shouldPrintExecution(_tid) && num_objects > 0)
351 {
352 if (_blocks_exec_printed.count(_subdomain))
353 return;
354 console << "[DBG] Ordering of " + objectType() + " Objects on block " << block_name << " ("
355 << _subdomain << ")" << std::endl;
356 if (_kernels.hasActiveBlockObjects(_subdomain, _tid))
357 {
358 console << "[DBG] Ordering of kernels:" << std::endl;
359 console << _kernels.activeObjectsToFormattedString() << std::endl;
360 }
361 if (_fv_kernels.size())
362 {
363 console << "[DBG] Ordering of FV elemental kernels:" << std::endl;
364 std::string fvkernels =
365 std::accumulate(_fv_kernels.begin() + 1,
366 _fv_kernels.end(),
367 _fv_kernels[0]->name(),
368 [](const std::string & str_out, FVElementalKernel * kernel)
369 { return str_out + " " + kernel->name(); });
370 console << ConsoleUtils::formatString(fvkernels, "[DBG]") << std::endl;
371 }
372 if (_dg_kernels.hasActiveBlockObjects(_subdomain, _tid))
373 {
374 console << "[DBG] Ordering of DG kernels:" << std::endl;
375 console << _dg_kernels.activeObjectsToFormattedString() << std::endl;
376 }
377 }
378 else if (_fe_problem.shouldPrintExecution(_tid) && num_objects == 0 &&
379 !_blocks_exec_printed.count(_subdomain))
380 console << "[DBG] No Active " + objectType() + " Objects on block " << block_name << " ("
381 << _subdomain << ")" << std::endl;
382
383 _blocks_exec_printed.insert(_subdomain);
384}

386void
387NonlinearThread::printBoundaryExecutionInformation(const unsigned int bid) const
388{
389 if (!_fe_problem.shouldPrintExecution(_tid) || _boundaries_exec_printed.count(bid) ||
390 (!_integrated_bcs.hasActiveBoundaryObjects(bid, _tid) &&
391 !_interface_kernels.hasActiveBoundaryObjects(bid, _tid)))
392 return;
393
394 const auto & console = _fe_problem.console();
395 const auto b_name = _mesh.getBoundaryName(bid);
396 console << "[DBG] Ordering of " + objectType() + " Objects on boundary " << b_name << " (" << bid
397 << ")" << std::endl;
398
399 if (_integrated_bcs.hasActiveBoundaryObjects(bid, _tid))
400 {
401 console << "[DBG] Ordering of integrated boundary conditions:" << std::endl;
402 console << _integrated_bcs.activeObjectsToFormattedString() << std::endl;
403 }
404
405 // We have not checked if we have a neighbor. This could be premature for saying we are executing
406 // interface kernels. However, we should assume the execution will happen on another side of the
407 // same boundary
408 if (_interface_kernels.hasActiveBoundaryObjects(bid, _tid))
409 {
410 console << "[DBG] Ordering of interface kernels:" << std::endl;
411 console << _interface_kernels.activeObjectsToFormattedString() << std::endl;
412 }
413
414 _boundaries_exec_printed.insert(bid);
