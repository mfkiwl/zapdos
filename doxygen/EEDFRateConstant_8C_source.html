43EEDFRateConstant::EEDFRateConstant(const InputParameters & parameters)
44 : Material(parameters),
45 _reaction_rate(declareProperty<Real>("k" + getParam<std::string>("number") + "_" +
46 getParam<std::string>("reaction"))),
47 _d_k_d_en(declareProperty<Real>("d_k_d_en_" + getParam<std::string>("reaction"))),
48 _sampler(isCoupled("sampler") ? coupledValue("sampler") : _zero),
49 _em(isCoupled("electrons") ? coupledValue("electrons") : _zero),
50 _mean_en(isCoupled("mean_energy") ? coupledValue("mean_energy") : _zero)
51{
52 if (!isCoupled("sampler"))
53 mooseError("Sampling variable is not coupled! Please input the variable (aux or nonlinear) "
54 "that will be used to sample from data files.");
55 std::vector<Real> val_x;
56 std::vector<Real> rate_coefficient;
57 std::string file_name =
58 getParam<std::string>("file_location") + "/" + getParam<FileName>("property_file");
59 MooseUtils::checkFileReadable(file_name);
60 const char * charPath = file_name.c_str();
61 std::ifstream myfile(charPath);
62 Real value;
63
64 if (myfile.is_open())
65 {
66 while (myfile >> value)
67 {
68 val_x.push_back(value);
69 myfile >> value;
70 rate_coefficient.push_back(value);
71 }
72 myfile.close();
73 }
74 else
75 mooseError("Unable to open file");
76
77 _coefficient_interpolation.setData(val_x, rate_coefficient);
78}
