88template <typename T>
89T
90pow(T x, int e)
91{
92 bool neg = false;
93 T result = 1.0;
94
95 if (e < 0)
96 {
97 neg = true;
98 e = -e;
99 }
100
101 while (e)
102 {
103 // if bit 0 is set multiply the current power of two factor of the exponent
104 if (e & 1)
105 result *= x;
106
107 // x is incrementally set to consecutive powers of powers of two
108 x *= x;
109
110 // bit shift the exponent down
111 e >>= 1;
112 }
113
114 return neg ? 1.0 / result : result;
115}

229template <typename C,
230 typename T,
231 typename R = typename libMesh::CompareTypes<typename C::value_type, T>::supertype>
232R
233poly(const C & c, const T x, const bool derivative = false)
234{
235 const auto size = c.size();
236 if (size == 0)
237 return 0.0;
238
239 R value = c[0];
240 if (derivative)
241 {
242 value *= size - 1;
243 for (std::size_t i = 1; i < size - 1; ++i)
244 value = value * x + c[i] * (size - i - 1);
245 }
246 else
247 {
248 for (std::size_t i = 1; i < size; ++i)
249 value = value * x + c[i];
250 }
251
252 return value;
253}

264template <typename C,
265 typename T,
266 typename R = typename libMesh::CompareTypes<typename C::value_type, T>::supertype>
267R
268polynomial(const C & c, const T x)
269{
270 auto size = c.size();
271 if (size == 0)
272 return 0.0;
273
274 size--;
275 R value = c[size];
276 for (std::size_t i = 1; i <= size; ++i)
277 value = value * x + c[size - i];
278
279 return value;
280}

285template <typename C,
286 typename T,
287 typename R = typename libMesh::CompareTypes<typename C::value_type, T>::supertype>
288R
289polynomialDerivative(const C & c, const T x)
290{
291 auto size = c.size();
292 if (size <= 1)
293 return 0.0;
294
295 size--;
296 R value = c[size] * size;
297 for (std::size_t i = 1; i < size; ++i)
298 value = value * x + c[size - i] * (size - i);
299
300 return value;
301}

314template <typename T, typename T2>
315T
316smootherStep(T x, T2 start, T2 end, bool derivative = false)
317{
318 mooseAssert("start < end", "Start value must be lower than end value for smootherStep");
319 if (x <= start)
320 return 0.0;
321 else if (x >= end)
322 {
323 if (derivative)
324 return 0.0;
325 else
326 return 1.0;
327 }
328 x = (x - start) / (end - start);
329 if (derivative)
330 return 30.0 * Utility::pow<2>(x) * (x * (x - 2.0) + 1.0) / (end - start);
331 return Utility::pow<3>(x) * (x * (x * 6.0 - 15.0) + 10.0);
332}

340template <ComputeType compute_type, typename X, typename S, typename E>
341auto
342smootherStep(const X & x, const S & start, const E & end)
343{
344 mooseAssert("start < end", "Start value must be lower than end value for smootherStep");
345 if (x <= start)
346 return 0.0;
347 else if (x >= end)
348 {
349 if constexpr (compute_type == ComputeType::derivative)
350 return 0.0;
351 if constexpr (compute_type == ComputeType::value)
352 return 1.0;
353 }
354 const auto u = (x - start) / (end - start);
355 if constexpr (compute_type == ComputeType::derivative)
356 return 30.0 * Utility::pow<2>(u) * (u * (u - 2.0) + 1.0) / (end - start);
357 if constexpr (compute_type == ComputeType::value)
358 return Utility::pow<3>(u) * (u * (u * 6.0 - 15.0) + 10.0);
359}
