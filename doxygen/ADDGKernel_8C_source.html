52 _subproblem.haveADObjects(true);
53
54 addMooseVariableDependency(mooseVariable());
55
56 _save_in.resize(_save_in_strings.size());
57 _diag_save_in.resize(_diag_save_in_strings.size());
58
59 for (unsigned int i = 0; i < _save_in_strings.size(); i++)
60 {
61 MooseVariableFEBase * var = &_subproblem.getVariable(_tid,
62 _save_in_strings[i],
63 Moose::VarKindType::VAR_AUXILIARY,
64 Moose::VarFieldType::VAR_FIELD_STANDARD);
65
66 if (_sys.hasVariable(_save_in_strings[i]))
67 mooseError("Trying to use solution variable " + _save_in_strings[i] +
68 " as a save_in variable in " + name());
69
70 if (var->feType() != _var.feType())
71 paramError(
72 "save_in",
73 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
74 moose::internal::incompatVarMsg(*var, _var));
75
76 _save_in[i] = var;
77 var->sys().addVariableToZeroOnResidual(_save_in_strings[i]);
78 addMooseVariableDependency(var);
79 }
80
81 _has_save_in = _save_in.size() > 0;
82
83 for (unsigned int i = 0; i < _diag_save_in_strings.size(); i++)
84 {
85 MooseVariableFEBase * var = &_subproblem.getVariable(_tid,
86 _diag_save_in_strings[i],
87 Moose::VarKindType::VAR_NONLINEAR,
88 Moose::VarFieldType::VAR_FIELD_STANDARD);
89
90 if (_sys.hasVariable(_diag_save_in_strings[i]))
91 mooseError("Trying to use solution variable " + _diag_save_in_strings[i] +
92 " as a diag_save_in variable in " + name());
93
94 if (var->feType() != _var.feType())
95 paramError(
96 "diag_save_in",
97 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
98 moose::internal::incompatVarMsg(*var, _var));
99
100 _diag_save_in[i] = var;
101 var->sys().addVariableToZeroOnJacobian(_diag_save_in_strings[i]);
102 addMooseVariableDependency(var);
103 }
104
105 _has_diag_save_in = _diag_save_in.size() > 0;
106}

108void
109ADDGKernel::computeElemNeighResidual(Moose::DGResidualType type)
110{
111 bool is_elem;
112 if (type == Moose::Element)
113 is_elem = true;
114 else
115 is_elem = false;
116
117 const VariableTestValue & test_space = is_elem ? _test : _test_neighbor;
118
119 if (is_elem)
120 prepareVectorTag(_assembly, _var.number());
121 else
122 prepareVectorTagNeighbor(_assembly, _var.number());
123
124 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
125 for (_i = 0; _i < test_space.size(); _i++)
126 _local_re(_i) += raw_value(_JxW[_qp] * _coord[_qp] * computeQpResidual(type));
127
128 accumulateTaggedLocalResidual();
129
130 if (_has_save_in)
131 {
132 Threads::spin_mutex::scoped_lock lock(_resid_vars_mutex);
133 for (const auto & var : _save_in)
134 {
135 const std::vector<dof_id_type> & dof_indices =
136 is_elem ? var->dofIndices() : var->dofIndicesNeighbor();
137 var->sys().solution().add_vector(_local_re, dof_indices);
138 }
139 }
140}

154void
155ADDGKernel::computeElemNeighJacobian(Moose::DGJacobianType type)
156{
157 mooseAssert(type == Moose::ElementElement || type == Moose::NeighborNeighbor,
158 "With AD you should need one call per side");
159
160 const VariableTestValue & test_space = type == Moose::ElementElement ? _test : _test_neighbor;
161
162 std::vector<ADReal> residuals(test_space.size(), 0);
163
164 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
165 for (_i = 0; _i < test_space.size(); _i++)
166 residuals[_i] +=
167 _JxW[_qp] * _coord[_qp] *
168 computeQpResidual(type == Moose::ElementElement ? Moose::Element : Moose::Neighbor);
169
170 addJacobian(_assembly,
171 residuals,
172 type == Moose::ElementElement ? _var.dofIndices() : _var.dofIndicesNeighbor(),
173 _var.scalingFactor());
174
175 if (_has_diag_save_in)
176 {
177 unsigned int rows = _local_ke.m();
178 DenseVector<Number> diag(rows);
179 for (unsigned int i = 0; i < rows; i++)
180 diag(i) = _local_ke(i, i);
181
182 Threads::spin_mutex::scoped_lock lock(_jacoby_vars_mutex);
183 for (const auto & var : _diag_save_in)
184 {
185 if (type == Moose::ElementElement)
186 var->sys().solution().add_vector(diag, var->dofIndices());
187 else
188 var->sys().solution().add_vector(diag, var->dofIndicesNeighbor());
189 }
190 }
191}

194ADDGKernel::computeOffDiagJacobian(const unsigned int jvar_num)
195{

197 // once
198 if (!excludeBoundary() && jvar_num == _var.number())
199 {
200 const auto & jvar = getVariable(jvar_num);
201 computeOffDiagElemNeighJacobian(Moose::ElementElement, jvar);
202 computeOffDiagElemNeighJacobian(Moose::NeighborNeighbor, jvar);
203 }
