63void
64Syntax::addDependencySets(const std::string & action_sets)
65{
66 std::vector<std::string> sets, prev_names, tasks;
67 MooseUtils::tokenize(action_sets, sets, 1, "()");
68
69 for (unsigned int i = 0; i < sets.size(); ++i)
70 {
71 tasks.clear();
72 MooseUtils::tokenize(sets[i], tasks, 0, ", ");
73 for (unsigned int j = 0; j < tasks.size(); ++j)
74 {
75 // Each line should depend on each item in the previous line
76 for (unsigned int k = 0; k < prev_names.size(); ++k)
77 addDependency(tasks[j], prev_names[k]);
78 }
79 // Copy the the current items to the previous items for the next iteration
80 std::swap(tasks, prev_names);
81 }
82}

199std::vector<std::string>
200Syntax::getSyntaxByAction(const std::string & action, const std::string & task)
201{
202 // See if the reverse multimap has been built yet, if not build it now
203 if (!_actions_to_syntax_valid)
204 {
205 std::transform(_syntax_to_actions.begin(),
206 _syntax_to_actions.end(),
207 std::inserter(_actions_to_syntax, _actions_to_syntax.begin()),
208 [](const std::pair<std::string, ActionInfo> pair) {
209 return std::make_pair(pair.second._action,
210 std::make_pair(pair.first, pair.second._task));
211 });
212 _actions_to_syntax_valid = true;
213 }
214
215 std::vector<std::string> syntax;
216 auto it_pair = _actions_to_syntax.equal_range(action);
217 for (const auto & syntax_pair : as_range(it_pair))
218 // If task is blank, return all syntax, otherwise filter by task
219 if (task == "" || syntax_pair.second.second == task)
220 syntax.emplace_back(syntax_pair.second.first);
221
222 return syntax;
223}

225std::string
226Syntax::isAssociated(const std::string & real_id,
227 bool * is_parent,
228 const std::map<std::string, std::set<std::string>> & alt_map) const
229{
230 // if non-empty alt_map was provided then traverse its syntax instead of _syntax_to_actions
231 std::set<std::string> syntax_to_traverse;
232 if (!alt_map.empty())
233 std::transform(alt_map.begin(),
234 alt_map.end(),
235 std::inserter(syntax_to_traverse, syntax_to_traverse.end()),
236 [](auto pair) { return pair.first; });
237 else
238 std::transform(_syntax_to_actions.begin(),
239 _syntax_to_actions.end(),
240 std::inserter(syntax_to_traverse, syntax_to_traverse.end()),
241 [](auto pair) { return pair.first; });
242
249 bool local_is_parent;
250 if (is_parent == nullptr)
251 is_parent = &local_is_parent; // Just so we don't have to keep checking below when we want to
252 // set the value
253 std::vector<std::string> real_elements, reg_elements;
254 std::string return_value;
255
256 MooseUtils::tokenize(real_id, real_elements);
257
258 *is_parent = false;
259 for (auto it = syntax_to_traverse.rbegin(); it != syntax_to_traverse.rend(); ++it)
260 {
261 std::string reg_id = *it;
262 if (reg_id == real_id)
263 {
264 *is_parent = false;
265 return reg_id;
266 }
267 reg_elements.clear();
268 MooseUtils::tokenize(reg_id, reg_elements);
269 if (real_elements.size() <= reg_elements.size())
270 {
271 bool keep_going = true;
272 for (unsigned int j = 0; keep_going && j < real_elements.size(); ++j)
273 {
274 if (real_elements[j] != reg_elements[j] && reg_elements[j] != std::string("*"))
275 keep_going = false;
276 }
277 if (keep_going)
278 {
279 if (real_elements.size() < reg_elements.size() && !*is_parent)
280 {
281 // We found a parent, the longest parent in fact but we need to keep
282 // looking to make sure that the real thing isn't registered
283 *is_parent = true;
284 return_value = reg_id;
285 }
286 else if (real_elements.size() == reg_elements.size())
287 {
288 *is_parent = false;
289 return reg_id;
290 }
291 }
292 }
293 }
294
295 if (*is_parent)
296 return return_value;
297 else
298 return "";
299}
