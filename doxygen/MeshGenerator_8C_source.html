59const MeshGeneratorName *
60MeshGenerator::getMeshGeneratorNameFromParam(const std::string & param_name,
61 const bool allow_invalid) const
62{
63 const auto valid_param = isParamValid(param_name);
64 if (!allow_invalid)
65 {
66 if (!valid_param)
67 mooseError("Failed to get a parameter with the name \"",
68 param_name,
69 "\" when getting a MeshGenerator.",
70 "\n\nKnown parameters:\n",
71 _pars);
72 if (!_pars.isType<MeshGeneratorName>(param_name))
73 paramError(param_name,
74 "Parameter of type \"",
75 _pars.type(param_name),
76 "\" is not an expected type for getting a MeshGenerator (should be of type "
77 "\"MeshGeneratorName\")");
78 }
79 else if (!valid_param)
80 return nullptr;
81
82 const auto & name = getParam<MeshGeneratorName>(param_name);
83 checkGetMesh(name, param_name);
84
85 return &name;
86}

88const std::vector<MeshGeneratorName> &
89MeshGenerator::getMeshGeneratorNamesFromParam(const std::string & param_name) const
90{
91 if (!isParamValid(param_name))
92 mooseError("Failed to get a parameter with the name \"",
93 param_name,
94 "\" when getting MeshGenerators.",
95 "\n\nKnown parameters:\n",
96 _pars);
97 if (!_pars.isType<std::vector<MeshGeneratorName>>(param_name))
98 paramError(param_name,
99 "Parameter of type \"",
100 _pars.type(param_name),
101 "\" is not an expected type for getting MeshGenerators (should be of type "
102 "\"std::vector<MeshGeneratorName>\")");
103
104 const auto & names = getParam<std::vector<MeshGeneratorName>>(param_name);
105 for (const auto & name : names)
106 checkGetMesh(name, param_name);
107
108 return names;
109}

111void
112MeshGenerator::checkGetMesh(const MeshGeneratorName & mesh_generator_name,
113 const std::string & param_name) const
114{
115 mooseAssert(!mesh_generator_name.empty(), "Empty name");
116 const auto & mg_sys = _app.getMeshGeneratorSystem();
117 if (!_app.constructingMeshGenerators())
118 mooseError("Cannot get a mesh outside of construction");
119 if (!mg_sys.hasMeshGenerator(mesh_generator_name) && !isNullMeshName(mesh_generator_name))
120 {
121 std::stringstream error;
122 error << "The requested MeshGenerator with name '" << mesh_generator_name << "' ";
123 if (mg_sys.hasMeshGeneratorParams(mesh_generator_name))
124 error << "was found, but has not been constructed yet.\n\nThis can occur when your "
125 "dependencies are not properly defined and we cannot infer the proper construction "
126 "order of your MeshGenerators.\n\nThe most likely case is a sub generator whose "
127 "input(s) are not declared as a sub dependency in the generator creating them.";
128 else
129 error << "was not found.";
130 if (param_name.size())
131 paramError(param_name, error.str());
132 else
133 mooseError(error.str());
134 }
135}

152std::unique_ptr<MeshBase> &
153MeshGenerator::getMeshByName(const MeshGeneratorName & mesh_generator_name)
154{
155 checkGetMesh(mesh_generator_name, "");
156 if (isNullMeshName(mesh_generator_name))
157 return _null_mesh;
158
159 _requested_mesh_generators.insert(mesh_generator_name);
160 auto & mesh = _app.getMeshGeneratorSystem().getMeshGeneratorOutput(mesh_generator_name);
161 _requested_meshes.emplace_back(mesh_generator_name, &mesh);
162 return mesh;
163}

225std::unique_ptr<MeshBase>
226MeshGenerator::generateInternal()
227{
228 libmesh_parallel_only(comm());
229 mooseAssert(comm().verify(type() + name()), "Inconsistent execution ordering");
230
231#ifndef NDEBUG
232 for (const auto & [name, mesh] : _requested_meshes)
233 mooseAssert(*mesh, "Null output from " + name);
234#endif
235
236 auto mesh = generate();
237 mooseAssert(mesh, "Null output");
238
239 for (const auto & [requested_name, requested_mesh] : _requested_meshes)
240 if (*requested_mesh)
241 mooseError(
242 "The mesh from input ",
243 _app.getMeshGenerator(requested_name).type(),
244 " '",
245 _app.getMeshGenerator(requested_name).name(),
246 "' was not moved.\n\nThe MeshGenerator system requires that the memory from all input "
247 "meshes\nare managed by the requesting MeshGenerator during the generate phase.\n\nThis "
248 "is achieved with a std::move() operation within the generate() method.");
249
250 if (getParam<bool>("show_info"))
251 {
252 const auto mesh_info = mesh->get_info(/* verbosity = */ 2);
253
254 // We will prefix all information with "type() 'name()':" because this could potentially
255 // output a ton of information and looks a bit better with a prefix
256 std::stringstream oss;
257 const auto split = MooseUtils::split(mesh_info, "\n");
258 if (split.size())
259 for (std::size_t i = 0; i < split.size() - 1; ++i) // ignore the last line break
260 oss << COLOR_CYAN << "" << type() << " '" << name() << "': " << COLOR_DEFAULT << split[i]
261 << std::endl;
262 _console << oss.str() << std::flush;
263 }
264
265 // output the current mesh block to file
266 if (hasOutput())
267 {
268 if (!mesh->is_prepared())
269 mesh->prepare_for_use();
270
271 if (!getParam<bool>("nemesis"))
272 {
273 ExodusII_IO exio(*mesh);
274
275 if (mesh->mesh_dimension() == 1)
276 exio.write_as_dimension(3);
277
278 // Default to non-HDF5 output for wider compatibility
279 exio.set_hdf5_writing(false);
280
281 exio.write(name() + "_in.e");
282 }
283 else
284 {
285 Nemesis_IO nemesis_io(*mesh);
286
287 // Default to non-HDF5 output for wider compatibility
288 nemesis_io.set_hdf5_writing(false);
289
290 nemesis_io.write(name() + "_in.e");
291 }
292 }
293
294 return mesh;
295}
