22InputParameters
23MultiAppTransfer::validParams()
24{
25 InputParameters params = Transfer::validParams();
26 params.addDeprecatedParam<MultiAppName>("multi_app",
27 "The name of the MultiApp to transfer data with",
28 "Use to_multiapp & from_multiapp parameters now");
29 params.addParam<MultiAppName>("from_multi_app", "The name of the MultiApp to receive data from");
30 params.addParam<MultiAppName>("to_multi_app", "The name of the MultiApp to transfer the data to");

40 "check_multiapp_execute_on",
41 true,
42 "When false the check between the multiapp and transfer execute on flags is not performed.");
43 params.addParam<bool>("displaced_source_mesh",
44 false,
45 "Whether or not to use the displaced mesh for the source mesh.");
46 params.addParam<bool>("displaced_target_mesh",
47 false,
48 "Whether or not to use the displaced mesh for the target mesh.");
49 addSkipCoordCollapsingParam(params);
50 return params;

80MultiAppTransfer::MultiAppTransfer(const InputParameters & parameters)
81 : Transfer(parameters),
82 _skip_coordinate_collapsing(getParam<bool>("skip_coordinate_collapsing")),
83 _displaced_source_mesh(getParam<bool>("displaced_source_mesh")),
84 _displaced_target_mesh(getParam<bool>("displaced_target_mesh")),
85 _bbox_factor(isParamValid("bbox_factor") ? getParam<Real>("bbox_factor") : 1)
86{
87 // Get the multiapps from their names
88 if (!isParamValid("multi_app"))
89 {
90 if (isParamValid("from_multi_app"))
91 {
92 _from_multi_app = _fe_problem.getMultiApp(getParam<MultiAppName>("from_multi_app"));
93 _multi_app = _from_multi_app;
94 }
95 if (isParamValid("to_multi_app"))
96 {
97 _to_multi_app = _fe_problem.getMultiApp(getParam<MultiAppName>("to_multi_app"));
98 _multi_app = _to_multi_app;
99 }
100 if (!isParamValid("direction") && !isParamValid("from_multi_app") &&
101 !isParamValid("to_multi_app"))
102 mooseError("from_multi_app and/or to_multi_app must be specified");
103 }
104 else
105 {
106 // Check deprecated direction parameter
107 for (const auto & dir : _directions)
108 {
109 if (dir == FROM_MULTIAPP)
110 {
111 _from_multi_app = _fe_problem.getMultiApp(getParam<MultiAppName>("multi_app"));
112 _multi_app = _from_multi_app;
113 }
114 else if (dir == TO_MULTIAPP)
115 {
116 _to_multi_app = _fe_problem.getMultiApp(getParam<MultiAppName>("multi_app"));
117 _multi_app = _to_multi_app;
118 }
119 else
120 paramError("direction",
121 "BETWEN_MULTIAPP transfers should be specified using to/from_multi_app");
122 }
123 }
124
125 if (getParam<bool>("check_multiapp_execute_on"))
126 checkMultiAppExecuteOn();
127
128 // Fill direction attributes, for backward compatibility but also convenience
129 if (!isParamValid("direction"))
130 {
131 if (_from_multi_app && (!_to_multi_app || _from_multi_app == _to_multi_app))
132 _directions.push_back("from_multiapp");
133 if (_to_multi_app && (!_from_multi_app || _from_multi_app == _to_multi_app))
134 _directions.push_back("to_multiapp");
135 if (_from_multi_app && _to_multi_app && _from_multi_app != _to_multi_app)
136 _directions.push_back("between_multiapp");
137
138 // So it's available in the next constructors
139 _direction = _directions[0];
140 _current_direction = _directions[0];
141 }
142
143 // Check for different number of subapps
144 if (_to_multi_app && _from_multi_app &&
145 _from_multi_app->numGlobalApps() != _to_multi_app->numGlobalApps())
146 mooseError(
147 "Between multiapp transfer is only supported with the same number of subapps per MultiApp");
148
149 // Handle deprecated parameters
150 if (parameters.isParamSetByUser("direction"))
151 {
152 if (!isParamValid("multi_app"))
153 paramError("direction",
154 "The deprecated 'direction' parameter is meant to be used in conjunction with the "
155 "'multi_app' parameter");
156 if (isParamValid("to_multi_app") || isParamValid("from_multi_app"))
157 paramError("direction",
158 "The deprecated 'direction' parameter is not meant to be used in conjunction with "
159 "the 'from_multi_app' or 'to_multi_app' parameters");
160 }
161}

181 getExecuteOnEnum() != _to_multi_app->getExecuteOnEnum())
182 mooseDoOnce(
183 mooseWarning("MultiAppTransfer execute_on flags do not match associated to_multi_app "
184 "and from_multi_app execute_on flags"));

187void
188MultiAppTransfer::variableIntegrityCheck(const AuxVariableName & var_name) const
189{
190 bool variable_found = false;
191 bool has_an_app = false;
192
193 // Check the from_multi_app for the variable
194 if (_from_multi_app)
195 for (unsigned int i = 0; i < _from_multi_app->numGlobalApps(); i++)
196 if (_from_multi_app->hasLocalApp(i))
197 {
198 has_an_app = true;
199 if (_from_multi_app->appProblemBase(i).hasVariable(var_name))
200 variable_found = true;
201 }
202
203 // Check the to_multi_app for the variable
204 if (_to_multi_app)
205 for (unsigned int i = 0; i < _to_multi_app->numGlobalApps(); i++)
206 if (_to_multi_app->hasLocalApp(i))
207 {
208 has_an_app = true;
209 if (_to_multi_app->appProblemBase(i).hasVariable(var_name))
210 variable_found = true;
211 }
212
213 if (!variable_found && has_an_app)
214 mooseError("Cannot find variable ", var_name, " for ", name(), " Transfer");
215}

228void
229MultiAppTransfer::getAppInfo()
230{
231 // I would like to do all of this in initialSetup, but it will fail with
232 // multiapps that reset. A reset deletes and rebuilds the FEProblems so all
233 // of the pointers will be broken.
234
235 // Clear the vectors since we've probably built them up from a previous call
236 _from_problems.clear();
237 _to_problems.clear();
238 _from_es.clear();
239 _to_es.clear();
240 _from_meshes.clear();
241 _to_meshes.clear();
242 _to_positions.clear();
243 _from_positions.clear();
244 _to_transforms.clear();
245 _from_transforms.clear();
246 // Clear this map since we build it from scratch every time we transfer
247 // Otherwise, this will cause two issues: 1) increasing memory usage
248 // for a simulation that requires many transfers, 2) producing wrong results
249 // when we do collective communication on this vector.
250 _to_local2global_map.clear();
251 _from_local2global_map.clear();
252
253 // Build the vectors for to problems, from problems, and subapps positions.
254 if (_current_direction == FROM_MULTIAPP)
255 {
256 _to_problems.push_back(&_from_multi_app->problemBase());
257 _to_positions.push_back(Point(0., 0., 0.));
258 getFromMultiAppInfo();
259 }
260 else if (_current_direction == TO_MULTIAPP)
261 {
262 _from_problems.push_back(&_to_multi_app->problemBase());
263 _from_positions.push_back(Point(0., 0., 0.));
264 getToMultiAppInfo();
265 }
266 else if (_current_direction == BETWEEN_MULTIAPP)
267 {
268 mooseAssert(&_from_multi_app->problemBase().coordTransform() ==
269 &_to_multi_app->problemBase().coordTransform(),
270 "I believe these should be the same. If not, then it will be difficult to define a "
271 "canonical reference frame.");
272 getToMultiAppInfo();
273 getFromMultiAppInfo();
274 }
275
276 // Build the from and to equation systems and mesh vectors.
277 for (unsigned int i = 0; i < _to_problems.size(); i++)
278 {
279 // TODO: Do I actually want es or displaced es?
280 _to_es.push_back(&_to_problems[i]->es());
281 if (_displaced_target_mesh && _to_problems[i]->getDisplacedProblem())
282 _to_meshes.push_back(&_to_problems[i]->getDisplacedProblem()->mesh());
283 else
284 _to_meshes.push_back(&_to_problems[i]->mesh());
285 }
286
287 for (unsigned int i = 0; i < _from_problems.size(); i++)
288 {
289 _from_es.push_back(&_from_problems[i]->es());
290 if (_displaced_source_mesh && _from_problems[i]->getDisplacedProblem())
291 _from_meshes.push_back(&_from_problems[i]->getDisplacedProblem()->mesh());
292 else
293 _from_meshes.push_back(&_from_problems[i]->mesh());
294 }
295
296 MooseAppCoordTransform::MinimalData from_app_transform_construction_data{};
297 if (_communicator.rank() == 0)
298 from_app_transform_construction_data =
299 _current_direction == TO_MULTIAPP
300 ? _to_multi_app->problemBase().coordTransform().minimalDataDescription()
301 : _from_multi_app->appProblemBase(0).coordTransform().minimalDataDescription();
302 _communicator.broadcast(from_app_transform_construction_data);
303 _from_moose_app_transform =
304 std::make_unique<MooseAppCoordTransform>(from_app_transform_construction_data);
305
306 MooseAppCoordTransform::MinimalData to_app_transform_construction_data{};
307 if (_communicator.rank() == 0)
308 to_app_transform_construction_data =
309 _current_direction == FROM_MULTIAPP
310 ? _from_multi_app->problemBase().coordTransform().minimalDataDescription()
311 : _to_multi_app->appProblemBase(0).coordTransform().minimalDataDescription();
312 _communicator.broadcast(to_app_transform_construction_data);
313 _to_moose_app_transform =
314 std::make_unique<MooseAppCoordTransform>(to_app_transform_construction_data);
315
316 /*
317 * skip_coordinate_collapsing: whether to set the transform to skip coordinate collapsing
318 * (from XYZ to RZ for example)
319 * transforms: vector of transforms to add the new transforms to
320 * moose_app_transform: base for the new transform
321 * is_parent_app_transform: whether working on the transform for the parent app (this app, the
322 * one creating the transfer) or for child apps
323 * multiapp: pointer to the multiapp to obtain the position of the child apps
324 */
325 auto create_multiapp_transforms = [this](auto & transforms,
326 const auto & moose_app_transform,
327 const bool is_parent_app_transform,
328 const MultiApp * const multiapp = nullptr)
329 {
330 mooseAssert(is_parent_app_transform || multiapp,
331 "Coordinate transform must be created either for child app or parent app");
332 if (is_parent_app_transform)
333 {
334 transforms.push_back(std::make_unique<MultiAppCoordTransform>(moose_app_transform));
335 transforms.back()->skipCoordinateCollapsing(_skip_coordinate_collapsing);
336 // zero translation
337 }
338 else
339 {
340 mooseAssert(transforms.size() == 0, "transforms should not be initialized at this point");
341 for (const auto i : make_range(multiapp->numGlobalApps()))
342 {
343 transforms.push_back(std::make_unique<MultiAppCoordTransform>(moose_app_transform));
344 auto & transform = transforms[i];
345 transform->skipCoordinateCollapsing(_skip_coordinate_collapsing);
346 if (multiapp->usingPositions())
347 transform->setTranslationVector(multiapp->position(i));
348 }
349 }
350 };
351
352 if (_current_direction == TO_MULTIAPP)
353 {
354 create_multiapp_transforms(
355 _to_transforms, *_to_moose_app_transform, false, _to_multi_app.get());
356 create_multiapp_transforms(_from_transforms, *_from_moose_app_transform, true);
357 }
358 if (_current_direction == FROM_MULTIAPP)
359 {
360 create_multiapp_transforms(_to_transforms, *_to_moose_app_transform, true);
361 create_multiapp_transforms(
362 _from_transforms, *_from_moose_app_transform, false, _from_multi_app.get());
363 }
364 if (_current_direction == BETWEEN_MULTIAPP)
365 {
366 create_multiapp_transforms(
367 _to_transforms, *_to_moose_app_transform, false, _to_multi_app.get());
368 create_multiapp_transforms(
369 _from_transforms, *_from_moose_app_transform, false, _from_multi_app.get());
370 }
371
372 auto check_transform_compatibility = [this](const MultiAppCoordTransform & transform)
373 {
374 if (transform.hasNonTranslationTransformation() && !usesMooseAppCoordTransform())
375 mooseWarning("Transfer '",
376 name(),
377 "' of type '",
378 type(),
379 "' has non-translation transformations but it does not implement coordinate "
380 "transformations using the 'MooseAppCoordTransform' class. Your data transfers "
381 "will not be performed in the expected transformed frame");
382 };
383
384 // set the destination coordinate systems for each transform for the purposes of determining
385 // coordinate collapsing. For example if TO is XYZ and FROM is RZ, then TO will have its XYZ
386 // coordinates collapsed into RZ and FROM will have a no-op for coordinate collapsing
387
388 for (const auto i : index_range(_from_transforms))
389 {
390 auto & from_transform = _from_transforms[i];
391 from_transform->setDestinationCoordTransform(*_to_moose_app_transform);
392 if (i == 0)
393 check_transform_compatibility(*from_transform);
394 }
395 for (const auto i : index_range(_to_transforms))
396 {
397 auto & to_transform = _to_transforms[i];
398 to_transform->setDestinationCoordTransform(*_from_moose_app_transform);
399 if (i == 0)
400 check_transform_compatibility(*to_transform);
401 }
402}

451void
452MultiAppTransfer::extendBoundingBoxes(const Real factor, std::vector<BoundingBox> & bboxes) const
453{
454 const auto extension_factor = factor - 1;
455
456 // Extend (or contract if the extension factor is negative) bounding boxes along all the
457 // directions by the same length. Greater than zero values of this member may be necessary because
458 // the nearest bounding box does not necessarily give you the closest node/element. It will depend
459 // on the partition and geometry. A node/element will more likely find its nearest source
460 // element/node by extending bounding boxes. If each of the bounding boxes covers the entire
461 // domain, a node/element will be able to find its nearest source element/node for sure, but at
462 // the same time, more communication will be involved and can be expensive.
463 for (auto & box : bboxes)
464 {
465 // libmesh set an invalid bounding box using this code
466 // for (unsigned int i=0; i<LIBMESH_DIM; i++)
467 // {
468 // this->first(i) = std::numeric_limits<Real>::max();
469 // this->second(i) = -std::numeric_limits<Real>::max();
470 // }
471 // If it is an invalid box, we should skip it
472 if (box.first(0) == std::numeric_limits<Real>::max())
473 continue;
474
475 auto width = box.second - box.first;
476 box.second += width * extension_factor;
477 box.first -= width * extension_factor;
478 }
479}

481std::vector<BoundingBox>
482MultiAppTransfer::getFromBoundingBoxes()
483{
484 std::vector<std::pair<Point, Point>> bb_points(_from_meshes.size());
485 for (unsigned int i = 0; i < _from_meshes.size(); i++)
486 {
487 // Get a bounding box around the mesh elements that are local to the current
488 // processor.
489 BoundingBox bbox = MeshTools::create_local_bounding_box(*_from_meshes[i]);
490
491 // Translate the bounding box to the from domain's position. We may have rotations so we must
492 // be careful in constructing the new min and max (first and second)
493 const auto from_global_num = getGlobalSourceAppIndex(i);
494 transformBoundingBox(bbox, *_from_transforms[from_global_num]);
495
496 // Cast the bounding box into a pair of points (so it can be put through
497 // MPI communication).
498 bb_points[i] = static_cast<std::pair<Point, Point>>(bbox);
499 }
500
501 // Serialize the bounding box points.
502 _communicator.allgather(bb_points);
503
504 // Recast the points back into bounding boxes and return.
505 std::vector<BoundingBox> bboxes(bb_points.size());
506 for (unsigned int i = 0; i < bb_points.size(); i++)
507 bboxes[i] = static_cast<BoundingBox>(bb_points[i]);
508
509 // possibly extend bounding boxes
510 extendBoundingBoxes(_bbox_factor, bboxes);
511
512 return bboxes;
513}

515std::vector<BoundingBox>
516MultiAppTransfer::getFromBoundingBoxes(BoundaryID boundary_id)
517{
518 std::vector<std::pair<Point, Point>> bb_points(_from_meshes.size());
519 const Real min_r = std::numeric_limits<Real>::lowest();
520 const Real max_r = std::numeric_limits<Real>::max();
521
522 for (unsigned int i = 0; i < _from_meshes.size(); i++)
523 {
524
525 Point min(max_r, max_r, max_r);
526 Point max(min_r, min_r, min_r);
527 bool at_least_one = false;
528
529 // TODO: Factor this into mesh_tools after adding new boundary bounding box routine.
530 const ConstBndNodeRange & bnd_nodes = *_from_meshes[i]->getBoundaryNodeRange();
531 for (const auto & bnode : bnd_nodes)
532 {
533 if (bnode->_bnd_id == boundary_id &&
534 bnode->_node->processor_id() == _from_meshes[i]->processor_id())
535 {
536 at_least_one = true;
537 const auto & node = *bnode->_node;
538 for (const auto i : make_range(Moose::dim))
539 {
540 min(i) = std::min(min(i), node(i));
541 max(i) = std::max(max(i), node(i));
542 }
543 }
544 }
545
546 BoundingBox bbox(min, max);
547 if (!at_least_one)
548 bbox.min() = max; // If we didn't hit any nodes, this will be _the_ minimum bbox
549 else
550 {
551 // Translate the bounding box to the from domain's position. We may have rotations so we must
552 // be careful in constructing the new min and max (first and second)
553 const auto from_global_num = getGlobalSourceAppIndex(i);
554 transformBoundingBox(bbox, *_from_transforms[from_global_num]);
555 }
556
557 // Cast the bounding box into a pair of points (so it can be put through
558 // MPI communication).
559 bb_points[i] = static_cast<std::pair<Point, Point>>(bbox);
560 }
561
562 // Serialize the bounding box points.
563 _communicator.allgather(bb_points);
564
565 // Recast the points back into bounding boxes and return.
566 std::vector<BoundingBox> bboxes(bb_points.size());
567 for (unsigned int i = 0; i < bb_points.size(); i++)
568 bboxes[i] = static_cast<BoundingBox>(bb_points[i]);
569
570 // possibly extend bounding boxes
571 extendBoundingBoxes(_bbox_factor, bboxes);
572
573 return bboxes;
574}

612Point
613MultiAppTransfer::getPointInTargetAppFrame(const Point & p,
614 unsigned int local_i_to,
615 const std::string & phase) const
616{
617 const auto & to_transform = _to_transforms[getGlobalTargetAppIndex(local_i_to)];
618 if (to_transform->hasCoordinateSystemTypeChange())
619 {
620 if (!_skip_coordinate_collapsing)
621 mooseInfo(phase + " cannot use the point in the target app frame due to the "
622 "non-uniqueness of the coordinate collapsing reverse mapping."
623 " Coordinate collapse is ignored for this operation");
624 to_transform->skipCoordinateCollapsing(true);
625 const auto target_point = to_transform->mapBack(p);
626 to_transform->skipCoordinateCollapsing(false);
627 return target_point;
628 }
629 else
630 return to_transform->mapBack(p);
631}
