16InputParameters
17ProjectionAux::validParams()
18{
19 InputParameters params = AuxKernel::validParams();
20 params.addClassDescription(
21 "Returns the specified variable as an auxiliary variable with a projection of the source "
22 "variable. If they are the same type, this amounts to a simple copy.");
23 params.addRequiredCoupledVar("v", "Variable to take the value of.");
24
25 // Technically possible to project from nodal to elemental and back
26 params.set<bool>("_allow_nodal_to_elemental_coupling") = true;
27
28 // We need some ghosting for all elemental to nodal projections
29 params.addParam<unsigned short>("ghost_layers", 1, "The number of layers of elements to ghost.");
30 params.addRelationshipManager("ElementPointNeighborLayers",
31 Moose::RelationshipManagerType::ALGEBRAIC,
32 [](const InputParameters & obj_params, InputParameters & rm_params)
33 {
34 rm_params.set<unsigned short>("layers") =
35 obj_params.get<unsigned short>("ghost_layers");
36 rm_params.set<bool>("use_displaced_mesh") =
37 obj_params.get<bool>("use_displaced_mesh");
38 });
39 return params;
40}

53Real
54ProjectionAux::computeValue()
55{
56 if (!isNodal() || (_source_variable.isNodal() && _source_variable.order() >= _var.order()))
57 return _v[_qp];
58 // projecting continuous elemental variable onto a nodal one
59 // AND nodal low order -> nodal higher order
60 else if (isNodal() && _source_variable.getContinuity() != DISCONTINUOUS &&
61 _source_variable.getContinuity() != SIDE_DISCONTINUOUS)
62 return _source_sys.system().point_value(
63 _source_variable.number(), *_current_node, elemOnNodeVariableIsDefinedOn());
64 // Handle discontinuous elemental variable projection into a nodal variable
65 else
66 {
67 // Custom projection rule : use nodal values, if discontinuous the one coming from each element,
68 // weighted by element volumes
69 // First, find all the elements that this node is part of
70 auto elem_ids = _mesh.nodeToElemMap().find(_current_node->id());
71 mooseAssert(elem_ids != _mesh.nodeToElemMap().end(),
72 "Should have found an element around node " + std::to_string(_current_node->id()));
73
74 // Get the neighbor element centroid values & element volumes
75 Real sum_weighted_values = 0;
76 Real sum_volumes = 0;
77 for (auto & id : elem_ids->second)
78 {
79 const auto & elem = _mesh.elemPtr(id);
80 if (_source_variable.hasBlocks(elem->subdomain_id()))
81 {
82 const auto elem_volume = elem->volume();
83 sum_weighted_values +=
84 _source_sys.system().point_value(_source_variable.number(), *_current_node, elem) *
85 elem_volume;
86 sum_volumes += elem_volume;
87 }
88 }
89 return sum_weighted_values / sum_volumes;
90 }
91}

99 mooseError("Source variable is not defined everywhere the target variable is");
100}
