65RealVectorValue
66VectorFunctionDirichletBC::computeQpResidual()
67{
68 if (_function)
69 _values = _function->vectorValue(_t, *_current_node);
70 else
71 _values = {_function_x.value(_t, *_current_node),
72 _function_y.value(_t, *_current_node),
73 _function_z.value(_t, *_current_node)};
74
75 return _u - _values;
76}
