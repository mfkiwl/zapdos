37MooseAppPtr
38AppFactory::createAppShared(const std::string & default_app_type,
39 int argc,
40 char ** argv,
41 MPI_Comm comm_world_in)
42{
43 auto command_line = std::make_shared<CommandLine>(argc, argv);
44 auto which_app_param = emptyInputParameters();
45 MooseApp::addAppParam(which_app_param);
46 command_line->addCommandLineOptionsFromParams(which_app_param);
47 std::string app_type;
48 if (!command_line->search("app_to_run", app_type))
49 app_type = default_app_type;
50
51 auto app_params = AppFactory::instance().getValidParams(app_type);
52
53 app_params.set<int>("_argc") = argc;
54 app_params.set<char **>("_argv") = argv;
55 app_params.set<std::shared_ptr<CommandLine>>("_command_line") = command_line;
56
57 return AppFactory::instance().createShared(app_type, "main", app_params, comm_world_in);
58}

60MooseAppPtr
61AppFactory::createShared(const std::string & app_type,
62 const std::string & name,
63 InputParameters parameters,
64 MPI_Comm comm_world_in)
65{
66 // Error if the application type is not located
67 const auto it = _name_to_build_info.find(app_type);
68 if (it == _name_to_build_info.end())
69 mooseError("Object '" + app_type + "' was not registered.");
70 auto & build_info = it->second;
71
72 // Take the app_type and add it to the parameters so that it can be retrieved in the Application
73 parameters.set<std::string>("_type") = app_type;
74
75 // Check to make sure that all required parameters are supplied
76 parameters.checkParams("");
77
78 auto comm = std::make_shared<Parallel::Communicator>(comm_world_in);
79
80 parameters.set<std::shared_ptr<Parallel::Communicator>>("_comm") = comm;
81 parameters.set<std::string>("_app_name") = name;
82
83 if (!parameters.isParamValid("_command_line"))
84 mooseError("Valid CommandLine object required");
85
86 std::shared_ptr<CommandLine> command_line =
87 parameters.get<std::shared_ptr<CommandLine>>("_command_line");
88 command_line->addCommandLineOptionsFromParams(parameters);
89 command_line->populateInputParams(parameters);
90
91 build_info->_app_creation_count++;
92
93 return build_info->build(parameters);
94}
