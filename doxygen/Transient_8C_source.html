296void
297Transient::execute()
298{
299 preExecute();
300
301 // Start time loop...
302 while (keepGoing())
303 {
304 incrementStepOrReject();
305 preStep();
306 computeDT();
307 takeStep();
308 endStep();
309 postStep();
310 }
311
312 if (lastSolveConverged())
313 {
314 _t_step++;
315
316 /*
317 * Call the multi-app executioners endStep and
318 * postStep methods when doing Picard or when not automatically advancing sub-applications for
319 * some other reason. We do not perform these calls for loose-coupling/auto-advancement
320 * problems because Transient::endStep and Transient::postStep get called from
321 * TransientMultiApp::solveStep in that case.
322 */
323 if (!_fixed_point_solve->autoAdvance())
324 {
325 _problem.finishMultiAppStep(EXEC_MULTIAPP_FIXED_POINT_BEGIN,
326 /*recurse_through_multiapp_levels=*/true);
327 _problem.finishMultiAppStep(EXEC_TIMESTEP_BEGIN, /*recurse_through_multiapp_levels=*/true);
328 _problem.finishMultiAppStep(EXEC_TIMESTEP_END, /*recurse_through_multiapp_levels=*/true);
329 _problem.finishMultiAppStep(EXEC_MULTIAPP_FIXED_POINT_END,
330 /*recurse_through_multiapp_levels=*/true);
331 }
332 }
333
334 if (!_app.halfTransient())
335 {
336 TIME_SECTION("final", 1, "Executing Final Objects");
337 _problem.execMultiApps(EXEC_FINAL);
338 _problem.finalizeMultiApps();
339 _problem.execute(EXEC_FINAL);
340 _problem.outputStep(EXEC_FINAL);
341 }
342
343 // This method is to finalize anything else we want to do on the problem side.
344 _problem.postExecute();
345
346 // This method can be overridden for user defined activities in the Executioner.
347 postExecute();
348}

356void
357Transient::incrementStepOrReject()
358{
359 if (lastSolveConverged())
360 {
361 if (!_xfem_repeat_step)
362 {
363#ifdef LIBMESH_ENABLE_AMR
364 if (_t_step != 0)
365 _problem.adaptMesh();
366#endif
367
368 _time_old = _time;
369 _t_step++;
370
371 _problem.advanceState();
372
373 if (_t_step == 1)
374 return;
375
376 /*
377 * Call the multi-app executioners endStep and
378 * postStep methods when doing Picard or when not automatically advancing sub-applications for
379 * some other reason. We do not perform these calls for loose-coupling/auto-advancement
380 * problems because Transient::endStep and Transient::postStep get called from
381 * TransientMultiApp::solveStep in that case.
382 */
383 if (!_fixed_point_solve->autoAdvance())
384 {
385 _problem.finishMultiAppStep(EXEC_MULTIAPP_FIXED_POINT_BEGIN);
386 _problem.finishMultiAppStep(EXEC_TIMESTEP_BEGIN);
387 _problem.finishMultiAppStep(EXEC_TIMESTEP_END);
388 _problem.finishMultiAppStep(EXEC_MULTIAPP_FIXED_POINT_END);
389 }
390
391 /*
392 * Ensure that we increment the sub-application time steps so that
393 * when dt selection is made in the master application, we are using
394 * the correct time step information
395 */
396 _problem.incrementMultiAppTStep(EXEC_MULTIAPP_FIXED_POINT_BEGIN);
397 _problem.incrementMultiAppTStep(EXEC_TIMESTEP_BEGIN);
398 _problem.incrementMultiAppTStep(EXEC_TIMESTEP_END);
399 _problem.incrementMultiAppTStep(EXEC_MULTIAPP_FIXED_POINT_END);
400 }
401 }
402 else
403 {
404 _problem.restoreMultiApps(EXEC_MULTIAPP_FIXED_POINT_BEGIN, true);
405 _problem.restoreMultiApps(EXEC_TIMESTEP_BEGIN, true);
406 _problem.restoreMultiApps(EXEC_TIMESTEP_END, true);
407 _problem.restoreMultiApps(EXEC_MULTIAPP_FIXED_POINT_END, true);
408 _time_stepper->rejectStep();
409 _time = _time_old;
410 }

413void
414Transient::takeStep(Real input_dt)
415{
416 _dt_old = _dt;
417
418 if (input_dt == -1.0)
419 _dt = computeConstrainedDT();
420 else
421 _dt = input_dt;
422
423 _time_stepper->preSolve();
424
425 // Increment time
426 _time = _time_old + _dt;
427
428 _problem.timestepSetup();
429
430 _problem.onTimestepBegin();
431
432 _time_stepper->step();
433 _xfem_repeat_step = _fixed_point_solve->XFEMRepeatStep();
434
435 _last_solve_converged = _time_stepper->converged();
436
437 if (!lastSolveConverged())
438 {
439 _console << "Aborting as solve did not converge" << std::endl;
440 return;
441 }
442
443 if (!(_problem.haveXFEM() && _fixed_point_solve->XFEMRepeatStep()))
444 {
445 if (lastSolveConverged())
446 _time_stepper->acceptStep();
447 else
448 _time_stepper->rejectStep();
449 }
450
451 _time = _time_old;
452
453 _time_stepper->postSolve();
454
455 _solution_change_norm =
456 relativeSolutionDifferenceNorm() / (_normalize_solution_diff_norm_by_dt ? _dt : Real(1));
457
458 return;
459}

461void
462Transient::endStep(Real input_time)
463{
464 if (input_time == -1.0)
465 _time = _time_old + _dt;
466 else
467 _time = input_time;
468
469 if (lastSolveConverged())
470 {
471 if (_xfem_repeat_step)
472 _time = _time_old;
473 else
474 {
475 _nl.getTimeIntegrator()->postStep();
476
477 // Compute the Error Indicators and Markers
478 _problem.computeIndicators();
479 _problem.computeMarkers();
480
481 // Perform the output of the current time step
482 _problem.outputStep(EXEC_TIMESTEP_END);
483
484 // output
485 if (_time_interval && (_time + _timestep_tolerance >= _next_interval_output_time))
486 _next_interval_output_time += _time_interval_output_interval;
487 }
488 }

491Real
492Transient::computeConstrainedDT()
493{
494 // // If start up steps are needed
495 // if (_t_step == 1 && _n_startup_steps > 1)
496 // _dt = _input_dt/(double)(_n_startup_steps);
497 // else if (_t_step == 1+_n_startup_steps && _n_startup_steps > 1)
498 // _dt = _input_dt;
499
500 Real dt_cur = _dt;
501 std::ostringstream diag;
502
503 // After startup steps, compute new dt
504 if (_t_step > _n_startup_steps)
505 dt_cur = getDT();
506
507 else
508 {
509 diag << "Timestep < n_startup_steps, using old dt: " << std::setw(9) << std::setprecision(6)
510 << std::setfill('0') << std::showpoint << std::left << _dt << " tstep: " << _t_step
511 << " n_startup_steps: " << _n_startup_steps << std::endl;
512 }
513 _unconstrained_dt = dt_cur;
514
515 if (_verbose)
516 _console << diag.str();
517
518 diag.str("");
519 diag.clear();
520
521 // Allow the time stepper to limit the time step
522 _at_sync_point = _time_stepper->constrainStep(dt_cur);
523
524 // Don't let time go beyond next time interval output if specified
525 if ((_time_interval) && (_time + dt_cur + _timestep_tolerance >= _next_interval_output_time))
526 {
527 dt_cur = _next_interval_output_time - _time;
528 _at_sync_point = true;
529
530 diag << "Limiting dt for time interval output at time: " << std::setw(9) << std::setprecision(6)
531 << std::setfill('0') << std::showpoint << std::left << _next_interval_output_time
532 << " dt: " << std::setw(9) << std::setprecision(6) << std::setfill('0') << std::showpoint
533 << std::left << dt_cur << std::endl;
534 }
535
536 // If a target time is set and the current dt would exceed it, limit dt to match the target
537 if (_target_time > -std::numeric_limits<Real>::max() + _timestep_tolerance &&
538 _time + dt_cur + _timestep_tolerance >= _target_time)
539 {
540 dt_cur = _target_time - _time;
541 _at_sync_point = true;
542
543 diag << "Limiting dt for target time: " << std::setw(9) << std::setprecision(6)
544 << std::setfill('0') << std::showpoint << std::left << _next_interval_output_time
545 << " dt: " << std::setw(9) << std::setprecision(6) << std::setfill('0') << std::showpoint
546 << std::left << dt_cur << std::endl;
547 }
548
549 // Constrain by what the multi apps are doing
550 constrainDTFromMultiApp(dt_cur, diag, EXEC_MULTIAPP_FIXED_POINT_BEGIN);
551 constrainDTFromMultiApp(dt_cur, diag, EXEC_TIMESTEP_BEGIN);
552 constrainDTFromMultiApp(dt_cur, diag, EXEC_TIMESTEP_END);
553 constrainDTFromMultiApp(dt_cur, diag, EXEC_MULTIAPP_FIXED_POINT_END);
554
555 if (_verbose)
556 _console << diag.str();
557
558 return dt_cur;
559}

583bool
584Transient::keepGoing()
585{
586 bool keep_going = !_problem.isSolveTerminationRequested();
587
588 // Check for stop condition based upon steady-state check flag:
589 if (lastSolveConverged())
590 {
591 if (!_xfem_repeat_step)
592 {
593 if (_steady_state_detection == true && _time > _steady_state_start_time)
594 {
595 // Check solution difference relative norm against steady-state tolerance
596 if (_solution_change_norm < _steady_state_tolerance)
597 {
598 _console << "Steady-State Solution Achieved at time: " << _time << std::endl;
599 // Output last solve if not output previously by forcing it
600 keep_going = false;
601 }
602 else // Keep going
603 {
604 // Print steady-state relative error norm
605 _console << "Steady-State Relative Differential Norm: " << _solution_change_norm
606 << std::endl;
607 }
608 }
609
610 // Check for stop condition based upon number of simulation steps and/or solution end time:
611 if (static_cast<unsigned int>(_t_step) >= _num_steps)
612 keep_going = false;
613
614 if ((_time >= _end_time) || (fabs(_time - _end_time) <= _timestep_tolerance))
615 keep_going = false;
616 }
617 }
618 else if (_abort)
619 {
620 _console << "Aborting as solve did not converge and input selected to abort" << std::endl;
621 keep_going = false;
622 }
623 else if (!_error_on_dtmin && _dt <= _dtmin)
624 {
625 _console << "Aborting as timestep already at or below dtmin" << std::endl;
626 keep_going = false;
627 }
628
629 return keep_going;
630}

661void
662Transient::setupTimeIntegrator()
663{
664 if (_pars.isParamSetByUser("scheme") && _problem.hasTimeIntegrator())
665 mooseError("You cannot specify time_scheme in the Executioner and independently add a "
666 "TimeIntegrator to the system at the same time");
667
668 if (!_problem.hasTimeIntegrator())
669 {
670 // backwards compatibility
671 std::string ti_str;
672 using namespace Moose;
673
674 switch (_time_scheme)
675 {
676 case TI_IMPLICIT_EULER:
677 ti_str = "ImplicitEuler";
678 break;
679 case TI_EXPLICIT_EULER:
680 ti_str = "ExplicitEuler";
681 break;
682 case TI_CRANK_NICOLSON:
683 ti_str = "CrankNicolson";
684 break;
685 case TI_BDF2:
686 ti_str = "BDF2";
687 break;
688 case TI_EXPLICIT_MIDPOINT:
689 ti_str = "ExplicitMidpoint";
690 break;
691 case TI_LSTABLE_DIRK2:
692 ti_str = "LStableDirk2";
693 break;
694 case TI_EXPLICIT_TVD_RK_2:
695 ti_str = "ExplicitTVDRK2";
696 break;
697 case TI_NEWMARK_BETA:
698 ti_str = "NewmarkBeta";
699 break;
700 default:
701 mooseError("Unknown scheme: ", _time_scheme);
702 break;
703 }
704
705 InputParameters params = _app.getFactory().getValidParams(ti_str);
706 _problem.addTimeIntegrator(ti_str, ti_str, params);
707 }

710std::string
711Transient::getTimeStepperName() const
712{
713 if (_time_stepper)
714 {
715 TimeStepper & ts = *_time_stepper;
716 return demangle(typeid(ts).name());
717 }
