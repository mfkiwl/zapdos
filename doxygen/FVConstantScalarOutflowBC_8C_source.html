29ADReal
30FVConstantScalarOutflowBC::computeQpResidual()
31{
32 mooseAssert(_normal * _velocity >= 0,
33 "This boundary condition is for outflow but the flow is in the opposite direction of "
34 "the boundary normal");
35
36 // This will either be second or first order accurate depending on whether the user has asked
37 // for a two term expansion in their input file
38 return _normal * _velocity * _var.getBoundaryFaceValue(*_face_info, determineState());
39}
