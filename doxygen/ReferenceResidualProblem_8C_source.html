31 "a user-supplied reference quantity rather than "
32 "the initial residual");

34 "solution_variables", "Set of solution variables to be checked for relative convergence");
35 params.addParam<std::vector<AuxVariableName>>(
36 "reference_residual_variables",
37 "Set of variables that provide reference residuals for relative convergence check");
38 params.addParam<TagName>("reference_vector", "The tag name of the reference residual vector.");
39 params.addParam<Real>("acceptable_multiplier",
40 1.0,
41 "Multiplier applied to relative tolerance for acceptable limit");
42 params.addParam<int>("acceptable_iterations",

44 "Iterations after which convergence to acceptable limits is accepted");
45 params.addParam<std::vector<std::vector<NonlinearVariableName>>>(
46 "group_variables",
47 "Name of variables that are grouped together to check convergence. (Multiple groups can be "
48 "provided, separated by semicolon)");

65 Lnorm.addDocumentation(
66 "global_Linf",
67 "Compare the L-infinity norm of the residual vector to the L-infinity norm of the "
68 "absolute reference vector to determine relative convergence");
69 Lnorm.addDocumentation("local_Linf",
70 "Compute the L-infinity norm of the residual vector divided componentwise "
71 "by the absolute reference "

73 "determine relative convergence");
74

88ReferenceResidualProblem::ReferenceResidualProblem(const InputParameters & params)
89 : FEProblem(params),
90 _use_group_variables(false),
91 _reference_vector(nullptr),
92 _converge_on(getParam<std::vector<NonlinearVariableName>>("converge_on")),
93 _zero_ref_type(
94 params.get<MooseEnum>("zero_reference_residual_treatment").getEnum<ZeroReferenceType>()),
95 _reference_vector_tag_id(Moose::INVALID_TAG_ID)
96{
97 if (params.isParamValid("solution_variables"))
98 {
99 if (params.isParamValid("reference_vector"))
100 mooseDeprecated("The `solution_variables` parameter is deprecated, has no effect when "
101 "the tagging system is used, and will be removed on January 1, 2020. "
102 "Please simply delete this parameter from your input file.");
103 _soln_var_names = params.get<std::vector<NonlinearVariableName>>("solution_variables");
104 }
105
106 if (params.isParamValid("reference_residual_variables") &&
107 params.isParamValid("reference_vector"))
108 mooseError(
109 "For `ReferenceResidualProblem` you can specify either the `reference_residual_variables` "
110 "or `reference_vector` parameter, not both. `reference_residual_variables` is deprecated "
111 "so we recommend using `reference_vector`");
112
113 if (params.isParamValid("reference_residual_variables"))
114 {
115 mooseDeprecated(
116 "The save-in method for composing reference residual quantities is deprecated "
117 "and will be removed on January 1, 2020. Please use the tagging system instead; "
118 "specifically, please assign a TagName to the `reference_vector` parameter");
119
120 _ref_resid_var_names = params.get<std::vector<AuxVariableName>>("reference_residual_variables");
121
122 if (_soln_var_names.size() != _ref_resid_var_names.size())
123 mooseError("In ReferenceResidualProblem, size of solution_variables (",
124 _soln_var_names.size(),
125 ") != size of reference_residual_variables (",
126 _ref_resid_var_names.size(),
127 ")");
128 }
129 else if (params.isParamValid("reference_vector"))
130 {
131 if (numNonlinearSystems() > 1)
132 paramError(
133 "nl_sys_names",
134 "reference residual problem does not currently support multiple nonlinear systems");
135 _reference_vector_tag_id = getVectorTagID(getParam<TagName>("reference_vector"));
136 _reference_vector = &getNonlinearSystemBase(0).getVector(_reference_vector_tag_id);
137 }
138 else
139 mooseInfo("Neither the `reference_residual_variables` nor `reference_vector` parameter is "
140 "specified for `ReferenceResidualProblem`, which means that no reference "
141 "quantites are set. Because of this, the standard technique of comparing the "
142 "norm of the full residual vector with its initial value will be used.");
143
144 if (params.isParamValid("group_variables"))
145 {
146 _group_variables =
147 params.get<std::vector<std::vector<NonlinearVariableName>>>("group_variables");
148 _use_group_variables = true;
149 }
150
151 _accept_mult = params.get<Real>("acceptable_multiplier");
152 _accept_iters = params.get<int>("acceptable_iterations");
153
154 const auto norm_type_enum =
155 params.get<MooseEnum>("normalization_type").getEnum<NormalizationType>();
156 if (norm_type_enum == NormalizationType::LOCAL_L2)
157 {
158 _norm_type = DISCRETE_L2;
159 _local_norm = true;
160 }
161 else if (norm_type_enum == NormalizationType::GLOBAL_L2)
162 {
163 _norm_type = DISCRETE_L2;
164 _local_norm = false;
165 }
166 else if (norm_type_enum == NormalizationType::LOCAL_LINF)
167 {
168 _norm_type = DISCRETE_L_INF;
169 _local_norm = true;
170 }
171 else if (norm_type_enum == NormalizationType::GLOBAL_LINF)
172 {
173 _norm_type = DISCRETE_L_INF;
174 _local_norm = false;
175 }
176 else
177 mooseError("Internal error");
178
179 if (_local_norm && !params.isParamValid("reference_vector"))
180 paramError("reference_vector", "If local norm is used, a reference_vector must be provided.");
181}

183void
184ReferenceResidualProblem::initialSetup()
185{
186 NonlinearSystemBase & nonlinear_sys = getNonlinearSystemBase(/*nl_sys=*/0);
187 AuxiliarySystem & aux_sys = getAuxiliarySystem();
188 System & s = nonlinear_sys.system();
189 auto & as = aux_sys.sys();
190
191 if (_soln_var_names.empty())
192 {
193 // If the user provides reference_vector, that implies that they want the
194 // individual variables compared against their reference quantities in the
195 // tag vector. The code depends on having _soln_var_names populated,
196 // so fill that out if they didn't specify solution_variables.
197 if (_reference_vector)
198 for (unsigned int var_num = 0; var_num < s.n_vars(); var_num++)
199 _soln_var_names.push_back(s.variable_name(var_num));
200
201 // If they didn't provide reference_vector, that implies that they
202 // want to skip the individual variable comparison, so leave it alone.
203 }
204 else if (_soln_var_names.size() != s.n_vars())
205 mooseError("In ReferenceResidualProblem, size of solution_variables (",
206 _soln_var_names.size(),
207 ") != number of variables in system (",
208 s.n_vars(),
209 ")");
210
211 const auto n_soln_vars = _soln_var_names.size();
212 _variable_group_num_index.resize(n_soln_vars);
213
214 if (!_converge_on.empty())
215 {
216 _converge_on_var.assign(n_soln_vars, false);
217 for (std::size_t i = 0; i < n_soln_vars; ++i)
218 for (const auto & c : _converge_on)
219 if (MooseUtils::globCompare(_soln_var_names[i], c))
220 {
221 _converge_on_var[i] = true;
222 break;
223 }
224 }
225 else
226 _converge_on_var.assign(n_soln_vars, true);
227
228 unsigned int group_variable_num = 0;
229 if (_use_group_variables)
230 {
231 for (unsigned int i = 0; i < _group_variables.size(); ++i)
232 {
233 group_variable_num += _group_variables[i].size();
234 if (_group_variables[i].size() == 1)
235 mooseError(" In the 'group_variables' parameter, variable ",
236 _group_variables[i][0],
237 " is not grouped with other variables.");
238 }
239
240 unsigned int size = n_soln_vars - group_variable_num + _group_variables.size();
241 _group_ref_resid.resize(size);
242 _group_resid.resize(size);
243 _group_output_resid.resize(size);
244 _group_soln_var_names.resize(size);
245 _group_ref_resid_var_names.resize(size);
246 }
247 else
248 {
249 _group_ref_resid.resize(n_soln_vars);
250 _group_resid.resize(n_soln_vars);
251 _group_output_resid.resize(n_soln_vars);
252 _group_soln_var_names.resize(n_soln_vars);
253 _group_ref_resid_var_names.resize(n_soln_vars);
254 }
255
256 std::set<std::string> check_duplicate;
257 if (_use_group_variables)
258 {
259 for (unsigned int i = 0; i < _group_variables.size(); ++i)
260 for (unsigned int j = 0; j < _group_variables[i].size(); ++j)
261 check_duplicate.insert(_group_variables[i][j]);
262
263 if (check_duplicate.size() != group_variable_num)
264 mooseError(
265 "A variable cannot be included in multiple groups in the 'group_variables' parameter.");
266 }
267
268 _soln_vars.clear();
269 for (unsigned int i = 0; i < n_soln_vars; ++i)
270 {
271 bool foundMatch = false;
272 for (unsigned int var_num = 0; var_num < s.n_vars(); var_num++)
273 if (_soln_var_names[i] == s.variable_name(var_num))
274 {
275 _soln_vars.push_back(var_num);
276 foundMatch = true;
277 break;
278 }
279
280 if (!foundMatch)
281 mooseError("Could not find solution variable '", _soln_var_names[i], "' in system");
282 }
283
284 if (!_reference_vector)
285 {
286 _ref_resid_vars.clear();
287 for (unsigned int i = 0; i < _ref_resid_var_names.size(); ++i)
288 {
289 bool foundMatch = false;
290 for (unsigned int var_num = 0; var_num < as.n_vars(); var_num++)
291 if (_ref_resid_var_names[i] == as.variable_name(var_num))
292 {
293 _ref_resid_vars.push_back(var_num);
294 foundMatch = true;
295 break;
296 }
297
298 if (!foundMatch)
299 mooseError("Could not find variable '", _ref_resid_var_names[i], "' in auxiliary system");
300 }
301 }
302
303 unsigned int ungroup_index = 0;
304 if (_use_group_variables)
305 ungroup_index = _group_variables.size();
306
307 for (unsigned int i = 0; i < _soln_vars.size(); ++i)
308 {
309 bool find_group = false;
310 if (_use_group_variables)
311 {
312 for (unsigned int j = 0; j < _group_variables.size(); ++j)
313 if (std::find(_group_variables[j].begin(),
314 _group_variables[j].end(),
315 s.variable_name(_soln_vars[i])) != _group_variables[j].end())
316 {
317 if (!_converge_on_var[i])
318 paramError("converge_on",
319 "You added variable '",
320 _soln_var_names[i],
321 "' to a group but excluded it from the convergence check. This is not "
322 "permitted.");
323
324 _variable_group_num_index[i] = j;
325 find_group = true;
326 break;
327 }
328
329 if (!find_group)
330 {
331 _variable_group_num_index[i] = ungroup_index;
332 ungroup_index++;
333 }
334 }
335 else
336 _variable_group_num_index[i] = i;
337 }
338
339 if (_use_group_variables)
340 {
341 for (unsigned int i = 0; i < _group_variables.size(); ++i)
342 {
343 unsigned int num_scalar_vars = 0;
344 unsigned int num_field_vars = 0;
345 if (_group_variables[i].size() > 1)
346 {
347 for (unsigned int j = 0; j < _group_variables[i].size(); ++j)
348 for (unsigned int var_num = 0; var_num < s.n_vars(); var_num++)
349 if (_group_variables[i][j] == s.variable_name(var_num))
350 {
351 if (nonlinear_sys.isScalarVariable(_soln_vars[var_num]))
352 ++num_scalar_vars;
353 else
354 ++num_field_vars;
355 break;
356 }
357 }
358 if (num_scalar_vars > 0 && num_field_vars > 0)
359 mooseWarning("In the 'group_variables' parameter, standard variables and scalar variables "
360 "are grouped together in group ",
361 i);
362 }
363 }
364
365 for (unsigned int i = 0; i < n_soln_vars; ++i)
366 {
367 if (_group_soln_var_names[_variable_group_num_index[i]].empty())
368 {
369 _group_soln_var_names[_variable_group_num_index[i]] = _soln_var_names[i];
370 if (_use_group_variables && _variable_group_num_index[i] < _group_variables.size())
371 _group_soln_var_names[_variable_group_num_index[i]] += " (grouped) ";
372 }
373
374 if (!_reference_vector && _group_ref_resid_var_names[_variable_group_num_index[i]].empty())
375 {
376 _group_ref_resid_var_names[_variable_group_num_index[i]] = _ref_resid_var_names[i];
377 if (_use_group_variables && _variable_group_num_index[i] < _group_variables.size())
378 _group_ref_resid_var_names[_variable_group_num_index[i]] += " (grouped) ";
379 }
380 }
381
382 if (!_reference_vector)
383 {
384 const unsigned int size_solnVars = _soln_vars.size();
385 _scaling_factors.resize(size_solnVars);
386 for (unsigned int i = 0; i < size_solnVars; ++i)
387 if (nonlinear_sys.isScalarVariable(_soln_vars[i]))
388 _scaling_factors[i] = nonlinear_sys.getScalarVariable(0, _soln_vars[i]).scalingFactor();
389 else
390 _scaling_factors[i] = nonlinear_sys.getVariable(/*tid*/ 0, _soln_vars[i]).scalingFactor();
391 }
392
393 FEProblemBase::initialSetup();
394}

396void
397ReferenceResidualProblem::updateReferenceResidual()
398{
399 mooseAssert(_current_nl_sys, "This should be non-null");
400 AuxiliarySystem & aux_sys = getAuxiliarySystem();
401 System & s = _current_nl_sys->system();
402 auto & as = aux_sys.sys();
403
404 std::fill(_group_resid.begin(), _group_resid.end(), 0.0);
405 std::fill(_group_output_resid.begin(), _group_output_resid.end(), 0.0);
406 if (_local_norm)
407 std::fill(_group_ref_resid.begin(), _group_ref_resid.end(), 1.0);
408 else
409 std::fill(_group_ref_resid.begin(), _group_ref_resid.end(), 0.0);
410
411 for (unsigned int i = 0; i < _soln_vars.size(); ++i)
412 {
413 Real resid = 0.0;
414 const auto group = _variable_group_num_index[i];
415 if (_local_norm)
416 {
417 mooseAssert(_current_nl_sys->RHS().size() == (*_reference_vector).size(),
418 "Sizes of nonlinear RHS and reference vector should be the same.");
419 mooseAssert((*_reference_vector).size(), "Reference vector must be provided.");
420 // Add a tiny number to the reference to prevent a divide by zero.
421 auto ref = _reference_vector->clone();
422 ref->add(std::numeric_limits<Number>::min());
423 auto div = _current_nl_sys->RHS().clone();
424 *div /= *ref;
425 resid = Utility::pow<2>(s.calculate_norm(*div, _soln_vars[i], _norm_type));
426 }
427 else
428 {
429 resid = Utility::pow<2>(s.calculate_norm(_current_nl_sys->RHS(), _soln_vars[i], _norm_type));
430 if (_reference_vector)
431 {
432 const auto ref_resid = s.calculate_norm(*_reference_vector, _soln_vars[i], _norm_type);
433 _group_ref_resid[group] += Utility::pow<2>(ref_resid);
434 }
435 }
436
437 _group_resid[group] += _converge_on_var[i] ? resid : 0;
438 _group_output_resid[group] += resid;
439 }
440
441 if (!_reference_vector)
442 {
443 for (unsigned int i = 0; i < _ref_resid_vars.size(); ++i)
444 {
445 const auto ref_resid =
446 as.calculate_norm(*as.current_local_solution, _ref_resid_vars[i], _norm_type) *
447 _scaling_factors[i];
448 _group_ref_resid[_variable_group_num_index[i]] += Utility::pow<2>(ref_resid);
449 }
450 }
451
452 for (unsigned int i = 0; i < _group_resid.size(); ++i)
453 {
454 _group_resid[i] = std::sqrt(_group_resid[i]);
455 _group_output_resid[i] = std::sqrt(_group_output_resid[i]);
456 _group_ref_resid[i] = std::sqrt(_group_ref_resid[i]);
457 }
458}

460void
461ReferenceResidualProblem::nonlinearConvergenceSetup()
462{
463 updateReferenceResidual();
464
465 std::ostringstream out;
466
467 if (_group_soln_var_names.size() > 0)
468 {
469 out << std::setprecision(2) << std::scientific
470 << " Solution, reference convergence variable norms:\n";
471 unsigned int maxwsv = 0;
472 unsigned int maxwrv = 0;
473 for (unsigned int i = 0; i < _group_soln_var_names.size(); ++i)
474 {
475 if (_group_soln_var_names[i].size() > maxwsv)
476 maxwsv = _group_soln_var_names[i].size();
477 if (!_reference_vector && _group_ref_resid_var_names[i].size() > maxwrv)
478 maxwrv = _group_ref_resid_var_names[i].size();
479 }
480 if (_reference_vector)
481 // maxwrv is the width of maxwsv plus the length of "_ref" (e.g. 4)
482 maxwrv = maxwsv + 4;
483
484 for (unsigned int i = 0; i < _group_soln_var_names.size(); ++i)
485 {
486 out << " " << std::setw(maxwsv + (_local_norm ? 5 : 2)) << std::left
487 << (_local_norm ? "norm " : "") + _group_soln_var_names[i] + ": ";
488
489 if (_group_output_resid[i] == _group_resid[i])
490 out << std::setw(8) << _group_output_resid[i];
491 else
492 out << std::setw(8) << _group_resid[i] << " (" << _group_output_resid[i] << ')';
493
494 if (!_local_norm)
495 {
496 const auto ref_var_name =
497 _reference_vector ? _group_soln_var_names[i] + "_ref" : _group_ref_resid_var_names[i];
498 out << " " << std::setw(maxwrv + 2) << ref_var_name + ":" << std::setw(8)
499 << _group_ref_resid[i] << " (" << std::setw(8)
500 << (_group_ref_resid[i] ? _group_resid[i] / _group_ref_resid[i] : _group_resid[i])
501 << ")";
502 }
503 out << '\n';
504 }
505
506 _console << out.str() << std::flush;
507 }

510bool
511ReferenceResidualProblem::checkRelativeConvergence(const PetscInt it,
512 const Real fnorm,
513 const Real the_residual,
514 const Real rtol,
515 const Real abstol,
516 std::ostringstream & oss)
517{
518 if (checkConvergenceIndividVars(fnorm, abstol, rtol, the_residual))
519 {
520 oss << "Converged due to function norm " << fnorm << " < relative tolerance (" << rtol
521 << ") or absolute tolerance (" << abstol << ") for all solution variables\n";
522 return true;
523 }
524 else if (it >= _accept_iters &&
525 checkConvergenceIndividVars(
526 fnorm, abstol * _accept_mult, rtol * _accept_mult, the_residual))
527 {
528 oss << "Converged due to function norm " << fnorm << " < acceptable relative tolerance ("
529 << rtol * _accept_mult << ") or acceptable absolute tolerance (" << abstol * _accept_mult
530 << ") for all solution variables\n";
531 _console << "Converged due to ACCEPTABLE tolerances" << std::endl;
532 return true;
533 }
534
535 return false;
536}

538bool
539ReferenceResidualProblem::checkConvergenceIndividVars(const Real fnorm,
540 const Real abstol,
541 const Real rtol,
542 const Real initial_residual_before_preset_bcs)
543{
544 // Convergence is checked via:
545 // 1) if group residual is less than group reference residual by relative tolerance
546 // 2) if group residual is less than absolute tolerance
547 // 3) if group reference residual is zero and:
548 // 3.1) Convergence type is ZERO_TOLERANCE and group residual is zero (rare, but possible, and
549 // historically implemented way)
550 // 3.2) Convergence type is RELATIVE_TOLERANCE and group residual
551 // is less than relative tolerance. (i.e., using the relative tolerance to check group
552 // convergence in an absolute way)
553
554 bool convergedRelative = true;
555 if (_group_resid.size() > 0)
556 {
557 for (unsigned int i = 0; i < _group_resid.size(); ++i)
558 convergedRelative &=
559 (_group_resid[i] < _group_ref_resid[i] * rtol || _group_resid[i] < abstol ||
560 (_group_ref_resid[i] == 0.0 &&
561 ((_zero_ref_type == ZeroReferenceType::ZERO_TOLERANCE && _group_resid[i] == 0.0) ||
562 (_zero_ref_type == ZeroReferenceType::RELATIVE_TOLERANCE &&
563 _group_resid[i] <= rtol))));
564 }
565
566 else if (fnorm > initial_residual_before_preset_bcs * rtol)
567 convergedRelative = false;
568
569 return convergedRelative;
570}
