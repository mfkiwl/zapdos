55std::unique_ptr<MeshBase>
56MeshRepairGenerator::generate()
57{
58 std::unique_ptr<MeshBase> mesh = std::move(_input);
59 mesh->prepare_for_use();
60
61 // Blanket ban on distributed. This can be relaxed for some operations if needed
62 if (!mesh->is_serial())
63 mooseError("MeshRepairGenerator requires a serial mesh. The mesh should not be distributed.");
64
65 if (_fix_overlapping_nodes)
66 fixOverlappingNodes(mesh);
67
68 // Flip orientation of elements to keep positive volumes
69 if (_fix_element_orientation)
70 MeshTools::Modification::orient_elements(*mesh);
71
72 // Disambiguate any block that has elements of multiple types
73 if (_elem_type_separation)
74 separateSubdomainsByElementType(mesh);
75
76 mesh->set_isnt_prepared();
77 return mesh;
78}

80void
81MeshRepairGenerator::fixOverlappingNodes(std::unique_ptr<MeshBase> & mesh) const
82{
83 unsigned int num_fixed_nodes = 0;
84 auto pl = mesh->sub_point_locator();
85 pl->set_close_to_point_tol(_node_overlap_tol);
86
87 std::unordered_set<dof_id_type> nodes_removed;
88 // loop on nodes
89 for (auto & node : mesh->node_ptr_range())
90 {
91 // this node has already been removed
92 if (nodes_removed.count(node->id()))
93 continue;
94
95 // find all the elements around this node
96 std::set<const Elem *> elements;
97 (*pl)(*node, elements);
98
99 for (auto & elem : elements)
100 {
101 bool found = false;
102 for (auto & elem_node : elem->node_ref_range())
103 {
104 if (node->id() == elem_node.id())
105 {
106 found = true;
107 break;
108 }
109 }
110 if (!found)
111 {
112 for (auto & elem_node : elem->node_ref_range())
113 {
114 if (elem_node.id() == node->id())
115 continue;
116 const Real tol = _node_overlap_tol;
117 // Compares the coordinates
118 const auto x_node = (*node)(0);
119 const auto x_elem_node = elem_node(0);
120 const auto y_node = (*node)(1);
121 const auto y_elem_node = elem_node(1);
122 const auto z_node = (*node)(2);
123 const auto z_elem_node = elem_node(2);
124
125 if (MooseUtils::absoluteFuzzyEqual(x_node, x_elem_node, tol) &&
126 MooseUtils::absoluteFuzzyEqual(y_node, y_elem_node, tol) &&
127 MooseUtils::absoluteFuzzyEqual(z_node, z_elem_node, tol))
128 {
129 // Merging two nodes from the same element is almost never a good idea
130 if (elem->get_node_index(node) != libMesh::invalid_uint)
131 {
132 _console << "Two overlapping nodes in element " << elem->id() << " right by "
133 << elem->vertex_average() << ".\n They will not be stitched" << std::endl;
134 continue;
135 }
136
137 // Coordinates are the same but it's not the same node
138 // Replace the node in the element
139 const_cast<Elem *>(elem)->set_node(elem->get_node_index(&elem_node)) = node;
140 nodes_removed.insert(elem_node.id());
141
142 num_fixed_nodes++;
143 if (num_fixed_nodes < 10)
144 _console << "Stitching nodes " << *node << " and " << elem_node
145 << std::endl;
146 else if (num_fixed_nodes == 10)
147 _console << "Node stitching will now proceed silently." << std::endl;
148 }
149 }
150 }
151 }
152 }
153 _console << "Number of overlapping nodes which got merged: " << num_fixed_nodes << std::endl;
154 if (mesh->allow_renumbering())
155 mesh->renumber_nodes_and_elements();
156 else
157 {
158 mesh->remove_orphaned_nodes();
159 mesh->update_parallel_id_counts();
160 }
161}

163void
164MeshRepairGenerator::separateSubdomainsByElementType(std::unique_ptr<MeshBase> & mesh) const
165{
166 std::set<subdomain_id_type> ids;
167 mesh->subdomain_ids(ids);
168 // loop on sub-domain
169 for (const auto id : ids)
170 {
171 // Gather all the element types and blocks
172 // ElemType defines an enum for geometric element types
173 std::set<ElemType> types;
174 // loop on elements within this sub-domain
175 for (auto & elem : mesh->active_subdomain_elements_ptr_range(id))
176 types.insert(elem->type());
177
178 // This call must be performed on all processes
179 auto next_block_id = MooseMeshUtils::getNextFreeSubdomainID(*mesh);
180
181 if (types.size() > 1)
182 {
183 subdomain_id_type i = 0;
184 for (const auto type : types)
185 {
186 auto new_id = next_block_id + i++;
187 // Create blocks when a block has multiple element types
188 mesh->subdomain_name(new_id) = mesh->subdomain_name(id) + "_" + Moose::stringify(type);
189
190 // Re-assign elements to the new blocks
191 for (auto elem : mesh->active_subdomain_elements_ptr_range(id))
192 if (elem->type() == type)
193 elem->subdomain_id() = new_id;
194 }
195 }
196 }
