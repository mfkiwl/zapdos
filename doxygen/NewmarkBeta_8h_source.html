57template <typename T, typename T2, typename T3, typename T4, typename T5>
58void
59NewmarkBeta::computeTimeDerivativeHelper(
60 T & u_dot, const T2 & u_old, const T3 & u_dot_old, T4 & u_dotdot, const T5 & u_dotdot_old) const
61{
62 // compute second derivative
63 // according to Newmark-Beta method
64 // u_dotdot = first_term - second_term - third_term
65 // first_term = (u - u_old) / beta / dt ^ 2
66 // second_term = u_dot_old / beta / dt
67 // third_term = u_dotdot_old * (1 / 2 / beta - 1)
68 u_dotdot -= u_old;
69 u_dotdot *= 1.0 / _beta / _dt / _dt;
70 MathUtils::addScaled(-1.0 / _beta / _dt, u_dot_old, u_dotdot);
71 MathUtils::addScaled(-0.5 / _beta + 1.0, u_dotdot_old, u_dotdot);
72
73 // compute first derivative
74 // according to Newmark-Beta method
75 // u_dot = first_term + second_term + third_term
76 // first_term = u_dot_old
77 // second_term = u_dotdot_old * (1 - gamma) * dt
78 // third_term = u_dotdot * gamma * dt
79 u_dot = u_dot_old;
80 MathUtils::addScaled((1.0 - _gamma) * _dt, u_dotdot_old, u_dot);
81 MathUtils::addScaled(_gamma * _dt, u_dotdot, u_dot);
82}
