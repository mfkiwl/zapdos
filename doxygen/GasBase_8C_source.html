153 else if (_potential_units.compare("kV") == 0)

155
156 std::vector<Real> actual_mean_energy;
157 std::vector<Real> alpha;
158 std::vector<Real> alphaEx;
159 std::vector<Real> alphaEl;
160 std::vector<Real> mu;
161 std::vector<Real> diff;
162
163 std::string file_name = getParam<FileName>("property_tables_file");
164 MooseUtils::checkFileReadable(file_name);
165 const char * charPath = file_name.c_str();
166 std::ifstream myfile(charPath);
167 Real value;
168
169 if (myfile.is_open())
170 {
171 while (myfile >> value)
172 {
173 actual_mean_energy.push_back(value);
174 myfile >> value;
175 alpha.push_back(value);
176 myfile >> value;
177 alphaEx.push_back(value);
178 myfile >> value;
179 alphaEl.push_back(value);
180 myfile >> value;
181 mu.push_back(value);
182 myfile >> value;
183 diff.push_back(value);
184 }
185 myfile.close();
186 }
187
188 else
189 mooseError("Unable to open file");
190
191 _alpha_interpolation.setData(actual_mean_energy, alpha);
192 _alphaEx_interpolation.setData(actual_mean_energy, alphaEx);
193 _alphaEl_interpolation.setData(actual_mean_energy, alphaEl);
194 _mu_interpolation.setData(actual_mean_energy, mu);
195 _diff_interpolation.setData(actual_mean_energy, diff);
196}

198void
199GasBase::computeQpProperties()
200{
201 _massem[_qp] = 9.11e-31;
202 _massGas[_qp] = 40.0 * 1.66e-27;
203 _T_gas[_qp] = _user_T_gas;
204 _p_gas[_qp] = _user_p_gas;
205 _k_boltz[_qp] = 1.38e-23;
206 _Avogadro[_qp] = 6.0221409E23;
207 if (_use_moles)
208 _n_gas[_qp] = _p_gas[_qp] / (8.3145 * _T_gas[_qp]);
209 else
210 _n_gas[_qp] = _p_gas[_qp] / (_k_boltz[_qp] * _T_gas[_qp]);
211 _se_coeff[_qp] = _user_se_coeff;
212 _work_function[_qp] = _user_work_function;
213 _field_enhancement[_qp] = _user_field_enhancement;
214
215 _Richardson_coefficient[_qp] = _user_Richardson_coefficient;
216 _cathode_temperature[_qp] = _user_cathode_temperature;
217
218 _se_energy[_qp] = 2. * 3. / 2.; // Emi uses 2 Volts coming off the wall (presumably for Te).
219 // Multiply by 3/2 to get mean_en
220 _e[_qp] = 1.6e-19;
221 _eps[_qp] = 8.85e-12;
222 _sgnem[_qp] = -1.;
223 _sgnmean_en[_qp] = -1.;
224 _diffpotential[_qp] = _eps[_qp];
225
226 // With the exception of temperature/energy (generally in eV), all properties are in standard SI
227 // units unless otherwise indicated
228
229 if (_interp_trans_coeffs)
230 {
231 if (_ramp_trans_coeffs)
232 {
233 _muem[_qp].value() =
234 (std::tanh(_t / 1e-6) *
235 _mu_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) +
236 (1. - std::tanh(_t / 1e-6)) * .0352) *
237 _voltage_scaling;
238 _muem[_qp].derivatives() =
239 std::tanh(_t / 1e-6) *
240 _mu_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
241 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
242 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _voltage_scaling;
243 _diffem[_qp].value() =
244 std::tanh(_t / 1e-6) *
245 _diff_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) +
246 (1. - std::tanh(_t / 1e-6)) * .30;
247 _diffem[_qp].derivatives() =
248 std::tanh(_t / 1e-6) *
249 _diff_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
250 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
251 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
252 }
253 else
254 {
255 _muem[_qp].value() =
256 _mu_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
257 _voltage_scaling;
258 _muem[_qp].derivatives() =
259 _mu_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
260 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
261 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _voltage_scaling;
262 _diffem[_qp].value() =
263 _diff_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value()));
264 _diffem[_qp].derivatives() =
265 _diff_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
266 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
267 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
268 }
269 }
270 else
271 {
272 // From bolos at atmospheric pressure and an EField of 2e5 V/m
273 _muem[_qp].value() =
274 0.0352103411399 * _voltage_scaling; // units of m^2/(kV*s) if _voltage_scaling = 1000
275 _muem[_qp].derivatives() = 0.0;
276 _diffem[_qp].value() = 0.297951680159;
277 _diffem[_qp].derivatives() = 0.0;
278 }
279
280 // From curve fitting with bolos
281 _iz_coeff_efield_a[_qp] = 1.43171672e-1;
282 _iz_coeff_efield_b[_qp] = 9.05925536e-1;
283 _iz_coeff_efield_c[_qp] = 3.04958892e+6;
284
285 _iz_coeff_energy_a[_qp] = 1.52165930e+8;
286 _iz_coeff_energy_b[_qp] = -2.87277596e-1;
287 _iz_coeff_energy_c[_qp] = 5.51972192e+1;
288
289 _actual_mean_energy[_qp] = std::exp(_mean_en[_qp] - _em[_qp]);
290 _alpha_iz[_qp].value() =
291 _alpha_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value()));
292 _alpha_iz[_qp].derivatives() =
293 _alpha_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
294 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
295 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
296 _alpha_ex[_qp].value() =
297 _alphaEx_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value()));
298 _alpha_ex[_qp].derivatives() =
299 _alphaEx_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
300 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
301 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
302 if (_interp_elastic_coeff)
303 {
304 _alpha_el[_qp].value() =
305 _alphaEl_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value()));
306 _alpha_el[_qp].derivatives() = _alphaEl_interpolation.sampleDerivative(
307 std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
308 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
309 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
310 }
311 else
312 {
313 _alpha_el[_qp].value() = 5e8;
314 _alpha_el[_qp].derivatives() = 0.0;
315 }
316
317 _el_coeff_energy_a[_qp] = 1.60638169e-13;
318 _el_coeff_energy_b[_qp] = 3.17917979e-1;
319 _el_coeff_energy_c[_qp] = 4.66301096;
320
321 _N_A[_qp] = 6.02e23;
322 _Eiz[_qp] = 15.76;
323 _Eex[_qp] = 11.5;
324
325 // From Hagelaar: The below approximations can be derived assumption Maxwell EEDF, const
326 // momentum-transfer frequency, and constant kinetic pressure.
327 _mumean_en[_qp].value() = 5.0 / 3.0 * _muem[_qp].value();
328 _diffmean_en[_qp].value() = 5.0 / 3.0 * _diffem[_qp].value();
329 if (_interp_trans_coeffs)
330 {
331 _mumean_en[_qp].derivatives() = 5.0 / 3.0 * _muem[_qp].derivatives();
332 _diffmean_en[_qp].derivatives() = 5.0 / 3.0 * _diffem[_qp].derivatives();
333 }
334 else
335 {
336 _mumean_en[_qp].derivatives() = 0.0;
337 _diffmean_en[_qp].derivatives() = 0.0;
338 }
339
340 _rate_coeff_elastic[_qp] = 1e-13;
341
342 _TemVolts[_qp] = 2. / 3. * std::exp(_mean_en[_qp] - _em[_qp]);
343 _Tem[_qp] = _e[_qp] * _TemVolts[_qp] / _k_boltz[_qp];
344
345 _kiz[_qp] = 2.34e-14 * std::pow(_TemVolts[_qp], .59) * std::exp(-17.44 / _TemVolts[_qp]);
346 _kex[_qp] = 2.48e-14 * std::pow(_TemVolts[_qp], .33) * std::exp(-12.78 / _TemVolts[_qp]);
347 _kArp[_qp] = 2.48e-14 * std::pow(_TemVolts[_qp], .33) * std::exp(-12.78 / _TemVolts[_qp]);
348
349 _kel[_qp].value() = 1e-13; // Approximate elastic rate coefficient
350 _kel[_qp].derivatives() = 0.;
351 if (_use_moles)
352 {
353 _kiz[_qp] = _kiz[_qp] * _N_A[_qp];
354 _kex[_qp] = _kex[_qp] * _N_A[_qp];
355 _kArp[_qp] = _kArp[_qp] * _N_A[_qp];
356 _kel[_qp] = _kel[_qp] * _N_A[_qp];
357 }
358}
