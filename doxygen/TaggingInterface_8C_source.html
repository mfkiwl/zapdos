19TaggingInterface::validParams()
20{
21
22 InputParameters params = emptyInputParameters();
23
24 // These are the default names for tags, but users will be able to add their own
25 MultiMooseEnum vtags("nontime time", "nontime", true);
26 MultiMooseEnum mtags("nontime system", "system", true);
27
28 params.addParam<MultiMooseEnum>(
29 "vector_tags", vtags, "The tag for the vectors this Kernel should fill");
30
31 params.addParam<MultiMooseEnum>(
32 "matrix_tags", mtags, "The tag for the matrices this Kernel should fill");
33
34 params.addParam<std::vector<TagName>>("extra_vector_tags",
35 "The extra tags for the vectors this Kernel should fill");
36
37 params.addParam<std::vector<TagName>>(
38 "absolute_value_vector_tags",
39 "The tags for the vectors this residual object should fill with the "
40 "absolute value of the residual contribution");
41
42 params.addParam<std::vector<TagName>>("extra_matrix_tags",
43 "The extra tags for the matrices this Kernel should fill");
44
45 params.addParamNamesToGroup(
46 "vector_tags matrix_tags extra_vector_tags extra_matrix_tags absolute_value_vector_tags",
47 "Tagging");
48

52TaggingInterface::TaggingInterface(const MooseObject * moose_object)
53 : _subproblem(*moose_object->parameters().getCheckedPointerParam<SubProblem *>("_subproblem")),
54 _moose_object(*moose_object),
55 _tag_params(_moose_object.parameters())
56{
57 auto & vector_tag_names = _tag_params.get<MultiMooseEnum>("vector_tags");
58
59 if (!vector_tag_names.isValid())
60 mooseError("MUST provide at least one vector_tag for Kernel: ", _moose_object.name());
61
62 for (auto & vector_tag_name : vector_tag_names)
63 {
64 const TagID vector_tag_id = _subproblem.getVectorTagID(vector_tag_name.name());
65 if (_subproblem.vectorTagType(vector_tag_id) != Moose::VECTOR_TAG_RESIDUAL)
66 mooseError("Vector tag '",
67 vector_tag_name.name(),
68 "' for Kernel '",
69 _moose_object.name(),
70 "' is not a residual vector tag");
71 _vector_tags.insert(vector_tag_id);
72 }
73
74 // Add extra vector tags. These tags should be created in the System already, otherwise
75 // we can not add the extra tags
76 auto & extra_vector_tags = _tag_params.get<std::vector<TagName>>("extra_vector_tags");
77
78 for (auto & vector_tag_name : extra_vector_tags)
79 {
80 const TagID vector_tag_id = _subproblem.getVectorTagID(vector_tag_name);
81 if (_subproblem.vectorTagType(vector_tag_id) != Moose::VECTOR_TAG_RESIDUAL)
82 mooseError("Extra vector tag '",
83 vector_tag_name,
84 "' for Kernel '",
85 _moose_object.name(),
86 "' is not a residual vector tag");
87 _vector_tags.insert(vector_tag_id);
88 }
89
90 // Add absolue value vector tags. These tags should be created in the System already, otherwise
91 // we can not add the extra tags
92 auto & abs_vector_tags = _tag_params.get<std::vector<TagName>>("absolute_value_vector_tags");
93
94 for (auto & vector_tag_name : abs_vector_tags)
95 {
96 const TagID vector_tag_id = _subproblem.getVectorTagID(vector_tag_name);
97 if (_subproblem.vectorTagType(vector_tag_id) != Moose::VECTOR_TAG_RESIDUAL)
98 mooseError("Absolute value vector tag '",
99 vector_tag_name,
100 "' for Kernel '",
101 _moose_object.name(),
102 "' is not a residual vector tag");
103 _abs_vector_tags.insert(vector_tag_id);
104 }
105
106 auto & matrix_tag_names = _tag_params.get<MultiMooseEnum>("matrix_tags");
107
108 if (!matrix_tag_names.isValid())
109 mooseError("MUST provide at least one matrix_tag for Kernel: ", _moose_object.name());
110
111 for (auto & matrix_tag_name : matrix_tag_names)
112 _matrix_tags.insert(_subproblem.getMatrixTagID(matrix_tag_name.name()));
113
114 auto & extra_matrix_tags = _tag_params.get<std::vector<TagName>>("extra_matrix_tags");
115
116 for (auto & matrix_tag_name : extra_matrix_tags)
117 _matrix_tags.insert(_subproblem.getMatrixTagID(matrix_tag_name));
118
119 _re_blocks.resize(_vector_tags.size());
120 _absre_blocks.resize(_abs_vector_tags.size());
121 _ke_blocks.resize(_matrix_tags.size());
122
123 const auto * const fe_problem =
124 moose_object->parameters().getCheckedPointerParam<FEProblemBase *>("_fe_problem_base");
125 if (const auto * const ref_problem = dynamic_cast<const ReferenceResidualProblem *>(fe_problem))
126 {
127 const auto reference_tag = ref_problem->referenceVectorTagID({});
128 auto create_tags_split =
129 [reference_tag](const auto & tags, auto & non_ref_tags, auto & ref_tags)
130 {
131 for (const auto tag : tags)
132 if (tag == reference_tag)
133 ref_tags.insert(tag);
134 else
135 non_ref_tags.insert(tag);
136 };
137 create_tags_split(_vector_tags, _non_ref_vector_tags, _ref_vector_tags);
138 create_tags_split(_abs_vector_tags, _non_ref_abs_vector_tags, _ref_abs_vector_tags);
139 }
140 else
141 {
142 _non_ref_vector_tags = _vector_tags;
143 _non_ref_abs_vector_tags = _abs_vector_tags;
144 }
145}

200void
201TaggingInterface::prepareVectorTagInternal(Assembly & assembly,
202 const unsigned int ivar,
203 const std::set<TagID> & vector_tags,
204 const std::set<TagID> & absolute_value_vector_tags)
205{
206 auto prepare = [this, ivar, &assembly](auto & re_blocks, const auto & tags)
207 {
208 re_blocks.clear();
209 re_blocks.reserve(tags.size());
210 for (const auto tag_id : tags)
211 {
212 const auto & tag = _subproblem.getVectorTag(tag_id);
213 re_blocks.push_back(&assembly.residualBlock(ivar, Assembly::LocalDataKey{}, tag._type_id));
214 }
215 };
216
217 prepare(_re_blocks, vector_tags);
218 prepare(_absre_blocks, absolute_value_vector_tags);
219
220 _local_re.resize(_re_blocks.empty()
221 ? (_absre_blocks.empty() ? std::size_t(0) : _absre_blocks[0]->size())
222 : _re_blocks[0]->size());
223}

225void
226TaggingInterface::prepareVectorTagNeighbor(Assembly & assembly, unsigned int ivar)
227{
228 _re_blocks.resize(_vector_tags.size());
229 mooseAssert(_vector_tags.size() >= 1, "we need at least one active tag");
230 auto vector_tag = _vector_tags.begin();
231 for (MooseIndex(_vector_tags) i = 0; i < _vector_tags.size(); i++, ++vector_tag)
232 {
233 const VectorTag & tag = _subproblem.getVectorTag(*vector_tag);
234 _re_blocks[i] = &assembly.residualBlockNeighbor(ivar, Assembly::LocalDataKey{}, tag._type_id);
235 }
236 _local_re.resize(_re_blocks[0]->size());
237
238 _absre_blocks.resize(_abs_vector_tags.size());
239 vector_tag = _abs_vector_tags.begin();
240 for (MooseIndex(_abs_vector_tags) i = 0; i < _abs_vector_tags.size(); i++, ++vector_tag)
241 {
242 const VectorTag & tag = _subproblem.getVectorTag(*vector_tag);
243 _absre_blocks[i] =
244 &assembly.residualBlockNeighbor(ivar, Assembly::LocalDataKey{}, tag._type_id);
245 }
246}

354void
355TaggingInterface::accumulateTaggedLocalResidual()
356{
357 for (auto & re : _re_blocks)
358 *re += _local_re;
359 for (auto & absre : _absre_blocks)
360 for (const auto i : index_range(_local_re))
361 (*absre)(i) += std::abs(_local_re(i));
362}

382TaggingInterface::accumulateTaggedLocalMatrix(Assembly & assembly,

386{

392 mooseAssert(_ke_blocks[0]->m() == k.m() && _ke_blocks[0]->n() == k.n(),
393 "Passed-in k must match the blocks we are about to sum into");
394 for (auto & ke : _ke_blocks)
395 *ke += k;
