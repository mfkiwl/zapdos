52EEDFElasticLog::computeQpResidual()
53{
54 Real Eel = -3.0 * _massem / _massTarget[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]);
55
56 return -_test[_i][_qp] * _reaction_coeff[_qp] * std::exp(_em[_qp] + _target[_qp]) * Eel;
57}

59Real
60EEDFElasticLog::computeQpJacobian()
61{
62 Real d_actual_mean_en_d_mean_en = std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
63 Real d_k_d_mean_en = _d_k_d_actual_mean_en[_qp] * d_actual_mean_en_d_mean_en;
64
65 Real Eel = -3.0 * _massem / _massTarget[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]);
66 Real d_Eel_d_mean_en = Eel * _phi[_j][_qp];
67
68 return -_test[_i][_qp] * std::exp(_em[_qp] + _target[_qp]) *
69 (d_k_d_mean_en * Eel + _reaction_coeff[_qp] * d_Eel_d_mean_en);
70}

72Real
73EEDFElasticLog::computeQpOffDiagJacobian(unsigned int jvar)
74{
75 Real d_actual_mean_en_d_em = -std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
76 Real d_k_d_em = _d_k_d_actual_mean_en[_qp] * d_actual_mean_en_d_em;
77
78 Real Eel = -3.0 * _massem / _massTarget[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]);
79 Real d_Eel_d_em =
80 -3.0 * _massem / _massTarget[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]) * -_phi[_j][_qp];
81
82 if (jvar == _em_id)
83 return -_test[_i][_qp] *
84 (d_k_d_em * std::exp(_em[_qp] + _target[_qp]) * Eel +
85 _reaction_coeff[_qp] * std::exp(_em[_qp] + _target[_qp]) * _phi[_j][_qp] * Eel +
86 _reaction_coeff[_qp] * std::exp(_em[_qp] + _target[_qp]) * d_Eel_d_em);
87 else if (jvar == _target_id)
88 return -_test[_i][_qp] * _reaction_coeff[_qp] * Eel * std::exp(_em[_qp] + _target[_qp]) *
89 _phi[_j][_qp];
90 else
91 return 0.0;
92}
