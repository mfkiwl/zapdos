29void
30CrankNicolson::computeTimeDerivatives()
31{
32 if (!_sys.solutionUDot())
33 mooseError("CrankNicolson: Time derivative of solution (`u_dot`) is not stored. Please set "
34 "uDotRequested() to true in FEProblemBase befor requesting `u_dot`.");
35
36 NumericVector<Number> & u_dot = *_sys.solutionUDot();
37 u_dot = *_solution;
38 computeTimeDerivativeHelper(u_dot, _solution_old);
39 u_dot.close();
40
41 _du_dot_du = 2. / _dt;
42}

52void
53CrankNicolson::init()
54{
55 if (!_sys.solutionUDot())
56 mooseError("CrankNicolson: Time derivative of solution (`u_dot`) is not stored. Please set "
57 "uDotRequested() to true in FEProblemBase befor requesting `u_dot`.");
58
59 // time derivative is assumed to be zero on initial
60 NumericVector<Number> & u_dot = *_sys.solutionUDot();
61 u_dot.zero();
62 _du_dot_du = 0;
63
64 // compute residual for the initial time step
65 // Note: we can not directly pass _residual_old in computeResidualTag because
66 // the function will call postResidual, which will cause _residual_old
67 // to be added on top of itself prohibited by PETSc.
68 // Objects executed on initial have been executed by FEProblem,
69 // so we can and should directly call NonlinearSystem residual evaluation.
70 _fe_problem.setCurrentResidualVectorTags({_nl.nonTimeVectorTag()});
71 _nl.computeResidualTag(_nl.RHS(), _nl.nonTimeVectorTag());
72 _fe_problem.clearCurrentResidualVectorTags();
73 _residual_old = _nl.RHS();
74}

76void
77CrankNicolson::postResidual(NumericVector<Number> & residual)
78{
79 // PETSc 3.19 insists on having closed vectors when doing VecAXPY,
80 // and that's probably a good idea with earlier versions too, but
81 // we don't always get here with _Re_time closed.
82 std::vector<unsigned char> inputs_closed = {
83 _Re_time.closed(), _Re_non_time.closed(), _residual_old.closed()};
84
85 // We might have done work on one processor but not all processors,
86 // so we have to sync our closed() checks. Congrats to the BISON
87 // folks for test coverage that caught that.
88 comm().min(inputs_closed);
89
90 if (!inputs_closed[0])
91 _Re_time.close();
92 if (!inputs_closed[1])
93 _Re_non_time.close();
94 if (!inputs_closed[2])
95 _residual_old.close();
96 residual += _Re_time;
97 residual += _Re_non_time;
98 residual += _residual_old;
99}
