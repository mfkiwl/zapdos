51Real
52SideIntegralPostprocessor::computeIntegral()
53{
54 Real sum = 0;
55 if (_qp_integration)
56 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
57 sum += _JxW[_qp] * _coord[_qp] * computeQpIntegral();
58 else
59 {
60 // Finite volume functors integration is over FaceInfo, not quadrature points
61 getFaceInfos();
62
63 for (auto & fi : _face_infos)
64 sum += fi->faceArea() * fi->faceCoord() * computeFaceInfoIntegral(fi);
65 }
66 return sum;
67}
