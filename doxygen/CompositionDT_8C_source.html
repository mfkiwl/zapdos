47CompositionDT::CompositionDT(const InputParameters & parameters)
48 : TimeStepper(parameters),
49 _has_initial_dt(isParamValid("initial_dt")),
50 _initial_dt(_has_initial_dt ? getParam<Real>("initial_dt") : 0.),
51 _lower_bound(getParam<std::vector<std::string>>("lower_bound").begin(),
52 getParam<std::vector<std::string>>("lower_bound").end()),
53 _current_time_stepper(nullptr),
54 _largest_bound_time_stepper(nullptr),
55 _closest_time_sequence_stepper(nullptr)
56{
57 // Make sure the steppers in "lower_bound" exist
58 const auto time_steppers = getTimeSteppers();
59 for (const auto & time_stepper_name : _lower_bound)
60 if (std::find_if(time_steppers.begin(),
61 time_steppers.end(),
62 [&time_stepper_name](const auto & ts)
63 { return ts->name() == time_stepper_name; }) == time_steppers.end() &&
64 _lower_bound.size() != 0)
65 paramError(
66 "lower_bound", "Failed to find a timestepper with the name '", time_stepper_name, "'");
67}

136Real
137CompositionDT::computeDT()
138{
139 const auto time_steppers = getTimeSteppers();
140 // Note : compositionDT requires other active time steppers as input so no active time steppers
141 // or only compositionDT is active is not allowed
142 if (time_steppers.size() < 1)
143 mooseError("No TimeStepper(s) are currently active to compute a timestep");
144
145 std::set<std::pair<Real, TimeStepper *>, CompareFirst> dts, bound_dt;
146
147 for (auto & ts : time_steppers)
148 if (!dynamic_cast<TimeSequenceStepperBase *>(ts))
149 {
150 ts->computeStep();
151 const auto dt = ts->getCurrentDT();
152
153 if (_lower_bound.count(ts->name()))
154 bound_dt.emplace(dt, ts);
155 else
156 dts.emplace(dt, ts);
157 }
158
159 _current_time_stepper = dts.size() ? dts.begin()->second : nullptr;
160 _largest_bound_time_stepper = bound_dt.size() ? (--bound_dt.end())->second : nullptr;
161
162 _dt = produceCompositionDT(dts, bound_dt);
163
164 return _dt;
165}

167Real
168CompositionDT::getSequenceSteppersNextTime()
169{
170 const auto time_steppers = getTimeSteppers();
171
172 std::vector<TimeSequenceStepperBase *> time_sequence_steppers;
173 for (auto & ts : time_steppers)
174 if (auto tss = dynamic_cast<TimeSequenceStepperBase *>(ts))
175 time_sequence_steppers.push_back(tss);
176
177 if (time_sequence_steppers.empty())
178 return 0;
179
180 Real next_time_to_hit = std::numeric_limits<Real>::max();
181 for (auto & tss : time_sequence_steppers)
182 {
183 Real ts_time_to_hit = tss->getNextTimeInSequence();
184 if (ts_time_to_hit - _time <= _dt_min)
185 {
186 tss->increaseCurrentStep();
187 ts_time_to_hit = tss->getNextTimeInSequence();
188 }
189 if (next_time_to_hit > ts_time_to_hit)
190 {
191 _closest_time_sequence_stepper = tss;
192 next_time_to_hit = ts_time_to_hit;
193 }
194 }
195 return next_time_to_hit;
196}

198Real
199CompositionDT::produceCompositionDT(
200 std::set<std::pair<Real, TimeStepper *>, CompareFirst> & dts,
201 std::set<std::pair<Real, TimeStepper *>, CompareFirst> & bound_dts)
202{
203 Real minDT, lower_bound, dt;
204 minDT = lower_bound = dt = 0.0;
205 if (!dts.empty())
206 minDT = dts.begin()->first;
207 if (!bound_dts.empty())
208 lower_bound = bound_dts.rbegin()->first;
209
210 if (minDT > lower_bound)
211 dt = minDT;
212 else
213 {
214 dt = lower_bound;
215 _current_time_stepper = _largest_bound_time_stepper;
216 }
217
218 auto ts = getSequenceSteppersNextTime();
219
220 if (ts != 0 && (ts - _time) < dt)
221 {
222 _current_time_stepper = _closest_time_sequence_stepper;
223 return std::min((ts - _time), dt);
224 }
225 else
226 return dt;
227}

229std::vector<TimeStepper *>
230CompositionDT::getTimeSteppers()
231{
232 std::vector<TimeStepper *> time_steppers;
233 _fe_problem.theWarehouse()
234 .query()
235 .condition<AttribSystem>("TimeStepper")
236 .queryInto(time_steppers);
237
238 // Remove CompositionDT from time_steppers vector to avoid recursive call
239 time_steppers.erase(std::remove(time_steppers.begin(), time_steppers.end(), this),
240 time_steppers.end());
241 return time_steppers;
242}
