44RateCoefficientProvider::RateCoefficientProvider(const InputParameters & parameters)
45 : GeneralUserObject(parameters),
46 _sampling_format(getParam<std::string>("sampling_format")),
47 _rate_format(getParam<std::string>("rate_format")),
48 _reduced_field_value(coupledScalarValue("reduced_field"))
49// _function(getFunction("function")),
50// _point(getParam<Point>("point"))
51// _reaction_rate(declareProperty<Real>("k_"+getParam<std::string>("reaction"))),
52// _d_k_d_en(declareProperty<Real>("d_k_d_en_"+getParam<std::string>("reaction"))),
53// _sampling_format(getParam<std::string>("sampling_format"))
54{
55 if (_rate_format == "EEDF")
56 {
57 std::vector<Real> reduced_field;
58 std::vector<Real> rate_coefficient;
59 std::string file_name =
60 getParam<std::string>("file_location") + "/" + getParam<FileName>("property_file");
61 MooseUtils::checkFileReadable(file_name);
62 const char * charPath = file_name.c_str();
63 std::ifstream myfile(charPath);
64 Real value;
65
66 if (myfile.is_open())
67 {
68 while (myfile >> value)
69 {
70 reduced_field.push_back(value);
71 myfile >> value;
72 rate_coefficient.push_back(value);
73 }
74 myfile.close();
75 }
76 else
77 mooseError("Unable to open file");
78
79 _coefficient_interpolation.setData(reduced_field, rate_coefficient);
80 }
81 else if (_rate_format == "Constant")
82 {
83 _rate_constant = getParam<Real>("rate_constant");
84 }
85 // else if (_rate_format == "Equation")
86 // {
87 // _rate_constant = _function.value(_t, _point);
88 // }
89 else if (_rate_format != "Equation")
90 {
91 mooseError("RateCoefficientProvider: Rate format " + _rate_format + " is not applicable.");
92 }
93}

95Real
96RateCoefficientProvider::reaction_coefficient() const
97{
98 Real reaction_rate = 0;
99 if (_rate_format == "EEDF")
100 {
101 if (_sampling_format == "electron_energy")
102 {
103 mooseError("Cannot sample with energy currently. Sample with reduced electric field.");
104 }
105 else if (_sampling_format == "reduced_field")
106 {
107 reaction_rate = _coefficient_interpolation.sample(_reduced_field_value[0]);
108
109 // reaction_rate = _coefficient_interpolation.sample(7.76697e-20);
110 // reaction_rate = _coefficient_interpolation.sample(5.0e-20);
111
112 reaction_rate = reaction_rate * 1e6;
113
114 // _reaction_rate[_qp] = _reaction_rate[_qp] * 6.022e23; // convert from [dens]/s to
115 // [dens]/mol/s _d_k_d_en[_qp] =
116 // _coefficient_interpolation.sampleDerivative(_reduced_field[_qp]);
117 }
118 }
119 else if (_rate_format == "Constant")
120 {
121 reaction_rate = _rate_constant;
122 }
123 else if (_rate_format == "Equation")
124 {
125 // this functionality is deprecated
126 // Does not need to be used
127 // reaction_rate = _function.value(_t, _point);
128 reaction_rate = 0; // This reaction rate is now calculated as an AuxVariable
129 }
130
131 return reaction_rate;
132}
