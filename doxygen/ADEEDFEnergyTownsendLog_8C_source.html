57ADReal
58ADEEDFEnergyTownsendLog::computeQpResidual()
59{
60 _electron_flux_mag = (std::exp(_em[_qp]) * (-_muem[_qp] * -_grad_potential[_qp] * _r_units -
61 _diffem[_qp] * _grad_em[_qp] * _r_units))
62 .norm();
63
64 return -_test[_i][_qp] * _alpha[_qp] * std::exp(_target[_qp]) * _electron_flux_mag *
65 _threshold_energy;
66}
