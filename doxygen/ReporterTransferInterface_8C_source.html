99void
100ReporterTransferInterface::declareClone(const ReporterName & rname,
101 FEProblemBase & problem,
102 const std::string & type,
103 const ReporterMode & mode)
104{
105 ReporterData & rdata = problem.getReporterData(ReporterData::WriteKey());
106 if (type == "bool")
107 rdata.declareReporterValue<bool, ReporterGeneralContext<bool>>(rname, mode, _rti_transfer);
108 else if (type == "integer")
109 rdata.declareReporterValue<int, ReporterGeneralContext<int>>(rname, mode, _rti_transfer);
110 else if (type == "real")
111 rdata.declareReporterValue<Real, ReporterGeneralContext<Real>>(rname, mode, _rti_transfer);
112 else if (type == "string")
113 rdata.declareReporterValue<std::string, ReporterGeneralContext<std::string>>(
114 rname, mode, _rti_transfer);
115 else
116 _rti_transfer.mooseError("Unknown reporter type, ", type, ".");
117
118 // Hide variables (if requested in parameters) if name is associated with a reporter object
119 hideVariableHelper(rname, problem);
120}

174std::vector<ReporterName>
175ReporterTransferInterface::getReporterNamesHelper(std::string prefix,
176 const std::string & obj_name,
177 const std::vector<ReporterName> & rep_names)
178{
179 if (!prefix.empty())
180 prefix += ":";
181 std::vector<ReporterName> rnames;
182 rnames.reserve(rep_names.size());
183 for (const auto & rn : rep_names)
184 rnames.emplace_back(obj_name, prefix + rn.getObjectName() + ":" + rn.getValueName());
185 return rnames;
186}
