34ADReal
35HphiRadialInterface::computeQpResidual(Moose::DGResidualType type)
36{
37 ADReal r = 0;
38
39 switch (type)
40 {
41 case Moose::Element:
42 r = _test[_i][_qp] * _eps_r[_qp] / _eps_r_neighbor[_qp] *
43 (_grad_neighbor_value[_qp] * _normals[_qp] +
44 _neighbor_value[_qp] / _q_point[_qp](0)) -
45 _test[_i][_qp] * _neighbor_value[_qp] / _q_point[_qp](0);
46 break;
47
48 case Moose::Neighbor:
49 r = 0.;
50 break;
51 }
52
53 return r;
54}
