87void
88PIDTransientControl::execute()
89{
90 if (_t >= _start_time && _t < _stop_time)
91 {
92 // Get the current value of the controllable parameter
93 Real value;
94 if (isParamValid("parameter"))
95 value = getControllableValue<Real>("parameter");
96 else
97 value = getPostprocessorValueByName(getParam<std::string>("parameter_pp"));
98
99 // Save integral and controlled value at each time step
100 // if the solver fails, a smaller time step will be used but _t_step is unchanged
101 if (_t_step != _t_step_old)
102 {
103 // Reset the error integral if PID is only used within each timestep
104 if (_reset_every_timestep)
105 _integral = 0;
106
107 _integral_old = _integral;
108 _value_old = value;
109 _t_step_old = _t_step;
110 _old_delta = 0;
111 }
112
113 // If there were coupling/Picard iterations during the transient and they failed,
114 // we need to reset the controlled value and the error integral to their initial value at the
115 // beginning of the coupling process
116 if (_app.getExecutioner()->picardSolve().numPicardIts() == 1)
117 {
118 _integral = _integral_old;
119 value = _value_old;
120 }
121
122 // Compute the delta between the current value of the postprocessor and the desired value
123 Real delta = _current - _target.value(_t);
124
125 // If desired, reset integral of the error if the error crosses zero
126 if (_reset_integral_windup && delta * _old_delta < 0)
127 _integral = 0;
128
129 // Compute the three error terms and add them to the controlled value
130 _integral += delta * _dt;
131 value += _Kint * _integral + _Kpro * delta;
132 if (_dt > 0)
133 value += _Kder * delta / _dt;
134
135 // Set the new value of the postprocessor
136 if (isParamValid("parameter"))
137 setControllableValue<Real>("parameter", value);
138 else
139 _fe_problem.setPostprocessorValueByName(getParam<std::string>("parameter_pp"), value);
140
141 // Keep track of the previous delta for integral windup control
142 _old_delta = delta;
143 }
144}
