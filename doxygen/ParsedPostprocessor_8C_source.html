38ParsedPostprocessor::ParsedPostprocessor(const InputParameters & parameters)
39 : GeneralPostprocessor(parameters),
40 FunctionParserUtils(parameters),
41 _n_pp(coupledPostprocessors("pp_names")),
42 _use_t(getParam<bool>("use_t")),
43 _value(0.0)
44{
45 // build postprocessors argument
46 std::string postprocessors;
47
48 // coupled postprocessors
49 std::vector<PostprocessorName> pp_names = getParam<std::vector<PostprocessorName>>("pp_names");
50 for (std::size_t i = 0; i < _n_pp; ++i)
51 postprocessors += (i == 0 ? "" : ",") + pp_names[i];
52
53 // add time if required
54 if (_use_t)
55 postprocessors += (postprocessors.empty() ? "" : ",") + std::string("t");
56
57 // base function object
58 _func_F = std::make_shared<SymFunction>();
59
60 // set FParser internal feature flags
61 setParserFeatureFlags(_func_F);
62
63 // add the constant expressions
64 addFParserConstants(_func_F,
65 getParam<std::vector<std::string>>("constant_names"),
66 getParam<std::vector<std::string>>("constant_expressions"));
67
68 // parse function
69 std::string function = getParam<std::string>("function");
70 if (_func_F->Parse(function, postprocessors) >= 0)
71 mooseError("Invalid parsed function\n", function, "\n", _func_F->ErrorMsg());
72
73 // optimize
74 if (!_disable_fpoptimizer)
75 _func_F->Optimize();
76
77 // just-in-time compile
78 if (_enable_jit)
79 {
80 // let rank 0 do the JIT compilation first
81 if (_communicator.rank() != 0)
82 _communicator.barrier();
83
84 _func_F->JITCompile();
85
86 // wait for ranks > 0 to catch up
87 if (_communicator.rank() == 0)
88 _communicator.barrier();
89 }
90
91 // reserve storage for parameter passing buffer
92 _func_params.resize(_n_pp + _use_t);
93 _pp_values.resize(_n_pp);
94 for (unsigned int i = 0; i < _n_pp; i++)
95 _pp_values[i] = &getPostprocessorValue("pp_names", i);
96}

112 _func_params[i] = *_pp_values[i];
