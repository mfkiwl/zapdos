33PlaneIDMeshGenerator::PlaneIDMeshGenerator(const InputParameters & parameters)
34 : MeshGenerator(parameters),
35 _input(getMesh("input")),
36 _axis_index(getParam<MooseEnum>("plane_axis")),
37 _element_id_name(getParam<std::string>("id_name"))
38{
39 if (!isParamValid("num_ids_per_plane"))
40 _planes = getParam<std::vector<Real>>("plane_coordinates");
41 else
42 {
43 _planes.clear();
44 std::vector<Real> base_planes = getParam<std::vector<Real>>("plane_coordinates");
45 std::vector<unsigned int> sublayers = getParam<std::vector<unsigned int>>("num_ids_per_plane");
46 if (base_planes.size() != sublayers.size() + 1)
47 paramError("plane_coordinates",
48 "Sizes of 'plane_coordinates' and 'num_ids_per_plane' disagree");
49 _planes.push_back(base_planes[0]);
50 for (unsigned int i = 0; i < sublayers.size(); ++i)
51 {
52 Real layer_size = (base_planes[i + 1] - base_planes[i]) / (Real)sublayers[i];
53 for (unsigned int j = 0; j < sublayers[i]; ++j)
54 _planes.push_back(_planes.back() + layer_size);
55 }
56 }
57 if (_planes.size() < 2)
58 paramError("plane_coordinates", "Size of 'plane_coordinates' should be at least two");
59}

61std::unique_ptr<MeshBase>
62PlaneIDMeshGenerator::generate()
63{
64 std::unique_ptr<MeshBase> mesh = std::move(_input);
65 if (mesh->mesh_dimension() < _axis_index + 1)
66 paramError("plane_axis", "PlaneIDMeshGenerator must operate on a proper plane axis");
67 unsigned int extra_id_index = 0;
68 if (!mesh->has_elem_integer(_element_id_name))
69 extra_id_index = mesh->add_elem_integer(_element_id_name);
70 else
71 {
72 extra_id_index = mesh->get_elem_integer_index(_element_id_name);
73 paramWarning(
74 "id_name", "An element integer with the name '", _element_id_name, "' already exists");
75 }
76
77 const Real tol = getParam<Real>("tolerance");
78 for (auto & elem : mesh->active_element_ptr_range())
79 {
80 const int layer_id = getPlaneID(elem->vertex_average());
81 for (unsigned int i = 0; i < elem->n_nodes(); ++i)
82 {
83 const Point & p = elem->point(i) - (elem->point(i) - elem->vertex_average()) * tol;
84 if (getPlaneID(p) != layer_id)
85 mooseError("Element at ", elem->vertex_average(), " is cut by the planes");
86 }
87 elem->set_extra_integer(extra_id_index, layer_id);
88 }
89 return dynamic_pointer_cast<MeshBase>(mesh);
90}

92int
93PlaneIDMeshGenerator::getPlaneID(const Point & p) const
94{
95 // check the input point is located within the plane range
96 if (p(_axis_index) < _planes[0] || p(_axis_index) > _planes.back())
97 mooseError("The planes do not cover element at ", p);
98 int id = 0;
99 // looping over each plane to find a plane ID corresponding to the input point
100 for (unsigned int layer_id = 0; layer_id < _planes.size() - 1; ++layer_id)
101 {
102 // check the input point is located between _planes[layer_id] and _planes[layer_id + 1]
103 if (_planes[layer_id] < p(_axis_index) && _planes[layer_id + 1] >= p(_axis_index))
104 {
105 id = layer_id;
106 break;
107 }
108 }
109 return id;
