24ArrayHFEMDirichletBC::ArrayHFEMDirichletBC(const InputParameters & parameters)
25 : ArrayLowerDIntegratedBC(parameters),
26 _value(isParamValid("value") ? getParam<RealEigenVector>("value")
27 : RealEigenVector::Zero(_count)),
28 _uhat_var(isParamValid("uhat") ? getArrayVar("uhat", 0) : nullptr),
29 _uhat(_uhat_var ? (_is_implicit ? &_uhat_var->slnLower() : &_uhat_var->slnLowerOld()) : nullptr)
30{
31 if (_value.size() != _count)
32 paramError(

35 if (_uhat_var)

38 paramError("uhat",

46

49 }

58void
59ArrayHFEMDirichletBC::computeLowerDQpResidual(RealEigenVector & r)
60{
61 if (_uhat)
62 r += (_u[_qp] - (*_uhat)[_qp]) * _test_lambda[_i][_qp];
63 else
64 r += (_u[_qp] - _value) * _test_lambda[_i][_qp];

67RealEigenVector
68ArrayHFEMDirichletBC::computeLowerDQpJacobian(Moose::ConstraintJacobianType type)
69{
70 RealEigenVector r = RealEigenVector::Zero(_count);
71 switch (type)
72 {
73 case Moose::LowerPrimary:
74 return RealEigenVector::Constant(_count, _test_lambda[_i][_qp] * _phi[_j][_qp]);
75
76 case Moose::PrimaryLower:
77 return RealEigenVector::Constant(_count, _phi_lambda[_j][_qp] * _test[_i][_qp]);
78
79 default:
80 break;
81 }
82
83 return r;
84}

86RealEigenMatrix
87ArrayHFEMDirichletBC::computeLowerDQpOffDiagJacobian(Moose::ConstraintJacobianType type,
88 const MooseVariableFEBase & jvar)
89{
90 if (_uhat_var && jvar.number() == _uhat_var->number() && type == Moose::LowerLower)
91 {
92 RealEigenVector v = RealEigenVector::Constant(_count, -_test_lambda[_i][_qp] * _phi[_j][_qp]);
93 RealEigenMatrix t = RealEigenMatrix::Zero(_var.count(), _var.count());
94 t.diagonal() = v;
95 return t;
96 }
97 else
98 return ArrayLowerDIntegratedBC::computeLowerDQpOffDiagJacobian(type, jvar);
99}
