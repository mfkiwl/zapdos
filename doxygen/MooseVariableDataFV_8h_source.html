466template <typename OutputType>
467inline bool
468MooseVariableDataFV<OutputType>::safeToComputeADUDot() const
469{
470 // If we don't have a time integrator then we have no way to calculate _ad_u_dot because we rely
471 // on calls to TimeIntegrator::computeADTimeDerivatives. Another potential situation where
472 // _ad_u_dot computation is potentially troublesome is if we are an auxiliary variable which uses
473 // the auxiliary system copy of the time integrator. Some derived time integrator classes do setup
474 // in their solve() method, and that solve() method only happens for the nonlinear system copy of
475 // the time integrator.
476 return _time_integrator && (_var.kind() == Moose::VAR_NONLINEAR);
477}

479template <typename OutputType>
480inline const ADTemplateVariableValue<OutputType> &
481MooseVariableDataFV<OutputType>::adUDot() const
482{
483 _need_ad = _need_ad_u_dot = true;
484
485 if (!safeToComputeADUDot())
486 // We will just copy the value of _u_dot into _ad_u_dot
487 _need_u_dot = true;
488
489 return _ad_u_dot;
490}
