31InputParameters
32Output::validParams()
33{
34 // Get the parameters from the parent object
35 InputParameters params = MooseObject::validParams();
36 params += SetupInterface::validParams();
37
38 // Displaced Mesh options
39 params.addParam<bool>(
40 "use_displaced", false, "Enable/disable the use of the displaced mesh for outputting");
41
42 // Output intervals and timing
43 params.addParam<unsigned int>(
44 "interval", 1, "The interval at which time steps are output to the solution file");
45 params.addParam<Real>(
46 "minimum_time_interval", 0.0, "The minimum simulation time between output steps");
47 params.addParam<std::vector<Real>>(
48 "sync_times", {}, "Times at which the output and solution is forced to occur");
49 params.addParam<TimesName>(
50 "sync_times_object",
51 "Times object providing the times at which the output and solution is forced to occur");
52 params.addParam<bool>("sync_only", false, "Only export results at sync times");
53 params.addParam<Real>("start_time", "Time at which this output object begins to operate");
54 params.addParam<Real>("end_time", "Time at which this output object stop operating");
55 params.addParam<int>("start_step", "Time step at which this output object begins to operate");
56 params.addParam<int>("end_step", "Time step at which this output object stop operating");
57 params.addParam<Real>(
58 "time_tolerance", 1e-14, "Time tolerance utilized checking start and end times");
59 params.addDeprecatedParam<FunctionName>(
60 "output_limiting_function",
61 "Piecewise base function that sets sync_times",
62 "Replaced by using the Times system with the sync_times_objects parameter");
63
64 // Update the 'execute_on' input parameter for output
65 ExecFlagEnum & exec_enum = params.set<ExecFlagEnum>("execute_on", true);
66 exec_enum = Output::getDefaultExecFlagEnum();
67 exec_enum = {EXEC_INITIAL, EXEC_TIMESTEP_END};
68 params.setDocString("execute_on", exec_enum.getDocString());
69
70 // Add ability to append to the 'execute_on' list
71 params.addParam<ExecFlagEnum>("additional_execute_on", exec_enum, exec_enum.getDocString());
72 params.set<ExecFlagEnum>("additional_execute_on").clear();
73 params.addParamNamesToGroup("execute_on additional_execute_on", "Execution scheduling");
74
75 // 'Timing' group
76 params.addParamNamesToGroup(
77 "time_tolerance interval sync_times sync_times_object sync_only start_time end_time "
78 "start_step end_step minimum_time_interval",
79 "Timing and frequency of output");
80
81 // Add a private parameter for indicating if it was created with short-cut syntax
82 params.addPrivateParam<bool>("_built_by_moose", false);
83
84 // Register this class as base class
85 params.declareControllable("enable");
86 params.registerBase("Output");
87
88 return params;
89}

99Output::Output(const InputParameters & parameters)
100 : MooseObject(parameters),
101 Restartable(this, "Output"),
102 MeshChangedInterface(parameters),
103 SetupInterface(this),
104 FunctionInterface(this),
105 PostprocessorInterface(this),
106 VectorPostprocessorInterface(this),
107 ReporterInterface(this),
108 PerfGraphInterface(this),
109 _problem_ptr(getParam<FEProblemBase *>("_fe_problem_base")),
110 _transient(_problem_ptr->isTransient()),
111 _use_displaced(getParam<bool>("use_displaced")),
112 _es_ptr(nullptr),
113 _mesh_ptr(nullptr),
114 _execute_on(getParam<ExecFlagEnum>("execute_on")),
115 _current_execute_flag(EXEC_NONE),
116 _time(_problem_ptr->time()),
117 _time_old(_problem_ptr->timeOld()),
118 _t_step(_problem_ptr->timeStep()),
119 _dt(_problem_ptr->dt()),
120 _dt_old(_problem_ptr->dtOld()),
121 _num(0),
122 _interval(getParam<unsigned int>("interval")),
123 _minimum_time_interval(getParam<Real>("minimum_time_interval")),
124 _sync_times(std::set<Real>(getParam<std::vector<Real>>("sync_times").begin(),
125 getParam<std::vector<Real>>("sync_times").end())),
126 _sync_times_object(isParamValid("sync_times_object")
127 ? static_cast<Times *>(&_problem_ptr->getUserObject<Times>(
128 getParam<TimesName>("sync_times_object")))
129 : nullptr),
130 _start_time(isParamValid("start_time") ? getParam<Real>("start_time")
131 : std::numeric_limits<Real>::lowest()),
132 _end_time(isParamValid("end_time") ? getParam<Real>("end_time")
133 : std::numeric_limits<Real>::max()),
134 _start_step(isParamValid("start_step") ? getParam<int>("start_step")
135 : std::numeric_limits<int>::lowest()),
136 _end_step(isParamValid("end_step") ? getParam<int>("end_step")
137 : std::numeric_limits<int>::max()),
138 _t_tol(getParam<Real>("time_tolerance")),
139 _sync_only(getParam<bool>("sync_only")),
140 _allow_output(true),
141 _is_advanced(false),
142 _advanced_execute_on(_execute_on, parameters),
143 _last_output_time(
144 declareRestartableData<Real>("last_output_time", std::numeric_limits<Real>::lowest()))
145{
146 if (_use_displaced)
147 {
148 std::shared_ptr<DisplacedProblem> dp = _problem_ptr->getDisplacedProblem();
149 if (dp != nullptr)
150 {
151 _es_ptr = &dp->es();
152 _mesh_ptr = &dp->mesh();
153 }
154 else
155 {
156 mooseWarning(
157 name(),
158 ": Parameter 'use_displaced' ignored, there is no displaced problem in your simulation.");
159 _es_ptr = &_problem_ptr->es();
160 _mesh_ptr = &_problem_ptr->mesh();
161 }
162 }
163 else
164 {
165 _es_ptr = &_problem_ptr->es();
166 _mesh_ptr = &_problem_ptr->mesh();
167 }
168
169 // Apply the additional output flags
170 if (isParamValid("additional_execute_on"))
171 {
172 const ExecFlagEnum & add = getParam<ExecFlagEnum>("additional_execute_on");
173 for (auto & me : add)
174 _execute_on.push_back(me);
175 }
176
177 if (isParamValid("output_limiting_function"))
178 {
179 const Function & olf = getFunction("output_limiting_function");
180 const PiecewiseBase * pwb_olf = dynamic_cast<const PiecewiseBase *>(&olf);
181 if (pwb_olf == nullptr)
182 mooseError("Function muse have a piecewise base!");
183
184 for (auto i = 0; i < pwb_olf->functionSize(); i++)
185 _sync_times.insert(pwb_olf->domain(i));
186 }
187
188 // Get sync times from Times object if using
189 if (_sync_times_object)
190 {
191 if (isParamValid("output_limiting_function") || isParamSetByUser("sync_times"))
192 paramError("sync_times_object",
193 "Only one method of specifying sync times is supported at a time");
194 else
195 // Sync times for the time steppers are taken from the output warehouse. The output warehouse
196 // takes sync times from the output objects immediately after the object is constructed. Hence
197 // we must ensure that we set the `_sync_times` in the constructor
198 _sync_times = _sync_times_object->getUniqueTimes();
199 }
200}

207void
208Output::outputStep(const ExecFlagType & type)
209{
210 // Output is not allowed
211 if (!_allow_output && type != EXEC_FORCED)
212 return;
213
214 // If recovering disable output of initial condition, it was already output
215 if (type == EXEC_INITIAL && _app.isRecovering())
216 return;
217
218 // Return if the current output is not on the desired interval
219 if (type != EXEC_FINAL && !onInterval())
220 return;
221
222 // store current simulation time
223 _last_output_time = _time;
224
225 // set current type
226 _current_execute_flag = type;
227
228 // Call the output method
229 if (shouldOutput())
230 {
231 TIME_SECTION("outputStep", 2, "Outputting Step");
232 output();
233 }
234
235 _current_execute_flag = EXEC_NONE;
236}

246bool
247Output::onInterval()
248{
249 // The output flag to return
250 bool output = false;
251
252 // Return true if the current step on the current output interval and within the output time range
253 // and within the output step range
254 if (_time >= _start_time && _time <= _end_time && _t_step >= _start_step &&
255 _t_step <= _end_step && (_t_step % _interval) == 0)
256 output = true;
257
258 // Return false if 'sync_only' is set to true
259 if (_sync_only)
260 output = false;
261
262 if (_sync_times_object)
263 {
264 const auto & sync_times = _sync_times_object->getUniqueTimes();
265 if (sync_times != _sync_times)
266 mooseError("The provided sync times object has changing time values. Only static time "
267 "values are supported since time steppers take sync times from the output "
268 "warehouse which determines its sync times at output construction time.");
269 }
270
271 // If sync times are not skipped, return true if the current time is a sync_time
272 if (_sync_times.find(_time) != _sync_times.end())
273 output = true;
274
275 // check if enough time has passed between outputs
276 if (_time > _last_output_time && _last_output_time + _minimum_time_interval > _time + _t_tol)
277 return false;
278
279 // Return the output status
280 return output;
281}
