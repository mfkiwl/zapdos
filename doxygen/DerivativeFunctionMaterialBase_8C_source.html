32template <bool is_ad>
33DerivativeFunctionMaterialBaseTempl<is_ad>::DerivativeFunctionMaterialBaseTempl(
34 const InputParameters & parameters)
35 : FunctionMaterialBase<is_ad>(parameters),
36 _third_derivatives(this->template getParam<unsigned int>("derivative_order") == 3)
37{
38 // reserve space for material properties and explicitly initialize to NULL
39 _prop_dF.resize(_nargs, NULL);
40 _prop_d2F.resize(_nargs);
41 _prop_d3F.resize(_nargs);
42 for (unsigned int i = 0; i < _nargs; ++i)
43 {
44 _prop_d2F[i].resize(_nargs, NULL);
45
46 if (_third_derivatives)
47 {
48 _prop_d3F[i].resize(_nargs);
49
50 for (unsigned int j = 0; j < _nargs; ++j)
51 _prop_d3F[i][j].resize(_nargs, NULL);
52 }
53 }
54
55 // initialize derivatives
56 for (unsigned int i = 0; i < _nargs; ++i)
57 {
58 if (!Coupleable::isCoupledConstant(_arg_names[i]))
59 _prop_dF[i] = &this->template declarePropertyDerivative<Real, is_ad>(_F_name, _arg_names[i]);
60
61 // second derivatives
62 for (unsigned int j = i; j < _nargs; ++j)
63 {
64 if (!Coupleable::isCoupledConstant(_arg_names[i]))
65 _prop_d2F[i][j] = _prop_d2F[j][i] = &this->template declarePropertyDerivative<Real, is_ad>(
66 _F_name, _arg_names[i], _arg_names[j]);
67
68 // third derivatives
69 if (_third_derivatives)
70 {
71 for (unsigned int k = j; k < _nargs; ++k)
72 {
73 // filling all permutations does not cost us much and simplifies access
74 // (no need to check i<=j<=k)
75 if (!Coupleable::isCoupledConstant(_arg_names[i]))
76 _prop_d3F[i][j][k] = _prop_d3F[k][i][j] = _prop_d3F[j][k][i] = _prop_d3F[k][j][i] =
77 _prop_d3F[j][i][k] = _prop_d3F[i][k][j] =
78 &this->template declarePropertyDerivative<Real, is_ad>(
79 _F_name, _arg_names[i], _arg_names[j], _arg_names[k]);
80 }
81 }
82 }
83 }

86template <bool is_ad>
87void
88DerivativeFunctionMaterialBaseTempl<is_ad>::initialSetup()
89{
90 // set the _prop_* pointers of all material properties that are not beeing used back to NULL
91 bool needs_third_derivatives = false;
92
93 if (!_fe_problem.isMatPropRequested(_F_name))
94 _prop_F = NULL;
95
96 for (unsigned int i = 0; i < _nargs; ++i)
97 {
98 if (!_fe_problem.isMatPropRequested(this->derivativePropertyNameFirst(_F_name, _arg_names[i])))
99 _prop_dF[i] = NULL;
100
101 // second derivatives
102 for (unsigned int j = i; j < _nargs; ++j)
103 {
104 if (!_fe_problem.isMatPropRequested(
105 this->derivativePropertyNameSecond(_F_name, _arg_names[i], _arg_names[j])))
106 _prop_d2F[i][j] = _prop_d2F[j][i] = NULL;
107
108 // third derivatives
109 if (_third_derivatives)
110 {
111 for (unsigned int k = j; k < _nargs; ++k)
112 {
113 if (!_fe_problem.isMatPropRequested(this->derivativePropertyNameThird(
114 _F_name, _arg_names[i], _arg_names[j], _arg_names[k])))
115 _prop_d3F[i][j][k] = _prop_d3F[k][i][j] = _prop_d3F[j][k][i] = _prop_d3F[k][j][i] =
116 _prop_d3F[j][i][k] = _prop_d3F[i][k][j] = NULL;
117 else
118 needs_third_derivatives = true;
119 }
120
121 if (!needs_third_derivatives)
122 mooseWarning("This simulation does not actually need the third derivatives of "
123 "DerivativeFunctionMaterialBaseTempl " +
124 name());
125 }
126 }
127 }

130template <bool is_ad>
131void
132DerivativeFunctionMaterialBaseTempl<is_ad>::computeProperties()
133{
134 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
135 {
136 // set function value
137 if (_prop_F)
138 (*_prop_F)[_qp] = computeF();
139
140 for (unsigned int i = 0; i < _nargs; ++i)
141 {
142 // set first derivatives
143 if (_prop_dF[i])
144 (*_prop_dF[i])[_qp] = computeDF(_arg_numbers[i]);
145
146 // second derivatives
147 for (unsigned int j = i; j < _nargs; ++j)
148 {
149 if (_prop_d2F[i][j])
150 (*_prop_d2F[i][j])[_qp] = computeD2F(_arg_numbers[i], _arg_numbers[j]);
151
152 // third derivatives
153 if (_third_derivatives)
154 {
155 for (unsigned int k = j; k < _nargs; ++k)
156 if (_prop_d3F[i][j][k])
157 (*_prop_d3F[i][j][k])[_qp] =
158 computeD3F(_arg_numbers[i], _arg_numbers[j], _arg_numbers[k]);
159 }
160 }
161 }
162 }
163}
