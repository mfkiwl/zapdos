25ElectronMobility::ElectronMobility(const InputParameters & parameters)
26 : AuxScalarKernel(parameters),
27 _reduced_field(coupledScalarValue("reduced_field"))
28{
29 std::string file_name = getParam<std::string>("file_location") + "/" + "electron_mobility.txt";
30 MooseUtils::checkFileReadable(file_name);
31 const char * charPath = file_name.c_str();
32 std::ifstream myfile(charPath);
33 Real value;
34
35 std::vector<Real> reduced_field;
36 std::vector<Real> mobility;
37 if (myfile.is_open())
38 {
39 while (myfile >> value)
40 {
41 reduced_field.push_back(value);
42 myfile >> value;
43 mobility.push_back(value);
44 }
45 myfile.close();
46 }
47 else
48 mooseError("Unable to open file");
49
50 _mobility.setData(reduced_field, mobility);
51}
