31ExplicitSSPRungeKutta::ExplicitSSPRungeKutta(const InputParameters & parameters)
32 : ExplicitTimeIntegrator(parameters),
33
34 _order(getParam<MooseEnum>("order")),
35
36 _stage(0),
37 _solution_intermediate_stage(_nl.addVector("solution_intermediate_stage", false, GHOSTED)),
38 _tmp_solution(_nl.addVector("tmp_solution", false, GHOSTED)),
39 _tmp_mass_solution_product(_nl.addVector("tmp_mass_solution_product", false, GHOSTED))
40{
41 // For SSPRK methods up to order 3, the number of stages equals the order
42 _n_stages = _order;
43
44 if (_order == 1)
45 {
46 _a = {{1.0}};
47 _b = {1.0};
48 _c = {0.0};
49 }
50 else if (_order == 2)
51 {
52 _a = {{1.0, 0.0}, {0.5, 0.5}};
53 _b = {1.0, 0.5};
54 _c = {0.0, 1.0};
55 }
56 else if (_order == 3)
57 {
58 _a = {{1.0, 0.0, 0.0}, {0.75, 0.25, 0.0}, {1.0 / 3.0, 0.0, 2.0 / 3.0}};
59 _b = {1.0, 0.25, 2.0 / 3.0};
60 _c = {0.0, 1.0, 0.5};
61 }
62 else
63 mooseError("Invalid time integration order.");
64
65 _solution_stage.resize(_n_stages + 1, nullptr);
66}

75void
76ExplicitSSPRungeKutta::computeADTimeDerivatives(ADReal & ad_u_dot,
77 const dof_id_type & dof,
78 DualReal & /*ad_u_dotdot*/) const
79{
80 // Note that if the solution for the current stage is not a nullptr, then neither
81 // are the previous stages.
82 if (_solution_stage[_stage])
83 {
84 for (unsigned int k = 0; k <= _stage; k++)
85 ad_u_dot -= _a[_stage][k] * (*(_solution_stage[k]))(dof);
86 ad_u_dot *= 1.0 / (_b[_stage] * _dt);
87 }
88 else
89 {
90 // We must be outside the solve loop in order to meet this criterion. In that case are we at
91 // timestep_begin or timestep_end? We don't know, so I don't think it's meaningful to compute
92 // derivatives here. Let's put in a quiet NaN which will only signal if we try to do something
93 // meaningful with it (and then we do want to signal because time derivatives may not be
94 // meaningful right now)
95 ad_u_dot = std::numeric_limits<typename ADReal::value_type>::quiet_NaN();
96 }
97}

99void
100ExplicitSSPRungeKutta::solve()
101{
102 // Reset iteration counts
103 _n_nonlinear_iterations = 0;
104 _n_linear_iterations = 0;
105
106 _current_time = _fe_problem.time();
107 const Real time_old = _fe_problem.timeOld();
108 const Real dt = _current_time - time_old;
109
110 bool converged = false;
111
112 _solution_stage[0] = &_solution_old;
113 for (_stage = 0; _stage < _n_stages; _stage++)
114 {
115 if (_stage == 0)
116 {
117 // Nothing needs to be done
118 }
119 else if (_stage == _n_stages - 1)
120 {
121 _solution_stage[_stage] = _solution;
122 }
123 else
124 {
125 // Else must be the intermediate stage of the 3-stage method
126 _solution_intermediate_stage = *_solution;
127 _solution_intermediate_stage.close();
128 _solution_stage[_stage] = &_solution_intermediate_stage;
129 }
130
131 // Set stage time for residual evaluation
132 _fe_problem.time() = time_old + _c[_stage] * dt;
133 _nonlinear_implicit_system->update();
134
135 converged = solveStage();
136 if (!converged)
137 return;
138 }
139}

141bool
142ExplicitSSPRungeKutta::solveStage()
143{
144 // Compute the mass matrix
145 _nl.computeTimeDerivatives();
146 auto & mass_matrix = _nonlinear_implicit_system->get_system_matrix();
147 _fe_problem.computeJacobianTag(
148 *_nonlinear_implicit_system->current_local_solution, mass_matrix, _Ke_time_tag);
149
150 // Compute RHS vector using previous stage solution in steady-state residual
151 _explicit_residual.zero();
152 _fe_problem.computeResidual(
153 *_nonlinear_implicit_system->current_local_solution, _explicit_residual, _nl.number());
154
155 // Move the residual to the RHS
156 _explicit_residual *= -1.0;
157
158 // Perform the linear solve
159 bool converged = performExplicitSolve(mass_matrix);
160
161 // Update the solution: u^(s) = u^(s-1) + du^(s)
162 (*_nonlinear_implicit_system->solution).zero();
163 *_nonlinear_implicit_system->solution = *(_solution_stage[_stage]);
164 *_nonlinear_implicit_system->solution += _solution_update;
165
166 // Enforce contraints on the solution
167 DofMap & dof_map = _nonlinear_implicit_system->get_dof_map();
168 dof_map.enforce_constraints_exactly(*_nonlinear_implicit_system,
169 _nonlinear_implicit_system->solution.get());
170 _nonlinear_implicit_system->update();
171
172 _nl.setSolution(*_nonlinear_implicit_system->current_local_solution);
173
174 _nonlinear_implicit_system->nonlinear_solver->converged = converged;
175
176 return converged;
177}

179void
180ExplicitSSPRungeKutta::postResidual(NumericVector<Number> & residual)
181{
182 // The time residual is not included in the steady-state residual
183 residual += _Re_non_time;
184
185 // Compute \sum_{k=0}^{s-1} a_{s,k} u^(k) - u^(s-1)
186 _tmp_solution.zero();
187 for (unsigned int k = 0; k <= _stage; k++)
188 _tmp_solution.add(_a[_stage][k], *(_solution_stage[k]));
189 _tmp_solution.add(-1.0, *(_solution_stage[_stage]));
190 _tmp_solution.close();
191
192 // Perform mass matrix product with the above vector
193 auto & mass_matrix = _nonlinear_implicit_system->get_system_matrix();
194 mass_matrix.vector_mult(_tmp_mass_solution_product, _tmp_solution);
195
196 // Finish computing residual vector (before modification by nodal BCs)
197 residual -= _tmp_mass_solution_product;
198
199 residual.close();
200
201 // Set time at which to evaluate nodal BCs
202 _fe_problem.time() = _current_time;
203}
