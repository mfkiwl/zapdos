45template <bool is_ad>
46std::pair<Real, Real>
47ElementExtremeFunctorValueTempl<is_ad>::getProxyValuePair()
48{
49 // Most element evaluations do not use skewness correction,
50 // but this could become a parameter in the future
51 Moose::ElemArg elem = makeElemArg(_current_elem);
52 return std::make_pair(MetaPhysicL::raw_value(_proxy_functor(elem, determineState())),
53 MetaPhysicL::raw_value(_functor(elem, determineState())));
54}
