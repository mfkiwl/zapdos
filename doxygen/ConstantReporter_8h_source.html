57template <typename T>
58std::vector<T *>
59ConstantReporter::declareConstantReporterValues(const std::string & prefix)
60{
61 std::string names_param(prefix + "_names");
62 std::string values_param(prefix + "_values");
63 std::vector<T *> data;
64
65 if (isParamValid(names_param) && !isParamValid(values_param))
66 paramError(names_param, "Must specify values using ", values_param);
67 else if (!isParamValid(names_param) && isParamValid(values_param))
68 paramError(values_param, "Use ", names_param, " to specify reporter names.");
69 else if (!isParamValid(names_param) && !isParamValid(values_param))
70 return data;
71
72 auto & names = getParam<std::vector<ReporterValueName>>(names_param);
73 auto & values = this->getParam<std::vector<T>>(values_param);
74 if (names.size() != values.size())
75 paramError(values_param,
76 "Number of names specified in ",
77 names_param,
78 " must match number of values specified in ",
79 values_param);
80
81 for (unsigned int i = 0; i < names.size(); ++i)
82 data.push_back(&this->declareValueByName<T>(names[i], values[i]));
83
84 return data;
85}
