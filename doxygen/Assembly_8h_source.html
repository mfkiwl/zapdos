2409 auto neighbor = elem->neighbor_ptr(side);

2417 return q;

2420 if (q->get_order() > q_neighbor->get_order())

3048Assembly::cacheJacobian(const Residuals & residuals,
3049 const Indices & input_row_indices,
3050 const Real scaling_factor,

3067#ifndef NDEBUG
