134template <typename T>
135template <typename PolymorphicLambda>
136void
137PiecewiseByBlockLambdaFunctor<T>::setFunctor(const MooseMesh & libmesh_dbg_var(mesh),
138 const std::set<SubdomainID> & block_ids,
139 PolymorphicLambda my_lammy)
140{
141 mooseAssert(&mesh == &_mesh,
142 "We should always be setting this functor with the same mesh. We may relax this "
143 "assertion later");
144
145 auto add_lammy = [this, my_lammy](const SubdomainID block_id)
146 {
147 auto pr = _elem_functor.emplace(block_id, my_lammy);
148 if (!pr.second)
149 mooseError("No insertion for the functor material property '",
150 this->functorName(),
151 "' for block id ",
152 block_id,
153 ". Another material must already declare this property on that block.");
154 _face_functor.emplace(block_id, my_lammy);
155 _elem_qp_functor.emplace(block_id, my_lammy);
156 _elem_side_qp_functor.emplace(block_id, my_lammy);
157 _elem_point_functor.emplace(block_id, my_lammy);
158 _node_functor.emplace(block_id, my_lammy);
159 };
160
161 for (const auto block_id : block_ids)
162 add_lammy(block_id);
163}

165template <typename T>
166bool
167PiecewiseByBlockLambdaFunctor<T>::isExtrapolatedBoundaryFace(const FaceInfo & fi,
168 const Elem *,
169 const Moose::StateArg &) const
170{
171 if (!fi.neighborPtr())
172 return true;
173
174 const bool defined_on_elem = _elem_functor.count(fi.elem().subdomain_id());
175 const bool defined_on_neighbor = _elem_functor.count(fi.neighbor().subdomain_id());
176 const bool extrapolated = (defined_on_elem + defined_on_neighbor) == 1;
177
178 mooseAssert(defined_on_elem || defined_on_neighbor,
179 "This shouldn't be called if we aren't defined on either side.");
180 return extrapolated;
181}

183template <typename T>
184bool
185PiecewiseByBlockLambdaFunctor<T>::hasBlocks(const SubdomainID id) const
186{
187 // If any of the maps has a functor for that block, it has the block
188 const bool has_blocks = _elem_functor.count(id);
189 mooseAssert(has_blocks == _face_functor.count(id),
190 "All functor sets should agree on whether we have this sub id");
191 mooseAssert(has_blocks == _elem_qp_functor.count(id),
192 "All functor sets should agree on whether we have this sub id");
193 mooseAssert(has_blocks == _elem_side_qp_functor.count(id),
194 "All functor sets should agree on whether we have this sub id");
195 return has_blocks;
196}

210template <typename T>
211typename PiecewiseByBlockLambdaFunctor<T>::ValueType
212PiecewiseByBlockLambdaFunctor<T>::evaluate(const Moose::ElemArg & elem_arg,
213 const Moose::StateArg & time) const
214{
215 const Elem * const elem = elem_arg.elem;
216 mooseAssert(elem && elem != libMesh::remote_elem,
217 "The element must be non-null and non-remote in functor material properties");
218 auto it = _elem_functor.find(elem->subdomain_id());
219 if (it == _elem_functor.end())
220 subdomainErrorMessage(elem->subdomain_id());
221
222 return it->second(elem_arg, time);
223}

225template <typename T>
226typename PiecewiseByBlockLambdaFunctor<T>::ValueType
227PiecewiseByBlockLambdaFunctor<T>::evaluate(const Moose::FaceArg & face,
228 const Moose::StateArg & time) const
229{
230 using namespace Moose::FV;
231
232 if (face.face_side)
233 {
234 const auto sub_id = face.face_side->subdomain_id();
235 auto it = _face_functor.find(sub_id);
236 if (it == _face_functor.end())
237 subdomainErrorMessage(sub_id);
238
239 return it->second(face, time);
240 }
241
242 mooseAssert(this->isInternalFace(*face.fi),
243 "If we did not have a face side, then we must be an internal face");
244 return interpolate(*this, face, time);
245}
