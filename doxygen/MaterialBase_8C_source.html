25 params += BlockRestrictable::validParams();
26 params += BoundaryRestrictable::validParams();
27 params += TransientInterface::validParams();

51 params.addParam<std::vector<std::string>>(
52 "output_properties",
53 {},
54 "List of material properties, from this material, to output (outputs "
55 "must also be defined to an output type)");
56 params.addParam<MaterialPropertyName>(
57 "declare_suffix",
58 "",
59 "An optional suffix parameter that can be appended to any declared properties. The suffix "
60 "will be prepended with a '_' character.");
61

63 params.declareControllable("enable");
64
65 params.addParamNamesToGroup("outputs output_properties", "Outputs");
66 params.addParamNamesToGroup("use_displaced_mesh", "Advanced");
67 params.registerBase("MaterialBase");

113void
114MaterialBase::initStatefulProperties(unsigned int n_points)
115{
116 for (_qp = 0; _qp < n_points; ++_qp)
117 initQpStatefulProperties();
118
119 // checking for statefulness of properties via this loop is necessary
120 // because owned props might have been promoted to stateful by calls to
121 // getMaterialProperty[Old/Older] from other objects. In these cases, this
122 // object won't otherwise know that it owns stateful properties.
123 for (auto & prop : _supplied_props)
124 if (materialData().getMaterialPropertyStorage().isStatefulProp(prop) &&
125 !_overrides_init_stateful_props)
126 mooseWarning(std::string("Material \"") + name() +
127 "\" provides one or more stateful "
128 "properties but initQpStatefulProperties() "
129 "was not overridden in the derived class.");
130}

148void
149MaterialBase::registerPropName(const std::string & prop_name, bool is_get, const unsigned int state)
150{
151 if (!is_get)
152 {
153 const auto property_id = materialData().getPropertyId(prop_name);
154
155 _supplied_props.insert(prop_name);
156 _supplied_prop_ids.insert(property_id);
157
158 // Store the minimum state declared
159 auto find_min_state = _props_to_min_states.find(property_id);
160 if (find_min_state == _props_to_min_states.end())
161 _props_to_min_states.emplace(property_id, state);
162 else
163 find_min_state->second = std::min(find_min_state->second, state);
164
165 // Store material properties for block ids
166 for (const auto & block_id : blockIDs())
167 _fe_problem.storeSubdomainMatPropName(block_id, prop_name);
168
169 // Store material properties for the boundary ids
170 for (const auto & boundary_id : boundaryIDs())
171 _fe_problem.storeBoundaryMatPropName(boundary_id, prop_name);
172 }
173
174 if (state > 0)
175 _has_stateful_property = true;
176}
