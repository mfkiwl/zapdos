50ADReal
51ElectronEnergyTermRate::computeQpResidual()
52{
53 if (isCoupled("v"))
54 return -_test[_i][_qp] * _rate_coefficient[_qp] * std::exp(_v[_qp]) * std::exp(_em[_qp]) *
55 _energy_change;
56 else
57 return -_test[_i][_qp] * _rate_coefficient[_qp] * _n_gas[_qp] * std::exp(_em[_qp]) *
58 _energy_change;
59}
