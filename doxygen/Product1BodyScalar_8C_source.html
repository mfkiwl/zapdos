45Real
46Product1BodyScalar::computeQpJacobian()
47{
48 Real power, eq_u_mult, gas_mult;
49
50 power = 0.0;
51 eq_u_mult = 1.0;
52 gas_mult = 1.0;
53
54 if (isCoupledScalar("v") && _v_eq_u)
55 {
56 power += 1.0;
57 eq_u_mult = _v[_i];
58 }
59 else
60 gas_mult *= _v[_i];
61
62 return -_stoichiometric_coeff * _rate_coefficient[_i] * gas_mult * power *
63 std::pow(eq_u_mult, power - 1);
64}

66Real
67Product1BodyScalar::computeQpOffDiagJacobian(unsigned int jvar)
68{
69 Real power, deriv_factor, other_factor;
70 power = 0;
71 other_factor = 1;
72 deriv_factor = 1;
73
74 if (jvar == _v_var && !_v_eq_u)
75 {
76 power += 1;
77 deriv_factor = _v[_i];
78 }
79 else
80 other_factor *= _v[_i];
81
82 return -_stoichiometric_coeff * _rate_coefficient[_i] * other_factor * power *
83 std::pow(deriv_factor, power - 1);
