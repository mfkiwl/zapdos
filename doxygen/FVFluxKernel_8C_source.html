85// Note the lack of quadrature point loops in the residual/jacobian compute
86// functions. This is because finite volumes currently only works with
87// constant monomial elements. We only have one quadrature point regardless of
88// problem dimension and just multiply by the face area.
89
90bool
91FVFluxKernel::skipForBoundary(const FaceInfo & fi) const
92{
93 // Boundaries to avoid come first, since they are always obeyed
94 if (avoidBoundary(fi))
95 return true;
96
97 // We're not on a boundary, so in practice we're not 'skipping'
98 if (!onBoundary(fi))
99 return false;
100
101 // Blanket forcing on boundary
102 if (_force_boundary_execution)
103 return false;
104
105 // Selected boundaries to force
106 for (const auto bnd_to_force : _boundaries_to_force)
107 if (fi.boundaryIDs().count(bnd_to_force))
108 return false;
109
110 // If we have flux bcs then we do skip
111 const auto & flux_pr = _var.getFluxBCs(fi);
112 if (flux_pr.first)
113 return true;
114
115 // If we don't have flux bcs *and* we do have dirichlet bcs then we don't skip. If we don't have
116 // either then we assume natural boundary condition and we should skip
117 return !_var.getDirichletBC(fi).first;
118}

120void
121FVFluxKernel::computeResidual(const FaceInfo & fi)
122{
123 if (skipForBoundary(fi))
124 return;
125
126 _face_info = &fi;
127 _normal = fi.normal();
128 _face_type = fi.faceType(std::make_pair(_var.number(), _var.sys().number()));
129 auto r = MetaPhysicL::raw_value(fi.faceArea() * fi.faceCoord() * computeQpResidual());
130
131 // residual contributions for a flux kernel go to both neighboring faces.
132 // They are equal in magnitude but opposite in direction due to the outward
133 // facing unit normals of the face for each neighboring elements being
134 // oriented oppositely. We calculate the residual contribution once using
135 // the lower-id-elem-oriented _normal and just use the resulting residual's
136 // negative for the contribution to the neighbor element.
137
138 // The fancy face type if condition checks here are because we might
139 // currently be running on a face for which this kernel's variable is only
140 // defined on one side. If this is the case, we need to only calculate+add
141 // the residual contribution if there is a dirichlet bc for the active
142 // face+variable. We always need to add the residual contribution when the
143 // variable is defined on both sides of the face. If the variable is only
144 // defined on one side and there is NOT a dirichlet BC, then there is either
145 // a flux BC or a natural BC - in either of those cases we don't want to add
146 // any residual contributions from regular flux kernels.
147 if (_face_type == FaceInfo::VarFaceNeighbors::ELEM ||
148 _face_type == FaceInfo::VarFaceNeighbors::BOTH)
149 {
150 // residual contribution of this kernel to the elem element
151 prepareVectorTag(_assembly, _var.number());
152 _local_re(0) = r;
153 accumulateTaggedLocalResidual();
154 }
155 if (_face_type == FaceInfo::VarFaceNeighbors::NEIGHBOR ||
156 _face_type == FaceInfo::VarFaceNeighbors::BOTH)
157 {
158 // residual contribution of this kernel to the neighbor element
159 prepareVectorTagNeighbor(_assembly, _var.number());
160 _local_re(0) = -r;
161 accumulateTaggedLocalResidual();
162 }

165void
166FVFluxKernel::computeJacobian(const FaceInfo & fi)
167{
168 if (skipForBoundary(fi))
169 return;
170
171 _face_info = &fi;
172 _normal = fi.normal();
173 _face_type = fi.faceType(std::make_pair(_var.number(), _var.sys().number()));
174 const ADReal r = fi.faceArea() * fi.faceCoord() * computeQpResidual();
175
176 // The fancy face type if condition checks here are because we might
177 // currently be running on a face for which this kernel's variable is only
178 // defined on one side. If this is the case, we need to only calculate+add
179 // the jacobian contribution if there is a dirichlet bc for the active
180 // face+variable. We always need to add the jacobian contribution when the
181 // variable is defined on both sides of the face. If the variable is only
182 // defined on one side and there is NOT a dirichlet BC, then there is either
183 // a flux BC or a natural BC - in either of those cases we don't want to add
184 // any jacobian contributions from regular flux kernels.
185 if (_face_type == FaceInfo::VarFaceNeighbors::ELEM ||
186 _face_type == FaceInfo::VarFaceNeighbors::BOTH)
187 {
188 mooseAssert(_var.dofIndices().size() == 1, "We're currently built to use CONSTANT MONOMIALS");
189
190 addResidualsAndJacobian(
191 _assembly, std::array<ADReal, 1>{{r}}, _var.dofIndices(), _var.scalingFactor());
192 }
193
194 if (_face_type == FaceInfo::VarFaceNeighbors::NEIGHBOR ||
195 _face_type == FaceInfo::VarFaceNeighbors::BOTH)
196 {
197 mooseAssert((_face_type == FaceInfo::VarFaceNeighbors::NEIGHBOR) ==
198 (_var.dofIndices().size() == 0),
199 "If the variable is only defined on the neighbor hand side of the face, then that "
200 "means it should have no dof indices on the elem element. Conversely if "
201 "the variable is defined on both sides of the face, then it should have a non-zero "
202 "number of degrees of freedom on the elem element");
203
204 // We switch the sign for the neighbor residual
205 ADReal neighbor_r = -r;
206
207 mooseAssert(_var.dofIndicesNeighbor().size() == 1,
208 "We're currently built to use CONSTANT MONOMIALS");
209
210 addResidualsAndJacobian(_assembly,
211 std::array<ADReal, 1>{{neighbor_r}},
212 _var.dofIndicesNeighbor(),
213 _var.scalingFactor());
214 }

270FVFluxKernel::adjustRMGhostLayers(const unsigned short ghost_layers) const
271{
272 auto & factory = _app.getFactory();
273
