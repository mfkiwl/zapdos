52ParsedVectorAux::ParsedVectorAux(const InputParameters & parameters)
53 : VectorAuxKernel(parameters),
54 FunctionParserUtils(parameters),
55 _function({getParam<std::string>("expression_x"),
56 getParam<std::string>("expression_y"),
57 getParam<std::string>("expression_z")}),
58 _nargs(coupledComponents("coupled_variables")),
59 _n_vector_args(coupledComponents("coupled_vector_variables")),
60 _args(coupledValues("coupled_variables")),
61 _vector_args(coupledVectorValues("coupled_vector_variables")),
62 _use_xyzt(getParam<bool>("use_xyzt"))
63{
64 _func_F.resize(LIBMESH_DIM);
65 _function.resize(LIBMESH_DIM);
66
67 for (const auto i : make_range(Moose::dim))
68 {
69 // build variables argument
70 std::string variables;
71
72 // coupled regular field variables
73 for (const auto i : make_range(_nargs))
74 variables += (i == 0 ? "" : ",") + getFieldVar("coupled_variables", i)->name();
75
76 // coupled vector field variables
77 for (const auto i : make_range(_n_vector_args))
78 variables +=
79 (variables.empty() ? "" : ",") + getFieldVar("coupled_vector_variables", i)->name();
80
81 // "system" variables
82 const std::vector<std::string> xyzt = {"x", "y", "z", "t"};
83 if (_use_xyzt)
84 for (auto & v : xyzt)
85 variables += (variables.empty() ? "" : ",") + v;
86
87 // base function object
88 _func_F[i] = std::make_shared<SymFunction>();
89
90 // set FParser internal feature flags
91 setParserFeatureFlags(_func_F[i]);
92
93 // add the constant expressions
94 auto constant_names = isParamValid("constant_names")
95 ? getParam<std::vector<std::vector<std::string>>>("constant_names")
96 : std::vector<std::vector<std::string>>{};
97 auto constant_expressions =
98 isParamValid("constant_expressions")
99 ? getParam<std::vector<std::vector<std::string>>>("constant_expressions")
100 : std::vector<std::vector<std::string>>{};
101
102 if (constant_names.size() != constant_expressions.size())
103 paramError("constant_names", "Must be same length as constant_expressions");
104
105 if (isParamValid("constant_names") && isParamValid("constant_expressions"))
106 addFParserConstants(_func_F[i], constant_names[i], constant_expressions[i]);
107
108 // parse function
109 if (_func_F[i]->Parse(_function[i], variables) >= 0)
110 mooseError("Invalid function\n",
111 _function[i],
112 "\nin ParsedVectorAux '",
113 name(),
114 "' component ",
115 i == 0 ? "x"
116 : i == 1 ? "y"
117 : "z",
118 ".\n",
119 _func_F[i]->ErrorMsg());
120
121 // optimize
122 if (!_disable_fpoptimizer)
123 _func_F[i]->Optimize();
124
125 // just-in-time compile
126 if (_enable_jit)
127 {
128 // let rank 0 do the JIT compilation first
129 if (_communicator.rank() != 0)
130 _communicator.barrier();
131
132 _func_F[i]->JITCompile();
133
134 // wait for ranks > 0 to catch up
135 if (_communicator.rank() == 0)
136 _communicator.barrier();
137 }
138 }
139 // reserve storage for parameter passing buffer
140 _func_params.resize(_nargs + _n_vector_args + (_use_xyzt ? 4 : 0));
141}

143RealVectorValue
144ParsedVectorAux::computeValue()
145{
146 RealVectorValue value;
147 for (const auto i : make_range(Moose::dim))
148 {
149 for (const auto j : make_range(_nargs))
150 _func_params[j] = (*_args[j])[_qp];
151
152 for (const auto j : make_range(_n_vector_args))
153 _func_params[_nargs + j] = (*_vector_args[j])[_qp](i);
154
155 if (_use_xyzt)
156 {
157 for (const auto j : make_range(Moose::dim))
158 _func_params[_nargs + _n_vector_args + j] =
159 isNodal() ? (*_current_node)(j) : _q_point[_qp](j);
160 _func_params[_nargs + _n_vector_args + 3] = _t;
161 }
162 value(i) = evaluate(_func_F[i]);
163 }
164 return value;
165}
