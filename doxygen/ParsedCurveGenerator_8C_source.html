78ParsedCurveGenerator::ParsedCurveGenerator(const InputParameters & parameters)
79 : MeshGenerator(parameters),
80 FunctionParserUtils<false>(parameters),
81 _function_x(getParam<std::string>("x_formula")),
82 _function_y(getParam<std::string>("y_formula")),
83 _function_z(getParam<std::string>("z_formula")),
84 _nums_segments(getParam<std::vector<unsigned int>>("nums_segments")),
85 _section_bounding_t_values(getParam<std::vector<Real>>("section_bounding_t_values")),
86 _is_closed_loop(getParam<bool>("is_closed_loop")),
87 _point_overlapping_tolerance(getParam<Real>("point_overlapping_tolerance")),
88 _forced_closing_num_segments(isParamValid("forced_closing_num_segments")
89 ? getParam<unsigned int>("forced_closing_num_segments")
90 : 0),
91 _oversample_factor(getParam<Real>("oversample_factor")),
92 _max_oversample_number_factor(getParam<unsigned int>("max_oversample_number_factor")),
93 _node_set_boundaries(getParam<std::vector<BoundaryName>>("edge_nodesets"))
94{
95 if (std::adjacent_find(_section_bounding_t_values.begin(), _section_bounding_t_values.end()) !=
96 _section_bounding_t_values.end())
97 paramError("section_bounding_t_values", "elements must be unique.");
98 if (!std::is_sorted(_section_bounding_t_values.begin(), _section_bounding_t_values.end()) &&
99 !std::is_sorted(_section_bounding_t_values.rbegin(), _section_bounding_t_values.rend()))
100 paramError("section_bounding_t_values", "elements must change monotonically.");
101 if (_nums_segments.size() != _section_bounding_t_values.size() - 1)
102 paramError(
103 "nums_segments",
104 "The size of this parameter must be one less than size of section_bounding_t_values.");
105 _func_Fx = std::make_shared<SymFunction>();
106 _func_Fy = std::make_shared<SymFunction>();
107 _func_Fz = std::make_shared<SymFunction>();
108
109 // set FParser internal feature flags
110 setParserFeatureFlags(_func_Fx);
111 setParserFeatureFlags(_func_Fy);
112 setParserFeatureFlags(_func_Fz);
113
114 // add the constant expressions; note that the three functions share one set of constants
115 if (isParamValid("constant_names") && isParamValid("constant_expressions"))
116 {
117 addFParserConstants(_func_Fx,
118 getParam<std::vector<std::string>>("constant_names"),
119 getParam<std::vector<std::string>>("constant_expressions"));
120 addFParserConstants(_func_Fy,
121 getParam<std::vector<std::string>>("constant_names"),
122 getParam<std::vector<std::string>>("constant_expressions"));
123 addFParserConstants(_func_Fz,
124 getParam<std::vector<std::string>>("constant_names"),
125 getParam<std::vector<std::string>>("constant_expressions"));
126 }
127
128 // parse functions
129 if (_func_Fx->Parse(_function_x, "t") >= 0)
130 mooseError("Invalid function x(t)\n",
131 _function_x,
132 "\nin ParsedCurveGenerator ",
133 name(),
134 ".\n",
135 _func_Fx->ErrorMsg());
136 if (_func_Fy->Parse(_function_y, "t") >= 0)
137 mooseError("Invalid function y(t)\n",
138 _function_y,
139 "\nin ParsedCurveGenerator ",
140 name(),
141 ".\n",
142 _func_Fy->ErrorMsg());
143 if (_func_Fz->Parse(_function_z, "t") >= 0)
144 mooseError("Invalid function z(t)\n",
145 _function_z,
146 "\nin ParsedCurveGenerator ",
147 name(),
148 ".\n",
149 _func_Fz->ErrorMsg());
150
151 _func_params.resize(1);
152
153 if (!_is_closed_loop && _forced_closing_num_segments > 0)
154 paramError("forced_closing_num_segments",
155 "this parameter is not needed if the curve to be generated is not a closed loop.");
156
157 if (_node_set_boundaries.size() > 1 && _is_closed_loop)
158 paramError("edge_nodesets", "Cannot add more than one edge nodeset on a closed loop");
159 if (_node_set_boundaries.size() && _node_set_boundaries.size() != 2 && !_is_closed_loop)
160 paramError("edge_nodesets", "Exactly two edges in an open loop");
161}

163std::unique_ptr<MeshBase>
164ParsedCurveGenerator::generate()
165{
166 auto mesh = buildReplicatedMesh(2);
167
168 // Do oversampling for each section of the curve as defined by "section_bounding_t_values"
169 for (unsigned int i = 0; i < _nums_segments.size(); i++)
170 {
171 std::vector<Real> t_sect_space;
172 std::vector<Real> dis_sect_space;
173 tSectionSpaceDefiner(_section_bounding_t_values[i],
174 _section_bounding_t_values[i + 1],
175 t_sect_space,
176 dis_sect_space,
177 _nums_segments[i],
178 _is_closed_loop && _nums_segments.size() == 1,
179 _oversample_factor);
180 if (i > 0)
181 {
182 // Remove the last t value and distance value of previous section to avert overlapping
183 _t_space.pop_back();
184 // Add the last distance value as this is a cumulative distance
185 for (auto & dis_elem : dis_sect_space)
186 dis_elem += _dis_space.back();
187 _dis_space.pop_back();
188 }
189 _t_space.insert(_t_space.end(), t_sect_space.begin(), t_sect_space.end());
190 _dis_space.insert(_dis_space.end(), dis_sect_space.begin(), dis_sect_space.end());
191 }
192
193 // Use linear interpolation to help achieve nearly uniform intervals for each section
194 std::unique_ptr<LinearInterpolation> linear_t;
195 linear_t = std::make_unique<LinearInterpolation>(_dis_space, _t_space);
196
197 std::vector<Node *> nodes(std::accumulate(_nums_segments.begin(), _nums_segments.end(), 0) + 1);
198 for (unsigned int i = 0; i < nodes.size(); i++)
199 {
200 _func_params[0] = linear_t->sample((Real)i);
201 Point side_p = Point(evaluate(_func_Fx), evaluate(_func_Fy), evaluate(_func_Fz));
202 nodes[i] = mesh->add_point(side_p, i);
203 }
204
205 // For a closed loop, need to check if the first and last Points are overlapped
206 if (_is_closed_loop)
207 {
208 if (MooseUtils::absoluteFuzzyEqual(
209 (*nodes.back() - *nodes.front()).norm(), 0.0, _point_overlapping_tolerance))
210 {
211 // Remove the overlapped nodes for a closed loop
212 mesh->delete_node(nodes.back());
213 nodes.resize(nodes.size() - 1);
214 if (_forced_closing_num_segments > 0)
215 paramError("forced_closing_num_segments",
216 "this parameter is not needed if the first and last points of the curve to be "
217 "generated are overlapped.");
218 }
219 else if (_forced_closing_num_segments > 1)
220 {
221 // Add extra nodes on the curve section used to forcefully close the loop
222 const Point start_pt(*nodes.back());
223 const Point end_pt(*nodes.front());
224 const unsigned int num_nodes_0(nodes.size());
225 for (unsigned int i = 1; i < _forced_closing_num_segments; i++)
226 {
227 Point side_p =
228 start_pt + (end_pt - start_pt) * (Real)i / (Real)_forced_closing_num_segments;
229 nodes.push_back(mesh->add_point(side_p, num_nodes_0 + i - 1));
230 }
231 }
232 }
233
234 for (unsigned int i = 0; i < nodes.size() - !_is_closed_loop; i++)
235 {
236 Elem * elem = mesh->add_elem(new Edge2);
237 elem->set_node(0) = nodes[i];
238 elem->set_node(1) = nodes[(i + 1) % nodes.size()];
239 elem->subdomain_id() = 1;
240 }
241
242 if (_node_set_boundaries.size())
243 {
244 // Add boundary nodesets to boundary info
245 BoundaryInfo & boundary_info = mesh->get_boundary_info();
246 int i = 0;
247 for (auto & side_name : _node_set_boundaries)
248 boundary_info.nodeset_name(i++) = side_name;
249
250 boundary_info.add_node(*nodes.begin(), boundary_info.get_id_by_name(_node_set_boundaries[0]));
251 if (_node_set_boundaries.size() > 1)
252 boundary_info.add_node(*(nodes.end() - 1),
253 boundary_info.get_id_by_name(_node_set_boundaries[1]));
254 }
255
256 return dynamic_pointer_cast<MeshBase>(mesh);
257}

259void
260ParsedCurveGenerator::tSectionSpaceDefiner(const Real t_start,
261 const Real t_end,
262 std::vector<Real> & t_sect_space,
263 std::vector<Real> & dis_sect_space,
264 unsigned int num_segments,
265 const bool is_closed_loop,
266 const Real oversample_factor)
267{
268 std::vector<Point> pt_sect_space;
269 t_sect_space.push_back(t_start);
270 t_sect_space.push_back(t_end);
271 pt_sect_space.push_back(pointCalculator(t_start));
272 pt_sect_space.push_back(pointCalculator(t_end));
273 const Real total_distance =
274 is_closed_loop
275 ? euclideanDistance(pt_sect_space.front(), pointCalculator((t_start + t_end) / 2.0))
276 : euclideanDistance(pt_sect_space.front(), pt_sect_space.back());
277 if (MooseUtils::absoluteFuzzyEqual(total_distance, 0.0))
278 paramError("section_bounding_t_values",
279 "The curve has at least one cross, which is not supported.");
280 const Real target_fine_interval = total_distance / (Real)num_segments / oversample_factor;
281 unsigned int t_space_id = 0;
282 dis_sect_space.push_back(0.0);
283 // Use a maximum oversampling points number and a counter to avert infinite loop
284 const unsigned int max_binary_search = _max_oversample_number_factor * (num_segments + 1);
285 unsigned int binary_search_counter(0);
286 // Use binary algorithm to make all the intervals smaller than the target value
287 while (t_space_id + 1 < t_sect_space.size() && binary_search_counter <= max_binary_search)
288 {
289 binary_search_counter++;
290 if (euclideanDistance(pt_sect_space[t_space_id], pt_sect_space[t_space_id + 1]) >
291 target_fine_interval)
292 {
293 const Real new_t_value = (t_sect_space[t_space_id] + t_sect_space[t_space_id + 1]) / 2.0;
294 t_sect_space.insert(t_sect_space.begin() + t_space_id + 1, new_t_value);
295 pt_sect_space.insert(pt_sect_space.begin() + t_space_id + 1, pointCalculator(new_t_value));
296 }
297 else
298 {
299 dis_sect_space.push_back(
300 dis_sect_space.back() +
301 euclideanDistance(pt_sect_space[t_space_id], pt_sect_space[t_space_id + 1]));
302 t_space_id++;
303 }
304 }
305 if (binary_search_counter > max_binary_search)
306 paramError(
307 "max_oversample_number_factor",
308 "Maximum oversampling points number has been exceeded. Please consider adding more t "
309 "values into 'section_bounding_t_values' or increase 'max_oversample_number_factor'.");
310 const Real total_dis_seg = dis_sect_space.back();
311 // Normalization to make the normalized length of the curve equals to number of segments
312 for (auto & dist : dis_sect_space)
313 dist = dist / total_dis_seg * (Real)num_segments;
314}
