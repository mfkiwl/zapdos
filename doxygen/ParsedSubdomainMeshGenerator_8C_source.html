56ParsedSubdomainMeshGenerator::ParsedSubdomainMeshGenerator(const InputParameters & parameters)
57 : MeshGenerator(parameters),
58 FunctionParserUtils<false>(parameters),
59 _input(getMesh("input")),
60 _function(parameters.get<std::string>("combinatorial_geometry")),
61 _block_id(parameters.get<SubdomainID>("block_id")),
62 _excluded_ids(isParamValid("excluded_subdomain_ids")
63 ? parameters.get<std::vector<subdomain_id_type>>("excluded_subdomain_ids")
64 : std::vector<subdomain_id_type>())
65{
66 // base function object
67 _func_F = std::make_shared<SymFunction>();
68
69 // set FParser internal feature flags
70 setParserFeatureFlags(_func_F);
71
72 // add the constant expressions
73 addFParserConstants(_func_F,
74 getParam<std::vector<std::string>>("constant_names"),
75 getParam<std::vector<std::string>>("constant_expressions"));
76
77 // parse function
78 if (_func_F->Parse(_function, "x,y,z") >= 0)
79 mooseError("Invalid function\n",
80 _function,
81 "\nin ParsedSubdomainMeshGenerator ",
82 name(),
83 ".\n",
84 _func_F->ErrorMsg());
85
86 _func_params.resize(3);
87}

89std::unique_ptr<MeshBase>
90ParsedSubdomainMeshGenerator::generate()
91{
92 std::unique_ptr<MeshBase> mesh = std::move(_input);
93
94 if (isParamValid("excluded_subdomains"))
95 {
96 auto excluded_subdomains = parameters().get<std::vector<SubdomainName>>("excluded_subdomains");
97
98 // check that the subdomains exist in the mesh
99 for (const auto & name : excluded_subdomains)
100 if (!MooseMeshUtils::hasSubdomainName(*mesh, name))
101 paramError("excluded_subdomains", "The block '", name, "' was not found in the mesh");
102
103 _excluded_ids = MooseMeshUtils::getSubdomainIDs(*mesh, excluded_subdomains);
104 }
105
106 // Loop over the elements
107 for (const auto & elem : mesh->active_element_ptr_range())
108 {
109 _func_params[0] = elem->vertex_average()(0);
110 _func_params[1] = elem->vertex_average()(1);
111 _func_params[2] = elem->vertex_average()(2);
112 bool contains = evaluate(_func_F);
113
114 if (contains && std::find(_excluded_ids.begin(), _excluded_ids.end(), elem->subdomain_id()) ==
115 _excluded_ids.end())
116 elem->subdomain_id() = _block_id;
117 }
118
119 // Assign block name, if provided
120 if (isParamValid("block_name"))
121 mesh->subdomain_name(_block_id) = getParam<SubdomainName>("block_name");
122
123 mesh->set_isnt_prepared();
124 return dynamic_pointer_cast<MeshBase>(mesh);
125}
