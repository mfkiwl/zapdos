59template <bool is_ad>
60Real
61PowerDepTempl<is_ad>::computeValue()
62{
63 _current =
64 _sgn[_qp] * 1.6e-19 * 6.02e23 *
65 (_sgn[_qp] * raw_value(_mu[_qp]) * -_grad_potential[_qp] * _r_units *
66 std::exp(_density_log[_qp]) -
67 raw_value(_diff[_qp]) * std::exp(_density_log[_qp]) * _grad_density_log[_qp] * _r_units);
68
69 if (_art_diff)
70 {
71 Real vd_mag = raw_value(_mu[_qp]) * _grad_potential[_qp].norm() * _r_units;
72 Real delta = vd_mag * _current_elem->hmax() / 2.;
73 _current += _sgn[_qp] * 1.6e-19 * 6.02e23 * -delta * std::exp(_density_log[_qp]) *
74 _grad_density_log[_qp] * _r_units;
75 }
76
77 return _current * -_grad_potential[_qp] * _r_units * _voltage_scaling;
78}
