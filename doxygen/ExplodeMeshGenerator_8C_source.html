38std::unique_ptr<MeshBase>
39ExplodeMeshGenerator::generate()
40{
41 std::unique_ptr<MeshBase> mesh = std::move(_input);
42
43 // check that the subdomain IDs exist in the mesh
44 for (const auto & id : _subdomains)
45 if (!MooseMeshUtils::hasSubdomainID(*mesh, id))
46 paramError("subdomains", "The block ID '", id, "' was not found in the mesh");
47
48 BoundaryInfo & boundary_info = mesh->get_boundary_info();
49 if (boundary_info.get_id_by_name(_interface_name) != Moose::INVALID_BOUNDARY_ID)
50 paramError("interface_name", "The specified interface name already exits in the mesh.");
51
52 const auto node_to_elem_map = buildSubdomainRestrictedNodeToElemMap(mesh, _subdomains);
53
54 duplicateNodes(mesh, node_to_elem_map);
55
56 createInterface(*mesh, node_to_elem_map);
57
58 Partitioner::set_node_processor_ids(*mesh);
59
60 return dynamic_pointer_cast<MeshBase>(mesh);
61}

63ExplodeMeshGenerator::NodeToElemMapType
64ExplodeMeshGenerator::buildSubdomainRestrictedNodeToElemMap(
65 std::unique_ptr<MeshBase> & mesh, const std::vector<SubdomainID> & subdomains) const
66{
67 NodeToElemMapType node_to_elem_map;
68 for (const auto & elem : mesh->active_element_ptr_range())
69 {
70 // Skip if the element is not in the specified subdomains
71 if (std::find(subdomains.begin(), subdomains.end(), elem->subdomain_id()) == subdomains.end())
72 continue;
73
74 std::set<const Elem *> neighbors;
75 elem->find_point_neighbors(neighbors);
76
77 for (auto n : make_range(elem->n_nodes()))
78 {
79 // if ANY neighboring element that contains this node is not in the specified subdomains,
80 // don't add this node to the map, i.e. don't split this node.
81 bool should_duplicate = true;
82 for (auto neighbor : neighbors)
83 if (neighbor->contains_point(elem->node_ref(n)) &&
84 std::find(subdomains.begin(), subdomains.end(), neighbor->subdomain_id()) ==
85 subdomains.end())
86 {
87 should_duplicate = false;
88 break;
89 }
90
91 if (should_duplicate)
92 node_to_elem_map[elem->node_id(n)].insert(elem->id());
93 }
94 }
95
96 return node_to_elem_map;
97}

100ExplodeMeshGenerator::duplicateNodes(std::unique_ptr<MeshBase> & mesh,
101 const NodeToElemMapType & node_to_elem_map) const
102{
103 for (const auto & [node_id, connected_elem_ids] : node_to_elem_map)
104 for (auto & connected_elem_id : connected_elem_ids)
105 if (connected_elem_id != *connected_elem_ids.begin())
106 duplicateNode(mesh, mesh->elem_ptr(connected_elem_id), mesh->node_ptr(node_id));

109void
110ExplodeMeshGenerator::duplicateNode(std::unique_ptr<MeshBase> & mesh,
111 Elem * elem,
112 const Node * node) const
113{
114 std::unique_ptr<Node> new_node = Node::build(*node, Node::invalid_id);
115 new_node->processor_id() = elem->processor_id();
116 Node * added_node = mesh->add_node(std::move(new_node));
117 for (const auto j : elem->node_index_range())
118 if (elem->node_id(j) == node->id())
119 {
120 elem->set_node(j) = added_node;
121 break;
122 }
123
124 // Add boundary info to the new node
125 BoundaryInfo & boundary_info = mesh->get_boundary_info();
126 std::vector<boundary_id_type> node_boundary_ids;
127 boundary_info.boundary_ids(node, node_boundary_ids);
128 boundary_info.add_node(added_node, node_boundary_ids);
129}

131void
132ExplodeMeshGenerator::createInterface(MeshBase & mesh,
133 const NodeToElemMapType & node_to_elem_map) const
134{
135 BoundaryInfo & boundary_info = mesh.get_boundary_info();
136 const auto & existing_boundary_ids = boundary_info.get_boundary_ids();
137 boundary_id_type interface_id =
138 existing_boundary_ids.empty() ? 0 : *existing_boundary_ids.rbegin() + 1;
139 boundary_info.sideset_name(interface_id) = _interface_name;
140
141 std::set<std::pair<dof_id_type, unsigned int>> sides_to_add;
142
143 for (const auto & node_to_elems : node_to_elem_map)
144 for (const auto & elem_id_i : node_to_elems.second)
145 {
146 Elem * elem_i = mesh.elem_ptr(elem_id_i);
147 for (const auto & elem_id_j : node_to_elems.second)
148 {
149 Elem * elem_j = mesh.elem_ptr(elem_id_j);
150 if (elem_i != elem_j && elem_id_i < elem_id_j && elem_i->has_neighbor(elem_j))
151 sides_to_add.insert(std::make_pair(elem_id_i, elem_i->which_neighbor_am_i(elem_j)));
152 }
153 }
154
155 for (const auto & [elem_id, side] : sides_to_add)
156 boundary_info.add_side(elem_id, side, interface_id);
157}
