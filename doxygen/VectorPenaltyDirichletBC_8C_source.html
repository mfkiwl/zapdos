41Real
42VectorPenaltyDirichletBC::computeQpResidual()
43{
44 RealVectorValue u_exact = {_exact_x.value(_t, _q_point[_qp]),
45 _exact_y.value(_t, _q_point[_qp]),
46 _exact_z.value(_t, _q_point[_qp])};
47
48 return _penalty * _test[_i][_qp] * ((_linear ? 0. : _u[_qp]) - u_exact);
49}
