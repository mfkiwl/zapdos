96bool
97FVBoundaryCondition::hasFaceSide(const FaceInfo & fi, bool fi_elem_side) const
98{
99 const auto ft = fi.faceType(std::make_pair(_var.number(), _var.sys().number()));
100 if (fi_elem_side)
101 return ft == FaceInfo::VarFaceNeighbors::ELEM || ft == FaceInfo::VarFaceNeighbors::BOTH;
102 else
103 return ft == FaceInfo::VarFaceNeighbors::NEIGHBOR || ft == FaceInfo::VarFaceNeighbors::BOTH;
104}
