104Real
105NeumannCircuitVoltageNew::computeQpResidual()
106{
107 Real d_curr_times_resist_d_potential = _current_sign * _current * _resistance / _voltage_scaling;
108
109 if (_use_area)
110 {
111 d_curr_times_resist_d_potential *= _area;
112 }
113
114 return _test[_i][_qp] * _r_units *
115 (_V_source.value(_t, _q_point[_qp]) + _u[_qp] + d_curr_times_resist_d_potential);
116}

118Real
119NeumannCircuitVoltageNew::computeQpJacobian()
120{
121 Real d_curr_times_resist_dv =
122 _current_sign * _current_jac[_var_dofs[_j]] * _resistance / _voltage_scaling;
123
124 if (_use_area)
125 {
126 d_curr_times_resist_dv *= _area;
127 }
128
129 return _test[_i][_qp] * _r_units * (_phi[_j][_qp] + d_curr_times_resist_dv);
130}

132Real
133NeumannCircuitVoltageNew::computeQpOffDiagJacobian(unsigned int jvar)
134{
135 Real d_curr_times_resist_dv;
136
137 if (jvar == _em_id)
138 {
139 d_curr_times_resist_dv =
140 _current_sign * _current_jac[_em_dofs[_j]] * _resistance / _voltage_scaling;
141 }
142 else if (jvar == _ip_id)
143 {
144 d_curr_times_resist_dv =
145 _current_sign * _current_jac[_ip_dofs[_j]] * _resistance / _voltage_scaling;
146 }
147 else if (jvar == _mean_en_id)
148 {
149 d_curr_times_resist_dv =
150 _current_sign * _current_jac[_mean_en_dofs[_j]] * _resistance / _voltage_scaling;
151 }
152 else
153 {
154 return 0;
155 }
156
157 if (_use_area)
158 {
159 d_curr_times_resist_dv *= _area;
160 }
161
162 return _test[_i][_qp] * _r_units * d_curr_times_resist_dv;
163}
