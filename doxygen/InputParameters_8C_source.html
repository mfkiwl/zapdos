103bool
104InputParameters::attemptPrintDeprecated(const std::string & name_in)
105{
106 const auto name = checkForRename(name_in);
107 if (_show_deprecated_message)
108 {
109 auto emit_deprecation_message =
110 [this](const auto & deprecated_name, const auto & deprecation_message)
111 {
112 // This is user-facing, no need for a backtrace
113 const auto current_show_trace = Moose::show_trace;
114 Moose::show_trace = false;
115 moose::internal::mooseDeprecatedStream(
116 Moose::out, false, true, errorPrefix(deprecated_name), ":\n", deprecation_message, "\n");
117 Moose::show_trace = current_show_trace;
118 return true;
119 };
120
121 if (_params.count(name) && !libmesh_map_find(_params, name)._deprecation_message.empty())
122 return emit_deprecation_message(name,
123 "The parameter '" + name + "' is deprecated.\n" +
124 libmesh_map_find(_params, name)._deprecation_message);
125 else if (auto it = _old_to_new_name_and_dep.find(name_in);
126 it != _old_to_new_name_and_dep.end() && !it->second.second.empty())
127 return emit_deprecation_message(name_in, it->second.second);
128 }
129 return false;
130}

138InputParameters &
139InputParameters::operator=(const InputParameters & rhs)
140{
141 // An error to help minimize the segmentation faults that occure when MooseObjects do not have the
142 // correct constructor
143 if (!rhs._allow_copy)
144 {
145 const std::string & name =
146 rhs.get<std::string>("_object_name"); // If _allow_parameter_copy is set then so is name
147 // (see InputParameterWarehouse::addInputParameters)
148 mooseError("Copying of the InputParameters object for the ",
149 name,
150 " object is not allowed.\n\nThe likely cause for this error ",
151 "is having a constructor that does not use a const reference, all constructors\nfor "
152 "MooseObject based classes should be as follows:\n\n",
153 " MyObject::MyObject(const InputParameters & parameters);");
154 }
155
156 Parameters::operator=(rhs);
157
158 _params = rhs._params;
159
160 _buildable_types = rhs._buildable_types;
161 _buildable_rm_types = rhs._buildable_rm_types;
162 _collapse_nesting = rhs._collapse_nesting;
163 _moose_object_syntax_visibility = rhs._moose_object_syntax_visibility;
164 _coupled_vars = rhs._coupled_vars;
165 _new_to_deprecated_coupled_vars = rhs._new_to_deprecated_coupled_vars;
166 _allow_copy = rhs._allow_copy;
167 _block_fullpath = rhs._block_fullpath;
168 _block_location = rhs._block_location;
169 _old_to_new_name_and_dep = rhs._old_to_new_name_and_dep;
170 _new_to_old_names = rhs._new_to_old_names;
171
172 return *this;
173}

175InputParameters &
176InputParameters::operator+=(const InputParameters & rhs)
177{
178 Parameters::operator+=(rhs);
179
180 // TODO: this is not a proper merge - if a particular parameter exists in both this and rhs,
181 // then we should actually smartly merge both metadata structs before storing in this.
182 for (auto it = rhs._params.begin(); it != rhs._params.end(); ++it)
183 _params[it->first] = it->second;
184
185 _buildable_types.insert(
186 _buildable_types.end(), rhs._buildable_types.begin(), rhs._buildable_types.end());
187 _buildable_rm_types.insert(
188 _buildable_rm_types.end(), rhs._buildable_rm_types.begin(), rhs._buildable_rm_types.end());
189
190 // Collapse nesting and moose object syntax hiding are not modified with +=

192 _new_to_deprecated_coupled_vars.insert(rhs._new_to_deprecated_coupled_vars.begin(),
193 rhs._new_to_deprecated_coupled_vars.end());

196 rhs._old_to_new_name_and_dep.end());
197 _new_to_old_names.insert(rhs._new_to_old_names.begin(), rhs._new_to_old_names.end());
198 return *this;

258void
259InputParameters::addDeprecatedCoupledVar(const std::string & old_name,
260 const std::string & new_name,
261 const std::string & removal_date /*=""*/)
262{
263 mooseDeprecated("Please use 'deprecateCoupledVar'");
264
265 _show_deprecated_message = false;
266
267 // Set the doc string if we are adding the deprecated var after the new var has already been added
268 auto params_it = _params.find(new_name);
269 std::string doc_string;
270 if (params_it != _params.end())
271 doc_string = params_it->second._doc_string;
272
273 addParam<std::vector<VariableName>>(old_name, doc_string);
274 _coupled_vars.insert(old_name);
275 _new_to_deprecated_coupled_vars.emplace(new_name, old_name);
276
277 std::string deprecation_message =
278 "The coupled variable parameter '" + old_name + "' has been deprecated";
279 if (!removal_date.empty())
280 deprecation_message += " and will be removed " + removal_date;
281 deprecation_message += ". Please use the '" + new_name + "' coupled variable parameter instead.";
282 _params[old_name]._deprecation_message = deprecation_message;
283 _show_deprecated_message = true;
284}

368InputParameters::isParamValid(const std::string & name_in) const
369{
370 const auto name = checkForRename(name_in);
371 if (have_parameter<MooseEnum>(name))
372 return get<MooseEnum>(name).isValid();
373 else if (have_parameter<std::vector<MooseEnum>>(name))
374 {
375 for (auto it = get<std::vector<MooseEnum>>(name).begin();
376 it != get<std::vector<MooseEnum>>(name).end();
377 ++it)
378 if (!it->isValid())
379 return false;
380 return true;
381 }
382 else if (have_parameter<MultiMooseEnum>(name))
383 return get<MultiMooseEnum>(name).isValid();
384 else if (have_parameter<ExecFlagEnum>(name))
385 return get<ExecFlagEnum>(name).isValid();
386 else
387 return _params.count(name) > 0 && _params.at(name)._valid;

552void
553InputParameters::checkParams(const std::string & parsing_syntax)
554{
555 std::string parampath = blockFullpath() != "" ? blockFullpath() : parsing_syntax;
556 std::ostringstream oss;
557 // Required parameters
558 for (const auto & it : *this)
559 {
560 const auto param_name = checkForRename(it.first);
561 if (!isParamValid(param_name) && isParamRequired(param_name))
562 {
563 // check if an old, deprecated name exists for this parameter that may be specified
564 auto oit = _new_to_deprecated_coupled_vars.find(param_name);
565 if (oit != _new_to_deprecated_coupled_vars.end() && isParamValid(oit->second))
566 continue;
567
568 oss << blockLocation() << ": missing required parameter '" << parampath + "/" + param_name
569 << "'\n";
570 oss << "\tDoc String: \"" + getDocString(param_name) + "\"" << std::endl;
571 }
572 }
573
574 // Range checked parameters
575 for (const auto & it : *this)
576 {
577 std::string long_name(parampath + "/" + it.first);
578
579 dynamicCastRangeCheck(Real, Real, long_name, it.first, it.second, oss);
580 dynamicCastRangeCheck(int, long, long_name, it.first, it.second, oss);
581 dynamicCastRangeCheck(long, long, long_name, it.first, it.second, oss);
582 dynamicCastRangeCheck(unsigned int, long, long_name, it.first, it.second, oss);
583 }
584
585 // Controllable parameters
586 for (const auto & param_name : getControllableParameters())
587 {
588 if (isPrivate(param_name))
589 oss << inputLocation(param_name) << ": private parameter '" << paramFullpath(param_name)
590 << "' marked controllable";
591
592 checkMooseType(NonlinearVariableName, param_name);
593 checkMooseType(AuxVariableName, param_name);
594 checkMooseType(VariableName, param_name);
595 checkMooseType(BoundaryName, param_name);
596 checkMooseType(SubdomainName, param_name);
597 checkMooseType(PostprocessorName, param_name);
598 checkMooseType(VectorPostprocessorName, param_name);
599 checkMooseType(UserObjectName, param_name);
600 checkMooseType(MaterialPropertyName, param_name);
601 }
602
603 if (!oss.str().empty())
604 mooseError(oss.str());
605}

697std::string
698InputParameters::type(const std::string & name_in) const
699{
700 const auto name = checkForRename(name_in);
701 if (!_values.count(name))
702 mooseError("Parameter \"", name, "\" not found.\n\n", *this);
703
704 if (_coupled_vars.find(name) != _coupled_vars.end())

818void
819InputParameters::applyParameters(const InputParameters & common,
820 const std::vector<std::string> exclude)
821{
822 // Loop through the common parameters
823 for (const auto & it : common)
824 {
825 // Common parameter name
826 const std::string & common_name = it.first;
827 // Continue to next parameter, if the current is in list of excluded parameters
828 if (std::find(exclude.begin(), exclude.end(), common_name) != exclude.end())
829 continue;
830
831 applyParameter(common, common_name);
832 }
833
834 // Loop through the coupled variables
835 for (std::set<std::string>::const_iterator it = common.coupledVarsBegin();
836 it != common.coupledVarsEnd();
837 ++it)
838 {
839 // Variable name
840 const std::string var_name = *it;
841
842 // Continue to next variable, if the current is in list of excluded parameters
843 if (std::find(exclude.begin(), exclude.end(), var_name) != exclude.end())
844 continue;
845
846 applyCoupledVar(common, var_name);
847 }
848}

850void
851InputParameters::applySpecificParameters(const InputParameters & common,
852 const std::vector<std::string> & include,
853 bool allow_private)
854{
855 // Loop through the common parameters
856 for (const auto & it : common)
857 {
858
859 // Common parameter name
860 const std::string & common_name = it.first;
861
862 // Continue to next parameter, if the current is not in list of included parameters
863 if (std::find(include.begin(), include.end(), common_name) == include.end())
864 continue;
865
866 applyParameter(common, common_name, allow_private);
867 }
868
869 // Loop through the coupled variables
870 for (std::set<std::string>::const_iterator it = common.coupledVarsBegin();
871 it != common.coupledVarsEnd();
872 ++it)
873 {
874 // Variable name
875 const std::string var_name = *it;
876
877 // Continue to next variable, if the current is not in list of included parameters
878 if (std::find(include.begin(), include.end(), var_name) == include.end())
879 continue;
880
881 applyCoupledVar(common, var_name);
882 }

885void
886InputParameters::applyCoupledVar(const InputParameters & common, const std::string & var_name)
887{
888 // Disable the display of deprecated message when applying common parameters, this avoids a dump
889 // of messages
890 _show_deprecated_message = false;
891
892 // If the local parameters has a coupled variable, populate it with the value from the common
893 // parameters, if the common parameters has the coupled variable too
894 if (hasCoupledValue(var_name))
895 {
896 if (common.hasDefaultCoupledValue(var_name))
897 {
898 // prepare a vector of default coupled values
899 std::vector<Real> defaults(common.numberDefaultCoupledValues(var_name));
900 for (unsigned int j = 0; j < common.numberDefaultCoupledValues(var_name); ++j)
901 defaults[j] = common.defaultCoupledValue(var_name, j);
902 addCoupledVar(var_name, defaults, common.getDocString(var_name));
903 }
904 else if (common.hasCoupledValue(var_name))
905 addCoupledVar(var_name, common.getDocString(var_name));
906 }
907
908 // Enable deprecated message printing
909 _show_deprecated_message = true;
910}

912void
913InputParameters::applyParameter(const InputParameters & common,
914 const std::string & common_name,
915 bool allow_private)
916{
917 // Disable the display of deprecated message when applying common parameters, this avoids a dump
918 // of messages
919 _show_deprecated_message = false;
920
921 const auto local_name = checkForRename(common_name);
922
923 // Extract the properties from the local parameter for the current common parameter name
924 const bool local_exist = _values.find(local_name) != _values.end();
925 const bool local_set = _params.count(local_name) > 0 && !_params[local_name]._set_by_add_param;
926 const bool local_priv = allow_private ? false : isPrivate(local_name);
927 const bool local_valid = isParamValid(local_name);
928
929 // Extract the properties from the common parameter
930 const bool common_exist = common._values.find(common_name) != common._values.end();
931 const bool common_priv = allow_private ? false : common.isPrivate(common_name);
932 const bool common_valid = common.isParamValid(common_name);
933
934 /* In order to apply common parameter 4 statements must be satisfied
935 * (1) A local parameter must exist with the same name as common parameter
936 * (2) Common parameter must valid and exist
937 * (3) Local parameter must be invalid OR not have been set from its default
938 * (4) Both cannot be private
939 */
940 if (local_exist && common_exist && common_valid && (!local_valid || !local_set) &&
941 (!common_priv || !local_priv))
942 {
943 remove(local_name);
944 _values[local_name] = common._values.find(common_name)->second->clone();
945 set_attributes(local_name, false);
946 _params[local_name]._set_by_add_param =
947 libmesh_map_find(common._params, common_name)._set_by_add_param;
948 }
949
950 // Enable deprecated message printing
951 _show_deprecated_message = true;
952}
