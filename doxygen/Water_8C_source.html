200void
201Water::computeQpProperties()
202{
203 _cw[_qp] = 56;
204 _electron_mult[_qp] = _user_electron_mult;
205 _potential_mult[_qp] = _user_potential_mult;
206 _N_A[_qp] = 6.02e23;
207 _eps_r[_qp] = 80.;
208 // _eps_r[_qp] = 1.;
209 _eps_0[_qp] = 8.85e-12;
210 _eps[_qp] = _eps_r[_qp] * _eps_0[_qp];
211 _e[_qp] = 1.6e-19; // coulombic charge
212 _k[_qp] = 1.38e-23; // Boltzmanns constant
213 _T[_qp] = 300; // Simulation temperature
214 _kemliq[_qp] = 1.9e1 * _cw[_qp]; // e + H2O-->H + OH-
215 _kem[_qp] = _kemliq[_qp];
216 // _kemliqemliq[_qp] = 6e11 * _cw[_qp] * _cw[_qp] / (_N_A[_qp] * 1000.); // e + H2Op-->H + OH
217 _kemliqemliq[_qp] =
218 1e8 * _cw[_qp] * _cw[_qp] / 1000.; // 2e + 2H2O-->H2 + 2OH-. Now has units of m^3 / (mol * s)
219 _kemem[_qp] = _kemliqemliq[_qp];
220 _k4[_qp] = 2.5e10; // e + H + H2O-->H2 + OH-
221 _k5[_qp] = 3e10; // e + OH-->OH-
222 _k6[_qp] = 2.2e10; // e + O- + H2O --> 2OH-
223 _k7[_qp] = 2.3e10; // e + H3O+ --> H2 + H2O
224 _k8[_qp] = 1.1e10; // e + H2O2 --> OH + OH-
225 _k9[_qp] = 3.5e9; // e + HO2- + H2O --> OH + 2OH-
226 _k10[_qp] = 1.9e10; // e + O2 --> O2-
227 _k11[_qp] = 1.9e10; // e + O --> O-
228 _k12[_qp] = 1.0e1; // H + H2O --> H2 + OH
229 _k13[_qp] = 7.5e9; // 2H --> H2
230 _k14[_qp] = 7.0e9; // H + OH --> H2O
231 _k15[_qp] = 2.2e7; // H + OH- --> H2O + e
232 _k16[_qp] = 9.0e7; // H + H2O2 --> OH + H2O
233 _k17[_qp] = 6.0e6; // H2 + H2O2 --> H + OH + H2O
234 _k18[_qp] = 2.1e10; // H + O2 --> HO2
235 _k19[_qp] = 1.0e10; // H + HO2 --> H2O2
236 _k20[_qp] = 1.3e4; // O + H2O --> 2OH
237 _k21[_qp] = 3.0e9; // O + O2 --> O3
238 _k22[_qp] = 5.5e9; // 2OH --> H2O2
239 _k23[_qp] = 2.0e10; // OH + O- --> HO2-
240 _k24[_qp] = 4.2e7; // OH + H2 --> H + H2O
241 _k25[_qp] = 1.3e10; // OH + OH- --> O- + H2O
242 _k26[_qp] = 6.0e9; // OH + HO2 --> H2O + O2
243 _k27[_qp] = 8.0e9; // OH + O2- --> OH- + O2
244 _k28[_qp] = 1.8e6; // ] O- + H2O --> OH- + OH
245 _k29[_qp] = 8.0e7; // O- + H2 --> OH- + H
246 _k30[_qp] = 5.0e8; // O- + H2O2 --> O2- + H2O
247 _k31[_qp] = 4.0e8; // O- + HO2- --> O2- + OH-
248 _k32[_qp] = 3.6e9; // O- + O2- --> O3-
249 _k33[_qp] = 6.0e8; // O- + O2- + H2O --> 2OH- + O2
250 _k34[_qp] = 2.7e7; // OH + H2O2 --> H2O + HO2
251 _k35[_qp] = 7.5e9; // OH + HO2- --> OH- + HO2
252 _k36[_qp] = 6.0e3; // H2O+ + H2O --> H3O+ + OH
253 _k37[_qp] = 6.0e10; // H3O+ + OH- --> H + OH + H2O
254 _k38[_qp] = 2.0e3; // HO2 + H2O --> H3O+ + O2-
255 _k39[_qp] = 6.0e1; // H3O+ O2- --> HO2 + H2O
256 _diffemliq[_qp] = 4.5e-9; // diffusivity of hydrated electron. Anbar (1965) from Ranga paper
257 _diffem[_qp] = _diffemliq[_qp];
258 // _diffem[_qp] = 0.297951680159;
259 _diffpotentialliq[_qp] = _eps[_qp];
260 _diffpotential[_qp] = _diffpotentialliq[_qp];
261 // _diffemliq[_qp] = 0.297951680159/4;
262 _DH[_qp] = 5e-9; // H radical
263 _diffOHm[_qp] = 5.27e-9; // OH- ion
264 // _diffOHm[_qp] = 2.98e-3; // Temporary high diffusivity taken from gas phase argon
265 _diffClm[_qp] = _diffOHm[_qp];
266 _DH2Op[_qp] = 5e-9; // H2O+ ion
267 _DOH[_qp] = 5e-9; // OH radical
268 _DH2[_qp] = 4.5e-9; // H2 molecule
269 _DOm[_qp] = 5e-9; // O- ion
270 _diffH3Op[_qp] = 9.3e-9; // H3O+ ion
271 // _diffH3Op[_qp] = 2.98e-3;
272 _diffNap[_qp] = _diffH3Op[_qp];
273 _DH2O2[_qp] = 5e-9; // H2O2 molecule
274 _DHO2m[_qp] = 5e-9; // HO2- ion
275 _DO2[_qp] = 2e-9; // O2 molecule
276 _DO2m[_qp] = 5e-9; // O2- ion
277 _DO[_qp] = 5e-9; // O radical
278 _DHO2[_qp] = 5e-9; // HO2 radical
279 _DO3[_qp] = 5e-9; // O3 molecule
280 _DO3m[_qp] = 5e-9; // O3- ion
281 _zem[_qp] = -1; // charge of hydrated electron
282 _zH[_qp] = 0; // H radical
283 _zOHm[_qp] = -1; // OH- ion
284 _zH2Op[_qp] = 1; // H2O+ ion
285 _zOH[_qp] = 0; // OH radical
286 _zH2[_qp] = 0; // H2 molecule
287 _zOm[_qp] = -1; // O- ion
288 _zH3Op[_qp] = 1; // H3O+ ion
289 _zH2O2[_qp] = 0; // H2O2 molecule
290 _zHO2m[_qp] = -1; // HO2- ion
291 _zO2[_qp] = 0; // O2 molecule
292 _zO2m[_qp] = -1; // O2- ion
293 _zO[_qp] = 0; // O radical
294 _zHO2[_qp] = 0; // HO2 radical
295 _zO3[_qp] = 0; // O3 molecule
296 _zO3m[_qp] = -1; // O3- ion
297 _sgnemliq[_qp] = -1;
298 _sgnem[_qp] = _sgnemliq[_qp];
299 _sgnOHm[_qp] = -1;
300 _sgnH3Op[_qp] = 1;
301 _sgnClm[_qp] = -1;
302 _sgnNap[_qp] = 1;
303 _muemliq[_qp] =
304 _e[_qp] * _diffemliq[_qp] / _k[_qp] / _T[_qp] * 1000.; // mobility of hydrated electron
305 _muem[_qp] = _muemliq[_qp];
306 // _muem[_qp] = 0.0352103411399 * 1000.; // units of m^2/(kV*s)
307 // _muemliq[_qp] = 0.0352103411399/4;
308 _muH[_qp] = _zH[_qp] * _e[_qp] * _DH[_qp] / _k[_qp] / _T[_qp]; // H radical
309 _muOHm[_qp] = _e[_qp] * _diffOHm[_qp] / _k[_qp] / _T[_qp] * 1000.; // OH- ion
310 _muClm[_qp] = _muOHm[_qp];
311 // _muOHm[_qp] = 3.52e-4;
312 _muH2Op[_qp] = _zH2Op[_qp] * _e[_qp] * _DH2Op[_qp] / _k[_qp] / _T[_qp]; // H2O+ ion
313 _muOH[_qp] = _zOH[_qp] * _e[_qp] * _DOH[_qp] / _k[_qp] / _T[_qp]; // OH radical
314 _muH2[_qp] = _zH2[_qp] * _e[_qp] * _DH2[_qp] / _k[_qp] / _T[_qp]; // H2 molecule
315 _muOm[_qp] = _zOm[_qp] * _e[_qp] * _DOm[_qp] / _k[_qp] / _T[_qp]; // O- ion
316 _muH3Op[_qp] = _e[_qp] * _diffH3Op[_qp] / _k[_qp] / _T[_qp] * 1000.; // H3O+ ion
317 _muNap[_qp] = _muH3Op[_qp];
318 // _muH3Op[_qp] = 3.52e-4;
319 _muH2O2[_qp] = _zH2O2[_qp] * _e[_qp] * _DH2O2[_qp] / _k[_qp] / _T[_qp]; // H2O2 molecule
320 _muHO2m[_qp] = _zHO2m[_qp] * _e[_qp] * _DHO2m[_qp] / _k[_qp] / _T[_qp]; // HO2- ion
321 _muO2[_qp] = _zO2[_qp] * _e[_qp] * _DO2[_qp] / _k[_qp] / _T[_qp]; // O2 molecule
322 _muO2m[_qp] = _zO2m[_qp] * _e[_qp] * _DO2m[_qp] / _k[_qp] / _T[_qp]; // O2- ion
323 _muO[_qp] = _zO[_qp] * _e[_qp] * _DO[_qp] / _k[_qp] / _T[_qp]; // O radical
324 _muHO2[_qp] = _zHO2[_qp] * _e[_qp] * _DHO2[_qp] / _k[_qp] / _T[_qp]; // HO2 radical
325 _muO3[_qp] = _zO3[_qp] * _e[_qp] * _DO3[_qp] / _k[_qp] / _T[_qp]; // O3 molecule
326 _muO3m[_qp] = _zO3m[_qp] * _e[_qp] * _DO3m[_qp] / _k[_qp] / _T[_qp]; // O3- ion
327 _Dunity[_qp] = 1.0;
328 _muunity[_qp] = 1.0;
329 _munegunity[_qp] = -1.0;
330
331 // Material version of aux variables
332
333 _EField[_qp] = -_grad_potential[_qp](0);
334 _OHm_lin[_qp] = std::exp(_OHm[_qp]);
335 _H3Op_lin[_qp] = std::exp(_H3Op[_qp]);
