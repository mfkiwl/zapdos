319template <typename T>
320ReporterState<T> &
321ReporterData::getReporterStateHelper(const ReporterName & reporter_name,
322 bool declare,
323 const MooseObject * moose_object /* = nullptr */) const
324{
325 if (hasReporterState(reporter_name))
326 {
327 const auto error_helper =
328 [this, &reporter_name, &moose_object, &declare](const std::string & suffix)
329 {
330 std::stringstream oss;
331 oss << "While " << (declare ? "declaring" : "requesting") << " a "
332 << reporter_name.specialTypeToName() << " value with the name \""
333 << reporter_name.getValueName() << "\"";
334 if (!reporter_name.isPostprocessor() && !reporter_name.isVectorPostprocessor())
335 oss << " and type \"" << MooseUtils::prettyCppType<T>() << "\"";
336 oss << ",\na Reporter with the same name " << suffix << ".\n\n";
337 oss << getReporterInfo(reporter_name);
338
339 if (moose_object)
340 moose_object->mooseError(oss.str());
341 else
342 mooseError(oss.str());
343 };
344
345 if (declare && hasReporterValue(reporter_name))
346 error_helper("has already been declared");
347 if (!hasReporterState<T>(reporter_name))
348 {
349 std::stringstream oss;
350 oss << "has been " << (declare || !hasReporterValue(reporter_name) ? "requested" : "declared")
351 << " with a different type";
352 error_helper(oss.str());
353 }
354 }
355
356 // Reporter states are stored as restartable data. The act of registering restartable data
357 // may be done multiple times with the same name, which will happen when more than one
358 // get value is done, or a get value and a declare is done. With this, we create a new
359 // state every time, but said created state may not be the actual state if this state
360 // is already registered as restartable data. Therefore, we create a state, and then
361 // cast the restartable data received back to a state (which may be different than
362 // the one we created, but that's okay)
363 auto state_unique_ptr = std::make_unique<ReporterState<T>>(reporter_name);
364 auto & restartable_value = getRestartableDataHelper(std::move(state_unique_ptr), declare);
365
366 auto * state = dynamic_cast<ReporterState<T> *>(&restartable_value);
367 mooseAssert(state, "Cast failed. The check above must be broken!");
368
369 // See declareReporterValue for a comment on what happens if a state for the same
370 // name is requested but with different special types. TLDR: ReporterNames with
371 // different special types are not unique so they'll be the same entry
372 _states.emplace(reporter_name, state);
373
374 return *state;
375}

389template <typename T, typename S, typename... Args>
390T &
391ReporterData::declareReporterValue(const ReporterName & reporter_name,
392 const ReporterMode & mode,
393 const MooseObject & producer,
394 Args &&... args)
395{
396 // Get/create the ReporterState
397 auto & state = getReporterStateHelper<T>(reporter_name, /* declare = */ true, &producer);
398
399 // They key in _states (ReporterName) is not unique by special type. This is done on purpose
400 // because we want to store reporter names a single name regardless of special type.
401 // Because of this, we have the case where someone could request a reporter value
402 // that is later declared as a pp or a vpp value. In this case, when it is first
403 // requested, the _state entry will have a key and name with a special type of ANY.
404 // When it's declared here (later), we will still find the correct entry because
405 // we don't check the special type in the key/name. But... we want the actual
406 // key and name to represent a pp or a vpp. Therefore, we'll set it properly,
407 // remove the entry in the map (which has the ANY key), and re-add it so that it
408 // has the pp/vpp key. This allows us to identify Reporters that really represent
409 // pps/vpps in output and in error reporting.
410 if (reporter_name.isPostprocessor() && !state.getReporterName().isPostprocessor())
411 {
412 state.setIsPostprocessor();
413 _states.erase(reporter_name);
414 _states.emplace(reporter_name, &state);
415 }
416 else if (reporter_name.isVectorPostprocessor() &&
417 !state.getReporterName().isVectorPostprocessor())
418 {
419 state.setIsVectorPostprocessor();
420 _states.erase(reporter_name);
421 _states.emplace(reporter_name, &state);
422 }
423
424 mooseAssert(!_context_ptrs.count(reporter_name), "Context already exists");
425
426 // Create the ReporterContext
427 auto context_ptr = std::make_unique<S>(_app, producer, state, args...);
428 context_ptr->init(mode); // initialize the mode, see ContextReporter
429 _context_ptrs.emplace(reporter_name, std::move(context_ptr));
430
431 return state.value();
432}

452template <typename T>
453const T &
454ReporterData::getReporterValue(const ReporterName & reporter_name,
455 const std::size_t time_index) const
456{
457 if (!hasReporterValue<T>(reporter_name))
458 mooseError("Reporter name \"",
459 reporter_name,
460 "\" with type \"",
461 MooseUtils::prettyCppType<T>(),
462 "\" is not declared.");
463
464 // Force the const version of value, which does not allow for increasing time index
465 return static_cast<const ReporterState<T> &>(
466 getReporterStateHelper<T>(reporter_name, /* declare = */ false))
467 .value(time_index);
468}
