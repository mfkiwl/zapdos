76void
77MaterialOutputAction::act()
78{
79 mooseAssert(_problem,
80 "FEProblemBase pointer is nullptr, it is needed for auto material property output");
81
82 // Do nothing if the application does not have output
83 if (!_app.actionWarehouse().hasActions("add_output"))
84 return;
85
86 bool get_names_only = _current_task == "add_output_aux_variables" ? true : false;
87
88 // Set the pointers to the MaterialData objects (Note, these pointers are not available at
89 // construction)
90 _block_material_data = &_problem->getMaterialData(Moose::BLOCK_MATERIAL_DATA);
91 _boundary_material_data = &_problem->getMaterialData(Moose::BOUNDARY_MATERIAL_DATA);
92
93 // A complete list of all MaterialBase objects
94 const auto & material_ptrs = _problem->getMaterialWarehouse().getObjects();
95
96 // Handle setting of material property output in [Outputs] sub-blocks
97 // Output objects can enable material property output, the following code examines the parameters
98 // for each Output object and sets a flag if any Output object has output set and also builds a
99 // list if the
100 // properties are limited via the 'show_material_properties' parameters
101 bool outputs_has_properties = false;
102 std::set<std::string> output_object_properties;
103
104 const auto & output_actions = _app.actionWarehouse().getActionListByName("add_output");
105 for (const auto & act : output_actions)
106 {
107 // Extract the Output action
108 AddOutputAction * action = dynamic_cast<AddOutputAction *>(act);
109 if (!action)
110 continue;
111
112 // Add the material property names from the output object parameters to the list of properties
113 // to output
114 InputParameters & params = action->getObjectParams();
115 if (params.isParamValid("output_material_properties") &&
116 params.get<bool>("output_material_properties"))
117 {
118 outputs_has_properties = true;
119 std::vector<std::string> prop_names =
120 params.get<std::vector<std::string>>("show_material_properties");
121 output_object_properties.insert(prop_names.begin(), prop_names.end());
122 }
123 }
124
125 // Loop through each material object
126 std::set<std::string> material_names;
127 std::set<std::string> unsupported_names;
128 for (const auto & mat : material_ptrs)
129 {
130 // Extract the names of the output objects to which the material properties will be exported
131 std::set<OutputName> outputs = mat->getOutputs();
132
133 // Extract the property names that will actually be output
134 std::vector<std::string> output_properties =
135 mat->getParam<std::vector<std::string>>("output_properties");
136
137 // Append the properties listed in the Outputs block
138 if (outputs_has_properties)
139 output_properties.insert(output_properties.end(),
140 output_object_properties.begin(),
141 output_object_properties.end());
142
143 // Clear the list of variable names for the current material object, this list will be populated
144 // with all the
145 // variables names for the current material object and is needed for purposes of controlling the
146 // which output objects
147 // show the material property data
148 _material_variable_names.clear();
149
150 // Create necessary outputs for the properties if:
151 // (1) The Outputs block has material output enabled
152 // (2) If the MaterialBase object itself has set the 'outputs' parameter
153 if (outputs_has_properties || outputs.find("none") == outputs.end())
154 {
155 // Add the material property for output if the name is contained in the 'output_properties'
156 // list
157 // or if the list is empty (all properties)
158 const std::set<std::string> names = mat->getSuppliedItems();
159 for (const auto & name : names)
160 {
161 // Add the material property for output
162 if (output_properties.empty() ||
163 std::find(output_properties.begin(), output_properties.end(), name) !=
164 output_properties.end())
165 {
166 auto curr_material_names = materialOutput(name, *mat, get_names_only);
167 if (curr_material_names.size() == 0)
168 unsupported_names.insert(name);
169 material_names.insert(curr_material_names.begin(), curr_material_names.end());
170 }
171
172 // If the material object has limited outputs, store the variables associated with the
173 // output objects
174 if (!outputs.empty())
175 for (const auto & output_name : outputs)
176 _material_variable_names_map[output_name].insert(_material_variable_names.begin(),
177 _material_variable_names.end());
178 }
179 }
180 }
181 if (unsupported_names.size() > 0 && get_names_only)
182 {
183 std::ostringstream oss;
184 for (const auto & name : unsupported_names)
185 oss << "\n " << name;
186 mooseWarning("The types for total ",
187 unsupported_names.size(),
188 " material properties:",
189 oss.str(),
190 "\nare not supported for automatic output by ",
191 type(),
192 ".");
193 }
194
195 if (_current_task == "add_output_aux_variables")
196 {
197 auto params = _factory.getValidParams("MooseVariableConstMonomial");
198 // currently only elemental variables are supported for material property output
199 params.set<MooseEnum>("order") = "CONSTANT";
200 params.set<MooseEnum>("family") = "MONOMIAL";
201
202 // Create the AuxVariables
203 std::ostringstream oss;
204 for (const auto & var_name : material_names)
205 {
206 oss << "\n " << var_name;
207 if (_problem->hasVariable(var_name))
208 mooseError("The material property output " + var_name +
209 " has the same name as an existing variable, either use the material"
210 " declare_suffix parameter to disambiguate or the output_properties parameter"
211 " to restrict the material properties to output");
212 _problem->addAuxVariable("MooseVariableConstMonomial", var_name, params);
213 }
214 if (material_names.size() > 0)
215 _console << COLOR_CYAN << "The following total " << material_names.size()
216 << " aux variables:" << oss.str() << "\nare added for automatic output by " << type()
217 << "." << COLOR_DEFAULT << std::endl;
218 }
219 else
220 {
221 // When a MaterialBase object has 'output_properties' defined all other properties not listed
222 // must be added to the hide list for the output objects so that properties that are not desired
223 // do not appear.
224 for (const auto & it : _material_variable_names_map)
225 {
226 std::set<std::string> hide;
227 std::set_difference(material_names.begin(),
228 material_names.end(),
229 it.second.begin(),
230 it.second.end(),
231 std::inserter(hide, hide.begin()));
232
233 _output_warehouse.addInterfaceHideVariables(it.first, hide);
234 }
235 }
236}

238std::vector<std::string>
239MaterialOutputAction::materialOutput(const std::string & property_name,
240 const MaterialBase & material,
241 bool get_names_only)
242{
243 std::vector<std::string> names;
244 if (hasProperty<Real>(property_name))
245 names = materialOutputHelper<Real>(property_name, material, get_names_only);
246
247 else if (hasADProperty<Real>(property_name))
248 names = materialOutputHelper<ADReal>(property_name, material, get_names_only);
249
250 else if (hasProperty<RealVectorValue>(property_name))
251 names = materialOutputHelper<RealVectorValue>(property_name, material, get_names_only);
252
253 else if (hasADProperty<RealVectorValue>(property_name))
254 names = materialOutputHelper<ADRealVectorValue>(property_name, material, get_names_only);
255
256 else if (hasProperty<RealTensorValue>(property_name))
257 names = materialOutputHelper<RealTensorValue>(property_name, material, get_names_only);
258
259 else if (hasProperty<RankTwoTensor>(property_name))
260 names = materialOutputHelper<RankTwoTensor>(property_name, material, get_names_only);
261
262 else if (hasADProperty<RankTwoTensor>(property_name))
263 names = materialOutputHelper<ADRankTwoTensor>(property_name, material, get_names_only);
264
265 else if (hasProperty<RankFourTensor>(property_name))
266 names = materialOutputHelper<RankFourTensor>(property_name, material, get_names_only);
267
268 return names;
269}

271InputParameters
272MaterialOutputAction::getParams(const std::string & type,
273 const std::string & property_name,
274 const std::string & variable_name,
275 const MaterialBase & material)
276{
277 // Append the list of output variables for the current material
278 _material_variable_names.insert(variable_name);
279
280 // Set the action parameters
281 InputParameters params = _factory.getValidParams(type);
282
283 params.set<MaterialPropertyName>("property") = property_name;
284 params.set<AuxVariableName>("variable") = variable_name;
285 if (_output_only_on_timestep_end)
286 params.set<ExecFlagEnum>("execute_on") = EXEC_TIMESTEP_END;
287 else
288 params.set<ExecFlagEnum>("execute_on") = {EXEC_INITIAL, EXEC_TIMESTEP_END};
289
290 if (material.boundaryRestricted())
291 params.set<std::vector<BoundaryName>>("boundary") = material.boundaryNames();
292 else
293 params.set<std::vector<SubdomainName>>("block") = material.blocks();
294
295 return params;
296}
