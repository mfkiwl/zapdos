38FVBoundedValueConstraint::computeQpResidual()
39{
40 if (_bound_type == BoundType::LOWER_THAN && _u[_qp] > _phi0)
41 return _var(makeElemArg(_current_elem), determineState()) - _phi0;
42 else if (_bound_type == BoundType::HIGHER_THAN && _u[_qp] < _phi0)
43 return _phi0 - _var(makeElemArg(_current_elem), determineState());
44 else
45 return 0;
46}
