47SplineInterpolation::errorCheck()
48{

50 mooseError("SplineInterpolation: vectors are not the same length");
51
52 bool error = false;
53 for (unsigned i = 0; !error && i + 1 < _x.size(); ++i)
54 if (_x[i] >= _x[i + 1])

56
