29InputParameters
30MultiAppGeneralFieldNearestLocationTransfer::validParams()
31{
32 InputParameters params = MultiAppGeneralFieldTransfer::validParams();
33 params.addClassDescription(
34 "Transfers field data at the MultiApp position by finding the value at the nearest "
35 "neighbor(s) in the origin application.");
36 params.addParam<unsigned int>("num_nearest_points",
37 1,
38 "Number of nearest source (from) points will be chosen to "
39 "construct a value for the target point. All points will be "
40 "selected from the same origin mesh!");
41
42 // Nearest node is historically more an extrapolation transfer
43 params.set<Real>("extrapolation_constant") = GeneralFieldTransfer::BetterOutOfMeshValue;
44 params.suppressParameter<Real>("extrapolation_constant");
45 // We dont keep track of both point distance to app and to nearest node
46 params.set<bool>("use_nearest_app") = false;
47 params.suppressParameter<bool>("use_nearest_app");
48
49 // the default of node/centroid switching based on the variable is causing lots of mistakes and
50 // bad results
51 std::vector<MooseEnum> source_types = {
52 MooseEnum("nodes centroids variable_default", "variable_default")};
53 params.addParam<std::vector<MooseEnum>>(
54 "source_type", source_types, "Where to get the source values from for each source variable");
55
56 // choose whether to include data from multiple apps when performing nearest-position/
57 // mesh-divisions based transfers
58 params.addParam<bool>("group_subapps",
59 false,
60 "Whether to group data from subapps "
61 "when considering a nearest-position algorithm");
62
63 return params;
64}

95void
96MultiAppGeneralFieldNearestLocationTransfer::initialSetup()
97{
98 MultiAppGeneralFieldTransfer::initialSetup();
99
100 // Handle the source types ahead of time
101 const auto & source_types = getParam<std::vector<MooseEnum>>("source_type");
102 _source_is_nodes.resize(_from_var_names.size());
103 _use_zero_dof_for_value.resize(_from_var_names.size());
104 if (source_types.size() != _from_var_names.size())
105 mooseError("Not enough source types specified for this number of variables. Source types must "
106 "be specified for transfers with multiple variables");
107 for (const auto var_index : index_range(_from_var_names))
108 {
109 // Local app does not own any of the source problems
110 if (_from_problems.empty())
111 break;
112
113 // Get some info on the source variable
114 FEProblemBase & from_problem = *_from_problems[0];
115 MooseVariableFieldBase & from_var =
116 from_problem.getVariable(0,
117 _from_var_names[var_index],
118 Moose::VarKindType::VAR_ANY,
119 Moose::VarFieldType::VAR_FIELD_ANY);
120 System & from_sys = from_var.sys().system();
121 const auto & fe_type = from_sys.variable_type(from_var.number());
122
123 // Select where to get the origin values
124 if (source_types[var_index] == "nodes")
125 _source_is_nodes[var_index] = true;
126 else if (source_types[var_index] == "centroids")
127 _source_is_nodes[var_index] = false;
128 else
129 {
130 // "Nodal" variables are continuous for sure at nodes
131 if (from_var.isNodal())
132 _source_is_nodes[var_index] = true;
133 // for everyone else, we know they are continuous at centroids
134 else
135 _source_is_nodes[var_index] = false;
136 }
137
138 // Some variables can be sampled directly at their 0 dofs
139 // - lagrange at nodes on a first order mesh
140 // - anything constant and elemental obviously has the 0-dof value at the centroid (or
141 // vertex-average). However, higher order elemental, even monomial, do not hold the centroid
142 // value at dof index 0 For example: pyramid has dof 0 at the center of the base, prism has dof
143 // 0 on an edge etc
144 if ((_source_is_nodes[var_index] && fe_type.family == LAGRANGE &&
145 !from_problem.mesh().hasSecondOrderElements()) ||
146 (!_source_is_nodes[var_index] && fe_type.order == CONSTANT))
147 _use_zero_dof_for_value[var_index] = true;
148 else
149 _use_zero_dof_for_value[var_index] = false;
150
151 // Check with the source variable that it is not discontinuous at the source
152 if (_source_is_nodes[var_index])
153 if (from_var.getContinuity() == DISCONTINUOUS ||
154 from_var.getContinuity() == SIDE_DISCONTINUOUS)
155 mooseError("Source variable cannot be sampled at nodes as it is discontinuous");
156
157 // Local app does not own any of the target problems
158 if (_to_problems.empty())
159 break;
160
161 // Check with the target variable that we are not doing awful projections
162 MooseVariableFieldBase & to_var =
163 _to_problems[0]->getVariable(0,
164 _to_var_names[var_index],
165 Moose::VarKindType::VAR_ANY,
166 Moose::VarFieldType::VAR_FIELD_ANY);
167 System & to_sys = to_var.sys().system();
168 const auto & to_fe_type = to_sys.variable_type(to_var.number());
169 if (_source_is_nodes[var_index])
170 {
171 if (to_fe_type.order == CONSTANT)
172 mooseWarning(
173 "Transfer is projecting from nearest-nodes to centroids. This is likely causing "
174 "floating point indetermination in the results because multiple nodes are 'nearest' to "
175 "a centroid. Please consider using a ProjectionAux to build an elemental source "
176 "variable (for example constant monomial) before transferring");
177 }
178 else if (to_var.isNodal())
179 mooseWarning(
180 "Transfer is projecting from nearest-centroids to nodes. This is likely causing "
181 "floating point indetermination in the results because multiple centroids are "
182 "'nearest' to a node. Please consider using a ProjectionAux to build a nodal source "
183 "variable (for example linear Lagrange) before transferring");
184 }
185
186 // We need to improve the indexing if we are to allow this
187 if (!_from_mesh_divisions.empty())
188 for (const auto mesh_div : _from_mesh_divisions)
189 if (mesh_div->getNumDivisions() != _from_mesh_divisions[0]->getNumDivisions())
190 paramError("from_mesh_division",
191 "This transfer has only been implemented with a uniform number of source mesh "
192 "divisions across all source applications");
193}

205void
206MultiAppGeneralFieldNearestLocationTransfer::buildKDTrees(const unsigned int var_index)
207{
208 const auto num_sources = getNumSources();
209 const auto num_apps_per_tree = getNumAppsPerTree();
210 _local_kdtrees.resize(num_sources);
211 _local_points.resize(num_sources);
212 _local_values.resize(num_sources);
213 unsigned int max_leaf_size = 0;
214
215 // Construct a local KDTree for each source. A source can be a single app or multiple apps
216 // combined (option for nearest-position / mesh-divisions)
217 for (const auto i_source : make_range(num_sources))
218 {
219 // Nest a loop on apps in case multiple apps contribute to the same KD-Tree source
220 for (const auto app_i : make_range(num_apps_per_tree))
221 {
222 // Get the current app index
223 const auto i_from = _group_subapps ? app_i : (i_source / getNumDivisions());
224 // Current position index, if using nearest positions
225 const auto i_pos = _group_subapps ? i_source : (i_source % getNumDivisions());
226
227 // Get access to the variable and some variable information
228 FEProblemBase & from_problem = *_from_problems[i_from];
229 auto & from_mesh = from_problem.mesh(_displaced_source_mesh);
230 MooseVariableFieldBase & from_var =
231 from_problem.getVariable(0,
232 _from_var_names[var_index],
233 Moose::VarKindType::VAR_ANY,
234 Moose::VarFieldType::VAR_FIELD_ANY);
235 System & from_sys = from_var.sys().system();
236 const unsigned int from_var_num = from_sys.variable_number(getFromVarName(var_index));
237
238 // Build KDTree from nodes and nodal values
239 if (_source_is_nodes[var_index])
240 {
241 for (const auto & node : from_mesh.getMesh().local_node_ptr_range())
242 {
243 if (node->n_dofs(from_sys.number(), from_var_num) < 1)
244 continue;
245
246 if (!_from_blocks.empty() && !inBlocks(_from_blocks, from_mesh, node))
247 continue;
248
249 if (!_from_boundaries.empty() && !onBoundaries(_from_boundaries, from_mesh, node))
250 continue;
251
252 // Handle the various source mesh divisions behaviors
253 // NOTE: This could be more efficient, as instead of rejecting points in the
254 // wrong division, we could just be adding them to the tree for the right division
255 if (!_from_mesh_divisions.empty())
256 {
257 const auto tree_division_index = i_source % getNumDivisions();
258 const auto node_div_index = _from_mesh_divisions[i_from]->divisionIndex(*node);
259
260 // Spatial restriction is always active
261 if (node_div_index == MooseMeshDivision::INVALID_DIVISION_INDEX)
262 continue;
263 // We fill one tree per division index for matching subapp index or division index. We
264 // only accept source data from the division index
265 else if ((_from_mesh_division_behavior ==
266 MeshDivisionTransferUse::MATCH_DIVISION_INDEX ||
267 _to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX ||
268 _from_mesh_division_behavior ==
269 MeshDivisionTransferUse::MATCH_SUBAPP_INDEX) &&
270 tree_division_index != node_div_index)
271 continue;
272 }
273
274 // Transformed node is in the reference space, as is the _nearest_positions_obj
275 const auto transformed_node = (*_from_transforms[getGlobalSourceAppIndex(i_from)])(*node);
276
277 // Only add to the KDTree nodes that are closest to the 'position'
278 // When querying values at a target point, the KDTree associated to the closest
279 // position to the target point is queried
280 if (_nearest_positions_obj && !closestToPosition(i_pos, transformed_node))
281 continue;
282
283 _local_points[i_source].push_back(transformed_node);
284 const auto dof = node->dof_number(from_sys.number(), from_var_num, 0);
285 _local_values[i_source].push_back((*from_sys.solution)(dof));
286 if (!_use_zero_dof_for_value[var_index])
287 flagInvalidSolution(
288 "Nearest-location is not implemented for this source variable type on "
289 "this mesh. Returning value at dof 0");
290 }
291 }
292 // Build KDTree from centroids and value at centroids
293 else
294 {
295 for (auto & elem : as_range(from_mesh.getMesh().local_elements_begin(),
296 from_mesh.getMesh().local_elements_end()))
297 {
298 if (elem->n_dofs(from_sys.number(), from_var_num) < 1)
299 continue;
300
301 if (!_from_blocks.empty() && !inBlocks(_from_blocks, elem))
302 continue;
303
304 if (!_from_boundaries.empty() && !onBoundaries(_from_boundaries, from_mesh, elem))
305 continue;
306
307 // Handle the various source mesh divisions behaviors
308 if (!_from_mesh_divisions.empty())
309 {
310 const auto tree_division_index = i_source % getNumDivisions();
311 const auto elem_div_index = _from_mesh_divisions[i_from]->divisionIndex(*elem);
312
313 // Spatial restriction is always active
314 if (elem_div_index == MooseMeshDivision::INVALID_DIVISION_INDEX)
315 continue;
316 // We fill one tree per division index for matching subapp index or division index. We
317 // only accept source data from the division index
318 else if ((_from_mesh_division_behavior ==
319 MeshDivisionTransferUse::MATCH_DIVISION_INDEX ||
320 _from_mesh_division_behavior ==
321 MeshDivisionTransferUse::MATCH_SUBAPP_INDEX) &&
322 tree_division_index != elem_div_index)
323 continue;
324 }
325
326 const auto vertex_average = elem->vertex_average();
327 // Transformed element vertex average is in the reference space, as is the
328 // _nearest_positions_obj
329 const auto transformed_vertex_average =
330 (*_from_transforms[getGlobalSourceAppIndex(i_from)])(vertex_average);
331
332 if (_nearest_positions_obj && !closestToPosition(i_pos, transformed_vertex_average))
333 continue;
334
335 _local_points[i_source].push_back(transformed_vertex_average);
336 if (_use_zero_dof_for_value[var_index])
337 {
338 auto dof = elem->dof_number(from_sys.number(), from_var_num, 0);
339 _local_values[i_source].push_back((*from_sys.solution)(dof));
340 }
341 else
342 _local_values[i_source].push_back(
343 from_sys.point_value(from_var_num, vertex_average, elem));
344 }
345 }
346 max_leaf_size = std::max(max_leaf_size, from_mesh.getMaxLeafSize());
347 }
348
349 // Make a KDTree from the accumulated points data
350 std::shared_ptr<KDTree> _kd_tree =
351 std::make_shared<KDTree>(_local_points[i_source], max_leaf_size);
352 _local_kdtrees[i_source] = _kd_tree;
353 }
354}

364void
365MultiAppGeneralFieldNearestLocationTransfer::evaluateInterpValuesNearestNode(
366 const std::vector<std::pair<Point, unsigned int>> & incoming_points,
367 std::vector<std::pair<Real, Real>> & outgoing_vals)
368{
369 dof_id_type i_pt = 0;
370 for (auto & [pt, mesh_div] : incoming_points)
371 {
372 // Reset distance
373 outgoing_vals[i_pt].second = std::numeric_limits<Real>::max();
374 bool point_found = false;
375
376 const unsigned int num_sources = getNumSources();
377
378 // Loop on all sources
379 for (const auto i_from : make_range(num_sources))
380 {
381 // Examine all restrictions for the point. This source (KDTree+values) could be ruled out
382 if (!checkRestrictionsForSource(pt, mesh_div, i_from))
383 continue;
384
385 // TODO: Pre-allocate these two work arrays. They will be regularly resized by the searches
386 std::vector<std::size_t> return_index(_num_nearest_points);
387 std::vector<Real> return_dist_sqr(_num_nearest_points);
388
389 // KD Tree can be empty if no points are within block/boundary/bounding box restrictions
390 if (_local_kdtrees[i_from]->numberCandidatePoints())
391 {
392 point_found = true;
393 // Note that we do not need to use the transformed_pt (in the source app frame)
394 // because the KDTree has been created in the reference frame
395 _local_kdtrees[i_from]->neighborSearch(
396 pt, _num_nearest_points, return_index, return_dist_sqr);
397 Real val_sum = 0, dist_sum = 0;
398 for (const auto index : return_index)
399 {
400 val_sum += _local_values[i_from][index];
401 dist_sum += (_local_points[i_from][index] - pt).norm();
402 }
403
404 // If the new value found is closer than for other sources, use it
405 const auto new_distance = dist_sum / return_dist_sqr.size();
406 if (new_distance < outgoing_vals[i_pt].second)
407 outgoing_vals[i_pt] = {val_sum / return_index.size(), new_distance};
408 }
409 }
410
411 // none of the source problem meshes were within the restrictions set
412 if (!point_found)
413 outgoing_vals[i_pt] = {GeneralFieldTransfer::BetterOutOfMeshValue,
414 GeneralFieldTransfer::BetterOutOfMeshValue};
415 else if (_search_value_conflicts)
416 {
417 // Local source meshes conflicts can happen two ways:
418 // - two (or more) problems' nearest nodes are equidistant to the target point
419 // - within a problem, the num_nearest_points'th closest node is as close to the target
420 // point as the num_nearest_points + 1'th closest node
421 unsigned int num_equidistant_problems = 0;
422
423 for (const auto i_from : make_range(num_sources))
424 {
425 // Examine all restrictions for the point. This source (KDTree+values) could be ruled out
426 if (!checkRestrictionsForSource(pt, mesh_div, i_from))
427 continue;
428
429 // TODO: Pre-allocate these two work arrays. They will be regularly resized by the searches
430 std::vector<std::size_t> return_index(_num_nearest_points + 1);
431 std::vector<Real> return_dist_sqr(_num_nearest_points + 1);
432
433 // NOTE: app_index is not valid if _group_subapps = true
434 const auto app_index = i_from / getNumDivisions();
435 const auto num_search = _num_nearest_points + 1;
436
437 if (_local_kdtrees[i_from]->numberCandidatePoints())
438 {
439 _local_kdtrees[i_from]->neighborSearch(pt, num_search, return_index, return_dist_sqr);
440 auto num_found = return_dist_sqr.size();
441
442 // Local coordinates only accessible when not using nearest-position
443 // as we did not keep the index of the source app, only the position index
444 const Point local_pt = getPointInLocalSourceFrame(app_index, pt);
445
446 if (!_nearest_positions_obj &&
447 _from_transforms[getGlobalSourceAppIndex(app_index)]->hasCoordinateSystemTypeChange())
448 if (!_skip_coordinate_collapsing)
449 mooseInfo("Search value conflict cannot find the origin point due to the "
450 "non-uniqueness of the coordinate collapsing reverse mapping");
451
452 // Look for too many equidistant nodes within a problem. First zip then sort by distance
453 std::vector<std::pair<Real, std::size_t>> zipped_nearest_points;
454 for (const auto i : make_range(num_found))
455 zipped_nearest_points.push_back(std::make_pair(return_dist_sqr[i], return_index[i]));
456 std::sort(zipped_nearest_points.begin(), zipped_nearest_points.end());
457
458 // If two furthest are equally far from target point, then we have an indetermination in
459 // what is sent in this communication round from this process. However, it may not
460 // materialize to an actual conflict, as values sent from another process for the
461 // desired target point could be closer (nearest). There is no way to know at this point
462 // in the communication that a closer value exists somewhere else
463 if (num_found > 1 && num_found == num_search &&
464 MooseUtils::absoluteFuzzyEqual(zipped_nearest_points[num_found - 1].first,
465 zipped_nearest_points[num_found - 2].first))
466 registerConflict(app_index, 0, local_pt, outgoing_vals[i_pt].second, true);
467
468 // Recompute the distance for this problem. If it matches the cached value more than
469 // once it means multiple problems provide equidistant values for this point
470 Real dist_sum = 0;
471 for (const auto i : make_range(num_search - 1))
472 {
473 auto index = zipped_nearest_points[i].second;
474 dist_sum += (_local_points[i_from][index] - pt).norm();
475 }
476
477 // Compare to the selected value found after looking at all the problems
478 if (MooseUtils::absoluteFuzzyEqual(dist_sum / return_dist_sqr.size(),
479 outgoing_vals[i_pt].second))
480 {
481 num_equidistant_problems++;
482 if (num_equidistant_problems > 1)
483 registerConflict(app_index, 0, local_pt, outgoing_vals[i_pt].second, true);
484 }
485 }
486 }
487 }
488
489 // Move to next point
490 i_pt++;
491 }

494bool
495MultiAppGeneralFieldNearestLocationTransfer::inBlocks(const std::set<SubdomainID> & blocks,
496 const MooseMesh & mesh,
497 const Elem * elem) const
498{
499 // We need to override the definition of block restriction for an element
500 // because we have to consider whether each node of an element is adjacent to a block
501 for (const auto & i_node : make_range(elem->n_nodes()))
502 {
503 const auto & node = elem->node_ptr(i_node);
504 const std::set<SubdomainID> & node_blocks = mesh.getNodeBlockIds(*node);
505 std::set<SubdomainID> u;
506 std::set_intersection(blocks.begin(),
507 blocks.end(),
508 node_blocks.begin(),
509 node_blocks.end(),
510 std::inserter(u, u.begin()));
511 if (!u.empty())
512 return true;
513 }
514 return false;
515}

517unsigned int
518MultiAppGeneralFieldNearestLocationTransfer::getNumSources() const
519{
520 // Number of source = number of KDTrees.
521 // Using mesh divisions or nearest-positions, for every app we use 1 tree per division
522 if (!_from_mesh_divisions.empty() || (_nearest_positions_obj && !_group_subapps))
523 return _from_problems.size() * getNumDivisions();
524 // If we group apps, then we only use one tree per division (nearest-position region)
525 else if (_nearest_positions_obj && _group_subapps)
526 return _nearest_positions_obj->getNumPositions(_fe_problem.getCurrentExecuteOnFlag() ==
527 EXEC_INITIAL);
528 // Regular case: 1 KDTree per app
529 else
530 return _from_problems.size();

569bool
570MultiAppGeneralFieldNearestLocationTransfer::checkRestrictionsForSource(
571 const Point & pt, const unsigned int mesh_div, const unsigned int i_from) const
572{
573 // Only use the KDTree from the closest position if in "nearest-position" mode
574 const auto i_pos = _group_subapps ? i_from : i_from % getNumDivisions();
575 if (_nearest_positions_obj && !closestToPosition(i_pos, pt))
576 return false;
577
578 const unsigned int app_index = i_from / getNumDivisions();
579
580 // Check mesh restriction before anything
581 if (_source_app_must_contain_point)
582 {
583 // Transform the point to place it in the local coordinate system
584 const auto local_pt = getPointInLocalSourceFrame(app_index, pt);
585 if (!inMesh(_from_point_locators[app_index].get(), local_pt))
586 return false;
587 }
588
589 // Check the mesh division. We have handled the restriction of the source locations when
590 // building the nearest-neighbor trees. We only need to check that we meet the required
591 // source division index.
592 if (!_from_mesh_divisions.empty())
593 {
594 mooseAssert(mesh_div != MooseMeshDivision::INVALID_DIVISION_INDEX,
595 "We should not be receiving point requests with an invalid "
596 "source mesh division index");
597 const unsigned int kd_div_index = i_from % getNumDivisions();
598
599 // If matching source mesh divisions to target apps, we check that the index of the target
600 // application, which was passed in the point request, is equal to the current mesh division
601 if (_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX &&
602 mesh_div != kd_div_index)
603 return false;
604 // If matching source mesh divisions to target mesh divisions, we check that the index of the
605 // target mesh division, which was passed in the point request, is equal to the current mesh
606 // division
607 else if ((_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX ||
608 _to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX) &&
609 mesh_div != kd_div_index)
610 return false;
611 }
612
613 // If matching target apps to source mesh divisions, we check that the global index of the
614 // application is equal to the target mesh division index, which was passed in the point request
615 if (_to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX &&
616 mesh_div != getGlobalSourceAppIndex(app_index))
617 return false;
618
619 return true;
