22Terminator::validParams()
23{
24 InputParameters params = GeneralUserObject::validParams();
25 params.addClassDescription("Requests termination of the current solve based on the evaluation of"
26 " a parsed logical expression of the Postprocessor value(s).");
27 params.addRequiredCustomTypeParam<std::string>(
28 "expression",
29 "FunctionExpression",
30 "FParser expression to process Postprocessor values into a boolean value. "

32 MooseEnum failModeOption("HARD SOFT", "HARD");
33 params.addParam<MooseEnum>(
34 "fail_mode",
35 failModeOption,
36 "Abort entire simulation (HARD) or just the current time step (SOFT).");
37 params.addParam<std::string>("message",
38 "An optional message to be output instead of the default message "
39 "when the termination condition is triggered");
40
41 MooseEnum errorLevel("INFO WARNING ERROR NONE", "INFO");
42 errorLevel.addDocumentation("INFO", "Output an information message once.");
43 errorLevel.addDocumentation("WARNING", "Output a warning message once.");
44 errorLevel.addDocumentation("ERROR",
45 "Throw a MOOSE error, resulting in the termination of the run.");
46 errorLevel.addDocumentation("NONE", "No message will be printed.");
47 params.addParam<MooseEnum>(
48 "error_level",
49 errorLevel,
50 "The error level for the message. A level of ERROR will always lead to a hard "

55Terminator::Terminator(const InputParameters & parameters)
56 : GeneralUserObject(parameters),
57 _fail_mode(getParam<MooseEnum>("fail_mode").getEnum<FailMode>()),
58 _error_level(getParam<MooseEnum>("error_level").getEnum<ErrorLevel>()),
59 _pp_names(),
60 _pp_values(),
61 _expression(getParam<std::string>("expression")),
62 _fp()
63{
64 // sanity check the parameters
65 if (_error_level == ErrorLevel::ERROR && _fail_mode == FailMode::SOFT)
66 paramError("error_level",
67 "Setting the error level to ERROR always causes a hard failure, which is "
68 "incompatible with `fail_mode=SOFT`.");
69 if (_error_level == ErrorLevel::NONE && isParamValid("message"))
70 paramError("error_level",
71 "Cannot specify `error_level=NONE` together with the `message` parameter.");
72
73 // build the expression object
74 if (_fp.ParseAndDeduceVariables(_expression, _pp_names) >= 0)
75 mooseError(std::string("Invalid function\n" + _expression + "\nin Terminator.\n") +
76 _fp.ErrorMsg());
77
78 _pp_num = _pp_names.size();
79 _pp_values.resize(_pp_num);
80
81 // get all necessary postprocessors
82 for (unsigned int i = 0; i < _pp_num; ++i)
83 _pp_values[i] = &getPostprocessorValueByName(_pp_names[i]);
84
85 _params.resize(_pp_num);
86}

88void
89Terminator::initialSetup()
90{
91 // Check execution schedule of the postprocessors
92 if (_fail_mode == FailMode::SOFT)
93 for (const auto i : make_range(_pp_num))
94 // Make sure the postprocessor is executed at least as often
95 {
96 const auto & pp_exec = _fe_problem.getUserObjectBase(_pp_names[i], _tid).getExecuteOnEnum();
97 for (const auto & flag : getExecuteOnEnum())
98 if (!pp_exec.contains(flag) && flag != EXEC_FINAL)
99 paramWarning("expression",
100 "Postprocessor '" + _pp_names[i] + "' is not executed on " + flag.name() +
101 ", which it really should be to serve in the criterion "
102 "expression for throwing.");
103 }
104}

106void
107Terminator::handleMessage()
108{
109 std::string message;
110 if (!isParamValid("message"))
111 {
112 message = "Terminator '" + name() + "' is causing ";
113 if (_fail_mode == FailMode::HARD)
114 message += "the execution to terminate.\n";
115 else
116 message += "a time step cutback by marking the current step as failed.\n";
117 }
118 else
119 message = getParam<std::string>("message");
120
121 switch (_error_level)
122 {
123 case ErrorLevel::INFO:
124 mooseInfoRepeated(message);
125 break;
126
127 case ErrorLevel::WARNING:
128 mooseWarning(message);
129 break;
130
131 case ErrorLevel::ERROR:
132 mooseError(message);
133 break;
134
135 default:
136 break;
137 }
138}

140void
141Terminator::execute()
142{
143 // copy current Postprocessor values into the FParser parameter buffer
144 for (unsigned int i = 0; i < _pp_num; ++i)
145 _params[i] = *(_pp_values[i]);
146
147 // request termination of the run or timestep in case the expression evaluates to true
148 if (_fp.Eval(_params.data()) != 0)
149 {
150 handleMessage();
151 if (_fail_mode == FailMode::HARD)
152 _fe_problem.terminateSolve();
153 else
154 {
155 // Within a nonlinear solve, trigger a solve fail
156 if (_fe_problem.getCurrentExecuteOnFlag() == EXEC_LINEAR ||
157 _fe_problem.getCurrentExecuteOnFlag() == EXEC_NONLINEAR)
158 _fe_problem.setFailNextNonlinearConvergenceCheck();
159 // Outside of a solve, trigger a time step fail
160 else
161 getMooseApp().getExecutioner()->fixedPointSolve().failStep();
162 }
163 }
164}
