26bool
27ADFParser::JITCompile()
28{
29#if LIBMESH_HAVE_FPARSER_JIT
30 std::string includes;
31 const auto type_hash = typeid(ADReal).hash_code();
32 bool result;
33
34 const auto include_path_env = std::getenv("MOOSE_ADFPARSER_JIT_INCLUDE");
35 if (include_path_env)
36 result = JITCompileHelper(
37 "ADReal", "", "#include \"" + std::string(include_path_env) + "\"\n", type_hash);
38 else
39 {
40 // check if we can find an installed version of the monolithic include
41 const auto include_path =
42 MooseUtils::pathjoin(Moose::getExecutablePath(), "../include/moose/ADRealMonolithic.h");
43 if (MooseUtils::checkFileReadable(include_path, false, false, false))
44 result = JITCompileHelper("ADReal", "", "#include \"" + include_path + "\"\n", type_hash);
45 else
46 // otherwise use the compiled in location from the source tree
47 result = JITCompileHelper("ADReal", ADFPARSER_INCLUDES, "#include \"ADReal.h\"\n", type_hash);
48 }
49
50 if (!result)
51#endif
52 mooseError("ADFParser::JITCompile() failed. Evaluation not possible.");
53
54 return true;
55}
