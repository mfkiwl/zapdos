109void
110ElementSubdomainModifier::execute()
111{
112 // First, compute the desired subdomain ID for the current element.
113 SubdomainID subdomain_id = computeSubdomainID();
114
115 // If the current element's subdomain ID isn't what we want
116 if (subdomain_id != std::numeric_limits<SubdomainID>::max() &&
117 _current_elem->subdomain_id() != subdomain_id)
118 {
119 _moving_boundary_subdomains.insert(subdomain_id);
120 _moving_boundary_subdomains.insert(_current_elem->subdomain_id());
121 // Current element ID, used to index both the element on the displaced and undisplaced meshes.
122 dof_id_type elem_id = _current_elem->id();
123 // Change the element's subdomain
124 Elem * elem = _mesh.elemPtr(elem_id);
125 Elem * displaced_elem =
126 _displaced_problem ? _displaced_problem->mesh().elemPtr(elem_id) : nullptr;
127
128 // Save the affected nodes so that we can later update/initialize the solution
129 for (unsigned int i = 0; i < elem->n_nodes(); ++i)
130 _moved_nodes.insert(elem->node_id(i));
131 _cached_subdomain_assignments.emplace_back(elem, subdomain_id);
132 _moved_elems.push_back(elem);
133 if (displaced_elem)
134 {
135 _cached_subdomain_assignments.emplace_back(displaced_elem, subdomain_id);
136 _moved_displaced_elems.push_back(displaced_elem);
137 }
138 // Change the parent's subdomain, if any
139 // We do not save the parent info since they are inactive
140 setAncestorsSubdomainIDs(subdomain_id, elem_id);
141 }

162void
163ElementSubdomainModifier::finalize()
164{
165 // apply cached subdomain changes
166 for (auto & [elem, subdomain_id] : _cached_subdomain_assignments)
167 elem->subdomain_id() = subdomain_id;
168
169 _ghost_sides_to_add.clear();
170 _complement_ghost_sides_to_add.clear();
171 _complement_ghost_sides_to_remove.clear();
172 _complement_ghost_nodes_to_remove.clear();
173
174 /*
175 Synchronize ghost element subdomain ID
176 Note: this needs to be done before updating boundary info because
177 updating boundary requires the updated element subdomain ids
178 */
179 SyncSubdomainIds sync(_mesh.getMesh());
180 Parallel::sync_dofobject_data_by_id(_mesh.getMesh().comm(),
181 _mesh.getMesh().elements_begin(),
182 _mesh.getMesh().elements_end(),
183 sync);
184 if (_moving_boundary_specified)
185 updateMovingBoundaryInfo(_mesh, _moved_elems);
186 if (!_moved_elems.empty() && _complement_moving_boundary_specified)
187 updateComplementBoundaryInfo(_mesh, _moved_elems);
188
189 // Similarly for the displaced mesh
190 if (_displaced_problem)
191 {
192 SyncSubdomainIds sync_displaced(_displaced_problem->mesh().getMesh());
193 Parallel::sync_dofobject_data_by_id(_displaced_problem->mesh().getMesh().comm(),
194 _displaced_problem->mesh().getMesh().elements_begin(),
195 _displaced_problem->mesh().getMesh().elements_end(),
196 sync_displaced);
197 if (_moving_boundary_specified)
198 updateMovingBoundaryInfo(_displaced_problem->mesh(), _moved_displaced_elems);
199 if (!_moved_displaced_elems.empty() && _complement_moving_boundary_specified)
200 updateComplementBoundaryInfo(_displaced_problem->mesh(), _moved_displaced_elems);
201 }
202
203 // Synchronize boundary information after mesh update
204 synchronizeBoundaryInfo(_mesh);
205
206 if (_displaced_problem)
207 synchronizeBoundaryInfo(_displaced_problem->mesh());
208
209 // Reinit equation systems
210 _fe_problem.meshChanged();
211
212 // Apply initial condition for the newly moved elements and boundary nodes
213 buildMovedElemsRange();
214 buildMovedBndNodesRange();
215
216 if (_apply_ic)
217 {
218 _fe_problem.projectInitialConditionOnCustomRange(movedElemsRange(), movedBndNodesRange());
219
220 // Set old and older solution on the initialized dofs
221 setOldAndOlderSolutionsForMovedNodes(_fe_problem.getNonlinearSystemBase(_sys.number()));
222 setOldAndOlderSolutionsForMovedNodes(_fe_problem.getAuxiliarySystem());
223 }
224
225 // Initialize stateful material properties for the newly activated elements
226 _fe_problem.initElementStatefulProps(movedElemsRange(), false);
227}

229void
230ElementSubdomainModifier::updateMovingBoundaryInfo(MooseMesh & mesh,
231 const std::vector<const Elem *> & moved_elems)
232{
233 if (_moving_boundary_subdomains.size())
234 mooseAssert(_moving_boundary_subdomains.size() == 2,
235 "The number of moving subdomains should be two");
236 /*
237 There are a couple of steps to reconstruct the moving boundary.
238 1) Retrieve all the active elements associated with the moving boundary
239 in a previous step. These elements and their neighbors will serve as
240 boundary element candidates for the current step.
241 2) Remove all the elements from the moving boundary. That literately
242 deletes the moving boundary from the database.
243 3) Append moved elements to the boundary element candidates.
244 4) Reconstruct the moving boundary using the boundary element candidates by
245 computing the sides that are shared by two subdomains
246 5) Delete the old nodeset.
247 6) Reconstruct a new nodeset using the new sideset.
248 7) Sync boundary information
249 */
250
251 BoundaryInfo & bnd_info = mesh.getMesh().get_boundary_info();
252
253 /*
254 * Names can be deleted in the previous step if there is no side on this boundary
255 */
256 bnd_info.sideset_name(_moving_boundary_id) = _moving_boundary_name;
257 bnd_info.nodeset_name(_moving_boundary_id) = _moving_boundary_name;
258
259 auto & elem_side_bnd_ids = bnd_info.get_sideset_map();
260 std::set<const Elem *> boundary_elem_candidates;
261 std::vector<std::pair<const Elem *, unsigned int>> elem_sides_to_be_cleared;
262 /*
263 Check all the elements in the current moving boundary.
264 If an element is active, add it to the boundary element list;
265 otherwise, add active family members.
266 At the same time, append all the elements, including the active and
267 inactive, to a list being deleted later.
268 */
269 for (const auto & [elem, side_bnd] : elem_side_bnd_ids)
270 {
271 auto side = side_bnd.first;
272 auto boundary_id = side_bnd.second;
273 if (boundary_id == _moving_boundary_id)
274 {
275 if (elem->active())
276 boundary_elem_candidates.insert(elem);
277 else
278 {
279 auto top_parent = elem->top_parent();
280 std::vector<const Elem *> active_family;
281 top_parent->active_family_tree(active_family);
282 for (auto felem : active_family)
283 boundary_elem_candidates.insert((Elem *)felem);
284 }
285 elem_sides_to_be_cleared.emplace_back(elem, side);
286 }
287 }
288
289 /* Delete the old moving boundary */
290 for (auto & [elem, side] : elem_sides_to_be_cleared)
291 bnd_info.remove_side(elem, side, _moving_boundary_id);
292
293 /* Append moved elements to the boundary element candidate list */
294 for (auto elem : moved_elems)
295 boundary_elem_candidates.insert(elem);
296
297 /* Go through the boundary element candidate list */
298 for (auto elem : boundary_elem_candidates)
299 {
300 for (auto side : elem->side_index_range())
301 {
302 const Elem * neighbor = elem->neighbor_ptr(side);
303
304 /*
305 If elem's neighbor is active and has a different subdomain, we add the current side
306 to the moving boundary
307 */
308 if (neighbor && neighbor->active() && neighbor != libMesh::remote_elem)
309 {
310 if (neighbor->subdomain_id() != elem->subdomain_id())
311 {
312 /*
313 * If there is no new moved element, and then the elements from the original
314 * moving boundary should be safe to use
315 * If there are some new moved elements, then we need to check whether or not the
316 * interface is between the moving subdomains
317 */
318 if (!_moving_boundary_subdomains.size() ||
319 (_moving_boundary_subdomains.size() &&
320 _moving_boundary_subdomains.find(neighbor->subdomain_id()) !=
321 _moving_boundary_subdomains.end() &&
322 _moving_boundary_subdomains.find(elem->subdomain_id()) !=
323 _moving_boundary_subdomains.end()))
324 {
325 bnd_info.add_side(elem, side, _moving_boundary_id);
326 }
327 }
328 }
329 /*
330 If elem's neighbor is not active, we need to check family members of the neighbor.
331 In this case, the neighbor's children are on the current side and the children are
332 "smaller" than the current element. We so add the neighboring children to the moving
333 boundary list. Assigning "smaller" elements to the moving_boundary list is necessary in
334 order to correctly represent the moving boundary.
335 */
336 else if (neighbor && !neighbor->active() && neighbor != libMesh::remote_elem)
337 {
338 std::vector<const Elem *> active_family;
339 auto top_parent = neighbor->top_parent();
340 top_parent->active_family_tree_by_neighbor(active_family, elem);
341 for (auto felem : active_family)
342 {
343 if (felem->subdomain_id() != elem->subdomain_id())
344 {
345 auto cside = felem->which_neighbor_am_i(elem);
346 // Add all the sides to the boundary first and remove excessive sides later
347 bnd_info.add_side(felem, cside, _moving_boundary_id);
348 if (felem->processor_id() != this->processor_id())
349 _ghost_sides_to_add[felem->processor_id()].emplace_back(felem->id(), cside);
350 }
351 }
352 }
353 }
354 }
355
356 /* Delete the corresponding nodeset as well */
357 auto & nodeset_map = bnd_info.get_nodeset_map();
358 std::vector<const Node *> nodes_elem_sides_to_be_cleared;
359 for (const auto & pair : nodeset_map)
360 if (pair.second == _moving_boundary_id)
361 nodes_elem_sides_to_be_cleared.push_back(pair.first);
362
363 for (const auto node : nodes_elem_sides_to_be_cleared)
364 bnd_info.remove_node(node, _moving_boundary_id);
365
366 /* Reconstruct a new nodeset from the updated sideset */
367 std::set<const Node *> boundary_nodes;
368 for (const auto & [elem, side_bnd] : elem_side_bnd_ids)
369 {
370 auto side = side_bnd.first;
371 auto boundary_id = side_bnd.second;
372 if (boundary_id == _moving_boundary_id)
373 {
374 auto nodes = elem->nodes_on_side(side);
375 for (auto node : nodes)
376 boundary_nodes.insert(&(elem->node_ref(node)));
377 }
378 }
379
380 for (const auto node : boundary_nodes)
381 bnd_info.add_node(node, _moving_boundary_id);
382}

384void
385ElementSubdomainModifier::updateComplementBoundaryInfo(
386 MooseMesh & mesh, const std::vector<const Elem *> & moved_elems)
387{
388 BoundaryInfo & bnd_info = mesh.getMesh().get_boundary_info();
389
390 // The logic below updates the side set and the node set associated with the moving boundary.
391 std::set<dof_id_type> added_nodes, removed_nodes;
392 for (auto elem : moved_elems)
393 {
394 // First loop over all the sides of the element
395 for (auto side : elem->side_index_range())
396 {
397 const Elem * neighbor = elem->neighbor_ptr(side);
398 bnd_info.remove_side(elem, side, _complement_moving_boundary_id);
399 if (neighbor && neighbor != libMesh::remote_elem)
400 {
401 // If the neighbor has a different subdomain ID, then this side should be added to
402 // the moving boundary
403 if (neighbor->subdomain_id() != elem->subdomain_id())
404 {
405
406 unsigned int neighbor_side = neighbor->which_neighbor_am_i(elem);
407 bnd_info.add_side(neighbor, neighbor_side, _complement_moving_boundary_id);
408 if (neighbor->processor_id() != this->processor_id())
409 {
410 _complement_ghost_sides_to_add[neighbor->processor_id()].emplace_back(neighbor->id(),
411 neighbor_side);
412 }
413 }
414 // Otherwise remove this side and the neighbor side from the boundary.
415 else
416 {
417 // this will destroy any overlapping third side sets!
418 bnd_info.remove_side(elem, side, _complement_moving_boundary_id);
419 unsigned int neighbor_side = neighbor->which_neighbor_am_i(elem);
420 bnd_info.remove_side(neighbor, neighbor_side);
421 if (neighbor->processor_id() != this->processor_id())
422 _complement_ghost_sides_to_remove[neighbor->processor_id()].emplace_back(neighbor->id(),
423 neighbor_side);
424 }
425 }
426 }
427
428 // Then loop over all the nodes of the element
429 for (auto node : elem->node_index_range())
430 {
431 // Find the point neighbors
432 std::set<const Elem *> neighbor_set;
433 elem->find_point_neighbors(elem->node_ref(node), neighbor_set);
434 for (auto neighbor : neighbor_set)
435 if (neighbor != libMesh::remote_elem)
436 {
437 // If the neighbor has a different subdomain ID, then this node should be added to
438 // the moving boundary
439 if (neighbor->subdomain_id() != elem->subdomain_id())
440 added_nodes.insert(elem->node_id(node));
441 // Otherwise remove this node from the boundary.
442 else
443 {
444 removed_nodes.insert(elem->node_id(node));
445 if (neighbor->processor_id() != this->processor_id())
446 _complement_ghost_nodes_to_remove[neighbor->processor_id()].push_back(
447 elem->node_id(node));
448 }
449 }
450 }
451 }
452
453 // make sure to remove only nodes that are not in the add set
454 std::set<dof_id_type> nodes_to_remove;
455 std::set_difference(removed_nodes.begin(),
456 removed_nodes.end(),
457 added_nodes.begin(),
458 added_nodes.end(),
459 std::inserter(nodes_to_remove, nodes_to_remove.end()));
460 for (auto node_id : nodes_to_remove)
461 {
462 if (_complement_moving_boundary_specified)
463 mesh.getMesh().get_boundary_info().remove_node(mesh.nodePtr(node_id),
464 _complement_moving_boundary_id);
465 }
466}

478void
479ElementSubdomainModifier::pushBoundarySideInfo(MooseMesh & moose_mesh)
480{
481 auto & mesh = moose_mesh.getMesh();
482 auto elem_remove_functor =
483 [&mesh, this](processor_id_type,
484 const std::vector<std::pair<dof_id_type, unsigned short int>> & received_elem)
485 {
486 // remove the side
487 for (const auto & pr : received_elem)
488 mesh.get_boundary_info().remove_side(
489 mesh.elem_ptr(pr.first), pr.second, _complement_moving_boundary_id);
490 };
491
492 // We create a tempalte functor to add with custom boundary IDs
493 auto elem_add_functor_with_boundary_id =
494 [&mesh](const std::vector<std::pair<dof_id_type, unsigned short int>> & received_elem,
495 const BoundaryID boundary_id)
496 {
497 // add the side
498 for (const auto & pr : received_elem)
499 mesh.get_boundary_info().add_side(mesh.elem_ptr(pr.first), pr.second, boundary_id);
500 };
501
502 // Then we use it with the regular and the complement boundary id
503 auto complement_elem_add_functor =
504 [this, elem_add_functor_with_boundary_id](
505 processor_id_type,
506 const std::vector<std::pair<dof_id_type, unsigned short int>> & received_elem)
507 { elem_add_functor_with_boundary_id(received_elem, _complement_moving_boundary_id); };
508
509 auto elem_add_functor =
510 [this, elem_add_functor_with_boundary_id](
511 processor_id_type,
512 const std::vector<std::pair<dof_id_type, unsigned short int>> & received_elem)
513 { elem_add_functor_with_boundary_id(received_elem, _moving_boundary_id); };
514
515 // Push/pull the ghost cell sides for the regular and complement boundaries
516 Parallel::push_parallel_vector_data(
517 mesh.get_boundary_info().comm(), _ghost_sides_to_add, elem_add_functor);
518 Parallel::push_parallel_vector_data(
519 mesh.get_boundary_info().comm(), _complement_ghost_sides_to_remove, elem_remove_functor);
520 if (_complement_moving_boundary_specified)
521 Parallel::push_parallel_vector_data(mesh.get_boundary_info().comm(),
522 _complement_ghost_sides_to_add,
523 complement_elem_add_functor);
524}

526void
527ElementSubdomainModifier::pushBoundaryNodeInfo(MooseMesh & moose_mesh)
528{
529 auto & mesh = moose_mesh.getMesh();
530 auto node_remove_functor =
531 [&mesh, this](processor_id_type, const std::vector<dof_id_type> & received_nodes)

533 for (const auto & pr : received_nodes)
534 {
535 if (_moving_boundary_specified)
536 mesh.get_boundary_info().remove_node(mesh.node_ptr(pr), _moving_boundary_id);
537 if (_complement_moving_boundary_specified)
538 mesh.get_boundary_info().remove_node(mesh.node_ptr(pr), _complement_moving_boundary_id);
539 }
540 };
541
542 Parallel::push_parallel_vector_data(
543 mesh.get_boundary_info().comm(), _complement_ghost_nodes_to_remove, node_remove_functor);
544}

546void
547ElementSubdomainModifier::buildMovedElemsRange()
548{
549 // Clear the object first
550 _moved_elems_range.reset();
551
552 // Make some fake element iterators defining this vector of elements
553 Elem * const * elem_itr_begin = const_cast<Elem * const *>(_moved_elems.data());
554 Elem * const * elem_itr_end = elem_itr_begin + _moved_elems.size();
555
556 const auto elems_begin = MeshBase::const_element_iterator(
557 elem_itr_begin, elem_itr_end, Predicates::NotNull<Elem * const *>());
558 const auto elems_end = MeshBase::const_element_iterator(
559 elem_itr_end, elem_itr_end, Predicates::NotNull<Elem * const *>());
560
561 _moved_elems_range = std::make_unique<ConstElemRange>(elems_begin, elems_end);
562}

564void
565ElementSubdomainModifier::buildMovedBndNodesRange()
566{
567 // This is more involved than building the element range, because not all moved nodes are
568 // necessarily associated with a boundary initial condition. We need to first build a set of
569 // boundary nodes. Clear the object first:
570 _moved_bnd_nodes_range.reset();
571
572 // create a vector of the newly activated nodes
573 std::set<const BndNode *> moved_bnd_nodes_set;
574 for (auto & bnd_node : *_mesh.getBoundaryNodeRange())
575 {
576 dof_id_type bnd_node_id = bnd_node->_node->id();
577 if (_moved_nodes.find(bnd_node_id) != _moved_nodes.end())
578 moved_bnd_nodes_set.insert(bnd_node);
579 }
580
581 // Dump all the boundary nodes into a vector so that we can build a range out of it
582 std::vector<const BndNode *> moved_bnd_nodes;
583 moved_bnd_nodes.assign(moved_bnd_nodes_set.begin(), moved_bnd_nodes_set.end());
584
585 // Make some fake node iterators defining this vector of nodes
586 BndNode * const * bnd_node_itr_begin = const_cast<BndNode * const *>(moved_bnd_nodes.data());
587 BndNode * const * bnd_node_itr_end = bnd_node_itr_begin + moved_bnd_nodes.size();
588
589 const auto bnd_nodes_begin = MooseMesh::const_bnd_node_iterator(
590 bnd_node_itr_begin, bnd_node_itr_end, Predicates::NotNull<const BndNode * const *>());
591 const auto bnd_nodes_end = MooseMesh::const_bnd_node_iterator(
592 bnd_node_itr_end, bnd_node_itr_end, Predicates::NotNull<const BndNode * const *>());
593
594 _moved_bnd_nodes_range = std::make_unique<ConstBndNodeRange>(bnd_nodes_begin, bnd_nodes_end);
595}

597void
598ElementSubdomainModifier::setOldAndOlderSolutionsForMovedNodes(SystemBase & sys)
599{
600 // Don't do anything if this is a steady simulation
601 if (!sys.hasSolutionState(1))
602 return;
603
604 ConstBndNodeRange & bnd_node_range = movedBndNodesRange();
605
606 NumericVector<Number> & current_solution = *sys.system().current_local_solution;
607 NumericVector<Number> & old_solution = sys.solutionOld();
608 NumericVector<Number> * older_solution = sys.hasSolutionState(2) ? &sys.solutionOlder() : nullptr;
609
610 DofMap & dof_map = sys.dofMap();
611
612 // Get dofs for the newly added elements
613 std::vector<dof_id_type> dofs;
614 for (auto & bnd_node : bnd_node_range)
615 {
616 std::vector<dof_id_type> bnd_node_dofs;
617 dof_map.dof_indices(bnd_node->_node, bnd_node_dofs);
618 dofs.insert(dofs.end(), bnd_node_dofs.begin(), bnd_node_dofs.end());
619 }
620
621 // Set the old and older solution to match the IC.
622 for (auto dof : dofs)
623 {
624 old_solution.set(dof, current_solution(dof));
625 if (older_solution)
626 older_solution->set(dof, current_solution(dof));
627 }
628
629 old_solution.close();
630 if (older_solution)
631 older_solution->close();
632}
