29LymberopoulosElectronBC::LymberopoulosElectronBC(const InputParameters & parameters)
30 : ADIntegratedBC(parameters),
31
32 _r_units(1. / getParam<Real>("position_units")),
33 _ks(getParam<Real>("ks")),
34 _gamma(getParam<Real>("gamma")),
35
36 // Coupled Variables
37 _grad_potential(adCoupledGradient("potential")),
38
39 _sign(1)
40{
41 _num_ions = coupledComponents("ion");
42
43 // Resize the vectors to store _num_ions values:
44 _ion.resize(_num_ions);
45 _ion_var.resize(_num_ions);
46 _muion.resize(_num_ions);
47 _sgnion.resize(_num_ions);
48
49 // Retrieve the values for each ion and store in the relevant vectors.
50 // Note that these need to be dereferenced to get the values inside the
51 // main body of the code.
52 // e.g. instead of "_ip[_qp]" it would be "(*_ip[i])[_qp]", where "i"
53 // refers to a single ion species.
54 for (unsigned int i = 0; i < _num_ions; ++i)
55 {
56 _ion_var[i] = getVar("ion", i);
57 _ion[i] = &adCoupledValue("ion", i);
58 _muion[i] = &getADMaterialProperty<Real>("mu" + (*getVar("ion", i)).name());
59 _sgnion[i] = &getMaterialProperty<Real>("sgn" + (*getVar("ion", i)).name());
60 }
61}

63ADReal
64LymberopoulosElectronBC::computeQpResidual()
65{
66 _ion_flux.zero();
67 for (unsigned int i = 0; i < _num_ions; ++i)
68 {
69 _ion_flux += (*_sgnion[i])[_qp] * (*_muion[i])[_qp] * -_grad_potential[_qp] * _r_units *
70 std::exp((*_ion[i])[_qp]);
71 }
72
73 return _test[_i][_qp] * _r_units *
74 (_sign * _ks * std::exp(_u[_qp]) * _normals[_qp] * _normals[_qp] -
75 _gamma * _ion_flux * _normals[_qp]);
76}
