31FVAnisotropicDiffusion::FVAnisotropicDiffusion(const InputParameters & params)
32 : FVFluxKernel(params), _coeff(getFunctor<ADRealVectorValue>("coeff"))
33{
34 const auto & interp_method = getParam<MooseEnum>("coeff_interp_method");
35 if (interp_method == "average")
36 _coeff_interp_method = Moose::FV::InterpMethod::Average;
37 else if (interp_method == "harmonic")
38 _coeff_interp_method = Moose::FV::InterpMethod::HarmonicAverage;
39
40 if ((_var.faceInterpolationMethod() == Moose::FV::InterpMethod::SkewCorrectedAverage) &&
41 (_tid == 0))
42 adjustRMGhostLayers(std::max((unsigned short)(3), _pars.get<unsigned short>("ghost_layers")));
43}

45ADReal
46FVAnisotropicDiffusion::computeQpResidual()
47{
48 const auto state = determineState();
49 const auto & grad_T = _var.adGradSln(*_face_info,
50 state,
51 _var.faceInterpolationMethod() ==
52 Moose::FV::InterpMethod::SkewCorrectedAverage);
53
54 ADRealVectorValue coeff;
55 // If we are on internal faces, we interpolate the diffusivity as usual
56 if (_var.isInternalFace(*_face_info))
57 interpolate(_coeff_interp_method,
58 coeff,
59 _coeff(elemArg(), state),
60 _coeff(neighborArg(), state),
61 *_face_info,
62 true);
63 // Else we just use the boundary values (which depend on how the diffusion
64 // coefficient is constructed)
65 else
66 {
67 const auto face = singleSidedFaceArg();
68 coeff = _coeff(face, state);
69 }
70
71 ADReal r = 0;
72 for (const auto i : make_range(Moose::dim))
73 r += _normal(i) * coeff(i) * grad_T(i);
74 return -r;
75}
