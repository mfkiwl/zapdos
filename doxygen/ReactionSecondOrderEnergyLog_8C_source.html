68Real
69ReactionSecondOrderEnergyLog::computeQpOffDiagJacobian(unsigned int jvar)
70{
71 Real power;
72 power = 0;
73
74 if (!_v_eq_u && jvar == _v_id)
75 power += 1;
76
77 if (!_w_eq_u && jvar == _w_id)
78 power += 1;
79
80 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] *
81 std::exp(_v[_qp] + _w[_qp]) * power * _phi[_j][_qp] * _threshold_energy;
82}
