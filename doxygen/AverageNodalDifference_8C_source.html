42void
43AverageNodalDifference::execute()
44{
45 Real val = _u[_qp];
46 Real other_val = _other_var[_qp];
47
48 _sum_of_squared_diff += (val - other_val) * (val - other_val);
49 _n++;
50}

58void
59AverageNodalDifference::finalize()
60{
61 gatherSum(_sum_of_squared_diff);
62 gatherSum(_n);
63
64 _value = std::sqrt(_sum_of_squared_diff / (_n * _n));
