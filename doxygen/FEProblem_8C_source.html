31FEProblem::FEProblem(const InputParameters & parameters)
32 : FEProblemBase(parameters), _use_nonlinear(getParam<bool>("use_nonlinear"))
33{
34 for (const auto i : index_range(_nl_sys_names))
35 {
36 const auto & sys_name = _nl_sys_names[i];
37 auto & nl = _nl[i];
38 nl = _use_nonlinear ? (std::make_shared<NonlinearSystem>(*this, sys_name))
39 : (std::make_shared<MooseEigenSystem>(*this, sys_name));
40 _nl_sys.push_back(std::dynamic_pointer_cast<NonlinearSystem>(nl));
41 }
42
43 // backwards compatibility for AD for objects that depend on initializing derivatives during
44 // construction
45 setCurrentNonlinearSystem(0);
46
47 _aux = std::make_shared<AuxiliarySystem>(*this, "aux0");
48
49 newAssemblyArray(_nl);
50
51 initNullSpaceVectors(parameters, _nl);
52
53 es().parameters.set<FEProblem *>("_fe_problem") = this;
54
55 // Create extra vectors and matrices if any
56 createTagVectors();
57
58 // Create extra solution vectors if any
59 createTagSolutions();
60}

71void
72FEProblem::addLineSearch(const InputParameters & parameters)
73{
74 MooseEnum line_search = parameters.get<MooseEnum>("line_search");
75 Moose::LineSearchType enum_line_search = Moose::stringToEnum<Moose::LineSearchType>(line_search);
76 if (enum_line_search == Moose::LS_CONTACT || enum_line_search == Moose::LS_PROJECT)
77 {
78 if (enum_line_search == Moose::LS_CONTACT)
79 {
80 InputParameters ls_params = _factory.getValidParams("PetscContactLineSearch");
81
82 bool affect_ltol = parameters.isParamValid("contact_line_search_ltol");
83 ls_params.set<bool>("affect_ltol") = affect_ltol;
84 ls_params.set<unsigned>("allowed_lambda_cuts") =
85 parameters.get<unsigned>("contact_line_search_allowed_lambda_cuts");
86 ls_params.set<Real>("contact_ltol") = affect_ltol
87 ? parameters.get<Real>("contact_line_search_ltol")
88 : parameters.get<Real>("l_tol");
89 ls_params.set<FEProblem *>("_fe_problem") = this;
90
91 _line_search =
92 _factory.create<LineSearch>("PetscContactLineSearch", "contact_line_search", ls_params);
93 }
94 else
95 {
96 InputParameters ls_params = _factory.getValidParams("PetscProjectSolutionOntoBounds");
97 ls_params.set<FEProblem *>("_fe_problem") = this;
98
99 _line_search = _factory.create<LineSearch>(
100 "PetscProjectSolutionOntoBounds", "project_solution_onto_bounds_line_search", ls_params);
101 }
102 }
103 else
104 mooseError("Requested line search ", line_search.operator std::string(), " is not supported");
105}
