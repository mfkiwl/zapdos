45IonsFromIonization::computeQpResidual()
46{
47 ADReal electron_flux_mag = (-_muem[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_em[_qp]) -
48 _diffem[_qp] * std::exp(_em[_qp]) * _grad_em[_qp] * _r_units)
49 .norm();
50 ADReal iz_term = _alpha_iz[_qp] * electron_flux_mag;
51
52 return -_test[_i][_qp] * iz_term;
53}
