46ADReal
47ElectronEnergyLossFromElastic::computeQpResidual()
48{
49 ADReal electron_flux_mag = (-_muem[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_em[_qp]) -
50 _diffem[_qp] * std::exp(_em[_qp]) * _grad_em[_qp] * _r_units)
51 .norm();
52 ADReal Eel = -3.0 * _massem[_qp] / _massGas[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]);
53 ADReal el_term = _alpha_el[_qp] * electron_flux_mag * Eel;
54
55 return -_test[_i][_qp] * el_term;
56}
