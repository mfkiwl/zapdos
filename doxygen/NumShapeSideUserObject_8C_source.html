42void
43NumShapeSideUserObject::executeJacobian(unsigned int jvar)
44{
45 // derivative of _integral w.r.t. u_j
46 if (jvar == _u_var)
47 {
48 // sum jacobian contributions over quadrature points
49 Real sum = 0.0;
50 for (unsigned int qp = 0; qp < _qrule->n_points(); ++qp)
51 sum += _JxW[qp] * _coord[qp] * _grad_phi[_j][qp] * _normals[qp];
52
53 // the user has to store the value of sum in a storage object indexed by global DOF _j_global
54 _jacobian_storage[_j_global] += sum;
55 }
