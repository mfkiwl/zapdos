25template <typename ComputeValueType>
26InputParameters
27AuxKernelTempl<ComputeValueType>::validParams()
28{
29 InputParameters params = MooseObject::validParams();
30 params += BlockRestrictable::validParams();
31 params += BoundaryRestrictable::validParams();
32 params += RandomInterface::validParams();
33 params += MeshChangedInterface::validParams();
34 params += MaterialPropertyInterface::validParams();
35 params += FunctorInterface::validParams();
36
37 // Add the SetupInterface parameter 'execute_on' with 'linear' and 'timestep_end'
38 params += SetupInterface::validParams();
39 ExecFlagEnum & exec_enum = params.set<ExecFlagEnum>("execute_on", true);
40 exec_enum.addAvailableFlags(EXEC_PRE_DISPLACE);
41 exec_enum = {EXEC_LINEAR, EXEC_TIMESTEP_END};
42 params.setDocString("execute_on", exec_enum.getDocString());
43
44 params.addRequiredParam<AuxVariableName>("variable",
45 "The name of the variable that this object applies to");
46
47 params.addParam<bool>("use_displaced_mesh",
48 false,
49 "Whether or not this object should use the "
50 "displaced mesh for computation. Note that "
51 "in the case this is true but no "
52 "displacements are provided in the Mesh block "
53 "the undisplaced mesh will still be used.");
54 params.addParamNamesToGroup("use_displaced_mesh", "Advanced");
55 params.addParam<bool>("check_boundary_restricted",
56 true,
57 "Whether to check for multiple element sides on the boundary "
58 "in the case of a boundary restricted, element aux variable. "
59 "Setting this to false will allow contribution to a single element's "
60 "elemental value(s) from multiple boundary sides on the same element "
61 "(example: when the restricted boundary exists on two or more sides "
62 "of an element, such as at a corner of a mesh");
63
64 // This flag is set to true if the AuxKernelTempl is being used on a boundary
65 params.addPrivateParam<bool>("_on_boundary", false);
66
67 params.declareControllable("enable"); // allows Control to enable/disable this type of object
68 params.registerBase("AuxKernel");
69
70 if (typeid(AuxKernelTempl<ComputeValueType>).name() == typeid(VectorAuxKernel).name())
71 params.registerBase("VectorAuxKernel");
72 if (typeid(AuxKernelTempl<ComputeValueType>).name() == typeid(ArrayAuxKernel).name())
73 params.registerBase("ArrayAuxKernel");
74 return params;
75}

77template <typename ComputeValueType>
78AuxKernelTempl<ComputeValueType>::AuxKernelTempl(const InputParameters & parameters)
79 : MooseObject(parameters),
80 MooseVariableInterface<ComputeValueType>(
81 this,
82 parameters.getCheckedPointerParam<SystemBase *>("_sys")
83 ->getVariable(parameters.get<THREAD_ID>("_tid"),
84 parameters.get<AuxVariableName>("variable"))
85 .isNodal(),
86 "variable",
87 Moose::VarKindType::VAR_AUXILIARY,
88 std::is_same<Real, ComputeValueType>::value
89 ? Moose::VarFieldType::VAR_FIELD_STANDARD
90 : (std::is_same<RealVectorValue, ComputeValueType>::value
91 ? Moose::VarFieldType::VAR_FIELD_VECTOR
92 : Moose::VarFieldType::VAR_FIELD_ARRAY)),
93 BlockRestrictable(this),
94 BoundaryRestrictable(this, mooseVariableBase()->isNodal()),
95 SetupInterface(this),
96 CoupleableMooseVariableDependencyIntermediateInterface(this, mooseVariableBase()->isNodal()),
97 FunctionInterface(this),
98 UserObjectInterface(this),
99 TransientInterface(this),
100 MaterialPropertyInterface(this, blockIDs(), boundaryIDs()),
101 PostprocessorInterface(this),
102 DependencyResolverInterface(),
103 RandomInterface(parameters,
104 *parameters.getCheckedPointerParam<FEProblemBase *>("_fe_problem_base"),
105 parameters.get<THREAD_ID>("_tid"),
106 mooseVariableBase()->isNodal()),
107 GeometricSearchInterface(this),
108 Restartable(this, "AuxKernels"),
109 MeshChangedInterface(parameters),
110 VectorPostprocessorInterface(this),
111 ElementIDInterface(this),
112 NonADFunctorInterface(this),
113 _check_boundary_restricted(getParam<bool>("check_boundary_restricted")),
114 _subproblem(*getCheckedPointerParam<SubProblem *>("_subproblem")),
115 _sys(*getCheckedPointerParam<SystemBase *>("_sys")),
116 _nl_sys(*getCheckedPointerParam<SystemBase *>("_nl_sys")),
117 _aux_sys(static_cast<AuxiliarySystem &>(_sys)),
118 _tid(parameters.get<THREAD_ID>("_tid")),
119 _var(_aux_sys.getActualFieldVariable<ComputeValueType>(
120 _tid, parameters.get<AuxVariableName>("variable"))),
121 _nodal(_var.isNodal()),
122 _u(_nodal ? _var.nodalValueArray() : _var.sln()),
123
124 _assembly(_subproblem.assembly(_tid, 0)),
125 _bnd(boundaryRestricted()),
126 _mesh(_subproblem.mesh()),
127
128 _test(_bnd ? _var.phiFace() : _var.phi()),
129 _q_point(_bnd ? _assembly.qPointsFace() : _assembly.qPoints()),
130 _qrule(_bnd ? _assembly.qRuleFace() : _assembly.qRule()),
131 _JxW(_bnd ? _assembly.JxWFace() : _assembly.JxW()),
132 _coord(_assembly.coordTransformation()),
133
134 _current_elem(_assembly.elem()),
135 _current_side(_assembly.side()),
136 _current_elem_volume(_assembly.elemVolume()),
137 _current_side_volume(_assembly.sideElemVolume()),
138
139 _current_node(_assembly.node()),
140 _current_boundary_id(_assembly.currentBoundaryID()),
141 _solution(_aux_sys.solution())
142{
143 addMooseVariableDependency(&_var);
144 _supplied_vars.insert(parameters.get<AuxVariableName>("variable"));
145
146 if (_bnd && !isNodal() && _check_boundary_restricted)
147 {
148 // when the variable is elemental and this aux kernel operates on boundaries,
149 // we need to check that no elements are visited more than once through visiting
150 // all the sides on the boundaries
151 auto boundaries = _mesh.getMesh().get_boundary_info().build_side_list();
152 std::set<dof_id_type> elements;
153 for (const auto & t : boundaries)
154 {
155 if (hasBoundary(std::get<2>(t)))
156 {
157 const auto eid = std::get<0>(t);
158 const auto stat = elements.insert(eid);
159 if (!stat.second) // already existed in the set
160 mooseError(
161 "Boundary restricted auxiliary kernel '",
162 name(),
163 "' has element (id=",
164 eid,
165 ") connected with more than one boundary sides.\nTo skip this error check, "
166 "set 'check_boundary_restricted = false'.\nRefer to the AuxKernel "
167 "documentation on boundary restricted aux kernels for understanding this error.");
168 }
169 }
170 }
171
172 // Check for supported variable types
173 // Any 'nodal' family that actually has DoFs outside of nodes, or gradient dofs at nodes is
174 // not properly set by AuxKernelTempl::compute
175 // NOTE: We could add a few exceptions, lower order from certain unsupported families and on
176 // certain element types only have value-DoFs on nodes
177 const auto type = _var.feType();
178 if (_var.isNodal() && !((type.family == LAGRANGE) || (type.order <= FIRST)))
179 paramError("variable",
180 "Variable family " + Moose::stringify(type.family) + " is not supported at order " +
181 Moose::stringify(type.order) + " by the AuxKernel system.");
182}

277template <typename ComputeValueType>
278void
279AuxKernelTempl<ComputeValueType>::compute()
280{
281 precalculateValue();
282
283 if (isNodal()) /* nodal variables */
284 {
285 if (_var.isNodalDefined())
286 {
287 _qp = 0;
288 ComputeValueType value = computeValue();
289 // update variable data, which is referenced by other kernels, so the value is up-to-date
290 _var.setNodalValue(value);
291 }
292 }
293 else /* elemental variables */
294 {
295 _n_local_dofs = _var.numberOfDofs();
296 if (_n_local_dofs == 1) /* p0 */
297 {
298 ComputeValueType value = 0;
299 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
300 value += _JxW[_qp] * _coord[_qp] * computeValue();
301 value /= (_bnd ? _current_side_volume : _current_elem_volume);
302 if (_var.isFV())
303 setDofValueHelper(value);
304 else
305 // update the variable data referenced by other kernels.
306 // Note that this will update the values at the quadrature points too
307 // (because this is an Elemental variable)
308 _var.setNodalValue(value);
309 }
310 else /* high-order */
311 {
312 _local_re.resize(_n_local_dofs);
313 _local_re.zero();
314 _local_ke.resize(_n_local_dofs, _n_local_dofs);
315 _local_ke.zero();
316
317 // assemble the local mass matrix and the load
318 for (unsigned int i = 0; i < _test.size(); i++)
319 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
320 {
321 ComputeValueType t = _JxW[_qp] * _coord[_qp] * _test[i][_qp];
322 _local_re(i) += t * computeValue();
323 for (unsigned int j = 0; j < _test.size(); j++)
324 _local_ke(i, j) += t * _test[j][_qp];
325 }
326 // mass matrix is always SPD but in case of boundary restricted, it will be rank deficient
327 _local_sol.resize(_n_local_dofs);
328 if (_bnd)
329 _local_ke.svd_solve(_local_re, _local_sol);
330 else
331 _local_ke.cholesky_solve(_local_re, _local_sol);
332
333 _var.setDofValues(_local_sol);
334 }
335 }
336}

338template <>
339void
340AuxKernelTempl<RealEigenVector>::compute()
341{
342 precalculateValue();
343
344 if (isNodal()) /* nodal variables */
345 {
346 if (_var.isNodalDefined())
347 {
348 _qp = 0;
349 RealEigenVector value = computeValue();
350 // update variable data, which is referenced by other kernels, so the value is up-to-date
351 _var.setNodalValue(value);
352 }
353 }
354 else /* elemental variables */
355 {
356 _n_local_dofs = _var.numberOfDofs();
357 if (_n_local_dofs == 1) /* p0 */
358 {
359 RealEigenVector value = RealEigenVector::Zero(_var.count());
360 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
361 value += _JxW[_qp] * _coord[_qp] * computeValue();
362 value /= (_bnd ? _current_side_volume : _current_elem_volume);
363 // update the variable data referenced by other kernels.
364 // Note that this will update the values at the quadrature points too
365 // (because this is an Elemental variable)
366 _var.setNodalValue(value);
367 }
368 else /* high-order */
369 {
370 _local_re.resize(_n_local_dofs);
371 for (unsigned int i = 0; i < _local_re.size(); ++i)
372 _local_re(i) = RealEigenVector::Zero(_var.count());
373 _local_ke.resize(_n_local_dofs, _n_local_dofs);
374 _local_ke.zero();
375
376 // assemble the local mass matrix and the load
377 for (unsigned int i = 0; i < _test.size(); i++)
378 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
379 {
380 Real t = _JxW[_qp] * _coord[_qp] * _test[i][_qp];
381 _local_re(i) += t * computeValue();
382 for (unsigned int j = 0; j < _test.size(); j++)
383 _local_ke(i, j) += t * _test[j][_qp];
384 }
385
386 // mass matrix is always SPD
387 _local_sol.resize(_n_local_dofs);
388 for (unsigned int i = 0; i < _local_re.size(); ++i)
389 _local_sol(i) = RealEigenVector::Zero(_var.count());
390 DenseVector<Number> re(_n_local_dofs);
391 DenseVector<Number> sol(_n_local_dofs);
392 for (unsigned int i = 0; i < _var.count(); ++i)
393 {
394 for (unsigned int j = 0; j < _n_local_dofs; ++j)
395 re(j) = _local_re(j)(i);
396
397 if (_bnd)
398 _local_ke.svd_solve(re, sol);
399 else
400 _local_ke.cholesky_solve(re, sol);
401
402 for (unsigned int j = 0; j < _n_local_dofs; ++j)
403 _local_sol(j)(i) = sol(j);
404 }
405
406 _var.setDofValues(_local_sol);
407 }
408 }
409}
