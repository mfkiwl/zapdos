57ReactionSecondOrderLogTempl<is_ad>::computeQpResidual()

63template <bool is_ad>
64Real
65ReactionSecondOrderLogTempl<is_ad>::computeQpJacobian()
66{
67 Real power;
68 power = 0.0;
69
70 if (_v_eq_u)
71 power += 1.0;
72
73 if (_w_eq_u)
74 power += 1.0;
75
76 if (!_v_eq_u && !_w_eq_u)
77 {
78 return 0.0;
79 }
80 else
81 {
82 return -_test[_i][_qp] * _stoichiometric_coeff * power *
83 raw_value(_reaction_coeff[_qp] * std::exp(_v[_qp] + _w[_qp])) * _phi[_j][_qp];
84 }
85}

87template <bool is_ad>
88Real
89ReactionSecondOrderLogTempl<is_ad>::computeQpOffDiagJacobian(unsigned int jvar)
90{
91 Real power;
92 power = 0;
93
94 if (!_v_eq_u && jvar == _v_id)
95 power += 1;
96
97 if (!_w_eq_u && jvar == _w_id)
98 power += 1;
99
100 return -_test[_i][_qp] * _stoichiometric_coeff *
101 raw_value(_reaction_coeff[_qp] * std::exp(_v[_qp] + _w[_qp])) * power * _phi[_j][_qp];
102}
