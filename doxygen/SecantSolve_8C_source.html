77void
78SecantSolve::saveVariableValues(const bool primary)
79{
80 TagID fxn_m1_tagid;
81 TagID xn_m1_tagid;
82 TagID fxn_m2_tagid;
83 TagID xn_m2_tagid;
84 if (primary)
85 {
86 fxn_m1_tagid = _fxn_m1_tagid;
87 xn_m1_tagid = _xn_m1_tagid;
88 fxn_m2_tagid = _fxn_m2_tagid;
89 xn_m2_tagid = _xn_m2_tagid;
90 }
91 else
92 {
93 fxn_m1_tagid = _secondary_fxn_m1_tagid;
94 xn_m1_tagid = _secondary_xn_m1_tagid;
95 fxn_m2_tagid = _secondary_fxn_m2_tagid;
96 xn_m2_tagid = _secondary_xn_m2_tagid;
97 }
98
99 // Save previous variable values
100 NumericVector<Number> & solution = _nl.solution();
101 NumericVector<Number> & fxn_m1 = _nl.getVector(fxn_m1_tagid);
102 NumericVector<Number> & xn_m1 = _nl.getVector(xn_m1_tagid);
103 NumericVector<Number> & fxn_m2 = _nl.getVector(fxn_m2_tagid);
104 NumericVector<Number> & xn_m2 = _nl.getVector(xn_m2_tagid);
105
106 // Advance one step
107 xn_m2 = xn_m1;
108 fxn_m2 = fxn_m1;
109
110 // Before a solve, solution is a sequence term, after a solve, solution is the evaluated term
111 xn_m1 = solution;
112}

114void
115SecantSolve::savePostprocessorValues(const bool primary)
116{
117 const std::vector<PostprocessorName> * transformed_pps;
118 std::vector<std::vector<PostprocessorValue>> * transformed_pps_values;
119 if (primary)
120 {
121 transformed_pps = &_transformed_pps;
122 transformed_pps_values = &_transformed_pps_values;
123 }
124 else
125 {
126 transformed_pps = &_secondary_transformed_pps;
127 transformed_pps_values = &_secondary_transformed_pps_values;
128 }
129
130 // Save previous postprocessor values
131 for (size_t i = 0; i < (*transformed_pps).size(); i++)
132 {
133 // Advance one step
134 (*transformed_pps_values)[i][3] = (*transformed_pps_values)[i][1];
135 (*transformed_pps_values)[i][2] = (*transformed_pps_values)[i][0];
136
137 // Save current value
138 // Primary: this is done before the timestep's solves and before timestep_begin transfers,
139 // so the value is the result of the previous Secant update (xn_m1)
140 // Secondary: this is done after the secondary solve, but before timestep_end postprocessors
141 // are computed, or timestep_end transfers are received.
142 // This value is the same as before the solve (xn_m1)
143 (*transformed_pps_values)[i][1] = getPostprocessorValueByName((*transformed_pps)[i]);
144 }
145}

157void
158SecantSolve::transformPostprocessors(const bool primary)
159{
160 Real relaxation_factor;
161 const std::vector<PostprocessorName> * transformed_pps;
162 std::vector<std::vector<PostprocessorValue>> * transformed_pps_values;
163 if (primary)
164 {
165 relaxation_factor = _relax_factor;
166 transformed_pps = &_transformed_pps;
167 transformed_pps_values = &_transformed_pps_values;
168 }
169 else
170 {
171 relaxation_factor = _secondary_relaxation_factor;
172 transformed_pps = &_secondary_transformed_pps;
173 transformed_pps_values = &_secondary_transformed_pps_values;
174 }
175
176 // Relax postprocessors for the main application
177 for (size_t i = 0; i < (*transformed_pps).size(); i++)
178 {
179 // Get new postprocessor value
180 const Real fxn_m1 = getPostprocessorValueByName((*transformed_pps)[i]);
181 const Real xn_m1 = (*transformed_pps_values)[i][1];
182 const Real fxn_m2 = (*transformed_pps_values)[i][2];
183 const Real xn_m2 = (*transformed_pps_values)[i][3];
184
185 // Save fxn_m1, received or computed before the solve
186 (*transformed_pps_values)[i][0] = fxn_m1;
187
188 // Compute and set relaxed value
189 Real new_value = fxn_m1;
190 if (!MooseUtils::absoluteFuzzyEqual(fxn_m1 - xn_m1 - fxn_m2 + xn_m2, 0))
191 new_value = xn_m1 - (fxn_m1 - xn_m1) * (xn_m1 - xn_m2) / (fxn_m1 - xn_m1 - fxn_m2 + xn_m2);
192
193 // Relax update if desired
194 new_value = relaxation_factor * new_value + (1 - relaxation_factor) * xn_m1;
195
196 _problem.setPostprocessorValueByName((*transformed_pps)[i], new_value);
197 }
198}

200void
201SecantSolve::transformVariables(const std::set<dof_id_type> & target_dofs, const bool primary)
202{
203 Real relaxation_factor;
204 TagID fxn_m1_tagid;
205 TagID xn_m1_tagid;
206 TagID fxn_m2_tagid;
207 TagID xn_m2_tagid;
208 if (primary)
209 {
210 relaxation_factor = _relax_factor;
211 fxn_m1_tagid = _fxn_m1_tagid;
212 xn_m1_tagid = _xn_m1_tagid;
213 fxn_m2_tagid = _fxn_m2_tagid;
214 xn_m2_tagid = _xn_m2_tagid;
215 }
216 else
217 {
218 relaxation_factor = _secondary_relaxation_factor;
219 fxn_m1_tagid = _secondary_fxn_m1_tagid;
220 xn_m1_tagid = _secondary_xn_m1_tagid;
221 fxn_m2_tagid = _secondary_fxn_m2_tagid;
222 xn_m2_tagid = _secondary_xn_m2_tagid;
223 }
224
225 NumericVector<Number> & solution = _nl.solution();
226 NumericVector<Number> & xn_m1 = _nl.getVector(xn_m1_tagid);
227 NumericVector<Number> & fxn_m2 = _nl.getVector(fxn_m2_tagid);
228 NumericVector<Number> & xn_m2 = _nl.getVector(xn_m2_tagid);
229
230 // Save the most recent evaluation of the coupled problem
231 NumericVector<Number> & fxn_m1 = _nl.getVector(fxn_m1_tagid);
232 fxn_m1 = solution;
233
234 for (const auto & dof : target_dofs)
235 {
236 // Avoid 0 denominator issue
237 Real new_value = fxn_m1(dof);
238 if (!MooseUtils::absoluteFuzzyEqual(solution(dof) - xn_m1(dof) - fxn_m2(dof) + xn_m2(dof), 0))
239 new_value = xn_m1(dof) - (solution(dof) - xn_m1(dof)) * (xn_m1(dof) - xn_m2(dof)) /
240 (solution(dof) - xn_m1(dof) - fxn_m2(dof) + xn_m2(dof));
241
242 // Relax update
243 new_value = relaxation_factor * new_value + (1 - relaxation_factor) * xn_m1(dof);
244
245 solution.set(dof, new_value);
246 }
247 solution.close();
248 _nl.update();
249}

251void
252SecantSolve::printFixedPointConvergenceHistory()
253{
254 _console << "\n 0 Secant initialization |R| = "
255 << Console::outputNorm(std::numeric_limits<Real>::max(), _fixed_point_initial_norm)
256 << '\n';
257
258 for (unsigned int i = 0; i <= _fixed_point_it; ++i)
259 {
260 Real max_norm =
261 std::max(_fixed_point_timestep_begin_norm[i], _fixed_point_timestep_end_norm[i]);
262 std::stringstream secant_prefix;
263 if (i < 2)
264 secant_prefix << " Secant initialization |R| = ";
265 else
266 secant_prefix << " Secant step |R| = ";
267
268 _console << std::setw(2) << i + 1 << secant_prefix.str()
269 << Console::outputNorm(_fixed_point_initial_norm, max_norm) << '\n';
270 }
