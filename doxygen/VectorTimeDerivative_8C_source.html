48void
49VectorTimeDerivative::computeJacobian()
50{
51 if (_lumping)
52 {
53 prepareMatrixTag(_assembly, _var.number(), _var.number());
54
55 precalculateJacobian();
56 for (_i = 0; _i < _test.size(); _i++)
57 for (_j = 0; _j < _phi.size(); _j++)
58 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
59 _local_ke(_i, _i) += _JxW[_qp] * _coord[_qp] * computeQpJacobian();
60
61 accumulateTaggedLocalMatrix();
62 }
63 else
64 VectorTimeKernel::computeJacobian();
65}
