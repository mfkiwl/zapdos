56ADReal
57ADEEDFElasticTownsendLog::computeQpResidual()
58{
59 ADReal electron_flux_mag = (std::exp(_em[_qp]) * (-_muem[_qp] * -_grad_potential[_qp] * _r_units -
60 _diffem[_qp] * _grad_em[_qp] * _r_units))
61 .norm();
62 ADReal Eel = -3.0 * _massem / _massGas[_qp] * 2.0 / 3. * std::exp(_u[_qp] - _em[_qp]);
63 ADReal el_term = _alpha[_qp] * std::exp(_target[_qp]) * electron_flux_mag * Eel;
64
65 return -_test[_i][_qp] * el_term;
66}
