22InputParameters
23PropertyReadFile::validParams()
24{
25 InputParameters params = GeneralUserObject::validParams();
26 params.addClassDescription("User Object to read property data from an external file and assign "
27 "to elements.");
28 params.addRequiredParam<FileName>("prop_file_name", "Name of the property file name");
29 params.addRequiredParam<unsigned int>("nprop", "Number of tabulated property values");
30 params.addParam<unsigned int>(
31 "nvoronoi", 0, "Number of voronoi tesselations/grains/nearest neighbor regions");
32 params.addDeprecatedParam<unsigned int>(
33 "ngrain", "Number of grains.", "ngrain is deprecated, use nvoronoi instead");
34 params.addParam<unsigned int>("nblock", 0, "Number of blocks");
35 params.addRequiredParam<MooseEnum>("read_type",
36 MooseEnum("element voronoi block node grain"),
37 "Type of property distribution: "
38 "element:by element "
39 "node: by node "
40 "voronoi:nearest neighbor / voronoi grain structure "
41 "block:by mesh block "
42 "grain: deprecated, use voronoi");
43 params.addParam<bool>("use_random_voronoi",
44 false,
45 "Wether to generate random positions for the Voronoi tesselation");
46 params.addParam<unsigned int>("rand_seed", 2000, "random seed");
47 params.addParam<MooseEnum>(
48 "rve_type",
49 MooseEnum("periodic none", "none"),
50 "Periodic or non-periodic grain distribution: Default is non-periodic");
51 params.addParam<bool>(
52 "use_zero_based_block_indexing", true, "Are the blocks numbered starting at zero?");
53 return params;
54}

57 : GeneralUserObject(parameters),
58 _prop_file_name(getParam<FileName>("prop_file_name")),
59 _reader(_prop_file_name),
60
61 _read_type(getParam<MooseEnum>("read_type").getEnum<ReadTypeEnum>()),
62 _use_random_tesselation(getParam<bool>("use_random_voronoi")),
63 _rand_seed(getParam<unsigned int>("rand_seed")),

65 _block_zero(getParam<bool>("use_zero_based_block_indexing")),

69 _nprop(getParam<unsigned int>("nprop")),
70 _nvoronoi(isParamValid("ngrain") ? getParam<unsigned int>("ngrain")
71 : getParam<unsigned int>("nvoronoi")),
72 _nblock(getParam<unsigned int>("nblock"))
73{
74 if (!_use_random_tesselation && parameters.isParamSetByUser("rand_seed"))
75 paramError("rand_seed",
76 "Random seeds should only be provided if random tesselation is desired");
77
78 Point mesh_min, mesh_max;
79 for (unsigned int i : make_range(Moose::dim))
80 {
81 mesh_min(i) = _mesh.getMinInDimension(i);
82 mesh_max(i) = _mesh.getMaxInDimension(i);
83 }
84 _bounding_box = MooseUtils::buildBoundingBox(mesh_min, mesh_max);
85
86 readData();
87}

89void
90PropertyReadFile::readData()
91{
92 if (_read_type == ReadTypeEnum::ELEMENT && _mesh.getMesh().allow_renumbering())
93 mooseWarning("CSV data is sorted by element, but mesh element renumbering is on, be careful!");
94
95 _reader.setFormatFlag(MooseUtils::DelimitedFileReader::FormatFlag::ROWS);
96 _reader.read();
97
98 unsigned int nobjects = 0;
99
100 switch (_read_type)
101 {
102 case ReadTypeEnum::ELEMENT:
103 nobjects = _mesh.nElem();
104 break;
105
106 case ReadTypeEnum::NODE:
107 nobjects = _mesh.nNodes();
108 break;
109
110 case ReadTypeEnum::VORONOI:
111 if (_nvoronoi <= 0)
112 paramError("nvoronoi", "Provide non-zero number of voronoi tesselations/grains.");
113 nobjects = _nvoronoi;
114 break;
115
116 // TODO Delete after Grizzly update see idaholab/Grizzly#182
117 case ReadTypeEnum::GRAIN:
118 if (_nvoronoi <= 0)
119 paramError("ngrain", "Provide non-zero number of voronoi tesselations/grains.");
120 nobjects = _nvoronoi;
121 break;
122
123 case ReadTypeEnum::BLOCK:
124 if (_nblock <= 0)
125 paramError("nblock", "Provide non-zero number of blocks.");
126 nobjects = _nblock;
127 break;
128 }
129
130 // make sure the data from file has enough rows and columns
131 if (_reader.getData().size() < nobjects)
132 mooseError(
133 "Data in ", _prop_file_name, " does not have enough rows for ", nobjects, " objects.");
134 if (_reader.getData().size() > nobjects)
135 mooseWarning("Data size in ",
136 _prop_file_name,
137 " is larger than ",
138 nobjects,
139 " objects, some data will not be used.");
140 for (unsigned int i = 0; i < nobjects; i++)
141 if (_reader.getData(i).size() < _nprop)
142 mooseError("Row ", i, " in ", _prop_file_name, " has number of data less than ", _nprop);
143
144 if (_read_type == ReadTypeEnum::VORONOI || _read_type == ReadTypeEnum::GRAIN)
145 initVoronoiCenterPoints();
146}

148void
149PropertyReadFile::initVoronoiCenterPoints()
150{
151 _center.resize(_nvoronoi);
152
153 // Generate a random tesselation
154 if (_use_random_tesselation)
155 {
156 MooseRandom::seed(_rand_seed);
157 for (const auto i : make_range(_nvoronoi))
158 for (const auto j : make_range(Moose::dim))
159 _center[i](j) = _bounding_box.min()(j) +
160 MooseRandom::rand() * (_bounding_box.max() - _bounding_box.min())(j);
161 }
162 // Read tesselation from file
163 else
164 {
165 for (const auto i : make_range(_nvoronoi))
166 for (const auto j : make_range(_mesh.dimension()))
167 _center[i](j) = _reader.getData(i)[j];
168 }
169}

171Real
172PropertyReadFile::getData(const Elem * const elem, const unsigned int prop_num) const
173{
174 if (prop_num > _nprop)
175 paramError(
176 "nprop", "Property number ", prop_num, " greater than total number of properties ", _nprop);
177
178 Real data = 0.0;
179 switch (_read_type)
180 {
181 case ReadTypeEnum::ELEMENT:
182 data = getElementData(elem, prop_num);
183 break;
184
185 case ReadTypeEnum::VORONOI:
186 data = getVoronoiData(elem->vertex_average(), prop_num);
187 break;
188
189 // TODO: Delete after Grizzly update
190 case ReadTypeEnum::GRAIN:
191 data = getVoronoiData(elem->vertex_average(), prop_num);
192 break;
193
194 case ReadTypeEnum::BLOCK:
195 data = getBlockData(elem, prop_num);
196 break;
197
198 case ReadTypeEnum::NODE:
199 mooseError(
200 "PropertyReadFile has data sorted by node, it should note be retrieved by element");
201 break;
202 }
203 return data;
204}

232Real
233PropertyReadFile::getBlockData(const Elem * elem, unsigned int prop_num) const
234{
235 unsigned int offset = 0;
236 if (!_block_zero)
237 offset = 1;
238
239 unsigned int elem_subdomain_id = elem->subdomain_id();
240 if (elem_subdomain_id >= _nblock + offset)
241 paramError("nblock",
242 "Element block id ",
243 elem_subdomain_id,
244 " greater than than total number of blocks in mesh: ",
245 _nblock,
246 ". Blocks should be numbered consecutively.");
247 return _reader.getData(elem_subdomain_id - offset)[prop_num];

250Real
251PropertyReadFile::getVoronoiData(const Point & point, const unsigned int prop_num) const
252{
253 Real min_dist = std::numeric_limits<Real>::max();
254 unsigned int ivoronoi = 0;
255
256 for (const auto i : make_range(_nvoronoi))
257 {
258 Real dist = 0.0;
259 switch (_rve_type)
260 {
261 case 0:
262 // Calculates minimum periodic distance when "periodic" is specified
263 // for rve_type
264 dist = minPeriodicDistance(_center[i], point);
265 break;
266
267 default:
268 // Calculates minimum distance when nothing is specified
269 // for rve_type
270 Point dist_vec = _center[i] - point;
271 dist = dist_vec.norm();
272 }
273
274 if (dist < min_dist)
275 {
276 min_dist = dist;
277 ivoronoi = i;
278 }
279 }
280 return _reader.getData(ivoronoi)[prop_num];
281}

283// TODO: this should probably use the built-in min periodic distance!
284Real
285PropertyReadFile::minPeriodicDistance(const Point & c, const Point & p) const
286{
287 Point dist_vec = c - p;
288 Real min_dist = dist_vec.norm();
289
290 Real fac[3] = {-1.0, 0.0, 1.0};
291 for (unsigned int i = 0; i < 3; i++)
292 for (unsigned int j = 0; j < 3; j++)
293 for (unsigned int k = 0; k < 3; k++)
294 {
295 Point p1;
296 p1(0) = p(0) + fac[i] * (_bounding_box.max() - _bounding_box.min())(0);
297 p1(1) = p(1) + fac[j] * (_bounding_box.max() - _bounding_box.min())(1);
298 p1(2) = p(2) + fac[k] * (_bounding_box.max() - _bounding_box.min())(2);
299
300 dist_vec = c - p1;
301 Real dist = dist_vec.norm();
302
303 if (dist < min_dist)
304 min_dist = dist;
305 }
306
307 return min_dist;
308}
