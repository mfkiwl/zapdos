53void
54CurrentDensityShapeSideUserObject::initialize()
55{
56 _integral = 0.0;
57
58 // Jacobian term storage is up to the user. One option is using an std::vector
59 // We resize it to the total number of DOFs in the system and zero it out.
60 // WARNING: this can be large number (smart sparse storage could be a future improvement)
61 if (_fe_problem.currentlyComputingJacobian())
62 _jacobian_storage.assign(_subproblem.es().n_dofs(), 0.0);
63}

65void
66CurrentDensityShapeSideUserObject::execute()
67{
68 for (unsigned int qp = 0; qp < _qrule->n_points(); ++qp)
69 {
70 RealVectorValue ion_current =
71 _e * (_muip[qp].value() * -_grad_potential[qp] * std::exp(_ip[qp]) -
72 _diffip[qp].value() * std::exp(_ip[qp]) * _grad_ip[qp]);
73 RealVectorValue electron_current =
74 -_e * (-_muem[qp].value() * -_grad_potential[qp] * std::exp(_em[qp]) -
75 _diffem[qp].value() * std::exp(_em[qp]) * _grad_em[qp]);
76 Real outgoing_current = _normals[qp] * (ion_current + electron_current);
77 if (_use_moles)
78 outgoing_current *= _avogadro;
79
80 _integral += _JxW[qp] * _coord[qp] * outgoing_current;
81 }
82}

84void
85CurrentDensityShapeSideUserObject::executeJacobian(unsigned int jvar)
86{
87 // derivative of _integral w.r.t. ip_j
88 if (jvar == _ip_id)
89 {
90 Real sum = 0.0;
91 for (unsigned int qp = 0; qp < _qrule->n_points(); ++qp)
92 {
93 RealVectorValue d_ion_current_d_ip =
94 _e * (_muip[qp].value() * -_grad_potential[qp] * std::exp(_ip[qp]) * _phi[_j][qp] -
95 _diffip[qp].value() * (std::exp(_ip[qp]) * _phi[_j][qp] * _grad_ip[qp] +
96 std::exp(_ip[qp]) * _grad_phi[_j][qp]));
97 if (_use_moles)
98 d_ion_current_d_ip *= _avogadro;
99 sum += _JxW[qp] * _coord[qp] * _normals[qp] * d_ion_current_d_ip;
100 }
101
102 _jacobian_storage[_j_global] += sum;
103 }
104
105 // derivative of _integral w.r.t. em_j
106 else if (jvar == _em_id)
107 {
108 Real sum = 0.0;
109 for (unsigned int qp = 0; qp < _qrule->n_points(); ++qp)
110 {
111 Real d_actual_mean_en_d_em = std::exp(_mean_en[qp] - _em[qp]) * -_phi[_j][qp];
112 Real d_muem_d_em = _muem[qp].derivatives()[jvar] * d_actual_mean_en_d_em;
113 Real d_diffem_d_em = _diffem[qp].derivatives()[jvar] * d_actual_mean_en_d_em;
114
115 RealVectorValue d_electron_current_d_em =
116 -_e * (-_muem[qp].value() * -_grad_potential[qp] * std::exp(_em[qp]) * _phi[_j][qp] -
117 d_muem_d_em * -_grad_potential[qp] * std::exp(_em[qp]) -
118 _diffem[qp].value() * (std::exp(_em[qp]) * _phi[_j][qp] * _grad_em[qp] +
119 std::exp(_em[qp]) * _grad_phi[_j][qp]) -
120 d_diffem_d_em * std::exp(_em[qp]) * _grad_em[qp]);
121 if (_use_moles)
122 d_electron_current_d_em *= _avogadro;
123 sum += _JxW[qp] * _coord[qp] * _normals[qp] * d_electron_current_d_em;
124 }
125
126 _jacobian_storage[_j_global] += sum;
127 }
128
129 // derivative of _integral w.r.t. potential_j
130 else if (jvar == _potential_id)
131 {
132 Real sum = 0.0;
133 for (unsigned int qp = 0; qp < _qrule->n_points(); ++qp)
134 {
135 RealVectorValue d_ion_current_d_potential =
136 _e * (_muip[qp].value() * -_grad_phi[_j][qp] * std::exp(_ip[qp]));
137 RealVectorValue d_electron_current_d_potential =
138 -_e * (-_muem[qp].value() * -_grad_phi[_j][qp] * std::exp(_em[qp]));
139 Real d_outgoing_current_d_potential =
140 _normals[qp] * (d_ion_current_d_potential + d_electron_current_d_potential);
141 if (_use_moles)
142 d_outgoing_current_d_potential *= _avogadro;
143 sum += _JxW[qp] * _coord[qp] * d_outgoing_current_d_potential;
144 }
145
146 _jacobian_storage[_j_global] += sum;
147 }
148
149 // derivative of _integral w.r.t. mean_en_j
150 else if (jvar == _mean_en_id)
151 {
152 Real sum = 0.0;
153 for (unsigned int qp = 0; qp < _qrule->n_points(); ++qp)
154 {
155 Real d_actual_mean_en_d_mean_en = std::exp(_mean_en[qp] - _em[qp]) * _phi[_j][qp];
156 Real d_muem_d_mean_en = _muem[qp].derivatives()[jvar] * d_actual_mean_en_d_mean_en;
157 Real d_diffem_d_mean_en = _diffem[qp].derivatives()[jvar] * d_actual_mean_en_d_mean_en;
158
159 RealVectorValue d_electron_current_d_mean_en =
160 -_e * (-d_muem_d_mean_en * -_grad_potential[qp] * std::exp(_em[qp]) -
161 d_diffem_d_mean_en * std::exp(_em[qp]) * _grad_em[qp]);
162 if (_use_moles)
163 d_electron_current_d_mean_en *= _avogadro;
164 sum += _JxW[qp] * _coord[qp] * _normals[qp] * d_electron_current_d_mean_en;
165 }
166
167 _jacobian_storage[_j_global] += sum;
168 }
169}

171void
172CurrentDensityShapeSideUserObject::finalize()
173{
174 gatherSum(_integral);
175
176 if (_fe_problem.currentlyComputingJacobian())
177 gatherSum(_jacobian_storage);
178}

180void
181CurrentDensityShapeSideUserObject::threadJoin(const UserObject & y)
182{
183 const CurrentDensityShapeSideUserObject & shp_uo =
184 dynamic_cast<const CurrentDensityShapeSideUserObject &>(y);
185 _integral += shp_uo._integral;
186
187 if (_fe_problem.currentlyComputingJacobian())
188 {
189 mooseAssert(_jacobian_storage.size() == shp_uo._jacobian_storage.size(),
190 "Jacobian storage size is inconsistent across threads");
191 for (unsigned int i = 0; i < _jacobian_storage.size(); ++i)
192 _jacobian_storage[i] += shp_uo._jacobian_storage[i];
193 }
194}
