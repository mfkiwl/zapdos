287
288 const std::set<SubdomainID> & subdomains = other_mesh.meshSubdomains();
289 for (const auto & sbd_id : subdomains)

291
292 // Get references to BoundaryInfo objects to make the code below cleaner...
293 const BoundaryInfo & other_boundary_info = other_mesh.getMesh().get_boundary_info();
294 BoundaryInfo & boundary_info = getMesh().get_boundary_info();
295
296 // Use the other BoundaryInfo object to build the list of side boundary ids
297 std::vector<BoundaryID> side_boundaries;
298 other_boundary_info.build_side_boundary_ids(side_boundaries);
299
300 // Assign those boundary ids in our BoundaryInfo object
301 for (const auto & side_bnd_id : side_boundaries)
302 boundary_info.sideset_name(side_bnd_id) = other_boundary_info.get_sideset_name(side_bnd_id);
303
304 // Do the same thing for node boundary ids
305 std::vector<BoundaryID> node_boundaries;
306 other_boundary_info.build_node_boundary_ids(node_boundaries);
307
308 for (const auto & node_bnd_id : node_boundaries)
309 boundary_info.nodeset_name(node_bnd_id) = other_boundary_info.get_nodeset_name(node_bnd_id);
310
311 _bounds.resize(other_mesh._bounds.size());
312 for (std::size_t i = 0; i < _bounds.size(); ++i)
313 {
314 _bounds[i].resize(other_mesh._bounds[i].size());
315 for (std::size_t j = 0; j < _bounds[i].size(); ++j)
316 _bounds[i][j] = other_mesh._bounds[i][j];
317 }
318
319 updateCoordTransform();
320}

360bool
361MooseMesh::prepare(const MeshBase * const mesh_to_clone)
362{
363 TIME_SECTION("prepare", 2, "Preparing Mesh", true);
364
365 bool called_prepare_for_use = false;
366
367 mooseAssert(_mesh, "The MeshBase has not been constructed");
368
369 if (!dynamic_cast<DistributedMesh *>(&getMesh()) || _is_nemesis)
370 // For whatever reason we do not want to allow renumbering here nor ever in the future?
371 getMesh().allow_renumbering(false);
372
373 if (mesh_to_clone)
374 {
375 mooseAssert(mesh_to_clone->is_prepared(),
376 "The mesh we wish to clone from must already be prepared");
377 _mesh = mesh_to_clone->clone();
378 _moose_mesh_prepared = false;
379 }
380 else if (!_mesh->is_prepared())
381 {
382 _mesh->prepare_for_use();
383 _moose_mesh_prepared = false;
384 called_prepare_for_use = true;
385 }
386
387 if (_moose_mesh_prepared)
388 return called_prepare_for_use;
389
390 // Collect (local) subdomain IDs
391 _mesh_subdomains.clear();
392 for (const auto & elem : getMesh().element_ptr_range())
393 _mesh_subdomains.insert(elem->subdomain_id());
394
395 // add explicitly requested subdomains
396 if (isParamValid("add_subdomain_ids"))
397 {
398 const auto add_subdomain_id = getParam<std::vector<SubdomainID>>("add_subdomain_ids");
399 _mesh_subdomains.insert(add_subdomain_id.begin(), add_subdomain_id.end());
400 }
401
402 // Make sure nodesets have been generated
403 buildNodeListFromSideList();
404
405 // Collect (local) boundary IDs
406 const std::set<BoundaryID> & local_bids = getMesh().get_boundary_info().get_boundary_ids();
407 _mesh_boundary_ids.insert(local_bids.begin(), local_bids.end());
408
409 const std::set<BoundaryID> & local_node_bids =
410 getMesh().get_boundary_info().get_node_boundary_ids();
411 _mesh_nodeset_ids.insert(local_node_bids.begin(), local_node_bids.end());
412
413 const std::set<BoundaryID> & local_side_bids =
414 getMesh().get_boundary_info().get_side_boundary_ids();
415 _mesh_sideset_ids.insert(local_side_bids.begin(), local_side_bids.end());
416
417 // Communicate subdomain and boundary IDs if this is a parallel mesh
418 if (!getMesh().is_serial())
419 {
420 _communicator.set_union(_mesh_subdomains);
421 _communicator.set_union(_mesh_boundary_ids);
422 _communicator.set_union(_mesh_nodeset_ids);
423 _communicator.set_union(_mesh_sideset_ids);
424 }
425
426 if (!_built_from_other_mesh)
427 {
428 if (!_coord_system_set)
429 setCoordSystem(_provided_coord_blocks, getParam<MultiMooseEnum>("coord_type"));
430 else if (_pars.isParamSetByUser("coord_type"))
431 mooseError(
432 "Trying to set coordinate system type information based on the user input file, but "
433 "the coordinate system type information has already been set programmatically! "
434 "Either remove your coordinate system type information from the input file, or contact "
435 "your application developer");
436 }
437
438 // Set general axisymmetric axes if provided
439 if (isParamValid("rz_coord_blocks") && isParamValid("rz_coord_origins") &&
440 isParamValid("rz_coord_directions"))
441 {
442 const auto rz_coord_blocks = getParam<std::vector<SubdomainName>>("rz_coord_blocks");
443 const auto rz_coord_origins = getParam<std::vector<Point>>("rz_coord_origins");
444 const auto rz_coord_directions = getParam<std::vector<RealVectorValue>>("rz_coord_directions");
445 if (rz_coord_origins.size() == rz_coord_blocks.size() &&
446 rz_coord_directions.size() == rz_coord_blocks.size())
447 {
448 std::vector<std::pair<Point, RealVectorValue>> rz_coord_axes;
449 for (unsigned int i = 0; i < rz_coord_origins.size(); ++i)
450 rz_coord_axes.push_back(std::make_pair(rz_coord_origins[i], rz_coord_directions[i]));
451
452 setGeneralAxisymmetricCoordAxes(rz_coord_blocks, rz_coord_axes);
453
454 if (isParamSetByUser("rz_coord_axis"))
455 mooseError("The parameter 'rz_coord_axis' may not be provided if 'rz_coord_blocks', "
456 "'rz_coord_origins', and 'rz_coord_directions' are provided.");
457 }
458 else
459 mooseError("The parameters 'rz_coord_blocks', 'rz_coord_origins', and "
460 "'rz_coord_directions' must all have the same size.");
461 }
462 else if (isParamValid("rz_coord_blocks") || isParamValid("rz_coord_origins") ||
463 isParamValid("rz_coord_directions"))
464 mooseError("If any of the parameters 'rz_coord_blocks', 'rz_coord_origins', and "
465 "'rz_coord_directions' are provided, then all must be provided.");
466
467 detectOrthogonalDimRanges();
468
469 update();
470
471 // Check if there is subdomain name duplication for the same subdomain ID
472 checkDuplicateSubdomainNames();
473
474 _moose_mesh_prepared = true;
475
476 return called_prepare_for_use;
477}

501void
502MooseMesh::buildLowerDMesh()
503{
504 auto & mesh = getMesh();
505
506 if (!mesh.is_serial())
507 mooseError(
508 "Hybrid finite element method must use replicated mesh.\nCurrently lower-dimensional mesh "
509 "does not support mesh re-partitioning and a debug assertion being hit related with "
510 "neighbors of lower-dimensional element, with distributed mesh.");
511
512 // Lower-D element build requires neighboring element information
513 if (!mesh.is_prepared())
514 mesh.find_neighbors();
515
516 // maximum number of sides of all elements
517 unsigned int max_n_sides = 0;
518
519 // remove existing lower-d element first
520 std::set<Elem *> deleteable_elems;
521 for (auto & elem : mesh.element_ptr_range())
522 if (elem->subdomain_id() == Moose::INTERNAL_SIDE_LOWERD_ID ||
523 elem->subdomain_id() == Moose::BOUNDARY_SIDE_LOWERD_ID)
524 deleteable_elems.insert(elem);
525 else if (elem->n_sides() > max_n_sides)
526 max_n_sides = elem->n_sides();
527
528 for (auto & elem : deleteable_elems)
529 mesh.delete_elem(elem);
530
531 mesh.comm().max(max_n_sides);
532
533 deleteable_elems.clear();
534
535 dof_id_type max_elem_id = mesh.max_elem_id();
536 unique_id_type max_unique_id = mesh.parallel_max_unique_id();
537
538 std::vector<Elem *> side_elems;
539 _higher_d_elem_side_to_lower_d_elem.clear();
540 for (const auto & elem : mesh.active_element_ptr_range())
541 {
542 // skip existing lower-d elements
543 if (elem->interior_parent())
544 continue;
545
546 for (const auto side : elem->side_index_range())
547 {
548 Elem * neig = elem->neighbor_ptr(side);
549
550 bool build_side = false;
551 if (!neig)
552 build_side = true;
553 else
554 {
555 mooseAssert(!neig->is_remote(), "We error if the mesh is not serial");
556 if (!neig->active())
557 build_side = true;
558 else if (neig->level() == elem->level() && elem->id() < neig->id())
559 build_side = true;
560 }
561
562 if (build_side)
563 {
564 std::unique_ptr<Elem> side_elem(elem->build_side_ptr(side, false));
565
566 // The side will be added with the same processor id as the parent.
567 side_elem->processor_id() = elem->processor_id();
568
569 // Add subdomain ID
570 if (neig)
571 side_elem->subdomain_id() = Moose::INTERNAL_SIDE_LOWERD_ID;
572 else
573 side_elem->subdomain_id() = Moose::BOUNDARY_SIDE_LOWERD_ID;
574
575 // set ids consistently across processors (these ids will be temporary)
576 side_elem->set_id(max_elem_id + elem->id() * max_n_sides + side);
577 side_elem->set_unique_id(max_unique_id + elem->id() * max_n_sides + side);
578
579 // Also assign the side's interior parent, so it is always
580 // easy to figure out the Elem we came from.
581 // Note: the interior parent could be a ghost element.
582 side_elem->set_interior_parent(elem);
583
584 side_elems.push_back(side_elem.release());
585
586 // add link between higher d element to lower d element
587 auto pair = std::make_pair(elem, side);
588 auto link = std::make_pair(pair, side_elems.back());
589 auto ilink = std::make_pair(side_elems.back(), side);
590 _lower_d_elem_to_higher_d_elem_side.insert(ilink);
591 _higher_d_elem_side_to_lower_d_elem.insert(link);
592 }
593 }
594 }
595
596 // finally, add the lower-dimensional element to the mesh
597 // Note: lower-d interior element will exist on a processor if its associated interior
598 // parent exists on a processor whether or not being a ghost. Lower-d elements will
599 // get its interior parent's processor id.
600 for (auto & elem : side_elems)
601 mesh.add_elem(elem);
602
603 _mesh_subdomains.insert(Moose::INTERNAL_SIDE_LOWERD_ID);
604 mesh.subdomain_name(Moose::INTERNAL_SIDE_LOWERD_ID) = "INTERNAL_SIDE_LOWERD_SUBDOMAIN";

855void
856MooseMesh::buildElemIDInfo()
857{
858 unsigned int n = getMesh().n_elem_integers() + 1;
859
860 _block_id_mapping.clear();
861 _max_ids.clear();
862 _min_ids.clear();
863 _id_identical_flag.clear();
864
865 _block_id_mapping.resize(n);
866 _max_ids.resize(n, std::numeric_limits<dof_id_type>::min());
867 _min_ids.resize(n, std::numeric_limits<dof_id_type>::max());
868 _id_identical_flag.resize(n, std::vector<bool>(n, true));
869 for (const auto & elem : getMesh().active_local_element_ptr_range())
870 for (unsigned int i = 0; i < n; ++i)
871 {
872 auto id = (i == n - 1 ? elem->subdomain_id() : elem->get_extra_integer(i));
873 _block_id_mapping[i][elem->subdomain_id()].insert(id);
874 if (id > _max_ids[i])
875 _max_ids[i] = id;
876 if (id < _min_ids[i])
877 _min_ids[i] = id;
878 for (unsigned int j = 0; j < n; ++j)
879 {
880 auto idj = (j == n - 1 ? elem->subdomain_id() : elem->get_extra_integer(j));
881 if (i != j && _id_identical_flag[i][j] && id != idj)
882 _id_identical_flag[i][j] = false;
883 }
884 }
885
886 for (unsigned int i = 0; i < n; ++i)
887 {
888 for (auto & blk : meshSubdomains())
889 comm().set_union(_block_id_mapping[i][blk]);
890 comm().min(_id_identical_flag[i]);
891 }
892 comm().max(_max_ids);
893 comm().min(_min_ids);
894}

896std::unordered_map<dof_id_type, std::set<dof_id_type>>
897MooseMesh::getElemIDMapping(const std::string & from_id_name, const std::string & to_id_name) const
898{
899 auto & mesh_base = getMesh();
900
901 if (!mesh_base.has_elem_integer(from_id_name))
902 mooseError("Mesh does not have the element integer name '", from_id_name, "'");
903 if (!mesh_base.has_elem_integer(to_id_name))
904 mooseError("Mesh does not have the element integer name '", to_id_name, "'");
905
906 const auto id1 = mesh_base.get_elem_integer_index(from_id_name);
907 const auto id2 = mesh_base.get_elem_integer_index(to_id_name);
908
909 std::unordered_map<dof_id_type, std::set<dof_id_type>> id_map;
910 for (const auto id : getAllElemIDs(id1))
911 id_map[id] = std::set<dof_id_type>();
912
913 for (const auto & elem : mesh_base.active_local_element_ptr_range())
914 id_map[elem->get_extra_integer(id1)].insert(elem->get_extra_integer(id2));
915
916 for (auto & [id, ids] : id_map)
917 {
918 libmesh_ignore(id); // avoid overzealous gcc 9.4 unused var warning
919 comm().set_union(ids);
920 }
921
922 return id_map;
923}

1169bool
1170MooseMesh::isBoundaryFullyExternalToSubdomains(BoundaryID bid,
1171 const std::set<SubdomainID> & blk_group) const
1172{
1173 mooseAssert(_bnd_elem_range, "Boundary element range is not initialized");
1174
1175 // Loop over all side elements of the mesh, select those on the boundary
1176 for (const auto & bnd_elem : *_bnd_elem_range)
1177 {
1178 const auto & [elem_ptr, elem_side, elem_bid] = *bnd_elem;
1179 if (elem_bid == bid)
1180 {
1181 // If an element is internal to the group of subdomain, check the neighbor
1182 if (blk_group.find(elem_ptr->subdomain_id()) != blk_group.end())
1183 {
1184 const auto * const neighbor = elem_ptr->neighbor_ptr(elem_side);
1185
1186 // If we did not ghost the neighbor, we cannot decide
1187 if (neighbor == libMesh::remote_elem)
1188 mooseError("Insufficient level of geometrical ghosting to determine "
1189 "if a boundary is internal to the mesh");
1190 // If the neighbor does not exist, then we are on the edge of the mesh
1191 if (!neighbor)
1192 continue;
1193 // If the neighbor is also in the group of subdomain,
1194 // then the boundary cuts the subdomains
1195 if (blk_group.find(neighbor->subdomain_id()) != blk_group.end())
1196 return false;
1197 }
1198 }
1199 }
1200 return true;

1203void
1204MooseMesh::cacheInfo()
1205{
1206 TIME_SECTION("cacheInfo", 3);
1207
1208 _sub_to_neighbor_subs.clear();
1209 _subdomain_boundary_ids.clear();
1210 _neighbor_subdomain_boundary_ids.clear();
1211 _block_node_list.clear();
1212 _higher_d_elem_side_to_lower_d_elem.clear();
1213
1214 // TODO: Thread this!
1215 for (const auto & elem : getMesh().element_ptr_range())
1216 {
1217 const Elem * ip_elem = elem->interior_parent();
1218
1219 if (ip_elem)
1220 {
1221 unsigned int ip_side = ip_elem->which_side_am_i(elem);
1222
1223 // For some grid sequencing tests: ip_side == libMesh::invalid_uint
1224 if (ip_side != libMesh::invalid_uint)
1225 {
1226 auto pair = std::make_pair(ip_elem, ip_side);
1227 _higher_d_elem_side_to_lower_d_elem.insert(
1228 std::pair<std::pair<const Elem *, unsigned short int>, const Elem *>(pair, elem));
1229 }
1230 }
1231
1232 for (unsigned int nd = 0; nd < elem->n_nodes(); ++nd)
1233 {
1234 Node & node = *elem->node_ptr(nd);
1235 _block_node_list[node.id()].insert(elem->subdomain_id());
1236 }
1237 }
1238
1239 for (const auto & elem : getMesh().active_local_element_ptr_range())
1240 {
1241 SubdomainID subdomain_id = elem->subdomain_id();
1242 for (unsigned int side = 0; side < elem->n_sides(); side++)
1243 {
1244 std::vector<BoundaryID> boundary_ids = getBoundaryIDs(elem, side);
1245 std::set<BoundaryID> & subdomain_set = _subdomain_boundary_ids[subdomain_id];
1246
1247 subdomain_set.insert(boundary_ids.begin(), boundary_ids.end());
1248
1249 Elem * neig = elem->neighbor_ptr(side);
1250 if (neig)
1251 {
1252 _neighbor_subdomain_boundary_ids[neig->subdomain_id()].insert(boundary_ids.begin(),
1253 boundary_ids.end());
1254 SubdomainID neighbor_subdomain_id = neig->subdomain_id();
1255 if (neighbor_subdomain_id != subdomain_id)
1256 _sub_to_neighbor_subs[subdomain_id].insert(neighbor_subdomain_id);
1257 }
1258 }
1259 }
1260

1381Node *
1382MooseMesh::addQuadratureNode(const Elem * elem,
1383 const unsigned short int side,
1384 const unsigned int qp,
1385 BoundaryID bid,
1386 const Point & point)
1387{
1388 Node * qnode;
1389
1390 if (_elem_to_side_to_qp_to_quadrature_nodes[elem->id()][side].find(qp) ==
1391 _elem_to_side_to_qp_to_quadrature_nodes[elem->id()][side].end())
1392 {
1393 // Create a new node id starting from the max node id and counting down. This will be the least
1394 // likely to collide with an existing node id.
1395 // Note that we are using numeric_limits<unsigned>::max even
1396 // though max_id is stored as a dof_id_type. I tried this with
1397 // numeric_limits<dof_id_type>::max and it broke several tests in
1398 // MOOSE. So, this is some kind of a magic number that we will
1399 // just continue to use...
1400 dof_id_type max_id = std::numeric_limits<unsigned int>::max() - 100;
1401 dof_id_type new_id = max_id - _quadrature_nodes.size();
1402
1403 if (new_id <= getMesh().max_node_id())
1404 mooseError("Quadrature node id collides with existing node id!");
1405
1406 qnode = new Node(point, new_id);
1407
1408 // Keep track of this new node in two different ways for easy lookup
1409 _quadrature_nodes[new_id] = qnode;
1410 _elem_to_side_to_qp_to_quadrature_nodes[elem->id()][side][qp] = qnode;
1411
1412 if (elem->active())
1413 {
1414 _node_to_elem_map[new_id].push_back(elem->id());
1415 _node_to_active_semilocal_elem_map[new_id].push_back(elem->id());
1416 }
1417 }
1418 else
1419 qnode = _elem_to_side_to_qp_to_quadrature_nodes[elem->id()][side][qp];
1420
1421 BndNode * bnode = new BndNode(qnode, bid);
1422 _bnd_nodes.push_back(bnode);
1423 _bnd_node_ids[bid].insert(qnode->id());
1424
1425 _extra_bnd_nodes.push_back(*bnode);
1426
1427 // Do this so the range will be regenerated next time it is accessed
1428 _bnd_node_range.reset();
1429
1430 return qnode;
1431}

1576void
1577MooseMesh::buildPeriodicNodeMap(std::multimap<dof_id_type, dof_id_type> & periodic_node_map,
1578 unsigned int var_number,
1579 PeriodicBoundaries * pbs) const
1580{
1581 TIME_SECTION("buildPeriodicNodeMap", 5);
1582
1583 // clear existing map
1584 periodic_node_map.clear();
1585
1586 // get periodic nodes
1587 std::vector<PeriodicNodeInfo> periodic_nodes;
1588 for (const auto & t : getMesh().get_boundary_info().build_node_list())
1589 {
1590 // unfortunately libMesh does not give us a pointer, so we have to look it up ourselves
1591 auto node = _mesh->node_ptr(std::get<0>(t));
1592 mooseAssert(node != nullptr,
1593 "libMesh::BoundaryInfo::build_node_list() returned an ID for a non-existing node");
1594 auto bc_id = std::get<1>(t);
1595 periodic_nodes.emplace_back(node, bc_id);
1596 }
1597
1598 // sort by boundary id
1599 std::sort(periodic_nodes.begin(),
1600 periodic_nodes.end(),
1601 [](const PeriodicNodeInfo & a, const PeriodicNodeInfo & b) -> bool
1602 { return a.second > b.second; });
1603
1604 // build kd-tree
1605 using KDTreeType = nanoflann::KDTreeSingleIndexAdaptor<
1606 nanoflann::L2_Simple_Adaptor<Real, PointListAdaptor<PeriodicNodeInfo>, Real, std::size_t>,
1607 PointListAdaptor<PeriodicNodeInfo>,
1608 LIBMESH_DIM,
1609 std::size_t>;
1610 const unsigned int max_leaf_size = 20; // slightly affects runtime
1611 auto point_list =
1612 PointListAdaptor<PeriodicNodeInfo>(periodic_nodes.begin(), periodic_nodes.end());
1613 auto kd_tree = std::make_unique<KDTreeType>(
1614 LIBMESH_DIM, point_list, nanoflann::KDTreeSingleIndexAdaptorParams(max_leaf_size));
1615 mooseAssert(kd_tree != nullptr, "KDTree was not properly initialized.");
1616 kd_tree->buildIndex();
1617
1618 // data structures for kd-tree search
1619 nanoflann::SearchParameters search_params;
1620 std::vector<nanoflann::ResultItem<std::size_t, Real>> ret_matches;
1621
1622 // iterate over periodic nodes (boundary ids are in contiguous blocks)
1623 PeriodicBoundaryBase * periodic = nullptr;
1624 BoundaryID current_bc_id = BoundaryInfo::invalid_id;
1625 for (auto & pair : periodic_nodes)
1626 {
1627 // entering a new block of boundary IDs
1628 if (pair.second != current_bc_id)
1629 {
1630 current_bc_id = pair.second;
1631 periodic = pbs->boundary(current_bc_id);
1632 if (periodic && !periodic->is_my_variable(var_number))
1633 periodic = nullptr;
1634 }
1635
1636 // variable is not periodic at this node, skip
1637 if (!periodic)
1638 continue;
1639
1640 // clear result buffer
1641 ret_matches.clear();
1642
1643 // id of the current node
1644 const auto id = pair.first->id();
1645
1646 // position where we expect a periodic partner for the current node and boundary
1647 Point search_point = periodic->get_corresponding_pos(*pair.first);
1648
1649 // search at the expected point
1650 kd_tree->radiusSearch(&(search_point)(0), libMesh::TOLERANCE, ret_matches, search_params);
1651 for (auto & match_pair : ret_matches)
1652 {
1653 const auto & match = periodic_nodes[match_pair.first];
1654 // add matched node if the boundary id is the corresponding id in the periodic pair
1655 if (match.second == periodic->pairedboundary)
1656 periodic_node_map.emplace(id, match.first->id());
1657 }
1658 }
1659}

1661void
1662MooseMesh::buildPeriodicNodeSets(std::map<BoundaryID, std::set<dof_id_type>> & periodic_node_sets,
1663 unsigned int var_number,
1664 PeriodicBoundaries * pbs) const
1665{
1666 TIME_SECTION("buildPeriodicNodeSets", 5);
1667
1668 periodic_node_sets.clear();
1669
1670 // Loop over all the boundary nodes adding the periodic nodes to the appropriate set
1671 for (const auto & t : getMesh().get_boundary_info().build_node_list())
1672 {
1673 auto node_id = std::get<0>(t);
1674 auto bc_id = std::get<1>(t);
1675
1676 // Is this current node on a known periodic boundary?
1677 if (periodic_node_sets.find(bc_id) != periodic_node_sets.end())
1678 periodic_node_sets[bc_id].insert(node_id);
1679 else // This still might be a periodic node but we just haven't seen this boundary_id yet
1680 {
1681 const PeriodicBoundaryBase * periodic = pbs->boundary(bc_id);
1682 if (periodic && periodic->is_my_variable(var_number))
1683 periodic_node_sets[bc_id].insert(node_id);
1684 }
1685 }
1686}
1687
1688bool
1689MooseMesh::detectOrthogonalDimRanges(Real tol)
1690{
1691 TIME_SECTION("detectOrthogonalDimRanges", 5);
1692
1693 if (_regular_orthogonal_mesh)
1694 return true;
1695
1696 std::vector<Real> min(3, std::numeric_limits<Real>::max());
1697 std::vector<Real> max(3, std::numeric_limits<Real>::min());
1698 unsigned int dim = getMesh().mesh_dimension();
1699
1700 // Find the bounding box of our mesh
1701 for (const auto & node : getMesh().node_ptr_range())
1702 // Check all coordinates, we don't know if this mesh might be lying in a higher dim even if the
1703 // mesh dimension is lower.
1704 for (const auto i : make_range(Moose::dim))
1705 {
1706 if ((*node)(i) < min[i])
1707 min[i] = (*node)(i);
1708 if ((*node)(i) > max[i])
1709 max[i] = (*node)(i);
1710 }
1711
1712 this->comm().max(max);
1713 this->comm().min(min);
1714
1715 _extreme_nodes.resize(8); // 2^LIBMESH_DIM
1716 // Now make sure that there are actual nodes at all of the extremes
1717 std::vector<bool> extreme_matches(8, false);
1718 std::vector<unsigned int> comp_map(3);
1719 for (const auto & node : getMesh().node_ptr_range())
1720 {
1721 // See if the current node is located at one of the extremes
1722 unsigned int coord_match = 0;
1723
1724 for (const auto i : make_range(Moose::dim))
1725 {
1726 if (std::abs((*node)(i)-min[i]) < tol)
1727 {
1728 comp_map[i] = MIN;
1729 ++coord_match;
1730 }
1731 else if (std::abs((*node)(i)-max[i]) < tol)
1732 {
1733 comp_map[i] = MAX;
1734 ++coord_match;
1735 }
1736 }
1737
1738 if (coord_match == LIBMESH_DIM) // Found a coordinate at one of the extremes
1739 {
1740 _extreme_nodes[comp_map[X] * 4 + comp_map[Y] * 2 + comp_map[Z]] = node;
1741 extreme_matches[comp_map[X] * 4 + comp_map[Y] * 2 + comp_map[Z]] = true;
1742 }
1743 }
1744
1745 // See if we matched all of the extremes for the mesh dimension
1746 this->comm().max(extreme_matches);
1747 if (std::count(extreme_matches.begin(), extreme_matches.end(), true) == (1 << dim))
1748 _regular_orthogonal_mesh = true;
1749
1750 // Set the bounds
1751 _bounds.resize(LIBMESH_DIM);
1752 for (const auto i : make_range(Moose::dim))
1753 {
1754 _bounds[i].resize(2);
1755 _bounds[i][MIN] = min[i];
1756 _bounds[i][MAX] = max[i];
1757 }
1758
1759 return _regular_orthogonal_mesh;
1760}
1761
1762void
1763MooseMesh::detectPairedSidesets()
1764{
1765 TIME_SECTION("detectPairedSidesets", 5);
1766
1767 // Loop over level-0 elements (since boundary condition information
1768 // is only directly stored for them) and find sidesets with normals
1769 // that point in the -x, +x, -y, +y, and -z, +z direction. If there
1770 // is a unique sideset id for each direction, then the paired
1771 // sidesets consist of (-x,+x), (-y,+y), (-z,+z). If there are
1772 // multiple sideset ids for a given direction, then we can't pick a
1773 // single pair for that direction. In that case, we'll just return
1774 // as was done in the original algorithm.
1775
1776 // Points used for direction comparison
1777 const Point minus_x(-1, 0, 0), plus_x(1, 0, 0), minus_y(0, -1, 0), plus_y(0, 1, 0),
1778 minus_z(0, 0, -1), plus_z(0, 0, 1);
1779
1780 // we need to test all element dimensions from dim down to 1
1781 const unsigned int dim = getMesh().mesh_dimension();
1782
1783 // boundary id sets for elements of different dimensions
1784 std::vector<std::set<BoundaryID>> minus_x_ids(dim), plus_x_ids(dim), minus_y_ids(dim),
1785 plus_y_ids(dim), minus_z_ids(dim), plus_z_ids(dim);
1786
1787 std::vector<std::unique_ptr<FEBase>> fe_faces(dim);
1788 std::vector<std::unique_ptr<QGauss>> qfaces(dim);
1789 for (unsigned side_dim = 0; side_dim < dim; ++side_dim)
1790 {
1791 // Face is assumed to be flat, therefore normal is assumed to be
1792 // constant over the face, therefore only compute it at 1 qp.
1793 qfaces[side_dim] = std::unique_ptr<QGauss>(new QGauss(side_dim, CONSTANT));
1794
1795 // A first-order Lagrange FE for the face.
1796 fe_faces[side_dim] = FEBase::build(side_dim + 1, FEType(FIRST, LAGRANGE));
1797 fe_faces[side_dim]->attach_quadrature_rule(qfaces[side_dim].get());
1798 }
1799
1800 // We need this to get boundary ids for each boundary face we encounter.
1801 BoundaryInfo & boundary_info = getMesh().get_boundary_info();
1802 std::vector<boundary_id_type> face_ids;
1803
1804 for (auto & elem : as_range(getMesh().level_elements_begin(0), getMesh().level_elements_end(0)))
1805 {
1806 // dimension of the current element and its normals
1807 unsigned int side_dim = elem->dim() - 1;
1808 const std::vector<Point> & normals = fe_faces[side_dim]->get_normals();
1809
1810 // loop over element sides
1811 for (unsigned int s = 0; s < elem->n_sides(); s++)
1812 {
1813 // If side is on the boundary
1814 if (elem->neighbor_ptr(s) == nullptr)
1815 {
1816 std::unique_ptr<Elem> side = elem->build_side_ptr(s);
1817
1818 fe_faces[side_dim]->reinit(elem, s);
1819
1820 // Get the boundary ID(s) for this side. If there is more
1821 // than 1 boundary id, then we already can't determine a
1822 // unique pairing of sides in this direction, but we'll just
1823 // keep going to keep the logic simple.
1824 boundary_info.boundary_ids(elem, s, face_ids);
1825
1826 // x-direction faces
1827 if (normals[0].absolute_fuzzy_equals(minus_x))
1828 minus_x_ids[side_dim].insert(face_ids.begin(), face_ids.end());
1829 else if (normals[0].absolute_fuzzy_equals(plus_x))
1830 plus_x_ids[side_dim].insert(face_ids.begin(), face_ids.end());
1831
1832 // y-direction faces
1833 else if (normals[0].absolute_fuzzy_equals(minus_y))
1834 minus_y_ids[side_dim].insert(face_ids.begin(), face_ids.end());
1835 else if (normals[0].absolute_fuzzy_equals(plus_y))
1836 plus_y_ids[side_dim].insert(face_ids.begin(), face_ids.end());
1837
1838 // z-direction faces
1839 else if (normals[0].absolute_fuzzy_equals(minus_z))
1840 minus_z_ids[side_dim].insert(face_ids.begin(), face_ids.end());
1841 else if (normals[0].absolute_fuzzy_equals(plus_z))
1842 plus_z_ids[side_dim].insert(face_ids.begin(), face_ids.end());
1843 }
1844 }
1845 }
1846
1847 // For a distributed mesh, boundaries may be distributed as well. We therefore collect information
1848 // from everyone. If the mesh is already serial, then there is no need to do an allgather. Note
1849 // that this is just going to gather information about what the periodic bc ids are. We are not
1850 // gathering any remote elements or anything like that. It's just that the GhostPointNeighbors
1851 // ghosting functor currently relies on the fact that every process agrees on whether we have
1852 // periodic boundaries; every process that thinks there are periodic boundaries will call
1853 // MeshBase::sub_point_locator which makes a parallel_object_only() assertion (right or wrong). So
1854 // we all need to go there (or not go there)
1855 if (_use_distributed_mesh && !_mesh->is_serial())
1856 {
1857 // Pack all data together so that we send them via one communication
1858 // pair: boundary side --> boundary ids.
1859 std::vector<std::pair<boundary_id_type, boundary_id_type>> vecdata;
1860 // We check boundaries on all dimensions
1861 for (unsigned side_dim = 0; side_dim < dim; ++side_dim)
1862 {
1863 // "6" means: we have at most 6 boundaries. It is true for generated simple mesh
1864 // "detectPairedSidesets" is designed for only simple meshes
1865 for (auto bd = minus_x_ids[side_dim].begin(); bd != minus_x_ids[side_dim].end(); bd++)
1866 vecdata.emplace_back(side_dim * 6 + 0, *bd);
1867
1868 for (auto bd = plus_x_ids[side_dim].begin(); bd != plus_x_ids[side_dim].end(); bd++)
1869 vecdata.emplace_back(side_dim * 6 + 1, *bd);
1870
1871 for (auto bd = minus_y_ids[side_dim].begin(); bd != minus_y_ids[side_dim].end(); bd++)
1872 vecdata.emplace_back(side_dim * 6 + 2, *bd);
1873
1874 for (auto bd = plus_y_ids[side_dim].begin(); bd != plus_y_ids[side_dim].end(); bd++)
1875 vecdata.emplace_back(side_dim * 6 + 3, *bd);
1876
1877 for (auto bd = minus_z_ids[side_dim].begin(); bd != minus_z_ids[side_dim].end(); bd++)
1878 vecdata.emplace_back(side_dim * 6 + 4, *bd);
1879
1880 for (auto bd = plus_z_ids[side_dim].begin(); bd != plus_z_ids[side_dim].end(); bd++)
1881 vecdata.emplace_back(side_dim * 6 + 5, *bd);
1882 }
1883
1884 _communicator.allgather(vecdata, false);
1885
1886 // Unpack data, and add them into minus/plus_x/y_ids
1887 for (auto pair = vecdata.begin(); pair != vecdata.end(); pair++)
1888 {
1889 // Convert data from the long vector, and add data to separated sets
1890 auto side_dim = pair->first / 6;
1891 auto side = pair->first % 6;
1892
1893 switch (side)
1894 {
1895 case 0:
1896 minus_x_ids[side_dim].insert(pair->second);
1897 break;
1898 case 1:
1899 plus_x_ids[side_dim].insert(pair->second);
1900 break;
1901 case 2:
1902 minus_y_ids[side_dim].insert(pair->second);
1903 break;
1904 case 3:
1905 plus_y_ids[side_dim].insert(pair->second);
1906 break;
1907 case 4:
1908 minus_z_ids[side_dim].insert(pair->second);
1909 break;
1910 case 5:
1911 plus_z_ids[side_dim].insert(pair->second);
1912 break;
1913 default:
1914 mooseError("Unknown boundary side ", side);
1915 }
1916 }
1917
1918 } // end if (_use_distributed_mesh && !_need_ghost_ghosted_boundaries)
1919
1920 for (unsigned side_dim = 0; side_dim < dim; ++side_dim)
1921 {
1922 // If unique pairings were found, fill up the _paired_boundary data
1923 // structure with that information.
1924 if (minus_x_ids[side_dim].size() == 1 && plus_x_ids[side_dim].size() == 1)
1925 _paired_boundary.emplace_back(
1926 std::make_pair(*(minus_x_ids[side_dim].begin()), *(plus_x_ids[side_dim].begin())));
1927
1928 if (minus_y_ids[side_dim].size() == 1 && plus_y_ids[side_dim].size() == 1)
1929 _paired_boundary.emplace_back(
1930 std::make_pair(*(minus_y_ids[side_dim].begin()), *(plus_y_ids[side_dim].begin())));
1931
1932 if (minus_z_ids[side_dim].size() == 1 && plus_z_ids[side_dim].size() == 1)
1933 _paired_boundary.emplace_back(
1934 std::make_pair(*(minus_z_ids[side_dim].begin()), *(plus_z_ids[side_dim].begin())));
1935 }

1996{
1997 for (unsigned int i = 0; i < dimension(); ++i)
1998 {
1999 // check to see if we're closer in real or periodic space in x, y, and z
2000 if (isTranslatedPeriodic(nonlinear_var_num, i))
2001 {
2002 // Need to test order before differencing
2003 if (p(i) > q(i))
2004 {
2005 if (p(i) - q(i) > _half_range(i))
2006 p(i) -= _half_range(i) * 2;
2007 }
2008 else
2009 {
2010 if (q(i) - p(i) > _half_range(i))
2011 p(i) += _half_range(i) * 2;
2012 }
2013 }
2014 }
2015
2016 return q - p;

2045{
2046 std::map<ElemType, Elem *> canonical_elems;
2047
2048 // First, loop over all elements and find a canonical element for each type
2049 // Doing it this way guarantees that this is going to work in parallel
2050 for (const auto & elem : getMesh().element_ptr_range()) // TODO: Thread this
2051 {
2052 ElemType type = elem->type();
2053
2054 if (canonical_elems.find(type) ==
2055 canonical_elems.end()) // If we haven't seen this type of elem before save it
2056 canonical_elems[type] = elem;
2057 else
2058 {
2059 Elem * stored = canonical_elems[type];
2060 if (elem->id() < stored->id()) // Arbitrarily keep the one with a lower id
2061 canonical_elems[type] = elem;
2062 }
2063 }
2064 // Now build the maps using these templates
2065 // Note: This MUST be done NOT threaded!
2066 for (const auto & can_it : canonical_elems)
2067 {
2068 Elem * elem = can_it.second;
2069
2070 // Need to do this just once to get the right qrules put in place
2071 assembly->setCurrentSubdomainID(elem->subdomain_id());
2072 assembly->reinit(elem);
2073 assembly->reinit(elem, 0);
2074 auto && qrule = assembly->writeableQRule();
2075 auto && qrule_face = assembly->writeableQRuleFace();
2076
2077 // Volume to volume projection for refinement
2078 buildRefinementMap(*elem, *qrule, *qrule_face, -1, -1, -1);
2079
2080 // Volume to volume projection for coarsening
2081 buildCoarseningMap(*elem, *qrule, *qrule_face, -1);
2082
2083 // Map the sides of children
2084 for (unsigned int side = 0; side < elem->n_sides(); side++)
2085 {
2086 // Side to side for sides that match parent's sides
2087 buildRefinementMap(*elem, *qrule, *qrule_face, side, -1, side);
2088 buildCoarseningMap(*elem, *qrule, *qrule_face, side);
2089 }
2090
2091 // Child side to parent volume mapping for "internal" child sides
2092 for (unsigned int child = 0; child < elem->n_children(); ++child)
2093 for (unsigned int side = 0; side < elem->n_sides();
2094 ++side) // Assume children have the same number of sides!
2095 if (!elem->is_child_on_side(child, side)) // Otherwise we already computed that map
2096 buildRefinementMap(*elem, *qrule, *qrule_face, -1, child, side);
2097 }

2461 // Find the closest parent_qp to each child_qp
2462 mapPoints(child_ref_points, parent_ref_points, qp_map);
2463 }
2464
2465 coarsen_map.resize(parent_ref_points.size());
2466
2467 // For each parent qp find the closest child qp
2468 for (unsigned int child = 0; child < n_children; child++)
2469 {
2470 if (parent_side != -1 && !elem->is_child_on_side(child, child_side))
2471 continue;
2472
2473 std::vector<Point> & child_ref_points = child_to_ref_points[child];
2474
2475 std::vector<QpMap> qp_map;
2476
2477 // Find all of the closest points from parent_qp to _THIS_ child's qp
2478 mapPoints(parent_ref_points, child_ref_points, qp_map);
2479
2480 // Check those to see if they are closer than what we currently have for each point
2481 for (unsigned int parent_qp = 0; parent_qp < parent_ref_points.size(); ++parent_qp)
2482 {
2483 std::pair<unsigned int, QpMap> & child_and_map = coarsen_map[parent_qp];
2484 unsigned int & closest_child = child_and_map.first;
2485 QpMap & closest_map = child_and_map.second;
2486
2487 QpMap & current_map = qp_map[parent_qp];
2488
2489 if (current_map._distance < closest_map._distance)
2490 {
2491 closest_child = child;
2492 closest_map = current_map;
2493 }
2494 }
2495 }

3053void
3054MooseMesh::ghostGhostedBoundaries()
3055{
3056 // No need to do this if using a serial mesh
3057 // We do not need to ghost boundary elements when _need_ghost_ghosted_boundaries
3058 // is not true. _need_ghost_ghosted_boundaries can be set by a mesh generator
3059 // where boundaries are already ghosted accordingly
3060 if (!_use_distributed_mesh || !_need_ghost_ghosted_boundaries)
3061 return;
3062
3063 TIME_SECTION("GhostGhostedBoundaries", 3);
3064
3065 parallel_object_only();
3066
3067 DistributedMesh & mesh = dynamic_cast<DistributedMesh &>(getMesh());
3068
3069 // We clear ghosted elements that were added by previous invocations of this
3070 // method but leave ghosted elements that were added by other code, e.g.
3071 // OversampleOutput, untouched
3072 mesh.clear_extra_ghost_elems(_ghost_elems_from_ghost_boundaries);
3073 _ghost_elems_from_ghost_boundaries.clear();
3074
3075 std::set<const Elem *, CompareElemsByLevel> boundary_elems_to_ghost;
3076 std::set<Node *> connected_nodes_to_ghost;
3077
3078 std::vector<const Elem *> family_tree;
3079
3080 for (const auto & t : mesh.get_boundary_info().build_side_list())
3081 {
3082 auto elem_id = std::get<0>(t);
3083 auto bc_id = std::get<2>(t);
3084
3085 if (_ghosted_boundaries.find(bc_id) != _ghosted_boundaries.end())
3086 {
3087 Elem * elem = mesh.elem_ptr(elem_id);
3088
3089#ifdef LIBMESH_ENABLE_AMR
3090 elem->family_tree(family_tree);
3091 Elem * parent = elem->parent();
3092 while (parent)
3093 {
3094 family_tree.push_back(parent);
3095 parent = parent->parent();
3096 }
3097#else
3098 family_tree.clear();
3099 family_tree.push_back(elem);
3100#endif
3101 for (const auto & felem : family_tree)
3102 {
3103 boundary_elems_to_ghost.insert(felem);
3104
3105 // The entries of connected_nodes_to_ghost need to be
3106 // non-constant, so that they will work in things like
3107 // UpdateDisplacedMeshThread. The container returned by
3108 // family_tree contains const Elems even when the Elem
3109 // it is called on is non-const, so once that interface
3110 // gets fixed we can remove this const_cast.
3111 for (unsigned int n = 0; n < felem->n_nodes(); ++n)
3112 connected_nodes_to_ghost.insert(const_cast<Node *>(felem->node_ptr(n)));
3113 }
3114 }
3115 }
3116
3117 // We really do want to store this by value instead of by reference
3118 const auto prior_ghost_elems = mesh.extra_ghost_elems();
3119
3120 mesh.comm().allgather_packed_range(&mesh,
3121 connected_nodes_to_ghost.begin(),
3122 connected_nodes_to_ghost.end(),
3123 extra_ghost_elem_inserter<Node>(mesh));
3124
3125 mesh.comm().allgather_packed_range(&mesh,
3126 boundary_elems_to_ghost.begin(),
3127 boundary_elems_to_ghost.end(),
3128 extra_ghost_elem_inserter<Elem>(mesh));
3129
3130 const auto & current_ghost_elems = mesh.extra_ghost_elems();
3131
3132 std::set_difference(current_ghost_elems.begin(),
3133 current_ghost_elems.end(),
3134 prior_ghost_elems.begin(),
3135 prior_ghost_elems.end(),
3136 std::inserter(_ghost_elems_from_ghost_boundaries,
3137 _ghost_elems_from_ghost_boundaries.begin()));

3475void
3476MooseMesh::buildFiniteVolumeInfo() const
3477{
3478 mooseAssert(!Threads::in_threads,
3479 "This routine has not been implemented for threads. Please query this routine before "
3480 "a threaded region or contact a MOOSE developer to discuss.");
3481 _finite_volume_info_dirty = false;
3482
3483 using Keytype = std::pair<const Elem *, unsigned short int>;
3484
3485 // create a map from elem/side --> boundary ids
3486 std::vector<std::tuple<dof_id_type, unsigned short int, boundary_id_type>> side_list =
3487 buildActiveSideList();
3488 std::map<Keytype, std::set<boundary_id_type>> side_map;
3489 for (auto & [elem_id, side, bc_id] : side_list)
3490 {
3491 const Elem * elem = _mesh->elem_ptr(elem_id);
3492 Keytype key(elem, side);
3493 auto & bc_set = side_map[key];
3494 bc_set.insert(bc_id);
3495 }
3496
3497 _face_info.clear();
3498 _all_face_info.clear();
3499 _elem_side_to_face_info.clear();
3500
3501 _elem_to_elem_info.clear();
3502 _elem_info.clear();
3503
3504 // by performing the element ID comparison check in the below loop, we are ensuring that we never
3505 // double count face contributions. If a face lies along a process boundary, the only process that
3506 // will contribute to both sides of the face residuals/Jacobians will be the process that owns the
3507 // element with the lower ID.
3508 auto begin = getMesh().active_elements_begin();
3509 auto end = getMesh().active_elements_end();
3510
3511 // We prepare a map connecting the Elem* and the corresponding ElemInfo
3512 // for the active elements.
3513 for (const Elem * elem : as_range(begin, end))
3514 _elem_to_elem_info.emplace(elem->id(), elem);
3515
3516 _linear_finite_volume_dofs_cached = false;
3517
3518 dof_id_type face_index = 0;
3519 for (const Elem * elem : as_range(begin, end))
3520 {
3521 for (unsigned int side = 0; side < elem->n_sides(); ++side)
3522 {
3523 // get the neighbor element
3524 const Elem * neighbor = elem->neighbor_ptr(side);
3525
3526 // Check if the FaceInfo shall belong to the element. If yes,
3527 // create and initialize the FaceInfo. We need this to ensure that
3528 // we do not duplicate FaceInfo-s.
3529 if (Moose::FV::elemHasFaceInfo(*elem, neighbor))
3530 {
3531 mooseAssert(!neighbor || (neighbor->level() < elem->level() ? neighbor->active() : true),
3532 "If the neighbor is coarser than the element, we expect that the neighbor must "
3533 "be active.");
3534
3535 // We construct the faceInfo using the elementinfo and side index
3536 _all_face_info.emplace_back(&_elem_to_elem_info[elem->id()], side, face_index++);
3537
3538 auto & fi = _all_face_info.back();
3539
3540 // get all the sidesets that this face is contained in and cache them
3541 // in the face info.
3542 std::set<boundary_id_type> & boundary_ids = fi.boundaryIDs();
3543 boundary_ids.clear();
3544
3545 // We initialize the weights/other information in faceInfo. If the neighbor does not exist
3546 // or is remote (so when we are on some sort of mesh boundary), we initialize the ghost
3547 // cell and use it to compute the weights corresponding to the faceInfo.
3548 if (!neighbor || neighbor == remote_elem)
3549 fi.computeBoundaryCoefficients();
3550 else
3551 fi.computeInternalCoefficients(&_elem_to_elem_info[neighbor->id()]);
3552
3553 auto lit = side_map.find(Keytype(&fi.elem(), fi.elemSideID()));
3554 if (lit != side_map.end())
3555 boundary_ids.insert(lit->second.begin(), lit->second.end());
3556
3557 if (fi.neighborPtr())
3558 {
3559 auto rit = side_map.find(Keytype(fi.neighborPtr(), fi.neighborSideID()));
3560 if (rit != side_map.end())
3561 boundary_ids.insert(rit->second.begin(), rit->second.end());
3562 }
3563 }
3564 }
3565 }
3566
3567 // Build the local face info and elem_side to face info maps. We need to do this after
3568 // _all_face_info is finished being constructed because emplace_back invalidates all iterators and
3569 // references if ever the new size exceeds capacity
3570 for (auto & fi : _all_face_info)
3571 {
3572 const Elem * const elem = &fi.elem();
3573 const auto side = fi.elemSideID();
3574
3575#ifndef NDEBUG
3576 auto pair_it =
3577#endif
3578 _elem_side_to_face_info.emplace(std::make_pair(elem, side), &fi);
3579 mooseAssert(pair_it.second, "We should be adding unique FaceInfo objects.");
3580
3581 // We will add the faces on processor boundaries to the list of face infos on each
3582 // associated processor.
3583 if (fi.elem().processor_id() == this->processor_id() ||
3584 (fi.neighborPtr() && (fi.neighborPtr()->processor_id() == this->processor_id())))
3585 _face_info.push_back(&fi);
3586 }
3587
3588 for (auto & ei : _elem_to_elem_info)
3589 if (ei.second.elem()->processor_id() == this->processor_id())
3590 _elem_info.push_back(&ei.second);
3591}
