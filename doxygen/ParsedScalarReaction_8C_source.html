35ParsedScalarReaction::ParsedScalarReaction(const InputParameters & parameters)
36 : ParsedODEKernel(parameters), _sampling_format(getParam<std::string>("sampling_format"))
37// _v_var(isCoupledScalar("v") ? coupledScalar("v") : -1),
38// _v(coupledScalarValue("v")),
39// _n_gas(getParam<Real>("n_gas")),
40// _stoichiometric_coeff(getParam<Real>("coefficient")),
41// _v_eq_u(getParam<bool>("v_eq_u"))
42{
43 std::vector<Real> reduced_field;
44 std::vector<Real> electron_temperature;
45 std::string file_name =
46 getParam<std::string>("file_location") + "/" + getParam<FileName>("property_file");
47 MooseUtils::checkFileReadable(file_name);
48 const char * charPath = file_name.c_str();
49 std::ifstream myfile(charPath);
50 Real value;
51
52 if (myfile.is_open())
53 {
54 while (myfile >> value)
55 {
56 reduced_field.push_back(value);
57 myfile >> value;
58 electron_temperature.push_back(value);
59 }
60 myfile.close();
61 }
62 else
63 mooseError("Unable to open file");
64
65 _temperature_interpolation.setData(reduced_field, electron_temperature);
66}
