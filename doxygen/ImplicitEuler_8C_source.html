28ImplicitEuler::computeTimeDerivatives()
29{
30 if (!_sys.solutionUDot())

32 "uDotRequested() to true in FEProblemBase befor requesting `u_dot`.");
33
34 NumericVector<Number> & u_dot = *_sys.solutionUDot();
35 u_dot = *_solution;
36 computeTimeDerivativeHelper(u_dot, _solution_old);
37 u_dot.close();

40}
