472template <typename SpaceArg>
473typename FunctorBase<T>::ValueType
474FunctorBase<T>::queryFVArgCache(std::map<SpaceArg, ValueType> & cache_data,
475 const SpaceArg & space) const
476{
477 // We don't want to evaluate if the key already exists, so instead we value initialize
478 auto [it, inserted] = cache_data.try_emplace(space, ValueType());
479 auto & value = it->second;
480
481 if (inserted)
482 // value not ready to go
483 // this function is only called from functions that assert we are in the current time state
484 value = evaluate(space, currentState());
485
486 return value;
487}

517template <typename T>
518template <typename SpaceArg, typename StateArg>
519typename FunctorBase<T>::ValueType
520FunctorBase<T>::queryQpCache(const unsigned int qp,
521 const QBase & qrule,
522 std::vector<std::pair<bool, ValueType>> & qp_cache_data,
523 const SpaceArg & space,
524 const StateArg & state) const
525{
526 // Check and see whether we even have sized for this quadrature point. If we haven't then we
527 // must evaluate
528 if (qp >= qp_cache_data.size())
529 {
530 qp_cache_data.resize(qrule.n_points(), std::make_pair(false, ValueType()));
531 auto & pr = qp_cache_data[qp];
532 pr.second = evaluate(space, state);
533 pr.first = true;
534 return pr.second;
535 }
536
537 // We've already sized for this qp, so let's see whether we have a valid cache value
538 auto & pr = qp_cache_data[qp];
539 if (pr.first)
540 return pr.second;
541
542 // No valid cache value so evaluate
543 pr.second = evaluate(space, state);
544 pr.first = true;
545 return pr.second;
546}

548template <typename T>
549typename FunctorBase<T>::ValueType
550FunctorBase<T>::operator()(const ElemQpArg & elem_qp, const StateArg & state) const
551{
552 if (_clearance_schedule.count(EXEC_ALWAYS))
553 return evaluate(elem_qp, state);
554
555 const auto elem_id = elem_qp.elem->id();
556 if (elem_id != _current_qp_map_key)
557 {
558 _current_qp_map_key = elem_id;
559 _current_qp_map_value = &_qp_to_value[elem_id];
560 }
561 auto & qp_data = *_current_qp_map_value;
562 const auto qp = elem_qp.qp;
563 const auto * const qrule = elem_qp.qrule;
564 mooseAssert(qrule, "qrule must be non-null");
565
566 return queryQpCache(qp, *qrule, qp_data, elem_qp, state);
567}

569template <typename T>
570typename FunctorBase<T>::ValueType
571FunctorBase<T>::operator()(const ElemSideQpArg & elem_side_qp, const StateArg & state) const
572{
573 if (_clearance_schedule.count(EXEC_ALWAYS))
574 return evaluate(elem_side_qp, state);
575
576 const Elem * const elem = elem_side_qp.elem;
577 mooseAssert(elem, "elem must be non-null");
578 const auto elem_id = elem->id();
579 if (elem_id != _current_side_qp_map_key)
580 {
581 _current_side_qp_map_key = elem_id;
582 _current_side_qp_map_value = &_side_qp_to_value[elem_id];
583 }
584 auto & side_qp_data = *_current_side_qp_map_value;
585 const auto side = elem_side_qp.side;
586 const auto qp = elem_side_qp.qp;
587 const auto * const qrule = elem_side_qp.qrule;
588 mooseAssert(qrule, "qrule must be non-null");
589
590 // Check and see whether we even have sized for this side
591 if (side >= side_qp_data.size())
592 side_qp_data.resize(elem->n_sides());
593
594 // Ok we were sized enough for our side
595 auto & qp_data = side_qp_data[side];
596 return queryQpCache(qp, *qrule, qp_data, elem_side_qp, state);
597}

620template <typename T>
621FaceArg
622FunctorBase<T>::checkFace(const Moose::FaceArg & face) const
623{
624 const Elem * const elem = face.face_side;
625 const FaceInfo * const fi = face.fi;
626 mooseAssert(fi, "face info should be non-null");
627 auto ret_face = face;
628 bool check_elem_def = false;
629 bool check_neighbor_def = false;
630 if (!elem)
631 {
632 if (!hasFaceSide(*fi, true))
633 {
634 ret_face.face_side = fi->neighborPtr();
635 check_neighbor_def = true;
636 }
637 else if (!hasFaceSide(*fi, false))
638 {
639 ret_face.face_side = fi->elemPtr();
640 check_elem_def = true;
641 }
642 }
643 else if (elem == fi->elemPtr())
644 check_elem_def = true;
645 else
646 {
647 mooseAssert(elem == fi->neighborPtr(), "This has to match something");
648 check_neighbor_def = true;
649 }
650
651 if (check_elem_def && !hasFaceSide(*fi, true))
652 {
653 std::string additional_message = "It is not defined on the neighbor side either.";
654 if (hasFaceSide(*fi, false))
655 additional_message = "It is however defined on the neighbor side.";
656 additional_message += " Face centroid: " + Moose::stringify(fi->faceCentroid());
657 mooseError(_functor_name,
658 " is not defined on the element side of the face information, but a face argument "
659 "producer "
660 "(e.g. residual object, postprocessor, etc.) has requested evaluation there.\n",
661 additional_message);
662 }
663 if (check_neighbor_def && !hasFaceSide(*fi, false))
664 {
665 std::string additional_message = "It is not defined on the element side either.";
666 if (hasFaceSide(*fi, true))
667 additional_message = "It is however defined on the element side.";
668 additional_message += " Face centroid: " + Moose::stringify(fi->faceCentroid());
669 mooseError(
670 _functor_name,
671 " is not defined on the neighbor side of the face information, but a face argument "
672 "producer (e.g. residual object, postprocessor, etc.) has requested evaluation there.\n",
673 additional_message);
674 }
675
676 return ret_face;
677}

679template <typename T>
680void
681FunctorBase<T>::clearCacheData()
682{
683 for (auto & map_pr : _qp_to_value)
684 for (auto & pr : map_pr.second)
685 pr.first = false;
686
687 for (auto & map_pr : _side_qp_to_value)
688 {
689 auto & side_vector = map_pr.second;
690 for (auto & qp_vector : side_vector)
691 for (auto & pr : qp_vector)
692 pr.first = false;
693 }
694
695 _current_qp_map_key = DofObject::invalid_id;
696 _current_qp_map_value = nullptr;
697 _current_side_qp_map_key = DofObject::invalid_id;
698 _current_side_qp_map_value = nullptr;
699
700 _elem_arg_to_value.clear();
701 _face_arg_to_value.clear();
702 _node_arg_to_value.clear();
703}
