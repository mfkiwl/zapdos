297BoltzmannSolverScalar::initialize()
298{
299 // Here we can write the input file based on input parameters
300 // Required input: gas composition fractions, gas temperature
301
302 // To rewrite file, we can use a bash command (using system()):
303 // sed -e "34s/.*/0.23 0.77 \/ Gas composition fraction/" -i '' temp_in.dat
304 // line # ^ [ ] <- replacement string
305 // If n steps have passed, write mole fractions and reduced field value into input file
306 if (_t <= _cutoff_time)
307 {
308 if (_timestep_number == _n_steps || _timestep_number == 0)
309 {
310 std::string edit_command;
311 edit_command = "sed -e \"34s/.*/";
312 // For each variable we add both the value (converted to a string) and a following space
313 // character.
314 for (unsigned int i = 0; i < _nargs; ++i)
315 {
316 _fractions_string[i] << std::setprecision(10) << (*_args[i])[0];
317 edit_command = edit_command + _fractions_string[i].str() + " ";
318
319 _fractions_string[i].str("");
320 }
321 // edit_command = edit_command + "\\/ Gas composition fraction/\" -i \'\' temp_in.dat";
322 edit_command = edit_command + "\\/ Gas composition fraction/\" -i \'\' " + _file_name;
323
324 // Real test1, test2;
325 // test1 = 0.3;
326 // test2 = 0.7;
327 // edit_command = "sed -e \"34s/.*/"+std::to_string(test1)+" "+std::to_string(test2)+" \\/ Gas
328 // composition fraction/\" -i \'\' temp_in.dat";
329 const char * command = edit_command.c_str();
330
331 auto err = system(command);
332 if (err < 0)
333 mooseError("failed command:", edit_command);
334
335 // Now the input file is edited with updated molar fractions!
336
337 // Update the reduced field line:

367void
368BoltzmannSolverScalar::execute()
369{
370 // Run BOLSIG+
371 // system("./bolsigminus example3.dat");
372 if (_t <= _cutoff_time)
373 {
374 if (_timestep_number == _n_steps || _timestep_number == 0)
375 {
376
377 _timestep_number = 1;
378
379 const char * command = _bolsig_run.c_str();
380 mooseInfo("Running BOLSIG+...");
381 auto err = system(command);
382 if (err < 0)
383 mooseError("failed command:", _bolsig_run);
384 mooseInfo("DONE running BOLSIG+");
385 std::fstream file(_output_file_name);
386
387 if (_output_table)
388 {
389 for (int i = 0; i < _num_reactions; ++i)
390 {
391 getLine(file, _reaction_line[i]);
392 for (int j = 0; j < _table_number; ++j)
393 {
394 file >> _x_val[j] >> _rate_coefficient[i][j];
395 _rate_coefficient[i][j] = _rate_coefficient[i][j] * _conversion_factor;
396 }
397
398 _coefficient_interpolation[i].setData(_x_val, _rate_coefficient[i]);
399 }
400
401 // Electron Temperature
402 getLine(file, _temperature_line);
403 for (int j = 0; j < _table_number; ++j)
404 {
405 file >> _tempstore >> _electron_temperature[j];
406 }
407
408 // Electron Mobility
409 getLine(file, _mobility_line);
410 for (int j = 0; j < _table_number; ++j)
411 {
412 // Note that BOLSIG+ outputs mobility as a mobility * N.
413 // Here the value is converted back to mobility alone.
414 // _conversion_factor simply converts to the desired units. Defaults to 1.0.
415 file >> _tempstore >> _electron_mobility[j];
416 _electron_mobility[j] = _electron_mobility[j] / (_plasma_density[0] * _conversion_factor);
417 }
418
419 // Now we interpolate the results to output a "table"
420 _temperature_interpolation.setData(_x_val, _electron_temperature);
421 _mobility_interpolation.setData(_x_val, _electron_mobility);
422 }
423 else
424 {
425 for (int i = 0; i < _num_reactions; ++i)
426 {
427 getLine(file, _reaction_line[i]);
428 file >> _tempstore >> _rate_coefficient[i][0];
429 // if (_rate_coefficient[i] == 0.0)
430 // mooseError("Zero rate coefficient");
431 }
432
433 // Electron temperature
434 getLine(file, _temperature_line);
435 file >> _tempstore >> _electron_temperature[0];
436
437 // Electron mobility
438 getLine(file, _mobility_line);
439 file >> _tempstore >> _electron_mobility[0];
440 _electron_mobility[0] = _electron_mobility[0] / (_plasma_density[0] * _conversion_factor);
441 }
442 }
443 else
444 {
445 _timestep_number = _timestep_number + 1;
446 }
447 // getLine(file, 113);
448
449 // file >> _x_val >> _rate_coefficient;
450 // Now we will fit everything into data tables (automatically...difficult.)
451 // Not sure how to do this easily. Ugh.
452 }
