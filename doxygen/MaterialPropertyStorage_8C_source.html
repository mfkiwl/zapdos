29void
30MaterialPropertyStorage::shallowSwapData(const std::vector<unsigned int> & stateful_prop_ids,
31 MaterialProperties & data,
32 MaterialProperties & data_from)
33{
34 for (const auto i : make_range(std::min(stateful_prop_ids.size(), data_from.size())))
35 if (stateful_prop_ids[i] < data.size() && data.hasValue(stateful_prop_ids[i]) &&
36 data_from.hasValue(i))
37 {
38 auto & prop = data[stateful_prop_ids[i]];
39 auto & prop_from = data_from[i];
40 prop.swap(prop_from);
41 }
42}

45MaterialPropertyStorage::shallowSwapDataBack(const std::vector<unsigned int> & stateful_prop_ids,

51 data_from.hasValue(stateful_prop_ids[i]))

53 auto & prop = data[i];

55 prop.swap(prop_from);

66void
67MaterialPropertyStorage::updateStatefulPropsForPRefinement(
68 const processor_id_type libmesh_dbg_var(pid),
69 const std::vector<QpMap> & p_refinement_map,
70 const QBase & qrule,
71 const QBase & qrule_face,
72 const THREAD_ID tid,
73 const Elem & elem,
74 const int input_side)
75{
76 unsigned int n_qpoints = 0;
77
78 // If we passed in -1 for these then we really need to store properties at 0
79 unsigned int side = input_side == -1 ? 0 : input_side;
80
81 if (input_side == -1) // Not doing side projection (ie, doing volume projection)
82 n_qpoints = qrule.n_points();
83 else
84 n_qpoints = qrule_face.n_points();
85
86 getMaterialData(tid).resize(n_qpoints);
87
88 mooseAssert(elem.active(), "We should be doing p-refinement on active elements only");
89 mooseAssert(elem.processor_id() == pid, "Prolongation should be occurring locally");
90 mooseAssert(p_refinement_map.size() == n_qpoints, "Refinement map not proper size");
91
92 initProps(tid, &elem, side, n_qpoints);
93
94 for (const auto state : stateIndexRange())
95 {
96 auto & props = setProps(&elem, side, state);
97 for (const auto i : index_range(_stateful_prop_id_to_prop_id))
98 {
99 auto & current_p_level_prop = props[i];
100 // We need to clone this property in order to not overwrite the values we're going to be
101 // reading from
102 auto previous_p_level_prop = current_p_level_prop.clone(current_p_level_prop.size());
103 // Cloning, despite its name, does not copy the data. Luckily since we are about to overwrite
104 // all of the current_p_level_prop data, we can just swap its data over to our
105 // previous_p_level_prop
106 previous_p_level_prop->swap(current_p_level_prop);
107 current_p_level_prop.resize(n_qpoints);
108 for (const auto qp : index_range(p_refinement_map))
109 current_p_level_prop.qpCopy(qp, *previous_p_level_prop, p_refinement_map[qp]._to);
110 }
111 }

114void
115MaterialPropertyStorage::prolongStatefulProps(
116 processor_id_type pid,
117 const std::vector<std::vector<QpMap>> & refinement_map,
118 const QBase & qrule,
119 const QBase & qrule_face,
120 MaterialPropertyStorage & parent_material_props,
121 const THREAD_ID tid,
122 const Elem & elem,
123 const int input_parent_side,
124 const int input_child,
125 const int input_child_side)
126{
127 mooseAssert(input_child != -1 || input_parent_side == input_child_side, "Invalid inputs!");
128
129 unsigned int n_qpoints = 0;
130
131 // If we passed in -1 for these then we really need to store properties at 0
132 unsigned int parent_side = input_parent_side == -1 ? 0 : input_parent_side;
133 unsigned int child_side = input_child_side == -1 ? 0 : input_child_side;
134
135 if (input_child_side == -1) // Not doing side projection (ie, doing volume projection)
136 n_qpoints = qrule.n_points();
137 else
138 n_qpoints = qrule_face.n_points();
139
140 getMaterialData(tid).resize(n_qpoints);
141
142 unsigned int n_children = elem.n_children();
143
144 std::vector<unsigned int> children;
145
146 if (input_child != -1) // Passed in a child explicitly
147 children.push_back(input_child);
148 else
149 {
150 children.resize(n_children);
151 for (unsigned int child = 0; child < n_children; child++)
152 children[child] = child;
153 }
154
155 for (const auto & child : children)
156 {
157 // If we're not projecting an internal child side, but we are projecting sides, see if this
158 // child is on that side
159 if (input_child == -1 && input_child_side != -1 && !elem.is_child_on_side(child, parent_side))
160 continue;
161
162 const Elem * child_elem = elem.child_ptr(child);
163
164 // If it's not a local child then it'll be prolonged where it is
165 // local
166 if (child_elem->processor_id() != pid)
167 continue;
168
169 mooseAssert(child < refinement_map.size(), "Refinement_map vector not initialized");
170 const std::vector<QpMap> & child_map = refinement_map[child];
171
172 initProps(tid, child_elem, child_side, n_qpoints);
173
174 for (const auto state : stateIndexRange())
175 {
176 const auto & parent_props = parent_material_props.props(&elem, parent_side, state);
177 auto & child_props = setProps(child_elem, child_side, state);
178 for (const auto i : index_range(_stateful_prop_id_to_prop_id))
179 for (const auto qp : index_range(refinement_map[child]))
180 child_props[i].qpCopy(qp, parent_props[i], child_map[qp]._to);
181 }
182 }
183}

185void
186MaterialPropertyStorage::restrictStatefulProps(
187 const std::vector<std::pair<unsigned int, QpMap>> & coarsening_map,
188 const std::vector<const Elem *> & coarsened_element_children,
189 const QBase & qrule,
190 const QBase & qrule_face,
191 const THREAD_ID tid,
192 const Elem & elem,
193 int input_side)
194{
195 unsigned int side;
196
197 bool doing_a_side = input_side != -1;
198
199 unsigned int n_qpoints = 0;
200
201 if (!doing_a_side)
202 {
203 side = 0; // Use 0 for the elem
204 n_qpoints = qrule.n_points();
205 }
206 else
207 {
208 side = input_side;
209 n_qpoints = qrule_face.n_points();
210 }
211
212 initProps(tid, &elem, side, n_qpoints);
213
214 std::vector<MaterialProperties *> parent_props(numStates());
215 for (const auto state : stateIndexRange())
216 parent_props[state] = &setProps(&elem, side, state);
217
218 // Copy from the child stateful properties
219 for (const auto qp : index_range(coarsening_map))
220 {
221 const std::pair<unsigned int, QpMap> & qp_pair = coarsening_map[qp];
222 unsigned int child = qp_pair.first;
223
224 mooseAssert(child < coarsened_element_children.size(),
225 "Coarsened element children vector not initialized");
226 const Elem * child_elem = coarsened_element_children[child];
227 const QpMap & qp_map = qp_pair.second;
228
229 for (const auto state : stateIndexRange())
230 {
231 const auto & child_props = props(child_elem, side, state);
232 for (const auto i : index_range(_stateful_prop_id_to_prop_id))
233 (*parent_props[state])[i].qpCopy(qp, child_props[i], qp_map._to);
234 }
235 }

238void
239MaterialPropertyStorage::initStatefulProps(const THREAD_ID tid,
240 const std::vector<std::shared_ptr<MaterialBase>> & mats,
241 const unsigned int n_qpoints,
242 const Elem & elem,
243 const unsigned int side /* = 0*/)
244{
245 // NOTE: since materials are storing their computed properties in MaterialData class, we need to
246 // juggle the memory between MaterialData and MaterialProperyStorage classes
247
248 initProps(tid, &elem, side, n_qpoints);
249
250 // copy from storage to material data
251 swap(tid, elem, side);
252 // run custom init on properties
253 for (const auto & mat : mats)
254 mat->initStatefulProperties(n_qpoints);
255
256 swapBack(tid, elem, side);
257
258 if (!hasStatefulProperties())
259 return;
260
261 // This second call to initProps covers cases where code in
262 // "init[Qp]StatefulProperties" may have called a get/declare for a stateful
263 // property affecting the _stateful_prop_id_to_prop_id vector among other
264 // things. This is necessary because a call to
265 // getMaterialProperty[Old/Older] can potentially trigger a material to
266 // become stateful that previously wasn't. This needs to go after the
267 // swapBack.
268 initProps(tid, &elem, side, n_qpoints);
269
270 // Copy to older states as needed
271 const auto & current_props = props(&elem, side, 0);
272 for (const auto state : statefulIndexRange())
273 {
274 auto & to_props = setProps(&elem, side, state);
275 for (const auto i : index_range(_stateful_prop_id_to_prop_id))
276 for (const auto qp : make_range(n_qpoints))
277 to_props[i].qpCopy(qp, current_props[i], qp);
278 }
279}

325void
326MaterialPropertyStorage::swap(const THREAD_ID tid, const Elem & elem, unsigned int side)
327{
328 Threads::spin_mutex::scoped_lock lock(this->_spin_mtx);
329
330 for (const auto state : stateIndexRange())
331 shallowSwapData(_stateful_prop_id_to_prop_id,
332 getMaterialData(tid).props(state),
333 // Would be nice to make this setProps()
334 initAndSetProps(&elem, side, state));
335}

337void
338MaterialPropertyStorage::swapBack(const THREAD_ID tid, const Elem & elem, unsigned int side)
339{
340 Threads::spin_mutex::scoped_lock lock(this->_spin_mtx);
341
342 for (const auto state : stateIndexRange())
343 shallowSwapDataBack(_stateful_prop_id_to_prop_id,
344 setProps(&elem, side, state),
345 getMaterialData(tid).props(state));
346
347 // Workaround for MOOSE difficulties in keeping materialless
348 // elements (e.g. Lower D elements in Mortar code) materials
349 for (const auto state : stateIndexRange())
350 if (props(&elem, side, state).empty())
351 setProps(state)[&elem].erase(side);
352}

354unsigned int
355MaterialPropertyStorage::addProperty(const std::string & prop_name, const unsigned int state)
356{
357 if (state > MaterialData::max_state)
358 mooseError("Material property state of ",
359 state,
360 " is not supported. Max state supported: ",
361 MaterialData::max_state);
362
363 // Increment state as needed
364 if (maxState() < state)
365 _max_state = state;
366
367 const auto prop_id = _registry.addOrGetID(prop_name, {});
368
369 if (state > 0)
370 {
371 if (std::find(_stateful_prop_id_to_prop_id.begin(),
372 _stateful_prop_id_to_prop_id.end(),
373 prop_id) == _stateful_prop_id_to_prop_id.end())
374 _stateful_prop_id_to_prop_id.push_back(prop_id);
375 _stateful_prop_names[prop_id] = prop_name;
376 }
377
378 return prop_id;
379}

391void
392MaterialPropertyStorage::initProps(const THREAD_ID tid,
393 const unsigned int state,
394 const Elem * elem,
395 unsigned int side,
396 unsigned int n_qpoints)
397{
398 auto & material_data = getMaterialData(tid);
399 material_data.resize(n_qpoints);
400
401 auto & mat_props = initAndSetProps(elem, side, state);
402
403 // In some special cases, material_data might be larger than n_qpoints
404 if (material_data.isOnlyResizeIfSmaller())
405 n_qpoints = material_data.nQPoints();
406
407 const auto n_props = _stateful_prop_id_to_prop_id.size();
408 if (mat_props.size() < n_props)
409 mat_props.resize(n_props, {});
410
411 // init properties (allocate memory. etc)
412 for (const auto i : index_range(_stateful_prop_id_to_prop_id))
413 if (!mat_props.hasValue(i))
414 {
415 const auto prop_id = _stateful_prop_id_to_prop_id[i];
416 mat_props.setPointer(i, material_data.props(0)[prop_id].clone(n_qpoints), {});
417 mooseAssert(mat_props[i].id() == prop_id, "Inconsistent id");
418 }
419}

421void
422dataStore(std::ostream & stream, MaterialPropertyStorage & storage, void * context)
423{
424 const auto num_states = storage.numStates();
425 dataStore(stream, num_states, nullptr);
426
427 // Store the material property ID -> name map for mapping back
428 const auto & registry = storage.getMaterialPropertyRegistry();
429 std::vector<std::string> ids_to_names(registry.idsToNamesBegin(), registry.idsToNamesEnd());
430 dataStore(stream, ids_to_names, nullptr);
431
432 // Store the stateful ID -> property ID map for mapping back
433 dataStore(stream, storage._stateful_prop_id_to_prop_id, nullptr);
434
435 // Store the stateful id -> name map
436 dataStore(stream, storage._stateful_prop_names, nullptr);
437
438 // Store every property
439 for (const auto state : storage.stateIndexRange())
440 {
441 std::size_t num_elems = storage.setProps(state).size();
442 dataStore(stream, num_elems, nullptr);
443
444 for (auto & elem_side_map_pair : storage.setProps(state))
445 {
446 const Elem * elem = elem_side_map_pair.first;
447 mooseAssert(elem, "Null element");
448 dataStore(stream, elem, context);
449
450 auto & side_map = elem_side_map_pair.second;
451 std::size_t num_sides = side_map.size();
452 dataStore(stream, num_sides, nullptr);
453
454 for (auto & [side, props] : side_map)
455 {
456 dataStore(stream, side, nullptr);
457
458 std::size_t num_props = props.size();
459 dataStore(stream, num_props, nullptr);
460 mooseAssert(num_props > 0, "No properties");
461
462 std::size_t n_q_points = 0;
463 for (const auto & entry : props)
464 if (entry.size() > n_q_points)
465 n_q_points = entry.size();
466 dataStore(stream, n_q_points, nullptr);
467
468 for (auto & entry : props)
469 dataStoreSkippable(stream, entry, nullptr);
470 }
471 }
472 }

475void
476dataLoad(std::istream & stream, MaterialPropertyStorage & storage, void * context)
477{
478 const auto & registry = storage.getMaterialPropertyRegistry();
479
480 decltype(storage.numStates()) num_states;
481 dataLoad(stream, num_states, nullptr);
482 if (num_states != storage.numStates())
483 mooseError("Stateful material properties up to state ",
484 num_states,
485 " were stored in checkpoint/backup,\nbut state ",
486 storage.numStates(),
487 " is now the maximum state requested.\n\nThis mismatch is not currently supported.");
488
489 std::vector<std::string> from_prop_ids_to_names;
490 dataLoad(stream, from_prop_ids_to_names, nullptr);
491
492 decltype(storage._stateful_prop_id_to_prop_id) from_stateful_prop_id_to_prop_id;
493 dataLoad(stream, from_stateful_prop_id_to_prop_id, nullptr);
494
495 decltype(storage._stateful_prop_names) from_stateful_prop_names;
496 dataLoad(stream, from_stateful_prop_names, nullptr);
497
498 {
499 const auto fill_names = [](const auto & from)
500 {
501 std::set<std::string> names;
502 for (const auto & id_name_pair : from)
503 names.insert(id_name_pair.second);
504 return names;
505 };
506
507 const auto from_names = fill_names(from_stateful_prop_names);
508 const auto to_names = fill_names(storage.statefulPropNames());
509
510 // This requirement currently comes from the fact that now when we do restart/recover,
511 // we do not call initStatefulQpProperties() anymore on each material. Which means that
512 // if you add more stateful properties, they would never be initialized. It seems like
513 // this capability (adding props) isn't currently used, so we're going to make it an
514 // error for now. There are plenty of ways around this, but the priority at the moment
515 // is enabling more flexible restart/recover, and this is one of the costs.
516 // TODO: check types
517 if (from_names != to_names)
518 {
519 std::stringstream err;
520 err << "There is a mismatch in the stateful material properties stored during "
521 "checkpoint/backup and declared in restart/recover.\n"
522 << "The current system requires that the stateful properties be equivalent.\n\n";
523 auto add_props = [&err](const auto & names)
524 {
525 for (const auto & name : names)
526 err << " " << name << "\n";
527 };
528 err << "Stored stateful properties:\n";
529 add_props(from_names);
530 err << "\nDeclared stateful properties:\n";
531 add_props(to_names);
532 mooseError(err.str());
533 }
534 }
535
536 std::vector<std::optional<unsigned int>> to_stateful_ids(storage.statefulProps().size());
537
538 // Fill the mapping from previous ID to current stateful ID
539 for (const auto from_stateful_id : index_range(from_stateful_prop_id_to_prop_id))
540 {
541 const auto from_prop_id = from_stateful_prop_id_to_prop_id[from_stateful_id];
542
543 mooseAssert(from_prop_id < from_prop_ids_to_names.size(), "Invalid ID map");
544 const auto & from_name = from_prop_ids_to_names[from_prop_id];
545
546 const auto to_prop_id = registry.getID(from_name);
547
548 const auto find_prop_id = std::find(storage._stateful_prop_id_to_prop_id.begin(),
549 storage._stateful_prop_id_to_prop_id.end(),
550 to_prop_id);
551 mooseAssert(find_prop_id != storage._stateful_prop_id_to_prop_id.end(), "Not found");
552
553 const std::size_t to_stateful_id =
554 std::distance(storage._stateful_prop_id_to_prop_id.begin(), find_prop_id);
555 to_stateful_ids[from_stateful_id] = to_stateful_id;
556 }
557
558 // Load the properties
559 for (const auto state : storage.stateIndexRange())
560 {
561 std::size_t num_elems;
562 dataLoad(stream, num_elems, nullptr);
563
564 for (std::size_t i_elem = 0; i_elem < num_elems; ++i_elem)
565 {
566 const Elem * elem;
567 dataLoad(stream, elem, context);
568 mooseAssert(elem, "Null element");
569
570 std::size_t num_sides;
571 dataLoad(stream, num_sides, nullptr);
572
573 for (std::size_t i_side = 0; i_side < num_sides; ++i_side)
574 {
575 unsigned int side;
576 dataLoad(stream, side, nullptr);
577
578 std::size_t num_props;
579 dataLoad(stream, num_props, nullptr);
580
581 std::size_t num_q_points;
582 dataLoad(stream, num_q_points, nullptr);
583
584 storage.initProps(0, state, elem, side, num_q_points);
585 auto & props = storage.setProps(elem, side, state);
586
587 for (const auto from_stateful_id : make_range(num_props))
588 {
589 const auto to_stateful_id = to_stateful_ids[from_stateful_id];
590 mooseAssert(to_stateful_id, "Skipping not supported");
591
592 // Eventually we should support skipping, and this will be how it's done
593 // if (!to_stateful_id)
594 // dataLoadSkip(stream);
595 // else
596
597 dataLoadSkippable(stream, props[*to_stateful_id], nullptr);
598 }
599 }
600 }
601 }
