2002{

2005 // boundary condition action, will be querying isDistributedMesh()

2011
2012 auto mesh = std::make_unique<T>(_communicator, dim);
2013

2015 mesh->allow_renumbering(false);

2018 _app.attachRelationshipManagers(*mesh, *this);
2019
