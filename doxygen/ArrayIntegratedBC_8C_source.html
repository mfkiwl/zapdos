27ArrayIntegratedBC::ArrayIntegratedBC(const InputParameters & parameters)
28 : IntegratedBCBase(parameters),
29 MooseVariableInterface<RealEigenVector>(this,
30 false,
31 "variable",
32 Moose::VarKindType::VAR_NONLINEAR,
33 Moose::VarFieldType::VAR_FIELD_ARRAY),
34 _var(*mooseVariable()),
35 _normals(_assembly.normals()),
36 _phi(_assembly.phiFace(_var)),
37 _test(_var.phiFace()),
38 _u(_is_implicit ? _var.sln() : _var.slnOld()),
39 _count(_var.count()),
40 _work_vector(_count)
41{
42 addMooseVariableDependency(mooseVariable());
43
44 _save_in.resize(_save_in_strings.size());
45 _diag_save_in.resize(_diag_save_in_strings.size());
46
47 for (unsigned int i = 0; i < _save_in_strings.size(); i++)
48 {
49 ArrayMooseVariable * var = &_subproblem.getArrayVariable(_tid, _save_in_strings[i]);
50
51 if (var->feType() != _var.feType())
52 paramError(
53 "save_in",
54 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
55 moose::internal::incompatVarMsg(*var, _var));
56 _save_in[i] = var;
57 var->sys().addVariableToZeroOnResidual(_save_in_strings[i]);
58 addMooseVariableDependency(var);
59 }
60
61 _has_save_in = _save_in.size() > 0;
62
63 for (unsigned int i = 0; i < _diag_save_in_strings.size(); i++)
64 {
65 ArrayMooseVariable * var = &_subproblem.getArrayVariable(_tid, _diag_save_in_strings[i]);
66
67 if (var->feType() != _var.feType())
68 paramError(
69 "diag_save_in",
70 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
71 moose::internal::incompatVarMsg(*var, _var));
72
73 _diag_save_in[i] = var;
74 var->sys().addVariableToZeroOnJacobian(_diag_save_in_strings[i]);
75 addMooseVariableDependency(var);
76 }
77
78 _has_diag_save_in = _diag_save_in.size() > 0;
79}

81void
82ArrayIntegratedBC::computeResidual()
83{
84 prepareVectorTag(_assembly, _var.number());
85
86 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
87 {
88 initQpResidual();
89 for (_i = 0; _i < _test.size(); _i++)
90 {
91 _work_vector.setZero();
92 computeQpResidual(_work_vector);
93 mooseAssert(_work_vector.size() == _count,
94 "Size of local residual is not equal to the number of array variable compoments");
95 _work_vector *= _JxW[_qp] * _coord[_qp];
96 _assembly.saveLocalArrayResidual(_local_re, _i, _test.size(), _work_vector);
97 }
98 }
99
100 accumulateTaggedLocalResidual();
101
102 if (_has_save_in)
103 {
104 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
105 for (const auto & var : _save_in)
106 {
107 auto * avar = dynamic_cast<ArrayMooseVariable *>(var);
108 if (avar)
109 avar->addSolution(_local_re);
110 else
111 mooseError("Save-in variable for an array kernel must be an array variable");
112 }
113 }
114}

116void
117ArrayIntegratedBC::computeJacobian()
118{
119 prepareMatrixTag(_assembly, _var.number(), _var.number());
120
121 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
122 {
123 initQpJacobian();
124 for (_i = 0; _i < _test.size(); _i++)
125 for (_j = 0; _j < _phi.size(); _j++)
126 {
127 RealEigenVector v = _JxW[_qp] * _coord[_qp] * computeQpJacobian();
128 _assembly.saveDiagLocalArrayJacobian(
129 _local_ke, _i, _test.size(), _j, _phi.size(), _var.number(), v);
130 }
131 }
132
133 accumulateTaggedLocalMatrix();
134
135 if (_has_diag_save_in)
136 {
137 DenseVector<Number> diag = _assembly.getJacobianDiagonal(_local_ke);
138 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
139 for (const auto & var : _diag_save_in)
140 {
141 auto * avar = dynamic_cast<ArrayMooseVariable *>(var);
142 if (avar)
143 avar->addSolution(diag);
144 else
145 mooseError("Save-in variable for an array kernel must be an array variable");
146 }
147 }
148}

156void
157ArrayIntegratedBC::computeOffDiagJacobian(const unsigned int jvar_num)
158{
159 const auto & jvar = getVariable(jvar_num);
160
161 bool same_var = jvar_num == _var.number();
162
163 prepareMatrixTag(_assembly, _var.number(), jvar_num);
164
165 // This (undisplaced) jvar could potentially yield the wrong phi size if this object is acting on
166 // the displaced mesh
167 auto phi_size = jvar.dofIndices().size();
168
169 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
170 {
171 initQpOffDiagJacobian(jvar);
172 for (_i = 0; _i < _test.size(); _i++)
173 for (_j = 0; _j < phi_size; _j++)
174 {
175 RealEigenMatrix v = _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobian(jvar);
176 _assembly.saveFullLocalArrayJacobian(
177 _local_ke, _i, _test.size(), _j, jvar.phiSize(), _var.number(), jvar_num, v);
178 }
179 }
180
181 accumulateTaggedLocalMatrix();
182
183 if (_has_diag_save_in && same_var)
184 {
185 DenseVector<Number> diag = _assembly.getJacobianDiagonal(_local_ke);
186 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
187 for (const auto & var : _diag_save_in)
188 {
189 auto * avar = dynamic_cast<ArrayMooseVariable *>(var);
190 if (avar)
191 avar->addSolution(diag);
192 else
193 mooseError("Save-in variable for an array kernel must be an array variable");
194 }
195 }
196}

207void
208ArrayIntegratedBC::computeOffDiagJacobianScalar(unsigned int jvar)
209{
210 prepareMatrixTag(_assembly, _var.number(), jvar);
211
212 const MooseVariableScalar & jv = _sys.getScalarVariable(_tid, jvar);
213 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
214 for (_i = 0; _i < _test.size(); _i++)
215 {
216 RealEigenMatrix v = _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobianScalar(jv);
217 _assembly.saveFullLocalArrayJacobian(
218 _local_ke, _i, _test.size(), 0, 1, _var.number(), jvar, v);
219 }
220
221 accumulateTaggedLocalMatrix();
222}
