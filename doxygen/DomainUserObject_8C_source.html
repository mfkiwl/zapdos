40 ElementIDInterface(this),
41 _mesh(_subproblem.mesh()),
42 _current_elem(_assembly.elem()),
43 _current_elem_volume(_assembly.elemVolume()),
44 _current_side(_assembly.side()),
45 _current_side_elem(_assembly.sideElem()),
46 _current_side_volume(_assembly.sideElemVolume()),
47 _neighbor_elem(_assembly.neighbor()),
48 _current_neighbor_volume(_assembly.neighborVolume()),
49 _current_boundary_id(_assembly.currentBoundaryID()),
50 _normals(_assembly.normals()),
51 _q_point(_assembly.qPoints()),
52 _qrule(_assembly.qRule()),
53 _JxW(_assembly.JxW()),
54 _q_point_face(_assembly.qPointsFace()),
55 _qrule_face(_assembly.qRuleFace()),
56 _JxW_face(_assembly.JxWFace()),
57 _coord(_assembly.coordTransformation())
58{
59 // Keep track of which variables are coupled so we know what we depend on
60 const std::vector<MooseVariableFEBase *> & coupled_vars = getCoupledMooseVars();
61 for (const auto & var : coupled_vars)
62 addMooseVariableDependency(var);
63
64 if (isParamValid("interface_boundaries"))
65 {
66 const auto & interface_boundaries = getParam<std::vector<BoundaryName>>("interface_boundaries");
67 _interface_bnd_ids = MooseMeshUtils::getBoundaryIDSet(_mesh, interface_boundaries, false);
68
69 for (const auto interface_bnd_id : _interface_bnd_ids)
70 {
71 // check on which side the interface is connected with the subdomain of this domain user
72 // object
73 const auto & primary_connected_blocks = _mesh.getBoundaryConnectedBlocks(interface_bnd_id);
74 bool has_full_primary_connection = true;
75 for (const auto interface_connected_block : primary_connected_blocks)
76 if (!hasBlocks(interface_connected_block))
77 has_full_primary_connection = false;
78
79 const auto & secondary_connected_blocks =
80 _mesh.getBoundaryConnectedSecondaryBlocks(interface_bnd_id);
81 bool has_full_secondary_connection = true;
82 for (const auto interface_connected_block : secondary_connected_blocks)
83 if (!hasBlocks(interface_connected_block))
84 has_full_secondary_connection = false;
85
86 // we push the subdomains on the other side for the interface
87 if (has_full_primary_connection)
88 _interface_connected_blocks[interface_bnd_id] = secondary_connected_blocks;
89 else if (has_full_secondary_connection)
90 _interface_connected_blocks[interface_bnd_id] = primary_connected_blocks;
91 else
92 paramError("interface_boundaries",
93 "Not all sides in the interface with ID ",
94 interface_bnd_id,
95 " is connected with the domain of the domain user object ",
96 name());
97 }
98 }

102DomainUserObject::getInterfaceFieldVar(const std::string & var_name,

105{
106 const auto * const field_var = getFieldVar(var_name, comp);
107 mooseAssert(field_var, "We should not be able to return a null variable");
108 if (interfaces)
109 _var_interfaces[field_var->name()] = *interfaces;

115void
116DomainUserObject::checkVariable(const MooseVariableFieldBase & variable) const
117{
118 auto it = _var_interfaces.find(variable.name());
119 if (it != _var_interfaces.end())
120 {
121 // we could have done this check in the constructor but to be consistent with other block
122 // restrictable checks, we'll do it here
123 auto & bnd_ids = it->second;
124 for (const auto bnd_id : bnd_ids)
125 {
126 const auto & connected_blocks = libmesh_map_find(_interface_connected_blocks, bnd_id);
127 for (const auto & bid : connected_blocks)
128 if (!variable.hasBlocks(bid))
129 mooseError("Variable '",
130 variable.name(),
131 "' is not defined on the interface connected block '",
132 _mesh.getSubdomainName(bid),
133 "'");
134 }
135 }
136 else
137 BlockRestrictable::checkVariable(variable);
138}
