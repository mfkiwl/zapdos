181bool
182FixedPointSolve::solve()
183{
184 TIME_SECTION("PicardSolve", 1);
185
186 Real current_dt = _problem.dt();
187
188 _fixed_point_timestep_begin_norm.clear();
189 _fixed_point_timestep_end_norm.clear();
190 _fixed_point_timestep_begin_norm.resize(_max_fixed_point_its);
191 _fixed_point_timestep_end_norm.resize(_max_fixed_point_its);
192
193 bool converged = true;
194
195 // need to back up multi-apps even when not doing fixed point iteration for recovering from failed
196 // multiapp solve
197 _problem.backupMultiApps(EXEC_MULTIAPP_FIXED_POINT_BEGIN);
198 _problem.backupMultiApps(EXEC_TIMESTEP_BEGIN);
199 _problem.backupMultiApps(EXEC_TIMESTEP_END);
200 _problem.backupMultiApps(EXEC_MULTIAPP_FIXED_POINT_END);
201
202 // Prepare to relax variables as a main app
203 std::set<dof_id_type> transformed_dofs;
204 if ((_relax_factor != 1.0 || !dynamic_cast<PicardSolve *>(this)) && _transformed_vars.size() > 0)
205 {
206 // Snag all of the local dof indices for all of these variables
207 AllLocalDofIndicesThread aldit(_problem, _transformed_vars);
208 ConstElemRange & elem_range = *_problem.mesh().getActiveLocalElementRange();
209 Threads::parallel_reduce(elem_range, aldit);
210
211 transformed_dofs = aldit.getDofIndices();
212 }
213
214 // Prepare to relax variables as a subapp
215 std::set<dof_id_type> secondary_transformed_dofs;
216 if (_secondary_relaxation_factor != 1.0 || !dynamic_cast<PicardSolve *>(this))
217 {
218 if (_secondary_transformed_variables.size() > 0)
219 {
220 // Snag all of the local dof indices for all of these variables
221 AllLocalDofIndicesThread aldit(_problem, _secondary_transformed_variables);
222 ConstElemRange & elem_range = *_problem.mesh().getActiveLocalElementRange();
223 Threads::parallel_reduce(elem_range, aldit);
224
225 secondary_transformed_dofs = aldit.getDofIndices();
226 }
227
228 // To detect a new time step
229 if (_old_entering_time == _problem.time())
230 {
231 // Keep track of the iteration number of the main app
232 _main_fixed_point_it++;
233
234 // Save variable values before the solve. Solving will provide new values
235 saveVariableValues(false);
236 }
237 else
238 _main_fixed_point_it = 0;
239 }
240
241 for (_fixed_point_it = 0; _fixed_point_it < _max_fixed_point_its; ++_fixed_point_it)
242 {
243 if (_has_fixed_point_its)
244 {
245 if (_fixed_point_it == 0)
246 {
247 if (_has_fixed_point_norm)
248 {
249 // First fixed point iteration - need to save off the initial nonlinear residual
250 _fixed_point_initial_norm = _problem.computeResidualL2Norm();
251 _console << COLOR_MAGENTA << "Initial fixed point residual norm: " << COLOR_DEFAULT;
252 if (_fixed_point_initial_norm == std::numeric_limits<Real>::max())
253 _console << " MAX ";
254 else
255 _console << std::scientific << _fixed_point_initial_norm;
256 _console << COLOR_DEFAULT << "\n" << std::endl;
257 }
258 }
259 else
260 {
261 // For every iteration other than the first, we need to restore the state of the MultiApps
262 _problem.restoreMultiApps(EXEC_MULTIAPP_FIXED_POINT_BEGIN);
263 _problem.restoreMultiApps(EXEC_TIMESTEP_BEGIN);
264 _problem.restoreMultiApps(EXEC_TIMESTEP_END);
265 _problem.restoreMultiApps(EXEC_MULTIAPP_FIXED_POINT_END);
266 }
267
268 _console << COLOR_MAGENTA << "Beginning fixed point iteration " << _fixed_point_it
269 << COLOR_DEFAULT << std::endl
270 << std::endl;
271 }
272
273 // Save last postprocessor value as value before solve
274 if (_fixed_point_custom_pp && _fixed_point_it > 0 && !getParam<bool>("direct_pp_value"))
275 _pp_old = *_fixed_point_custom_pp;
276
277 // Solve a single application for one time step
278 bool solve_converged = solveStep(_fixed_point_timestep_begin_norm[_fixed_point_it],
279 _fixed_point_timestep_end_norm[_fixed_point_it],
280 transformed_dofs);
281
282 // Get new value and print history for the custom postprocessor convergence criterion
283 if (_fixed_point_custom_pp)
284 computeCustomConvergencePostprocessor();
285
286 if (solve_converged)
287 {
288 if (_has_fixed_point_its)
289 {
290 if (_has_fixed_point_norm)
291 // Print the evolution of the main app residual over the fixed point iterations
292 printFixedPointConvergenceHistory();
293
294 // Examine convergence metrics & properties and set the convergence reason
295 bool break_out = examineFixedPointConvergence(converged);
296 if (break_out)
297 break;
298 }
299 }
300 else
301 {
302 // If the last solve didn't converge then we need to exit this step completely (even in the
303 // case of coupling). So we can retry...
304 converged = false;
305 break;
306 }
307
308 if (converged)
309 {
310 // Fixed point iteration loop ends right above
311 _problem.execute(EXEC_MULTIAPP_FIXED_POINT_END);
312 _problem.execTransfers(EXEC_MULTIAPP_FIXED_POINT_END);
313 if (!_problem.execMultiApps(EXEC_MULTIAPP_FIXED_POINT_END, autoAdvance()))
314 {
315 _fixed_point_status = MooseFixedPointConvergenceReason::DIVERGED_FAILED_MULTIAPP;
316 return false;
317 }
318 _problem.outputStep(EXEC_MULTIAPP_FIXED_POINT_END);
319 }
320
321 _problem.dt() =
322 current_dt; // _dt might be smaller than this at this point for multistep methods
323 }
324
325 // Save postprocessors after the solve and their potential timestep_end execution
326 // The postprocessors could be overwritten at timestep_begin, which is why they are saved
327 // after the solve. They could also be saved right after the transfers.
328 if (_old_entering_time == _problem.time())
329 savePostprocessorValues(false);
330
331 if (converged)
332 {
333 // Update the subapp using the fixed point algorithm
334 if (_secondary_transformed_variables.size() > 0 &&
335 useFixedPointAlgorithmUpdateInsteadOfPicard(false) && _old_entering_time == _problem.time())
336 transformVariables(secondary_transformed_dofs, false);
337
338 // Update the entering time, used to detect failed solves
339 _old_entering_time = _problem.time();
340 }
341
342 if (_has_fixed_point_its)
343 printFixedPointConvergenceReason();
344
345 // clear history to avoid displaying it again on next solve that can happen for example during
346 // transient
347 _pp_history.str("");
348
349 return converged;
350}

359bool
360FixedPointSolve::solveStep(Real & begin_norm,
361 Real & end_norm,
362 const std::set<dof_id_type> & transformed_dofs)
363{
364 bool auto_advance = autoAdvance();
365
366 // Compute previous norms for coloring the norm output
367 Real begin_norm_old = (_fixed_point_it > 0 ? _fixed_point_timestep_begin_norm[_fixed_point_it - 1]
368 : std::numeric_limits<Real>::max());
369 Real end_norm_old = (_fixed_point_it > 0 ? _fixed_point_timestep_end_norm[_fixed_point_it - 1]
370 : std::numeric_limits<Real>::max());
371
372 _executioner.preSolve();
373 _problem.execTransfers(EXEC_TIMESTEP_BEGIN);
374
375 if (_fixed_point_it == 0)
376 {
377 _problem.execute(EXEC_MULTIAPP_FIXED_POINT_BEGIN);
378 _problem.execTransfers(EXEC_MULTIAPP_FIXED_POINT_BEGIN);
379 if (!_problem.execMultiApps(EXEC_MULTIAPP_FIXED_POINT_BEGIN, autoAdvance()))
380 {
381 _fixed_point_status = MooseFixedPointConvergenceReason::DIVERGED_FAILED_MULTIAPP;
382 return false;
383 }
384 _problem.outputStep(EXEC_MULTIAPP_FIXED_POINT_BEGIN);
385 }
386
387 if (!_problem.execMultiApps(EXEC_TIMESTEP_BEGIN, auto_advance))
388 {
389 _fixed_point_status = MooseFixedPointConvergenceReason::DIVERGED_FAILED_MULTIAPP;
390 return false;
391 }
392
393 if (_problem.haveXFEM() && _update_xfem_at_timestep_begin)
394 _problem.updateMeshXFEM();
395
396 _problem.execute(EXEC_TIMESTEP_BEGIN);
397
398 // Transform the fixed point postprocessors before solving, but after the timestep_begin transfers
399 // have been received
400 if (_transformed_pps.size() > 0 && useFixedPointAlgorithmUpdateInsteadOfPicard(true))
401 transformPostprocessors(true);
402 if (_secondary_transformed_pps.size() > 0 && useFixedPointAlgorithmUpdateInsteadOfPicard(false) &&
403 _problem.time() == _old_entering_time)
404 transformPostprocessors(false);
405
406 if (_has_fixed_point_its && _has_fixed_point_norm)
407 if (_problem.hasMultiApps(EXEC_TIMESTEP_BEGIN) || _fixed_point_force_norms)
408 {
409 begin_norm = _problem.computeResidualL2Norm();
410
411 _console << COLOR_MAGENTA << "Fixed point residual norm after TIMESTEP_BEGIN MultiApps: "
412 << Console::outputNorm(begin_norm_old, begin_norm) << std::endl;
413 }
414
415 // Perform output for timestep begin
416 _problem.outputStep(EXEC_TIMESTEP_BEGIN);
417
418 // Update warehouse active objects
419 _problem.updateActiveObjects();
420
421 // Save the current values of variables and postprocessors, before the solve
422 saveAllValues(true);
423
424 if (_has_fixed_point_its)
425 _console << COLOR_MAGENTA << "\nMain app solve:" << COLOR_DEFAULT << std::endl;
426 if (!_inner_solve->solve())
427 {
428 _fixed_point_status = MooseFixedPointConvergenceReason::DIVERGED_NONLINEAR;
429
430 // Perform the output of the current, failed time step (this only occurs if desired)
431 _problem.outputStep(EXEC_FAILED);
432 return false;
433 }
434 else
435 _fixed_point_status = MooseFixedPointConvergenceReason::CONVERGED_NONLINEAR;
436
437 // Use the fixed point algorithm if the conditions (availability of values, etc) are met
438 if (_transformed_vars.size() > 0 && useFixedPointAlgorithmUpdateInsteadOfPicard(true))
439 transformVariables(transformed_dofs, true);
440
441 if (_problem.haveXFEM() && (_xfem_update_count < _max_xfem_update) && _problem.updateMeshXFEM())
442 {
443 _console << "\nXFEM modified mesh, repeating step" << std::endl;
444 _xfem_repeat_step = true;
445 ++_xfem_update_count;
446 }
447 else
448 {
449 if (_problem.haveXFEM())
450 {
451 _xfem_repeat_step = false;
452 _xfem_update_count = 0;
453 _console << "\nXFEM did not modify mesh, continuing" << std::endl;
454 }
455
456 _problem.onTimestepEnd();
457 _problem.execute(EXEC_TIMESTEP_END);
458
459 _problem.execTransfers(EXEC_TIMESTEP_END);
460 if (!_problem.execMultiApps(EXEC_TIMESTEP_END, auto_advance))
461 {
462 _fixed_point_status = MooseFixedPointConvergenceReason::DIVERGED_FAILED_MULTIAPP;
463 return false;
464 }
465 }
466
467 if (_fail_step)
468 {
469 _fail_step = false;
470 return false;
471 }
472
473 _executioner.postSolve();
474
475 if (_has_fixed_point_its && _has_fixed_point_norm)
476 if (_problem.hasMultiApps(EXEC_TIMESTEP_END) || _fixed_point_force_norms)
477 {
478 end_norm = _problem.computeResidualL2Norm();
479
480 _console << COLOR_MAGENTA << "Fixed point residual norm after TIMESTEP_END MultiApps: "
481 << Console::outputNorm(end_norm_old, end_norm) << std::endl;
482 }
483
484 return true;
485}

501bool
502FixedPointSolve::examineFixedPointConvergence(bool & converged)
503{
504 if (_fixed_point_it + 2 > _min_fixed_point_its)
505 {
506 Real max_norm = std::max(_fixed_point_timestep_begin_norm[_fixed_point_it],
507 _fixed_point_timestep_end_norm[_fixed_point_it]);
508
509 Real max_relative_drop = max_norm / _fixed_point_initial_norm;
510
511 if (_has_fixed_point_norm && max_norm < _fixed_point_abs_tol)
512 {
513 _fixed_point_status = MooseFixedPointConvergenceReason::CONVERGED_ABS;
514 return true;
515 }
516 if (_has_fixed_point_norm && max_relative_drop < _fixed_point_rel_tol)
517 {
518 _fixed_point_status = MooseFixedPointConvergenceReason::CONVERGED_RELATIVE;
519 return true;
520 }
521 }
522 if (std::abs(_pp_new - _pp_old) < _custom_abs_tol)
523 {
524 _fixed_point_status = MooseFixedPointConvergenceReason::CONVERGED_CUSTOM;
525 return true;
526 }
527 if (std::abs((_pp_new - _pp_old) / _pp_scaling) < _custom_rel_tol)
528 {
529 _fixed_point_status = MooseFixedPointConvergenceReason::CONVERGED_CUSTOM;
530 return true;
531 }
532 if (_fixed_point_it + 1 == _max_fixed_point_its)
533 {
534 if (_accept_max_it)
535 {
536 _fixed_point_status = MooseFixedPointConvergenceReason::REACH_MAX_ITS;
537 converged = true;
538 }
539 else
540 {
541 _fixed_point_status = MooseFixedPointConvergenceReason::DIVERGED_MAX_ITS;
542 converged = false;
543 }
544 return true;
545 }
546 return false;
547}

549void
550FixedPointSolve::printFixedPointConvergenceReason()
551{
552 _console << "Fixed point convergence reason: ";
553 switch (_fixed_point_status)
554 {
555 case MooseFixedPointConvergenceReason::CONVERGED_ABS:
556 _console << "CONVERGED_ABS";
557 break;
558 case MooseFixedPointConvergenceReason::CONVERGED_RELATIVE:
559 _console << "CONVERGED_RELATIVE";
560 break;
561 case MooseFixedPointConvergenceReason::CONVERGED_CUSTOM:
562 _console << "CONVERGED_CUSTOM";
563 break;
564 case MooseFixedPointConvergenceReason::REACH_MAX_ITS:
565 _console << "REACH_MAX_ITS";
566 break;
567 case MooseFixedPointConvergenceReason::DIVERGED_MAX_ITS:
568 _console << "DIVERGED_MAX_ITS";
569 break;
570 case MooseFixedPointConvergenceReason::DIVERGED_NONLINEAR:
571 _console << "DIVERGED_NONLINEAR";
572 break;
573 case MooseFixedPointConvergenceReason::DIVERGED_FAILED_MULTIAPP:
574 _console << "DIVERGED_FAILED_MULTIAPP";
575 break;
576 default:
577 // UNSOLVED and CONVERGED_NONLINEAR should not be hit when coupling
578 // iteration is not on here
579 mooseError("Internal error: wrong fixed point status!");
580 break;
581 }
582 _console << std::endl;
583}

585bool
586FixedPointSolve::autoAdvance() const
587{
588 bool auto_advance = !(_has_fixed_point_its && _problem.isTransient());
589
590 if (dynamic_cast<EigenExecutionerBase *>(&_executioner) && _has_fixed_point_its)
591 auto_advance = true;
592
593 if (_auto_advance_set_by_user)
594 auto_advance = _auto_advance_user_value;
595
596 return auto_advance;
