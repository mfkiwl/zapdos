34ParsedAuxScalar::ParsedAuxScalar(const InputParameters & parameters)
35 : AuxScalarKernel(parameters),
36 FunctionParserUtils<false>(parameters),
37 _function(getParam<std::string>("function")),
38 _nargs(coupledScalarComponents("args")),
39 _args(_nargs)
40{
41 // build variables argument
42 std::string variables;
43 for (unsigned int i = 0; i < _nargs; ++i)
44 {
45 variables += (i == 0 ? "" : ",") + getScalarVar("args", i)->name();
46 _args[i] = &coupledScalarValue("args", i);
47 }
48
49 // base function object
50 _func_F = SymFunctionPtr(std::make_shared<SymFunction>());
51
52 // set FParser interneal feature flags
53 setParserFeatureFlags(_func_F);
54
55 // add the constant expressions
56 addFParserConstants(_func_F,
57 getParam<std::vector<std::string>>("constant_names"),
58 getParam<std::vector<std::string>>("constant_expressions"));
59
60 // parse function
61 if (_func_F->Parse(_function, variables) >= 0)
62 mooseError("Invalid function\n",
63 _function,
64 "\nin ParsedAuxScalar ",
65 name(),
66 ".\n",
67 _func_F->ErrorMsg());
68
69 // optimize
70 if (!_disable_fpoptimizer)
71 _func_F->Optimize();
72
73 // just-in-time compile
74 if (_enable_jit)
75 _func_F->JITCompile();
76
77 // reserve storage for parameter passing bufefr
78 _func_params.resize(_nargs);
79}
