37DGCoeffDiffusion::computeQpResidual(Moose::DGResidualType type)
38{
39 ADReal r = 0;
40
41 const unsigned int elem_b_order = static_cast<unsigned int>(_var.order());
42 const double h_elem =
43 _current_elem->volume() / _current_side_elem->volume() * 1. / std::pow(elem_b_order, 2.);
44
45 switch (type)
46 {
47 case Moose::Element:
48 r += 0.5 * (-_diff[_qp] * std::exp(_u[_qp]) * _grad_u[_qp] * _normals[_qp] * _test[_i][_qp] +
49 _epsilon * _grad_test[_i][_qp] * _normals[_qp] * std::exp(_u[_qp]));
50 r += _sigma / h_elem * std::exp(_u[_qp]) * _test[_i][_qp];
51
52 r += 0.5 * (-_diff_neighbor[_qp] * std::exp(_u_neighbor[_qp]) * _grad_u_neighbor[_qp] *
53 _normals[_qp] * _test[_i][_qp] -
54 _epsilon * _grad_test[_i][_qp] * _normals[_qp] * std::exp(_u_neighbor[_qp]));
55 r += -_sigma / h_elem * std::exp(_u_neighbor[_qp]) * _test[_i][_qp];
56 break;
57
58 case Moose::Neighbor:
59 r += 0.5 *
60 (_diff[_qp] * std::exp(_u[_qp]) * _grad_u[_qp] * _normals[_qp] +
61 _diff_neighbor[_qp] * std::exp(_u_neighbor[_qp]) * _grad_u_neighbor[_qp] *
62 _normals[_qp]) *
63 _test_neighbor[_i][_qp];
64 r += _epsilon * 0.5 * _grad_test_neighbor[_i][_qp] * _normals[_qp] *
65 (std::exp(_u[_qp]) - std::exp(_u_neighbor[_qp]));
66 r -= _sigma / h_elem * (std::exp(_u[_qp]) - std::exp(_u_neighbor[_qp])) *
67 _test_neighbor[_i][_qp];
68 break;
69 }
70
71 return r;
72}
