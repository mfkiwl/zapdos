68ADReal
69SakiyamaEnergySecondaryElectronBC::computeQpResidual()
70{
71 _ion_flux.zero();
72 for (unsigned int i = 0; i < _num_ions; ++i)
73 {
74 if (_normals[_qp] * (*_sgnip[i])[_qp] * -_grad_potential[_qp] > 0.0)
75 {
76 _a = 1.0;
77 }
78 else
79 {
80 _a = 0.0;
81 }
82
83 _ion_flux += _a * (*_sgnip[i])[_qp] * (*_muip[i])[_qp] * -_grad_potential[_qp] * _r_units *
84 std::exp((*_ip[i])[_qp]);
85 }
86
87 if (Te_dependent)
88 {
89 _se_energy = std::exp(_u[_qp] - _em[_qp]);
90 }
91 else
92 {
93 _se_energy = _user_se_energy;
94 }
95
96 return -_test[_i][_qp] * _r_units * _se_coeff * (5.0 / 3.0) * _se_energy * _ion_flux *
97 _normals[_qp];
98}
