39Real
40MaxIncrement::computeQpDamping()
41{
42 Real inc = std::abs(_u_increment[_qp]);
43 if (_increment_type == IncrementTypeEnum::fractional)
44 inc /= std::abs(_u[_qp]);
45
46 if (inc > _max_increment)
47 return _max_increment / inc;
48
49 return 1.0;
50}
