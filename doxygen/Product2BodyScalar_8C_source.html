50Real
51Product2BodyScalar::computeQpJacobian()
52{
53 Real power, eq_u_mult, gas_mult;
54 power = 0.0;
55 eq_u_mult = 1.0;
56 gas_mult = 1.0;
57
58 if (isCoupledScalar("v") && _v_eq_u)
59 {
60 power += 1.0;
61 eq_u_mult = _v[_i];
62 }
63 else
64 gas_mult *= _v[_i];
65
66 if (isCoupledScalar("w") && _w_eq_u)
67 {
68 power += 1.0;
69 eq_u_mult = _w[_i];
70 }
71 else
72 gas_mult *= _w[_i];
73
74 return -_stoichiometric_coeff * _rate_coefficient[_i] * gas_mult * power *
75 std::pow(eq_u_mult, power - 1);
76}

78Real
79Product2BodyScalar::computeQpOffDiagJacobian(unsigned int jvar)
80{
81 Real power, deriv_factor, other_factor;
82 power = 0;
83 other_factor = 1;
84 deriv_factor = 1;
85
86 if (jvar == _v_var && !_v_eq_u)
87 {
88 power += 1;
89 deriv_factor = _v[_i];
90 }
91 else
92 other_factor *= _v[_i];
93
94 if (jvar == _w_var && !_w_eq_u)
95 {
96 power += 1;
97 deriv_factor = _w[_i];
98 }
99 else
100 other_factor *= _w[_i];
101
102 return -_stoichiometric_coeff * _rate_coefficient[_i] * other_factor * power *
103 std::pow(deriv_factor, power - 1);
104}
