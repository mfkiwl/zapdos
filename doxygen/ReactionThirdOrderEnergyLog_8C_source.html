73Real
74ReactionThirdOrderEnergyLog::computeQpOffDiagJacobian(unsigned int jvar)
75{
76 Real power;
77 power = 0;
78
79 if (!_v_eq_u && jvar == _v_id)
80 power += 1;
81
82 if (!_w_eq_u && jvar == _w_id)
83 power += 1;
84
85 if (!_x_eq_u && jvar == _x_id)
86 power += 1;
87
88 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] *
89 std::exp(_v[_qp] + _w[_qp] + _x[_qp]) * power * _phi[_j][_qp] * _threshold_energy;
90}
