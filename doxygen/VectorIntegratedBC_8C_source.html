42void
43VectorIntegratedBC::computeResidual()
44{
45 prepareVectorTag(_assembly, _var.number());
46
47 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
48 for (_i = 0; _i < _test.size(); _i++)
49 {
50 Real residual = _JxW[_qp] * _coord[_qp] * computeQpResidual();
51 _local_re(_i) += residual;
52 }
53
54 accumulateTaggedLocalResidual();
55}

57void
58VectorIntegratedBC::computeJacobian()
59{
60 prepareMatrixTag(_assembly, _var.number(), _var.number());
61
62 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
63 for (_i = 0; _i < _test.size(); _i++)
64 for (_j = 0; _j < _phi.size(); _j++)
65 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeQpJacobian();
66
67 accumulateTaggedLocalMatrix();
68}

70void
71VectorIntegratedBC::computeOffDiagJacobian(const unsigned int jvar_num)
72{
73 const auto & jvar = getVariable(jvar_num);
74
75 prepareMatrixTag(_assembly, _var.number(), jvar_num);
76
77 // This (undisplaced) jvar could potentially yield the wrong phi size if this object is acting on
78 // the displaced mesh
79 auto phi_size = jvar.dofIndices().size();
80
81 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
82 for (_i = 0; _i < _test.size(); _i++)
83 for (_j = 0; _j < phi_size; _j++)
84 {
85 if (_var.number() == jvar_num)
86 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeQpJacobian();
87 else
88 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobian(jvar_num);
89 }
90
91 accumulateTaggedLocalMatrix();
92}

94void
95VectorIntegratedBC::computeOffDiagJacobianScalar(unsigned int jvar)
96{
97 prepareMatrixTag(_assembly, _var.number(), jvar);
98
99 MooseVariableScalar & jv = _sys.getScalarVariable(_tid, jvar);
100 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
101 for (_i = 0; _i < _test.size(); _i++)
102 for (_j = 0; _j < jv.order(); _j++)
103 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobianScalar(jvar);
104
105 accumulateTaggedLocalMatrix();
106}
