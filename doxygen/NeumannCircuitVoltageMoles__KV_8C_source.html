38NeumannCircuitVoltageMoles_KV::NeumannCircuitVoltageMoles_KV(const InputParameters & parameters)
39 : ADIntegratedBC(parameters),
40 _r_units(1. / getParam<Real>("position_units")),
41 _V_bat(getFunction("function")),
42 _data(getUserObject<ProvideMobility>("data_provider")),
43 _mean_en(adCoupledValue("mean_en")),
44 _em(adCoupledValue("em")),
45 _se_coeff(getMaterialProperty<Real>("se_coeff")),
46 _eps(getMaterialProperty<Real>("eps")),
47 _N_A(getMaterialProperty<Real>("N_A")),
48 _muem(getADMaterialProperty<Real>("muem")),
49 _e(getMaterialProperty<Real>("e")),
50 _massem(getMaterialProperty<Real>("massem")),
51 _kb(getMaterialProperty<Real>("k_boltz")),
52
53 _potential_units(getParam<std::string>("potential_units")),
54 _r(getParam<Real>("r"))
55{
56 _ion_flux.zero();
57 _n_gamma = 0.0;
58 _actual_mean_en = 0.0;
59 _v_e_th = 0.0;
60 _v_i_th = 0.0;
61 _a = 0.0;
62 _b = 0.0;
63 _numerator = 0.0;
64 _denominator = 0.0;
65 if (_potential_units.compare("V") == 0)
66 _voltage_scaling = 1.;
67 else if (_potential_units.compare("kV") == 0)
68 _voltage_scaling = 1000;
69
70 // First we need to initialize all of the ion densities and material properties.
71 // Find the number of ions coupled into this BC:
72 _num_ions = coupledComponents("ip");
73
74 // Resize the vectors to store _num_ions values:
75 _ip.resize(_num_ions);
76 _grad_ip.resize(_num_ions);
77 _T_heavy.resize(_num_ions);
78 _muip.resize(_num_ions);
79 _Dip.resize(_num_ions);
80 _sgnip.resize(_num_ions);
81 _mass.resize(_num_ions);
82
83 // Retrieve the values for each ion and store in the relevant vectors.
84 // Note that these need to be dereferenced to get the values inside the
85 // main body of the code.
86 // e.g. instead of "_ip[_qp]" it would be "(*_ip[i])[_qp]", where "i"
87 // refers to a single ion species.
88 for (unsigned int i = 0; i < _ip.size(); ++i)
89 {
90 _ip[i] = &adCoupledValue("ip", i);
91 _grad_ip[i] = &adCoupledGradient("ip", i);
92 _T_heavy[i] = &getADMaterialProperty<Real>("T" + (*getVar("ip", i)).name());
93 _muip[i] = &getADMaterialProperty<Real>("mu" + (*getVar("ip", i)).name());
94 _Dip[i] = &getADMaterialProperty<Real>("diff" + (*getVar("ip", i)).name());
95 _sgnip[i] = &getMaterialProperty<Real>("sgn" + (*getVar("ip", i)).name());
96 _mass[i] = &getMaterialProperty<Real>("mass" + (*getVar("ip", i)).name());
97 }
98}

100ADReal
101NeumannCircuitVoltageMoles_KV::computeQpResidual()
102{
103 if (_normals[_qp] * -1.0 * -_grad_u[_qp] > 0.0)
104 {
105 _a = 1.0;
106 _b = 0.;
107 }
108 else
109 {
110 _a = 0.0;
111 _b = 1.;
112 }
113
114 _ion_flux.zero();
115 _n_gamma = 0;
116 _secondary_ion = 0;
117 _ion_drift = 0;
118 for (unsigned int i = 0; i < _num_ions; ++i)
119 {
120 _ion_flux +=
121 (*_sgnip[i])[_qp] * (*_muip[i])[_qp] * -_grad_u[_qp] * _r_units * std::exp((*_ip[i])[_qp]) -
122 (*_Dip[i])[_qp] * std::exp((*_ip[i])[_qp]) * (*_grad_ip[i])[_qp] * _r_units;
123
124 _secondary_ion += std::exp((*_ip[i])[_qp]) * (*_muip[i])[_qp];
125
126 _ion_drift += std::sqrt(8 * _kb[_qp] * (*_T_heavy[i])[_qp] / (M_PI * (*_mass[i])[_qp])) *
127 std::exp((*_ip[i])[_qp]);
128 }
129 _n_gamma = (1. - _a) * _se_coeff[_qp] * _ion_flux * _normals[_qp] /
130 (_muem[_qp] * -_grad_u[_qp] * _r_units * _normals[_qp]);
131
132 _v_e_th = std::sqrt(8 * _data.coulomb_charge() * 2.0 / 3 * std::exp(_mean_en[_qp] - _em[_qp]) /
133 (M_PI * _massem[_qp]));
134
135 return _test[_i][_qp] * _r_units * _eps[_qp] *
136 (-2. * (1. + _r) * _u[_qp] - 2. * (1. + _r) * -_V_bat.value(_t, _q_point[_qp]) +
137 _data.electrode_area() * _data.coulomb_charge() * _data.ballast_resist() /
138 _voltage_scaling * (-1. + _r) *
139 ((-1. + (-1. + _a) * _se_coeff[_qp]) * _N_A[_qp] * _ion_drift +
140 _N_A[_qp] * (std::exp(_em[_qp]) - _n_gamma) * _v_e_th)) /
141 (2. * _data.electrode_area() * _data.coulomb_charge() *
142 ((-1. + 2. * _a) * _muem[_qp] / _voltage_scaling * _N_A[_qp] *
143 (std::exp(_em[_qp]) - _n_gamma) -
144 (-1. + 2. * _b) * (-1. + (-1. + _a) * _se_coeff[_qp]) * _secondary_ion /
145 _voltage_scaling * _N_A[_qp]) *
146 _data.ballast_resist() * (-1. + _r));
147}
