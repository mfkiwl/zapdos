39void
40SolutionIC::initialSetup()
41{
42 // can only check blocks when the solution UO uses Exodus
43 if (_solution_object.getSolutionFileType() == "exodusII")
44 {
45 // remap block names this IC is defined on into the ExodusII file block IDs
46 const auto & block_names_to_ids_from = _solution_object.getBlockNamesToIds();
47
48 const std::vector<SubdomainID> all_block_ids(meshBlockIDs().begin(), meshBlockIDs().end());
49 const auto blocks_to_check =
50 blockRestricted() ? blocks() : _sys.mesh().getSubdomainNames(all_block_ids);
51
52 for (auto & blk_name : blocks_to_check)
53 {
54 auto it = block_names_to_ids_from.find(blk_name);
55 if (it != block_names_to_ids_from.end())
56 _exo_block_ids.insert(it->second);
57 else
58 {
59 auto blk_id = _sys.mesh().getSubdomainID(blk_name);
60 // use the ids, it may be that the source file does not have block names
61 // and that we are using the id as the block name (block = 0 for example)
62 const auto & block_ids_to_names_from = _solution_object.getBlockIdsToNames();
63 if (block_ids_to_names_from.find(blk_id) != block_ids_to_names_from.end() &&
64 block_ids_to_names_from.find(blk_id)->second.empty())
65 _exo_block_ids.insert(blk_id);
66 else
67 mooseError("Block '",
68 blk_name,
69 "' does not exist in the file '",
70 _solution_object.getMeshFileName(),
71 "'.");
72 }
73 }
74 }
75}
