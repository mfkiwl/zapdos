379inline const MaterialPropertyStorage::PropsType &
380MaterialPropertyStorage::props(const unsigned int state) const
381{
382 mooseAssert(state < _storage.size(), "Invalid material property state " + std::to_string(state));
383 return _storage[state];
384}

386inline const MaterialProperties &
387MaterialPropertyStorage::props(const Elem * elem, unsigned int side, const unsigned int state) const
388{
389 const auto find_elem = props(state).find(elem);
390 mooseAssert(find_elem != props(state).end(), "Material property does not have elem entry");
391 const auto find_side = find_elem->second.find(side);
392 mooseAssert(find_side != find_elem->second.end(), "Material property does not have side entry");
393 return find_side->second;
394}
