75void
76ExplicitTimeIntegrator::meshChanged()
77{
78 // Can only be done after the system is initialized
79 if (_solve_type == LUMPED || _solve_type == LUMP_PRECONDITIONED)
80 *_ones = 1.;
81
82 if (_solve_type == CONSISTENT || _solve_type == LUMP_PRECONDITIONED)
83 _linear_solver = LinearSolver<Number>::build(comm());
84
85 if (_solve_type == LUMP_PRECONDITIONED)
86 {
87 _preconditioner = std::make_unique<LumpedPreconditioner>(_mass_matrix_diag);
88 _linear_solver->attach_preconditioner(_preconditioner.get());
89 _linear_solver->init();
90 }
91
92 if (_solve_type == CONSISTENT || _solve_type == LUMP_PRECONDITIONED)
93 Moose::PetscSupport::setLinearSolverDefaults(_fe_problem, *_linear_solver);
94}

96bool
97ExplicitTimeIntegrator::performExplicitSolve(SparseMatrix<Number> & mass_matrix)
98{
99 bool converged = false;
100
101 switch (_solve_type)
102 {
103 case CONSISTENT:
104 {
105 converged = solveLinearSystem(mass_matrix);
106
107 break;
108 }
109 case LUMPED:
110 {
111 // Computes the sum of each row (lumping)
112 // Note: This is actually how PETSc does it
113 // It's not "perfectly optimal" - but it will be fast (and universal)
114 mass_matrix.vector_mult(_mass_matrix_diag, *_ones);
115
116 // "Invert" the diagonal mass matrix
117 _mass_matrix_diag.reciprocal();
118
119 // Multiply the inversion by the RHS
120 _solution_update.pointwise_mult(_mass_matrix_diag, _explicit_residual);
121
122 // Check for convergence by seeing if there is a nan or inf
123 auto sum = _solution_update.sum();
124 converged = std::isfinite(sum);
125
126 // The linear iteration count remains zero
127 _n_linear_iterations = 0;
128
129 break;
130 }
131 case LUMP_PRECONDITIONED:
132 {
133 mass_matrix.vector_mult(_mass_matrix_diag, *_ones);
134 _mass_matrix_diag.reciprocal();
135
136 converged = solveLinearSystem(mass_matrix);
137
138 break;
139 }
140 default:
141 mooseError("Unknown solve_type in ExplicitTimeIntegrator.");
142 }
143
144 return converged;
145}

147bool
148ExplicitTimeIntegrator::solveLinearSystem(SparseMatrix<Number> & mass_matrix)
149{
150 auto & es = _fe_problem.es();
151
152 const auto num_its_and_final_tol =
153 _linear_solver->solve(mass_matrix,
154 _solution_update,
155 _explicit_residual,
156 es.parameters.get<Real>("linear solver tolerance"),
157 es.parameters.get<unsigned int>("linear solver maximum iterations"));
158
159 _n_linear_iterations += num_its_and_final_tol.first;
160
161 const bool converged = checkLinearConvergence();
162
163 return converged;
164}

166bool
167ExplicitTimeIntegrator::checkLinearConvergence()
168{
169 auto reason = _linear_solver->get_converged_reason();
170
171 switch (reason)
172 {
173 case CONVERGED_RTOL_NORMAL:
174 case CONVERGED_ATOL_NORMAL:
175 case CONVERGED_RTOL:
176 case CONVERGED_ATOL:
177 case CONVERGED_ITS:
178 case CONVERGED_CG_NEG_CURVE:
179 case CONVERGED_CG_CONSTRAINED:
180 case CONVERGED_STEP_LENGTH:
181 case CONVERGED_HAPPY_BREAKDOWN:
182 return true;
183 case DIVERGED_NULL:
184 case DIVERGED_ITS:
185 case DIVERGED_DTOL:
186 case DIVERGED_BREAKDOWN:
187 case DIVERGED_BREAKDOWN_BICG:
188 case DIVERGED_NONSYMMETRIC:
189 case DIVERGED_INDEFINITE_PC:
190 case DIVERGED_NAN:
191 case DIVERGED_INDEFINITE_MAT:
192 case CONVERGED_ITERATING:
193 case DIVERGED_PCSETUP_FAILED:
194 return false;
195 default:
196 mooseError("Unknown convergence reason in ExplicitTimeIntegrator.");
197 }
