52void
53GhostingAux::precalculateValue()
54{
55 if (_current_elem->processor_id() == _pid && _include_local)
56 _value = 1;
57 else if (_current_elem->processor_id() != _pid)
58 _value = _ghost_uo.getElementalValue(_current_elem, _rm_type, _pid);
59 else
60 _value = 0.;
61}
