48void
49MassLumpedTimeDerivative::computeJacobian()
50{
51 prepareMatrixTag(_assembly, _var.number(), _var.number());
52 for (_i = 0; _i < _test.size(); _i++)
53 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
54 _local_ke(_i, _i) += _JxW[_qp] * _coord[_qp] * computeQpJacobian();
55 accumulateTaggedLocalMatrix();
56}
