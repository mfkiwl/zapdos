38template <bool is_ad>
39void
40PiecewiseConstantByBlockMaterialTempl<is_ad>::computeQpProperties()
41{
42 if (!_bnd)
43 {
44 mooseAssert(_current_elem,
45 "We should be on a block which means we should definitely have a current element");
46 auto it = _sub_id_to_prop.find(_current_elem->subdomain_id());
47 mooseAssert(it != _sub_id_to_prop.end(),
48 "Block restriction must match the subdomain names passed in the "
49 "subdomain_to_prop_value parameter");
50 _prop[_qp] = it->second;
51
52 return;
53 }
54
55 if (_current_elem)
56 {
57 _face_info = _mesh.faceInfo(_current_elem, _current_side);
58 if (!_face_info)
59 {
60 const Elem * const neighbor = _current_elem->neighbor_ptr(_current_side);
61 mooseAssert(neighbor, "Should be non-null");
62 const auto neighbor_side = neighbor->which_neighbor_am_i(_current_elem);
63 _face_info = _mesh.faceInfo(neighbor, neighbor_side);
64 mooseAssert(_face_info, "We need to have retrieved something.");
65 }
66 mooseAssert(_current_elem->build_side_ptr(_current_side)->vertex_average() ==
67 _face_info->faceCentroid(),
68 "Making sure we're in the right place");
69 auto it = _sub_id_to_prop.find(_current_elem->subdomain_id());
70 if (it == _sub_id_to_prop.end())
71 {
72 // We may be a ghosted material
73 const bool current_elem_is_fi_elem = _current_elem == &_face_info->elem();
74 const Elem * const other_elem_to_try =
75 current_elem_is_fi_elem ? _face_info->neighborPtr() : &_face_info->elem();
76 mooseAssert(other_elem_to_try, "This should be non-null");
77 it = _sub_id_to_prop.find(other_elem_to_try->subdomain_id());
78 mooseAssert(it != _sub_id_to_prop.end(),
79 "Block restriction must match the subdomain names passed in the "
80 "subdomain_to_prop_value parameter");
81 }
82 _prop[_qp] = it->second;
83 return;
84 }
85
86 mooseAssert(
87 _face_info,
88 "We must have set a face info object in order for the PiecewiseConstantByBlockMaterial "
89 "class to work on faces");
90
91 // We must be off the domain
92 auto it = _sub_id_to_prop.find(_face_info->elem().subdomain_id());
93 mooseAssert(it != _sub_id_to_prop.end(),
94 "Block restriction must match the subdomain names passed in the "
95 "subdomain_to_prop_value parameter");
96 _prop[_qp] = it->second;
97}
