85template <typename T, typename P>
86T
87PiecewiseLinearFromVectorPostprocessor::valueInternal(const T & t, const P & p) const
88{
89 if (_argument_column.empty())
90 return 0.0;
91
92 const std::tuple<Real, unsigned int, unsigned int> now = {
93 MetaPhysicL::raw_value(t),
94 _fe_problem.nNonlinearIterations(/*nl_sys=*/0),
95 _fe_problem.nLinearIterations(/*nl_sys=*/0)};
96
97 if (now != _last_update)
98 {
99 _linear_interp->setData(_argument_column, _value_column);
100 _last_update = now;
101 }
102
103 const T x = _component == 3 ? t : p(_component);
104 return _linear_interp->sample(x);
105}
