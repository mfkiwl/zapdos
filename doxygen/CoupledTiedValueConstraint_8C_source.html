44Real
45CoupledTiedValueConstraint::computeQpResidual(Moose::ConstraintType type)
46{
47 Real scaling_factor = _var.scalingFactor();
48 Real secondary_resid = 0;
49 Real retVal = 0;
50 switch (type)
51 {
52 case Moose::Secondary:
53 retVal = (_u_secondary[_qp] - _u_primary[_qp]) * _test_secondary[_i][_qp] * _scaling;
54 break;
55 case Moose::Primary:
56 secondary_resid =
57 _residual_copy(_current_node->dof_number(0, _var.number(), 0)) / scaling_factor;
58 retVal = secondary_resid * _test_primary[_i][_qp];
59 break;
60 default:
61 break;
62 }
63 return retVal;
64}

66Real
67CoupledTiedValueConstraint::computeQpJacobian(Moose::ConstraintJacobianType type)
68{
69 Real scaling_factor = _var.scalingFactor();
70 Real secondary_jac = 0;
71 Real retVal = 0;
72 switch (type)
73 {
74 case Moose::SecondarySecondary:
75 retVal = _phi_secondary[_j][_qp] * _test_secondary[_i][_qp] * _scaling;
76 break;
77 case Moose::SecondaryPrimary:
78 retVal = 0;
79 break;
80 case Moose::PrimarySecondary:
81 secondary_jac =
82 (*_jacobian)(_current_node->dof_number(0, _var.number(), 0), _connected_dof_indices[_j]);
83 retVal = secondary_jac * _test_primary[_i][_qp] / scaling_factor;
84 break;
85 case Moose::PrimaryPrimary:
86 retVal = 0;
87 break;
88 default:
89 mooseError("Unsupported type");
90 break;
91 }
92 return retVal;
93}

96CoupledTiedValueConstraint::computeQpOffDiagJacobian(Moose::ConstraintJacobianType type,
97 unsigned int jvar)
98{
99 Real retVal = 0;
100
101 if (jvar == _primary_var_num)
102 {
103 switch (type)
104 {
105 case Moose::SecondarySecondary:
106 retVal = 0;
107 break;
108 case Moose::SecondaryPrimary:
109 retVal = -_phi_primary[_j][_qp] * _test_secondary[_i][_qp] * _scaling;
110 break;
111 case Moose::PrimarySecondary:
112 retVal = 0;
113 break;
114 case Moose::PrimaryPrimary:
115 retVal = 0;
116 break;
117 default:
118 mooseError("Unsupported type");
119 break;
120 }
121 }
122
123 return retVal;
124}
