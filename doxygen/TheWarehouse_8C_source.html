56class VecStore : public Storage
57{
58public:
59 virtual void add(std::size_t obj_id, std::vector<std::unique_ptr<Attribute>> attribs) override
60 {
61 std::lock_guard<std::mutex> l(_mutex);
62 if (obj_id != _data.size())
63 throw std::runtime_error("object with id " + std::to_string(obj_id) + " already added");
64 _data.push_back(std::move(attribs));
65 }
66
67 virtual std::vector<std::size_t>
68 query(const std::vector<std::unique_ptr<Attribute>> & conds) override
69 {
70 std::vector<std::size_t> ids;
71 std::lock_guard<std::mutex> l(_mutex);
72 for (std::size_t i = 0; i < _data.size(); i++)
73 {
74 auto & data = _data[i];
75 bool ismatch = true;
76 for (auto & cond : conds)
77 {
78 if (!data[cond->id()]->isMatch(*cond))
79 {
80 ismatch = false;
81 break;
82 }
83 }
84 if (ismatch)
85 ids.push_back(i);
86 }
87 return ids;
88 }
89
90 virtual void set(std::size_t obj_id, std::vector<std::unique_ptr<Attribute>> attribs) override
91 {
92 if (obj_id > _data.size())
93 throw std::runtime_error("unknown object id " + std::to_string(obj_id));
94
95 std::lock_guard<std::mutex> l(_mutex);
96
97 auto & dst = _data[obj_id];
98 for (auto & attrib : attribs)
99 dst[attrib->id()] = std::move(attrib);
100 }
101
102private:
103 std::mutex _mutex;
104 std::vector<std::vector<std::unique_ptr<Attribute>>> _data;
105};

159int
160TheWarehouse::prepare(std::vector<std::unique_ptr<Attribute>> conds)
161{
162 bool sort = false;
163 std::unique_ptr<Attribute> sorted_attrib;
164 if (!conds.empty() && dynamic_cast<AttribSorted *>(conds.back().get()))
165 {
166 sorted_attrib = std::move(conds.back());
167 static const AttribSorted sorted_attrib_true(*this, true);
168 sort = sorted_attrib->isMatch(sorted_attrib_true);
169 // Remove the sorted condition temporarily
170 conds.pop_back();
171 }
172
173#ifdef DEBUG
174 for (auto & cond : conds)
175 mooseAssert(!dynamic_cast<AttribSorted *>(cond.get()),
176 "There should be no sorted attributes in this container.");
177#endif
178
179 auto obj_ids = _store->query(conds);
180 if (sorted_attrib)
181 conds.push_back(std::move(sorted_attrib));
182
183 std::lock_guard<std::mutex> lock(_obj_cache_mutex);
184 _obj_cache.push_back({});
185 auto query_id = _obj_cache.size() - 1;
186 auto & vec = _obj_cache.back();
187 {
188 std::lock_guard<std::mutex> lock(_query_cache_mutex);
189 _query_cache[std::move(conds)] = query_id;
190 }
191
192 std::lock_guard<std::mutex> o_lock(_obj_mutex);
193 for (auto & id : obj_ids)
194 vec.push_back(_objects[id].get());
195
196 if (sort && !vec.empty() && dynamic_cast<DependencyResolverInterface *>(vec[0]))
197 {
198 std::vector<DependencyResolverInterface *> dependers;
199 for (auto obj : vec)
200 {
201 auto d = dynamic_cast<DependencyResolverInterface *>(obj);
202 if (!d)
203 {
204 dependers.clear();
205 break;
206 }
207 dependers.push_back(d);
208 }
209
210 try
211 {
212 DependencyResolverInterface::sort(dependers);
213 }
214 catch (CyclicDependencyException<DependencyResolverInterface *> & e)
215 {
216 DependencyResolverInterface::cyclicDependencyError<UserObject *>(
217 e, "Cyclic dependency detected in object ordering");
218 }
219
220 for (unsigned int i = 0; i < dependers.size(); i++)
221 vec[i] = dynamic_cast<MooseObject *>(dependers[i]);
222 }
223
224 return query_id;
225}

235std::size_t
236TheWarehouse::queryID(const std::vector<std::unique_ptr<Attribute>> & conds)
237{
238 {
239 std::lock_guard<std::mutex> lock(_query_cache_mutex);
240 auto it = _query_cache.find(conds);
241 if (it != _query_cache.end())
242 return it->second;
243 }
244
245 std::vector<std::unique_ptr<Attribute>> conds_clone;
246 conds_clone.resize(conds.size());
247 for (std::size_t i = 0; i < conds.size(); i++)
248 conds_clone[i] = conds[i]->clone();
249 return prepare(std::move(conds_clone));
250}

252std::size_t
253TheWarehouse::count(const std::vector<std::unique_ptr<Attribute>> & conds)
254{
255 auto query_id = queryID(conds);
256 std::lock_guard<std::mutex> lock(_obj_cache_mutex);
257 auto & objs = query(query_id);
258 std::size_t count = 0;
259 for (auto obj : objs)
260 if (obj->enabled())
261 count++;

276void
277isValid(MooseObject * obj)
278{
279 auto blk = dynamic_cast<BlockRestrictable *>(obj);
280 if (!blk)
281 return;
282
283 // Check variables
284 auto c_ptr = dynamic_cast<Coupleable *>(obj);
285 if (c_ptr)
286 for (MooseVariableFEBase * var : c_ptr->getCoupledMooseVars())
287 blk->checkVariable(*var);
288
289 const InputParameters & parameters = obj->parameters();
290
291 SubProblem & problem = *parameters.get<SubProblem *>("_subproblem");
292
293 THREAD_ID tid = parameters.get<THREAD_ID>("_tid");
294
295 if (parameters.isParamValid("variable"))
296 {
297 // Try the scalar version first
298 std::string variable_name = parameters.getMooseType("variable");
299 if (variable_name == "")
300 // When using vector variables, we are only going to use the first one in the list at the
301 // interface level...
302 variable_name = parameters.getVecMooseType("variable")[0];
303
304 blk->checkVariable(problem.getVariable(
305 tid, variable_name, Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY));
306 }
