34DriftDiffusionDoNothingBC::DriftDiffusionDoNothingBC(const InputParameters & parameters)
35 : ADIntegratedBC(parameters),
36
37 _r_units(1. / getParam<Real>("position_units")),
38
39 _mu(getParam<bool>("use_material_props") ? getADMaterialProperty<Real>("mu" + _var.name())
40 : _user_mu),
41 _sign(getParam<bool>("use_material_props") ? getMaterialProperty<Real>("sgn" + _var.name())
42 : _user_sign),
43 _diffusivity(getParam<bool>("use_material_props")
44 ? getADMaterialProperty<Real>("diff" + _var.name())
45 : _user_diff),
46
47 // Coupled variables
48 _grad_potential(isCoupled("potential") ? adCoupledGradient("potential") : _minus_e_field)
49{
50 if (!(isCoupled("potential") || parameters.isParamSetByUser("EField")))
51 mooseError("You must either couple in a potential variable or set an EField.");
52
53 auto max_qps = _fe_problem.getMaxQps();
54 _user_diff.resize(max_qps);
55 _user_mu.resize(max_qps);
56 _user_sign.resize(max_qps);
57 if (!(isCoupled("potential")))
58 _minus_e_field.resize(max_qps);
59 for (decltype(max_qps) qp = 0; qp < max_qps; ++qp)
60 {
61 _user_diff[qp] = getParam<Real>("diff");
62 _user_mu[qp] = getParam<Real>("mu");
63 _user_sign[qp] = getParam<Real>("sign");
64 if (!(isCoupled("potential")))
65 _minus_e_field[qp] = RealGradient(-getParam<Real>("EField"));
66 }
67}

69ADReal
70DriftDiffusionDoNothingBC::computeQpResidual()
71{
72 return _mu[_qp] * _sign[_qp] * std::exp(_u[_qp]) * -_grad_potential[_qp] * _r_units *
73 _normals[_qp] * _test[_i][_qp] * _r_units -
74 _diffusivity[_qp] * std::exp(_u[_qp]) * _grad_u[_qp] * _r_units * _normals[_qp] *
75 _test[_i][_qp] * _r_units;
76}
