57
58 // Group for converting outputs
59 params.addParamNamesToGroup("postprocessors_as_reporters vectorpostprocessors_as_reporters",
60 "Conversions before output");

64 true,
65 "Enable/disable the output of postprocessors",
66 "'execute_postprocessors_on' has replaced this parameter");

69 "Enable/disable the output of vector postprocessors",

152void
153AdvancedOutput::init()
154{
155 // Initialize the execution flags
156 for (auto & [name, input] : _advanced_execute_on)
157 initExecutionTypes(name, input);
158
159 // Clear existing execute information lists
160 _execute_data.reset();
161
162 // Initialize the available output
163 initAvailableLists();
164
165 // Separate the hide/show list into components
166 initShowHideLists(getParam<std::vector<VariableName>>("show"),
167 getParam<std::vector<VariableName>>("hide"));
168
169 // If 'elemental_as_nodal = true' the elemental variable names must be appended to the
170 // nodal variable names. Thus, when libMesh::EquationSystem::build_solution_vector is called
171 // it will create the correct nodal variable from the elemental
172 if (_elemental_as_nodal)
173 {
174 OutputData & nodal = _execute_data["nodal"];
175 OutputData & elemental = _execute_data["elemental"];
176 nodal.show.insert(elemental.show.begin(), elemental.show.end());
177 nodal.hide.insert(elemental.hide.begin(), elemental.hide.end());
178 nodal.available.insert(elemental.available.begin(), elemental.available.end());
179 }
180
181 // Similarly as above, if 'scalar_as_nodal = true' append the elemental variable lists
182 if (_scalar_as_nodal)
183 {
184 OutputData & nodal = _execute_data["nodal"];
185 OutputData & scalar = _execute_data["scalars"];
186 nodal.show.insert(scalar.show.begin(), scalar.show.end());
187 nodal.hide.insert(scalar.hide.begin(), scalar.hide.end());
188 nodal.available.insert(scalar.available.begin(), scalar.available.end());
189 }
190
191 // Initialize the show/hide/output lists for each of the types of output
192 for (auto & it : _execute_data)
193 initOutputList(it.second);
194}

273void
274AdvancedOutput::output()
275{
276 const auto & type = _current_execute_flag;
277
278 // (re)initialize the list of available items for output
279 init();
280
281 // Call the various output types, if data exists

283 {

287
288 if (wantOutput("elemental", type))
289 {
290 outputElementalVariables();
291 _last_execute_time["elemental"] = _time;
292 }
293
294 if (wantOutput("postprocessors", type))
295 {
296 outputPostprocessors();
297 _last_execute_time["postprocessors"] = _time;
298 }
299
300 if (wantOutput("vector_postprocessors", type))
301 {
302 outputVectorPostprocessors();
303 _last_execute_time["vector_postprocessors"] = _time;
304 }
305
306 if (wantOutput("scalars", type))
307 {
308 outputScalarVariables();
309 _last_execute_time["scalars"] = _time;
310 }
311
312 if (wantOutput("system_information", type))
313 {
314 outputSystemInformation();
315 _last_execute_time["system_information"] = _time;
316 }
317
318 if (wantOutput("input", type))
319 {
320 outputInput();
321 _last_execute_time["input"] = _time;
322 }
323
324 if (wantOutput("reporters", type))
325 {
326 outputReporters();
327 _last_execute_time["reporters"] = _time;

331bool
332AdvancedOutput::wantOutput(const std::string & name, const ExecFlagType & type)
333{
334 // Ignore EXEC_FORCED for system information and input, there is no reason to force this
335 if (type == EXEC_FORCED && (name == "system_information" || name == "input"))
336 return false;
337
338 // Do not output if the 'none' is contained by the execute_on
339 if (_advanced_execute_on.contains(name) && _advanced_execute_on[name].contains("none"))
340 return false;
341
342 // Data output flag, true if data exists to be output
343 bool execute_data_flag = true;
344
345 // Set flag to false, if the OutputData exists and the output variable list is empty
346 std::map<std::string, OutputData>::const_iterator iter = _execute_data.find(name);
347 if (iter != _execute_data.end() && iter->second.output.empty())
348 execute_data_flag = false;
349
350 // Set flag to false, if the OutputOnWarehouse DOES NOT contain an entry
351 if (!_advanced_execute_on.contains(name))
352 execute_data_flag = false;
353
354 // Force the output, if there is something to output and the time has not been output
355 if (type == EXEC_FORCED && execute_data_flag && _last_execute_time[name] != _time)
356 return true;
357
358 // Return true (output should occur) if three criteria are satisfied, else do not output:
359 // (1) The execute_data_flag = true (i.e, there is data to output)
360 // (2) The current output type is contained in the list of output execution types
361 // (3) The current execution time is "final" or "forced" and the data has not already been
362 // output
363 if (execute_data_flag && _advanced_execute_on[name].contains(type) &&
364 !(type == EXEC_FINAL && _last_execute_time[name] == _time))
365 return true;
366 else
367 return false;
368}

370bool
371AdvancedOutput::hasOutput(const ExecFlagType & type)
372{
373 // If any of the component outputs are true, then there is some output to perform
374 for (const auto & it : _advanced_execute_on)
375 if (wantOutput(it.first, type))
376 return true;
377
378 // There is nothing to output
379 return false;
380}

382bool
383AdvancedOutput::hasOutput()
384{
385 // Test that variables exist for output AND that output execution flags are valid
386 for (const auto & it : _execute_data)
387 if (!(it.second).output.empty() && _advanced_execute_on.contains(it.first) &&
388 _advanced_execute_on[it.first].isValid())
389 return true;
390
391 // Test execution flags for non-variable output
392 if (_advanced_execute_on.contains("system_information") &&
393 _advanced_execute_on["system_information"].isValid())
394 return true;
395 if (_advanced_execute_on.contains("input") && _advanced_execute_on["input"].isValid())
396 return true;
397
398 return false;
399}

401void
402AdvancedOutput::initAvailableLists()
403{
404 // Initialize Postprocessor list
405 // This flag is set to true if any postprocessor has the 'outputs' parameter set, it is then used
406 // to produce an warning if postprocessor output is disabled
407 if (!_postprocessors_as_reporters)
408 initPostprocessorOrVectorPostprocessorLists<Postprocessor>("postprocessors");
409
410 // Initialize vector postprocessor list
411 // This flag is set to true if any vector postprocessor has the 'outputs' parameter set, it is
412 // then used
413 // to produce an warning if vector postprocessor output is disabled
414 if (!_vectorpostprocessors_as_reporters)
415 initPostprocessorOrVectorPostprocessorLists<VectorPostprocessor>("vector_postprocessors");
416
417 // Get a list of the available variables
418 std::vector<VariableName> variables = _problem_ptr->getVariableNames();
419
420 // Loop through the variables and store the names in the correct available lists
421 for (const auto & var_name : variables)
422 {
423 if (_problem_ptr->hasVariable(var_name))
424 {
425 MooseVariableFEBase & var = _problem_ptr->getVariable(
426 0, var_name, Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY);
427 const FEType type = var.feType();
428 for (unsigned int i = 0; i < var.count(); ++i)
429 {
430 VariableName vname = var_name;
431 if (var.isArray())
432 vname = SubProblem::arrayVariableComponent(var_name, i);
433
434 // A note that if we have p-refinement we assume "worst-case" scenario that our constant
435 // monomial/monomial-vec families have been refined and we can no longer write them as
436 // elemental
437 if (type.order == CONSTANT && !_problem_ptr->havePRefinement() &&
438 type.family != MONOMIAL_VEC)
439 _execute_data["elemental"].available.insert(vname);
440 else if (type.family == NEDELEC_ONE || type.family == LAGRANGE_VEC ||
441 type.family == MONOMIAL_VEC || type.family == RAVIART_THOMAS)
442 {
443 const auto geom_type = ((type.family == MONOMIAL_VEC) && (type.order == CONSTANT) &&
444 !_problem_ptr->havePRefinement())
445 ? "elemental"
446 : "nodal";
447 switch (_es_ptr->get_mesh().spatial_dimension())
448 {
449 case 0:
450 case 1:
451 _execute_data[geom_type].available.insert(vname);
452 break;
453 case 2:
454 _execute_data[geom_type].available.insert(vname + "_x");
455 _execute_data[geom_type].available.insert(vname + "_y");
456 break;
457 case 3:
458 _execute_data[geom_type].available.insert(vname + "_x");
459 _execute_data[geom_type].available.insert(vname + "_y");
460 _execute_data[geom_type].available.insert(vname + "_z");
461 break;
462 }
463 }
464 else
465 _execute_data["nodal"].available.insert(vname);
466 }
467 }
468
469 else if (_problem_ptr->hasScalarVariable(var_name))
470 _execute_data["scalars"].available.insert(var_name);
471 }
472
473 // Initialize Reporter name list
474 for (auto && r_name : _reporter_data.getReporterNames())
475 if ((_postprocessors_as_reporters || !r_name.isPostprocessor()) &&
476 (_vectorpostprocessors_as_reporters || !r_name.isVectorPostprocessor()))
477 _execute_data["reporters"].available.insert(r_name);
478}

500void
501AdvancedOutput::initShowHideLists(const std::vector<VariableName> & show,
502 const std::vector<VariableName> & hide)
503{
504
505 // Storage for user-supplied input that is unknown as a variable or postprocessor
506 std::set<std::string> unknown;
507
508 // If a show hide/list exists, let the data warehouse know about it. This allows for the proper
509 // handling of output lists (see initOutputList)
510 if (show.size() > 0)
511 _execute_data.setHasShowList(true);
512
513 // Populate the show lists
514 for (const auto & var_name : show)
515 {
516 if (_problem_ptr->hasVariable(var_name))
517 {
518 MooseVariableFEBase & var = _problem_ptr->getVariable(
519 0, var_name, Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY);
520 const FEType type = var.feType();
521 for (unsigned int i = 0; i < var.count(); ++i)
522 {
523 VariableName vname = var_name;
524 if (var.isArray())
525 vname = SubProblem::arrayVariableComponent(var_name, i);
526
527 if (type.order == CONSTANT)
528 _execute_data["elemental"].show.insert(vname);
529 else if (type.family == NEDELEC_ONE || type.family == LAGRANGE_VEC ||
530 type.family == MONOMIAL_VEC || type.family == RAVIART_THOMAS)
531 {
532 const auto geom_type =
533 ((type.family == MONOMIAL_VEC) && (type.order == CONSTANT)) ? "elemental" : "nodal";
534 switch (_es_ptr->get_mesh().spatial_dimension())
535 {
536 case 0:
537 case 1:
538 _execute_data[geom_type].show.insert(vname);
539 break;
540 case 2:
541 _execute_data[geom_type].show.insert(vname + "_x");
542 _execute_data[geom_type].show.insert(vname + "_y");
543 break;
544 case 3:
545 _execute_data[geom_type].show.insert(vname + "_x");
546 _execute_data[geom_type].show.insert(vname + "_y");
547 _execute_data[geom_type].show.insert(vname + "_z");
548 break;
549 }
550 }
551 else
552 _execute_data["nodal"].show.insert(vname);
553 }
554 }
555 else if (_problem_ptr->hasScalarVariable(var_name))
556 _execute_data["scalars"].show.insert(var_name);
557 else if (hasPostprocessorByName(var_name))
558 _execute_data["postprocessors"].show.insert(var_name);
559 else if (hasVectorPostprocessorByName(var_name))
560 _execute_data["vector_postprocessors"].show.insert(var_name);
561 else if ((var_name.find("/") != std::string::npos) &&
562 (hasReporterValueByName(ReporterName(var_name))))
563 _execute_data["reporters"].show.insert(var_name);
564 else
565 unknown.insert(var_name);
566 }
567
568 // Populate the hide lists
569 for (const auto & var_name : hide)
570 {
571 if (_problem_ptr->hasVariable(var_name))
572 {
573 MooseVariableFEBase & var = _problem_ptr->getVariable(
574 0, var_name, Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY);
575 const FEType type = var.feType();
576 for (unsigned int i = 0; i < var.count(); ++i)
577 {
578 VariableName vname = var_name;
579 if (var.isArray())
580 vname = SubProblem::arrayVariableComponent(var_name, i);
581
582 if (type.order == CONSTANT)
583 _execute_data["elemental"].hide.insert(vname);
584 else if (type.family == NEDELEC_ONE || type.family == LAGRANGE_VEC ||
585 type.family == MONOMIAL_VEC || type.family == RAVIART_THOMAS)
586 {
587 switch (_es_ptr->get_mesh().spatial_dimension())
588 {
589 case 0:
590 case 1:
591 _execute_data["nodal"].hide.insert(vname);
592 break;
593 case 2:
594 _execute_data["nodal"].hide.insert(vname + "_x");
595 _execute_data["nodal"].hide.insert(vname + "_y");
596 break;
597 case 3:
598 _execute_data["nodal"].hide.insert(vname + "_x");
599 _execute_data["nodal"].hide.insert(vname + "_y");
600 _execute_data["nodal"].hide.insert(vname + "_z");
601 break;
602 }
603 }
604 else
605 _execute_data["nodal"].hide.insert(vname);
606 }
607 }
608 else if (_problem_ptr->hasScalarVariable(var_name))
609 _execute_data["scalars"].hide.insert(var_name);
610 else if (hasPostprocessorByName(var_name))
611 _execute_data["postprocessors"].hide.insert(var_name);
612 else if (hasVectorPostprocessorByName(var_name))
613 _execute_data["vector_postprocessors"].hide.insert(var_name);
614 else if ((var_name.find("/") != std::string::npos) &&
615 (hasReporterValueByName(ReporterName(var_name))))
616 _execute_data["reporters"].hide.insert(var_name);
617
618 else
619 unknown.insert(var_name);
620 }
621
622 // Error if an unknown variable or postprocessor is found
623 if (!unknown.empty())
624 {
625 std::ostringstream oss;
626 oss << "Output(s) do not exist (must be variable, scalar, postprocessor, or vector "
627 "postprocessor): ";
628 std::copy(unknown.begin(), unknown.end(), infix_ostream_iterator<std::string>(oss, " "));
629 mooseError(oss.str());
630 }
631}

633void
634AdvancedOutput::initOutputList(OutputData & data)
635{
636 // References to the vectors of variable names
637 std::set<std::string> & hide = data.hide;
638 std::set<std::string> & show = data.show;
639 std::set<std::string> & avail = data.available;
640 std::set<std::string> & output = data.output;
641
642 // Append to the hide list from OutputInterface objects
643 std::set<std::string> interface_hide;
644 _app.getOutputWarehouse().buildInterfaceHideVariables(name(), interface_hide);
645 hide.insert(interface_hide.begin(), interface_hide.end());
646
647 // Both show and hide are empty and no show/hide settings were provided (show all available)
648 if (show.empty() && hide.empty() && !_execute_data.hasShowList())
649 output = avail;
650
651 // Only hide is empty (show all the variables listed)
652 else if (!show.empty() && hide.empty())
653 output = show;
654
655 // Only show is empty (show all except those hidden)
656 else if (show.empty() && !hide.empty())
657 std::set_difference(avail.begin(),
658 avail.end(),
659 hide.begin(),
660 hide.end(),
661 std::inserter(output, output.begin()));
662
663 // Both hide and show are present (show all those listed)
664 else
665 {
666 // Check if variables are in both, which is invalid
667 std::vector<std::string> tmp;
668 std::set_intersection(
669 hide.begin(), hide.end(), show.begin(), show.end(), std::inserter(tmp, tmp.begin()));
670 if (!tmp.empty())
671 {
672 std::ostringstream oss;
673 oss << "Output(s) specified to be both shown and hidden: ";
674 std::copy(tmp.begin(), tmp.end(), infix_ostream_iterator<std::string>(oss, " "));
675 mooseError(oss.str());
676 }
677
678 // Define the output variable list
679 output = show;
680 }
681}

683void
684AdvancedOutput::addValidParams(InputParameters & params, const MultiMooseEnum & types)
685{
686 ExecFlagEnum empty_execute_on = MooseUtils::getDefaultExecFlagEnum();
687 empty_execute_on.addAvailableFlags(EXEC_FAILED);
688
689 // Nodal output
690 if (types.contains("nodal"))
691 {
692 params.addParam<ExecFlagEnum>(
693 "execute_nodal_on", empty_execute_on, "Control the output of nodal variables");
694 params.addParamNamesToGroup("execute_nodal_on", "Selection/restriction of output");
695 }
696
697 // Elemental output
698 if (types.contains("elemental"))
699 {
700 params.addParam<ExecFlagEnum>(
701 "execute_elemental_on", empty_execute_on, "Control the output of elemental variables");
702 params.addParamNamesToGroup("execute_elemental_on", "Selection/restriction of output");
703
704 // Add material output control, which are output via elemental variables
705 params.addParam<bool>("output_material_properties",
706 false,
707 "Flag indicating if material properties should be output");
708 params.addParam<std::vector<std::string>>(
709 "show_material_properties",
710 "List of material properties that should be written to the output");
711 params.addParamNamesToGroup("output_material_properties show_material_properties", "Materials");
712
713 // Add mesh extra element id control, which are output via elemental variables
714 params.addParam<bool>(
715 "output_extra_element_ids",
