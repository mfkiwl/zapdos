45Real
46DiffusionFluxAux::computeValue()
47{
48 const Real gradient = _use_normal ? _grad_u[_qp] * _normals[_qp] : _grad_u[_qp](_component);
49 const Real diffusion_coef = _diffusion_coef ? (*_diffusion_coef)[_qp]
50 : MetaPhysicL::raw_value((*_ad_diffusion_coef)[_qp]);
51 return -diffusion_coef * gradient;
52}
