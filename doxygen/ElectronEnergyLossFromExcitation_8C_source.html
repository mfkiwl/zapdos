46ADReal
47ElectronEnergyLossFromExcitation::computeQpResidual()
48{
49 ADReal electron_flux_mag = (-_muem[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_em[_qp]) -
50 _diffem[_qp] * std::exp(_em[_qp]) * _grad_em[_qp] * _r_units)
51 .norm();
52 ADReal ex_term = _alpha_ex[_qp] * electron_flux_mag;
53
54 return -_test[_i][_qp] * ex_term * -_Eex[_qp];
55}
