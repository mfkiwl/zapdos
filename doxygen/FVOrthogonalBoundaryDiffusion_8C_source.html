44ADReal
45FVOrthogonalBoundaryDiffusion::computeQpResidual()
46{
47 const bool elem_is_interior =
48 _face_info->faceType(std::make_pair(_var.number(), _var.sys().number())) ==
49 FaceInfo::VarFaceNeighbors::ELEM;
50
51 const auto & diff_quant = elem_is_interior ? _diff_quant_elem[_qp] : _diff_quant_neighbor[_qp];
52 const auto & coeff = elem_is_interior ? _coeff_elem[_qp] : _coeff_neighbor[_qp];
53 const auto & interior_centroid =
54 elem_is_interior ? _face_info->elemCentroid() : _face_info->neighborCentroid();
55
56 return -coeff * (_function.value(_t, _face_info->faceCentroid()) - diff_quant) /
57 (_face_info->faceCentroid() - interior_centroid).norm();
58}
