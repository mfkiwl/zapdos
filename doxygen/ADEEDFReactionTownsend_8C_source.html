60ADReal
61ADEEDFReactionTownsendLog::computeQpResidual()
62{
63 return -_test[_i][_qp] * _alpha[_qp] * std::exp(_target[_qp]) *
64 (std::exp(_em[_qp]) * (-_muem[_qp] * -_grad_potential[_qp] * _r_units -
65 _diffem[_qp] * _grad_em[_qp] * _r_units))
66 .norm() *
67 _coefficient;
68}
