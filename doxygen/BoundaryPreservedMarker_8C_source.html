31BoundaryPreservedMarker::BoundaryPreservedMarker(const InputParameters & parameters)
32 : Marker(parameters),
33 _marker_name(parameters.get<MarkerName>("marker")),
34 _marker(&getMarkerValue(_marker_name))
35{
36 _mesh.errorIfDistributedMesh(type());
37 BoundaryName boundary_name = getParam<BoundaryName>("preserved_boundary");
38 auto boundary_ids = MooseMeshUtils::getBoundaryIDs(_mesh, {boundary_name}, true);
39 mooseAssert(boundary_ids.size() == 1, "Boundary does not exist");
40 _preserved_boundary = boundary_ids[0];
41}

43bool
44BoundaryPreservedMarker::preserveBoundary(const Elem * const & current_elem)
45{
46 auto & elem_side_bnd_ids = _mesh.getMesh().get_boundary_info().get_sideset_map();
47
48 // Do not coarsen the elements when they are connected to the preserved boundary
49 for (const auto & pr : as_range(elem_side_bnd_ids.equal_range(current_elem)))
50 if (pr.second.second == _preserved_boundary)
51 return true;
52
53 return false;
54}

56Marker::MarkerValue
57BoundaryPreservedMarker::computeElementMarker()
58{
59 MarkerValue marker_value = static_cast<MarkerValue>((*_marker)[0]);
60 if (marker_value == COARSEN && preserveBoundary(_current_elem))
61 return DO_NOTHING;
62 else
63 return marker_value;
64}
