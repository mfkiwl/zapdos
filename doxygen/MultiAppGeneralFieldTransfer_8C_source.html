163 _from_mesh_division_behavior(getParam<MooseEnum>("from_mesh_division_usage")),
164 _to_mesh_division_behavior(getParam<MooseEnum>("to_mesh_division_usage")),

178 paramError("variable", "The number of variables to transfer to and from should be equal");

182 paramError("source_variable_components",

207 if (!isParamSetByUser("bbox_factor") && !isParamSetByUser("fixed_bounding_box_size"))
208 mooseWarning(
209 "Extrapolation (nearest-source options, outside-app source) parameters have "
210 "been passed, but no subapp bounding box expansion parameters have been passed.");

214 paramError("use_bounding_boxes",
215 "Cannot pass additional bounding box parameters (sizes, expansion, etc) if we are "
216 "not using bounding boxes");

219void
220MultiAppGeneralFieldTransfer::initialSetup()
221{
222 MultiAppConservativeTransfer::initialSetup();
223
224 // Use IDs for block and boundary restriction
225 // Loop over all source problems
226 for (const auto i_from : index_range(_from_problems))
227 {
228 const auto & from_moose_mesh = _from_problems[i_from]->mesh(_displaced_source_mesh);
229 if (isParamValid("from_blocks"))
230 {
231 auto & blocks = getParam<std::vector<SubdomainName>>("from_blocks");
232 std::vector<SubdomainID> ids = from_moose_mesh.getSubdomainIDs(blocks);
233 _from_blocks.insert(ids.begin(), ids.end());
234 if (_from_blocks.size() != blocks.size())
235 paramError("from_blocks", "Some blocks were not found in the mesh");
236 }
237
238 if (isParamValid("from_boundaries"))
239 {
240 auto & boundary_names = getParam<std::vector<BoundaryName>>("from_boundaries");
241 std::vector<BoundaryID> boundary_ids = from_moose_mesh.getBoundaryIDs(boundary_names);
242 _from_boundaries.insert(boundary_ids.begin(), boundary_ids.end());
243 if (_from_boundaries.size() != boundary_names.size())
244 paramError("from_boundaries", "Some boundaries were not found in the mesh");
245 }
246
247 if (isParamValid("from_mesh_division"))
248 {
249 const auto & mesh_div_name = getParam<MeshDivisionName>("from_mesh_division");
250 _from_mesh_divisions.push_back(&_from_problems[i_from]->getMeshDivision(mesh_div_name));
251 // Check that the behavior set makes sense
252 if (_from_mesh_division_behavior == MeshDivisionTransferUse::RESTRICTION)
253 {
254 if (_from_mesh_divisions[i_from]->coversEntireMesh())
255 mooseInfo("'from_mesh_division_usage' is set to use a spatial restriction but the "
256 "'from_mesh_division' for source app of global index " +
257 std::to_string(getGlobalSourceAppIndex(i_from)) +
258 " covers the entire mesh. Do not expect any restriction from a mesh "
259 "division that covers the entire mesh");
260 }
261 else if (_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX &&
262 !isParamValid("to_mesh_division"))
263 paramError("to_mesh_division_usage",
264 "Source mesh division cannot match target mesh division if no target mesh "
265 "division is specified");
266 else if (_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX)
267 {
268 if (!hasToMultiApp())
269 paramError("from_mesh_division_usage",
270 "Cannot match source mesh division index to target subapp index if there is "
271 "only one target: the parent app (not a subapp)");
272 else if (getToMultiApp()->numGlobalApps() !=
273 _from_mesh_divisions[i_from]->getNumDivisions())
274 mooseWarning("Attempting to match target subapp index with the number of source mesh "
275 "divisions, which is " +
276 std::to_string(_from_mesh_divisions[i_from]->getNumDivisions()) +
277 " while there are " + std::to_string(getToMultiApp()->numGlobalApps()) +
278 " target subapps");
279 if (_to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX)
280 // We do not support it because it would require sending the point + target app index +
281 // target app division index, and we only send the Point + one number
282 paramError("from_mesh_division_usage",
283 "We do not support using target subapp index for source division behavior and "
284 "matching the division index for the target mesh division behavior.");
285 }
286 else if (_from_mesh_division_behavior == "none")
287 paramError("from_mesh_division_usage", "User must specify a 'from_mesh_division_usage'");
288 }
289 else if (_from_mesh_division_behavior != "none")
290 paramError("from_mesh_division",
291 "'from_mesh_division' must be specified if the usage method is specified");
292 }
293
294 // Loop over all target problems
295 for (const auto i_to : index_range(_to_problems))
296 {
297 const auto & to_moose_mesh = _to_problems[i_to]->mesh(_displaced_target_mesh);
298 if (isParamValid("to_blocks"))
299 {
300 auto & blocks = getParam<std::vector<SubdomainName>>("to_blocks");
301 std::vector<SubdomainID> ids = to_moose_mesh.getSubdomainIDs(blocks);
302 _to_blocks.insert(ids.begin(), ids.end());
303 if (_to_blocks.size() != blocks.size())
304 paramError("to_blocks", "Some blocks were not found in the mesh");
305 }
306
307 if (isParamValid("to_boundaries"))
308 {
309 auto & boundary_names = getParam<std::vector<BoundaryName>>("to_boundaries");
310 std::vector<BoundaryID> boundary_ids = to_moose_mesh.getBoundaryIDs(boundary_names);
311 _to_boundaries.insert(boundary_ids.begin(), boundary_ids.end());
312 if (_to_boundaries.size() != boundary_names.size())
313 paramError("to_boundaries", "Some boundaries were not found in the mesh");
314 }
315
316 if (isParamValid("to_mesh_division"))
317 {
318 const auto & mesh_div_name = getParam<MeshDivisionName>("to_mesh_division");
319 _to_mesh_divisions.push_back(&_to_problems[i_to]->getMeshDivision(mesh_div_name));
320 // Check that the behavior set makes sense
321 if (_to_mesh_division_behavior == MeshDivisionTransferUse::RESTRICTION)
322 {
323 if (_to_mesh_divisions[i_to]->coversEntireMesh())
324 mooseInfo("'to_mesh_division_usage' is set to use a spatial restriction but the "
325 "'to_mesh_division' for target application of global index " +
326 std::to_string(getGlobalSourceAppIndex(i_to)) +
327 " covers the entire mesh. Do not expect any restriction from a mesh "
328 "division that covers the entire mesh");
329 }
330 else if (_to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX)
331 {
332 if (!isParamValid("from_mesh_division"))
333 paramError("to_mesh_division_usage",
334 "Target mesh division cannot match source mesh division if no source mesh "
335 "division is specified");
336 else if ((*_from_mesh_divisions.begin())->getNumDivisions() !=
337 _to_mesh_divisions[i_to]->getNumDivisions())
338 mooseWarning("Source and target mesh divisions do not have the same number of bins. If "
339 "this is what you expect, please reach out to a MOOSE or app developer to "
340 "ensure appropriate use");
341 }
342 else if (_to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX)
343 {
344 if (!hasFromMultiApp())
345 paramError(
346 "to_mesh_division_usage",
347 "Cannot match target mesh division index to source subapp index if there is only one "
348 "source: the parent app (not a subapp)");
349 else if (getFromMultiApp()->numGlobalApps() != _to_mesh_divisions[i_to]->getNumDivisions())
350 mooseWarning("Attempting to match source subapp index with the number of target mesh "
351 "divisions, which is " +
352 std::to_string(_to_mesh_divisions[i_to]->getNumDivisions()) +
353 " while there are " + std::to_string(getFromMultiApp()->numGlobalApps()) +
354 " source subapps");
355 if (_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX)
356 paramError(
357 "from_mesh_division_usage",
358 "We do not support using source subapp index for the target division behavior and "
359 "matching the division index for the source mesh division behavior.");
360 }
361 else if (_to_mesh_division_behavior == "none")
362 paramError("to_mesh_division_usage", "User must specify a 'to_mesh_division_usage'");
363 }
364 else if (_to_mesh_division_behavior != "none")
365 paramError("to_mesh_division",
366 "'to_mesh_division' must be specified if usage method '" +
367 Moose::stringify(_to_mesh_division_behavior) + "' is specified");
368 }
369
370 // Check if components are set correctly if using an array variable
371 for (const auto i_from : index_range(_from_problems))
372 {
373 for (const auto var_index : make_range(_from_var_names.size()))
374 {
375 MooseVariableFieldBase & from_var =
376 _from_problems[i_from]->getVariable(0,
377 _from_var_names[var_index],
378 Moose::VarKindType::VAR_ANY,
379 Moose::VarFieldType::VAR_FIELD_ANY);
380 if (from_var.count() > 1 && _from_var_components.empty())
381 paramError("source_variable_components", "Component must be passed for an array variable");
382 if (_from_var_components.size() && from_var.count() < _from_var_components[var_index])
383 paramError("source_variable_components",
384 "Component passed is larger than size of variable");
385 }
386 }
387 for (const auto i_to : index_range(_to_problems))
388 {
389 for (const auto var_index : make_range(_to_var_names.size()))
390 {
391 MooseVariableFieldBase & to_var =
392 _to_problems[i_to]->getVariable(0,
393 _to_var_names[var_index],
394 Moose::VarKindType::VAR_ANY,
395 Moose::VarFieldType::VAR_FIELD_ANY);
396 if (to_var.count() > 1 && _to_var_components.empty())
397 paramError("target_variable_components", "Component must be passed for an array variable");
398 if (_to_var_components.size() && to_var.count() < _to_var_components[var_index])
399 paramError("target_variable_components",
400 "Component passed is larger than size of variable");
401 }
402 }
403
404 // Cache some quantities to avoid having to get them on every transferred point
405 if (_to_problems.size())
406 {
407 _to_variables.resize(_to_var_names.size());
408 for (const auto i_var : index_range(_to_var_names))
409 _to_variables[i_var] = &_to_problems[0]->getVariable(
410 0, _to_var_names[i_var], Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY);
411 }

467void
468MultiAppGeneralFieldTransfer::transferVariable(unsigned int i)
469{
470 mooseAssert(i < _var_size, "The variable of index " << i << " does not exist");
471
472 // Find outgoing target points
473 // We need to know what points we need to send which processors
474 // One processor will receive many points from many processors
475 // One point may go to different processors
476 ProcessorToPointVec outgoing_points;
477 extractOutgoingPoints(i, outgoing_points);
478
479 if (_from_var_names.size())
480 prepareEvaluationOfInterpValues(i);
481 else
482 prepareEvaluationOfInterpValues(-1);
483
484 // Fill values and app ids for incoming points
485 // We are responsible to compute values for these incoming points
486 auto gather_functor =
487 [this](processor_id_type /*pid*/,
488 const std::vector<std::pair<Point, unsigned int>> & incoming_locations,
489 std::vector<std::pair<Real, Real>> & outgoing_vals)
490 {
491 outgoing_vals.resize(
492 incoming_locations.size(),
493 {GeneralFieldTransfer::BetterOutOfMeshValue, GeneralFieldTransfer::BetterOutOfMeshValue});
494 // Evaluate interpolation values for these incoming points
495 evaluateInterpValues(incoming_locations, outgoing_vals);
496 };
497
498 DofobjectToInterpValVec dofobject_to_valsvec(_to_problems.size());
499 InterpCaches interp_caches(_to_problems.size(), getMaxToProblemsBBoxDimensions());
500 InterpCaches distance_caches(_to_problems.size(), getMaxToProblemsBBoxDimensions());
501
502 // Copy data out to incoming_vals_ids
503 auto action_functor = [this, &i, &dofobject_to_valsvec, &interp_caches, &distance_caches](
504 processor_id_type pid,
505 const std::vector<std::pair<Point, unsigned int>> & my_outgoing_points,
506 const std::vector<std::pair<Real, Real>> & incoming_vals)
507 {
508 auto & pointInfoVec = _processor_to_pointInfoVec[pid];
509
510 // Cache interpolation values for each dof object / points
511 cacheIncomingInterpVals(pid,
512 i,
513 pointInfoVec,
514 my_outgoing_points,
515 incoming_vals,
516 dofobject_to_valsvec,
517 interp_caches,
518 distance_caches);
519 };
520
521 // We assume incoming_vals_ids is ordered in the same way as outgoing_points
522 // Hopefully, pull_parallel_vector_data will not mess up this
523 const std::pair<Real, Real> * ex = nullptr;
524 libMesh::Parallel::pull_parallel_vector_data(
525 comm(), outgoing_points, gather_functor, action_functor, ex);
526
527 // Check for conflicts and overlaps from the maps that were built during the transfer
528 if (_search_value_conflicts)
529 outputValueConflicts(i, dofobject_to_valsvec, distance_caches);
530
531 // Set cached values into solution vector
532 setSolutionVectorValues(i, dofobject_to_valsvec, interp_caches);
533}

535void
536MultiAppGeneralFieldTransfer::locatePointReceivers(const Point point,
537 std::set<processor_id_type> & processors)
538{
539 // Check which processors have apps that may include or be near this point
540 // A point may be close enough to several problems, hosted on several processes
541 bool found = false;
542
543 // Additional process-restriction techniques we could use (TODOs):
544 // - nearest_positions/_app could use its own heuristic
545 // - from_mesh_divisions could be polled for which divisions they possess on each
546 // process, depending on the behavior chosen. This could limit potential senders.
547 // This should be done ahead of this function call, for all points at once
548
549 // Select the method for determining the apps receiving points/sending values
550 if (_use_bounding_boxes)
551 {
552 // We examine all bounding boxes and find the maximum distance within a bounding box
553 // from the point. This creates a sphere around the point of interest. Any app with a bounding
554 // box that intersects this sphere (with a bboxMinDistance < nearest_max_distance) will be
555 // considered a potential source.
556 // NOTE: This is a heuristic. We could try others
557 // NOTE: from_bboxes are in the reference space, as is the point
558 Real nearest_max_distance = std::numeric_limits<Real>::max();
559 for (const auto & bbox : _from_bboxes)
560 {
561 Real distance = bboxMaxDistance(point, bbox);
562 if (distance < nearest_max_distance)
563 nearest_max_distance = distance;
564 }
565
566 unsigned int from0 = 0;
567 for (processor_id_type i_proc = 0; i_proc < n_processors();
568 from0 += _froms_per_proc[i_proc], ++i_proc)
569 for (unsigned int i_from = from0; i_from < from0 + _froms_per_proc[i_proc]; ++i_from)
570 {
571 Real distance = bboxMinDistance(point, _from_bboxes[i_from]);
572 // We will not break here because we want to send a point to all possible source domains
573 if (_greedy_search || distance <= nearest_max_distance ||
574 _from_bboxes[i_from].contains_point(point))
575 {
576 processors.insert(i_proc);
577 found = true;
578 }
579 }
580 }
581 // Greedy search will contact every single processor. It's not scalable, but if there's valid data
582 // on any subapp on any process, it will find it
583 else if (_greedy_search)
584 {
585 found = true;
586 for (const auto i_proc : make_range(n_processors()))
587 processors.insert(i_proc);
588 }
589 // Since we indicated that we only wanted values from a subapp with the same global index as the
590 // target mesh division, we might as well only communicate with the process that owns this app
591 else if (!_to_mesh_divisions.empty() &&
592 _to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX)
593 {
594 // The target point could have a different index in each target mesh division. So on paper, we
595 // would need to check all of them.
596 auto saved_target_div = MooseMeshDivision::INVALID_DIVISION_INDEX;
597 for (const auto i_to : index_range(_to_meshes))
598 {
599 const auto target_div = _to_mesh_divisions[i_to]->divisionIndex(
600 _to_transforms[getGlobalTargetAppIndex(i_to)]->mapBack(point));
601 // If it's the same division index, do not redo the search
602 if (target_div == saved_target_div)
603 continue;
604 else
605 saved_target_div = target_div;
606
607 // Look for the processors owning a source-app with an index equal to the target mesh division
608 for (const auto i_proc : make_range(n_processors()))
609 for (const auto i_from : make_range(_froms_per_proc[i_proc]))
610 if (target_div == _global_app_start_per_proc[i_proc] + i_from)
611 {
612 processors.insert(i_proc);
613 found = true;
614 }
615 }
616 }
617 else
618 mooseError("No algorithm were selected to find which processes may send value data "
619 "for a each target point. Please either specify using bounding boxes, "
620 "greedy search, or to_mesh_division-based parameters");
621
622 // Error out if we could not find this point when ask us to do so
623 if (!found && _error_on_miss)
624 mooseError("Cannot locate point ", point, " \n ", "mismatched meshes are used");
625}

627void
628MultiAppGeneralFieldTransfer::cacheOutgoingPointInfo(const Point point,
629 const dof_id_type dof_object_id,
630 const unsigned int problem_id,
631 ProcessorToPointVec & outgoing_points)
632{
633 std::set<processor_id_type> processors;
634 // Find which processors will receive point data so they can send back value data
635 // The list can be larger than needed, depending on the heuristic / algorithm used to make
636 // the call on whether a processor (and the apps it runs) should be involved
637 processors.clear();
638 locatePointReceivers(point, processors);
639
640 // We need to send this location data to these processors so they can send back values
641 for (const auto pid : processors)
642 {
643 // Select which from_mesh_division the source data must come from for this point
644 unsigned int required_source_division = 0;
645 if (_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX)
646 required_source_division = getGlobalTargetAppIndex(problem_id);
647 else if (_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX ||
648 _to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX ||
649 _to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX)
650 required_source_division = _to_mesh_divisions[problem_id]->divisionIndex(
651 _to_transforms[getGlobalTargetAppIndex(problem_id)]->mapBack(point));
652
653 // Skip if we already know we don't want the point
654 if (required_source_division == MooseMeshDivision::INVALID_DIVISION_INDEX)
655 continue;
656
657 // Store outgoing information for every source process
658 outgoing_points[pid].push_back(std::pair<Point, unsigned int>(point, required_source_division));
659
660 // Store point information locally for processing received data
661 // We can use these information when insert values to solution vector
662 PointInfo pointinfo;
663 pointinfo.problem_id = problem_id;
664 pointinfo.dof_object_id = dof_object_id;
665 _processor_to_pointInfoVec[pid].push_back(pointinfo);
666 }

669void
670MultiAppGeneralFieldTransfer::extractOutgoingPoints(const unsigned int var_index,
671 ProcessorToPointVec & outgoing_points)
672{
673 // Get the variable name, with the accommodation for array/vector names
674 const auto & var_name = getToVarName(var_index);
675
676 // Clean up the map from processor to pointInfo vector
677 // This map should be consistent with outgoing_points
678 _processor_to_pointInfoVec.clear();
679
680 // Loop over all problems
681 for (const auto i_to : index_range(_to_problems))
682 {
683 const auto global_i_to = getGlobalTargetAppIndex(i_to);
684
685 // libMesh EquationSystems
686 auto & es = getEquationSystem(*_to_problems[i_to], _displaced_target_mesh);
687 // libMesh system that has this variable
688 System * to_sys = find_sys(es, var_name);
689 auto sys_num = to_sys->number();
690 auto var_num = _to_variables[var_index]->number();
691 auto & fe_type = _to_variables[var_index]->feType();
692 bool is_nodal = _to_variables[var_index]->isNodal();
693
694 // Moose mesh
695 const auto & to_moose_mesh = _to_problems[i_to]->mesh(_displaced_target_mesh);
696 const auto & to_mesh = to_moose_mesh.getMesh();
697
698 // We support more general variables via libMesh GenericProjector
699 if (fe_type.order > CONSTANT && !is_nodal)
700 {
701 GeneralFieldTransfer::RecordRequests<Number> f;
702 GeneralFieldTransfer::RecordRequests<Gradient> g;
703 GeneralFieldTransfer::NullAction<Number> nullsetter;
704 const std::vector<unsigned int> varvec(1, var_num);
705
706 libMesh::GenericProjector<GeneralFieldTransfer::RecordRequests<Number>,
707 GeneralFieldTransfer::RecordRequests<Gradient>,
708 Number,
709 GeneralFieldTransfer::NullAction<Number>>
710 request_gather(*to_sys, f, &g, nullsetter, varvec);
711
712 // Defining only boundary values will not be enough to describe the variable, disallow it
713 if (_to_boundaries.size() && (_to_variables[var_index]->getContinuity() == DISCONTINUOUS))
714 mooseError("Higher order discontinuous elemental variables are not supported for "
715 "target-boundary "
716 "restricted transfers");
717
718 // Not implemented as the target mesh division could similarly be cutting elements in an
719 // arbitrary way with not enough requested points to describe the target variable
720 if (!_to_mesh_divisions.empty() && !_to_mesh_divisions[i_to]->coversEntireMesh())
721 mooseError("Higher order variable support not implemented for target mesh division "
722 "unless the mesh is fully covered / indexed in the mesh division. This must be "
723 "set programmatically in the MeshDivision object used.");
724
725 // We dont look at boundary restriction, not supported for higher order target variables
726 // Same for mesh divisions
727 const auto & to_begin = _to_blocks.empty()
728 ? to_mesh.active_local_elements_begin()
729 : to_mesh.active_local_subdomain_set_elements_begin(_to_blocks);
730
731 const auto & to_end = _to_blocks.empty()
732 ? to_mesh.active_local_elements_end()
733 : to_mesh.active_local_subdomain_set_elements_end(_to_blocks);
734
735 ConstElemRange to_elem_range(to_begin, to_end);
736
737 request_gather.project(to_elem_range);
738
739 dof_id_type point_id = 0;
740 for (const Point & p : f.points_requested())
741 // using the point number as a "dof_object_id" will serve to identify the point if we ever
742 // rework interp/distance_cache into the dof_id_to_value maps
743 this->cacheOutgoingPointInfo(
744 (*_to_transforms[global_i_to])(p), point_id++, i_to, outgoing_points);
745
746 // This is going to require more complicated transfer work
747 if (!g.points_requested().empty())
748 mooseError("We don't currently support variables with gradient degrees of freedom");
749 }
750 else if (is_nodal)
751 {
752 for (const auto & node : to_mesh.local_node_ptr_range())
753 {
754 // Skip this node if the variable has no dofs at it.
755 if (node->n_dofs(sys_num, var_num) < 1)
756 continue;
757
758 // Skip if it is a block restricted transfer and current node does not have
759 // specified blocks
760 if (!_to_blocks.empty() && !inBlocks(_to_blocks, to_moose_mesh, node))
761 continue;
762
763 if (!_to_boundaries.empty() && !onBoundaries(_to_boundaries, to_moose_mesh, node))
764 continue;
765
766 // Skip if the node does not meet the target mesh division behavior
767 // We cannot know from which app the data will come from so we cannot know
768 // the source mesh division index and the source app global index
769 if (!_to_mesh_divisions.empty() && _to_mesh_divisions[i_to]->divisionIndex(*node) ==
770 MooseMeshDivision::INVALID_DIVISION_INDEX)
771 continue;
772
773 // Cache point information
774 // We will use this information later for setting values back to solution vectors
775 cacheOutgoingPointInfo(
776 (*_to_transforms[global_i_to])(*node), node->id(), i_to, outgoing_points);
777 }
778 }
779 else // Elemental, constant monomial
780 {
781 for (const auto & elem :
782 as_range(to_mesh.local_elements_begin(), to_mesh.local_elements_end()))
783 {
784 // Skip this element if the variable has no dofs at it.
785 if (elem->n_dofs(sys_num, var_num) < 1)
786 continue;
787
788 // Skip if the element is not inside the block restriction
789 if (!_to_blocks.empty() && !inBlocks(_to_blocks, elem))
790 continue;
791
792 // Skip if the element does not have a side on the boundary
793 if (!_to_boundaries.empty() && !onBoundaries(_to_boundaries, to_moose_mesh, elem))
794 continue;
795
796 // Skip if the element is not indexed within the mesh division
797 if (!_to_mesh_divisions.empty() && _to_mesh_divisions[i_to]->divisionIndex(*elem) ==
798 MooseMeshDivision::INVALID_DIVISION_INDEX)
799 continue;
800
801 // Cache point information
802 // We will use this information later for setting values back to solution vectors
803 cacheOutgoingPointInfo((*_to_transforms[global_i_to])(elem->vertex_average()),
804 elem->id(),
805 i_to,
806 outgoing_points);
807 } // for
808 } // else
809 } // for
810}

826void
827MultiAppGeneralFieldTransfer::cacheIncomingInterpVals(
828 processor_id_type pid,
829 const unsigned int var_index,
830 std::vector<PointInfo> & pointInfoVec,
831 const std::vector<std::pair<Point, unsigned int>> & point_requests,
832 const std::vector<std::pair<Real, Real>> & incoming_vals,
833 DofobjectToInterpValVec & dofobject_to_valsvec,
834 InterpCaches & interp_caches,
835 InterpCaches & distance_caches)
836{
837 mooseAssert(pointInfoVec.size() == incoming_vals.size(),
838 "Number of dof objects does not equal to the number of incoming values");
839
840 dof_id_type val_offset = 0;
841 for (const auto & pointinfo : pointInfoVec)
842 {
843 // Retrieve target information from cached point infos
844 const auto problem_id = pointinfo.problem_id;
845 const auto dof_object_id = pointinfo.dof_object_id;
846
847 auto & fe_type = _to_variables[var_index]->feType();
848 bool is_nodal = _to_variables[var_index]->isNodal();
849
850 // In the higher order elemental variable case, we receive point values, not nodal or
851 // elemental. We use an InterpCache to store the values. The distance_cache is necessary to
852 // choose between multiple origin problems sending values. This code could be unified with the
853 // lower order order case by using the dofobject_to_valsvec
854 if (fe_type.order > CONSTANT && !is_nodal)
855 {
856 // Cache solution on target mesh in its local frame of reference
857 InterpCache & value_cache = interp_caches[problem_id];
858 InterpCache & distance_cache = distance_caches[problem_id];
859 Point p = _to_transforms[getGlobalTargetAppIndex(problem_id)]->mapBack(
860 point_requests[val_offset].first);
861 const Number val = incoming_vals[val_offset].first;
862
863 // Initialize distance to be able to compare
864 if (!distance_cache.hasKey(p))
865 distance_cache[p] = std::numeric_limits<Real>::max();
866
867 // We should only have one closest value for each variable at any given point.
868 // While there are shared Qps, on vertices for higher order variables usually,
869 // the generic projector only queries each point once
870 if (_search_value_conflicts && !GeneralFieldTransfer::isBetterOutOfMeshValue(val) &&
871 value_cache.hasKey(p) != 0 && !MooseUtils::absoluteFuzzyEqual(value_cache[p], val) &&
872 MooseUtils::absoluteFuzzyEqual(distance_cache[p], incoming_vals[val_offset].second))
873 registerConflict(problem_id, dof_object_id, p, incoming_vals[val_offset].second, false);
874
875 if ((!GeneralFieldTransfer::isBetterOutOfMeshValue(val) || _use_nearest_app) &&
876 MooseUtils::absoluteFuzzyGreaterThan(distance_cache[p], incoming_vals[val_offset].second))
877 {
878 // NOTE: We store the distance as well as the value. We really only need the
879 // value to construct the variable, but the distance is used to make decisions in nearest
880 // node schemes on which value to use
881 value_cache[p] = val;
882 distance_cache[p] = incoming_vals[val_offset].second;
883 }
884 }
885 else
886 {
887 // Using the dof object pointer, so we can handle
888 // both element and node using the same code
889#ifndef NDEBUG
890 auto var_num = _to_variables[var_index]->number();
891 auto & to_sys = _to_variables[var_index]->sys();
892
893 const MeshBase & to_mesh = _to_problems[problem_id]->mesh(_displaced_target_mesh).getMesh();
894 const DofObject * dof_object_ptr = nullptr;
895 const auto sys_num = to_sys.number();
896 // It is a node
897 if (is_nodal)
898 dof_object_ptr = to_mesh.node_ptr(dof_object_id);
899 // It is an element
900 else
901 dof_object_ptr = to_mesh.elem_ptr(dof_object_id);
902
903 // We should only be supporting nodal and constant elemental
904 // variables in this code path; if we see multiple DoFs on one
905 // object we should have been using GenericProjector
906 mooseAssert(dof_object_ptr->n_dofs(sys_num, var_num) == 1,
907 "Unexpectedly found " << dof_object_ptr->n_dofs(sys_num, var_num)
908 << "dofs instead of 1");
909#endif
910
911 auto & dofobject_to_val = dofobject_to_valsvec[problem_id];
912
913 // Check if we visited this dof object earlier
914 auto values_ptr = dofobject_to_val.find(dof_object_id);
915 // We did not visit this
916 if (values_ptr == dofobject_to_val.end())
917 {
918 // Values for this dof object
919 auto & val = dofobject_to_val[dof_object_id];
920 // Interpolation value
921 val.interp = incoming_vals[val_offset].first;
922 // Where this value came from
923 val.pid = pid;
924 // Distance
925 val.distance = incoming_vals[val_offset].second;
926 }
927 else
928 {
929 auto & val = values_ptr->second;
930
931 // Look for value conflicts
932 if (detectConflict(val.interp,
933 incoming_vals[val_offset].first,
934 val.distance,
935 incoming_vals[val_offset].second))
936 {
937 // Keep track of distance and value
938 const Point p =
939 getPointInTargetAppFrame(point_requests[val_offset].first,
940 problem_id,
941 "Registration of received equi-distant value conflict");
942 registerConflict(problem_id, dof_object_id, p, incoming_vals[val_offset].second, false);
943 }
944
945 // We adopt values that are, in order of priority
946 // - valid
947 // - closest distance
948 // - the smallest rank with the same distance
949 if (!GeneralFieldTransfer::isBetterOutOfMeshValue(incoming_vals[val_offset].first) &&
950 (MooseUtils::absoluteFuzzyGreaterThan(val.distance, incoming_vals[val_offset].second) ||
951 ((val.pid > pid) &&
952 MooseUtils::absoluteFuzzyEqual(val.distance, incoming_vals[val_offset].second))))
953 {
954 val.interp = incoming_vals[val_offset].first;
955 val.pid = pid;
956 val.distance = incoming_vals[val_offset].second;
957 }
958 }
959 }
960
961 // Move it to next position
962 val_offset++;
963 }
964}

977void
978MultiAppGeneralFieldTransfer::examineReceivedValueConflicts(
979 const unsigned int var_index,
980 const DofobjectToInterpValVec & dofobject_to_valsvec,
981 const InterpCaches & distance_caches)
982{
983 const auto var_name = getToVarName(var_index);
984 // We must check a posteriori because we could have two
985 // equidistant points with different values from two different problems, but a third point from
986 // another problem is actually closer, so there is no conflict because only that last one
987 // matters We check here whether the potential conflicts actually were the nearest points Loop
988 // over potential conflicts
989 for (auto conflict_it = _received_conflicts.begin(); conflict_it != _received_conflicts.end();)
990 {
991 const auto potential_conflict = *conflict_it;
992 bool overlap_found = false;
993
994 // Extract info for the potential conflict
995 const unsigned int problem_id = std::get<0>(potential_conflict);
996 const dof_id_type dof_object_id = std::get<1>(potential_conflict);
997 const Point p = std::get<2>(potential_conflict);
998 const Real distance = std::get<3>(potential_conflict);
999
1000 // Extract target variable info
1001 auto & es = getEquationSystem(*_to_problems[problem_id], _displaced_target_mesh);
1002 System * to_sys = find_sys(es, var_name);
1003 auto var_num = to_sys->variable_number(var_name);
1004 auto & fe_type = to_sys->variable_type(var_num);
1005 bool is_nodal = _to_variables[var_index]->isNodal();
1006
1007 // Higher order elemental
1008 if (fe_type.order > CONSTANT && !is_nodal)
1009 {
1010 auto cached_distance = distance_caches[problem_id].find(p);
1011 if (cached_distance == distance_caches[problem_id].end())
1012 mooseError("Conflict point was not found in the map of all origin-target distances");
1013 // Distance is still the distance when we detected a potential overlap
1014 if (MooseUtils::absoluteFuzzyEqual(cached_distance->second, distance))
1015 overlap_found = true;
1016 }
1017 // Nodal and const monomial variable
1018 else if (MooseUtils::absoluteFuzzyEqual(
1019 dofobject_to_valsvec[problem_id].find(dof_object_id)->second.distance, distance))
1020 overlap_found = true;
1021
1022 // Map will only keep the actual overlaps
1023 if (!overlap_found)
1024 _received_conflicts.erase(conflict_it);
1025 else
1026 ++conflict_it;
1027 }
1028}

1030void
1031MultiAppGeneralFieldTransfer::examineLocalValueConflicts(
1032 const unsigned int var_index,
1033 const DofobjectToInterpValVec & dofobject_to_valsvec,
1034 const InterpCaches & distance_caches)
1035{
1036 const auto var_name = getToVarName(var_index);
1037 // We must check a posteriori because we could have:
1038 // - two equidistant points with different values from two different problems
1039 // - two (or more) equidistant points with different values from the same problem
1040 // but a third point/value couple from another problem is actually closer, so there is no
1041 // conflict because only that last one matters. We check here whether the potential conflicts
1042 // actually were the nearest points. We use several global reductions. If there are not too many
1043 // potential conflicts (and there should not be in a well-posed problem) it should be manageably
1044 // expensive
1045
1046 // Move relevant conflict info (location, distance) to a smaller data structure
1047 std::vector<std::tuple<Point, Real>> potential_conflicts;
1048 potential_conflicts.reserve(_local_conflicts.size());
1049
1050 // Loop over potential conflicts to broadcast all the conflicts
1051 for (auto conflict_it = _local_conflicts.begin(); conflict_it != _local_conflicts.end();
1052 ++conflict_it)
1053 {
1054 // Extract info for the potential conflict
1055 const auto potential_conflict = *conflict_it;
1056 const unsigned int i_from = std::get<0>(potential_conflict);
1057 Point p = std::get<2>(potential_conflict);
1058 const Real distance = std::get<3>(potential_conflict);
1059 // If not using nearest-positions: potential conflict was saved in the source frame
1060 // If using nearest-positions: potential conflict was saved in the reference frame
1061 if (!_nearest_positions_obj)
1062 {
1063 const auto from_global_num = getGlobalSourceAppIndex(i_from);
1064 p = (*_from_transforms[from_global_num])(p);
1065 }
1066
1067 // Send data in the global frame of reference
1068 potential_conflicts.push_back(std::make_tuple(p, distance));
1069 }
1070 _communicator.allgather(potential_conflicts, false);
1071 // conflicts could have been reported multiple times within a tolerance
1072 std::sort(potential_conflicts.begin(), potential_conflicts.end());
1073 potential_conflicts.erase(unique(potential_conflicts.begin(),
1074 potential_conflicts.end(),
1075 [](auto l, auto r)
1076 {
1077 return std::get<0>(l).absolute_fuzzy_equals(std::get<0>(r)) &&
1078 std::abs(std::get<1>(l) - std::get<1>(r)) < TOLERANCE;
1079 }),
1080 potential_conflicts.end());
1081
1082 std::vector<std::tuple<Point, Real>> real_conflicts;
1083 real_conflicts.reserve(potential_conflicts.size());
1084
1085 // For each potential conflict, we need to identify what problem asked for that value
1086 for (auto conflict_it = potential_conflicts.begin(); conflict_it != potential_conflicts.end();
1087 ++conflict_it)
1088 {
1089 // Extract info for the potential conflict
1090 auto potential_conflict = *conflict_it;
1091 const Point p = std::get<0>(potential_conflict);
1092 const Real distance = std::get<1>(potential_conflict);
1093
1094 // Check all the problems to try to find this requested point in the data structures filled
1095 // with the received information
1096 bool target_found = false;
1097 bool conflict_real = false;
1098 for (const auto i_to : index_range(_to_problems))
1099 {
1100 // Extract variable info
1101 auto & es = getEquationSystem(*_to_problems[i_to], _displaced_target_mesh);
1102 System * to_sys = find_sys(es, var_name);
1103 auto var_num = to_sys->variable_number(var_name);
1104 auto & fe_type = to_sys->variable_type(var_num);
1105 bool is_nodal = _to_variables[var_index]->isNodal();
1106
1107 // Move to the local frame of reference for the target problem
1108 Point local_p =
1109 getPointInTargetAppFrame(p, i_to, "Resolution of local value conflicts detected");
1110
1111 // Higher order elemental
1112 if (fe_type.order > CONSTANT && !is_nodal)
1113 {
1114 // distance_caches finds use a binned floating point search
1115 auto cached_distance = distance_caches[i_to].find(local_p);
1116 if (cached_distance != distance_caches[i_to].end())
1117 {
1118 target_found = true;
1119 // Distance between source & target is still the distance we found in the sending
1120 // process when we detected a potential overlap while gathering values to send
1121 if (MooseUtils::absoluteFuzzyEqual(cached_distance->second, distance))
1122 conflict_real = true;
1123 }
1124 }
1125 // Nodal-value-dof-only and const monomial variable
1126 else
1127 {
1128 // Find the dof id for the variable to be set
1129 dof_id_type dof_object_id = std::numeric_limits<dof_id_type>::max();
1130 auto pl = _to_problems[i_to]->mesh().getPointLocator();
1131 pl->enable_out_of_mesh_mode();
1132 if (is_nodal)
1133 {
1134 auto node = pl->locate_node(local_p);
1135 if (node)
1136 // this is not the dof_id for the variable, but the dof_object_id
1137 dof_object_id = node->id();
1138 }
1139 else
1140 {
1141 auto elem = (*pl)(local_p);
1142 if (elem)
1143 dof_object_id = elem->id();
1144 }
1145 pl->disable_out_of_mesh_mode();
1146
1147 // point isn't even in mesh
1148 if (dof_object_id == std::numeric_limits<dof_id_type>::max())
1149 continue;
1150
1151 // this dof was not requested by this problem on this process
1152 if (dofobject_to_valsvec[i_to].find(dof_object_id) == dofobject_to_valsvec[i_to].end())
1153 continue;
1154
1155 target_found = true;
1156 // Check the saved distance in the vector of saved results. If the same, then the local
1157 // conflict we detected with that distance is still an issue after receiving all values
1158 if (MooseUtils::absoluteFuzzyEqual(
1159 dofobject_to_valsvec[i_to].find(dof_object_id)->second.distance, distance))
1160 conflict_real = true;
1161 }
1162 }
1163 // Only keep the actual conflicts / overlaps
1164 if (target_found && conflict_real)
1165 real_conflicts.push_back(potential_conflict);
1166 }
1167
1168 // Communicate real conflicts to all so they can be checked by every process
1169 _communicator.allgather(real_conflicts, false);
1170
1171 // Delete potential conflicts that were resolved
1172 // Each local list of conflicts will now be updated. It's important to keep conflict lists local
1173 // so we can give more context like the sending processor id (the domain of which can be
1174 // inspected by the user)
1175 for (auto conflict_it = _local_conflicts.begin(); conflict_it != _local_conflicts.end();)
1176 {
1177 // Extract info for the potential conflict
1178 const auto potential_conflict = *conflict_it;
1179 const unsigned int i_from = std::get<0>(potential_conflict);
1180 Point p = std::get<2>(potential_conflict);
1181 const Real distance = std::get<3>(potential_conflict);
1182 if (!_nearest_positions_obj)
1183 {
1184 const auto from_global_num = getGlobalSourceAppIndex(i_from);
1185 p = (*_from_transforms[from_global_num])(p);
1186 }
1187
1188 // If not in the vector of real conflicts, was not real so delete it
1189 if (std::find_if(real_conflicts.begin(),
1190 real_conflicts.end(),
1191 [p, distance](const auto & item)
1192 {
1193 return std::get<0>(item).absolute_fuzzy_equals(p) &&
1194 std::abs(std::get<1>(item) - distance) < TOLERANCE;
1195 }) == real_conflicts.end())
1196 _local_conflicts.erase(conflict_it);
1197 else
1198 ++conflict_it;
1199 }
1200}

1328void
1329MultiAppGeneralFieldTransfer::setSolutionVectorValues(
1330 const unsigned int var_index,
1331 const DofobjectToInterpValVec & dofobject_to_valsvec,
1332 const InterpCaches & interp_caches)
1333{
1334 // Get the variable name, with the accommodation for array/vector names
1335 const auto & var_name = getToVarName(var_index);
1336
1337 for (const auto problem_id : index_range(_to_problems))
1338 {
1339 auto & dofobject_to_val = dofobject_to_valsvec[problem_id];
1340
1341 // libMesh EquationSystems
1342 // NOTE: we would expect to set variables from the displaced equation system here
1343 auto & es = getEquationSystem(*_to_problems[problem_id], false);
1344
1345 // libMesh system
1346 System * to_sys = find_sys(es, var_name);
1347
1348 // libMesh mesh
1349 const MeshBase & to_mesh = _to_problems[problem_id]->mesh(_displaced_target_mesh).getMesh();
1350 auto var_num = to_sys->variable_number(var_name);
1351 auto sys_num = to_sys->number();
1352
1353 auto & fe_type = _to_variables[var_index]->feType();
1354 bool is_nodal = _to_variables[var_index]->isNodal();
1355
1356 if (fe_type.order > CONSTANT && !is_nodal)
1357 {
1358 // We may need to use existing data values in places where the
1359 // from app domain doesn't overlap
1360 MeshFunction to_func(es, *to_sys->current_local_solution, to_sys->get_dof_map(), var_num);
1361 to_func.init();
1362
1363 GeneralFieldTransfer::CachedData<Number> f(
1364 interp_caches[problem_id], to_func, _default_extrapolation_value);
1365 libMesh::VectorSetAction<Number> setter(*to_sys->solution);
1366 const std::vector<unsigned int> varvec(1, var_num);
1367
1368 libMesh::GenericProjector<GeneralFieldTransfer::CachedData<Number>,
1369 GeneralFieldTransfer::CachedData<Gradient>,
1370 Number,
1371 libMesh::VectorSetAction<Number>>
1372 set_solution(*to_sys, f, nullptr, setter, varvec);
1373
1374 // We dont look at boundary restriction, not supported for higher order target variables
1375 const auto & to_begin = _to_blocks.empty()
1376 ? to_mesh.active_local_elements_begin()
1377 : to_mesh.active_local_subdomain_set_elements_begin(_to_blocks);
1378
1379 const auto & to_end = _to_blocks.empty()
1380 ? to_mesh.active_local_elements_end()
1381 : to_mesh.active_local_subdomain_set_elements_end(_to_blocks);
1382
1383 ConstElemRange active_local_elem_range(to_begin, to_end);
1384
1385 set_solution.project(active_local_elem_range);
1386 }
1387 else
1388 {
1389 for (const auto & val_pair : dofobject_to_val)
1390 {
1391 const auto dof_object_id = val_pair.first;
1392
1393 const DofObject * dof_object = nullptr;
1394 if (is_nodal)
1395 dof_object = to_mesh.node_ptr(dof_object_id);
1396 else
1397 dof_object = to_mesh.elem_ptr(dof_object_id);
1398
1399 const auto dof = dof_object->dof_number(sys_num, var_num, 0);
1400 const auto val = val_pair.second.interp;
1401
1402 // This will happen if meshes are mismatched
1403 if (_error_on_miss && GeneralFieldTransfer::isBetterOutOfMeshValue(val))
1404 {
1405 const auto target_location =
1406 hasToMultiApp()
1407 ? " on target app " + std::to_string(getGlobalTargetAppIndex(problem_id))
1408 : " on parent app ";
1409 if (is_nodal)
1410 mooseError("No source value could be found for node ",
1411 dof_object_id,
1412 target_location,
1413 "could not be located. Node details:\n",
1414 _to_meshes[problem_id]->nodePtr(dof_object_id)->get_info());
1415 else
1416 mooseError("No source value could be found for element ",
1417 dof_object_id,
1418 target_location,
1419 "could not be located. Element details:\n",
1420 _to_meshes[problem_id]->elemPtr(dof_object_id)->get_info());
1421 }
1422
1423 // We should not put garbage into our solution vector
1424 // but it can be that we want to set it to a different value than what was already there
1425 // for example: the source app has been displaced and was sending an indicator of its
1426 // position
1427 if (GeneralFieldTransfer::isBetterOutOfMeshValue(val))
1428 {
1429 if (!GeneralFieldTransfer::isBetterOutOfMeshValue(_default_extrapolation_value))
1430 to_sys->solution->set(dof, _default_extrapolation_value);
1431 continue;
1432 }
1433
1434 to_sys->solution->set(dof, val);
1435 }
1436 }
1437
1438 to_sys->solution->close();
1439 // Sync local solutions
1440 to_sys->update();
1441 }

1444bool
1445MultiAppGeneralFieldTransfer::acceptPointInOriginMesh(unsigned int i_from,
1446 const std::vector<BoundingBox> & local_bboxes,
1447 const Point & pt,
1448 const unsigned int only_from_mesh_div,
1449 Real & distance) const
1450{
1451 if (_use_bounding_boxes && !local_bboxes[i_from].contains_point(pt))
1452 return false;
1453 else
1454 {
1455 auto * pl = _from_point_locators[i_from].get();
1456 const auto from_global_num = getGlobalSourceAppIndex(i_from);
1457 const auto transformed_pt = _from_transforms[from_global_num]->mapBack(pt);
1458
1459 // Check point against source block restriction
1460 if (!_from_blocks.empty() && !inBlocks(_from_blocks, pl, transformed_pt))
1461 return false;
1462
1463 // Check point against source boundary restriction. Block restriction will speed up the search
1464 if (!_from_boundaries.empty() &&
1465 !onBoundaries(_from_boundaries, _from_blocks, *_from_meshes[i_from], pl, transformed_pt))
1466 return false;
1467
1468 // Check point against the source mesh division
1469 if ((!_from_mesh_divisions.empty() || !_to_mesh_divisions.empty()) &&
1470 !acceptPointMeshDivision(transformed_pt, i_from, only_from_mesh_div))
1471 return false;
1472
1473 // Get nearest position (often a subapp position) for the target point
1474 // We want values from the child app that is closest to the same position as the target
1475 Point nearest_position_source;
1476 if (_nearest_positions_obj)
1477 {
1478 const bool initial = _fe_problem.getCurrentExecuteOnFlag() == EXEC_INITIAL;
1479 // The search for the nearest position is done in the reference frame
1480 const Point nearest_position = _nearest_positions_obj->getNearestPosition(pt, initial);
1481 nearest_position_source = _nearest_positions_obj->getNearestPosition(
1482 (*_from_transforms[from_global_num])(Point(0, 0, 0)), initial);
1483
1484 if (!_skip_coordinate_collapsing &&
1485 _from_transforms[from_global_num]->hasNonTranslationTransformation())
1486 mooseError("Rotation and scaling currently unsupported with nearest positions transfer.");
1487
1488 // Source (usually app position) is not closest to the same positions as the target, dont
1489 // send values
1490 if (nearest_position != nearest_position_source)
1491 return false;
1492
1493 // Set the distance as the distance from the nearest position to the target point
1494 distance = (pt - nearest_position_source).norm();
1495 }
1496
1497 // Check that the app actually contains the origin point
1498 // We dont need to check if we already found it in a block or a boundary
1499 if (_from_blocks.empty() && _from_boundaries.empty() && _source_app_must_contain_point &&
1500 !inMesh(pl, transformed_pt))
1501 return false;
1502 }
1503 return true;
1504}

1624bool
1625MultiAppGeneralFieldTransfer::acceptPointMeshDivision(
1626 const Point & pt, const unsigned int i_local, const unsigned int only_from_this_mesh_div) const
1627{
1628 // This routine can also be called to examine if the to_mesh_division index matches the current
1629 // source subapp index
1630 unsigned int source_mesh_div = MooseMeshDivision::INVALID_DIVISION_INDEX - 1;
1631 if (!_from_mesh_divisions.empty())
1632 source_mesh_div = _from_mesh_divisions[i_local]->divisionIndex(pt);
1633
1634 // If the point is not indexed in the source division
1635 if (!_from_mesh_divisions.empty() && source_mesh_div == MooseMeshDivision::INVALID_DIVISION_INDEX)
1636 return false;
1637 // If the point is not the at the same index in the target and the origin meshes, reject
1638 else if ((_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX ||
1639 _to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_DIVISION_INDEX) &&
1640 source_mesh_div != only_from_this_mesh_div)
1641 return false;
1642 // If the point is at a certain division index that is not the same as the index of the subapp
1643 // we wanted the information to be from for that point, reject
1644 else if (_from_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX &&
1645 source_mesh_div != only_from_this_mesh_div)
1646 return false;
1647 else if (_to_mesh_division_behavior == MeshDivisionTransferUse::MATCH_SUBAPP_INDEX &&
1648 only_from_this_mesh_div != getGlobalSourceAppIndex(i_local))
1649 return false;
1650 else
1651 return true;
1652}

1723 Point max(min_r, min_r, min_r);
1724 bool at_least_one = false;
1725 const auto & from_mesh = _from_problems[j]->mesh(_displaced_source_mesh);
1726
1727 for (const auto & elem : as_range(from_mesh.getMesh().local_elements_begin(),
1728 from_mesh.getMesh().local_elements_end()))
1729 {
1730 if (!_from_blocks.empty() && !inBlocks(_from_blocks, from_mesh, elem))
1731 continue;
1732
1733 for (const auto & node : elem->node_ref_range())
1734 {
1735 if (!_from_boundaries.empty() && !onBoundaries(_from_boundaries, from_mesh, &node))
1736 continue;
1737
1738 at_least_one = true;
1739 for (const auto i : make_range(LIBMESH_DIM))
1740 {
1741 min(i) = std::min(min(i), node(i));
1742 max(i) = std::max(max(i), node(i));
1743 }
1744 }
1745 }
1746
1747 // For 2D RZ problems, we need to amend the bounding box to cover the whole XYZ projection
1748 // - The XYZ-Y axis is assumed aligned with the RZ-Z axis
1749 // - RZ systems also cover negative coordinates hence the use of the maximum R
1750 // NOTE: We will only support the case where there is only one coordinate system
1751 if ((from_mesh.getUniqueCoordSystem() == Moose::COORD_RZ) && (LIBMESH_DIM == 3))
1752 {
1753 min(0) = -max(0);
1754 min(2) = -max(0);
1755 max(2) = max(0);
1756 }
1757
1758 BoundingBox bbox(min, max);
1759 if (!at_least_one)
1760 bbox.min() = max; // If we didn't hit any nodes, this will be _the_ minimum bbox
1761 else
1762 {
1763 // Translate the bounding box to the from domain's position. We may have rotations so we
1764 // must be careful in constructing the new min and max (first and second)
1765 const auto from_global_num = getGlobalSourceAppIndex(j);
1766 transformBoundingBox(bbox, *_from_transforms[from_global_num]);
1767 }
1768
1769 // Cast the bounding box into a pair of points (so it can be put through
1770 // MPI communication).
1771 bb_points[j] = static_cast<std::pair<Point, Point>>(bbox);
1772 }
1773
1774 // Serialize the bounding box points.
1775 _communicator.allgather(bb_points);
1776
1777 // Recast the points back into bounding boxes and return.
1778 std::vector<BoundingBox> bboxes(bb_points.size());
1779 for (const auto i : make_range(bb_points.size()))
1780 bboxes[i] = static_cast<BoundingBox>(bb_points[i]);
1781
1782 // TODO move up
1783 // Check for a user-set fixed bounding box size and modify the sizes as appropriate
1784 if (_fixed_bbox_size != std::vector<Real>(3, 0))
1785 for (const auto i : make_range(LIBMESH_DIM))
1786 if (!MooseUtils::absoluteFuzzyEqual(_fixed_bbox_size[i], 0))
1787 for (const auto j : make_range(bboxes.size()))
1788 {
1789 const auto current_width = (bboxes[j].second - bboxes[j].first)(i);
1790 bboxes[j].first(i) -= (_fixed_bbox_size[i] - current_width) / 2;
1791 bboxes[j].second(i) += (_fixed_bbox_size[i] - current_width) / 2;
1792 }
1793
1794 return bboxes;
1795}
