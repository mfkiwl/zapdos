105MortarConsumerInterface::MortarConsumerInterface(const MooseObject * moose_object)
106 : _mci_fe_problem(*moose_object->getCheckedPointerParam<FEProblemBase *>("_fe_problem_base")),
107 _mci_subproblem(*moose_object->getCheckedPointerParam<SubProblem *>("_subproblem")),
108 _mci_tid(moose_object->getParam<THREAD_ID>("_tid")),
109 _mci_mesh(_mci_subproblem.mesh()),
110 // all geometric assembly information should be correct for nl system number 0
111 _mci_assembly(_mci_subproblem.assembly(_mci_tid, 0)),
112 _mortar_data(_mci_fe_problem.mortarData()),
113 _secondary_id(
114 _mci_mesh.getBoundaryID(moose_object->getParam<BoundaryName>("secondary_boundary"))),
115 _primary_id(_mci_mesh.getBoundaryID(moose_object->getParam<BoundaryName>("primary_boundary"))),
116 _secondary_subdomain_id(
117 _mci_mesh.getSubdomainID(moose_object->getParam<SubdomainName>("secondary_subdomain"))),
118 _primary_subdomain_id(
119 _mci_mesh.getSubdomainID(moose_object->getParam<SubdomainName>("primary_subdomain"))),
120 _interpolate_normals(moose_object->getParam<bool>("interpolate_normals")),
121 _phys_points_secondary(_mci_assembly.qPointsFace()),
122 _phys_points_primary(_mci_assembly.qPointsFaceNeighbor()),
123 _qrule_msm(_mci_assembly.qRuleMortar()),
124 _qrule_face(_mci_assembly.qRuleFace()),
125 _lower_secondary_elem(_mci_assembly.lowerDElem()),
126 _lower_primary_elem(_mci_assembly.neighborLowerDElem()),
127 _JxW_msm(_mci_assembly.jxWMortar()),
128 _msm_elem(_mci_assembly.msmElem())
129{
130 const bool displaced = moose_object->isParamValid("use_displaced_mesh")
131 ? moose_object->getParam<bool>("use_displaced_mesh")
132 : false;
133
134 // Create the mortar interface if it hasn't already been created
135 _mci_fe_problem.createMortarInterface(
136 std::make_pair(_primary_id, _secondary_id),
137 std::make_pair(_primary_subdomain_id, _secondary_subdomain_id),
138 displaced,
139 moose_object->getParam<bool>("periodic"),
140 moose_object->getParam<bool>("debug_mesh"),
141 moose_object->getParam<bool>("correct_edge_dropping"),
142 moose_object->getParam<Real>("minimum_projection_angle"));
143
144 _amg = &_mci_fe_problem.getMortarInterface(
145 std::make_pair(_primary_id, _secondary_id),
146 std::make_pair(_primary_subdomain_id, _secondary_subdomain_id),
147 displaced);
148
149 const auto & secondary_set = _mortar_data.getHigherDimSubdomainIDs(_secondary_subdomain_id);
150 const auto & primary_set = _mortar_data.getHigherDimSubdomainIDs(_primary_subdomain_id);
151
152 std::set_union(secondary_set.begin(),
153 secondary_set.end(),
154 primary_set.begin(),
155 primary_set.end(),
156 std::inserter(_higher_dim_subdomain_ids, _higher_dim_subdomain_ids.begin()));
157 _boundary_ids = {_secondary_id, _primary_id};
158}

169void
170MortarConsumerInterface::trimDerivative(const dof_id_type remove_derivative_index,
171 ADReal & dual_number)
172{
173 auto md_it = dual_number.derivatives().nude_data().begin();
174 auto mi_it = dual_number.derivatives().nude_indices().begin();
175
176 auto d_it = dual_number.derivatives().nude_data().begin();
177
178 for (auto i_it = dual_number.derivatives().nude_indices().begin();
179 i_it != dual_number.derivatives().nude_indices().end();
180 ++i_it, ++d_it)
181 if (*i_it != remove_derivative_index)
182 {
183 *mi_it = *i_it;
184 *md_it = *d_it;
185 ++mi_it;
186 ++md_it;
187 }
188
189 std::size_t n_indices = md_it - dual_number.derivatives().nude_data().begin();
190 dual_number.derivatives().nude_indices().resize(n_indices);
191 dual_number.derivatives().nude_data().resize(n_indices);
192}
