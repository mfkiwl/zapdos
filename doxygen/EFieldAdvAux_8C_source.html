51template <bool is_ad>
52Real
53EFieldAdvAuxTempl<is_ad>::computeValue()
54{
55 return _sgn[_qp] * raw_value(_mu[_qp]) * std::exp(_density_log[_qp]) * -_grad_potential[_qp](0) *
56 _r_units * 6.02e23;
57}
