59SideTotFluxIntegral::computeQpIntegral()
60{
61 // Output units for base case are: mol / (m^2 * s)
62
63 if (_user_velocity > 0.)
64 _v_thermal = _user_velocity;
65 else
66 _v_thermal = std::sqrt(8 * _kb[_qp] * _T_heavy[_qp] / (M_PI * _mass[_qp]));
67
68 if (_normals[_qp] * _sgn[_qp] * -_grad_potential[_qp] > 0.0)
69 _a = 1.0;
70 else
71 _a = 0.0;
72
73 // return -_mobility_coef[_qp] * _grad_potential[_qp] * _normals[_qp] * std::exp(_u[_qp])
74 // -_diffusion_coef[_qp] * std::exp(_u[_qp]) * _grad_u[_qp] * _normals[_qp];
75
76 return (1. - _r) / (1. + _r) * 0.5 * _v_thermal * std::exp(_u[_qp]) +
77 (1. - _r) / (1. + _r) *
78 ((2 * _a - 1) * _sgn[_qp] * _mobility_coef[_qp] * -_grad_potential[_qp] * _r_units *
79 std::exp(_u[_qp]) * _normals[_qp]);
80}
