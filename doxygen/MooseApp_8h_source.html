1474template <typename T>
1475const T &
1476MooseApp::getRenamedParam(const std::string & old_name, const std::string & new_name) const
1477{
1478 // this enables having a default on the new parameter but bypassing it with the old one
1479 // Most important: accept new parameter
1480 if (isParamSetByUser(new_name) && !isParamValid(old_name))
1481 return InputParameters::getParamHelper(new_name, _pars, static_cast<T *>(0));
1482 // Second most: accept old parameter
1483 else if (isParamValid(old_name) && !isParamSetByUser(new_name))
1484 return InputParameters::getParamHelper(old_name, _pars, static_cast<T *>(0));
1485 // Third most: accept default for new parameter
1486 else if (isParamValid(new_name) && !isParamValid(old_name))
1487 return InputParameters::getParamHelper(new_name, _pars, static_cast<T *>(0));
1488 // Refuse: no default, no value passed
1489 else if (!isParamValid(old_name) && !isParamValid(new_name))
1490 mooseError(_pars.blockFullpath() + ": parameter '" + new_name +
1491 "' is being retrieved without being set.\n"
1492 "Did you mispell it?");
1493 // Refuse: both old and new parameters set by user
1494 else
1495 mooseError(_pars.blockFullpath() + ": parameter '" + new_name +
1496 "' may not be provided alongside former parameter '" + old_name + "'");
1497}
