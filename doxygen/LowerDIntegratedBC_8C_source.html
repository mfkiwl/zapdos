28LowerDIntegratedBC::LowerDIntegratedBC(const InputParameters & parameters)
29 : IntegratedBC(parameters),

76void
77LowerDIntegratedBC::computeLowerDJacobian(Moose::ConstraintJacobianType type)
78{
79 mooseAssert(type == Moose::LowerLower || type == Moose::LowerPrimary ||
80 type == Moose::PrimaryLower,
81 "Jacobian types must have lower in computeLowerDJacobian");
82
83 const auto & test_space =
84 (type == Moose::LowerLower || type == Moose::LowerPrimary) ? _test_lambda : _test;
85 unsigned int ivar = (type == Moose::LowerLower || type == Moose::LowerPrimary)
86 ? _lowerd_var.number()
87 : _var.number();
88
89 const auto & loc_phi =
90 (type == Moose::LowerLower || type == Moose::PrimaryLower) ? _phi_lambda : _phi;
91 const auto jvar = (type == Moose::LowerLower || type == Moose::PrimaryLower)
92 ? _lowerd_var.number()
93 : _var.number();
94
95 // need to transform the type for assembling Jacobian on boundary to be consistent with
96 // Assembly::addJacobianLowerD() and Assembly::prepareLowerD().
97 Moose::ConstraintJacobianType type_tr =
98 (type == Moose::LowerLower
99 ? type
100 : (type == Moose::LowerPrimary ? Moose::LowerSecondary : Moose::SecondaryLower));
101 prepareMatrixTagLower(_assembly, ivar, jvar, type_tr);
102
103 if (_local_ke.n() == 0 || _local_ke.m() == 0)
104 return;
105
106 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
107 {
108 initLowerDQpJacobian(type);
109 for (_i = 0; _i < test_space.size(); _i++)
110 for (_j = 0; _j < loc_phi.size(); _j++)
111 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeLowerDQpJacobian(type);
112 }
113
114 accumulateTaggedLocalMatrix();
115}

117void
118LowerDIntegratedBC::computeOffDiagJacobian(const unsigned int jvar_num)
119{
120 if (jvar_num == variable().number())
121 {
122 IntegratedBC::computeJacobian();
123 computeLowerDJacobian(Moose::LowerPrimary);
124 }
125 else if (jvar_num == _lowerd_var.number())
126 {
127 computeLowerDJacobian(Moose::LowerLower);
128 computeLowerDJacobian(Moose::PrimaryLower);
129 }
130 else
131 {
132 IntegratedBC::computeOffDiagJacobian(jvar_num);
133 computeLowerDOffDiagJacobian(Moose::LowerLower, jvar_num);
134 computeLowerDOffDiagJacobian(Moose::LowerPrimary, jvar_num);
135 computeLowerDOffDiagJacobian(Moose::PrimaryLower, jvar_num);
136 }
137}

139void
140LowerDIntegratedBC::computeLowerDOffDiagJacobian(Moose::ConstraintJacobianType type,
141 const unsigned int jvar_num)
142{
143 mooseAssert(type == Moose::LowerLower || type == Moose::LowerPrimary ||
144 type == Moose::PrimaryLower,
145 "Jacobian types must have lower in computeLowerDJacobian");
146
147 const auto & test_space =
148 (type == Moose::LowerLower || type == Moose::LowerPrimary) ? _test_lambda : _test;
149 const auto ivar = (type == Moose::LowerLower || type == Moose::LowerPrimary)
150 ? _lowerd_var.number()
151 : _var.number();
152
153 const auto & jvar = getVariable(jvar_num);
154
155 prepareMatrixTagLower(_assembly, ivar, jvar_num, type);
156 if (_local_ke.n() == 0 || _local_ke.m() == 0)
157 return;
158
159 if (jvar.fieldType() == Moose::VarFieldType::VAR_FIELD_STANDARD)
160 {
161 const auto & jv0 = static_cast<const MooseVariable &>(jvar);
162 const auto & loc_phi =
163 (type == Moose::LowerLower || type == Moose::PrimaryLower) ? jv0.phiLower() : jv0.phiFace();
164
165 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
166 {
167 initLowerDQpOffDiagJacobian(type, jvar);
168 for (_i = 0; _i < test_space.size(); _i++)
169 for (_j = 0; _j < loc_phi.size(); _j++)
170 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeLowerDQpOffDiagJacobian(type, jvar);
171 }
172 }
173 else if (jvar.fieldType() == Moose::VarFieldType::VAR_FIELD_ARRAY)
174 mooseError("Array variable cannot be coupled into integrated BC currently");
175 else
176 mooseError("Vector variable cannot be coupled into integrated BC currently");
177
178 accumulateTaggedLocalMatrix();
179}
