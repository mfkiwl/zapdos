204inline const Elem &
205FaceInfo::neighbor() const
206{
207 mooseAssert(_neighbor_info,

213inline FaceInfo::VarFaceNeighbors
214FaceInfo::faceType(const std::pair<unsigned int, unsigned int> & var_sys) const
215{
216 auto it = _face_types_by_var.find(var_sys);
217 if (it == _face_types_by_var.end())
218 mooseError("Variable number ",
219 var_sys.first,
220 " in system number ",
221 var_sys.second,
222 " not found in variable to VarFaceNeighbors map");
223 return it->second;
224}
