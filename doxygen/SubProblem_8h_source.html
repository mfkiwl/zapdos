1284template <typename T>
1285void
1286SubProblem::addFunctor(const std::string & name,
1287 const Moose::FunctorBase<T> & functor,
1288 const THREAD_ID tid)
1289{
1290 constexpr bool added_functor_is_ad =
1291 !std::is_same<T, typename MetaPhysicL::RawType<T>::value_type>::value;
1292
1293 mooseAssert(tid < _functors.size(), "Too large a thread ID");
1294
1295 auto & functor_to_request_info = _functor_to_request_info[tid];
1296 auto & functors = _functors[tid];
1297 auto it = functors.find("wraps_" + name);
1298 if (it != functors.end())
1299 {
1300 // We have this functor already. If it's a null functor, we want to replace it with the valid
1301 // functor we have now. If it's not then we'll add a new entry into the multimap and then we'll
1302 // error later if a user requests a functor because their request is ambiguous. This is the
1303 // reason that the functors container is a multimap: for nice error messages
1304 auto * const existing_wrapper_base =
1305 added_functor_is_ad ? std::get<2>(it->second).get() : std::get<1>(it->second).get();
1306 auto * const existing_wrapper = dynamic_cast<Moose::Functor<T> *>(existing_wrapper_base);
1307 if (existing_wrapper && existing_wrapper->template wrapsType<Moose::NullFunctor<T>>())
1308 {
1309 // Sanity check
1310 auto [request_info_it, request_info_end_it] = functor_to_request_info.equal_range(name);
1311 if (request_info_it == request_info_end_it)
1312 mooseError("We are wrapping a NullFunctor but we don't have any unfilled functor request "
1313 "info. This doesn't make sense.");
1314
1315 // Check for valid requests
1316 while (request_info_it != request_info_end_it)
1317 {
1318 auto & [requested_functor_is_ad, requestor_is_ad] = request_info_it->second;
1319 if (!requested_functor_is_ad && requestor_is_ad && added_functor_is_ad)
1320 mooseError("We are requesting a non-AD functor from an AD object, but the true functor "
1321 "is AD. This means we could be dropping important derivatives. We will not "
1322 "allow this");
1323 // We're going to eventually check whether we've fulfilled all functor requests and our
1324 // check will be that the multimap is empty. This request is fulfilled, so erase it from the
1325 // map now
1326 request_info_it = functor_to_request_info.erase(request_info_it);
1327 }
1328
1329 // Ok we didn't have the functor before, so we will add it now
1330 std::get<0>(it->second) =
1331 added_functor_is_ad ? SubProblem::TrueFunctorIs::AD : SubProblem::TrueFunctorIs::NONAD;
1332 existing_wrapper->assign(functor);
1333 // Finally we create the non-AD or AD complement of the just added functor
1334 if constexpr (added_functor_is_ad)
1335 {
1336 typedef typename MetaPhysicL::RawType<T>::value_type NonADType;
1337 auto * const existing_non_ad_wrapper_base = std::get<1>(it->second).get();
1338 auto * const existing_non_ad_wrapper =
1339 dynamic_cast<Moose::Functor<NonADType> *>(existing_non_ad_wrapper_base);
1340 mooseAssert(existing_non_ad_wrapper->template wrapsType<Moose::NullFunctor<NonADType>>(),
1341 "Both members of pair should have been wrapping a NullFunctor");
1342 existing_non_ad_wrapper->assign(
1343 std::make_unique<Moose::RawValueFunctor<NonADType>>(functor));
1344 }
1345 else
1346 {
1347 typedef typename Moose::ADType<T>::type ADType;
1348 auto * const existing_ad_wrapper_base = std::get<2>(it->second).get();
1349 auto * const existing_ad_wrapper =
1350 dynamic_cast<Moose::Functor<ADType> *>(existing_ad_wrapper_base);
1351 mooseAssert(existing_ad_wrapper->template wrapsType<Moose::NullFunctor<ADType>>(),
1352 "Both members of pair should have been wrapping a NullFunctor");
1353 existing_ad_wrapper->assign(std::make_unique<Moose::ADWrapperFunctor<ADType>>(functor));
1354 }
1355 return;
1356 }
1357 }
1358
1359 // We are a new functor, create the opposite ADType one and store it with other functors
1360 if constexpr (added_functor_is_ad)
1361 {
1362 typedef typename MetaPhysicL::RawType<T>::value_type NonADType;
1363 auto new_non_ad_wrapper = std::make_unique<Moose::Functor<NonADType>>(
1364 std::make_unique<Moose::RawValueFunctor<NonADType>>(functor));
1365 auto new_ad_wrapper = std::make_unique<Moose::Functor<T>>(functor);
1366 _functors[tid].emplace("wraps_" + name,
1367 std::make_tuple(SubProblem::TrueFunctorIs::AD,
1368 std::move(new_non_ad_wrapper),
1369 std::move(new_ad_wrapper)));
1370 }
1371 else
1372 {
1373 typedef typename Moose::ADType<T>::type ADType;
1374 auto new_non_ad_wrapper = std::make_unique<Moose::Functor<T>>((functor));
1375 auto new_ad_wrapper = std::make_unique<Moose::Functor<ADType>>(
1376 std::make_unique<Moose::ADWrapperFunctor<ADType>>(functor));
1377 _functors[tid].emplace("wraps_" + name,
1378 std::make_tuple(SubProblem::TrueFunctorIs::NONAD,
1379 std::move(new_non_ad_wrapper),
1380 std::move(new_ad_wrapper)));
1381 }
