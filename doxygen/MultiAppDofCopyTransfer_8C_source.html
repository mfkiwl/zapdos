54void
55MultiAppDofCopyTransfer::initialSetup()
56{
57 MultiAppFieldTransfer::initialSetup();
58
59 const FEProblemBase * from_problem;
60 const FEProblemBase * to_problem;
61
62 if (_current_direction == FROM_MULTIAPP)
63 {
64 // Subdomain and variable type information is shared on all subapps
65 from_problem = &getFromMultiApp()->appProblemBase(getFromMultiApp()->firstLocalApp());
66 to_problem = &getFromMultiApp()->problemBase();
67 }
68 else if (_current_direction == TO_MULTIAPP)
69 {
70 from_problem = &getToMultiApp()->problemBase();
71 to_problem = &getToMultiApp()->appProblemBase(getToMultiApp()->firstLocalApp());
72 }
73 else
74 {
75 from_problem = &getFromMultiApp()->appProblemBase(getFromMultiApp()->firstLocalApp());
76 to_problem = &getToMultiApp()->appProblemBase(getToMultiApp()->firstLocalApp());
77 }
78
79 // Convert block names to block IDs, fill with all blocks if unspecified
80 if (_has_block_restrictions)
81 {
82 const auto & from_block_names = getParam<std::vector<SubdomainName>>("from_blocks");
83 for (const auto & b : from_block_names)
84 if (!MooseMeshUtils::hasSubdomainName(const_cast<MeshBase &>(from_problem->mesh().getMesh()),
85 b))
86 paramError("from_blocks", "The block '", b, "' was not found in the mesh");
87
88 if (from_block_names.size())
89 {
90 if (from_problem)
91 {
92 const auto block_vec = from_problem->mesh().getSubdomainIDs(from_block_names);
93 _from_blocks = std::set<SubdomainID>(block_vec.begin(), block_vec.end());
94 }
95 // We dont even own any of these subapps
96 else
97 _from_blocks = {Moose::INVALID_BLOCK_ID};
98 }
99 else
100 _from_blocks = from_problem->mesh().meshSubdomains();
101
102 const auto & to_block_names = getParam<std::vector<SubdomainName>>("to_blocks");
103 for (const auto & b : to_block_names)
104 if (!MooseMeshUtils::hasSubdomainName(const_cast<MeshBase &>(to_problem->mesh().getMesh()),
105 b))
106 paramError("to_blocks", "The block '", b, "' was not found in the mesh");
107
108 if (to_block_names.size())
109 {
110 if (to_problem)
111 {
112 const auto block_vec = to_problem->mesh().getSubdomainIDs(to_block_names);
113 _to_blocks = std::set<SubdomainID>(block_vec.begin(), block_vec.end());
114 }
115 // We dont even own any of these subapps
116 else
117 _from_blocks = {Moose::INVALID_BLOCK_ID};
118 }
119 else
120 _to_blocks = to_problem->mesh().meshSubdomains();
121 }
122
123 // Forbid block restriction on nodal variables as currently not supported
124 if (_from_blocks.size())
125 for (auto & from_var : getFromVarNames())
126 if (from_problem
127 ->getVariable(
128 0, from_var, Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY)
129 .hasDoFsOnNodes())
130 paramError("from_blocks", "Block restriction is not implemented for nodal variables");
131 if (_to_blocks.size())
132 for (auto & to_var : getToVarNames())
133 if (to_problem
134 ->getVariable(
135 0, to_var, Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY)
136 .hasDoFsOnNodes())
137 paramError("to_blocks", "Block restriction is not implemented for nodal variables");
138}

140void
141MultiAppDofCopyTransfer::transfer(FEProblemBase & to_problem, FEProblemBase & from_problem)
142{
143 // Perform error checking
144 if (!getToVarNames().size())
145 mooseError("No transferred variables were specified, neither programmatically or through the "
146 "'source_variable' parameter");
147 if (getToVarNames().size() != getFromVarNames().size())
148 mooseError("Number of variables transferred must be same in both systems.");
149 for (auto & to_var : getToVarNames())
150 checkVariable(to_problem, to_var);
151 for (auto & from_var : getFromVarNames())
152 checkVariable(from_problem, from_var);
153
154 for (unsigned int v = 0; v < getToVarNames().size(); ++v)
155 {
156 // Populate the to/from variables needed to perform the transfer
157 MooseVariableFEBase & to_var = to_problem.getVariable(
158 0, getToVarNames()[v], Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY);
159 MeshBase & to_mesh = to_problem.mesh().getMesh();
160
161 MooseVariableFEBase & from_var = from_problem.getVariable(
162 0, getFromVarNames()[v], Moose::VarKindType::VAR_ANY, Moose::VarFieldType::VAR_FIELD_ANY);
163 MeshBase & from_mesh = from_problem.mesh().getMesh();
164
165 auto & to_solution = isParamValid("to_solution_tag")
166 ? to_var.sys().getVector(
167 to_problem.getVectorTagID(getParam<TagName>("to_solution_tag")))
168 : to_var.sys().solution();
169 auto & from_solution = isParamValid("from_solution_tag")
170 ? from_var.sys().getVector(from_problem.getVectorTagID(
171 getParam<TagName>("from_solution_tag")))
172 : from_var.sys().solution();
173
174 // Check integrity
175 if (to_var.feType() != from_var.feType())
176 mooseError("MultiAppFieldTransfer '",
177 name(),
178 "'requires that the target variable '",
179 to_var.name(),
180 "' and the source variable'",
181 from_var.name(),
182 "' must be the same type "
183 "(order and family): ",
184 libMesh::Utility::enum_to_string<FEFamily>(to_var.feType().family),
185 moose::internal::incompatVarMsg(to_var, from_var));
186 if (to_var.fieldType() != from_var.fieldType())
187 mooseError(
188 "Corresponding transfer variables must be same field type (STANDARD | VECTOR | ARRAY).");
189 if (to_var.count() != from_var.count())
190 mooseError("Corresponding transfer variables must have same number of components.");
191
192 if ((to_mesh.n_nodes() != from_mesh.n_nodes()) || (to_mesh.n_elem() != from_mesh.n_elem()))
193 mooseError("The meshes must be identical to utilize MultiAppDofCopyTransfer::transfer.");
194
195 // Transfer node dofs. Block restriction is not supported, forbidden in initialSetup
196 for (const auto & node : as_range(to_mesh.local_nodes_begin(), to_mesh.local_nodes_end()))
197 transferDofObject(
198 node, from_mesh.node_ptr(node->id()), to_var, from_var, to_solution, from_solution);
199
200 // Transfer elem dofs
201 for (auto & to_elem : as_range(to_mesh.local_elements_begin(), to_mesh.local_elements_end()))
202 {
203 Elem * from_elem = from_mesh.elem_ptr(to_elem->id());
204 mooseAssert(to_elem->type() == from_elem->type(), "The elements must be the same type.");
205
206 // Examine block restriction
207 if (_has_block_restrictions)
208 {
209 SubdomainID from_block = from_elem->subdomain_id();
210 if (std::find(_from_blocks.begin(), _from_blocks.end(), from_block) == _from_blocks.end())
211 continue;
212
213 SubdomainID to_block = to_elem->subdomain_id();
214 if (std::find(_to_blocks.begin(), _to_blocks.end(), to_block) == _to_blocks.end())
215 continue;
216 }
217
218 transferDofObject(to_elem, from_elem, to_var, from_var, to_solution, from_solution);
219 }
220
221 to_solution.close();
222 to_var.sys().update();
223 }

226void
227MultiAppDofCopyTransfer::transferDofObject(libMesh::DofObject * to_object,
228 libMesh::DofObject * from_object,
229 MooseVariableFEBase & to_var,
230 MooseVariableFEBase & from_var,
231 NumericVector<Number> & to_solution,
232 NumericVector<Number> & from_solution)
233{
234 for (unsigned int vc = 0; vc < to_var.count(); ++vc)
235 // Transfer from one solution vector to another
236 if (to_object->n_dofs(to_var.sys().number(), to_var.number() + vc) >
237 0) // If this variable has dofs at this node
238 for (unsigned int comp = 0;
239 comp < to_object->n_comp(to_var.sys().number(), to_var.number() + vc);
240 ++comp)
241 {
242 dof_id_type dof = to_object->dof_number(to_var.sys().number(), to_var.number() + vc, comp);
243 dof_id_type from_dof =
244 from_object->dof_number(from_var.sys().number(), from_var.number() + vc, comp);
245 Real from_value = from_solution(from_dof);
246 to_solution.set(dof, from_value);
247 }
