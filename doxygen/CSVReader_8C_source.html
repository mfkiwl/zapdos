20CSVReader::validParams()
21{
22 InputParameters params = GeneralVectorPostprocessor::validParams();
23 params.addClassDescription(
24 "Converts columns of a CSV file into vectors of a VectorPostprocessor.");
25 params.addRequiredParam<FileName>("csv_file",
26 "The name of the CSV file to read. Currently, with "
27 "the exception of the header row, only numeric "
28 "values are supported.");
29 params.addParam<bool>("header",
30 "When true it is assumed that the first row contains column headers, these "
31 "headers are used as the VectorPostprocessor vector names. If false the "
32 "file is assumed to contain only numbers and the vectors are named "
33 "automatically based on the column number (e.g., 'column_0000', "
34 "'column_0001'). If not supplied the reader attempts to auto detect the "
35 "headers.");
36 params.addParam<std::string>("delimiter",
37 "The column delimiter. Despite the name this can read files "
38 "separated by delimiter other than a comma. If this options is "
39 "omitted it will read comma or space separated files.");
40 params.addParam<bool>(
41 "ignore_empty_lines", true, "When true new empty lines in the file are ignored.");
42 params.set<bool>("contains_complete_history") = true;
43 params.suppressParameter<bool>("contains_complete_history");
44 params.set<ExecFlagEnum>("execute_on", true) = EXEC_NONE;
45 params.suppressParameter<ExecFlagEnum>("execute_on");
46
47 // The value from this VPP is naturally already on every processor
48 // TODO: Make this not the case! See #11415
49 params.set<bool>("_auto_broadcast") = false;
50
51 return params;
52}

54CSVReader::CSVReader(const InputParameters & params) : GeneralVectorPostprocessor(params)
55{
57 MooseUtils::DelimitedFileReader csv_reader(getParam<FileName>("csv_file"), &_communicator);
58 csv_reader.setIgnoreEmptyLines(getParam<bool>("ignore_empty_lines"));
59 if (isParamValid("header"))
60 csv_reader.setHeaderFlag(getParam<bool>("header")
61 ? MooseUtils::DelimitedFileReader::HeaderFlag::ON
62 : MooseUtils::DelimitedFileReader::HeaderFlag::OFF);
63 if (isParamValid("delimiter"))
64 csv_reader.setDelimiter(getParam<std::string>("delimiter"));
65
66 csv_reader.read();
67 const std::vector<std::string> & names = csv_reader.getNames();
68 const std::vector<std::vector<double>> & data = csv_reader.getData();
69 for (std::size_t i = 0; i < data.size(); ++i)
70 {
71 _column_data[names[i]] = &declareVector(names[i]);
72 _column_data[names[i]]->assign(data[i].begin(), data[i].end());
73 }
74}
