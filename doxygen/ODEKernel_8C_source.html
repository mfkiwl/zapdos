43ODEKernel::computeJacobian()
44{
45 prepareMatrixTag(_assembly, _var.number(), _var.number());
46
47 for (_i = 0; _i < _var.order(); _i++)
48 for (_j = 0; _j < _var.order(); _j++)
49 _local_ke(_i, _j) += computeQpJacobian();
50
51 accumulateTaggedLocalMatrix();
52
53 // compute off-diagonal jacobians wrt scalar variables
54 const std::vector<MooseVariableScalar *> & scalar_vars = _sys.getScalarVariables(_tid);
55 for (const auto & var : scalar_vars)
56 computeOffDiagJacobianScalar(var->number());
57}

60ODEKernel::computeOffDiagJacobianScalar(unsigned int jvar)
61{
62 prepareMatrixTag(_assembly, _var.number(), jvar);
63
64 MooseVariableScalar & var_j = _sys.getScalarVariable(_tid, jvar);
65 for (_i = 0; _i < _var.order(); _i++)
66 for (_j = 0; _j < var_j.order(); _j++)
67 {
68 if (jvar != _var.number())
69 _local_ke(_i, _j) += computeQpOffDiagJacobianScalar(jvar);
70 }
71
72 accumulateTaggedLocalMatrix();
73}
