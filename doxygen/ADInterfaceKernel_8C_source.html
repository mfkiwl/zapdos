71template <typename T>
72void
73ADInterfaceKernelTempl<T>::computeElemNeighResidual(Moose::DGResidualType type)
74{
75 bool is_elem;
76 if (type == Moose::Element)
77 is_elem = true;
78 else
79 is_elem = false;
80
81 const ADTemplateVariableTestValue<T> & test_space = is_elem ? _test : _test_neighbor;
82
83 if (is_elem)
84 prepareVectorTag(_assembly, _var.number());
85 else
86 prepareVectorTagNeighbor(_assembly, _neighbor_var.number());
87
88 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
89 {
90 initQpResidual(type);
91 for (_i = 0; _i < test_space.size(); _i++)
92 _local_re(_i) += raw_value(_ad_JxW[_qp] * _ad_coord[_qp] * computeQpResidual(type));
93 }
94
95 accumulateTaggedLocalResidual();
96}

98template <typename T>
99void
100ADInterfaceKernelTempl<T>::computeResidual()
101{
102 // in the gmsh mesh format (at least in the version 2 format) the "sideset" physical entities are
103 // associated only with the lower-dimensional geometric entity that is the boundary between two
104 // higher-dimensional element faces. It does not have a sidedness to it like the exodus format
105 // does. Consequently we may naively try to execute an interface kernel twice, one time where _var
106 // has dofs on _current_elem *AND* _neighbor_var has dofs on _neighbor_elem, and the other time
107 // where _var has dofs on _neighbor_elem and _neighbor_var has dofs on _current_elem. We only want
108 // to execute in the former case. In the future we should remove this and add some kind of "block"
109 // awareness to interface kernels to avoid all the unnecessary reinit that happens before we hit
110 // this return
111 if (!_var.activeOnSubdomain(_current_elem->subdomain_id()) ||
112 !_neighbor_var.activeOnSubdomain(_neighbor_elem->subdomain_id()))
113 return;
114
115 precalculateResidual();
116
117 // Compute the residual for this element
118 computeElemNeighResidual(Moose::Element);
119
120 // Compute the residual for the neighbor
121 computeElemNeighResidual(Moose::Neighbor);
122}

124template <typename T>
125void
126ADInterfaceKernelTempl<T>::computeElemNeighJacobian(Moose::DGJacobianType type)
127{
128 mooseAssert(type == Moose::ElementElement || type == Moose::NeighborNeighbor,
129 "With AD you should need one call per side");
130
131 const ADTemplateVariableTestValue<T> & test_space =
132 (type == Moose::ElementElement || type == Moose::ElementNeighbor) ? _test : _test_neighbor;
133
134 std::vector<ADReal> residuals(test_space.size(), 0);
135
136 switch (type)
137 {
138 case Moose::ElementElement:
139 resid_type = Moose::Element;
140 break;
141 case Moose::ElementNeighbor:
142 resid_type = Moose::Element;
143 break;
144 case Moose::NeighborElement:
145 resid_type = Moose::Neighbor;
146 break;
147 case Moose::NeighborNeighbor:
148 resid_type = Moose::Neighbor;
149 break;
150 default:
151 mooseError("Unknown DGJacobianType ", type);
152 }
153
154 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
155 {
156 initQpResidual(resid_type);
157 for (_i = 0; _i < test_space.size(); _i++)
158 residuals[_i] += _ad_JxW[_qp] * _ad_coord[_qp] * computeQpResidual(resid_type);
159 }
160
161 const bool element_var_is_var = (type == Moose::ElementElement || type == Moose::ElementNeighbor);
162 addJacobian(_assembly,
163 residuals,
164 element_var_is_var ? _var.dofIndices() : _neighbor_var.dofIndicesNeighbor(),
165 element_var_is_var ? _var.scalingFactor() : _neighbor_var.scalingFactor());
166}

193ADInterfaceKernelTempl<T>::computeOffDiagElemNeighJacobian(Moose::DGJacobianType type, unsigned int)
194{
195 mooseAssert(type == Moose::ElementElement || type == Moose::NeighborNeighbor,
196 "With AD you should need one call per side");
197

199 type == Moose::ElementElement ? _test : _test_neighbor;

203 else

207

226{

228 // associated only with the lower-dimensional geometric entity that is the boundary between two
229 // higher-dimensional element faces. It does not have a sidedness to it like the exodus format
230 // does. Consequently we may naively try to execute an interface kernel twice, one time where _var
231 // has dofs on _current_elem *AND* _neighbor_var has dofs on _neighbor_elem, and the other time
232 // where _var has dofs on _neighbor_elem and _neighbor_var has dofs on _current_elem. We only want

237 !_neighbor_var.activeOnSubdomain(_neighbor_elem->subdomain_id()))

248 computeOffDiagElemNeighJacobian(Moose::ElementElement, jvar);
