34InterfaceDiffusion::computeQpResidual(Moose::DGResidualType type)
35{
36 Real r = 0;
37
38 switch (type)
39 {
40 case Moose::Element:
41 r = _test[_i][_qp] * -_D_neighbor[_qp] * _grad_neighbor_value[_qp] * _normals[_qp];
42 break;
43
44 case Moose::Neighbor:
45 r = _test_neighbor[_i][_qp] * _D[_qp] * _grad_u[_qp] * _normals[_qp];
46 break;
47 }
48
49 return r;
50}

53InterfaceDiffusion::computeQpJacobian(Moose::DGJacobianType type)
54{
55 Real jac = 0;
56
57 switch (type)
58 {
59 case Moose::ElementElement:
60 case Moose::NeighborNeighbor:
61 break;
62
63 case Moose::NeighborElement:
64 jac = _test_neighbor[_i][_qp] * _D[_qp] * _grad_phi[_j][_qp] * _normals[_qp];
65 break;
66
67 case Moose::ElementNeighbor:
68 jac = _test[_i][_qp] * -_D_neighbor[_qp] * _grad_phi_neighbor[_j][_qp] * _normals[_qp];
69 break;
70 }
71
72 return jac;
73}
