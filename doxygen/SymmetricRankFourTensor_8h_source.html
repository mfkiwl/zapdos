461template <typename T>
462template <typename T2>
463auto
464SymmetricRankFourTensorTempl<T>::operator*(const SymmetricRankTwoTensorTempl<T2> & b) const
465 -> SymmetricRankTwoTensorTempl<decltype(T() * T2())>
466{
467 typedef decltype(T() * T2()) ValueType;
468 SymmetricRankTwoTensorTempl<ValueType> result;
469
470 std::size_t index = 0;
471 for (const auto i : make_range(N))
472 {
473 ValueType tmp = 0.0;
474 for (const auto j : make_range(N))
475 tmp += _vals[index++] * b._vals[j];
476 result._vals[i] = tmp;
477 }
478
479 return result;
480}

495template <typename T>
496template <typename T2>
497void
498SymmetricRankFourTensorTempl<T>::fillSymmetric9FromInputVector(const T2 & input)
499{
500 mooseAssert(LIBMESH_DIM == 3, "This method assumes LIBMESH_DIM == 3");
501 mooseAssert(input.size() == 9,
502 "To use fillSymmetric9FromInputVector, your input must have size 9.");
503 zero();
504
505 _vals[0] = input[0]; // C1111
506 _vals[7] = input[3]; // C2222
507 _vals[14] = input[5]; // C3333
508
509 _vals[1] = _vals[6] = input[1]; // C1122
510 _vals[2] = _vals[12] = input[2]; // C1133
511 _vals[8] = _vals[13] = input[4]; // C2233
512
513 static constexpr std::size_t C2323 = 21;
514 static constexpr std::size_t C1313 = 28;
515 static constexpr std::size_t C1212 = 35;
516
517 _vals[C2323] = 2.0 * input[6];
518 _vals[C1313] = 2.0 * input[7];
519 _vals[C1212] = 2.0 * input[8];
520}

522template <typename T>
523template <typename T2>
524void
525SymmetricRankFourTensorTempl<T>::fillSymmetric21FromInputVector(const T2 & input)
526{
527 // C1111 C1122 C1133 C1123 C1113 C1112
528 // C2222 C2233 C2223 C2213 C2212
529 // C3333 C3323 C3313 C3312
530 // C2323 C2313 C2312
531 // C1313 C1312
532 // C1212
533
534 mooseAssert(LIBMESH_DIM == 3, "This method assumes LIBMESH_DIM == 3");
535 mooseAssert(input.size() == 21,
536 "To use fillSymmetric21FromInputVector, your input must have size 21.");
537 std::size_t index = 0;
538 for (const auto i : make_range(N))
539 for (const auto j : make_range(i, N))
540 {
541 _vals[i + N * j] = mandelFactor(i, j) * input[index];
542 _vals[j + N * i] = mandelFactor(j, i) * input[index];
543 index++;
544 }
545}

547template <typename T>
548SymmetricRankFourTensorTempl<T>
549SymmetricRankFourTensorTempl<T>::invSymm() const
550{
551 SymmetricRankFourTensorTempl<T> result(initNone);
552
553 if constexpr (SymmetricRankFourTensorTempl<T>::N2 * sizeof(T) > EIGEN_STACK_ALLOCATION_LIMIT)
554 {
555 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> mat(N, N);
556 for (const auto i : make_range(N))
557 for (const auto j : make_range(N))
558 mat(i, j) = (*this)(i, j);
559 mat = mat.inverse();
560 for (const auto i : make_range(N))
561 for (const auto j : make_range(N))
562 result(i, j) = mat(i, j);
563 }
564 else
565 {
566 const Eigen::Map<const Eigen::Matrix<T, N, N, Eigen::RowMajor>> mat(&_vals[0]);
567 Eigen::Map<Eigen::Matrix<T, N, N, Eigen::RowMajor>> res(&result._vals[0]);
568 res = mat.inverse();
569 }
570
571 return result;
572}
