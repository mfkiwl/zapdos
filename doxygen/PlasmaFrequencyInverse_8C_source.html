58void
59PlasmaFrequencyInverse::finalize()
60{
61 gatherMax(_value);
62
63 if (_use_moles)
64 {
65 _em_density = std::exp(_value) * 6.022e23;
66 }
67 else
68 {
69 _em_density = std::exp(_value);
70 }
71
72 Real _plasma_freq = std::sqrt(std::pow(1.6022e-19, 2) * _em_density / (8.8542e-12 * 9.1094e-31));
73
74 _inverse = 1. / _plasma_freq;
75}
