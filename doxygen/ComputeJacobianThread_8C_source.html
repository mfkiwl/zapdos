40void
41ComputeJacobianThread::compute(KernelBase & kernel)
42{
43 if (kernel.isImplicit())
44 {
45 kernel.prepareShapes(kernel.variable().number());
46 kernel.computeJacobian();
47 if (_fe_problem.checkNonlocalCouplingRequirement() && !_fe_problem.computingScalingJacobian())
48 mooseError("Nonlocal kernels only supported for non-diagonal coupling. Please specify an SMP "
49 "preconditioner, with appropriate row-column coupling or specify full = true.");
50 }
51}

97void
98ComputeJacobianThread::determineObjectWarehouses()
99{
100 // If users pass a empty vector or a full size of vector,
101 // we take all kernels
102 if (!_tags.size() || _tags.size() == _fe_problem.numMatrixTags())
103 {
104 _tag_kernels = &_kernels;
105 _dg_warehouse = &_dg_kernels;
106 _ibc_warehouse = &_integrated_bcs;
107 _ik_warehouse = &_interface_kernels;
108 }
109 // If we have one tag only,
110 // We call tag based storage
111 else if (_tags.size() == 1)
112 {
113 _tag_kernels = &(_kernels.getMatrixTagObjectWarehouse(*(_tags.begin()), _tid));
114 _dg_warehouse = &(_dg_kernels.getMatrixTagObjectWarehouse(*(_tags.begin()), _tid));
115 _ibc_warehouse = &(_integrated_bcs.getMatrixTagObjectWarehouse(*(_tags.begin()), _tid));
116 _ik_warehouse = &(_interface_kernels.getMatrixTagObjectWarehouse(*(_tags.begin()), _tid));
117 }
118 // This one may be expensive, and hopefully we do not use it so often
119 else
120 {
121 _tag_kernels = &(_kernels.getMatrixTagsObjectWarehouse(_tags, _tid));
122 _dg_warehouse = &(_dg_kernels.getMatrixTagsObjectWarehouse(_tags, _tid));
123 _ibc_warehouse = &(_integrated_bcs.getMatrixTagsObjectWarehouse(_tags, _tid));
124 _ik_warehouse = &(_interface_kernels.getMatrixTagsObjectWarehouse(_tags, _tid));
125 }
126}
