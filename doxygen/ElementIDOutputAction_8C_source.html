29void
30ElementIDOutputAction::act()
31{
32 // Do nothing if the application does not have output
33 if (!_app.actionWarehouse().hasActions("add_output"))
34 return;
35
36 if (_current_task == "add_aux_kernel")
37 {
38 const auto & output_actions = _app.actionWarehouse().getActionListByName("add_output");
39 for (const auto & act : output_actions)
40 {
41 // Extract the Output action
42 AddOutputAction * action = dynamic_cast<AddOutputAction *>(act);
43 if (!action)
44 continue;
45
46 InputParameters & params = action->getObjectParams();
47 if (params.isParamValid("output_extra_element_ids") &&
48 params.get<bool>("output_extra_element_ids"))
49 {
50 bool has_element_id_names = params.isParamValid("extra_element_ids_to_output");
51 std::vector<std::string> element_id_names;
52 if (has_element_id_names)
53 element_id_names = params.get<std::vector<std::string>>("extra_element_ids_to_output");
54
55 auto var_params = _factory.getValidParams("MooseVariableConstMonomial");
56 auto kernel_params = _factory.getValidParams("ExtraElementIDAux");
57 kernel_params.set<ExecFlagEnum>("execute_on") = EXEC_INITIAL;
58 for (unsigned int i = 0; i < _mesh->getMesh().n_elem_integers(); ++i)
59 {
60 auto & var_name = _mesh->getMesh().get_elem_integer_name(i);
61 if (!has_element_id_names ||
62 (std::find(element_id_names.begin(), element_id_names.end(), var_name) !=
63 element_id_names.end()))
64 {
65 // Create aux variables based on the extra element id name
66 _problem->addAuxVariable("MooseVariableConstMonomial", var_name, var_params);
67
68 // Create aux kernels based on the extra element id name
69 kernel_params.set<AuxVariableName>("variable") = var_name;
70 kernel_params.set<std::vector<ExtraElementIDName>>("extra_id_name") = {var_name};
71 _problem->addAuxKernel("ExtraElementIDAux", "_output_" + var_name, kernel_params);
72 }
73 }
74 }
75 }
76 }
77}
