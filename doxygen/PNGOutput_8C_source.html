66// Funtion for making the _mesh_function object.
67void
68PNGOutput::makeMeshFunc()
69{
70
71 // The number assigned to the variable. Used to build the correct mesh. Default is 0.
72 unsigned int variable_number = 0;
73
74 // PNGOutput does not currently scale for running in parallel.
75 if (processor_id() != 0)
76 mooseInfo("PNGOutput is not currently scalable.");
77
78 bool var_found = false;
79 if (_problem_ptr->getAuxiliarySystem().hasVariable(_variable))
80 {
81 variable_number = _problem_ptr->getAuxiliarySystem().getVariable(0, _variable).number();
82 var_found = true;
83 }
84
85 else
86 for (const auto nl_sys_num : make_range(_problem_ptr->numNonlinearSystems()))
87 if (_problem_ptr->getNonlinearSystem(nl_sys_num).hasVariable(_variable))
88 {
89 variable_number =
90 _problem_ptr->getNonlinearSystem(nl_sys_num).getVariable(0, _variable).number();
91 _nl_sys_num = nl_sys_num;
92 var_found = true;
93 }
94
95 if (!var_found)
96 paramError("variable", "This doesn't exist.");
97
98 const std::vector<unsigned int> var_nums = {variable_number};
99
100 // If we want the background to be transparent, we need a number over 1.
101 if (_transparent_background)
102 _out_bounds_shade = 2;
103
104 // Find the values that will be used for rescaling purposes.
105 calculateRescalingValues();
106
107 // Set up the mesh_function
108 if (_nl_sys_num == libMesh::invalid_uint)
109 _mesh_function =
110 std::make_unique<MeshFunction>(*_es_ptr,
111 _problem_ptr->getAuxiliarySystem().serializedSolution(),
112 _problem_ptr->getAuxiliarySystem().dofMap(),
113 var_nums);
114 else
115 _mesh_function = std::make_unique<MeshFunction>(
116 *_es_ptr,
117 _problem_ptr->getNonlinearSystem(_nl_sys_num).serializedSolution(),
118 _problem_ptr->getNonlinearSystem(_nl_sys_num).dofMap(),
119 var_nums);
120 _mesh_function->init();
121
122 // Need to enable out of mesh with the given control color scaled in reverse
123 // so when scaling is done, this value retains it's original value.
124 _mesh_function->enable_out_of_mesh_mode(reverseScale(_out_bounds_shade));
125}

127// Function to find the min and max values so that all the values can be scaled between the two.
128void
129PNGOutput::calculateRescalingValues()
130{
131 // The max and min.
132 // If the max value wasn't specified in the input file, find it from the system.
133 if (!_pars.isParamSetByUser("max"))
134 {
135 if (_nl_sys_num == libMesh::invalid_uint)
136 _scaling_max = _problem_ptr->getAuxiliarySystem().serializedSolution().max();
137 else
138 _scaling_max = _problem_ptr->getNonlinearSystem(_nl_sys_num).serializedSolution().max();
139 }
140 else
141 _scaling_max = _max;
142
143 // If the min value wasn't specified in the input file, find it from the system.
144 if (!_pars.isParamSetByUser("min"))
145 {
146 if (_nl_sys_num == libMesh::invalid_uint)
147 _scaling_min = _problem_ptr->getAuxiliarySystem().serializedSolution().min();
148 else
149 _scaling_min = _problem_ptr->getNonlinearSystem(_nl_sys_num).serializedSolution().min();
150 }
151 else
152 _scaling_min = _min;
153
154 // The amount the values will need to be shifted.
155 _shift_value = 0;
156
157 // Get the shift value.
158 if (_scaling_min != 0)
159 {
160 // Shiftvalue will be the same magnitude, but
161 // going in the opposite direction of the scalingMin
162 _shift_value -= _scaling_min;
163 }
164
165 // Shift the max.
166 _scaling_max += _shift_value;
167}

169// Function to apply the scale to the data points.
170// Needed to be able to see accurate images that cover the appropriate color spectrum.
171inline Real
172PNGOutput::applyScale(Real value_to_scale)
173{
174 return ((value_to_scale + _shift_value) / _scaling_max);
175}
176
177// Function to reverse the scaling that happens to a value.
178// Needed to be able to accurately control the _out_bounds_shade.
179inline Real
180PNGOutput::reverseScale(Real value_to_unscale)
181{
182 return ((value_to_unscale * _scaling_max) - _shift_value);
183}

185// Function that controls the colorization of the png image for non-grayscale images.
186void
187PNGOutput::setRGB(png_byte * rgb, Real selection)
188{
189 // With this system we have a color we start with when the value is 0 and another it approaches as
190 // the value increases all the way to 255. If we want it to approach another color from that new
191 // color, it will do so for the next 255, so the transition is from 256 - 511. For each
192 // additional color we want to transition to, we need another 255. Transitioning from no color, or
193 // black to Red then Green then Blue then the values of from black as it becomes Red would be 0 -
194 // 255, Red to Green as 256 - 511 and then Green to Blue as 512 - 767 which gives us our total
195 // colorSpectrum of 0 - 767, which includes those colors and each of their states in the
196 // transistion.
197 unsigned int number_of_destination_colors = 1;
198 switch (_color)
199 {
200 // BRYW. Three destination colors (R,Y,W).
201 case 1:
202 number_of_destination_colors = 3;
203 break;
204
205 // BWR. Two destination colors (W,R).
206 case 2:
207 number_of_destination_colors = 2;
208 break;
209
210 // RWB. Two destination colors (W,B).
211 case 3:
212 number_of_destination_colors = 2;
213 break;
214
215 // BR. One destination color (R).
216 case 4:
217 number_of_destination_colors = 1;
218 break;
219 }
220
221 // We need to convert the number of colors into the spectrum max, then convert the value from the
222 // mesh to a point somewhere in the range of 0 to color_spectrum_max.
223 auto color_spectrum_max = (256 * number_of_destination_colors) - 1;
224 auto color = (unsigned int)(selection * color_spectrum_max);
225
226 // Unless we specifically say some part is transparent, we want the whole image to be opaque.
227 auto tran = (unsigned int)(_transparency * 255);
228
229 // Make sure everything is within our colorSpectrum. If it's bigger, then we want a
230 // transparent background.
231 if (color > color_spectrum_max)
232 {
233 color = color_spectrum_max;
234 tran = 0;
235 }
236
237 auto magnitude = color % 256;
238
239 switch (_color)
240 {
241 // Current color scheme: Blue->Red->Yellow->White
242 case 1:
243 // Blue->Red
244 if (color < 256)
245 {
246 rgb[0] = magnitude;
247 rgb[1] = 0;
248 rgb[2] = 50; // 255 - magnitude;
249 }
250 // Red->Yellow
251 else if (color < 512)
252 {
253 rgb[0] = 255;
254 rgb[1] = magnitude;
255 rgb[2] = 0;
256 }
257 // Yellow->White
258 else
259 {
260 rgb[0] = 255;
261 rgb[1] = 255;
262 rgb[2] = magnitude;
263 }
264 break;
265
266 // Color Scheme: Blue->White->Red
267 // Using the RGB values found in Paraview
268 case 2:
269 // Blue->White
270 if (color < 256)
271 {
272 rgb[0] = (int)(255.0 * (0.231373 + (0.002485 * (float)magnitude)));
273 rgb[1] = (int)(255.0 * (0.298039 + (0.002223 * (float)magnitude)));
274 rgb[2] = (int)(255.0 * (0.752941 + (0.000439 * (float)magnitude)));
275 }
276 // White->Red
277 else
278 {
279 rgb[0] = (int)(255.0 * (0.865003 - (0.000624 * (float)magnitude)));
280 rgb[1] = (int)(255.0 * (0.865003 - (0.003331 * (float)magnitude)));
281 rgb[2] = (int)(255.0 * (0.865003 - (0.002808 * (float)magnitude)));
282 }
283 break;
284
285 // Red->White->Blue
286 case 3:
287 // Red->White
288 if (color < 256)
289 {
290 rgb[0] = 255;
291 rgb[1] = magnitude;
292 rgb[2] = magnitude;
293 }
294 // White->Blue
295 else
296 {
297 rgb[0] = 255 - magnitude;
298 rgb[1] = 255 - magnitude;
299 rgb[2] = 255;
300 }
301 break;
302
303 // Blue->Red
304 case 4:
305 // Blue->Red
306 rgb[0] = magnitude;
307 rgb[1] = 0;
308 rgb[2] = 255 - magnitude;
309 break;
310 }
311 // Add any transparency.
312 rgb[3] = tran;
313}

326// Function the writes the PNG out to the appropriate filename.
327void
328PNGOutput::makePNG()
329{
330 // Get the max and min of the BoundingBox
331 Point max_point = _box.max();
332 Point min_point = _box.min();
333
334 // The the total distance on the x and y axes.
335 Real dist_x = max_point(0) - min_point(0);
336 Real dist_y = max_point(1) - min_point(1);
337
338 // Width and height for the PNG image.
339 Real width;
340 Real height;
341
342 // Variable to record the resolution variable after normalized to work with pixels in longest
343 // direction.
344 Real normalized_resolution;
345
346 // The longer dimension becomes the value to which we scale the other.
347 if (dist_x > dist_y)
348 {
349 width = _resolution;
350 height = (_resolution / dist_x) * dist_y;
351 normalized_resolution = (((Real)_resolution) / dist_x);
352 }
353 else
354 {
355 height = _resolution;
356 width = (_resolution / dist_y) * dist_x;
357 normalized_resolution = (((Real)_resolution) / dist_y);
358 }
359
360 // Create the filename based on base and the test step number.
361 std::ostringstream png_file;
362 png_file << _file_base << "_" << std::setfill('0') << std::setw(3) << _t_step << ".png";
363
364 // libpng is built on C, so by default it takes FILE*.
365 FILE * fp = nullptr;
366 png_structp pngp = nullptr;
367 png_infop infop = nullptr;
368 // Required depth for proper image clarity.
369 Real depth = 8;
370 // Allocate resources.
371 std::vector<png_byte> row((width + 1) * 4);
372
373 // Check if we can open and write to the file.
374 MooseUtils::checkFileWriteable(png_file.str());
375
376 // Open the file with write and bit modes.
377 fp = fopen(png_file.str().c_str(), "wb");
378
379 pngp = png_create_write_struct(PNG_LIBPNG_VER_STRING, nullptr, nullptr, nullptr);
380 if (!pngp)
381 mooseError("Failed to make the pointer string for the png.");
382
383 infop = png_create_info_struct(pngp);
384 if (!infop)
385 mooseError("Failed to make an info pointer for the png.");
386
387 // Initializes the IO for the png. Needs FILE* to compile.
388 png_init_io(pngp, fp);
389
390 // Set up the PNG header.
391 png_set_IHDR(pngp,
392 infop,
393 width,
394 height,
395 depth,
396 (_color ? PNG_COLOR_TYPE_RGBA : PNG_COLOR_TYPE_GRAY),
397 PNG_INTERLACE_NONE,
398 PNG_COMPRESSION_TYPE_DEFAULT,
399 PNG_FILTER_TYPE_DEFAULT);
400
401 png_write_info(pngp, infop);
402
403 // Initiallizing the point that will be used for populating the mesh values.
404 // Initializing x, y, z to zero so that we don't access the point before it's
405 // been set. z = 0 for all the png's.
406 Point pt(0, 0, 0);
407
408 // Dense vector that we can pass into the _mesh_function to fill with a value for a given point.
409 DenseVector<Number> dv(0);
410
411 // Loop through to create the image.
412 for (Real y = max_point(1); y >= min_point(1); y -= 1. / normalized_resolution)
413 {
414 pt(1) = y;
415 unsigned int index = 0;
416 for (Real x = min_point(0); x <= max_point(0); x += 1. / normalized_resolution)
417 {
418 pt(0) = x;
419 (*_mesh_function)(pt, _time, dv, nullptr);
420
421 // Determine whether to create the PNG in color or grayscale
422 if (_color)
423 setRGB(&row.data()[index * 4], applyScale(dv(0)));
424 else
425 row.data()[index] = applyScale(dv(0)) * 255;
426
427 index++;
428 }
429 png_write_row(pngp, row.data());
430 }
431
432 // Close the file and take care of some other png end stuff.
433 png_write_end(pngp, nullptr);
434 if (fp != nullptr)
435 fclose(fp);
436 if (infop != nullptr)
437 png_free_data(pngp, infop, PNG_FREE_ALL, -1);
438 if (pngp != nullptr)
439 png_destroy_write_struct(&pngp, &infop);
440}
