55DenseVector<Real>
56ReporterData::getAllRealReporterValues() const
57{
58 DenseVector<Real> all_values;
59
60 std::vector<Real> & output = all_values.get_values();
61
62 for (const auto & name_context_pair : _context_ptrs)
63 {
64 const ReporterName & rname = name_context_pair.first;
65
66 if (hasReporterValue<Real>(rname))
67 output.push_back(getReporterValue<Real>(rname.getCombinedName()));
68
69 if (hasReporterValue<std::vector<Real>>(rname))
70 {
71 const auto & vec = getReporterValue<std::vector<Real>>(rname.getCombinedName());
72 for (const auto & v : vec)
73 output.push_back(v);
74 }
75 }
76
77 return all_values;
78}

81ReporterData::getAllRealReporterFullNames() const

177std::string
178ReporterData::getReporterInfo(const ReporterStateBase & state, const ReporterContextBase * context)
179{
180 std::stringstream oss;
181
182 const auto & name = state.getReporterName();
183
184 if (name.isPostprocessor())
185 oss << "Postprocessor \"" << name.getObjectName() << "\":\n";
186 else
187 {
188 oss << name.specialTypeToName() << " \"" << name.getCombinedName() << "\":\n Type:\n "
189 << state.valueType() << "\n";
190 }
191 oss << " Producer:\n ";
192 if (context)
193 {
194 oss << context->getProducer().type() << " \"" << context->getProducer().name() << "\"";
195 oss << "\n Context type:\n " << context->contextType();
196 }
197 else
198 oss << "None";
199 oss << "\n Consumer(s):\n";
200 if (state.getConsumers().empty())
201 oss << " None\n";
202 else
203 for (const auto & mode_object_pair : state.getConsumers())
204 {
205 const ReporterMode mode = mode_object_pair.first;
206 const MooseObject * object = mode_object_pair.second;
207 oss << " " << object->typeAndName() << " (mode: " << mode << ")\n";
208 }
209
210 return oss.str();
211}
