47ADReal
48JouleHeating::computeQpResidual()
49{
50 return _test[_i][_qp] * -_grad_potential[_qp] * _r_units * _voltage_scaling *
51 (-_mu[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_em[_qp]) -
52 _diff[_qp] * std::exp(_em[_qp]) * _grad_em[_qp] * _r_units);
53}
