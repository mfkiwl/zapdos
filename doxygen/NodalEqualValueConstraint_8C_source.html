35NodalEqualValueConstraint::computeResidual()
36{
37 prepareVectorTag(_assembly, _var.number());
38
39 for (unsigned int k = 0; k < _value.size(); k++)
40 _local_re(k) = (*_value[k])[0] - (*_value[k])[1];
41
42 assignTaggedLocalResidual();
43}

45void
46NodalEqualValueConstraint::computeJacobian()
47{
48 prepareMatrixTag(_assembly, _var.number(), _var.number());
49
50 // put zeroes on the diagonal (we have to do it, otherwise PETSc will complain!)
51 for (unsigned int i = 0; i < _local_ke.m(); i++)
52 for (unsigned int j = 0; j < _local_ke.n(); j++)
53 _local_ke(i, j) = 0.;
54
55 assignTaggedLocalMatrix();
56
57 for (unsigned int k = 0; k < _value.size(); k++)
58 {
59 prepareMatrixTag(_assembly, _var.number(), _val_number[k]);
60
61 _local_ke(k, 0) = 1.;
62 _local_ke(k, 1) = -1.;
63
64 assignTaggedLocalMatrix();
65 }
66}
