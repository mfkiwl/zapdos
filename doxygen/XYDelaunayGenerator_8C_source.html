115std::unique_ptr<MeshBase>
116XYDelaunayGenerator::generate()
117{
118 // Put the boundary mesh in a local pointer
119 std::unique_ptr<UnstructuredMesh> mesh =
120 dynamic_pointer_cast<UnstructuredMesh>(std::move(_bdy_ptr));
121
122 // Get ready to triangulate the line segments we extract from it
123 Poly2TriTriangulator poly2tri(*mesh);
124 poly2tri.triangulation_type() = TriangulatorInterface::PSLG;
125
126 // If we're using a user-requested subset of boundaries on that
127 // mesh, get their ids.
128 std::set<std::size_t> bdy_ids;
129
130 if (isParamValid("input_boundary_names"))
131 {
132 if (isParamValid("input_subdomain_names"))
133 paramError(
134 "input_subdomain_names",
135 "input_boundary_names and input_subdomain_names cannot both specify an outer boundary.");
136
137 const auto & boundary_names = getParam<std::vector<BoundaryName>>("input_boundary_names");
138 for (const auto & name : boundary_names)
139 {
140 auto bcid = MooseMeshUtils::getBoundaryID(name, *mesh);
141 if (bcid == BoundaryInfo::invalid_id)
142 paramError("input_boundary_names", name, " is not a boundary name in the input mesh");
143
144 bdy_ids.insert(bcid);
145 }
146 }
147
148 if (isParamValid("input_subdomain_names"))
149 {
150 const auto & subdomain_names = getParam<std::vector<SubdomainName>>("input_subdomain_names");
151
152 const auto subdomain_ids = MooseMeshUtils::getSubdomainIDs(*mesh, subdomain_names);
153
154 // Check that the requested subdomains exist in the mesh
155 std::set<SubdomainID> subdomains;
156 mesh->subdomain_ids(subdomains);
157
158 for (auto i : index_range(subdomain_ids))
159 {
160 if (subdomain_ids[i] == Moose::INVALID_BLOCK_ID || !subdomains.count(subdomain_ids[i]))
161 paramError(
162 "input_subdomain_names", subdomain_names[i], " was not found in the boundary mesh");
163
164 bdy_ids.insert(subdomain_ids[i]);
165 }
166 }
167
168 if (!bdy_ids.empty())
169 poly2tri.set_outer_boundary_ids(bdy_ids);
170
171 poly2tri.set_interpolate_boundary_points(_add_nodes_per_boundary_segment);
172 poly2tri.set_refine_boundary_allowed(_refine_bdy);
173 poly2tri.set_verify_hole_boundaries(_verify_holes);
174
175 poly2tri.desired_area() = _desired_area;
176 poly2tri.minimum_angle() = 0; // Not yet supported
177 poly2tri.smooth_after_generating() = _smooth_tri;
178
179 if (_desired_area_func != "")
180 {
181 // poly2tri will clone this so it's fine going out of scope
182 ParsedFunction<Real> area_func{_desired_area_func};
183 poly2tri.set_desired_area_function(&area_func);
184 }
185
186 std::vector<TriangulatorInterface::MeshedHole> meshed_holes;
187 std::vector<TriangulatorInterface::Hole *> triangulator_hole_ptrs(_hole_ptrs.size());
188 std::vector<std::unique_ptr<MeshBase>> hole_ptrs(_hole_ptrs.size());
189
190 // Make sure pointers here aren't invalidated by a resize
191 meshed_holes.reserve(_hole_ptrs.size());
192 for (auto hole_i : index_range(_hole_ptrs))
193 {
194 hole_ptrs[hole_i] = std::move(*_hole_ptrs[hole_i]);
195 meshed_holes.emplace_back(*hole_ptrs[hole_i]);
196 if (hole_i < _refine_holes.size())
197 meshed_holes.back().set_refine_boundary_allowed(_refine_holes[hole_i]);
198
199 triangulator_hole_ptrs[hole_i] = &meshed_holes.back();
200 }
201
202 if (!triangulator_hole_ptrs.empty())
203 poly2tri.attach_hole_list(&triangulator_hole_ptrs);
204
205 poly2tri.triangulate();
206
207 if (isParamValid("output_subdomain_name"))
208 {
209 auto output_subdomain_name = getParam<SubdomainName>("output_subdomain_name");
210 _output_subdomain_id = MooseMeshUtils::getSubdomainID(output_subdomain_name, *mesh);
211
212 if (_output_subdomain_id == Elem::invalid_subdomain_id)
213 {
214 // We'll probably need to make a new ID, then
215 _output_subdomain_id = MooseMeshUtils::getNextFreeSubdomainID(*mesh);
216
217 // But check the hole meshes for our output subdomain name too
218 for (auto & hole_ptr : hole_ptrs)
219 {
220 auto possible_sbdid = MooseMeshUtils::getSubdomainID(output_subdomain_name, *hole_ptr);
221 // Huh, it was in one of them
222 if (possible_sbdid != Elem::invalid_subdomain_id)
223 {
224 _output_subdomain_id = possible_sbdid;
225 break;
226 }
227 _output_subdomain_id =
228 std::max(_output_subdomain_id, MooseMeshUtils::getNextFreeSubdomainID(*hole_ptr));
229 }
230
231 mesh->subdomain_name(_output_subdomain_id) = output_subdomain_name;
232 }
233 }
234
235 if (_smooth_tri || _output_subdomain_id)
236 for (auto elem : mesh->element_ptr_range())
237 {
238 mooseAssert(elem->type() == TRI3, "Unexpected non-Tri3 found in triangulation");
239
240 elem->subdomain_id() = _output_subdomain_id;
241
242 // I do not trust Laplacian mesh smoothing not to invert
243 // elements near reentrant corners. Eventually we'll add better
244 // smoothing options, but even those might have failure cases.
245 // Better to always do extra tests here than to ever let users
246 // try to run on a degenerate mesh.
247 if (_smooth_tri)
248 {
249 auto cross_prod = (elem->point(1) - elem->point(0)).cross(elem->point(2) - elem->point(0));
250
251 if (cross_prod(2) <= 0)
252 mooseError("Inverted element found in triangulation.\n"
253 "Laplacian smoothing can create these at reentrant corners; disable it?");
254 }
255 }
256
257 // Assign new subdomain name, if provided
258 if (isParamValid("output_subdomain_name"))
259 mesh->subdomain_name(_output_subdomain_id) = getParam<SubdomainName>("output_subdomain_name");
260
261 const bool use_binary_search = (_algorithm == "BINARY");
262
263 // The hole meshes are specified by the user, so they could have any
264 // BCID or no BCID or any combination of BCIDs on their outer
265 // boundary, so we'll have to set our own BCID to use for stitching
266 // there. We'll need to check all the holes for used BCIDs, if we
267 // want to pick a new ID on hole N that doesn't conflict with any
268 // IDs on hole M < N (or with the IDs on the new triangulation)
269
270 // The new triangulation by default assigns BCID i+1 to hole i ...
271 // but we can't even use this for mesh stitching, because we can't
272 // be sure it isn't also already in use on the hole's mesh and so we
273 // won't be able to safely clear it afterwards.
274 const boundary_id_type end_bcid = hole_ptrs.size() + 1;
275 boundary_id_type new_hole_bcid = end_bcid;
276
277 // We might be overriding the default bcid numbers. We have to be
278 // careful about how we renumber, though. We pick unused temporary
279 // numbers because e.g. "0->2, 2->0" is impossible to do
280 // sequentially, but "0->N, 2->N+2, N->2, N+2->0" works.
281 if (isParamValid("output_boundary"))
282 libMesh::MeshTools::Modification::change_boundary_id(*mesh, 0, end_bcid);
283
284 if (isParamValid("hole_boundaries"))
285 {
286 auto hole_boundaries = getParam<std::vector<BoundaryName>>("hole_boundaries");
287 auto hole_boundary_ids = MooseMeshUtils::getBoundaryIDs(*mesh, hole_boundaries, true);
288
289 if (hole_boundary_ids.size() != hole_ptrs.size())
290 paramError("hole_boundary_ids", "Need one hole_boundary_ids entry per hole, if specified.");
291
292 for (auto h : index_range(hole_ptrs))
293 libMesh::MeshTools::Modification::change_boundary_id(*mesh, h + 1, h + 1 + end_bcid);
294
295 for (auto h : index_range(hole_ptrs))
296 {
297 libMesh::MeshTools::Modification::change_boundary_id(
298 *mesh, h + 1 + end_bcid, hole_boundary_ids[h]);
299 mesh->get_boundary_info().sideset_name(hole_boundary_ids[h]) = hole_boundaries[h];
300 new_hole_bcid = std::max(new_hole_bcid, boundary_id_type(hole_boundary_ids[h] + 1));
301 }
302 }
303
304 if (isParamValid("output_boundary"))
305 {
306 const BoundaryName output_boundary = getParam<BoundaryName>("output_boundary");
307 const std::vector<BoundaryID> output_boundary_id =
308 MooseMeshUtils::getBoundaryIDs(*mesh, {output_boundary}, true);
309
310 libMesh::MeshTools::Modification::change_boundary_id(*mesh, end_bcid, output_boundary_id[0]);
311 mesh->get_boundary_info().sideset_name(output_boundary_id[0]) = output_boundary;
312
313 new_hole_bcid = std::max(new_hole_bcid, boundary_id_type(output_boundary_id[0] + 1));
314 }
315
316 bool doing_stitching = false;
317
318 for (auto hole_i : index_range(hole_ptrs))
319 {
320 const MeshBase & hole_mesh = *hole_ptrs[hole_i];
321 auto & hole_boundary_info = hole_mesh.get_boundary_info();
322 const std::set<boundary_id_type> & local_hole_bcids = hole_boundary_info.get_boundary_ids();
323
324 if (!local_hole_bcids.empty())
325 new_hole_bcid = std::max(new_hole_bcid, boundary_id_type(*local_hole_bcids.rbegin() + 1));
326 hole_mesh.comm().max(new_hole_bcid);
327
328 if (hole_i < _stitch_holes.size() && _stitch_holes[hole_i])
329 doing_stitching = true;
330 }
331
332 const boundary_id_type inner_bcid = new_hole_bcid + 1;
333
334 // libMesh mesh stitching still requires a serialized mesh, and it's
335 // cheaper to do that once than to do it once-per-hole
336 MeshSerializer serial(*mesh, doing_stitching);
337
338 for (auto hole_i : index_range(hole_ptrs))
339 {
340 if (hole_i < _stitch_holes.size() && _stitch_holes[hole_i])
341 {
342 UnstructuredMesh & hole_mesh = dynamic_cast<UnstructuredMesh &>(*hole_ptrs[hole_i]);
343 auto & hole_boundary_info = hole_mesh.get_boundary_info();
344
345 // Our algorithm here requires a serialized Mesh. To avoid
346 // redundant serialization and deserialization (libMesh
347 // MeshedHole and stitch_meshes still also require
348 // serialization) we'll do the serialization up front.
349 MeshSerializer serial_hole(hole_mesh);
350
351 // It would have been nicer for MeshedHole to add the BCID
352 // itself, but we want MeshedHole to work with a const mesh.
353 // We'll still use MeshedHole, for its code distinguishing
354 // outer boundaries from inner boundaries on a
355 // hole-with-holes.
356 TriangulatorInterface::MeshedHole mh{hole_mesh};
357
358 // We have to translate from MeshedHole points to mesh
359 // sides.
360 std::unordered_map<Point, Point> next_hole_boundary_point;
361 const int np = mh.n_points();
362 for (auto pi : make_range(1, np))
363 next_hole_boundary_point[mh.point(pi - 1)] = mh.point(pi);
364 next_hole_boundary_point[mh.point(np - 1)] = mh.point(0);
365
366#ifndef NDEBUG
367 int found_hole_sides = 0;
368#endif
369 for (auto elem : hole_mesh.element_ptr_range())
370 {
371 if (elem->dim() != 2)
372 mooseError("Non 2-D element found in hole; stitching is not supported.");
373
374 auto ns = elem->n_sides();
375 for (auto s : make_range(ns))
376 {
377 auto it_s = next_hole_boundary_point.find(elem->point(s));
378 if (it_s != next_hole_boundary_point.end())
379 if (it_s->second == elem->point((s + 1) % ns))
380 {
381 hole_boundary_info.add_side(elem, s, new_hole_bcid);
382#ifndef NDEBUG
383 ++found_hole_sides;
384#endif
385 }
386 }
387 }
388 mooseAssert(found_hole_sides == np, "Failed to find full outer boundary of meshed hole");
389
390 auto & mesh_boundary_info = mesh->get_boundary_info();
391#ifndef NDEBUG
392 int found_inner_sides = 0;
393#endif
394 for (auto elem : mesh->element_ptr_range())
395 {
396 auto ns = elem->n_sides();
397 for (auto s : make_range(ns))
398 {
399 auto it_s = next_hole_boundary_point.find(elem->point((s + 1) % ns));
400 if (it_s != next_hole_boundary_point.end())
401 if (it_s->second == elem->point(s))
402 {
403 mesh_boundary_info.add_side(elem, s, inner_bcid);
404#ifndef NDEBUG
405 ++found_inner_sides;
406#endif
407 }
408 }
409 }
410 mooseAssert(found_inner_sides == np, "Failed to find full boundary around meshed hole");
411
412 mesh->stitch_meshes(hole_mesh,
413 inner_bcid,
414 new_hole_bcid,
415 TOLERANCE,
416 /*clear_stitched_bcids*/ true,
417 _verbose_stitching,
418 use_binary_search);
419 }
420 }
421 mesh->prepare_for_use();
422 return mesh;
423}
