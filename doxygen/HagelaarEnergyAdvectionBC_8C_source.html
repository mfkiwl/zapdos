53ADReal
54HagelaarEnergyAdvectionBC::computeQpResidual()
55{
56 if (_normals[_qp] * -1.0 * -_grad_potential[_qp] > 0.0)
57 {
58 _a = 1.0;
59 }
60 else
61 {
62 _a = 0.0;
63 }
64
65 _ion_flux = _sgnip[_qp] * _muip[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_ip[_qp]) -
66 _Dip[_qp] * std::exp(_ip[_qp]) * _grad_ip[_qp] * _r_units;
67
68 return _test[_i][_qp] * _r_units / (6. * (_r + 1.)) *
69 (10. * _ion_flux * _normals[_qp] * _se_energy[_qp] * _se_coeff[_qp] * (_a - 1.) *
70 (_r + 1.) +
71 (_r - 1.) * (std::exp(_u[_qp]) - _se_energy[_qp] * _n_gamma) *
72 (6. * -_grad_potential[_qp] * _r_units * _normals[_qp] * _mumean_en[_qp] *
73 (2. * _a - 1.) -
74 5. * _v_thermal));
75}
