19ReactionThirdOrder::validParams()
20{
21 InputParameters params = Kernel::validParams();
22 params.addRequiredCoupledVar("v", "The first variable that is reacting.");
23 params.addRequiredCoupledVar("w", "The second variable that is reacting.");
24 params.addRequiredCoupledVar("x", "The third variable that is reacting.");
25 params.addRequiredParam<std::string>("reaction", "The full reaction equation.");
26 params.addRequiredParam<Real>("coefficient", "The stoichiometric coeffient.");
27 params.addParam<bool>("_v_eq_u", false, "Whether or not v and u are the same variable.");
28 params.addParam<bool>("_w_eq_u", false, "Whether or not w and u are the same variable.");
29 params.addParam<bool>("_x_eq_u", false, "Whether or not x and u are the same variable.");
30 params.addParam<std::string>(
31 "number",
32 "",
33 "The reaction number. Optional, just for material property naming purposes. If a single "
34 "reaction has multiple different rate coefficients (frequently the case when multiple "

127ReactionThirdOrder::computeQpOffDiagJacobian(unsigned int jvar)
128{
129 Real aval;
130 aval = -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * _phi[_j][_qp];
131
132 if (!_v_eq_u && !_w_eq_u && !_x_eq_u)
133 {
134 if (jvar == _v_id && jvar == _w_id && jvar == _x_id)
135 return aval * 3.0 * _v[_qp] * _v[_qp]; // This means that v, w, and x are the same variable.
136 else if (jvar == _v_id && jvar != _w_id && jvar != _x_id)
137 return aval * _w[_qp] * _x[_qp];
138 else if (jvar == _v_id && jvar != _w_id && jvar == _x_id) // v and x are the same
139 return aval * 2.0 * _v[_qp] * _w[_qp];
140 else if (jvar == _v_id && jvar == _w_id && jvar != _x_id)
141 return aval * 2.0 * _v[_qp] * _x[_qp];
142 else if (jvar != _v_id && jvar == _w_id && jvar == _x_id)
143 return aval * _v[_qp] * 2.0 * _w[_qp];
144 else if (jvar != _v_id && jvar != _w_id && jvar == _x_id)
145 return aval * _v[_qp] * _w[_qp];
146 else if (jvar != _v_id && jvar == _w_id && jvar != _x_id)
147 return aval * _v[_qp] * _x[_qp];
148 else
149 return 0.0;
150 }
151 else if (!_v_eq_u && _w_eq_u && _x_eq_u)
152 {
153 if (jvar == _v_id)
154 return aval * _w[_qp] * _x[_qp];
155 else
156 return 0.0;
157 }
158 else if (!_v_eq_u && _w_eq_u && !_x_eq_u)
