57std::unique_ptr<MeshBase>
58RefineSidesetGenerator::generate()
59{
60 // Get the list of boundary ids from the boundary names
61 const auto boundary_ids = MooseMeshUtils::getBoundaryIDs(
62 *_input, getParam<std::vector<BoundaryName>>("boundaries"), false);
63
64 // Check that the boundary ids/names exist in the mesh
65 for (std::size_t i = 0; i < boundary_ids.size(); ++i)
66 if (boundary_ids[i] == Moose::INVALID_BOUNDARY_ID)
67 paramError("boundaries",
68 "The boundary '",
69 getParam<std::vector<BoundaryName>>("boundaries")[i],
70 "' was not found within the mesh");
71 std::unique_ptr<MeshBase> mesh = std::move(_input);
72 int max = *std::max_element(_refinement.begin(), _refinement.end());
73 return recursive_refine(boundary_ids, mesh, _refinement, max);
74}

76std::unique_ptr<MeshBase>
77RefineSidesetGenerator::recursive_refine(std::vector<boundary_id_type> boundary_ids,
78 std::unique_ptr<MeshBase> & mesh,
79 std::vector<int> refinement,
80 int max,
81 int ref_step)
82{
83 // If the refinement step has reached the largest value in the _refinement array, return the mesh,
84 // as we are done.
85 if (ref_step == max)
86 return dynamic_pointer_cast<MeshBase>(mesh);
87
88 const bool old_allow_remote_element_removal = mesh->allow_remote_element_removal();
89 if (mesh->is_serial())
90 // If our mesh is already serial then we will keep it that way to avoid doing communication
91 mesh->allow_remote_element_removal(false);
92 if (!mesh->is_prepared())
93 mesh->prepare_for_use();
94
95 std::vector<std::tuple<dof_id_type, unsigned short int, boundary_id_type>> sideList =
96 mesh->get_boundary_info().build_active_side_list();
97
98 // If on one process a sideset element is semilocal, then any semilocal active neighbors might be
99 // marked for refinement. However, if on another process that sideset element is *not* semilocal,
100 // then obviously no neighboring elements can be marked for refinement. With no communication this
101 // can lead to elements on the former process being flagged for refinement while on the latter
102 // they will be marked to do nothing. This is a violation of parallel consistency, so we make sure
103 // to communicate neighbors that need to be refined
104 std::vector<dof_id_type> neighbors_to_refine;
105
106 for (std::size_t i = 0; i < boundary_ids.size(); i++)
107 {
108 if (refinement[i] > 0 && refinement[i] > ref_step)
109 {
110 for (std::tuple<dof_id_type, unsigned short int, boundary_id_type> tuple : sideList)
111 {
112 if (std::get<2>(tuple) == boundary_ids[i])
113 {
114 Elem * elem = mesh->elem_ptr(std::get<0>(tuple));
115 if (_boundary_side[i] == "primary" || _boundary_side[i] == "both")
116 elem->set_refinement_flag(Elem::REFINE);
117 if (_boundary_side[i] == "secondary" || _boundary_side[i] == "both")
118 {
119 auto neighbor = elem->neighbor_ptr(std::get<1>(tuple));
120 // when we have multiple domains, domains will only refine the elements that they own
121 // since there can be instances where there is no neighbor, this null_ptr check is
122 // necessary
123 if (neighbor)
124 {
125 if (neighbor->active())
126 {
127 neighbor->set_refinement_flag(Elem::REFINE);
128 if (!mesh->is_serial())
129 neighbors_to_refine.push_back(neighbor->id());
130 }
131 else
132 {
133 std::vector<Elem *> family_tree;
134 neighbor->active_family_tree_by_neighbor(family_tree, elem);
135 for (auto child_elem : family_tree)
136 {
137 child_elem->set_refinement_flag(Elem::REFINE);
138 if (!mesh->is_serial())
139 neighbors_to_refine.push_back(child_elem->id());
140 }
141 }
142 }
143 }
144 }
145 }
146 }
147 }
148
149 if (!mesh->is_serial())
150 {
151 mesh->comm().allgather(neighbors_to_refine);
152 std::sort(neighbors_to_refine.begin(), neighbors_to_refine.end());
153 auto new_last = std::unique(neighbors_to_refine.begin(), neighbors_to_refine.end());
154 neighbors_to_refine.erase(new_last, neighbors_to_refine.end());
155 for (const auto id : neighbors_to_refine)
156 if (Elem * const elem = mesh->query_elem_ptr(id))
157 elem->set_refinement_flag(Elem::REFINE);
158 }
159
160 MeshRefinement refinedmesh(*mesh);
161 if (!_enable_neighbor_refinement)
162 refinedmesh.face_level_mismatch_limit() = 0;
163
164 // This calls prepare_for_use
165 refinedmesh.refine_elements();
166 mesh->allow_remote_element_removal(old_allow_remote_element_removal);
167 ref_step++;
168 return recursive_refine(boundary_ids, mesh, refinement, max, ref_step);
169}
