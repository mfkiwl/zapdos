34 : DGKernelBase(parameters),
35 NeighborMooseVariableInterface(
36 this, false, Moose::VarKindType::VAR_NONLINEAR, Moose::VarFieldType::VAR_FIELD_ARRAY),
37 _var(*mooseVariable()),
38 _u(_is_implicit ? _var.sln() : _var.slnOld()),
39 _grad_u(_is_implicit ? _var.gradSln() : _var.gradSlnOld()),

63 _work_vector(_count)
64{
65 addMooseVariableDependency(mooseVariable());
66
67 _save_in.resize(_save_in_strings.size());
68 _diag_save_in.resize(_diag_save_in_strings.size());
69
70 for (unsigned int i = 0; i < _save_in_strings.size(); i++)
71 {
72 MooseVariableFEBase * var = &_subproblem.getVariable(_tid,
73 _save_in_strings[i],
74 Moose::VarKindType::VAR_AUXILIARY,
75 Moose::VarFieldType::VAR_FIELD_ARRAY);
76
77 if (_sys.hasVariable(_save_in_strings[i]))
78 mooseError("Trying to use solution variable " + _save_in_strings[i] +
79 " as a save_in variable in " + name());
80
81 if (var->feType() != _var.feType())
82 paramError(
83 "save_in",
84 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
85 moose::internal::incompatVarMsg(*var, _var));
86
87 _save_in[i] = var;
88 var->sys().addVariableToZeroOnResidual(_save_in_strings[i]);
89 addMooseVariableDependency(var);
90 }
91
92 _has_save_in = _save_in.size() > 0;
93
94 for (unsigned int i = 0; i < _diag_save_in_strings.size(); i++)
95 {
96 MooseVariableFEBase * var = &_subproblem.getVariable(_tid,
97 _diag_save_in_strings[i],
98 Moose::VarKindType::VAR_AUXILIARY,
99 Moose::VarFieldType::VAR_FIELD_ARRAY);
100
101 if (_sys.hasVariable(_diag_save_in_strings[i]))
102 mooseError("Trying to use solution variable " + _diag_save_in_strings[i] +
103 " as a diag_save_in variable in " + name());
104
105 if (var->feType() != _var.feType())
106 paramError(
107 "diag_save_in",
108 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
109 moose::internal::incompatVarMsg(*var, _var));
110
111 _diag_save_in[i] = var;
112 var->sys().addVariableToZeroOnJacobian(_diag_save_in_strings[i]);
113 addMooseVariableDependency(var);
114 }
115

117}

119void
120ArrayDGKernel::computeElemNeighResidual(Moose::DGResidualType type)
121{
122 bool is_elem;
123 if (type == Moose::Element)
124 is_elem = true;
125 else
126 is_elem = false;
127
128 const ArrayVariableTestValue & test_space = is_elem ? _test : _test_neighbor;
129
130 if (is_elem)
131 prepareVectorTag(_assembly, _var.number());
132 else
133 prepareVectorTagNeighbor(_assembly, _var.number());
134
135 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
136 {
137 initQpResidual(type);
138 for (_i = 0; _i < test_space.size(); _i++)
139 {
140 _work_vector.setZero();
141 computeQpResidual(type, _work_vector);
142 mooseAssert(_work_vector.size() == _count,
143 "Size of local residual is not equal to the number of array variable compoments");
144 _work_vector *= _JxW[_qp] * _coord[_qp];
145 _assembly.saveLocalArrayResidual(_local_re, _i, test_space.size(), _work_vector);
146 }
147 }
148
149 accumulateTaggedLocalResidual();
150
151 if (_has_save_in)
152 {
153 Threads::spin_mutex::scoped_lock lock(_resid_vars_mutex);
154 for (const auto & var : _save_in)
155 {
156 auto * avar = dynamic_cast<ArrayMooseVariable *>(var);
157 if (!avar)
158 mooseError("Save-in variable for an array kernel must be an array variable");
159
160 if (is_elem)
161 avar->addSolution(_local_re);
162 else
163 avar->addSolutionNeighbor(_local_re);
164 }
165 }
166}

168void
169ArrayDGKernel::computeElemNeighJacobian(Moose::DGJacobianType type)
170{
171 const ArrayVariableTestValue & test_space =
172 (type == Moose::ElementElement || type == Moose::ElementNeighbor) ? _test : _test_neighbor;
173 const ArrayVariableTestValue & loc_phi =
174 (type == Moose::ElementElement || type == Moose::NeighborElement) ? _phi : _phi_neighbor;
175
176 if (type == Moose::ElementElement)
177 prepareMatrixTag(_assembly, _var.number(), _var.number());
178 else
179 prepareMatrixTagNeighbor(_assembly, _var.number(), _var.number(), type);
180
181 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
182 {
183 initQpJacobian(type);
184 for (_i = 0; _i < test_space.size(); _i++)
185 for (_j = 0; _j < loc_phi.size(); _j++)
186 {
187 RealEigenVector v = _JxW[_qp] * _coord[_qp] * computeQpJacobian(type);
188 _assembly.saveDiagLocalArrayJacobian(
189 _local_ke, _i, test_space.size(), _j, loc_phi.size(), _var.number(), v);
190 }
191 }
192
193 accumulateTaggedLocalMatrix();
194
195 if (_has_diag_save_in && (type == Moose::ElementElement || type == Moose::NeighborNeighbor))
196 {
197 DenseVector<Number> diag = _assembly.getJacobianDiagonal(_local_ke);
198 Threads::spin_mutex::scoped_lock lock(_jacoby_vars_mutex);
199 for (const auto & var : _diag_save_in)
200 {
201 auto * avar = dynamic_cast<ArrayMooseVariable *>(var);
202 if (!avar)
203 mooseError("Save-in variable for an array kernel must be an array variable");
204
205 if (type == Moose::ElementElement)
206 avar->addSolution(diag);
207 else
208 avar->addSolutionNeighbor(diag);
209 }
210 }
211}

241void
242ArrayDGKernel::computeOffDiagElemNeighJacobian(Moose::DGJacobianType type,
243 const MooseVariableFEBase & jvar)
244{
245 const ArrayVariableTestValue & test_space =
246 (type == Moose::ElementElement || type == Moose::ElementNeighbor) ? _test : _test_neighbor;
247
248 if (type == Moose::ElementElement)
249 prepareMatrixTag(_assembly, _var.number(), jvar.number());
250 else
251 prepareMatrixTagNeighbor(_assembly, _var.number(), jvar.number(), type);
252
253 if (_local_ke.n() == 0 || _local_ke.m() == 0)
254 return;
255
256 if (jvar.fieldType() == Moose::VarFieldType::VAR_FIELD_STANDARD)
257 {
258 const auto & jv0 = static_cast<const MooseVariable &>(jvar);
259 const VariableTestValue & loc_phi =
260 (type == Moose::ElementElement || type == Moose::NeighborElement) ? jv0.phiFace()
261 : jv0.phiFaceNeighbor();
262
263 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
264 {
265 initQpOffDiagJacobian(type, jvar);
266 for (_i = 0; _i < test_space.size(); _i++)
267 for (_j = 0; _j < loc_phi.size(); _j++)
268 {
269 RealEigenMatrix v = _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobian(type, jvar);
270 _assembly.saveFullLocalArrayJacobian(_local_ke,
271 _i,
272 test_space.size(),
273 _j,
274 loc_phi.size(),
275 _var.number(),
276 jvar.number(),
277 v);
278 }
279 }
280 }
281 else if (jvar.fieldType() == Moose::VarFieldType::VAR_FIELD_ARRAY)
282 {
283 const auto & jv1 = static_cast<const ArrayMooseVariable &>(jvar);
284 const ArrayVariableTestValue & loc_phi =
285 (type == Moose::ElementElement || type == Moose::NeighborElement) ? jv1.phiFace()
286 : jv1.phiFaceNeighbor();
287
288 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
289 {
290 initQpOffDiagJacobian(type, jvar);
291 for (_i = 0; _i < test_space.size(); _i++)
292 for (_j = 0; _j < loc_phi.size(); _j++)
293 {
294 RealEigenMatrix v = _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobian(type, jvar);
295 _assembly.saveFullLocalArrayJacobian(_local_ke,
296 _i,
297 test_space.size(),
298 _j,
299 loc_phi.size(),
300 _var.number(),
301 jvar.number(),
302 v);
303 }
304 }
305 }
306 else
307 mooseError("Vector variable cannot be coupled into array DG kernel currently");
308
309 accumulateTaggedLocalMatrix();
310
311 if (_has_diag_save_in && (type == Moose::ElementElement || type == Moose::NeighborNeighbor) &&
312 _var.number() == jvar.number())
313 {
314 DenseVector<Number> diag = _assembly.getJacobianDiagonal(_local_ke);
315 Threads::spin_mutex::scoped_lock lock(_jacoby_vars_mutex);
316 for (const auto & var : _diag_save_in)
317 {
318 auto * avar = dynamic_cast<ArrayMooseVariable *>(var);
319 if (!avar)
320 mooseError("Save-in variable for an array kernel must be an array variable");
321
322 if (type == Moose::ElementElement)
323 avar->addSolution(diag);
324 else
325 avar->addSolutionNeighbor(diag);
326 }
327 }
328}
