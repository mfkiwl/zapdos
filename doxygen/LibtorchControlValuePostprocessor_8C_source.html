38void
39LibtorchControlValuePostprocessor::initialSetup()
40{
41 GeneralPostprocessor::initialSetup();
42
43 _libtorch_nn_control = dynamic_cast<LibtorchNeuralNetControl *>(
44 _fe_problem.getControlWarehouse()
45 .getActiveObject(getParam<std::string>("control_name"))
46 .get());
47 if (!_libtorch_nn_control)
48 paramError("control_name",
49 "The supplied control object is not derived from LibtorchNeuralNetControl!");
50
51 if (_libtorch_nn_control->numberOfControlSignals() <= _signal_index)
52 paramError("signal_index",
53 "The given control object only has ",
54 _libtorch_nn_control->numberOfControlSignals(),
55 " control signals!");
56}
