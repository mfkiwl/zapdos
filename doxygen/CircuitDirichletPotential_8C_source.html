43CircuitDirichletPotential::CircuitDirichletPotential(const InputParameters & parameters)
44 : ADNodalBC(parameters),
45 _current(getPostprocessorValue("current")),
46 _surface_potential(getFunction("surface_potential")),
47 _surface(getParam<std::string>("surface")),
48 _resist(getParam<Real>("resist")),
49 _coulomb_charge(1.6e-19),
50 _N_A(6.02e23),
51 _potential_units(getParam<std::string>("potential_units")),
52 _r_units(1. / getParam<Real>("position_units")),
53 _convert_moles(getParam<bool>("use_moles")),
54 _A(getParam<Real>("A"))
55{
56 if (_surface.compare("anode") == 0)
57 _current_sign = -1.;
58 else if (_surface.compare("cathode") == 0)
59 _current_sign = 1.;
60 if (_potential_units.compare("V") == 0)
61 _voltage_scaling = 1.;
62 else if (_potential_units.compare("kV") == 0)
63 _voltage_scaling = 1000;
64}

66ADReal
67CircuitDirichletPotential::computeQpResidual()
68{
69 if (_convert_moles)
70 return _surface_potential.value(_t, *_current_node) - _u +
71 _current_sign * _current / std::pow(_r_units, 2.) * _resist / _voltage_scaling *
72 _coulomb_charge * _A * _N_A;
73 else
74 return _surface_potential.value(_t, *_current_node) - _u +
75 _current_sign * _current / std::pow(_r_units, 2.) * _resist / _voltage_scaling *
76 _coulomb_charge * _A;
77}
