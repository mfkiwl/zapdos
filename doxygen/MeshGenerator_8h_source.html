418template <typename T, typename... Args>
419T &
420MeshGenerator::declareMeshProperty(const std::string & data_name, Args &&... args)
421{
422 if (!_app.constructingMeshGenerators())
423 mooseError("Can only call declareMeshProperty() during MeshGenerator construction");
424
425 // We sort construction ordering so that we _must_ declare before retrieving
426 if (hasMeshProperty(data_name))
427 mooseError("While declaring mesh property '",
428 data_name,
429 "' with type '",
430 MooseUtils::prettyCppType<T>(),
431 "',\nsaid property has already been declared with type '",
432 setMeshPropertyHelper(data_name).type(),
433 "'");
434
435 const auto full_name = meshPropertyName(data_name);
436 auto new_T_value =
437 std::make_unique<RestartableData<T>>(full_name, nullptr, std::forward<Args>(args)...);
438 auto value =
439 &_app.registerRestartableData(std::move(new_T_value), 0, false, MooseApp::MESH_META_DATA);
440 mooseAssert(value->declared(), "Should be declared");
441
442 RestartableData<T> * T_value = dynamic_cast<RestartableData<T> *>(value);
443 mooseAssert(T_value, "Bad cast");
444
445 return T_value->set();
446}

448template <typename T, typename... Args>
449T &
450MeshGenerator::setMeshProperty(const std::string & data_name, Args &&... args)
451{
452 if (_app.actionWarehouse().getCurrentTaskName() != "execute_mesh_generators")
453 mooseError("Updating mesh meta data with setMeshProperty() can only be called during "
454 "MeshGenerator::generate()");
455
456 if (!hasMeshProperty(data_name))
457 mooseError("Failed to get the mesh property '", data_name, "'");
458 RestartableDataValue * value = &setMeshPropertyHelper(data_name);
459 RestartableData<T> * T_value = dynamic_cast<RestartableData<T> *>(value);
460 if (!T_value)
461 mooseError("While retrieving mesh property '",
462 data_name,
463 "' with type '",
464 MooseUtils::prettyCppType<T>(),
465 "',\nthe property was found with type '",
466 value->type(),
467 "'");
468
469 // Set the value if someone provided arguments to set it to
470 if constexpr (sizeof...(args) > 0)
471 T_value->set() = T(std::forward<Args>(args)...);
472
473 return T_value->set();
474}
