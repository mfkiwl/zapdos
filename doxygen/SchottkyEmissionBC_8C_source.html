74ADReal
75SchottkyEmissionBC::computeQpResidual()
76{
77 ADReal dPhi;
78 Real kB;
79 ADReal jRD;
80 ADReal jSE;
81 ADReal F;
82 Real _relaxation_Expr;
83
84 _v_thermal =
85 std::sqrt(8 * _e[_qp] * 2.0 / 3 * std::exp(_mean_en[_qp] - _u[_qp]) / (M_PI * _massem[_qp]));
86
87 if (_normals[_qp] * -1.0 * -_grad_potential[_qp] > 0.0)
88 {
89 _a = 1.0;
90 return 0;
91 }
92 else
93 {
94 _a = 0.0;
95
96 _ion_flux = _sgnip[_qp] * _muip[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_ip[_qp]) -
97 _Dip[_qp] * std::exp(_ip[_qp]) * _grad_ip[_qp] * _r_units;
98
99 // Schottky emission
100 // je = AR * T^2 * exp(-(wf - dPhi) / (kB * T))
101 // dPhi = _dPhi_over_F * sqrt(F) // eV
102
103 F = -(1 - _a) * _field_enhancement[_qp] * _normals[_qp] * _grad_potential[_qp] * _r_units;
104
105 kB = 8.617385E-5; // eV/K
106 dPhi = _dPhi_over_F * std::sqrt(F);
107
108 jRD = _Richardson_coefficient[_qp] * std::pow(_cathode_temperature[_qp], 2) *
109 std::exp(-(_work_function[_qp] - dPhi) / (kB * _cathode_temperature[_qp]));
110 jSE = _e[_qp] * 6.02E23 * _se_coeff[_qp] * _ion_flux * _normals[_qp];
111
112 if (_relax)
113 {
114 _relaxation_Expr = std::tanh(_t / _tau);
115 }
116 else
117 {
118 _relaxation_Expr = 1.0;
119 }
120
121 return _relaxation_Expr * _test[_i][_qp] * _r_units * 2. / (1. + _r) * (1 - _a) * (-jRD - jSE) /
122 (_e[_qp] * 6.02E23);
123 }
124}
