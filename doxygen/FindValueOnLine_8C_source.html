73void
74FindValueOnLine::execute()
75{
76 Real s;
77 Real s_left = 0.0;
78 Real left = getValueAtPoint(_start_point);
79 Real s_right = 1.0;
80 Real right = getValueAtPoint(_end_point);
81
86 bool left_to_right = left < right;
87 // Initial bounds check
88 if ((left_to_right && _target < left) || (!left_to_right && _target < right))
89 {
90 if (_error_if_not_found)
91 {
92 mooseError("Target value \"",
93 _target,
94 "\" is less than the minimum sampled value \"",
95 std::min(left, right),
96 "\"");
97 }
98 else
99 {
100 _position = _default_value;
101 return;
102 }
103 }
104 if ((left_to_right && _target > right) || (!left_to_right && _target > left))
105 {
106 if (_error_if_not_found)
107 {
108 mooseError("Target value \"",
109 _target,
110 "\" is greater than the maximum sampled value \"",
111 std::max(left, right),
112 "\"");
113 }
114 else
115 {
116 _position = _default_value;
117 return;
118 }
119 }
120
121 bool found_it = false;
122 Real value = 0;
123 for (unsigned int i = 0; i < _depth; ++i)
124 {
125 // find midpoint
126 s = (s_left + s_right) / 2.0;
127 Point p = s * (_end_point - _start_point) + _start_point;
128
129 // sample value
130 value = getValueAtPoint(p);
131
132 // have we hit the target value yet?
133 if (MooseUtils::absoluteFuzzyEqual(value, _target, _tol))
134 {
135 found_it = true;
136 break;
137 }
138
139 // bisect
140 if ((left_to_right && _target < value) || (!left_to_right && _target > value))
141 // to the left
142 s_right = s;
143 else
144 // to the right
145 s_left = s;
146 }
147
148 // Return error if target value (within tol) was not found within depth bisections
149 if (!found_it)
150 mooseError("Target value \"",
151 std::setprecision(10),
152 _target,
153 "\" not found on line within tolerance, last sample: ",
154 value,
155 ".");
156
157 _position = s * _length;
158}

160Real
161FindValueOnLine::getValueAtPoint(const Point & p)
162{
163 const Elem * elem = (*_pl)(p);
164
165 processor_id_type elem_proc_id = elem ? elem->processor_id() : DofObject::invalid_processor_id;
166 _communicator.min(elem_proc_id);
167
168 if (elem_proc_id == DofObject::invalid_processor_id)
169 {
170 // there is no element
171 mooseError("No element found at the current search point. Please make sure the sampling line "
172 "stays inside the mesh completely.");
173 }
174
175 Real value = 0;
176
177 if (elem)
178 {
179 if (elem->processor_id() == processor_id())
180 {
181 // element is local
182 _point_vec[0] = p;
183 _subproblem.reinitElemPhys(elem, _point_vec, 0);
184 value = _coupled_var.sln()[0];
185 }
186 }
187
188 // broadcast value
189 _communicator.broadcast(value, elem_proc_id);
190 return value;
191}
