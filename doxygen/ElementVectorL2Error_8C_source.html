48Real
49ElementVectorL2Error::computeQpIntegral()
50{
51 RealVectorValue sol_val(0.0, 0.0, 0.0);
52 RealVectorValue func_val(0.0, 0.0, 0.0);
53
54 sol_val(0) = _u[_qp]; // required variable
55 func_val(0) = _funcx.value(_t, _q_point[_qp]); // required function
56
57 sol_val(1) = _v[_qp];
58 sol_val(2) = _w[_qp];
59
60 func_val(1) = _funcy.value(_t, _q_point[_qp]);
61 func_val(2) = _funcz.value(_t, _q_point[_qp]);
62
63 return (sol_val - func_val).norm_sq(); // dot product of difference vector
64}
