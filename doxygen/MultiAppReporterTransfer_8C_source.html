16MultiAppReporterTransfer::validParams()
17{
18 InputParameters params = MultiAppTransfer::validParams();
19 params += ReporterTransferInterface::validParams();
20 params.addClassDescription("Transfers reporter data between two applications.");
21 params.addRequiredParam<std::vector<ReporterName>>(
22 "from_reporters",
23 "List of the reporter names (object_name/value_name) to transfer the value from.");
24 params.addRequiredParam<std::vector<ReporterName>>(
25 "to_reporters",
26 "List of the reporter names (object_name/value_name) to transfer the value to.");
27 params.addParam<unsigned int>(
28 "subapp_index",
29 std::numeric_limits<unsigned int>::max(),
30 "The MultiApp object sub-application index to use when transferring to/from the "
31 "sub-application. If unset and transferring to the sub-applications then all "
32 "sub-applications will receive data. The value must be set when transferring from a "
33 "sub-application.");
34 return params;
35}

37MultiAppReporterTransfer::MultiAppReporterTransfer(const InputParameters & parameters)
38 : MultiAppTransfer(parameters),
39 ReporterTransferInterface(this),
40 _from_reporter_names(getParam<std::vector<ReporterName>>("from_reporters")),
41 _to_reporter_names(getParam<std::vector<ReporterName>>("to_reporters")),
42 _subapp_index(getParam<unsigned int>("subapp_index"))
43{
44 if (_from_reporter_names.size() != _to_reporter_names.size())
45 paramError("to_reporters", "from_reporters and to_reporters must be the same size.");
46
47 if (_directions.size() > 1)
48 paramError("direction", "This transfer only supports a single direction.");
49
50 if (isParamValid("to_multi_app") && isParamValid("from_multi_app") &&
51 _subapp_index != std::numeric_limits<unsigned int>::max())
52 paramError("subapp_index",
53 "The subapp_index parameter is not supported for transfers between two multiapps");
54
55 if (hasFromMultiApp())
56 {
57 const auto multi_app = getFromMultiApp();
58 // Errors for sub app index.
59 if (_subapp_index != std::numeric_limits<unsigned int>::max() &&
60 _subapp_index >= multi_app->numGlobalApps())
61 paramError(
62 "subapp_index",
63 "The supplied sub-application index is greater than the number of sub-applications.");
64 else if (_directions.contains(FROM_MULTIAPP) &&
65 _subapp_index == std::numeric_limits<unsigned int>::max() &&
66 multi_app->numGlobalApps() > 1)
67 paramError("from_multi_app",
68 "subapp_index must be provided when more than one subapp is present.");
69 }
70 else if (hasToMultiApp())
71 {
72 // Errors for sub app index.
73 if (_subapp_index != std::numeric_limits<unsigned int>::max() &&
74 _subapp_index >= getToMultiApp()->numGlobalApps())
75 paramError(
76 "subapp_index",
77 "The supplied sub-application index is greater than the number of sub-applications.");
78 }
79}

81void
82MultiAppReporterTransfer::initialSetup()
83{
84 MultiAppTransfer::initialSetup();
85
86 // We need to get a reference to the data now so we can tell ReporterData
87 // that we consume a replicated version.
88 // Find proper FEProblem
89 FEProblemBase * problem_ptr = nullptr;
90 if (_directions.contains(TO_MULTIAPP))
91 problem_ptr = &getToMultiApp()->problemBase();
92 else if (_subapp_index == std::numeric_limits<unsigned int>::max() &&
93 getFromMultiApp()->hasLocalApp(0))
94 problem_ptr = &getFromMultiApp()->appProblemBase(0);
95 else if (getFromMultiApp()->hasLocalApp(_subapp_index))
96 problem_ptr = &getFromMultiApp()->appProblemBase(_subapp_index);
97
98 // Tell ReporterData to consume with replicated
99 if (problem_ptr)
100 for (const auto & fn : _from_reporter_names)
101 addReporterTransferMode(fn, REPORTER_MODE_REPLICATED, *problem_ptr);
102}

104void
105MultiAppReporterTransfer::executeToMultiapp()
106{
107 if (!hasToMultiApp())
108 return;
109
110 std::vector<unsigned int> indices;
111 if (_subapp_index == std::numeric_limits<unsigned int>::max())
112 {
113 indices.resize(getToMultiApp()->numGlobalApps());
114 std::iota(indices.begin(), indices.end(), 0);
115 }
116 else
117 indices = {_subapp_index};
118
119 for (const auto & ind : indices)
120 if (getToMultiApp()->hasLocalApp(ind) &&
121 (!hasFromMultiApp() || getFromMultiApp()->hasLocalApp(ind)))
122 for (unsigned int n = 0; n < _from_reporter_names.size(); ++n)
123 transferReporter(_from_reporter_names[n],
124 _to_reporter_names[n],
125 hasFromMultiApp() ? getFromMultiApp()->appProblemBase(ind)
126 : getToMultiApp()->problemBase(),
127 getToMultiApp()->appProblemBase(ind));
128}

130void
131MultiAppReporterTransfer::executeFromMultiapp()
132{
133 if (!hasFromMultiApp())
134 return;
135
136 unsigned int ind = _subapp_index != std::numeric_limits<unsigned int>::max() ? _subapp_index : 0;
137 if (getFromMultiApp()->hasLocalApp(ind) &&
138 (!hasToMultiApp() || getToMultiApp()->hasLocalApp(ind)))
139 for (unsigned int n = 0; n < _from_reporter_names.size(); ++n)
140 transferReporter(_from_reporter_names[n],
141 _to_reporter_names[n],
142 getFromMultiApp()->appProblemBase(ind),
143 hasToMultiApp() ? getToMultiApp()->appProblemBase(ind)
144 : getFromMultiApp()->problemBase());
145}
