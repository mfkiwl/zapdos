305template <typename T>
306void
307ReporterContext<T>::finalize()
308{
309 // Automatic parallel operation to perform
310 ReporterContext::AutoOperation auto_operation = ReporterContext::AutoOperation::NONE;
311
312 // Set the default producer mode to ROOT
313 if (!_producer_enum.isValid())
314 _producer_enum.assign(REPORTER_MODE_ROOT);
315
316 // Determine auto parallel operation to perform
317 const auto & producer = _producer_enum; // convenience
318 for (const auto & pair : _state.getConsumers())
319 {
320 const ReporterMode consumer = pair.first;
321 const MooseObject * moose_object = pair.second;
322
323 // The following sets up the automatic operations and performs error checking for the various
324 // modes for the producer and consumer
325 //
326 // The data is correct and requires no operation for the following conditions (PRODUCER ->
327 // CONSUMER)
328 // ROOT -> ROOT
329 // REPLICATED -> REPLICATED
330 // DISTRIBUTED -> DISTRIBUTED
331 // REPLICATED -> ROOT
332
333 // Perform broadcast in the case
334 // ROOT -> REPLICATED
335 if (producer == REPORTER_MODE_ROOT && consumer == REPORTER_MODE_REPLICATED)
336 auto_operation = ReporterContext::AutoOperation::BROADCAST;
337
338 // The following are not support and create an error
339 // ROOT -> DISTRIBUTED
340 // REPLICATED -> DISTRIBUTED
341 // DISTRIBUTED -> ROOT
342 // DISTRIBUTED -> REPLICATED
343 else if ((producer == REPORTER_MODE_ROOT && consumer == REPORTER_MODE_DISTRIBUTED) ||
344 (producer == REPORTER_MODE_REPLICATED && consumer == REPORTER_MODE_DISTRIBUTED) ||
345 (producer == REPORTER_MODE_DISTRIBUTED && consumer == REPORTER_MODE_ROOT) ||
346 (producer == REPORTER_MODE_DISTRIBUTED && consumer == REPORTER_MODE_REPLICATED))
347 mooseError("The Reporter value \"",
348 name(),
349 "\" is being produced in ",
350 producer,
351 " mode, but the ",
352 moose_object->typeAndName(),
353 " is requesting to consume it in ",
354 consumer,
355 " mode, which is not supported.");
356 }
357
358 // Perform desired auto parallel operation
359 if (auto_operation == ReporterContext::AutoOperation::BROADCAST)
360 this->broadcast();
361}

532{

540 this->comm().scatter(_values, this->_state.value());
