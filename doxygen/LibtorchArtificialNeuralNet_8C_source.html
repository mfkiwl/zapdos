18LibtorchArtificialNeuralNet::LibtorchArtificialNeuralNet(
19 const std::string name,
20 const unsigned int num_inputs,
21 const unsigned int num_outputs,
22 const std::vector<unsigned int> & num_neurons_per_layer,
23 const std::vector<std::string> & activation_function)
24 : _name(name),
25 _num_inputs(num_inputs),
26 _num_outputs(num_outputs),
27 _num_neurons_per_layer(num_neurons_per_layer),
28 _activation_function(MultiMooseEnum("relu sigmoid elu gelu linear", "relu"))
29{
30 _activation_function = activation_function;
31
32 // Check if the number of activation functions matches the number of hidden layers
33 if ((_activation_function.size() != 1) &&
34 (_activation_function.size() != _num_neurons_per_layer.size()))
35 mooseError("The number of activation functions should be either one or the same as the number "
36 "of hidden layers");
37 constructNeuralNetwork();
38}

40LibtorchArtificialNeuralNet::LibtorchArtificialNeuralNet(
41 const Moose::LibtorchArtificialNeuralNet & nn)
42 : torch::nn::Module(),
43 _name(nn.name()),

45 _num_outputs(nn.numOutputs()),

59void
60LibtorchArtificialNeuralNet::constructNeuralNetwork()
61{
62 // Adding hidden layers
63 unsigned int inp_neurons = _num_inputs;
64 for (unsigned int i = 0; i < numHiddenLayers(); ++i)
65 {
66 std::unordered_map<std::string, unsigned int> parameters = {
67 {"inp_neurons", inp_neurons}, {"out_neurons", _num_neurons_per_layer[i]}};
68 addLayer("hidden_layer_" + std::to_string(i + 1), parameters);
69
70 // Necessary to retain double precision (and error-free runs)
71 _weights[i]->to(at::kDouble);
72 inp_neurons = _num_neurons_per_layer[i];
73 }
74 // Adding output layer
75 std::unordered_map<std::string, unsigned int> parameters = {{"inp_neurons", inp_neurons},
76 {"out_neurons", _num_outputs}};
77 addLayer("output_layer_", parameters);
78 _weights.back()->to(at::kDouble);
79}

81torch::Tensor
82LibtorchArtificialNeuralNet::forward(torch::Tensor & x)
83{
84 torch::Tensor output(x);
85 for (unsigned int i = 0; i < _weights.size() - 1; ++i)
86 {
87 std::string activation =
88 _activation_function.size() > 1 ? _activation_function[i] : _activation_function[0];
89 if (activation == "relu")
90 output = torch::relu(_weights[i]->forward(output));
91 else if (activation == "sigmoid")
92 output = torch::sigmoid(_weights[i]->forward(output));
93 else if (activation == "elu")
94 output = torch::elu(_weights[i]->forward(output));
95 else if (activation == "gelu")
96 output = torch::gelu(_weights[i]->forward(output));
97 else if (activation == "linear")
98 output = _weights[i]->forward(output);
99 }
100
101 output = _weights[_weights.size() - 1]->forward(output);
102
103 return output;
104}

106void
107LibtorchArtificialNeuralNet::addLayer(
108 const std::string & layer_name,
109 const std::unordered_map<std::string, unsigned int> & parameters)
110{
111 auto it = parameters.find("inp_neurons");
112 if (it == parameters.end())
113 ::mooseError("Number of input neurons not found during the construction of "
114 "LibtorchArtificialNeuralNet!");
115 unsigned int inp_neurons = it->second;
116
117 it = parameters.find("out_neurons");
118 if (it == parameters.end())
119 ::mooseError("Number of output neurons not found during the construction of "
120 "LibtorchArtificialNeuralNet!");
121 unsigned int out_neurons = it->second;
122
123 _weights.push_back(register_module(layer_name, torch::nn::Linear(inp_neurons, out_neurons)));
124}

127LibtorchArtificialNeuralNet::store(nlohmann::json & json) const
128{
129 const auto & named_params = this->named_parameters();
130 for (const auto & param_i : make_range(named_params.size()))
131 {
132 // We cast the parameters into a 1D vector
133 json[named_params[param_i].key()] = std::vector<Real>(
134 named_params[param_i].value().data_ptr<Real>(),
135 named_params[param_i].value().data_ptr<Real>() + named_params[param_i].value().numel());
136 }
