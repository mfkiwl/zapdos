14/*
15This acceleration scheme is used in paper
16DOI: https://doi.org/10.1116/1.587101, with a more
17detailed description in papr DOI:
18https://doi.org/10.1016/0009-2509(92)85133-V.
19
20This method assumes a periodic solution and uses
21that assumption to accelerate to the simulation.
22This is done by converting the periodic solution
23into a boundary problem in time (i.e. the solution
24at the beginning of the period must be equal to the
25solution at the end of the period). So, the current
26solution is increased by the different of the solution
27at the start and end of the cycle times a variable
28representative of the sensitivity of the problem. This
29sensitivity is calculated in a sub app.
30*/

61ADReal
62ShootMethodLog::computeQpResidual()
63{
64
65 ADReal _acceleration_growth = std::pow((1. - _sensitivity[_qp]), -1.);
66
67 if (_limit == 0.0)
68 {
69 _acceleration = _acceleration_growth *
70 (std::exp(_density_at_start_cycle[_qp]) - std::exp(_density_at_end_cycle[_qp]));
71 }
72 else
73 {
74 if (_acceleration_growth > _limit)
75 {
76 _acceleration =
77 _limit * (std::exp(_density_at_start_cycle[_qp]) - std::exp(_density_at_end_cycle[_qp]));
78 }
79 else
80 {
81 _acceleration = _acceleration_growth * (std::exp(_density_at_start_cycle[_qp]) -
82 std::exp(_density_at_end_cycle[_qp]));
83 }
84 }
85
86 return _test[_i][_qp] *
87 (std::exp(_u[_qp]) - std::exp(_density_at_start_cycle[_qp]) + _acceleration);
88}
