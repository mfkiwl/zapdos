156 if (_potential_units.compare("V") == 0)
157 _voltage_scaling = 1.;
158 else if (_potential_units.compare("kV") == 0)
159 _voltage_scaling = 1000;
160
161 std::vector<Real> actual_mean_energy;
162 std::vector<Real> mu;
163 std::vector<Real> diff;
164
165 std::string file_name = getParam<FileName>("property_tables_file");
166 MooseUtils::checkFileReadable(file_name);
167 const char * charPath = file_name.c_str();
168 std::ifstream myfile(charPath);
169 Real value;
170
171 if (myfile.is_open())
172 {
173 while (myfile >> value)
174 {
175 actual_mean_energy.push_back(value);
176 myfile >> value;
177 mu.push_back(value);
178 myfile >> value;
179 diff.push_back(value);
180 }
181 myfile.close();
182 }
183
184 else
185 mooseError("Unable to open file");
186
187 _mu_interpolation.setData(actual_mean_energy, mu);
188 _diff_interpolation.setData(actual_mean_energy, diff);
189}

191void
192GasElectronMoments::computeQpProperties()
193{
194 _massem[_qp] = 9.11e-31;
195 _massGas[_qp] = 4.0 * 1.66e-27;
196 _T_gas[_qp] = _user_T_gas;
197 _p_gas[_qp] = _user_p_gas[_qp];
198 _k_boltz[_qp] = 1.38e-23;
199 _Avogadro[_qp] = 6.0221409E23;
200 if (_use_moles)
201 _n_gas[_qp] = _p_gas[_qp] / (8.3145 * _T_gas[_qp]);
202 else
203 _n_gas[_qp] = _p_gas[_qp] / (_k_boltz[_qp] * _T_gas[_qp]);
204 Real _N_inverse = (_k_boltz[_qp] * _T_gas[_qp]) / _p_gas[_qp];
205
206 _se_coeff[_qp] = _user_se_coeff;
207 _work_function[_qp] = _user_work_function;
208 _field_enhancement[_qp] = _user_field_enhancement;
209
210 _Richardson_coefficient[_qp] = _user_Richardson_coefficient;
211 _cathode_temperature[_qp] = _user_cathode_temperature;
212
213 _se_energy[_qp] = 2. * 3. / 2.; // Emi uses 2 Volts coming off the wall (presumably for Te).
214 // Multiply by 3/2 to get mean_en
215 _e[_qp] = 1.6e-19;
216 _eps[_qp] = 8.85e-12;
217 _sgnem[_qp] = -1.;
218 _sgnmean_en[_qp] = -1.;
219 _diffpotential[_qp] = _eps[_qp];
220
221 // With the exception of temperature/energy (generally in eV), all properties are in standard SI
222 // units unless otherwise indicated
223
224 if (_pressure_dependent) // If the mobility and diff. does depend on pressure
225 {
226
227 if (_interp_trans_coeffs)
228 {
229 if (_ramp_trans_coeffs)
230 {
231 _muem[_qp].value() =
232 (std::tanh(_t / 1e-6) *
233 _mu_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) +
234 (1. - std::tanh(_t / 1e-6)) * .0352) *
235 _voltage_scaling * _time_units * _N_inverse;
236 _muem[_qp].derivatives() =
237 std::tanh(_t / 1e-6) *
238 _mu_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
239 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
240 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _voltage_scaling *
241 _time_units * _N_inverse;
242 _diffem[_qp].value() =
243 std::tanh(_t / 1e-6) *
244 _diff_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) +
245 (1. - std::tanh(_t / 1e-6)) * .30 * _time_units * _N_inverse;
246 _diffem[_qp].derivatives() = std::tanh(_t / 1e-6) *
247 _diff_interpolation.sampleDerivative(
248 std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
249 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
250 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) *
251 _time_units * _N_inverse;
252 }
253 else
254 {
255 _muem[_qp].value() =
256 _mu_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
257 _voltage_scaling * _time_units * _N_inverse;
258 _muem[_qp].derivatives() =
259 _mu_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
260 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
261 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _voltage_scaling *
262 _time_units * _N_inverse;
263 _diffem[_qp].value() =
264 _diff_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
265 _time_units * _N_inverse;
266 _diffem[_qp].derivatives() = _diff_interpolation.sampleDerivative(
267 std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
268 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
269 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) *
270 _time_units * _N_inverse;
271 }
272 }
273 else
274 {
275 _muem[_qp].value() = _user_muem * _voltage_scaling * _time_units * _N_inverse;
276 _muem[_qp].derivatives() = 0.0;
277 _diffem[_qp].value() = _user_diffem * _time_units * _N_inverse;
278 _diffem[_qp].derivatives() = 0.0;
279 }
280 }
281
282 else // If the mobility and diff. does not depend on pressure
283 {
284
285 if (_interp_trans_coeffs)
286 {
287 if (_ramp_trans_coeffs)
288 {
289 _muem[_qp].value() =
290 (std::tanh(_t / 1e-6) *
291 _mu_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) +
292 (1. - std::tanh(_t / 1e-6)) * .0352) *
293 _voltage_scaling * _time_units;
294 _muem[_qp].derivatives() =
295 std::tanh(_t / 1e-6) *
296 _mu_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
297 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
298 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _voltage_scaling * _time_units;
299 _diffem[_qp].value() =
300 std::tanh(_t / 1e-6) *
301 _diff_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) +
302 (1. - std::tanh(_t / 1e-6)) * .30 * _time_units;
303 _diffem[_qp].derivatives() = std::tanh(_t / 1e-6) *
304 _diff_interpolation.sampleDerivative(
305 std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
306 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
307 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) *
308 _time_units;
309 }
310 else
311 {
312 _muem[_qp].value() =
313 _mu_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
314 _voltage_scaling * _time_units;
315 _muem[_qp].derivatives() =
316 _mu_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
317 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
318 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _voltage_scaling * _time_units;
319 _diffem[_qp].value() =
320 _diff_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
321 _time_units;
322 _diffem[_qp].derivatives() = _diff_interpolation.sampleDerivative(
323 std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
324 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
325 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) *
326 _time_units;
327 }
328 }
329 else
330 {
331 _muem[_qp].value() = _user_muem * _voltage_scaling * _time_units;
332 _muem[_qp].derivatives() = 0.0;
333 _diffem[_qp].value() = _user_diffem * _time_units;
334 _diffem[_qp].derivatives() = 0.0;
335 }
336 }
337
338 if (_use_moles)
339 {
340 _N_A[_qp] = 6.02e23;
341 }
342 else
343 {
344 _N_A[_qp] = 1.0;
345 }
346
347 _Ar[_qp] = 1.01e5 / (300 * 1.38e-23);
348 _Eiz[_qp] = 24.58;
349 _Eex[_qp] = 19.80;
350
351 // From Hagelaar: The below approximations can be derived assumption Maxwell EEDF, const
352 // momentum-transfer frequency, and constant kinetic pressure.
353 _mumean_en[_qp].value() = 5.0 / 3.0 * _muem[_qp].value();
354 _diffmean_en[_qp].value() = 5.0 / 3.0 * _diffem[_qp].value();
355 if (_interp_trans_coeffs)
356 {
357 _mumean_en[_qp].derivatives() = 5.0 / 3.0 * _muem[_qp].derivatives();
358 _diffmean_en[_qp].derivatives() = 5.0 / 3.0 * _diffem[_qp].derivatives();
359 }
360 else
361 {
362 _mumean_en[_qp].derivatives() = 0.0;
363 _diffmean_en[_qp].derivatives() = 0.0;
364 }
365
366 // Might needed to change
367 _rate_coeff_elastic[_qp] = 1e-13;
368
369 _TemVolts[_qp] = 2. / 3. * std::exp(_mean_en[_qp] - _em[_qp]);
370 _Tem[_qp] = _e[_qp] * _TemVolts[_qp] / _k_boltz[_qp];
371
372 _kiz[_qp] = 2.34e-14 * std::pow(_TemVolts[_qp], .59) * std::exp(-17.44 / _TemVolts[_qp]);
373 _kex[_qp] = 2.48e-14 * std::pow(_TemVolts[_qp], .33) * std::exp(-12.78 / _TemVolts[_qp]);
374 _kel[_qp].value() = 1e-13; // Approximate elastic rate coefficient
375 _kel[_qp].derivatives() = 0.;
376 if (_use_moles)
377 {
378 _kiz[_qp] = _kiz[_qp] * _N_A[_qp];
379 _kex[_qp] = _kex[_qp] * _N_A[_qp];
380 _kel[_qp] = _kel[_qp] * _N_A[_qp];
381 }
382}
