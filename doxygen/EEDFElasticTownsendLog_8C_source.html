61Real
62EEDFElasticTownsendLog::computeQpResidual()
63{
64 Real electron_flux_mag = (std::exp(_em[_qp]) * (-_muem[_qp] * -_grad_potential[_qp] * _r_units -
65 _diffem[_qp] * _grad_em[_qp] * _r_units))
66 .norm();
67 Real Eel = -3.0 * _massem / _massGas[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]);
68 Real el_term = _alpha[_qp] * electron_flux_mag * Eel;
69
70 return -_test[_i][_qp] * el_term * std::exp(_target[_qp]);
71}

73Real
74EEDFElasticTownsendLog::computeQpJacobian()
75{
76 Real d_actual_mean_en_d_mean_en = std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
77 Real d_el_d_mean_en = _d_el_d_actual_mean_en[_qp] * d_actual_mean_en_d_mean_en;
78 Real d_muem_d_mean_en = _d_muem_d_actual_mean_en[_qp] * d_actual_mean_en_d_mean_en;
79 Real d_diffem_d_mean_en = _d_diffem_d_actual_mean_en[_qp] * d_actual_mean_en_d_mean_en;
80
81 RealVectorValue electron_flux =
82 std::exp(_em[_qp]) *
83 (-_muem[_qp] * -_grad_potential[_qp] * _r_units - _diffem[_qp] * _grad_em[_qp] * _r_units);
84 RealVectorValue d_electron_flux_d_mean_en =
85 std::exp(_em[_qp]) * (-d_muem_d_mean_en * -_grad_potential[_qp] * _r_units -
86 d_diffem_d_mean_en * _grad_em[_qp] * _r_units);
87 Real electron_flux_mag = electron_flux.norm();
88 Real d_electron_flux_mag_d_mean_en = electron_flux * d_electron_flux_d_mean_en /
89 (electron_flux_mag + std::numeric_limits<double>::epsilon());
90
91 Real Eel = -3.0 * _massem / _massGas[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]);
92 Real d_Eel_d_mean_en =
93 -3.0 * _massem / _massGas[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
94 Real d_el_term_d_mean_en =
95 (electron_flux_mag * d_el_d_mean_en + _alpha[_qp] * d_electron_flux_mag_d_mean_en) * Eel +
96 electron_flux_mag * _alpha[_qp] * d_Eel_d_mean_en;
97
98 return -_test[_i][_qp] * d_el_term_d_mean_en * std::exp(_target[_qp]);
99}

101Real
102EEDFElasticTownsendLog::computeQpOffDiagJacobian(unsigned int jvar)
103{
104 Real d_actual_mean_en_d_em = -std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
105 Real d_el_d_em = _d_el_d_actual_mean_en[_qp] * d_actual_mean_en_d_em;
106 Real d_muem_d_em = _d_muem_d_actual_mean_en[_qp] * d_actual_mean_en_d_em;
107 Real d_diffem_d_em = _d_diffem_d_actual_mean_en[_qp] * d_actual_mean_en_d_em;
108
109 RealVectorValue electron_flux =
110 std::exp(_em[_qp]) *
111 (-_muem[_qp] * -_grad_potential[_qp] * _r_units - _diffem[_qp] * _grad_em[_qp] * _r_units);
112 RealVectorValue d_electron_flux_d_potential =
113 -_muem[_qp] * -_grad_phi[_j][_qp] * _r_units * std::exp(_em[_qp]);
114 RealVectorValue d_electron_flux_d_em =
115 std::exp(_em[_qp]) * (-d_muem_d_em * -_grad_potential[_qp] * _r_units -
116 _muem[_qp] * -_grad_potential[_qp] * _r_units * _phi[_j][_qp] -
117 d_diffem_d_em * _grad_em[_qp] * _r_units -
118 _diffem[_qp] * _phi[_j][_qp] * _grad_em[_qp] * _r_units -
119 _diffem[_qp] * _grad_phi[_j][_qp] * _r_units);
120 Real electron_flux_mag = electron_flux.norm();
121 Real d_electron_flux_mag_d_potential =
122 electron_flux * d_electron_flux_d_potential /
123 (electron_flux_mag + std::numeric_limits<double>::epsilon());
124 Real d_electron_flux_mag_d_em = electron_flux * d_electron_flux_d_em /
125 (electron_flux_mag + std::numeric_limits<double>::epsilon());
126
127 Real Eel = -3.0 * _massem / _massGas[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]);
128 Real d_Eel_d_em =
129 -3.0 * _massem / _massGas[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]) * -_phi[_j][_qp];
130 Real d_Eel_d_potential = 0.0;
131 Real d_el_term_d_em =
132 (electron_flux_mag * d_el_d_em + _alpha[_qp] * d_electron_flux_mag_d_em) * Eel +
133 electron_flux_mag * _alpha[_qp] * d_Eel_d_em;
134 Real d_el_term_d_potential = (_alpha[_qp] * d_electron_flux_mag_d_potential) * Eel +
135 electron_flux_mag * _alpha[_qp] * d_Eel_d_potential;
136
137 if (jvar == _potential_id)
138 return -_test[_i][_qp] * d_el_term_d_potential * std::exp(_target[_qp]);
139 else if (jvar == _em_id)
140 return -_test[_i][_qp] * d_el_term_d_em * std::exp(_target[_qp]);
141 else if (jvar == _target_id)
142 return -_test[_i][_qp] * _alpha[_qp] * std::exp(_target[_qp]) * electron_flux_mag *
143 _phi[_j][_qp];
144
145 else
146 return 0.0;
147}
