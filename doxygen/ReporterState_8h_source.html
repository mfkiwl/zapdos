158template <typename T>
159T &
160ReporterState<T>::value(const std::size_t time_index)
161{
162 // Initialize the data; the first entry is the "current" data
163 if (this->get().empty())
164 this->set().resize(1);
165
166 // Initialize old, older, ... data
167 if (this->get().size() <= time_index)
168 this->set().resize(time_index + 1, this->get().back());
169
170 return *(std::next(this->set().begin(), time_index));
171}

173template <typename T>
174const T &
175ReporterState<T>::value(const std::size_t time_index) const
176{
177 if (this->get().size() <= time_index)
178 mooseError("The desired time index ",
179 time_index,
180 " does not exists for the '",
181 getReporterName(),
182 "' Reporter value, which contains ",
183 this->get().size(),
184 " old value(s). The getReporterValue method must be called with the desired time "
185 "index to be able to access data.");
186 return *(std::next(this->get().begin(), time_index));
187}

189template <typename T>
190void
191ReporterState<T>::copyValuesBack()
192{
193 std::list<T> & values = this->set();
194 for (typename std::list<T>::reverse_iterator iter = values.rbegin();
195 std::next(iter) != values.rend();
196 ++iter)
197 (*iter) = (*std::next(iter));
198}

202ReporterState<T>::storeInternal(std::ostream & stream)

204 // Store the container size
205 std::size_t size = this->get().size();

213template <typename T>
214void
215ReporterState<T>::loadInternal(std::istream & stream)
216{
217 // Read the container size
218 std::size_t size = 0;
219 dataLoad(stream, size, nullptr);
220
221 auto & values = this->set();
222
223 // If the current container is undersized, expand it to fit the loaded data
224 if (values.size() < size)
225 values.resize(size);
226
227 // Load each entry of the list directly into the storage
228 // Because we don't shrink the container if the stored size is smaller than
229 // our declared size, we have the odd iterator combo you see below
230 for (auto & val : as_range(values.begin(), std::next(values.begin(), size)))
231 loadHelper(stream, val, nullptr);
232}
