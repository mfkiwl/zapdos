46AddPeriodicRelativeNodalDifference::validParams()
47{
48 MooseEnum families(AddVariableAction::getNonlinearVariableFamilies());

51 InputParameters params = AddVariableAction::validParams();

80void
81AddPeriodicRelativeNodalDifference::act()
82{
83 MooseSharedPointer<Action> action;
84 MooseSharedPointer<MooseObjectAction> moose_object_action;
85
86 std::vector<NonlinearVariableName> var_log =
87 getParam<std::vector<NonlinearVariableName>>("periodic_variable_log");
88 std::vector<NonlinearVariableName> var =
89 getParam<std::vector<NonlinearVariableName>>("periodic_variable");
90
91 unsigned int number_var_log = var_log.size();
92 unsigned int number_var = var.size();
93
94 // The variable type for the aux variables
95 auto fe_type = AddVariableAction::feType(_pars);
96 auto type = AddVariableAction::variableType(fe_type);
97 auto var_params = _factory.getValidParams(type);
98 var_params.set<MooseEnum>("order") = "FIRST";
99 var_params.set<MooseEnum>("family") = "LAGRANGE";
100 var_params.set<std::vector<SubdomainName>>("block") =
101 getParam<std::vector<SubdomainName>>("block");
102
103 // Defineding the indexs needed for the periodic controller
104 _enable_start.resize(3 * (number_var_log + number_var));
105 _enable_end.resize(number_var_log + number_var);
106
107 for (unsigned int i = 0; i < number_var_log; i++)
108 {
109 std::string var_log_name = var_log[i];
110
111 _enable_start[3 * i] = "*::" + var_log_name + "_previous_cycle_sudo";
112 _enable_start[3 * i + 1] = "*::" + var_log_name + "_normalized";
113 _enable_start[3 * i + 2] = "*::" + var_log_name + "_previous_cycle_normalized";
114 }
115 for (unsigned int i = number_var_log; i < (number_var + number_var_log); i++)
116 {
117 std::string var_name = var[i - number_var_log];
118
119 _enable_start[3 * i] = "*::" + var_name + "_previous_cycle_sudo";
120 _enable_start[3 * i + 1] = "*::" + var_name + "_normalized";
121 _enable_start[3 * i + 2] = "*::" + var_name + "_previous_cycle_normalized";
122 }
123 for (unsigned int i = 0; i < number_var_log; i++)
124 {
125 std::string var_log_name = var_log[i];
126
127 _enable_end[i] = "*::" + var_log_name + "_previous_cycle";
128 }
129 for (unsigned int i = number_var_log; i < (number_var + number_var_log); i++)
130 {
131 std::string var_name = var[i - number_var_log];
132
133 _enable_end[i] = "*::" + var_name + "_previous_cycle";
134 }
135
136 _enable_start_start_time_index.resize(_enable_start.size());
137 _enable_start_end_time_index.resize(_enable_start.size());
138 _enable_end_start_time_index.resize(_enable_end.size());
139 _enable_end_end_time_index.resize(_enable_end.size());
140
141 // Adding Variables to store previous solutions and normalized solutions
142 // Note: A sudo previous solution is needed so that the '_previous_cycle'
143 // solutions are not overwritten too soon.
144 if (_current_task == "add_variable")
145 {
146 // Dummy variable for the denominator in QuotientAux
147 _problem->addAuxVariable(type, "Dummy_denominator", var_params);
148
149 for (unsigned int cur_num = 0; cur_num < number_var_log; cur_num++)
150 {
151 std::string var_log_name = var_log[cur_num];
152 _problem->addAuxVariable(type, var_log_name + "_previous_cycle", var_params);
153 _problem->addAuxVariable(type, var_log_name + "_previous_cycle_sudo", var_params);
154
155 _problem->addAuxVariable(type, var_log_name + "_normalized", var_params);
156 _problem->addAuxVariable(type, var_log_name + "_previous_cycle_normalized", var_params);
157 }
158 for (unsigned int cur_num = 0; cur_num < number_var; cur_num++)
159 {
160 std::string var_name = var[cur_num];
161 _problem->addAuxVariable(type, var_name + "_previous_cycle", var_params);
162 _problem->addAuxVariable(type, var_name + "_previous_cycle_sudo", var_params);
163
164 _problem->addAuxVariable(type, var_name + "_normalized", var_params);
165 _problem->addAuxVariable(type, var_name + "_previous_cycle_normalized", var_params);
166 }
167 }
168
169 // Adding IC for the previous solution to avoid a 'divided by zero' for
170 // the first cycle.
171 else if (_current_task == "add_ic")
172 {
173 for (unsigned int cur_num = 0; cur_num < number_var_log; cur_num++)
174 {
175 std::string var_log_name = var_log[cur_num];
176 Real initial = -21.;
177
178 addPerviousSolutionsIC(var_log_name + "_previous_cycle", initial);
179 addPerviousSolutionsIC(var_log_name + "_previous_cycle_sudo", initial);
180 }
181 for (unsigned int cur_num = 0; cur_num < number_var; cur_num++)
182 {
183 std::string var_name = var[cur_num];
184
185 Real initial = 0.0001;
186
187 addPerviousSolutionsIC(var_name + "_previous_cycle", initial);
188 addPerviousSolutionsIC(var_name + "_previous_cycle_sudo", initial);
189 }
190 }
191
192 else if (_current_task == "add_aux_kernel")
193 {
194 // Adding the Dummy kernel for the denominator in QuotientAux.
195 InputParameters params = _factory.getValidParams("ConstantAux");
196 params.set<AuxVariableName>("variable") = {"Dummy_denominator"};
197 params.set<Real>("value") = 1.;
198 params.set<std::vector<SubdomainName>>("block") = getParam<std::vector<SubdomainName>>("block");
199 _problem->addAuxKernel("ConstantAux", "Dummy_denominator", params);
200
201 // Adding Kernels to store previous solutions and to normalize solutions
202 for (unsigned int cur_num = 0; cur_num < number_var_log; cur_num++)
203 {
204 std::string var_log_name = var_log[cur_num];
205
206 addPerviousSolutionsKernels(var_log_name + "_previous_cycle_sudo", var_log_name);
207 addPerviousSolutionsKernels(var_log_name + "_previous_cycle",
208 var_log_name + "_previous_cycle_sudo");
209
210 addNormalizationKernels(
211 var_log_name + "_normalized", var_log_name, var_log_name + "_NodalAverage", true);
212 addNormalizationKernels(var_log_name + "_previous_cycle_normalized",
213 var_log_name + "_previous_cycle",
214 var_log_name + "_previous_cycle_NodalAverage",
215 true);
216 }
217
218 for (unsigned int cur_num = 0; cur_num < number_var; cur_num++)
219 {
220 std::string var_name = var[cur_num];
221
222 addPerviousSolutionsKernels(var_name + "_previous_cycle_sudo", var_name);
223 addPerviousSolutionsKernels(var_name + "_previous_cycle", var_name + "_previous_cycle_sudo");
224
225 addNormalizationKernels(
226 var_name + "_normalized", var_name, var_name + "_NodalAverage", false);
227 addNormalizationKernels(var_name + "_previous_cycle_normalized",
228 var_name + "_previous_cycle",
229 var_name + "_previous_cycle_NodalAverage",
230 false);
231 }
232 }
233
234 // Adding Postprocessor to calculate nodal average and difference
235 else if (_current_task == "add_postprocessor")
236 {
237 for (unsigned int cur_num = 0; cur_num < number_var_log; cur_num++)
238 {
239 std::string var_log_name = var_log[cur_num];
240
241 addAverageNodalPP(var_log_name, true);
242 addAverageNodalPP(var_log_name + "_previous_cycle", true);
243
244 addRelativePeriodicDiffPP(
245 var_log_name + "_normalized", var_log_name + "_previous_cycle_normalized", var_log_name);
246 }
247
248 for (unsigned int cur_num = 0; cur_num < number_var; cur_num++)
249 {
250 std::string var_name = var[cur_num];
251
252 addAverageNodalPP(var_name, false);
253 addAverageNodalPP(var_name + "_previous_cycle", false);
254
255 addRelativePeriodicDiffPP(
256 var_name + "_normalized", var_name + "_previous_cycle_normalized", var_name);
257 }
258 }
259
260 // Adding a periodic control, so Auxkernels only are on once par cycle
261 else if (_current_task == "add_control")
262 {
263 for (unsigned int i = 0; i < _num_controller_set; ++i)
264 {
265 for (MooseIndex(_enable_start) j = 0; j < _enable_start.size(); ++j)
266 {
267 _enable_start_start_time_index[j] = _start_time + _period * i;
268 _enable_start_end_time_index[j] = _start_time + _period * i + (_period * 0.0001);
269 }
270 for (MooseIndex(_enable_end) j = 0; j < _enable_end.size(); ++j)
271 {
272 _enable_end_start_time_index[j] =
273 (_start_time + _period) + _period * i + (_period * 0.0001);
274 _enable_end_end_time_index[j] = (_start_time + _period) + _period * i + (_period * 0.0002);
275 }
276
277 if (_enable_start.size() > 0)
278 {
279 std::string name_num = "PeriodicRelativeDifference_Enable_Begin_" + std::to_string(i);
280 bool first_controller;
281
282 if (i == 0)
283 {
284 first_controller = true;
285 }
286 else
287 {
288 first_controller = false;
289 }
290
291 AddTimePeriod(_enable_start,
292 _enable_start_start_time_index,
293 _enable_start_end_time_index,
294 name_num,
295 first_controller);
296 }
297
298 if (_enable_end.size() > 0)
299 {
300 std::string name_num = "PeriodicRelativeDifference_Enable_End_" + std::to_string(i);
301 bool first_controller;
302
303 if (i == 0)
304 {
305 first_controller = true;
306 }
307 else
308 {
309 first_controller = false;
310 }
311
312 AddTimePeriod(_enable_end,
313 _enable_end_start_time_index,
314 _enable_end_end_time_index,
315 name_num,
316 first_controller);
317 }
318 }
319 }
320}

366 }
367 else
368 {
369 InputParameters params = _factory.getValidParams("NormalizationAux");
370 params.set<AuxVariableName>("variable") = {variable_name};
371 params.set<std::vector<VariableName>>("source_variable") = {source};
372 params.set<PostprocessorName>("normalization") = {averaged};
373 params.set<bool>("enable") = false;
