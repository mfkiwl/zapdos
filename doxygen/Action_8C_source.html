86bool
87Action::addRelationshipManager(
88 Moose::RelationshipManagerType /*input_rm_type*/,
89 const InputParameters & moose_object_pars,
90 std::string rm_name,
91 Moose::RelationshipManagerType rm_type,
92 Moose::RelationshipManagerInputParameterCallback rm_input_parameter_func,
93 Moose::RMSystemType)
94{
95 // These need unique names
96 static unsigned int unique_object_id = 0;
97
98 auto new_name = moose_object_pars.get<std::string>("_moose_base") + '_' + name() + '_' + rm_name +
99 "_" + Moose::stringify(rm_type) + " " + std::to_string(unique_object_id);
100
101 auto rm_params = _factory.getValidParams(rm_name);
102 rm_params.set<Moose::RelationshipManagerType>("rm_type") = rm_type;
103
104 rm_params.set<std::string>("for_whom") = name();
105
106 // If there is a callback for setting the RM parameters let's use it
107 if (rm_input_parameter_func)
108 rm_input_parameter_func(moose_object_pars, rm_params);
109
110 rm_params.set<MooseMesh *>("mesh") = _mesh.get();
111
112 if (!rm_params.areAllRequiredParamsValid())
113 mooseError("Missing required parameters for RelationshipManager " + rm_name + " for object " +
114 name());
115
116 auto rm_obj = _factory.create<RelationshipManager>(rm_name, new_name, rm_params);
117
118 const bool added = _app.addRelationshipManager(rm_obj);
119
120 // Delete the resources created on behalf of the RM if it ends up not being added to the App.
121 if (!added)
122 _factory.releaseSharedObjects(*rm_obj);
123 else // we added it
124 unique_object_id++;
125
126 return added;
127}

134bool
135Action::addRelationshipManagers(Moose::RelationshipManagerType input_rm_type,
136 const InputParameters & moose_object_pars)
137{
138 const auto & buildable_types = moose_object_pars.getBuildableRelationshipManagerTypes();
139
140 bool added = false;
141
142 for (const auto & buildable_type : buildable_types)
143 {
144 auto & rm_name = std::get<0>(buildable_type);
145 auto & rm_type = std::get<1>(buildable_type);
146 auto rm_input_parameter_func = std::get<2>(buildable_type);
147
148 added = addRelationshipManager(
149 input_rm_type, moose_object_pars, rm_name, rm_type, rm_input_parameter_func) ||
150 added;
151 }
152
153 return added;
154}
