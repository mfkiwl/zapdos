28void
29SlepcEigenSolverConfiguration::configure_solver()
30{
31 PetscErrorCode ierr = 0;
32
33 if (_eigen_problem.isNonlinearEigenvalueSolver())
34 {
35 // Set custom monitors for SNES and KSP
36 _eigen_problem.initPetscOutput();
37 // Let us remove extra "eps_power" from SNES since users do not like it
38 ierr = Moose::SlepcSupport::mooseSlepcEPSSNESSetUpOptionPrefix(_slepc_solver.eps());
39 LIBMESH_CHKERR(ierr);
40 // Let us hook up a customize PC if users ask. Users still can use PETSc options to override
41 // this setting
42 if (_eigen_problem.solverParams()._customized_pc_for_eigen)
43 {
44 ierr = Moose::SlepcSupport::mooseSlepcEPSSNESSetCustomizePC(_slepc_solver.eps());
45 LIBMESH_CHKERR(ierr);
46 }
47 // Let set a default PC side. I would like to have the setting be consistent with
48 // what we do in regular nonlinear executioner. Petsc options are able to override
49 // this setting
50 Moose::SlepcSupport::mooseSlepcEPSSNESKSPSetPCSide(_eigen_problem, _slepc_solver.eps());
51 // A customized stopping test for nonlinear free power iterations.
52 // Nonlinear power iterations need to be marked as converged in EPS to
53 // retrieve solution from SLEPc EPS.
54 ierr = EPSSetStoppingTestFunction(
55 _slepc_solver.eps(), Moose::SlepcSupport::mooseSlepcStoppingTest, &_eigen_problem, NULL);
56 LIBMESH_CHKERR(ierr);
57
58 // Remove all SLEPc monitors.
59 ierr = EPSMonitorCancel(_slepc_solver.eps());
60 LIBMESH_CHKERR(ierr);
61 // A customized EPS monitor in moose. We need to print only eigenvalue
62 ierr = EPSMonitorSet(
63 _slepc_solver.eps(), Moose::SlepcSupport::mooseSlepcEPSMonitor, &_eigen_problem, NULL);
64 LIBMESH_CHKERR(ierr);
65 }
66}
