340 template <typename T, typename... Args>
341 std::vector<T *> & queryIntoHelper(std::vector<T *> & results, const bool sort, Args &&... args)
342 {
343 std::lock_guard<std::mutex> lock(_cache_mutex);
344 setKeysInner<0, KeyType<Attribs>...>(args...);
345
346 std::size_t query_id;
347 const auto entry = _cache.find(std::make_pair(sort, _key_tup));
348 if (entry == _cache.end())
349 {
350 setAttribsInner<0, KeyType<Attribs>...>(args...);
351 // add the sort attribute. No need (I think) to clear the cache because the base query is
352 // not changing
353 _attribs.emplace_back(new AttribSorted(*_w, sort));
354 query_id = _w->queryID(_attribs);
355 _cache[std::make_pair(sort, _key_tup)] = query_id;
356 // remove the sort attribute
357 _attribs.pop_back();
358 }
359 else
360 query_id = entry->second;
361
362 return _w->queryInto(query_id, results);
363 }

485 std::vector<T *> & queryInto(int query_id, std::vector<T *> & results, bool show_all = false)
486 {
487 std::lock_guard<std::mutex> lock(_obj_cache_mutex);
488 auto & objs = query(query_id);
489 results.clear();
490 results.reserve(objs.size());
491 for (auto & obj : objs)
492 {
493 mooseAssert(obj, "Null object");
494 auto cast_obj = dynamic_cast<T *>(obj);
495 if (obj)
496 mooseAssert(cast_obj,
497 "Queried object has incompatible c++ type for object named " + obj->name());
498 if (show_all || obj->enabled())
499 results.push_back(cast_obj);
500 }
501 return results;
502 }
