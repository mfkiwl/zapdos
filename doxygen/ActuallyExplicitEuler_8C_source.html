42ActuallyExplicitEuler::computeTimeDerivatives()

64void
65ActuallyExplicitEuler::solve()
66{
67 // Reset iteration counts
68 _n_nonlinear_iterations = 0;
69 _n_linear_iterations = 0;
70
71 _current_time = _fe_problem.time();
72
73 // Set time to the time at which to evaluate the residual
74 _fe_problem.time() = _fe_problem.timeOld();
75 _nonlinear_implicit_system->update();
76
77 // Compute the residual
78 _explicit_residual.zero();
79 _fe_problem.computeResidual(
80 *_nonlinear_implicit_system->current_local_solution, _explicit_residual, _nl.number());
81
82 // Move the residual to the RHS
83 _explicit_residual *= -1.0;
84
85 // Compute the mass matrix
86 auto & mass_matrix = _nonlinear_implicit_system->get_system_matrix();
87 if (!_constant_mass || (_constant_mass && _t_step == 1))
88 _fe_problem.computeJacobianTag(
89 *_nonlinear_implicit_system->current_local_solution, mass_matrix, _Ke_time_tag);
90
91 // Perform the linear solve
92 bool converged = performExplicitSolve(mass_matrix);
93
94 // Update the solution
95 *_nonlinear_implicit_system->solution = _nl.solutionOld();
96 *_nonlinear_implicit_system->solution += _solution_update;
97
98 // Enforce contraints on the solution
99 DofMap & dof_map = _nonlinear_implicit_system->get_dof_map();
100 dof_map.enforce_constraints_exactly(*_nonlinear_implicit_system,
101 _nonlinear_implicit_system->solution.get());
102 _nonlinear_implicit_system->update();
103
104 _nl.setSolution(*_nonlinear_implicit_system->current_local_solution);
105
106 _nonlinear_implicit_system->nonlinear_solver->converged = converged;
107}

110ActuallyExplicitEuler::postResidual(NumericVector<Number> & residual)
111{
