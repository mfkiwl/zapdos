51void
52ComputeUserObjectsThread::subdomainChanged()
53{
54 // for the current thread get block objects for the current subdomain and *all* side objects
55 std::vector<UserObject *> objs;
56 querySubdomain(Interfaces::ElementUserObject | Interfaces::InternalSideUserObject |
57 Interfaces::InterfaceUserObject | Interfaces::DomainUserObject,
58 objs);
59
60 _query.clone()
61 .condition<AttribThread>(_tid)
62 .condition<AttribInterfaces>(Interfaces::DomainUserObject)
63 .queryInto(_all_domain_objs);
64
65 std::vector<UserObject *> side_objs;
66 _query.clone()
67 .condition<AttribThread>(_tid)
68 .condition<AttribInterfaces>(Interfaces::SideUserObject)
69 .queryInto(side_objs);
70
71 objs.insert(objs.begin(), side_objs.begin(), side_objs.end());
72
73 // collect dependencies and run subdomain setup
74 _fe_problem.subdomainSetup(_subdomain, _tid);
75
76 std::set<MooseVariableFEBase *> needed_moose_vars;
77 std::set<unsigned int> needed_mat_props;
78 std::set<TagID> needed_fe_var_vector_tags;
79 for (const auto obj : objs)
80 {
81 auto v_obj = dynamic_cast<MooseVariableDependencyInterface *>(obj);
82 if (v_obj)
83 {
84 const auto & v_deps = v_obj->getMooseVariableDependencies();
85 needed_moose_vars.insert(v_deps.begin(), v_deps.end());
86 }
87
88 auto m_obj = dynamic_cast<MaterialPropertyInterface *>(obj);
89 if (m_obj)
90 {
91 auto & m_deps = m_obj->getMatPropDependencies();
92 needed_mat_props.insert(m_deps.begin(), m_deps.end());
93 }
94
95 auto c_obj = dynamic_cast<Coupleable *>(obj);
96 if (c_obj)
97 {
98 const auto & tag_deps = c_obj->getFEVariableCoupleableVectorTags();
99 needed_fe_var_vector_tags.insert(tag_deps.begin(), tag_deps.end());
100 }
101
102 obj->subdomainSetup();
103 }
104 _fe_problem.getMaterialWarehouse().updateBlockFEVariableCoupledVectorTagDependency(
105 _subdomain, needed_fe_var_vector_tags, _tid);
106
107 _fe_problem.setActiveElementalMooseVariables(needed_moose_vars, _tid);
108 _fe_problem.setActiveMaterialProperties(needed_mat_props, _tid);
109 _fe_problem.setActiveFEVariableCoupleableVectorTags(needed_fe_var_vector_tags, _tid);
110 _fe_problem.prepareMaterials(_subdomain, _tid);
111
112 querySubdomain(Interfaces::InternalSideUserObject, _internal_side_objs);
113 querySubdomain(Interfaces::ElementUserObject, _element_objs);
114 querySubdomain(Interfaces::ShapeElementUserObject, _shape_element_objs);
115 querySubdomain(Interfaces::DomainUserObject, _domain_objs);
116}

118void
119ComputeUserObjectsThread::onElement(const Elem * elem)
120{
121 _fe_problem.prepare(elem, _tid);
122 _fe_problem.reinitElem(elem, _tid);
123
124 // Set up Sentinel class so that, even if reinitMaterials() throws, we
125 // still remember to swap back during stack unwinding.
126 SwapBackSentinel sentinel(_fe_problem, &FEProblem::swapBackMaterials, _tid);
127 _fe_problem.reinitMaterials(_subdomain, _tid);
128
129 for (const auto & uo : _element_objs)
130 {
131 uo->execute();
132
133 // update the aux solution vector if writable coupled variables are used
134 if (uo->hasWritableCoupledVariables())
135 {
136 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
137 for (auto * var : uo->getWritableCoupledVariables())
138 var->insert(_aux_sys.solution());
139 }
140 }
141
142 for (auto & uo : _domain_objs)
143 {
144 uo->preExecuteOnElement();
145 uo->executeOnElement();
146 }
147
148 // UserObject Jacobians
149 if (_fe_problem.currentlyComputingJacobian() && _shape_element_objs.size() > 0)
150 {
151 // Prepare shape functions for ShapeElementUserObjects
152 const auto & jacobian_moose_vars = _fe_problem.getUserObjectJacobianVariables(_tid);
153 for (const auto & jvar : jacobian_moose_vars)
154 {
155 unsigned int jvar_id = jvar->number();
156 auto && dof_indices = jvar->dofIndices();
157
158 _fe_problem.prepareShapes(jvar_id, _tid);
159 for (const auto uo : _shape_element_objs)
160 uo->executeJacobianWrapper(jvar_id, dof_indices);
161 }
162 }

165void
166ComputeUserObjectsThread::onBoundary(const Elem * elem,
167 unsigned int side,
168 BoundaryID bnd_id,
169 const Elem * lower_d_elem /*=nullptr*/)
170{
171 std::vector<UserObject *> userobjs;
172 queryBoundary(Interfaces::SideUserObject, bnd_id, userobjs);
173 if (userobjs.size() == 0 && _domain_objs.size() == 0)
174 return;
175
176 _fe_problem.reinitElemFace(elem, side, bnd_id, _tid);
177
178 // Reinitialize lower-dimensional variables for use in boundary Materials
179 if (lower_d_elem)
180 _fe_problem.reinitLowerDElem(lower_d_elem, _tid);
181
182 // Set up Sentinel class so that, even if reinitMaterialsFace() throws, we
183 // still remember to swap back during stack unwinding.
184 SwapBackSentinel sentinel(_fe_problem, &FEProblem::swapBackMaterialsFace, _tid);
185 _fe_problem.reinitMaterialsFace(_subdomain, _tid);
186 _fe_problem.reinitMaterialsBoundary(bnd_id, _tid);
187
188 for (const auto & uo : userobjs)
189 uo->execute();
190
191 for (auto & uo : _domain_objs)
192 {
193 uo->preExecuteOnBoundary();
194 uo->executeOnBoundary();
195 }
196
197 // UserObject Jacobians
198 std::vector<ShapeSideUserObject *> shapers;
199 queryBoundary(Interfaces::ShapeSideUserObject, bnd_id, shapers);
200 if (_fe_problem.currentlyComputingJacobian() && shapers.size() > 0)
201 {
202 // Prepare shape functions for ShapeSideUserObjects
203 const auto & jacobian_moose_vars = _fe_problem.getUserObjectJacobianVariables(_tid);
204 for (const auto & jvar : jacobian_moose_vars)
205 {
206 unsigned int jvar_id = jvar->number();
207 auto && dof_indices = jvar->dofIndices();
208
209 _fe_problem.prepareFaceShapes(jvar_id, _tid);
210
211 for (const auto & uo : shapers)
212 uo->executeJacobianWrapper(jvar_id, dof_indices);
213 }
214 }

217void
218ComputeUserObjectsThread::onInternalSide(const Elem * elem, unsigned int side)
219{
220 // Pointer to the neighbor we are currently working on.
221 const Elem * neighbor = elem->neighbor_ptr(side);
222
223 // Get the global id of the element and the neighbor
224 const dof_id_type elem_id = elem->id(), neighbor_id = neighbor->id();
225
226 if (_internal_side_objs.size() == 0 && _domain_objs.size() == 0)
227 return;
228 if (!((neighbor->active() && (neighbor->level() == elem->level()) && (elem_id < neighbor_id)) ||
229 (neighbor->level() < elem->level())))
230 return;
231
232 _fe_problem.prepareFace(elem, _tid);
233 _fe_problem.reinitNeighbor(elem, side, _tid);
234
235 // Set up Sentinels so that, even if one of the reinitMaterialsXXX() calls throws, we
236 // still remember to swap back during stack unwinding.
237 SwapBackSentinel face_sentinel(_fe_problem, &FEProblem::swapBackMaterialsFace, _tid);
238 _fe_problem.reinitMaterialsFace(elem->subdomain_id(), _tid);
239
240 SwapBackSentinel neighbor_sentinel(_fe_problem, &FEProblem::swapBackMaterialsNeighbor, _tid);
241 _fe_problem.reinitMaterialsNeighbor(neighbor->subdomain_id(), _tid);
242
243 for (const auto & uo : _internal_side_objs)
244 if (!uo->blockRestricted() || uo->hasBlocks(neighbor->subdomain_id()))
245 uo->execute();
246
247 for (auto & uo : _domain_objs)
248 if (!uo->blockRestricted() || uo->hasBlocks(neighbor->subdomain_id()))
249 {
250 uo->preExecuteOnInternalSide();
251 uo->executeOnInternalSide();
252 }

255void
256ComputeUserObjectsThread::onInterface(const Elem * elem, unsigned int side, BoundaryID bnd_id)
257{
258 // Pointer to the neighbor we are currently working on.
259 const Elem * neighbor = elem->neighbor_ptr(side);
260 if (!(neighbor->active()))
261 return;
262
263 std::vector<UserObject *> interface_objs;
264 queryBoundary(Interfaces::InterfaceUserObject, bnd_id, interface_objs);
265
266 bool has_domain_objs = false;
267 // we need to check all domain user objects because a domain user object may not be active
268 // on the current subdomain but should be executed on the interface that it attaches to
269 for (const auto * const domain_uo : _all_domain_objs)
270 if (domain_uo->shouldExecuteOnInterface())
271 {
272 has_domain_objs = true;
273 break;
274 }
275
276 // if we do not have any interface user objects and domain user objects on the current
277 // interface
278 if (interface_objs.empty() && !has_domain_objs)
279 return;
280
281 _fe_problem.prepareFace(elem, _tid);
282 _fe_problem.reinitNeighbor(elem, side, _tid);
283
284 // Set up Sentinels so that, even if one of the reinitMaterialsXXX() calls throws, we
285 // still remember to swap back during stack unwinding.
286
287 SwapBackSentinel face_sentinel(_fe_problem, &FEProblem::swapBackMaterialsFace, _tid);
288 _fe_problem.reinitMaterialsFace(elem->subdomain_id(), _tid);
289 _fe_problem.reinitMaterialsBoundary(bnd_id, _tid);
290
291 SwapBackSentinel neighbor_sentinel(_fe_problem, &FEProblem::swapBackMaterialsNeighbor, _tid);
292 _fe_problem.reinitMaterialsNeighbor(neighbor->subdomain_id(), _tid);
293
294 // Has to happen after face and neighbor properties have been computed. Note that we don't use
295 // a sentinel here because FEProblem::swapBackMaterialsFace is going to handle face materials,
296 // boundary materials, and interface materials (e.g. it queries the boundary material data
297 // with the current element and side
298 _fe_problem.reinitMaterialsInterface(bnd_id, _tid);
299
300 for (const auto & uo : interface_objs)
301 uo->execute();
302
303 for (auto & uo : _all_domain_objs)
304 if (uo->shouldExecuteOnInterface())
305 {
306 uo->preExecuteOnInterface();
307 uo->executeOnInterface();
308 }

353void
354ComputeUserObjectsThread::printBlockExecutionInformation() const
355{
356 if (!_fe_problem.shouldPrintExecution(_tid))
357 return;
358
359 // Gather all user objects that may execute
360 // TODO: restrict this gathering of boundary objects to boundaries that are present
361 // in the current block
362 std::vector<ShapeSideUserObject *> shapers;
363 const_cast<ComputeUserObjectsThread *>(this)->queryBoundary(
364 Interfaces::ShapeSideUserObject, Moose::ANY_BOUNDARY_ID, shapers);
365
366 std::vector<SideUserObject *> side_uos;
367 const_cast<ComputeUserObjectsThread *>(this)->queryBoundary(
368 Interfaces::SideUserObject, Moose::ANY_BOUNDARY_ID, side_uos);
369
370 std::vector<InterfaceUserObject *> interface_objs;
371 const_cast<ComputeUserObjectsThread *>(this)->queryBoundary(
372 Interfaces::InterfaceUserObject, Moose::ANY_BOUNDARY_ID, interface_objs);
373
374 std::vector<const DomainUserObject *> domain_interface_uos;
375 for (const auto * const domain_uo : _domain_objs)
376 if (domain_uo->shouldExecuteOnInterface())
377 domain_interface_uos.push_back(domain_uo);
378
379 // Approximation of the number of user objects currently executing
380 const auto num_objects = _element_objs.size() + _domain_objs.size() + _shape_element_objs.size() +
381 side_uos.size() + shapers.size() + _internal_side_objs.size() +
382 interface_objs.size() + domain_interface_uos.size();
383
384 const auto & console = _fe_problem.console();
385 const auto & execute_on = _fe_problem.getCurrentExecuteOnFlag();
386
387 if (num_objects > 0)
388 {
389 if (_blocks_exec_printed.count(_subdomain))
390 return;
391
392 console << "[DBG] Ordering of User Objects on block " << _subdomain << std::endl;
393 // Output specific ordering of objects
394 printExecutionOrdering<ElementUserObject>(_element_objs, "element user objects");
395 printExecutionOrdering<DomainUserObject>(_domain_objs, "domain user objects");
396 if (_fe_problem.currentlyComputingJacobian())
397 printExecutionOrdering<ShapeElementUserObject>(
398 _shape_element_objs, "element user objects contributing to the Jacobian");
399 printExecutionOrdering<SideUserObject>(side_uos, "side user objects");
400 if (_fe_problem.currentlyComputingJacobian())
401 printExecutionOrdering<ShapeSideUserObject>(shapers,
402 "side user objects contributing to the Jacobian");
403 printExecutionOrdering<InternalSideUserObject>(_internal_side_objs,
404 "internal side user objects");
405 printExecutionOrdering<InterfaceUserObject>(interface_objs, "interface user objects");
406 console << "[DBG] Only user objects active on local element/sides are executed" << std::endl;
407 }
408 else if (num_objects == 0 && !_blocks_exec_printed.count(_subdomain))
409 console << "[DBG] No User Objects on block " << _subdomain << " on " << execute_on.name()
410 << std::endl;
411
412 // Mark subdomain as having printed to avoid printing again
413 _blocks_exec_printed.insert(_subdomain);
414}
