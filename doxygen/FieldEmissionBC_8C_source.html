72ADReal
73FieldEmissionBC::computeQpResidual()
74{
75 ADReal v;
76 ADReal f;
77 ADReal jFE;
78 ADReal jSE;
79 ADReal F;
80 Real _relaxation_Expr;
81
82 if (_normals[_qp] * -1.0 * -_grad_potential[_qp] > 0.0)
83 {
84 _a = 1.0;
85 return 0;
86 }
87 else
88 {
89 _a = 0.0;
90
91 _ion_flux = _sgnip[_qp] * _muip[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_ip[_qp]) -
92 _Dip[_qp] * std::exp(_ip[_qp]) * _grad_ip[_qp] * _r_units;
93
94 // Fowler-Nordheim
95 // jFE = (a / wf) * F^2 * exp(-v(f) * b * wf^1.5 / F)
96 // a = 1.541434E-6 A eV/V^2
97 // b = 6.830890E9 V/m-eV^1.5
98 // c = 1.439964E-9 eV^2-m/V
99 // v(f) = 1 - f + (f/6)*ln(f)
100 // f = c*(F/wf^2)
101
102 F = -(1 - _a) * _field_enhancement[_qp] * _normals[_qp] * _grad_potential[_qp] * _r_units;
103
104 f = FE_c * F / std::pow(_work_function[_qp], 2);
105 v = 1 - f + (f / 6) * std::log(f);
106
107 jFE = (FE_a / (_work_function[_qp])) * std::pow(F, 2) *
108 std::exp(-v * FE_b * std::pow(_work_function[_qp], 1.5) / F);
109 jSE = _e[_qp] * 6.02E23 * _se_coeff[_qp] * _ion_flux * _normals[_qp];
110
111 if (_relax == true)
112 _relaxation_Expr = std::tanh(_t / _tau);
113 else
114 _relaxation_Expr = 1.0;
115
116 return _relaxation_Expr * _test[_i][_qp] * _r_units * 2. / (1. + _r) * (1 - _a) * (-jFE - jSE) /
117 (_e[_qp] * 6.02E23);
118 }
119}
