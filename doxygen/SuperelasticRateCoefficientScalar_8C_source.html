38{
39 Real equilibrium_constant;
40 Real Tgas;
41 if (isCoupledScalar("Tgas"))
42 Tgas = _Tgas[_i];
43 else
44 Tgas = _Tgas_const;
45
46 equilibrium_constant = std::pow(1.0, _polynomial.power_coefficient()) * std::exp(_polynomial.delta_a(0)*(std::log(Tgas - 1)) +
47 (_polynomial.delta_a(1)/2.0)*Tgas + (_polynomial.delta_a(2)/6.0)*std::pow(Tgas, 2.0) + (_polynomial.delta_a(3)/12.0)*std::pow(Tgas, 3.0) +
48 (_polynomial.delta_a(4)/20.0)*std::pow(Tgas, 4.0) - _polynomial.delta_a(5)*std::pow(Tgas, -1.0) + _polynomial.delta_a(6));
49
50 return _forward_coefficient[_i] / equilibrium_constant;
