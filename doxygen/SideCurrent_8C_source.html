36SideCurrent::SideCurrent(const InputParameters & parameters)
37 : SideIntegralVariablePostprocessor(parameters),
38 _mobility(parameters.get<std::string>("mobility")),
39 _mobility_coef(getMaterialProperty<Real>(_mobility)),
40 _r_units(1. / getParam<Real>("position_units")),
41 _r(getParam<Real>("r")),
42 _kb(getMaterialProperty<Real>("k_boltz")),
43 _T_heavy(getMaterialProperty<Real>("TArp")),
44 _mass(getMaterialProperty<Real>("massArp")),
45 _e(getMaterialProperty<Real>("e")),
46 _sgn(getMaterialProperty<Real>("sgnArp")),
47 _a(0.5),
48 _grad_potential(coupledGradient("potential")),
49 _mean_en(coupledValue("mean_en")),
50 _Arp(coupledValue("Arp")),
51 _muArp(getMaterialProperty<Real>("muArp"))
52{
53 _num_ions = coupledComponents("ions");
54
55 // Resize the vectors to store _num_ions pointers
56 _ions.resize(_num_ions);
57 _mu_ions.resize(_num_ions);
58 _sgn_ions.resize(_num_ions);
59 _mass_ions.resize(_num_ions);
60 _T_ions.resize(_num_ions);
61 _grad_ions.resize(_num_ions);
62
63 for (unsigned int i = 0; i < _ions.size(); ++i)
64 {
65 _ions[i] = &coupledValue("ions", i);
66 _grad_ions[i] = &coupledGradient("ions", i);
67 _mu_ions[i] = &getMaterialProperty<Real>("mu" + (*getVar("ions", i)).name());
68 _sgn_ions[i] = &getMaterialProperty<Real>("sgn" + (*getVar("ions", i)).name());
69 _mass_ions[i] = &getMaterialProperty<Real>("mass" + (*getVar("ions", i)).name());
70 _T_ions[i] = &getMaterialProperty<Real>("T" + (*getVar("ions", i)).name());
71 }
72}

74Real
75SideCurrent::computeQpIntegral()
76{
77 // Output units for base case are: mol / (m^2 * s)
78 // _v_thermal = std::sqrt(8 * _kb[_qp] * _T_heavy[_qp] / (M_PI * _mass[_qp]));
79
80 if (_normals[_qp] * _sgn[_qp] * -_grad_potential[_qp] > 0.0)
81 {
82 _a = 1.0;
83 _b = 0.0;
84 }
85 else
86 {
87 _a = 0.0;
88 _b = 1.0;
89 }
90
91 _ve_thermal =
92 std::sqrt(8 * 1.602e-19 * 2.0 / 3 * std::exp(_mean_en[_qp] - _u[_qp]) / (M_PI * 9.11e-31));
93
94 _ion_flux = 0.0;
95
96 for (unsigned int i = 0; i < _num_ions; ++i)
97 _ion_flux += 0.5 *
98 std::sqrt(8 * _kb[_qp] * (*_T_ions[i])[_qp] / (M_PI * (*_mass_ions[i])[_qp])) *
99 std::exp((*_ions[i])[_qp]) +
100 (2 * _a - 1) * (*_sgn_ions[i])[_qp] * (*_mu_ions[i])[_qp] * -_grad_potential[_qp] *
101 _r_units * std::exp((*_ions[i])[_qp]) * _normals[_qp];
102
103 return ((1. - _r) / (1. + _r) * _ion_flux +
104 (1. - _r) / (1. + _r) *
105 (-(2 * _b - 1) * _mobility_coef[_qp] * -_grad_potential[_qp] * _r_units *
106 std::exp(_u[_qp]) * _normals[_qp] +
107 0.5 * _ve_thermal * std::exp(_u[_qp]))) *
108 6.022e23 * 1.602e-19 * _r_units;
109 /*
110 return ((1. - _r) / (1. + _r) * 0.5 * _v_thermal * std::exp(_Arp[_qp]) +
111 (1. - _r) / (1. + _r) *
112 ((2 * _a - 1) * _sgn[_qp] * _muArp[_qp] * -_grad_potential[_qp] * _r_units *
113 std::exp(_Arp[_qp]) * _normals[_qp]) +
114 (1. - _r) / (1. + _r) *
115 (-(2 * _b - 1) * _mobility_coef[_qp] * -_grad_potential[_qp] * _r_units *
116 std::exp(_u[_qp]) * _normals[_qp] +
117 0.5 * _ve_thermal * std::exp(_u[_qp]))) *
118 6.022e23 * 1.602e-19 * _r_units;
119 */
120}
