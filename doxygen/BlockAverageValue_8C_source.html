38BlockAverageValue::averageValue(SubdomainID block) const
39{
40 // Note that we can't use operator[] for a std::map in a const function!
41 if (_average_values.find(block) != _average_values.end())

79void
80BlockAverageValue::threadJoin(const UserObject & y)
81{
82 ElementIntegralVariablePostprocessor::threadJoin(y);
83
84 // We are joining with another class like this one so do a cast so we can get to it's data
85 const BlockAverageValue & bav = dynamic_cast<const BlockAverageValue &>(y);
86
87 for (std::map<SubdomainID, Real>::const_iterator it = bav._integral_values.begin();
88 it != bav._integral_values.end();
89 ++it)
90 _integral_values[it->first] += it->second;
91
92 for (std::map<SubdomainID, Real>::const_iterator it = bav._volume_values.begin();
93 it != bav._volume_values.end();
94 ++it)
95 _volume_values[it->first] += it->second;
96
97 for (std::map<SubdomainID, Real>::const_iterator it = bav._average_values.begin();
98 it != bav._average_values.end();
99 ++it)
100 _average_values[it->first] += it->second;
101}

103void
104BlockAverageValue::finalize()
105{
106 // Loop over the integral values and sum them up over the processors
107 for (std::map<SubdomainID, Real>::iterator it = _integral_values.begin();
108 it != _integral_values.end();
109 ++it)
110 gatherSum(it->second);
111
112 // Loop over the volumes and sum them up over the processors
113 for (std::map<SubdomainID, Real>::iterator it = _volume_values.begin();
114 it != _volume_values.end();
115 ++it)
116 gatherSum(it->second);
117
118 // Now everyone has the correct data so everyone can compute the averages properly:
119 for (std::map<SubdomainID, Real>::iterator it = _average_values.begin();
120 it != _average_values.end();
121 ++it)
122 {
123 SubdomainID id = it->first;
124 _average_values[id] = _integral_values[id] / _volume_values[id];
125 }
