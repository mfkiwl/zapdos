21CylindricalGridDivision::validParams()
22{
23 InputParameters params = MeshDivision::validParams();
24 params.addClassDescription(
25 "Divide the mesh along a cylindrical grid. The innermost numbering of divisions is the "
26 "radial bins, then comes the azimuthal bins, then the axial bins");
27
28 // Definition of the cylinder(s)
29 params.addRequiredParam<Point>("axis_direction", "Direction of the cylinder's axis");
30 params.addRequiredParam<Point>(
31 "azimuthal_start",
32 "Direction of the 0-azimuthal-angle vector, normal to the cylinder's axis");
33 params.addParam<Point>("center",
34 "Point on the cylinder's axis, acting as the center of this local "
35 "R-theta-Z coordinate based division");
36 params.addParam<PositionsName>("center_positions",
37 "Positions of the points on the cylinders' respective axis, "
38 "acting as the center of the local "
39 "R-theta-Z coordinate based divisions");
40
41 // Spatial bounds of the cylinder(s)
42 params.addRangeCheckedParam<Real>(
43 "r_min", 0, "r_min>=0", "Minimum radial coordinate (for a hollow cylinder)");
44 params.addRequiredRangeCheckedParam<Real>("r_max", "r_max>0", "Maximum radial coordinate");
45 params.addParam<Real>(
46 "cylinder_axial_min", std::numeric_limits<Real>::lowest(), "Minimum axial coordinate");
47 params.addParam<Real>(
48 "cylinder_axial_max", std::numeric_limits<Real>::max(), "Maximum axial coordinate");
49
50 // Number of bins
51 params.addRequiredRangeCheckedParam<unsigned int>(
52 "n_radial", "n_radial>0", "Number of divisions in the cylinder radial direction");
53 params.addRangeCheckedParam<unsigned int>(
54 "n_azimuthal", 1, "n_azimuthal>0", "Number of divisions in the azimuthal direction");
55 params.addRangeCheckedParam<unsigned int>(
56 "n_axial", 1, "n_axial>0", "Number of divisions in the cylinder axial direction");
57
58 params.addParam<bool>("assign_domain_outside_grid_to_border",
59 false,
60 "Whether to map the domain outside the grid back to the border of the grid "
61 "(radially or axially)");

66CylindricalGridDivision::CylindricalGridDivision(const InputParameters & parameters)
67 : MeshDivision(parameters),
68 _direction(getParam<Point>("axis_direction")),
69 _center(isParamValid("center") ? &getParam<Point>("center") : nullptr),
70 _center_positions(
71 isParamValid("center_positions")
72 ? &_fe_problem->getPositionsObject(getParam<PositionsName>("center_positions"))
73 : nullptr),
74 _azim_dir(getParam<Point>("azimuthal_start")),
75 _min_r(getParam<Real>("r_min")),
76 _max_r(getParam<Real>("r_max")),
77 _min_z(getParam<Real>("cylinder_axial_min")),
78 _max_z(getParam<Real>("cylinder_axial_max")),
79 _n_radial(getParam<unsigned int>("n_radial")),
80 _n_azim(getParam<unsigned int>("n_azimuthal")),
81 _n_axial(getParam<unsigned int>("n_axial")),
82 _outside_grid_counts_as_border(getParam<bool>("assign_domain_outside_grid_to_border"))
83{
84 CylindricalGridDivision::initialize();
85
86 // Check that we know the centers
87 if (!_center && !_center_positions)
88 paramError("center", "You must pass a parameter for the center of the cylindrical frame");
89
90 // Check axis
91 if (!MooseUtils::absoluteFuzzyEqual(_direction.norm_sq(), 1))
92 paramError("axis_direction", "Axis must have a norm of 1");
93 if (!MooseUtils::absoluteFuzzyEqual(_azim_dir.norm_sq(), 1))
94 paramError("azimuthal_start", "Azimuthal axis must have a norm of 1");
95
96 // Check non-negative size
97 if (_max_r < _min_r)
98 paramError("r_min", "Maximum radius must be larger than minimum radius");
99 if (_max_z < _min_z)
100 paramError("cylinder_axial_min", "Maximum axial extent must be larger than minimum");
101
102 // Check non-zero size if subdivided
103 if (_n_radial > 1 && MooseUtils::absoluteFuzzyEqual(_min_r, _max_r))
104 paramError("n_radial", "Zero-thickness cylinder cannot be subdivided radially");
105 if (_n_axial > 1 && MooseUtils::absoluteFuzzyEqual(_min_z, _max_z))
106 paramError("n_axial", "Zero-height cylinder cannot be subdivided axially");
107
108 // Check non-infinite size if subdivided
109 if ((_min_z == std::numeric_limits<Real>::lowest() ||
110 _max_z == std::numeric_limits<Real>::max()) &&
111 _n_axial > 1)
112 paramError("n_axial",
113 "Infinite-height cylinder cannot be subdivided axially. Please specify both a "
114 "cylinder axial minimum and maximum extent");
115}

117void
118CylindricalGridDivision::initialize()
119{
120 setNumDivisions(_n_radial * _n_azim * _n_axial);
121
122 // Check that the grid is well-defined
123 if (_center_positions)
124 {
125 Real min_dist = 2 * _max_r;
126 Real min_center_dist = _center_positions->getMinDistanceBetweenPositions();
127 // Note that if the positions are not co-planar, the distance reported would be bigger but there
128 // could still be an overlap. Looking at min_center_dist is not enough
129 if (min_dist > min_center_dist)
130 mooseError(
131 "Cylindrical grids centered on the positions are too close to each other (min distance: ",
132 min_center_dist,
133 "), closer than the radial extent of each grid. Mesh division is ill-defined");
134 }
135
136 // We could alternatively check every point in the mesh but it seems expensive
137 // A bounding box check could also suffice but the cylinders would need to be excessively large to
138 // enclose the entire mesh
139 _mesh_fully_indexed = true;
140 if (!_outside_grid_counts_as_border || _center_positions)
141 _mesh_fully_indexed = false;
142}

150unsigned int
151CylindricalGridDivision::divisionIndex(const Point & pt) const
152{
153 // Compute coordinates of the point in the cylindrical coordinates
154 Point pc;
155 Point in_plane;
156 unsigned int offset = 0;
157 if (_center)
158 {
159 pc(2) = (pt - *_center) * _direction;
160 in_plane = (pt - *_center) - pc(2) * _direction;
161 }
162 else
163 {
164 // If dividing using positions, find the closest position
165 const bool initial = _fe_problem->getCurrentExecuteOnFlag() == EXEC_INITIAL;
166 const auto nearest_center_index = _center_positions->getNearestPositionIndex(pt, initial);
167 offset = nearest_center_index * getNumDivisions();
168 const auto new_center = _center_positions->getPosition(nearest_center_index, initial);
169 pc(2) = (pt - new_center) * _direction;
170 in_plane = (pt - new_center) - pc(2) * _direction;
171 }
172
173 pc(0) = in_plane.norm();
174 const Point pi2_dir = _azim_dir.cross(_direction);
175 pc(1) = std::atan2(in_plane * pi2_dir, in_plane * _azim_dir) + libMesh::pi;
176
177 if (!_outside_grid_counts_as_border)
178 {
179 if (MooseUtils::absoluteFuzzyLessThan(pc(0), _min_r) ||
180 MooseUtils::absoluteFuzzyGreaterThan(pc(0), _max_r))
181 return MooseMeshDivision::INVALID_DIVISION_INDEX;
182 if (MooseUtils::absoluteFuzzyLessThan(pc(2), _min_z) ||
183 MooseUtils::absoluteFuzzyGreaterThan(pc(2), _max_z))
184 return MooseMeshDivision::INVALID_DIVISION_INDEX;
185 }
186
187 const auto not_found = MooseMeshDivision::INVALID_DIVISION_INDEX;
188 auto ir = not_found, ia = not_found, iz = not_found;
189 const Point widths(_max_r - _min_r, 2 * libMesh::pi, _max_z - _min_z);
190
191 // Look inside the grid and on the left / back / bottom
192 for (const auto jr : make_range(_n_radial + 1))
193 {
194 const auto border_r = _min_r + widths(0) * jr / _n_radial;
195 if (jr > 0 && jr < _n_radial && MooseUtils::absoluteFuzzyEqual(border_r, pc(0)))
196 mooseWarning(
197 "Querying the division index for a point of a boundary between two regions radially: " +
198 Moose::stringify(pt));
199 if (border_r >= pc(0))
200 {
201 ir = jr > 0 ? jr - 1 : 0;
202 break;
203 }
204 }
205 for (const auto ja : make_range(_n_azim + 1))
206 {
207 const auto border_a = widths(1) * ja / _n_azim;
208 if (ja > 0 && ja < _n_azim && MooseUtils::absoluteFuzzyEqual(border_a, pc(1)))
209 mooseWarning("Querying the division index for a point of a boundary between two regions "
210 "azimuthally : " +
211 Moose::stringify(pt));
212 if (border_a >= pc(1))
213 {
214 ia = ja > 0 ? ja - 1 : 0;
215 break;
216 }
217 }
218 for (const auto jz : make_range(_n_axial + 1))
219 {
220 const auto border_z = _min_z + widths(2) * jz / _n_axial;
221 if (jz > 0 && jz < _n_axial && MooseUtils::absoluteFuzzyEqual(border_z, pc(2)))
222 mooseWarning("Querying the division index for a point of a boundary between two axial "
223 "regions along the cylinder axis: " +
224 Moose::stringify(pt));
225 if (border_z >= pc(2))
226 {
227 iz = jz > 0 ? jz - 1 : 0;
228 break;
229 }
230 }
231
232 // Look beyond the edges of the grid
233 if (MooseUtils::absoluteFuzzyGreaterEqual(pc(0), _max_r))
234 ir = _n_radial - 1;
235 if (MooseUtils::absoluteFuzzyGreaterEqual(pc(2), _max_z))
236 iz = _n_axial - 1;
237
238 // Handle edge case on widths
239 if (ir == not_found && MooseUtils::absoluteFuzzyEqual(widths(0), 0))
240 ir = 0;
241 if (iz == not_found && MooseUtils::absoluteFuzzyEqual(widths(2), 0))
242 iz = 0;
243 mooseAssert(ir != not_found, "We should have found a mesh division bin radially");
244 mooseAssert(ia != not_found, "We should have found a mesh division bin azimuthally");
245 mooseAssert(iz != not_found, "We should have found a mesh division bin axially");
246
247 return offset + ir + _n_radial * ia + iz * _n_radial * _n_azim;
248}
