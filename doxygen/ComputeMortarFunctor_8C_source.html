53void
54ComputeMortarFunctor::operator()(const Moose::ComputeType compute_type,
55 const std::set<TagID> & vector_tag_ids,
56 const std::set<TagID> & /*matrix_tag_ids*/)
57{
58 libmesh_parallel_only(_fe_problem.comm());
59
60 unsigned int num_cached = 0;
61 const auto & vector_tags = _fe_problem.getVectorTags(vector_tag_ids);
62
63 const auto & secondary_elems_to_mortar_segments = _amg.secondariesToMortarSegments();
64 typedef decltype(secondary_elems_to_mortar_segments.begin()) it_type;
65
66 std::vector<it_type> iterators;
67 for (auto it = secondary_elems_to_mortar_segments.begin();
68 it != secondary_elems_to_mortar_segments.end();
69 ++it)
70 {
71 auto * const secondary_elem = _subproblem.mesh().getMesh().query_elem_ptr(it->first);
72
73 if (secondary_elem && secondary_elem->processor_id() == _subproblem.processor_id() &&
74 !it->second.empty())
75 {
76 // This is local and the mortar segment set isn't empty, so include
77 iterators.push_back(it);
78 mooseAssert(secondary_elem->active(),
79 "We loop over active elements when building the mortar segment mesh, so we golly "
80 "well hope this is active.");
81 }
82 }
83
84 auto act_functor = [this, &num_cached, compute_type, &vector_tags]()
85 {
86 ++num_cached;
87
88 switch (compute_type)
89 {
90 case Moose::ComputeType::Residual:
91 {
92 for (auto * const mc : _mortar_constraints)
93 {
94 mc->setNormals();
95 mc->computeResidual();
96 }
97
98 _assembly.cacheResidual(Assembly::GlobalDataKey{}, vector_tags);
99 _assembly.cacheResidualNeighbor(Assembly::GlobalDataKey{}, vector_tags);
100 _assembly.cacheResidualLower(Assembly::GlobalDataKey{}, vector_tags);
101
102 if (num_cached % 20 == 0)
103 _assembly.addCachedResiduals(Assembly::GlobalDataKey{}, vector_tags);
104
105 break;
106 }
107
108 case Moose::ComputeType::Jacobian:
109 {
110 for (auto * const mc : _mortar_constraints)
111 {
112 mc->setNormals();
113 mc->computeJacobian();
114 }
115
116 _assembly.cacheJacobianMortar(Assembly::GlobalDataKey{});
117
118 if (num_cached % 20 == 0)
119 _assembly.addCachedJacobian(Assembly::GlobalDataKey{});
120 break;
121 }
122
123 case Moose::ComputeType::ResidualAndJacobian:
124 {
125 for (auto * const mc : _mortar_constraints)
126 {
127 mc->setNormals();
128 mc->computeResidualAndJacobian();
129 }
130
131 _assembly.cacheResidual(Assembly::GlobalDataKey{}, vector_tags);
132 _assembly.cacheResidualNeighbor(Assembly::GlobalDataKey{}, vector_tags);
133 _assembly.cacheResidualLower(Assembly::GlobalDataKey{}, vector_tags);
134 _assembly.cacheJacobianMortar(Assembly::GlobalDataKey{});
135
136 if (num_cached % 20 == 0)
137 {
138 _assembly.addCachedResiduals(Assembly::GlobalDataKey{}, vector_tags);
139 _assembly.addCachedJacobian(Assembly::GlobalDataKey{});
140 }
141 break;
142 }
143 }
144 };
145
146 PARALLEL_TRY
147 {
148 try
149 {
150 Moose::Mortar::loopOverMortarSegments(iterators,
151 _assembly,
152 _subproblem,
153 _fe_problem,
154 _amg,
155 _displaced,
156 _mortar_constraints,
157 0,
158 _secondary_ip_sub_to_mats,
159 _primary_ip_sub_to_mats,
160 _secondary_boundary_mats,
161 act_functor,
162 /*reinit_mortar_user_objects=*/true);
163 }
164 catch (libMesh::LogicError & e)
165 {
166 _fe_problem.setException("We caught a libMesh::LogicError: " + std::string(e.what()));
167 }
168 catch (MooseException & e)
169 {
170 _fe_problem.setException(e.what());
171 }
172 catch (MetaPhysicL::LogicError & e)
173 {
174 moose::translateMetaPhysicLError(e);
175 }
176 }
177 PARALLEL_CATCH;
178
179 // Call any post operations for our mortar constraints
180 for (auto * const mc : _mortar_constraints)
181 {
182 if (_amg.incorrectEdgeDropping())
183 mc->incorrectEdgeDroppingPost(_amg.getInactiveLMNodes());
184 else
185 mc->post();
186
187 mc->zeroInactiveLMDofs(_amg.getInactiveLMNodes(), _amg.getInactiveLMElems());
188 }
189
190 // Make sure any remaining cached residuals/Jacobians get added
191 if (_assembly.computingResidual())
192 _assembly.addCachedResiduals(Assembly::GlobalDataKey{}, vector_tags);
193 if (_assembly.computingJacobian())
194 _assembly.addCachedJacobian(Assembly::GlobalDataKey{});
195}
