41DCIonBC::computeQpResidual()
42{
43 if (_normals[_qp] * _sgn[_qp] * -_grad_potential[_qp] > 0.0)
44 {
45 _a = 1.0;
46 }
47 else
48 {
49 _a = 0.0;
50 }
51
52 return _test[_i][_qp] * _r_units *
53 (_a * _mu[_qp] * _sgn[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_u[_qp]) *
54 _normals[_qp]);
55}
