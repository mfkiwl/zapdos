59void
60SetupQuadratureAction::act()
61{
62 if (_problem.get() == nullptr)
63 return;
64
65 // add default/global quadrature rules
66 _problem->createQRules(
67 _type, _order, _element_order, _side_order, Moose::ANY_BLOCK_ID, _allow_negative_qweights);
68
69 // add custom block-specific quadrature rules
70 for (const auto & [block, order] : _custom_block_orders)
71 _problem->createQRules(_type,
72 _order,
73 Moose::stringToEnum<Order>(order),
74 Moose::stringToEnum<Order>(order),
75 block,
76 _allow_negative_qweights);
77}
