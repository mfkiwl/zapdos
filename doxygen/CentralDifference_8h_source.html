54CentralDifference::computeTimeDerivativeHelper(T & u_dot,
55 T2 & u_dotdot,
56 const T3 & u_old,
57 const T4 & u_old_old,
58 const T5 & u_old_old_old) const
59{
60 // computing first derivative
61 // using the Central Difference method
62 // u_dot_old = (first_term - second_term) / 2 / dt
63 // first_term = u
64 // second_term = u_older
65 u_dot = u_old;
66 u_dot -= u_old_old_old; // 'older than older' solution
67 u_dot *= 1.0 / (2.0 * _dt);
68
69 // computing second derivative
70 // using the Central Difference method
71 // u_dotdot_old = (first_term - second_term + third_term) / dt / dt
72 // first_term = u
73 // second_term = 2 * u_old
74 // third_term = u_older
75 u_dotdot = u_old;
76 u_dotdot -= u_old_old;
77 u_dotdot -= u_old_old;
78 u_dotdot += u_old_old_old;
79 u_dotdot *= 1.0 / (_dt * _dt);
80}
