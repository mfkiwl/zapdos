66EEDFEnergyTownsendLog::computeQpResidual()
67{
68 Real electron_flux_mag = (std::exp(_em[_qp]) * (-_muem[_qp] * -_grad_potential[_qp] * _r_units -
69 _diffem[_qp] * _grad_em[_qp] * _r_units))
70 .norm();
71 Real iz_term = _alpha[_qp] * std::exp(_target[_qp]) * electron_flux_mag;
72
73 return -_test[_i][_qp] * iz_term * _threshold_energy;
74}

76Real
77EEDFEnergyTownsendLog::computeQpJacobian()
78{
79 Real d_actual_mean_en_d_mean_en = std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
80 Real d_iz_d_mean_en = _d_iz_d_actual_mean_en[_qp] * d_actual_mean_en_d_mean_en;
81 Real d_muem_d_mean_en = _d_muem_d_actual_mean_en[_qp] * d_actual_mean_en_d_mean_en;
82 Real d_diffem_d_mean_en = _d_diffem_d_actual_mean_en[_qp] * d_actual_mean_en_d_mean_en;
83
84 RealVectorValue electron_flux =
85 std::exp(_em[_qp]) *
86 (-_muem[_qp] * -_grad_potential[_qp] * _r_units - _diffem[_qp] * _grad_em[_qp] * _r_units);
87 RealVectorValue d_electron_flux_d_mean_en =
88 std::exp(_em[_qp]) * (-d_muem_d_mean_en * -_grad_potential[_qp] * _r_units -
89 d_diffem_d_mean_en * _grad_em[_qp] * _r_units);
90 Real electron_flux_mag = electron_flux.norm();
91 Real d_electron_flux_mag_d_mean_en = electron_flux * d_electron_flux_d_mean_en /
92 (electron_flux_mag + std::numeric_limits<double>::epsilon());
93 Real d_iz_term_d_mean_en =
94 (electron_flux_mag * d_iz_d_mean_en + _alpha[_qp] * d_electron_flux_mag_d_mean_en);
95
96 return -_test[_i][_qp] * d_iz_term_d_mean_en * std::exp(_target[_qp]) * _threshold_energy;
97}

99Real
100EEDFEnergyTownsendLog::computeQpOffDiagJacobian(unsigned int jvar)
101{
102 Real d_actual_mean_en_d_em = -std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
103 Real d_iz_d_em = _d_iz_d_actual_mean_en[_qp] * d_actual_mean_en_d_em;
104 Real d_muem_d_em = _d_muem_d_actual_mean_en[_qp] * d_actual_mean_en_d_em;
105 Real d_diffem_d_em = _d_diffem_d_actual_mean_en[_qp] * d_actual_mean_en_d_em;
106
107 RealVectorValue electron_flux =
108 std::exp(_em[_qp]) *
109 (-_muem[_qp] * -_grad_potential[_qp] * _r_units - _diffem[_qp] * _grad_em[_qp] * _r_units);
110 RealVectorValue d_electron_flux_d_potential =
111 -_muem[_qp] * -_grad_phi[_j][_qp] * _r_units * std::exp(_em[_qp]);
112 RealVectorValue d_electron_flux_d_em =
113 std::exp(_em[_qp]) * (-d_muem_d_em * -_grad_potential[_qp] * _r_units -
114 _muem[_qp] * -_grad_potential[_qp] * _r_units * _phi[_j][_qp] -
115 d_diffem_d_em * _grad_em[_qp] * _r_units -
116 _diffem[_qp] * _phi[_j][_qp] * _grad_em[_qp] * _r_units -
117 _diffem[_qp] * _grad_phi[_j][_qp] * _r_units);
118 Real electron_flux_mag = electron_flux.norm();
119 Real d_electron_flux_mag_d_potential =
120 electron_flux * d_electron_flux_d_potential /
121 (electron_flux_mag + std::numeric_limits<double>::epsilon());
122 Real d_electron_flux_mag_d_em = electron_flux * d_electron_flux_d_em /
123 (electron_flux_mag + std::numeric_limits<double>::epsilon());
124
125 Real d_iz_term_d_potential = (_alpha[_qp] * d_electron_flux_mag_d_potential);
126 Real d_iz_term_d_em = (electron_flux_mag * d_iz_d_em + _alpha[_qp] * d_electron_flux_mag_d_em);
127
128 if (jvar == _potential_id)
129 return -_test[_i][_qp] * d_iz_term_d_potential * _threshold_energy * std::exp(_target[_qp]);
130
131 else if (jvar == _em_id)
132 {
133 return -_test[_i][_qp] * d_iz_term_d_em * _threshold_energy * std::exp(_target[_qp]);
134 }
135 else if (jvar == _target_id)
136 {
137 return -_test[_i][_qp] * _alpha[_qp] * std::exp(_target[_qp]) * electron_flux.norm() *
138 _phi[_j][_qp] * _threshold_energy;
139 }
140
141 else
142 return 0.0;
143}
