36void
37CouplingFunctorCheckAction::act()
38{
39 // If we're doing Jacobian-free, then we have no matrix and we can just return
40 if (_problem->solverParams()._type == Moose::ST_JFNK)
41 return;
42
43 for (const auto i : make_range(_problem->numNonlinearSystems()))
44 {
45 auto & nl = _problem->getNonlinearSystemBase(i);
46 auto & dgs = nl.getDGKernelWarehouse();
47 auto & iks = nl.getInterfaceKernelWarehouse();
48
49 // to reduce typing
50 auto algebraic = Moose::RelationshipManagerType::ALGEBRAIC;
51 auto coupling = Moose::RelationshipManagerType::COUPLING;
52
53 // If we have any DGKernels or InterfaceKernels we need one layer of sparsity
54 if (dgs.size() || iks.size())
55 {
56 // We are going to add the algebraic ghosting and coupling functors one at a time because then
57 // we can keep track of whether we need to redistribute the dofs or not
58
59 // Add the algebraic ghosting functors if we are running in parallel
60 if (_communicator.size() > 1 &&
61 addRelationshipManagers(algebraic, RelationshipManager::oneLayerGhosting(algebraic)))
62 _app.attachRelationshipManagers(algebraic);
63
64 // Add the coupling functor. This plays a role regardless of whether we are running serially
65 // or in parallel
66 if (addRelationshipManagers(coupling, RelationshipManager::oneLayerGhosting(coupling)))
67 _app.attachRelationshipManagers(coupling);
68
69 // If any nonlinear system added ghosting, then in our current MOOSE ghosting strategy we
70 // added it for all systems so we can safely exit this loop
71 break;
72 }
73 }
74}
