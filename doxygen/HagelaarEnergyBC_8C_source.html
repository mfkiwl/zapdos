46HagelaarEnergyBC::computeQpResidual()
47{
48 if (_normals[_qp] * -1.0 * -_grad_potential[_qp] > 0.0)
49 {
50 _a = 1.0;
51 }
52 else
53 {
54 _a = 0.0;
55 }
56 _v_thermal =
57 std::sqrt(8 * _e[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]) / (M_PI * _massem[_qp]));
58
59 return _test[_i][_qp] * _r_units * (1. - _r) / (1. + _r) *
60 (-(2. * _a - 1.) * _mumean_en[_qp] * -_grad_potential[_qp] * _r_units * _normals[_qp] +
61 5. / 6. * _v_thermal) *
62 std::exp(_u[_qp]);
63}
