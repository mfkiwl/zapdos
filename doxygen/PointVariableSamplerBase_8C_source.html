70void
71PointVariableSamplerBase::execute()
72{
73 BoundingBox bbox = _mesh.getInflatedProcessorBoundingBox();
74
76 std::vector<Point> point_vec(1);
77
78 for (MooseIndex(_points) i = 0; i < _points.size(); ++i)
79 {
80 Point & p = _points[i];
81
82 // Do a bounding box check so we're not doing unnecessary PointLocator lookups
83 // In the discontinuous case all ranks must proceed to get a global consensus
84 // on who owns face points in getLocalElemContainingPoint()
85 if (bbox.contains_point(p) || _discontinuous_at_faces)
86 {
87 auto & values = _point_values[i];
88
89 if (values.empty())
90 values.resize(_coupled_moose_vars.size());
91
92 // First find the element the hit lands in
93 const Elem * elem = getLocalElemContainingPoint(p);
94
95 if (elem)
96 {
97 // We have to pass a vector of points into reinitElemPhys
98 point_vec[0] = p;
99
100 _subproblem.setCurrentSubdomainID(elem, 0);
101 _subproblem.reinitElemPhys(elem, point_vec, 0); // Zero is for tid
102
103 for (MooseIndex(_coupled_moose_vars) j = 0; j < _coupled_moose_vars.size(); ++j)
104 values[j] = (dynamic_cast<MooseVariableField<Real> *>(_coupled_moose_vars[j]))->sln()[0] *
105 _pp_value; // The zero is for the "qp"
106
107 _found_points[i] = true;
108 }
109 }
110 }
