108bool
109RelationshipManager::baseGreaterEqual(const RelationshipManager & rhs) const
110{
111 // !attach early = attach late
112 // If we are supposed to be attached late, then we should take precedence over the rhs. You can
113 // think of this as being >= because we if we are attaching late, then we are asking that *all*
114 // elements be geometrically ghosted during the initial mesh preparation phase. We will only allow
115 // remote elements to be deleted later on after addition of late geomeric ghosting functors (at
116 // the same time as algebraic and coupling)
117 return isType(rhs._rm_type) && (!_attach_geometric_early >= !rhs._attach_geometric_early);
118}

154void
155RelationshipManager::init(const MeshBase & mesh, const DofMap * const dof_map)
156{
157 mooseAssert(_dof_map ? dof_map == _dof_map : true,
158 "Trying to initialize with a different dof map");
159
160 _dof_map = dof_map;
161
162 // It is conceivable that this init method gets called twice, once during early geometric setup
163 // and later when we're doing algebraic/coupling (and late geometric) setup. There might be new
164 // information available during the latter call that the derived RelationshipManager can
165 // leverage, so we should make sure we call through to internal init both times
166 internalInitWithMesh(mesh);
167
168 // One would hope that internalInitWithMesh set the mesh, but we can't be sure
169 set_mesh(&mesh);
170
171 _inited = true;
172}
