85void
86PatternedMesh::buildMesh()
87{
88 // Read in all of the meshes
89 for (MooseIndex(_files) i = 0; i < _files.size(); ++i)
90 {
91 _meshes.emplace_back(std::make_unique<ReplicatedMesh>(_communicator));
92 auto & mesh = _meshes.back();
93
94 mesh->read(_files[i]);
95 }
96
97 // Create a mesh for all n-1 rows, the first row is the original mesh
98 _row_meshes.reserve(_pattern.size() - 1);
99 for (MooseIndex(_pattern) i = 0; i < _pattern.size() - 1; ++i)
100 _row_meshes.emplace_back(std::make_unique<ReplicatedMesh>(_communicator));
101
102 // Local pointers to simplify algorithm
103 std::vector<ReplicatedMesh *> row_meshes;
104 row_meshes.reserve(_pattern.size());
105 // First row is the original mesh
106 row_meshes.push_back(static_cast<ReplicatedMesh *>(&getMesh()));
107 // Copy the remaining raw pointers into the local vector
108 for (const auto & row_mesh : _row_meshes)
109 row_meshes.push_back(row_mesh.get());
110
111 BoundaryID left = getBoundaryID(getParam<BoundaryName>("left_boundary"));
112 BoundaryID right = getBoundaryID(getParam<BoundaryName>("right_boundary"));
113 BoundaryID top = getBoundaryID(getParam<BoundaryName>("top_boundary"));
114 BoundaryID bottom = getBoundaryID(getParam<BoundaryName>("bottom_boundary"));
115
116 // Build each row mesh
117 for (MooseIndex(_pattern) i = 0; i < _pattern.size(); ++i)
118 for (MooseIndex(_pattern) j = 0; j < _pattern[i].size(); ++j)
119 {
120 Real deltax = j * _x_width, deltay = i * _y_width;
121
122 // If this is the first cell of the row initialize the row mesh
123 if (j == 0)
124 {
125 row_meshes[i]->read(_files[_pattern[i][j]]);
126
127 MeshTools::Modification::translate(*row_meshes[i], deltax, -deltay, 0);
128
129 continue;
130 }
131
132 ReplicatedMesh & cell_mesh = *_meshes[_pattern[i][j]];
133
134 // Move the mesh into the right spot. -i because we are starting at the top
135 MeshTools::Modification::translate(cell_mesh, deltax, -deltay, 0);
136
137 row_meshes[i]->stitch_meshes(dynamic_cast<ReplicatedMesh &>(cell_mesh),
138 right,
139 left,
140 TOLERANCE,
141 /*clear_stitched_boundary_ids=*/true);
142
143 // Undo the translation
144 MeshTools::Modification::translate(cell_mesh, -deltax, deltay, 0);
145 }
146
147 // Now stitch together the rows
148 // We're going to stitch them all to row 0 (which is the real mesh)
149 for (MooseIndex(_pattern) i = 1; i < _pattern.size(); i++)
150 row_meshes[0]->stitch_meshes(
151 *row_meshes[i], bottom, top, TOLERANCE, /*clear_stitched_boundary_ids=*/true);
152}
