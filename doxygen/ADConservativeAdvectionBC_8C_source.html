64ADReal
65ADConservativeAdvectionBC::computeQpResidual()
66{
67 const auto vdotn =
68 (_velocity_mat_prop ? (*_velocity_mat_prop)[_qp]
69 : ADRealVectorValue(_velocity_function->vectorValue(_t, _q_point[_qp]))) *
70 _normals[_qp];
71
72 if (_primal_dirichlet)
73 return _test[_i][_qp] * vdotn * _primal_dirichlet->value(_t, _q_point[_qp]) *
74 _primal_coeff[_qp];
75 else
76 return _test[_i][_qp] * vdotn * _adv_quant[_qp];
77}
