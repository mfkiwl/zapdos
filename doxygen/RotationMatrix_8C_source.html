13RotationMatrix::rotVecToZ(RealVectorValue vec)
14// provides a rotation matrix that will rotate the vector vec to the z axis (the "2" direction)
15{
16 // ensure that vec is normalised
17 vec /= std::sqrt(vec * vec);
18
19 // construct v0 and v1 to be orthonormal to vec
20 // and form a RH basis, that is, so v1 x vec = v0
21
22 // Use Gram-Schmidt method to find v1.
23 RealVectorValue v1;
24 // Need a prototype for v1 first, and this is done by looking at the smallest component of vec
25 RealVectorValue w(std::abs(vec(0)), std::abs(vec(1)), std::abs(vec(2)));
26 if ((w(2) >= w(1) && w(1) >= w(0)) || (w(1) >= w(2) && w(2) >= w(0)))
27 // vec(0) is the smallest component
28 v1(0) = 1;
29 else if ((w(2) >= w(0) && w(0) >= w(1)) || (w(0) >= w(2) && w(2) >= w(1)))
30 // vec(1) is the smallest component
31 v1(1) = 1;
32 else
33 // vec(2) is the smallest component
34 v1(2) = 1;
35 // now Gram-Schmidt
36 v1 -= (v1 * vec) * vec;
37 v1 /= std::sqrt(v1 * v1);
38
39 // now use v0 = v1 x vec
40 RealVectorValue v0;
41 v0(0) = v1(1) * vec(2) - v1(2) * vec(1);
42 v0(1) = v1(2) * vec(0) - v1(0) * vec(2);
43 v0(2) = v1(0) * vec(1) - v1(1) * vec(0);
44
45 // the desired rotation matrix is just
46 RealTensorValue rot(v0(0), v0(1), v0(2), v1(0), v1(1), v1(2), vec(0), vec(1), vec(2));
47 return rot;
48}

51RotationMatrix::rotVec1ToVec2(RealVectorValue vec1, RealVectorValue vec2)
52// provides a rotation matrix that will rotate the vector vec1 to the vector vec2
53{
54 RealTensorValue rot1_to_z = rotVecToZ(vec1);
55 RealTensorValue rot2_to_z = rotVecToZ(vec2);
56 return rot2_to_z.transpose() * rot1_to_z;
57}
