159void
160CommonOutputAction::act()
161{
162 if (_current_task == "common_output")
163 {
164 // Store the common output parameters in the OutputWarehouse
165 _app.getOutputWarehouse().setCommonParameters(&_pars);
166
167// Create the actions for the short-cut methods
168#ifdef LIBMESH_HAVE_EXODUS_API
169 if (getParam<bool>("exodus"))
170 create("Exodus");
171#else
172 if (getParam<bool>("exodus"))
173 mooseWarning("Exodus output requested but not enabled through libMesh");
174#endif
175
176#ifdef LIBMESH_HAVE_NEMESIS_API
177 if (getParam<bool>("nemesis"))
178 create("Nemesis");
179#else
180 if (getParam<bool>("nemesis"))
181 mooseWarning("Nemesis output requested but not enabled through libMesh");
182#endif
183
184 // Only create a Console if screen output was not created
185 if (getParam<bool>("console") && !hasConsole())
186 create("Console");
187
188 if (getParam<bool>("csv"))
189 create("CSV");
190
191 if (getParam<bool>("xml"))
192 create("XMLOutput");
193
194 if (getParam<bool>("json"))
195 create("JSON");
196
197#ifdef LIBMESH_HAVE_VTK
198 if (getParam<bool>("vtk"))
199 create("VTK");
200#else
201 if (getParam<bool>("vtk"))
202 mooseWarning("VTK output requested but not enabled through libMesh");
203#endif
204
205 if (getParam<bool>("xda"))
206 create("XDA");
207
208 if (getParam<bool>("xdr"))
209 create("XDR");
210
211 if (getParam<bool>("checkpoint"))
212 create("Checkpoint");
213
214 if (getParam<bool>("gmv"))
215 create("GMV");
216
217 if (getParam<bool>("tecplot"))
218 create("Tecplot");
219
220 if (getParam<bool>("gnuplot"))
221 create("Gnuplot");
222
223 if (getParam<bool>("solution_history"))
224 create("SolutionHistory");
225
226 if (getParam<bool>("progress"))
227 create("Progress");
228
229 if (getParam<bool>("dofmap"))
230 create("DOFMap");
231
232 if (getParam<bool>("controls") || _app.getParam<bool>("show_controls"))
233 create("ControlOutput");
234
235 if (!_app.getParam<bool>("no_timing") &&
236 (getParam<bool>("perf_graph") || getParam<bool>("print_perf_log") ||
237 _app.getParam<bool>("timing")))
238 create("PerfGraphOutput");
239
240 if (!_app.getParam<bool>("no_timing") && getParam<bool>("perf_graph_live"))
241 perfGraph().setLivePrintActive(true);
242 else
243 perfGraph().setLivePrintActive(false);
244
245 perfGraph().setLiveTimeLimit(getParam<Real>("perf_graph_live_time_limit"));
246 perfGraph().setLiveMemoryLimit(getParam<unsigned int>("perf_graph_live_mem_limit"));
247
248 if (!getParam<bool>("color"))
249 Moose::setColorConsole(false);
250 }
251 else if (_current_task == "add_output")
252 {
253 // More like remove output. This can't be done during the "common_output" task because the
254 // finite element problem is not yet constructed
255
256 if (!getParam<bool>("console") || (isParamValid("print_nonlinear_converged_reason") &&
257 !getParam<bool>("print_nonlinear_converged_reason")))
258 Moose::PetscSupport::disableNonlinearConvergedReason(*_problem);
259
260 if (!getParam<bool>("console") || (isParamValid("print_linear_converged_reason") &&
261 !getParam<bool>("print_linear_converged_reason")))
262 Moose::PetscSupport::disableLinearConvergedReason(*_problem);
263 }
264 else
265 mooseError("unrecognized task ", _current_task, " in CommonOutputAction.");
266}

268void
269CommonOutputAction::create(std::string object_type)
270{
271 // Set the 'type =' parameters for the desired object
272 _action_params.set<std::string>("type") = object_type;
273
274 // Create the complete object name (uses lower case of type)
275 std::transform(object_type.begin(), object_type.end(), object_type.begin(), ::tolower);
276
277 // Create the action
278 std::shared_ptr<MooseObjectAction> action = std::static_pointer_cast<MooseObjectAction>(
279 _action_factory.create("AddOutputAction", object_type, _action_params));
280
281 // Set flag indicating that the object to be created was created with short-cut syntax
282 action->getObjectParams().set<bool>("_built_by_moose") = true;
283
284 // Add the action to the warehouse
285 _awh.addActionBlock(action);
286}

288bool
289CommonOutputAction::hasConsole()
290{
291
292 // Loop through all of the actions for adding output objects
293 for (ActionIterator it = _awh.actionBlocksWithActionBegin("add_output");
294 it != _awh.actionBlocksWithActionEnd("add_output");
295 it++)
296 {
297 MooseObjectAction * moa = dynamic_cast<MooseObjectAction *>(*it);
298 if (!moa)
299 continue;
300
301 const std::string & type = moa->getMooseObjectType();
302 InputParameters & params = moa->getObjectParams();
303 if (type.compare("Console") == 0 && params.get<bool>("output_screen"))
304 return true;
305 }
306
307 return false;
308}
