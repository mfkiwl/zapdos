33DGKernel::DGKernel(const InputParameters & parameters)
34 : DGKernelBase(parameters),
35 NeighborMooseVariableInterface(
36 this, false, Moose::VarKindType::VAR_NONLINEAR, Moose::VarFieldType::VAR_FIELD_STANDARD),
37 _var(*mooseVariable()),
38 _u(_is_implicit ? _var.sln() : _var.slnOld()),
39 _grad_u(_is_implicit ? _var.gradSln() : _var.gradSlnOld()),
40
41 _phi(_assembly.phiFace(_var)),
42 _grad_phi(_assembly.gradPhiFace(_var)),
43
44 _test(_var.phiFace()),
45 _grad_test(_var.gradPhiFace()),
46
47 _phi_neighbor(_assembly.phiFaceNeighbor(_var)),
48 _grad_phi_neighbor(_assembly.gradPhiFaceNeighbor(_var)),
49
50 _test_neighbor(_var.phiFaceNeighbor()),
51 _grad_test_neighbor(_var.gradPhiFaceNeighbor()),
52
53 _u_neighbor(_is_implicit ? _var.slnNeighbor() : _var.slnOldNeighbor()),
54 _grad_u_neighbor(_is_implicit ? _var.gradSlnNeighbor() : _var.gradSlnOldNeighbor())
55{
56 addMooseVariableDependency(mooseVariable());
57
58 _save_in.resize(_save_in_strings.size());
59 _diag_save_in.resize(_diag_save_in_strings.size());
60
61 for (unsigned int i = 0; i < _save_in_strings.size(); i++)
62 {
63 MooseVariableFEBase * var = &_subproblem.getVariable(_tid,
64 _save_in_strings[i],
65 Moose::VarKindType::VAR_AUXILIARY,
66 Moose::VarFieldType::VAR_FIELD_STANDARD);
67
68 if (_sys.hasVariable(_save_in_strings[i]))
69 mooseError("Trying to use solution variable " + _save_in_strings[i] +
70 " as a save_in variable in " + name());
71
72 if (var->feType() != _var.feType())
73 paramError(
74 "save_in",
75 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
76 moose::internal::incompatVarMsg(*var, _var));
77
78 _save_in[i] = var;
79 var->sys().addVariableToZeroOnResidual(_save_in_strings[i]);
80 addMooseVariableDependency(var);
81 }
82
83 _has_save_in = _save_in.size() > 0;
84
85 for (unsigned int i = 0; i < _diag_save_in_strings.size(); i++)
86 {
87 MooseVariableFEBase * var = &_subproblem.getVariable(_tid,
88 _diag_save_in_strings[i],
89 Moose::VarKindType::VAR_AUXILIARY,
90 Moose::VarFieldType::VAR_FIELD_STANDARD);
91
92 if (_sys.hasVariable(_diag_save_in_strings[i]))
93 mooseError("Trying to use solution variable " + _diag_save_in_strings[i] +
94 " as a diag_save_in variable in " + name());
95
96 if (var->feType() != _var.feType())
97 paramError(
98 "diag_save_in",
99 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
100 moose::internal::incompatVarMsg(*var, _var));
101
102 _diag_save_in[i] = var;
103 var->sys().addVariableToZeroOnJacobian(_diag_save_in_strings[i]);
104 addMooseVariableDependency(var);
105 }
106
107 _has_diag_save_in = _diag_save_in.size() > 0;
108}

110void
111DGKernel::computeElemNeighResidual(Moose::DGResidualType type)
112{
113 bool is_elem;
114 if (type == Moose::Element)
115 is_elem = true;
116 else
117 is_elem = false;
118
119 const VariableTestValue & test_space = is_elem ? _test : _test_neighbor;
120
121 if (is_elem)
122 prepareVectorTag(_assembly, _var.number());
123 else
124 prepareVectorTagNeighbor(_assembly, _var.number());
125
126 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
127 {
128 precalculateQpResidual(type);
129 for (_i = 0; _i < test_space.size(); _i++)
130 _local_re(_i) += _JxW[_qp] * _coord[_qp] * computeQpResidual(type);
131 }
132
133 accumulateTaggedLocalResidual();
134
135 if (_has_save_in)
136 {
137 Threads::spin_mutex::scoped_lock lock(_resid_vars_mutex);
138 for (const auto & var : _save_in)
139 {
140 const std::vector<dof_id_type> & dof_indices =
141 is_elem ? var->dofIndices() : var->dofIndicesNeighbor();
142 var->sys().solution().add_vector(_local_re, dof_indices);
143 }
144 }
145}

147void
148DGKernel::computeElemNeighJacobian(Moose::DGJacobianType type)
149{
150 const VariableTestValue & test_space =
151 (type == Moose::ElementElement || type == Moose::ElementNeighbor) ? _test : _test_neighbor;
152 const VariableTestValue & loc_phi =
153 (type == Moose::ElementElement || type == Moose::NeighborElement) ? _phi : _phi_neighbor;
154
155 if (type == Moose::ElementElement)
156 prepareMatrixTag(_assembly, _var.number(), _var.number());
157 else
158 prepareMatrixTagNeighbor(_assembly, _var.number(), _var.number(), type);
159
160 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
161 {
162 precalculateQpJacobian(type);
163 for (_i = 0; _i < test_space.size(); _i++)
164 for (_j = 0; _j < loc_phi.size(); _j++)
165 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeQpJacobian(type);
166 }
167
168 accumulateTaggedLocalMatrix();
169
170 if (_has_diag_save_in && (type == Moose::ElementElement || type == Moose::NeighborNeighbor))
171 {
172 unsigned int rows = _local_ke.m();
173 DenseVector<Number> diag(rows);
174 for (unsigned int i = 0; i < rows; i++)
175 diag(i) = _local_ke(i, i);
176
177 Threads::spin_mutex::scoped_lock lock(_jacoby_vars_mutex);
178 for (const auto & var : _diag_save_in)
179 {
180 if (type == Moose::ElementElement)
181 var->sys().solution().add_vector(diag, var->dofIndices());
182 else
183 var->sys().solution().add_vector(diag, var->dofIndicesNeighbor());
184 }
185 }
186}

188void
189DGKernel::computeOffDiagElemNeighJacobian(Moose::DGJacobianType type,
190 const MooseVariableFEBase & jvar)
191{
192 const VariableTestValue & test_space =
193 (type == Moose::ElementElement || type == Moose::ElementNeighbor) ? _test : _test_neighbor;
194 const VariableTestValue & loc_phi =
195 (type == Moose::ElementElement || type == Moose::NeighborElement) ? _phi : _phi_neighbor;
196
197 if (type == Moose::ElementElement)
198 prepareMatrixTag(_assembly, _var.number(), jvar.number());
199 else
200 prepareMatrixTagNeighbor(_assembly, _var.number(), jvar.number(), type);
201
202 if (_local_ke.n() == 0 || _local_ke.m() == 0)
203 return;
204
205 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
206 {
207 precalculateQpOffDiagJacobian(type, jvar);
208 for (_i = 0; _i < test_space.size(); _i++)
209 for (_j = 0; _j < loc_phi.size(); _j++)
210 _local_ke(_i, _j) +=
211 _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobian(type, jvar.number());
212 }
213
214 accumulateTaggedLocalMatrix();
215}
