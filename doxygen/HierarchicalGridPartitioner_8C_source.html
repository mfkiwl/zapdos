71void
72HierarchicalGridPartitioner::_do_partition(MeshBase & mesh, const unsigned int /*n*/)
73{
74 const auto dim = mesh.spatial_dimension();
75 if (_ny_procs == 0 && dim > 1)
76 paramError("ny_procs", "Required for ", dim, "D meshes");
77 if (_ny_nodes == 0 && dim > 1)
78 paramError("ny_nodes", "Required for ", dim, "D meshes");
79 if (_nz_procs == 0 && dim == 3)
80 paramError("nz_procs", "Required for 3D meshes");
81 if (_nz_nodes == 0 && dim == 3)
82 paramError("nz_nodes", "Required for 3D meshes");
83
84 auto total_nodes = _nx_nodes;
85
86 if (mesh.spatial_dimension() >= 2)
87 total_nodes *= _ny_nodes;
88
89 if (mesh.spatial_dimension() == 3)
90 total_nodes *= _nz_nodes;
91
92 auto procs_per_node = _nx_procs;
93
94 if (mesh.spatial_dimension() >= 2)
95 procs_per_node *= _ny_procs;
96
97 if (mesh.spatial_dimension() == 3)
98 procs_per_node *= _nz_procs;
99
100 if (procs_per_node * total_nodes != mesh.n_partitions())
101 mooseError("Partitioning in ",
102 name(),
103 " doesn't add up to the total number of processors: ",
104 procs_per_node * total_nodes);
105
106 // Figure out the physical bounds of the given mesh
107 auto nodes_bounding_box = MeshTools::create_bounding_box(mesh);
108 const auto & nodes_min = nodes_bounding_box.min();
109 const auto & nodes_max = nodes_bounding_box.max();
110
111 // Bound the coarse mesh (n_nodes * n_nodes)
112 auto nodes_mesh = std::make_unique<ReplicatedMesh>(this->_communicator);
113 nodes_mesh->partitioner() = std::make_unique<LinearPartitioner>();
114
115 if (mesh.spatial_dimension() == 2)
116 MeshTools::Generation::build_cube(*nodes_mesh,
117 _nx_nodes,
118 _ny_nodes,
119 _nz_nodes,
120 nodes_min(0),
121 nodes_max(0),
122 nodes_min(1),
123 nodes_max(1),
124 nodes_min(2),
125 nodes_max(2),
126 QUAD4);
127 else
128 MeshTools::Generation::build_cube(*nodes_mesh,
129 _nx_nodes,
130 _ny_nodes,
131 _nz_nodes,
132 nodes_min(0),
133 nodes_max(0),
134 nodes_min(1),
135 nodes_max(1),
136 nodes_min(2),
137 nodes_max(2),
138 HEX8);
139
140 // Now build the procs meshes
141 std::vector<std::unique_ptr<ReplicatedMesh>> procs_meshes(nodes_mesh->n_elem());
142
143 for (const auto & elem_ptr : nodes_mesh->active_element_ptr_range())
144 {
145 // Need to find the bounds of the elem
146 Point min(std::numeric_limits<Real>::max(),
147 std::numeric_limits<Real>::max(),
148 std::numeric_limits<Real>::max());
149 Point max(-std::numeric_limits<Real>::max(),
150 -std::numeric_limits<Real>::max(),
151 -std::numeric_limits<Real>::max());
152
153 // Loop over all the nodes
154 for (const auto & node : elem_ptr->node_ref_range())
155 {
156 min(0) = std::min(min(0), node(0));
157 min(1) = std::min(min(1), node(1));
158 min(2) = std::min(min(2), node(2));
159
160 max(0) = std::max(max(0), node(0));
161 max(1) = std::max(max(1), node(1));
162 max(2) = std::max(max(2), node(2));
163 }
164
165 auto procs_mesh = std::make_unique<ReplicatedMesh>(this->_communicator);
166 procs_mesh->partitioner() = std::make_unique<LinearPartitioner>();
167
168 if (mesh.spatial_dimension() == 2)
169 MeshTools::Generation::build_cube(*procs_mesh,
170 _nx_procs,
171 _ny_procs,
172 _nz_procs,
173 min(0),
174 max(0),
175 min(1),
176 max(1),
177 min(2),
178 max(2),
179 QUAD4);
180 else
181 MeshTools::Generation::build_cube(*procs_mesh,
182 _nx_procs,
183 _ny_procs,
184 _nz_procs,
185 min(0),
186 max(0),
187 min(1),
188 max(1),
189 min(2),
190 max(2),
191 HEX8);
192
193 procs_meshes[elem_ptr->id()] = std::move(procs_mesh);
194 }
195
196 auto nodes_point_locator_ptr = nodes_mesh->sub_point_locator();
197
198 std::vector<std::unique_ptr<PointLocatorBase>> procs_point_locators(procs_meshes.size());
199
200 for (unsigned int i = 0; i < procs_meshes.size(); i++)
201 procs_point_locators[i] = procs_meshes[i]->sub_point_locator();
202
203 // Loop over all of the elements in the given mesh
204 for (auto & elem_ptr : mesh.active_element_ptr_range())
205 {
206 auto elem_centroid = elem_ptr->vertex_average();
207
208 // Find the element it lands in in the Nodes mesh
209 auto nodes_elem_ptr = (*nodes_point_locator_ptr)(elem_centroid);
210
211 auto nodes_elem_id = nodes_elem_ptr->id();
212
213 // True if we found something
214 if (nodes_elem_ptr)
215 {
216 // Now see where it lands within the procs mesh of that node
217 auto procs_elem_ptr = (*procs_point_locators[nodes_elem_id])(elem_centroid);
218
219 // Assign the _id_ of the cell to the processor_id (plus an offset for which node we're in)
220 elem_ptr->processor_id() = procs_elem_ptr->id() + (nodes_elem_id * procs_per_node);
221 }
222 else // Should never happen (seriously - we create bounding boxes that should disallow this!)
223 mooseError("HierarchicalGridPartitioner unable to locate element within the grid!");
224 }
225}
