54void
55BlockWeightedPartitioner::initialize(MeshBase & mesh)
56{
57 // Get the IDs from the supplied names
58 const auto block_ids = MooseMeshUtils::getSubdomainIDs(mesh, _blocks);
59
60 // Make sure all of the blocks exist
61 std::set<subdomain_id_type> mesh_block_ids;
62 mesh.subdomain_ids(mesh_block_ids);
63 for (const auto block_id : block_ids)
64 if (!mesh_block_ids.count(block_id))
65 paramError("block", "The block ", block_id, " was not found on the mesh");
66
67 // Setup the block -> weight map for use in computeElementWeight
68 _blocks_to_weights.reserve(_weights.size());
69 for (MooseIndex(block_ids.size()) i = 0; i < block_ids.size(); i++)
70 _blocks_to_weights[block_ids[i]] = _weights[i];
71}
