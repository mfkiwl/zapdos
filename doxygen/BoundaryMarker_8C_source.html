42Marker::MarkerValue
43BoundaryMarker::computeElementMarker()
44{
45 if (_distance == 0.0)
46 {
47 // is the current element member of any selected boundary element set?
48 for (const auto boundary : _boundary_ids)
49 if (_mesh.isBoundaryElem(_current_elem->id(), boundary))
50 return _mark;
51
52 return DONT_MARK;
53 }
54 else
55 {
56 for (const auto boundary : _boundary_ids)
57 {
58 const auto it = _bnd_elem_ids.find(boundary);
59 if (it != _bnd_elem_ids.end())
60 for (const auto id : it->second)
61 {
62 // shortcut if we are checing the current element itself
63 if (id == _current_elem->id())
64 return _mark;
65
66 // otherwise compute distance to the boundary elements
67 const auto elem = _mesh.elemPtr(id);
68 const auto r = _current_elem->vertex_average() - elem->vertex_average();
69 if (r.norm() < _distance)
70 return _mark;
71 }
72 }
73 return DONT_MARK;
74 }
75}
