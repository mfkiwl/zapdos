44const MeshGenerator &
45MeshGeneratorSystem::appendMeshGenerator(const std::string & type,
46 const std::string & name,
47 InputParameters params)
48{
49 if (!appendingMeshGenerators())
50 mooseError("Can only call appendMeshGenerator() during the append_mesh_generator task");
51
52 // Make sure this mesh generator has one and _only_ one input, as "input"
53 const auto param_name_mg_name_pairs = getMeshGeneratorParamDependencies(params, true);
54 if (param_name_mg_name_pairs.size() != 1 || param_name_mg_name_pairs[0].first != "input")
55 mooseError("While adding ",
56 type,
57 " '",
58 name,
59 "' via appendMeshGenerator():\nCan only append a mesh generator that takes a "
60 "single input mesh generator via the parameter named 'input'");
61
62 // If no final generator is set, we need to make sure that we have one; we will hit
63 // this the first time we add an appended MeshGenerator and only need to do it once.
64 // We'll then generate the final ordering within the execution phase. We'll also
65 // clear the ordering because it could be invalid if we append any more generators,
66 // and we'll be re-ordering within executeMeshgenerators() anyway (where we don't
67 // keep track of any state for the sake of simpler logic)
68 if (_final_generator_name.empty())
69 {
70 if (_mesh_generators.empty())
71 mooseError("Cannot use appendMeshGenerator() because there is not a generator to append to!");
72
73 createMeshGeneratorOrder();
74 _ordered_mesh_generators.clear();
75 }
76
77 // Set the final generator as the input
78 mooseAssert(hasMeshGenerator(_final_generator_name), "Missing final generator");
79 params.set<MeshGeneratorName>("input") = _final_generator_name;
80
81 // Keep track of the new final generator
82 _final_generator_name = name;
83
84 // Need to add this to the param map so that createMeshGenerator can use it
85 mooseAssert(!_mesh_generator_params.count(name), "Already exists");
86 _mesh_generator_params.emplace(
87 std::piecewise_construct, std::forward_as_tuple(name), std::forward_as_tuple(type, params));
88
89 return *createMeshGenerator(name);
90}

92std::vector<std::pair<std::string, MeshGeneratorName>>
93MeshGeneratorSystem::getMeshGeneratorParamDependencies(const InputParameters & params,
94 const bool allow_empty /* = false */) const
95{
96 std::vector<std::pair<std::string, MeshGeneratorName>> dependencies;
97
98 auto add_dependency =
99 [&dependencies, &allow_empty](const auto & param_name, const auto & dependency)
100 {
101 if (dependency.size() || allow_empty)
102 dependencies.emplace_back(param_name, dependency);
103 };
104
105 for (const auto & [name, param] : params)
106 if (const auto dependency =
107 dynamic_cast<const Parameters::Parameter<MeshGeneratorName> *>(param.get()))
108 add_dependency(name, dependency->get());
109 else if (const auto dependencies =
110 dynamic_cast<const Parameters::Parameter<std::vector<MeshGeneratorName>> *>(
111 param.get()))
112 {
113 if (allow_empty && dependencies->get().empty())
114 add_dependency(name, std::string());
115 for (const auto & dependency : dependencies->get())
116 add_dependency(name, dependency);
117 }
118
119 return dependencies;
120}

122void
123MeshGeneratorSystem::createAddedMeshGenerators()
124{
125 mooseAssert(_app.actionWarehouse().getCurrentTaskName() == "create_added_mesh_generators",
126 "Should not run now");
127
128 // No generators were added via addMeshGenerator()
129 if (_mesh_generator_params.empty())
130 return;
131
132 DependencyResolver<std::string> resolver;
133
134 // Define the dependencies known so far
135 for (const auto & [name, type_params_pair] : _mesh_generator_params)
136 {
137 resolver.addItem(name);
138 for (const auto & param_dependency_pair :
139 getMeshGeneratorParamDependencies(type_params_pair.second))
140 resolver.addEdge(param_dependency_pair.second, name);
141 }
142
143 std::vector<std::vector<std::string>> ordered_generators;
144 try
145 {
146 ordered_generators = resolver.getSortedValuesSets();
147 }
148 catch (CyclicDependencyException<std::string> & e)
149 {
150 mooseError("Cyclic dependencies detected in mesh generation: ",
151 MooseUtils::join(e.getCyclicDependencies(), " <- "));
152 }
153
154 // Construct all of the mesh generators that we know exist
155 for (const auto & generator_names : ordered_generators)
156 for (const auto & generator_name : generator_names)
157 if (_mesh_generator_params.count(generator_name))
158 createMeshGenerator(generator_name);
159
160 mooseAssert(_mesh_generator_params.empty(), "Should be empty");
161 mooseAssert(_final_generator_name.empty(), "Should be unset at this point");
162
163 // Set the final generator if we have one set by the user
164 // and if so make sure it also exists
165 const auto & moose_mesh = _app.actionWarehouse().getMesh();
166 if (moose_mesh->parameters().have_parameter<std::string>("final_generator") &&
167 moose_mesh->isParamValid("final_generator"))
168 {
169 mooseAssert(moose_mesh->type() == "MeshGeneratorMesh",
170 "Assumption for mesh type is now invalid");
171
172 _final_generator_name = moose_mesh->getParam<std::string>("final_generator");
173 if (!hasMeshGenerator(_final_generator_name))
174 moose_mesh->paramError("final_generator",
175 "The forced final MeshGenerator '",
176 _final_generator_name,
177 "' does not exist");
178 }

181void
182MeshGeneratorSystem::createMeshGeneratorOrder()
183{
184 mooseAssert(_app.constructingMeshGenerators() ||
185 _app.actionWarehouse().getCurrentTaskName() == "execute_mesh_generators",
186 "Incorrect call time");
187 mooseAssert(_ordered_mesh_generators.empty(), "Already ordered");
188 mooseAssert(_mesh_generators.size(), "No mesh generators to order");
189
190 TIME_SECTION("createMeshGeneratorOrder", 1, "Ordering Mesh Generators");
191
192 // We dare not sort these based on address!
193 DependencyResolver<MeshGenerator *, MeshGenerator::Comparator> resolver;
194
195 // The mesh generators that must be called
196 // This is needed to mark the generators that could be cut due to a
197 // final generator being set, but should still be called because they're
198 // either being saved in memory or as output
199 std::vector<std::string> required_generators;
200 for (const auto & it : _mesh_generators)
201 if (it.second->hasSaveMesh() || it.second->hasOutput())
202 required_generators.push_back(it.second->name());
203
204 // The mesh generator tree should have all the mesh generators that
205 // The final generator depends on and all the mesh generators
206 // with 'save in' flag. Here we loop over all the mesh generators
207 // and conditionally add all of the dependencies into the resolver
208 for (const auto & it : _mesh_generators)
209 {
210 MeshGenerator * mg = it.second.get();
211
212 // The mesh generator has to meet one of the following conditions:
213 // Final mesh generator is not set, so we build the whole tree
214 if (_final_generator_name.empty() ||
215 // This mesh generator is the final generator
216 mg->name() == _final_generator_name ||
217 // This needs to be saved or output
218 mg->hasSaveMesh() || mg->hasOutput() ||
219 // Final mesh generator set and is a child of this generator
220 (_final_generator_name.size() && mg->isChildMeshGenerator(_final_generator_name, false)) ||
221 // This is a dependency of a generator that needs to be saved or output
222 std::find_if(required_generators.begin(),
223 required_generators.end(),
224 [&mg](const auto & name) { return mg->isChildMeshGenerator(name, false); }) !=
225 required_generators.end())
226 {
227 resolver.addItem(mg);
228 for (const auto & dep_mg : mg->getParentMeshGenerators())
229 resolver.addEdge(&getMeshGeneratorInternal(dep_mg->name()), mg);
230 }
231 }
232
233 // ...and sort them
234 try
235 {
236 _ordered_mesh_generators = resolver.getSortedValuesSets();
237 }
238 // It is _quite_ hard to trigger this to test it. I've tried to no avail.
239 // Now that we...
240 // - check and sort up front
241 // - know if dependencies exist at the time of requesting them
242 // - require that sub generators depend only on other sub generators in an object's
243 // tree + input dependencies that we explicitly declare
244 // I don't think it's possible. But we'll leave it here anyway and it
245 // definitely will not be covered
246 catch (CyclicDependencyException<MeshGenerator *, MeshGenerator::Comparator> & e)
247 {
248 const auto & cycle = e.getCyclicDependencies();
249 std::vector<std::string> names(cycle.size());
250 for (const auto i : index_range(cycle))
251 names[i] = cycle[i]->name();
252
253 mooseError("Cyclic dependencies detected in mesh generation: ",
254 MooseUtils::join(names, " <- "));
255 }
256
257 mooseAssert(_ordered_mesh_generators.size(), "No mesh generators found");
258
259 const auto & final_generators = _ordered_mesh_generators.back();
260
261 // We haven't forced a final generator yet
262 if (_final_generator_name.empty())
263 {
264 mooseAssert(final_generators.size(), "Empty vector");
265
266 // See if we have multiple independent trees of generators
267 const auto ancestor_list = resolver.getAncestors(final_generators.back());
268 if (ancestor_list.size() != resolver.size() && _final_generator_name.empty())
269 {
270 // Need to remove duplicates and possibly perform a difference so we'll import our list
271 // into a set for these operations.
272 std::set<MeshGenerator *, MeshGenerator::Comparator> ancestors(ancestor_list.begin(),
273 ancestor_list.end());
274 // Get all of the items from the resolver so we can compare against the tree from the
275 // final generator we just pulled.
276 const auto & allValues = resolver.getSortedValues();
277 decltype(ancestors) all(allValues.begin(), allValues.end());
278
279 decltype(ancestors) ind_tree;
280 std::set_difference(all.begin(),
281 all.end(),
282 ancestors.begin(),
283 ancestors.end(),
284 std::inserter(ind_tree, ind_tree.end()));
285
286 std::ostringstream oss;
287 oss << "Your MeshGenerator tree contains multiple possible generator outputs :\n\""
288 << final_generators.back()->name()
289 << " and one or more of the following from an independent set: \"";
290 bool first = true;
291 for (const auto & gen : ind_tree)
292 {
293 if (!first)
294 oss << ", ";
295 else
296 first = false;
297
298 oss << gen->name();
299 }
300 oss << "\"\n\nThis may be due to a missing dependency or may be intentional. Please either\n"
301 "- check that all the mesh generators are connected as a tree and culminate in a "
302 "single final mesh. Having one wrong 'input=mg' parameter is the most common error\n"
303 "- add additional dependencies to remove the ambiguity if you are using a user-built "
304 "MeshGenerator\n"
305 "- if you intend to execute a subset of the defined generators (uncommon), select the"
306 " final MeshGenerator in the [Mesh] block with the \"final_generator\" parameter.";
307 mooseError(oss.str());
308 }
309
310 _final_generator_name = final_generators.back()->name();
311 }
312 else
313 mooseAssert(hasMeshGenerator(_final_generator_name), "Missing the preset final generator");
314}

316void
317MeshGeneratorSystem::executeMeshGenerators()
318{
319 libmesh_parallel_only(comm());
320
321 // we do not need to do this when there are no mesh generators
322 if (_mesh_generators.empty())
323 return;
324
325 // Order the generators
326 createMeshGeneratorOrder();
327
328 std::map<std::string, std::unique_ptr<MeshBase> *> to_save_in_meshes;
329
330 // Loop over the MeshGenerators and save all meshes marked to to_save_in_meshes
331 for (const auto & generator_set : _ordered_mesh_generators)
332 for (const auto & generator : generator_set)
333 if (generator->hasSaveMesh())
334 {
335 if (_final_generator_name == generator->name())
336 generator->paramError("save_with_name",
337 "Cannot use the save in capability with the final mesh generator");
338 to_save_in_meshes.emplace(generator->getSavedMeshName(),
339 &getMeshGeneratorOutput(generator->name()));
340 }
341
342 // Grab the outputs from the final generator so MeshGeneratorMesh can pick it up
343 to_save_in_meshes.emplace(mainMeshGeneratorName(),
344 &getMeshGeneratorOutput(_final_generator_name));
345
346 // Run the MeshGenerators in the proper order
347 for (const auto & generator_set : _ordered_mesh_generators)
348 {
349 for (const auto & generator : generator_set)
350 {
351 const auto & name = generator->name();
352
353 auto current_mesh = generator->generateInternal();
354#ifdef DEBUG
355 // Assert that the mesh is either marked as not prepared or if it is marked as prepared,
356 // that it's *actually* prepared
357 if (!_has_bmbb && !MeshTools::valid_is_prepared(*current_mesh))
358 generator->mooseError("The generated mesh is marked as prepared but is not actually "
359 "prepared. Please edit the '",
360 generator->type(),
361 "' class to call 'set_isnt_prepared()'");
362#endif
363
364 // Now we need to possibly give this mesh to downstream generators
365 auto & outputs = _mesh_generator_outputs[name];
366
367 if (outputs.size())
368 {
369 auto & first_output = *outputs.begin();
370
371 first_output = std::move(current_mesh);
372
373 const auto & copy_from = *first_output;
374
375 auto output_it = ++outputs.begin();
376
377 // For all of the rest we need to make a copy
378 for (; output_it != outputs.end(); ++output_it)
379 (*output_it) = copy_from.clone();
380 }
381 }
382 }
383
384 // Grab all the valid save in meshes from the temporary map to_save_in_meshes
385 // and store them in _save_in_meshes
386 for (auto & [name, mesh_ptr] : to_save_in_meshes)
387 {
388 mooseAssert(mesh_ptr, "Invalid pointer");
389 mooseAssert(*mesh_ptr, "Invalid pointer");
390 if (name != mainMeshGeneratorName())
391 mooseAssert(_save_in_meshes.count(name), "Mesh has not been requested for save");
392 _save_in_meshes[name] = std::move(*mesh_ptr);
393 }
394}

396std::shared_ptr<MeshGenerator>
397MeshGeneratorSystem::createMeshGenerator(const std::string & generator_name)
398{
399 libmesh_parallel_only(comm());
400 mooseAssert(_app.constructingMeshGenerators(), "Should not run now");
401
402 const auto find_params = _mesh_generator_params.find(generator_name);
403 mooseAssert(find_params != _mesh_generator_params.end(), "Not added");
404 const auto & [type, params] = find_params->second;
405 mooseAssert(comm().verify(type + generator_name), "Inconsistent construction order");
406
407 std::shared_ptr<MeshGenerator> mg =
408 _app.getFactory().create<MeshGenerator>(type, generator_name, params);
409
410 if (mg->hasSaveMesh())
411 {
412 if (_save_in_meshes.count(mg->getSavedMeshName()))
413 mg->paramError("save_with_name",
414 "The save with name '",
415 mg->getSavedMeshName(),
416 "' has already been used");
417 _save_in_meshes.emplace(mg->getSavedMeshName(), nullptr);
418 }
419
420 // Setup the children and parents
421 for (const auto & dependency : mg->getRequestedMeshGenerators())
422 {
423 // We _shouldn't_ hit this; now that we enforce construction ordering we do
424 // all of this error checking at construction time because the info is available
425 mooseAssert(hasMeshGenerator(dependency), "Missing dependency");
426
427 auto & dependency_mg = getMeshGeneratorInternal(dependency);
428 mg->addParentMeshGenerator(dependency_mg, MeshGenerator::AddParentChildKey());
429 dependency_mg.addChildMeshGenerator(*mg, MeshGenerator::AddParentChildKey());
430 }
431
432 // Loop through all of the MeshGeneratorName and std::vector<MeshGeneratorName>
433 // parameters (meshes that we should depend on), and make sure that either:
434 // - We directly depend on them and requested a mesh from them
435 // - We created a sub generator that depends on them and we declared it
436 for (const auto & param_dependency_pair : getMeshGeneratorParamDependencies(mg->parameters()))
437 {
438 const auto & param_name = param_dependency_pair.first;
439 const auto & dependency_name = param_dependency_pair.second;
440
441 if (mg->isNullMeshName(dependency_name))
442 continue;
443
444 // True if this dependency was requested and is a parent
445 if (mg->isParentMeshGenerator(dependency_name))
446 {
447 mooseAssert(mg->getRequestedMeshGenerators().count(dependency_name), "Wasn't requested");
448 continue;
449 }
450
451 // Whether or not this is a dependency of at least one SubGenerator
452 auto find_sub_dependency = std::find_if(mg->getSubMeshGenerators().begin(),
453 mg->getSubMeshGenerators().end(),
454 [&dependency_name](const auto & mg)
455 { return mg->isParentMeshGenerator(dependency_name); });
456 const auto is_sub_dependency = find_sub_dependency != mg->getSubMeshGenerators().end();
457
458 // This should be used by a sub generator
459 if (mg->getRequestedMeshGeneratorsForSub().count(dependency_name))
460 {
461 if (!is_sub_dependency)
462 mg->mooseError("The sub generator dependency declared from MeshGenerator '",
463 dependency_name,
464 "' from the parameter '",
465 param_name,
466 "' was not used.\n\nDependencies that are declared by declareMeshForSub() "
467 "must be used as an input to a sub generator created by this object.");
468 }
469 // This was used by a sub generator but wasn't declared
470 else if (is_sub_dependency)
471 mg->mooseError(
472 "The MeshGenerator '",
473 dependency_name,
474 "' was referenced in the parameter '",
475 param_name,
476 "' and used in the sub generator ",
477 (*find_sub_dependency)->type(),
478 " '",
479 (*find_sub_dependency)->name(),
480 "', but was not declared as a sub dependency.\n\nTo correct this, modify the code of ",
481 mg->type(),
482 " to include a call to declareMesh(es)ForSub(\"",
483 param_name,
484 "\") in the constructor.");
485 // Isn't used at all
486 else
487 mg->mooseError(
488 "You failed to request the generated mesh(es) for the parameter '",
489 param_name,
490 "'.\n\nIn specific, the mesh from MeshGenerator '",
491 dependency_name,
492 "' was not requested.\n\nTo correct this, you should remove the parameter if the "
493 "mesh(es)\nare not needed, or request the mesh(es) with getMesh()/getMeshes().");
494 }
495
496 mooseAssert(!_mesh_generators.count(generator_name), "Already created");
497 _mesh_generators.emplace(generator_name, mg);
498 mooseAssert(!_mesh_generator_outputs.count(generator_name), "Already exists");
499 _mesh_generator_outputs[generator_name];
500 _mesh_generator_params.erase(find_params);
501
502 return mg;
503}
