39void
40NodalStatistics::execute()
41{
42 // Get value to to update statistics
43 Real value = computeValue();
44
45 if (_max < value)
46 _max = value;
47
48 if (_min > value)
49 _min = value;
50
51 // Update the total and the number to get the average when "finalizing"
52 _average += value;
53 _number_nodes++;
54}

66void
67NodalStatistics::finalize()
68{
