45void
46PolynomialCoefficients::initialize()
47{
48 // This material follows the method outlined in:
49 // Mikhail S Finko et al 2017 J. Phys. D: Appl. Phys. 50 485201
50 // Section 2.2, Equation 13
51
52 // Read the participant species' coefficients from files
53 std::string file_name;
54 // std::vector<std::vector<Real>> polynomial_coefficients;
55 _polynomial_coefficients.resize(_participants.size());
56 _power_coefficient = 0.0;
57 for (unsigned int i = 0; i < _participants.size(); ++i)
58 {
59 _power_coefficient += _coefficients[i]; // Finko, equation 7
60 file_name = getParam<std::string>("file_location") + "/" + _participants[i] + ".txt";
61 MooseUtils::checkFileReadable(file_name);
62 const char * charPath = file_name.c_str();
63 std::ifstream myfile(charPath);
64 Real value;
65
66 if (myfile.is_open())
67 {
68 while (myfile >> value)
69 {
70 _polynomial_coefficients[i].push_back(value);
71 }
72 myfile.close();
73 }
74 else
75 {
76 mooseError("Unable to open file: " + file_name);
77 }
78 }
79}

81void
82PolynomialCoefficients::execute()
83{
84 // Sum the coefficients
85 _delta_a.resize(_polynomial_coefficients[0].size(), 0.0);
86 for (unsigned int i = 0; i < _polynomial_coefficients.size(); ++i)
87 {
88 for (unsigned int j = 0; j < _polynomial_coefficients[i].size(); ++j)
89 {
90 _delta_a[j] += _coefficients[i] * _polynomial_coefficients[i][j];
91 }
92 }
93}
