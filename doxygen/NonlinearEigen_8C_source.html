45void
46NonlinearEigen::init()
47{
48 if (_app.isRecovering())
49 {
50 _console << "\nCannot recover NonlinearEigen solves!\nExiting...\n" << std::endl;
51 return;
52 }
53
54 EigenExecutionerBase::init();
55
56 // Write the initial.
57 // Note: We need to tempararily change the system time to make the output system work properly.
58 _problem.timeStep() = 0;
59 Real t = _problem.time();
60 _problem.time() = _problem.timeStep();
61 _problem.outputStep(EXEC_INITIAL);
62 _problem.time() = t;
63
64 if (_free_iter > 0)
65 {
66 // save the initial guess
67 _problem.advanceState();
68
69 // free power iterations
70 _console << " Free power iteration starts" << std::endl;
71
72 Real initial_res;
73 inversePowerIteration(_free_iter,
74 _free_iter,
75 _free_l_tol,
76 false,
77 std::numeric_limits<Real>::min(),
78 true,
79 "",
80 std::numeric_limits<Real>::max(),
81 _eigenvalue,
82 initial_res);
83
84 _problem.onTimestepEnd();
85 _problem.execute(EXEC_TIMESTEP_END);
86
87 if (_output_after_pi)
88 {
89 // output initial guess created by free power iterations
90 _problem.timeStep()++;
91 Real t = _problem.time();
92 _problem.time() = _problem.timeStep();
93 _problem.outputStep(EXEC_TIMESTEP_END);
94 _problem.time() = t;
95 }
96 }
97}

112void
113NonlinearEigen::takeStep()
114{
115 _console << " Nonlinear iteration starts" << std::endl;
116
117 preSolve();
118 _problem.timestepSetup();
119 _problem.advanceState();
120 _problem.execute(EXEC_TIMESTEP_BEGIN);
121
122 _last_solve_converged = nonlinearSolve(_nl_rel_tol, _nl_abs_tol, _l_tol, _eigenvalue);
123 postSolve();
124
125 if (lastSolveConverged())
126 {
127 printEigenvalue();
128
129 _problem.onTimestepEnd();
130 _problem.execute(EXEC_TIMESTEP_END);
131 }
