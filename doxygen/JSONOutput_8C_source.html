43std::string
44JSONOutput::filename()
45{
46 std::ostringstream file_name;
47 file_name << _file_base;
48
49 if (_one_file_per_timestep)
50 file_name << '_' << std::setw(_padding) << std::setprecision(0) << std::setfill('0')
51 << std::right << timeStep();
52
53 if (processor_id() > 0)
54 {
55 int digits = MooseUtils::numDigits(n_processors());
56 file_name << ".json"
57 << "." << std::setw(digits) << std::setfill('0') << processor_id();
58 }
59 else
60 file_name << ".json";
61
62 return file_name.str();
63}

79void
80JSONOutput::outputReporters()
81{
82 // Set of ReporterNames for output
83 std::set<ReporterName> r_names;
84 for (const std::string & c_name : getReporterOutput())
85 r_names.emplace(c_name);
86
87 // Is there ANY distributed data
88 _has_distributed = std::any_of(
89 r_names.begin(),
90 r_names.end(),
91 [this](const ReporterName & n)
92 { return _reporter_data.hasReporterWithMode(n.getObjectName(), REPORTER_MODE_DISTRIBUTED); });
93 if (processor_id() == 0 || _has_distributed)
94 {
95 // Create the current output node
96 auto & current_node = _json["time_steps"].emplace_back();
97
98 // Add time/iteration information
99 current_node["time"] = _problem_ptr->time();
100 current_node["time_step"] = _problem_ptr->timeStep();
101 if (_execute_enum.contains(EXEC_LINEAR) && !_on_nonlinear_residual)
102 current_node["linear_iteration"] = _linear_iter;
103 if (_execute_enum.contains(EXEC_NONLINEAR))
104 current_node["nonlinear_iteration"] = _nonlinear_iter;
105
106 // Inject processor info
107 if (n_processors() > 1 && _has_distributed)
108 {
109 _json["part"] = processor_id();
110 _json["number_of_parts"] = n_processors();
111 }
112
113 // Add Reporter values to the current node
114 auto & r_node = _json["reporters"]; // non-accidental insert
115 for (const auto & r_name : r_names)
116 {
117 // If this value is produced in root mode and we're not on root, don't report this value
118 const auto & context = _reporter_data.getReporterContextBase(r_name);
119 if (context.getProducerModeEnum() == REPORTER_MODE_ROOT && processor_id() != 0)
120 continue;
121
122 // Create/get object node
123 auto obj_node_pair = r_node.emplace(r_name.getObjectName(), nlohmann::json());
124 auto & obj_node = *(obj_node_pair.first);
125
126 // Whether or not we should store this Reporter's value or have it be null
127 bool should_store = true;
128
129 // If the object node was created insert the class level information
130 if (obj_node_pair.second)
131 {
132 // Query the TheWarehouse for all Reporter objects with the given name. The attributes and
133 // QueryID are used to allow the TheWarehouse::queryInto method be called with the
134 // "show_all" option set to true. This returns all objects, regardless of "enabled" state,
135 // which is what is needed to ensure that output always happens, even if an object is
136 // disabled.
137 std::vector<Reporter *> objs;
138 auto attr = _problem_ptr->theWarehouse()
139 .query()
140 .condition<AttribInterfaces>(Interfaces::Reporter)
141 .condition<AttribName>(r_name.getObjectName())
142 .attributes();
143 auto qid = _problem_ptr->theWarehouse().queryID(attr);
144 _problem_ptr->theWarehouse().queryInto(qid, objs, true);
145
146 // There can now be multiple reporter objects with the same name, but
147 // there will only be one reporter that stores all the data.
148 if (!objs.empty())
149 {
150 auto & reporter = *objs.front();
151
152 // It is possible to have a Reporter value without a reporter objects (i.e., VPPs, PPs),
153 // which is why objs can be empty.
154 reporter.store(obj_node);
155
156 // GeneralReporters have the option to only store JSON data when the execute flag
157 // matches an execute flag that is within the GeneralReporter; this captures that
158 should_store = reporter.shouldStore();
159 }
160 }
161
162 // Create/get value node
163 auto value_node_pair = obj_node["values"].emplace(r_name.getValueName(), nlohmann::json());
164 // If the value node was created insert the value information
165 if (value_node_pair.second)
166 // Store value meta data
167 context.storeInfo(*value_node_pair.first);
168
169 // Insert reporter value
170 auto & node = current_node[r_name.getObjectName()][r_name.getValueName()];
171 if (should_store)
172 context.store(node);
173 else
174 node = "null";
175 }
176 }
177}
