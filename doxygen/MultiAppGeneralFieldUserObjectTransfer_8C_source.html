26InputParameters
27MultiAppGeneralFieldUserObjectTransfer::validParams()
28{
29 InputParameters params = MultiAppGeneralFieldTransfer::validParams();
30 params.addClassDescription(
31 "Transfers user object spatial evaluations from an origin app onto a variable in the target "
32 "application.");
33
34 params.set<std::vector<VariableName>>("source_variable") = std::vector<VariableName>{};
35 params.suppressParameter<std::vector<VariableName>>("source_variable");
36 params.addRequiredParam<UserObjectName>("source_user_object",
37 "The UserObject you want to transfer values from. "
38 "It must implement the SpatialValue() class routine");
39
40 // Blanket ban on origin boundary restriction. User objects tend to extend beyond boundaries,
41 // and be able to be evaluated within a volume rather than only on a boundary
42 // This could be re-enabled for spatial user objects that are only defined on boundaries
43 params.suppressParameter<std::vector<BoundaryName>>("from_boundaries");
44 return params;
45}

47MultiAppGeneralFieldUserObjectTransfer::MultiAppGeneralFieldUserObjectTransfer(
48 const InputParameters & parameters)
49 : MultiAppGeneralFieldTransfer(parameters),
50 _user_object_name(getParam<UserObjectName>("source_user_object"))
51{
52 if (_to_var_names.size() > 1)
53 paramError("variable", "Only one variable at a time is supported by this transfer");
54
55 // Block restriction does not make sense if we're ok with extrapolating
56 if (isParamValid("from_blocks") && !_source_app_must_contain_point &&
57 !parameters.isParamSetByUser("extrapolation_constant"))
58 paramError("from_app_must_contain_point",
59 "Source block restriction cannot be used at the same type as allowing extrapolation"
60 " of values for a user object transfer (with 'from_app_must_contain_point=false') "
61 " unless an extrapolation constant is provided (with 'extrapolation_constant')");
62
63 // Nearest point isn't well defined for sending app-based data from main app to a multiapp
64 if (_nearest_positions_obj && isParamValid("to_multi_app") && !isParamValid("from_multi_app"))
65 paramError("use_nearest_position",
66 "Cannot use nearest-position algorithm when sending from the main application");
67}

86void
87MultiAppGeneralFieldUserObjectTransfer::evaluateInterpValuesWithUserObjects(
88 const std::vector<BoundingBox> & local_bboxes,
89 const std::vector<std::pair<Point, unsigned int>> & incoming_points,
90 std::vector<std::pair<Real, Real>> & outgoing_vals)
91{
92 dof_id_type i_pt = 0;
93 for (auto & [pt, mesh_div] : incoming_points)
94 {
95 bool point_found = false;
96 outgoing_vals[i_pt].second = GeneralFieldTransfer::BetterOutOfMeshValue;
97
98 // Loop on all local origin problems until:
99 // - we've found the point in an app and the value at that point is valid
100 // - or if looking for conflicts between apps, we must check them all
101 for (MooseIndex(_from_problems.size()) i_from = 0;
102 i_from < _from_problems.size() &&
103 (!point_found || _search_value_conflicts || _nearest_positions_obj);
104 ++i_from)
105 {
106 // User object spatialValue() evaluations do not provide a distance
107 Real distance = 1;
108 // Check spatial restrictions
109 if (!acceptPointInOriginMesh(i_from, local_bboxes, pt, mesh_div, distance))
110 continue;
111 else
112 {
113 const auto from_global_num = getGlobalSourceAppIndex(i_from);
114
115 // Get user object from the local problem
116 const UserObject & user_object =
117 _from_problems[i_from]->getUserObjectBase(_user_object_name);
118
119 // Use spatial value routine to compute the origin value to transfer
120 auto val = user_object.spatialValue(_from_transforms[from_global_num]->mapBack(pt));
121
122 // Look for overlaps. The check is not active outside of overlap search because in that
123 // case we accept the first value from the lowest ranked process
124 // NOTE: There is no guarantee this will be the final value used among all problems
125 // but we register an overlap as soon as two values are possible from this rank
126 if (detectConflict(val, outgoing_vals[i_pt].first, distance, outgoing_vals[i_pt].second))
127 registerConflict(i_from, 0, _from_transforms[from_global_num]->mapBack(pt), 1, true);
128
129 // No need to consider decision factors if value is invalid
130 if (val == GeneralFieldTransfer::BetterOutOfMeshValue)
131 continue;
132 else
133 point_found = true;
134
135 // Assign value
136 if (distance < outgoing_vals[i_pt].second)
137 {
138 outgoing_vals[i_pt].first = val;
139 outgoing_vals[i_pt].second = distance;
140 }
141 }
142 }
143
144 if (!point_found)
145 outgoing_vals[i_pt] = {GeneralFieldTransfer::BetterOutOfMeshValue,
146 GeneralFieldTransfer::BetterOutOfMeshValue};
147
148 // Move to next point
149 i_pt++;
150 }
151}
