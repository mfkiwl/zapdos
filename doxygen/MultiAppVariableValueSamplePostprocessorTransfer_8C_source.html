29InputParameters
30MultiAppVariableValueSamplePostprocessorTransfer::validParams()
31{
32 InputParameters params = MultiAppTransfer::validParams();
33 params.addClassDescription(
34 "Samples the value of a variable within the main application at each sub-application "
35 "position and transfers the value to a postprocessor on the sub-application(s) when "
36 "performing the to-multiapp transfer. Reconstructs the value of a CONSTANT MONOMIAL "
37 "variable associating the value of each element to the value of the postprocessor "
38 "in the closest sub-application whem performing the from-multiapp transfer.");
39 params.addRequiredParam<PostprocessorName>(
40 "postprocessor",
41 "The name of the postprocessor in the MultiApp to transfer the value to. "
42 "This should most likely be a Reciever Postprocessor.");
43 params.addRequiredParam<VariableName>("source_variable", "The variable to transfer from.");
44 params.addParam<unsigned int>(
45 "source_variable_component",
46 0,
47 "The component of source variable, may be non-zero for array variables.");
48 params.addParam<bool>(
49 "map_array_variable_components_to_child_apps",
50 false,
51 "When true, groups of sub-applications will be associated with different components of the "
52 "supplied array variable in 'source_variable'. For instance, if there are 9 sub-applications "
53 "and 3 components in the variable, sub-apps 0-2 will go to component 0, 3-5 will go to 1, "
54 "and 6-8 will go to 2.");
55 return params;
56}

58MultiAppVariableValueSamplePostprocessorTransfer::MultiAppVariableValueSamplePostprocessorTransfer(
59 const InputParameters & parameters)
60 : MultiAppTransfer(parameters),
61 MeshChangedInterface(parameters),
62 _postprocessor_name(getParam<PostprocessorName>("postprocessor")),
63 _var_name(getParam<VariableName>("source_variable")),
64 _comp(getParam<unsigned int>("source_variable_component")),
65 _var(_fe_problem.getVariable(0, _var_name)),
66 _map_comp_to_child(getParam<bool>("map_array_variable_components_to_child_apps"))
67{
68 if (_directions.size() != 1)
69 paramError("direction", "This transfer is only unidirectional");
70
71 if (_directions.contains("from_multiapp"))
72 {
73 // Check that the variable is a CONSTANT MONOMIAL.
74 auto & fe_type = _var.feType();
75 if (fe_type.order != CONSTANT || fe_type.family != MONOMIAL)
76 paramError("source_variable",
77 "Variable must be in CONSTANT MONOMIAL when transferring from a postprocessor "
78 "from sub-apps.");
79
80 if (!_fe_problem.getAuxiliarySystem().hasVariable(_var_name))
81 paramError("source_variable", "Variable must be an auxiliary variable");
82 }
83 else if (_directions.contains("between_multiapp"))
84 mooseError("MultiAppVariableValueSamplePostprocessorTransfer has not been made to support "
85 "sibling transfers");
86
87 if (_map_comp_to_child && !_var.isArray())
88 paramError("map_array_variable_components_to_child_apps",
89 "'source_variable' must be an array variable when mapping array variable components "
90 "to child applications.");
91 if (_map_comp_to_child && parameters.isParamSetByUser("source_variable_component"))
92 paramError("map_array_variable_components_to_child_apps",
93 "'source_variable_component' is invalid when mapping array variable components to "
94 "child applications.");
95}

97void
98MultiAppVariableValueSamplePostprocessorTransfer::setupPostprocessorCommunication()
99{
100 if (!_directions.contains("from_multiapp"))
101 return;
102
103 const auto num_global_apps = getFromMultiApp()->numGlobalApps();
104
105 // Setup the communication pattern
106 _postprocessor_to_processor_id.resize(num_global_apps,
107 std::numeric_limits<processor_id_type>::max());
108 for (const auto i : make_range(num_global_apps))
109 if (getFromMultiApp()->hasLocalApp(i))
110 _postprocessor_to_processor_id[i] = this->processor_id();
111
112 _communicator.min(_postprocessor_to_processor_id);
113#ifdef DEBUG
114 for (const auto i : make_range(num_global_apps))
115 {
116 mooseAssert(_postprocessor_to_processor_id[i] != std::numeric_limits<processor_id_type>::max(),
117 "Every element in the vector should have been set.");
118 if (getFromMultiApp()->hasLocalApp(i))
119 mooseAssert(_postprocessor_to_processor_id[i] == this->processor_id(),
120 "If I owned this app, then the processor id value should be my own");
121 }
122#endif
123}

125void
126MultiAppVariableValueSamplePostprocessorTransfer::cacheElemToPostprocessorData()
127{
128 if (!_directions.contains("from_multiapp"))
129 return;
130
131 // Cache the Multiapp position ID for every element.
132 auto & mesh = _fe_problem.mesh().getMesh();
133 unsigned int multiapp_pos_id = 0;
134 for (auto & elem : as_range(mesh.active_local_elements_begin(), mesh.active_local_elements_end()))
135 // Exclude the elements without dofs.
136 if (_var.hasBlocks(elem->subdomain_id()))
137 {
138 // The next two loops will loop through all the sub-applications
139 // The first loop is over each component of the source variable we are transferring to/from
140 unsigned int j = 0; // Indicates sub-app index
141 for (unsigned int g = 0; g < getFromMultiApp()->numGlobalApps() / _apps_per_component; ++g)
142 {
143 Real distance = std::numeric_limits<Real>::max();
144 unsigned int count = 0;
145 // The second loop is over all the sub-apps the given component is associated with
146 for (unsigned int c = 0; c < _apps_per_component; ++c, ++j)
147 {
148 Real current_distance = (getFromMultiApp()->position(j) - elem->true_centroid()).norm();
149 if (MooseUtils::absoluteFuzzyLessThan(current_distance, distance))
150 {
151 distance = current_distance;
152 multiapp_pos_id = j;
153 count = 0;
154 }
155 else if (MooseUtils::absoluteFuzzyEqual(current_distance, distance))
156 ++count;
157 }
158 if (count > 0)
159 mooseWarning(
160 "The distances of an element to more than one sub-applications are too close "
161 " in transfer '",
162 name(),
163 "'. The code chooses the sub-application with the smallest ID to set "
164 "the variable on the element, which may created undesired variable solutions."
165 "\nHaving different positions for sub-applications, "
166 "a centroid-based MultiApp or adding block restriction to the variable can "
167 "be used to resolve this warning.");
168
169 // Note: in case of count>0, the sub-application with smallest id will be used for the
170 // transfer.
171 _cached_multiapp_pos_ids.push_back(multiapp_pos_id);
172 _needed_postprocessors.insert(multiapp_pos_id);
173 }
174 }

177void
178MultiAppVariableValueSamplePostprocessorTransfer::initialSetup()
179{
180 MultiAppTransfer::initialSetup();
181
182 unsigned int num_apps = _directions.contains("from_multiapp") ? getFromMultiApp()->numGlobalApps()
183 : getToMultiApp()->numGlobalApps();
184 if (_map_comp_to_child && num_apps % _var.count() != 0)
185 paramError("map_array_variable_components_to_child_apps",
186 "The number of sub-applications (",
187 num_apps,
188 ") is not divisible by the number of components in '",
189 _var_name,
190 "' (",
191 _var.count(),
192 ").");
193 _apps_per_component = _map_comp_to_child ? num_apps / _var.count() : num_apps;
194
195 setupPostprocessorCommunication();
196 cacheElemToPostprocessorData();
197}

205void
206MultiAppVariableValueSamplePostprocessorTransfer::execute()
207{
208 TIME_SECTION("MultiAppVariableValueSamplePostprocessorTransfer::execute()",
209 5,
210 "Transferring a variable to a postprocessor through sampling");
211
212 switch (_current_direction)
213 {
214 case TO_MULTIAPP:
215 {
216 const ArrayMooseVariable * array_var = nullptr;
217 const MooseVariableField<Real> * standard_var = nullptr;
218 if (_var.isArray())
219 array_var = &_fe_problem.getArrayVariable(0, _var_name);
220 else if (!_var.isVector())
221 standard_var = static_cast<MooseVariableField<Real> *>(&_var);
222 else
223 mooseError("MultiAppVariableValueSamplePostprocessorTransfer does not support transfer of "
224 "vector variables");
225
226 MooseMesh & from_mesh = _fe_problem.mesh();
227
228 std::unique_ptr<PointLocatorBase> pl = from_mesh.getPointLocator();
229
230 pl->enable_out_of_mesh_mode();
231
232 for (unsigned int i = 0; i < getToMultiApp()->numGlobalApps(); i++)
233 {
234 Real value = -std::numeric_limits<Real>::max();
235
236 { // Get the value of the variable at the point where this multiapp is in the master domain
237
238 Point multi_app_position = getToMultiApp()->position(i);
239
240 std::vector<Point> point_vec(1, multi_app_position);
241
242 // First find the element the hit lands in
243 const Elem * elem = (*pl)(multi_app_position);
244
245 if (elem && elem->processor_id() == from_mesh.processor_id())
246 {
247 _fe_problem.setCurrentSubdomainID(elem, 0);
248 _fe_problem.reinitElemPhys(elem, point_vec, 0);
249
250 if (array_var)
251 {
252 value = array_var->sln()[0](getVariableComponent(i));
253 mooseAssert(
254 getVariableComponent(i) < array_var->count(),
255 "Component must be smaller than the number of components of array variable!");
256 mooseAssert(array_var->sln().size() == 1, "No values in u!");
257 }
258 else
259 {
260 value = standard_var->sln()[0];
261 mooseAssert(standard_var->sln().size() == 1, "No values in u!");
262 }
263 }
264
265 _communicator.max(value);
266 }
267
268 if (getToMultiApp()->hasLocalApp(i))
269 getToMultiApp()->appProblemBase(i).setPostprocessorValueByName(_postprocessor_name,
270 value);
271 }
272
273 break;
274 }
275 case FROM_MULTIAPP:
276 {
277 auto & mesh = _fe_problem.mesh().getMesh();
278 auto & solution = _var.sys().solution();
279
280 // Get the required postprocessor values
281 const unsigned int n_subapps = getFromMultiApp()->numGlobalApps();
282 std::vector<Real> pp_values(n_subapps, std::numeric_limits<Real>::max());
283 for (const auto i : make_range(n_subapps))
284 if (getFromMultiApp()->hasLocalApp(i))
285 pp_values[i] = getFromMultiApp()->appPostprocessorValue(i, _postprocessor_name);
286
287 // Gather all the multiapps postprocessor values that we need
288 std::unordered_map<processor_id_type, std::vector<unsigned int>> postprocessor_queries;
289 for (const auto needed_postprocessor : _needed_postprocessors)
290 {
291 const auto proc_id = _postprocessor_to_processor_id[needed_postprocessor];
292 if (proc_id != this->processor_id())
293 postprocessor_queries[proc_id].push_back(needed_postprocessor);
294 }
295
296 auto gather_data = [&pp_values
297#ifndef NDEBUG
298 ,
299 this
300#endif
301 ](processor_id_type libmesh_dbg_var(pid),
302 const std::vector<unsigned int> & postprocessor_ids,
303 std::vector<Real> & postprocessor_values)
304 {
305 mooseAssert(pid != this->processor_id(), "Should not be pulling from self");
306 postprocessor_values.resize(postprocessor_ids.size());
307 for (const auto i : index_range(postprocessor_ids))
308 {
309 const auto pp_id = postprocessor_ids[i];
310 const auto pp_value = pp_values[pp_id];
311 mooseAssert(
312 pp_value != std::numeric_limits<Real>::max(),
313 "If we are getting queried for postprocessor data, then we better have a valid"
314 "postprocesor value.");
315 postprocessor_values[i] = pp_value;
316 }
317 };
318
319 auto act_on_data = [&pp_values
320#ifndef NDEBUG
321 ,
322 this
323#endif
324 ](processor_id_type libmesh_dbg_var(pid),
325 const std::vector<unsigned int> & postprocessor_ids,
326 const std::vector<Real> & postprocessor_values)
327 {
328 mooseAssert(pid != this->processor_id(), "Should not be returning a query from self");
329 mooseAssert(postprocessor_ids.size() == postprocessor_values.size(),
330 "should be a 1-to-1 query-to-response");
331 for (const auto i : index_range(postprocessor_ids))
332 {
333 const auto pp_id = postprocessor_ids[i];
334 const auto pp_value = postprocessor_values[i];
335 mooseAssert(pp_value != std::numeric_limits<Real>::max(),
336 "If we are returning postprocessor data, then we better have a valid"
337 "postprocesor value.");
338 pp_values[pp_id] = pp_value;
339 }
340 };
341
342 constexpr Real example = 0;
343 TIMPI::pull_parallel_vector_data(
344 _communicator, postprocessor_queries, gather_data, act_on_data, &example);
345
346 // Assign the multiapps postprocessor values to the local elements.
347 unsigned int i = 0;
348 for (auto & elem :
349 as_range(mesh.active_local_elements_begin(), mesh.active_local_elements_end()))
350 {
351 // Exclude the elements without dofs
352 if (_var.hasBlocks(elem->subdomain_id()))
353 {
354 std::vector<dof_id_type> dof_indices;
355 _var.getDofIndices(elem, dof_indices);
356 mooseAssert(dof_indices.size() == 1,
357 "The variable must be a constant monomial with one DoF on an element");
358 mooseAssert(pp_values[_cached_multiapp_pos_ids[i]] != std::numeric_limits<Real>::max(),
359 "We should have pulled all the data we needed.");
360 for (unsigned int c = 0; c < n_subapps / _apps_per_component; ++c)
361 {
362 solution.set(dof_indices[0] + getVariableComponent(_cached_multiapp_pos_ids[i]),
363 pp_values[_cached_multiapp_pos_ids[i]]);
364 ++i;
365 }
366 }
367 }
368 solution.close();
369 break;
370 }
371 }
