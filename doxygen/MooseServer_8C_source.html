50bool
51MooseServer::parseDocumentForDiagnostics(wasp::DataArray & diagnosticsList)
52{
53 // strip prefix from document uri if it exists to get parse file path
54 std::string parse_file_path = document_path;
55 pcrecpp::RE("(.*://)(.*)").Replace("\\2", &parse_file_path);
56
57 // copy parent application parameters and modify to set up input check
58 InputParameters app_params = _moose_app.parameters();
59 app_params.set<std::vector<std::string>>("input_file") = {parse_file_path};
60 app_params.set<std::string>("_input_text") = document_text;
61 app_params.set<bool>("check_input") = true;
62 app_params.set<bool>("error_unused") = true;
63 app_params.set<bool>("error") = true;
64 app_params.set<bool>("error_deprecated") = true;
65 app_params.set<std::string>("color") = "off";
66 app_params.set<bool>("disable_perf_graph_live") = true;
67 app_params.remove("language_server");
68
69 // turn output off so input check application does not affect messages
70 std::streambuf * cached_output_buffer = Moose::out.rdbuf(nullptr);
71
72 // create new application with parameters modified for input check run
73 _check_app = AppFactory::instance().createShared(
74 _moose_app.type(), _moose_app.name(), app_params, _moose_app.getCommunicator()->get());
75
76 // disable logs and enable error exceptions with initial values cached
77 bool cached_logging_enabled = Moose::perf_log.logging_enabled();
78 Moose::perf_log.disable_logging();
79 bool cached_throw_on_error = Moose::_throw_on_error;
80 Moose::_throw_on_error = true;
81
82 bool pass = true;
83
84 // run input check application converting caught errors to diagnostics
85 try
86 {
87 _check_app->run();
88 }
89 catch (std::exception & err)
90 {
91 int line_number = 1;
92 int column_number = 1;
93
94 std::istringstream caught_msg(err.what());
95
96 // walk over caught message line by line adding each as a diagnostic
97 for (std::string error_line; std::getline(caught_msg, error_line);)
98 {
99 // skip over lines that are blank or consist totally of whitespace
100 if (error_line.find_first_not_of(" \t") == std::string::npos)
101 continue;
102
103 // check if this error line already has the input file path prefix
104 if (error_line.rfind(parse_file_path + ":", 0) == 0)
105 {
106 // strip input file path and colon prefix off of this error line
107 error_line.erase(0, parse_file_path.size() + 1);
108
109 int match_line_number;
110 int match_column_number;
111 std::string match_error_line;
112
113 // get line and column number from this error line if both exist
114 if (pcrecpp::RE("^(\\d+)\\.(\\d+)\\-?\\d*:\\s*(.*)$")
115 .FullMatch(error_line, &match_line_number, &match_column_number, &match_error_line))
116 {
117 line_number = match_line_number;
118 column_number = match_column_number;
119 error_line = match_error_line;
120 }
121
122 // otherwise get line number off of this error line if it exists
123 else if (pcrecpp::RE("^(\\d+):\\s*(.*)$")
124 .FullMatch(error_line, &match_line_number, &match_error_line))
125 {
126 line_number = match_line_number;
127 column_number = 1;
128 error_line = match_error_line;
129 }
130 }
131
132 // build zero based line and column diagnostic and add to the list
133 diagnosticsList.push_back(wasp::DataObject());
134 wasp::DataObject * diagnostic = diagnosticsList.back().to_object();
135 pass &= wasp::lsp::buildDiagnosticObject(*diagnostic,
136 errors,
137 line_number - 1,
138 column_number - 1,
139 line_number - 1,
140 column_number - 1,
141 1,
142 "moose_srv",
143 "check_inp",
144 error_line);
145 }
146 }
147
148 // reset behaviors of performance logging and error exception throwing
149 if (cached_logging_enabled)
150 Moose::perf_log.enable_logging();
151 Moose::_throw_on_error = cached_throw_on_error;
152
153 // turn output back on since it was turned off and input check is done
154 Moose::out.rdbuf(cached_output_buffer);
155
156 return pass;
157}

173bool
174MooseServer::gatherDocumentCompletionItems(wasp::DataArray & completionItems,
175 bool & is_incomplete,
176 int line,
177 int character)
178{
179 // add only root level blocks to completion list when parser root is null
180 if (!_check_app || !_check_app->parser()._root ||
181 _check_app->parser()._root->getNodeView().is_null())
182 return addSubblocksToList(completionItems, "/", line, character);
183
184 // find hit node for zero based request line and column number from input
185 wasp::HITNodeView view_root = _check_app->parser()._root->getNodeView();
186 wasp::HITNodeView request_context =
187 wasp::findNodeUnderLineColumn(view_root, line + 1, character + 1);
188
189 // get object context and value of type parameter for request if provided
190 wasp::HITNodeView object_context = request_context;
191 while (object_context.type() != wasp::OBJECT && object_context.has_parent())
192 object_context = object_context.parent();
193 const std::string & object_path = object_context.path();
194 wasp::HITNodeView type_node = object_context.first_child_by_name("type");
195 const std::string & object_type = type_node.is_null() ? "" : type_node.last_as_string();
196
197 // get set of all parameter and subblock names already specified in input
198 std::set<std::string> existing_params, existing_subblocks;
199 getExistingInput(object_context, existing_params, existing_subblocks);
200
201 // set used to gather all parameters valid from object context of request
202 InputParameters valid_params = emptyInputParameters();
203
204 // set used to gather MooseObjectAction tasks to verify object parameters
205 std::set<std::string> obj_act_tasks;
206
207 // get set of global parameters, action parameters, and object parameters
208 getAllValidParameters(valid_params, object_path, object_type, obj_act_tasks);
209
210 bool pass = true;
211
212 // use context of request to pick types of items added to completion list
213 if (request_context.type() == wasp::OBJECT || request_context.type() == wasp::DOCUMENT_ROOT)
214 {
215 // add gathered parameters to completion list at request input location
216 pass &= addParametersToList(completionItems, valid_params, existing_params, line, character);
217
218 // add all valid subblock objects to completion list for request object
219 pass &= addSubblocksToList(completionItems, object_path, line, character);
220 }
221 else if (request_context.type() == wasp::VALUE)
222 {
223 // get name of parameter in input and zero based line and column ranges
224 std::string param_name = request_context.has_parent() ? request_context.parent().name() : "";
225 int replace_line_start = request_context.line() - 1;
226 int replace_char_start = request_context.column() - 1;
227 int replace_line_end = request_context.last_line() - 1;
228 int replace_char_end = request_context.last_column();
229
230 // add options to completion list for request value if parameter exists
231 for (const auto & valid_params_iter : valid_params)
232 {
233 const std::string & valid_param_name = valid_params_iter.first;
234
235 if (param_name == valid_param_name)
236 {
237 pass &= addValuesToList(completionItems,
238 valid_params,
239 existing_subblocks,
240 param_name,
241 obj_act_tasks,
242 replace_line_start,
243 replace_char_start,
244 replace_line_end,
245 replace_char_end);
246 break;
247 }
248 }
249 }
250
251 is_incomplete = !pass;
252
253 return pass;
254}

256void
257MooseServer::getExistingInput(wasp::HITNodeView parent_node,
258 std::set<std::string> & existing_params,
259 std::set<std::string> & existing_subblocks)
260{
261 // gather names of all parameters and subblocks provided in input context
262 for (auto itr = parent_node.begin(); itr != parent_node.end(); itr.next())
263 {
264 auto child_node = itr.get();
265
266 // add key value or array type as parameter and object type as subblock
267 if (child_node.type() == wasp::KEYED_VALUE || child_node.type() == wasp::ARRAY)
268 existing_params.insert(child_node.name());
269 else if (child_node.type() == wasp::OBJECT)
270 existing_subblocks.insert(child_node.name());
271 }
272}

286void
287MooseServer::getActionParameters(InputParameters & valid_params,
288 const std::string & object_path,
289 std::set<std::string> & obj_act_tasks)
290{
291 Syntax & syntax = _check_app->syntax();
292 ActionFactory & action_factory = _check_app->getActionFactory();
293
294 // get registered syntax path identifier using actual object context path
295 bool is_parent;
296 std::string registered_syntax = syntax.isAssociated(object_path, &is_parent);
297
298 // use is_parent to skip action parameters when not explicitly registered
299 if (!is_parent)
300 {
301 // get action objects associated with registered syntax path identifier
302 auto action_range = syntax.getActions(registered_syntax);
303
304 // traverse action objects for syntax to gather valid action parameters
305 for (auto action_iter = action_range.first; action_iter != action_range.second; action_iter++)
306 {
307 const std::string & action_name = action_iter->second._action;
308
309 // use action name to get set of valid parameters from action factory
310 InputParameters action_params = action_factory.getValidParams(action_name);
311
312 // gather all MooseObjectAction tasks for verifying object parameters
313 if (action_params.have_parameter<bool>("isObjectAction"))
314 {
315 if (action_params.get<bool>("isObjectAction"))
316 {
317 std::set<std::string> tasks_by_actions = action_factory.getTasksByAction(action_name);
318 obj_act_tasks.insert(tasks_by_actions.begin(), tasks_by_actions.end());
319 }
320
321 // filter parameter from completion list as it is not used in input
322 action_params.remove("isObjectAction");
323 }
324
325 // add parameters from action to full valid collection being gathered
326 valid_params += action_params;
327 }
328 }

331void
332MooseServer::getObjectParameters(InputParameters & valid_params,
333 std::string object_type,
334 const std::set<std::string> & obj_act_tasks)
335{
336 Syntax & syntax = _check_app->syntax();
337 Factory & factory = _check_app->getFactory();
338
339 // use type parameter default if it exists and is not provided from input
340 if (object_type.empty() && valid_params.have_parameter<std::string>("type") &&
341 !valid_params.get<std::string>("type").empty())
342 {
343 object_type = valid_params.get<std::string>("type");
344
345 // make type parameter not required in input since it has default value
346 valid_params.makeParamNotRequired("type");
347 }
348
349 // check if object type has been registered to prevent unregistered error
350 if (factory.isRegistered(object_type))
351 {
352 // use object type to get set of valid parameters registered in factory
353 InputParameters object_params = factory.getValidParams(object_type);
354
355 // check if object has base associated with any MooseObjectAction tasks
356 if (object_params.have_parameter<std::string>("_moose_base"))
357 {
358 const std::string & moose_base = object_params.get<std::string>("_moose_base");
359
360 for (const auto & obj_act_task : obj_act_tasks)
361 {
362 if (syntax.verifyMooseObjectTask(moose_base, obj_act_task))
363 {
364 // add parameters from object to valid collection if base matches
365 valid_params += object_params;
366 break;
367 }
368 }
369 }
370 }
371
372 // make parameters from list of those set by action not required in input
373 if (valid_params.have_parameter<std::vector<std::string>>("_object_params_set_by_action"))
374 {
375 auto names = valid_params.get<std::vector<std::string>>("_object_params_set_by_action");
376 for (const auto & name : names)
377 valid_params.makeParamNotRequired(name);
378
379 // filter parameter from completion list since it is not used for input
380 valid_params.remove("_object_params_set_by_action");
381 }
382}

384bool
385MooseServer::addParametersToList(wasp::DataArray & completionItems,
386 const InputParameters & valid_params,
387 const std::set<std::string> & existing_params,
388 int request_line,
389 int request_char)
390{
391 bool pass = true;
392
393 // walk over collection of all valid parameters and build completion list
394 for (const auto & valid_params_iter : valid_params)
395 {
396 const std::string & param_name = valid_params_iter.first;
397 bool deprecated = valid_params.isParamDeprecated(param_name);
398 bool is_private = valid_params.isPrivate(param_name);
399
400 // filter out parameters that are deprecated, private, or already exist
401 if (deprecated || is_private || existing_params.count(param_name))
402 continue;
403
404 // process parameter description and type to use in input default value
405 std::string dirty_type = valid_params.type(param_name);
406 std::string clean_type = MooseUtils::prettyCppType(dirty_type);
407 std::string basic_type = JsonSyntaxTree::basicCppType(clean_type);
408 std::string doc_string = valid_params.getDocString(param_name);
409 MooseUtils::escape(doc_string);
410
411 // use basic type to decide if parameter is array and quotes are needed
412 std::string array_quote = basic_type.compare(0, 6, "Array:") == 0 ? "'" : "";
413
414 // remove any array prefixes from basic type string and leave base type
415 pcrecpp::RE("(Array:)*(.*)").GlobalReplace("\\2", &basic_type);
416
417 // prepare clean cpp type string to be used for key to find input paths
418 pcrecpp::RE(".+<([A-Za-z0-9_' ':]*)>.*").GlobalReplace("\\1", &clean_type);
419
420 // decide completion item kind that client may use to display list icon
421 int complete_kind = getCompletionItemKind(valid_params, param_name, clean_type, true);
422
423 // default value for completion to be built using parameter information
424 std::string default_value;
425
426 // first if parameter default is set then use it to build default value
427 if (valid_params.isParamValid(param_name))
428 {
429 default_value = JsonSyntaxTree::buildOutputString(valid_params_iter);
430 default_value = MooseUtils::trim(default_value);
431 }
432
433 // otherwise if parameter has coupled default then use as default value
434 else if (valid_params.hasDefaultCoupledValue(param_name))
435 {
436 std::ostringstream oss;
437 oss << valid_params.defaultCoupledValue(param_name);
438 default_value = oss.str();
439 }
440
441 // otherwise if parameter is enum then use first item for default value
442 if (default_value.empty())
443 {
444 std::map<std::string, std::string> options_and_descs;
445
446 if (valid_params.have_parameter<MooseEnum>(param_name))
447 getEnumsAndDocs(valid_params.get<MooseEnum>(param_name), options_and_descs);
448 else if (valid_params.have_parameter<MultiMooseEnum>(param_name))
449 getEnumsAndDocs(valid_params.get<MultiMooseEnum>(param_name), options_and_descs);
450 else if (valid_params.have_parameter<ExecFlagEnum>(param_name))
451 getEnumsAndDocs(valid_params.get<ExecFlagEnum>(param_name), options_and_descs);
452 else if (valid_params.have_parameter<std::vector<MooseEnum>>(param_name))
453 getEnumsAndDocs(valid_params.get<std::vector<MooseEnum>>(param_name)[0], options_and_descs);
454
455 if (!options_and_descs.empty())
456 default_value = options_and_descs.begin()->first;
457 }
458
459 // otherwise use parameter basic type to pick appropriate default value
460 if (default_value.empty())
461 {
462 if (basic_type == "Boolean")
463 default_value = "true";
464 else if (basic_type == "Integer")
465 default_value = "0";
466 else if (basic_type == "Real")
467 default_value = "0.0";
468 else
469 default_value = "value";
470 }
471
472 // switch 1 to true or 0 to false if boolean parameter as default value
473 if (basic_type == "Boolean" && default_value == "1")
474 default_value = "true";
475 else if (basic_type == "Boolean" && default_value == "0")
476 default_value = "false";
477
478 // finally build full insertion from parameter name, quote, and default
479 std::string embed_text = param_name + " = " + array_quote + default_value + array_quote;
480
481 // add parameter label, insert text, and description to completion list
482 completionItems.push_back(wasp::DataObject());
483 wasp::DataObject * item = completionItems.back().to_object();
484 pass &= wasp::lsp::buildCompletionObject(*item,
485 errors,
486 param_name,
487 request_line,
488 request_char,
489 request_line,
490 request_char,
491 embed_text,
492 complete_kind,
493 "",
494 doc_string,
495 false,
496 false);
497 }
498
499 return pass;
500}

502bool
503MooseServer::addSubblocksToList(wasp::DataArray & completionItems,
504 const std::string & object_path,
505 int request_line,
506 int request_char)
507{
508 Syntax & syntax = _check_app->syntax();
509
510 // set used to prevent reprocessing syntax paths for more than one action
511 std::set<std::string> syntax_paths_processed;
512
513 // build map of all syntax paths to names for subblocks and save to reuse
514 if (_syntax_to_subblocks.empty())
515 {
516 for (const auto & syntax_path_iter : syntax.getAssociatedActions())
517 {
518 std::string syntax_path = "/" + syntax_path_iter.first;
519
520 // skip current syntax path if already processed for different action
521 if (!syntax_paths_processed.insert(syntax_path).second)
522 continue;
523
524 // walk backward through syntax path adding subblock names to parents
525 for (std::size_t last_sep; (last_sep = syntax_path.find_last_of("/")) != std::string::npos;)
526 {
527 std::string subblock_name = syntax_path.substr(last_sep + 1);
528 syntax_path = syntax_path.substr(0, last_sep);
529 _syntax_to_subblocks[syntax_path].insert(subblock_name);
530 }
531 }
532 }
533
534 // get registered syntax from object path using map of paths to subblocks
535 std::string registered_syntax = syntax.isAssociated(object_path, nullptr, _syntax_to_subblocks);
536
537 bool pass = true;
538
539 // walk over subblock names if found or at root and build completion list
540 if (!registered_syntax.empty() || object_path == "/")
541 {
542 for (const auto & subblock_name : _syntax_to_subblocks[registered_syntax])
543 {
544 std::string doc_string;
545 std::string embed_text;
546 int complete_kind;
547
548 // customize description and insert text for star and named subblocks
549 if (subblock_name == "*")
550 {
551 doc_string = "custom user named block";
552 embed_text = "[block_name]\n \n[]";
553 complete_kind = wasp::lsp::m_comp_kind_variable;
554 }
555 else
556 {
557 doc_string = "application named block";
558 embed_text = "[" + subblock_name + "]\n \n[]";
559 complete_kind = wasp::lsp::m_comp_kind_struct;
560 }
561
562 // add subblock name, insert text, and description to completion list
563 completionItems.push_back(wasp::DataObject());
564 wasp::DataObject * item = completionItems.back().to_object();
565 pass &= wasp::lsp::buildCompletionObject(*item,
566 errors,
567 subblock_name,
568 request_line,
569 request_char,
570 request_line,
571 request_char,
572 embed_text,
573 complete_kind,
574 "",
575 doc_string,
576 false,
577 false);
578 }
579 }
580
581 return pass;
582}

584bool
585MooseServer::addValuesToList(wasp::DataArray & completionItems,
586 const InputParameters & valid_params,
587 const std::set<std::string> & existing_subblocks,
588 const std::string & param_name,
589 const std::set<std::string> & obj_act_tasks,
590 int replace_line_start,
591 int replace_char_start,
592 int replace_line_end,
593 int replace_char_end)
594{
595 Syntax & syntax = _check_app->syntax();
596 Factory & factory = _check_app->getFactory();
597
598 // get clean type for path associations and basic type for boolean values
599 std::string dirty_type = valid_params.type(param_name);
600 std::string clean_type = MooseUtils::prettyCppType(dirty_type);
601 std::string basic_type = JsonSyntaxTree::basicCppType(clean_type);
602
603 // remove any array prefixes from basic type string and replace with base
604 pcrecpp::RE("(Array:)*(.*)").GlobalReplace("\\2", &basic_type);
605
606 // prepare clean cpp type string to be used for a key to find input paths
607 pcrecpp::RE(".+<([A-Za-z0-9_' ':]*)>.*").GlobalReplace("\\1", &clean_type);
608
609 // decide completion item kind that client may use to display a list icon
610 int complete_kind = getCompletionItemKind(valid_params, param_name, clean_type, false);
611
612 // map used to gather options and descriptions for value completion items
613 std::map<std::string, std::string> options_and_descs;
614
615 // first if parameter name is active or inactive then use input subblocks
616 if (param_name == "active" || param_name == "inactive")
617 for (const auto & subblock_name : existing_subblocks)
618 options_and_descs[subblock_name] = "subblock name";
619
620 // otherwise if parameter type is boolean then use true and false strings
621 else if (basic_type == "Boolean")
622 {
623 options_and_descs["true"];
624 options_and_descs["false"];
625 }
626
627 // otherwise if parameter type is one of the enums then use valid options
628 else if (valid_params.have_parameter<MooseEnum>(param_name))
629 getEnumsAndDocs(valid_params.get<MooseEnum>(param_name), options_and_descs);
630 else if (valid_params.have_parameter<MultiMooseEnum>(param_name))
631 getEnumsAndDocs(valid_params.get<MultiMooseEnum>(param_name), options_and_descs);
632 else if (valid_params.have_parameter<ExecFlagEnum>(param_name))
633 getEnumsAndDocs(valid_params.get<ExecFlagEnum>(param_name), options_and_descs);
634 else if (valid_params.have_parameter<std::vector<MooseEnum>>(param_name))
635 getEnumsAndDocs(valid_params.get<std::vector<MooseEnum>>(param_name)[0], options_and_descs);
636
637 // otherwise if parameter name is type then use all verified object names
638 else if (param_name == "type")
639 {
640 // walk over entire set of objects that have been registered in factory
641 for (const auto & objects_iter : factory.registeredObjects())
642 {
643 const std::string & object_name = objects_iter.first;
644 const InputParameters & object_params = objects_iter.second->buildParameters();
645
646 // check if object has registered base parameter that can be verified
647 if (!object_params.have_parameter<std::string>("_moose_base"))
648 continue;
649 const std::string & moose_base = object_params.get<std::string>("_moose_base");
650
651 // walk over gathered MooseObjectAction tasks and add if base matches
652 for (const auto & obj_act_task : obj_act_tasks)
653 {
654 if (!syntax.verifyMooseObjectTask(moose_base, obj_act_task))
655 continue;
656 std::string type_description = object_params.getClassDescription();
657 MooseUtils::escape(type_description);
658 options_and_descs[object_name] = type_description;
659 break;
660 }
661 }
662 }
663
664 // otherwise if parameter type has any associated syntax then use lookups
665 else
666 {
667 // build map of parameter types to input lookup paths and save to reuse
668 if (_type_to_input_paths.empty())
669 {
670 for (const auto & associated_types_iter : syntax.getAssociatedTypes())
671 {
672 const std::string & type = associated_types_iter.second;
673 const std::string & path = associated_types_iter.first;
674 _type_to_input_paths[type].insert(path);
675 }
676 }
677
678 // check for input lookup paths that are associated with parameter type
679 const auto & input_path_iter = _type_to_input_paths.find(clean_type);
680
681 if (input_path_iter != _type_to_input_paths.end())
682 {
683 wasp::HITNodeView view_root = _check_app->parser()._root->getNodeView();
684
685 // walk over all syntax paths that are associated with parameter type
686 for (const auto & input_path : input_path_iter->second)
687 {
688 // use wasp siren to gather all input values at current lookup path
689 wasp::SIRENInterpreter<> selector;
690 if (!selector.parseString(input_path))
691 continue;
692 wasp::SIRENResultSet<wasp::HITNodeView> results;
693 std::size_t count = selector.evaluate(view_root, results);
694
695 // walk over results and add each input value found at current path
696 for (std::size_t i = 0; i < count; i++)
697 if (results.adapted(i).type() == wasp::OBJECT)
698 options_and_descs[results.adapted(i).name()] = "from /" + input_path;
699 }
700 }
701 }
702
703 bool pass = true;
704
705 // walk over pairs of options with descriptions and build completion list
706 for (const auto & option_and_desc : options_and_descs)
707 {
708 const std::string & option = option_and_desc.first;
709 const std::string & dscrpt = option_and_desc.second;
710
711 // add option name, insertion range, and description to completion list
712 completionItems.push_back(wasp::DataObject());
713 wasp::DataObject * item = completionItems.back().to_object();
714 pass &= wasp::lsp::buildCompletionObject(*item,
715 errors,
716 option,
717 replace_line_start,
718 replace_char_start,
719 replace_line_end,
720 replace_char_end,
721 option,
722 complete_kind,
723 "",
724 dscrpt,
725 false,
726 false);
727 }
728
729 return pass;

745bool
746MooseServer::gatherDocumentDefinitionLocations(wasp::DataArray & definitionLocations,
747 int line,
748 int character)
749{
750 Factory & factory = _check_app->getFactory();
751
752 // return without any definition locations added when parser root is null
753 if (!_check_app || !_check_app->parser()._root ||
754 _check_app->parser()._root->getNodeView().is_null())
755 return true;
756
757 // find hit node for zero based request line and column number from input
758 wasp::HITNodeView view_root = _check_app->parser()._root->getNodeView();
759 wasp::HITNodeView request_context =
760 wasp::findNodeUnderLineColumn(view_root, line + 1, character + 1);
761
762 // return without any definition locations added when node not value type
763 if (request_context.type() != wasp::VALUE)
764 return true;
765
766 // get name of parameter node parent of value and value string from input
767 std::string param_name = request_context.has_parent() ? request_context.parent().name() : "";
768 std::string val_string = request_context.last_as_string();
769
770 // add source code location if type parameter with registered object name
771 if (param_name == "type" && factory.isRegistered(val_string))
772 {
773 // get file path and line number of source code registering object type
774 FileLineInfo file_line_info = factory.getLineInfo(val_string);
775
776 // return without any definition locations added if file cannot be read
777 if (!file_line_info.isValid() ||
778 !MooseUtils::checkFileReadable(file_line_info.file(), false, false, false))
779 return true;
780
781 // add file scheme prefix to front of file path to build definition uri
782 auto location_uri = wasp::lsp::m_uri_prefix + file_line_info.file();
783
784 // add file uri and zero based line and column range to definition list
785 definitionLocations.push_back(wasp::DataObject());
786 wasp::DataObject * location = definitionLocations.back().to_object();
787 return wasp::lsp::buildLocationObject(*location,
788 errors,
789 location_uri,
790 file_line_info.line() - 1,
791 0,
792 file_line_info.line() - 1,
793 1000);
794 }
795
796 // get object context and value of type parameter for request if provided
797 wasp::HITNodeView object_context = request_context;
798 while (object_context.type() != wasp::OBJECT && object_context.has_parent())
799 object_context = object_context.parent();
800 const std::string & object_path = object_context.path();
801 wasp::HITNodeView type_node = object_context.first_child_by_name("type");
802 const std::string & object_type = type_node.is_null() ? "" : type_node.last_as_string();
803
804 // set used to gather all parameters valid from object context of request
805 InputParameters valid_params = emptyInputParameters();
806
807 // set used to gather MooseObjectAction tasks to verify object parameters
808 std::set<std::string> obj_act_tasks;
809
810 // get set of global parameters, action parameters, and object parameters
811 getAllValidParameters(valid_params, object_path, object_type, obj_act_tasks);
812
813 // set used to gather nodes from input lookups custom sorted by locations
814 SortedLocationNodes location_nodes(
815 [](const wasp::HITNodeView & l, const wasp::HITNodeView & r)
816 {
817 const std::string & l_file = l.node_pool()->stream_name();
818 const std::string & r_file = r.node_pool()->stream_name();
819 return (l_file < r_file || (l_file == r_file && l.line() < r.line()) ||
820 (l_file == r_file && l.line() == r.line() && l.column() < r.column()));
821 });
822
823 // gather all lookup path nodes matching value if parameter name is valid
824 for (const auto & valid_params_iter : valid_params)
825 {
826 if (valid_params_iter.first == param_name)
827 {
828 // get cpp type and prepare string for use as key finding input paths
829 std::string dirty_type = valid_params.type(param_name);
830 std::string clean_type = MooseUtils::prettyCppType(dirty_type);
831 pcrecpp::RE(".+<([A-Za-z0-9_' ':]*)>.*").GlobalReplace("\\1", &clean_type);
832
833 // get set of nodes from associated path lookups matching input value
834 getInputLookupDefinitionNodes(location_nodes, clean_type, val_string);
835 break;
836 }
837 }
838
839 // add parameter declarator to set if none were gathered by input lookups
840 if (location_nodes.empty() && request_context.has_parent() &&
841 request_context.parent().child_count_by_name("decl"))
842 location_nodes.insert(request_context.parent().first_child_by_name("decl"));
843
844 // add locations to definition list using lookups or parameter declarator
845 return addLocationNodesToList(definitionLocations, location_nodes);
846}
