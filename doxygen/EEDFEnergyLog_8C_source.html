51EEDFEnergyLog::computeQpResidual()
52{
53 return -_test[_i][_qp] * _reaction_coeff[_qp] * std::exp(_em[_qp] + _target[_qp]) *

57Real
58EEDFEnergyLog::computeQpJacobian()
59{
60 Real d_actual_mean_en_d_mean_en = std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
61 Real d_k_d_mean_en = _d_k_d_actual_mean_en[_qp] * d_actual_mean_en_d_mean_en;
62
63 return -_test[_i][_qp] * d_k_d_mean_en * std::exp(_em[_qp] + _target[_qp]) * _threshold_energy;
64}

66Real
67EEDFEnergyLog::computeQpOffDiagJacobian(unsigned int jvar)
68{
69 Real d_actual_mean_en_d_em = -std::exp(_u[_qp] - _em[_qp]) * _phi[_j][_qp];
70 Real d_k_d_em = _d_k_d_actual_mean_en[_qp] * d_actual_mean_en_d_em;
71
72 if (jvar == _em_id)
73 {
74 return -_test[_i][_qp] * d_k_d_em * _threshold_energy * std::exp(_em[_qp] + _target[_qp]);
75 }
76 else if (jvar == _target_id)
77 {
78 return -_test[_i][_qp] * d_k_d_em * _threshold_energy * std::exp(_em[_qp] + _target[_qp]) *
79 _phi[_j][_qp];
80 }
81 else
82 return 0.0;
83}
