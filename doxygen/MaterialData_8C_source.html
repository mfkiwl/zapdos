19void
20MaterialData::resize(unsigned int n_qpoints)
21{
22 if (n_qpoints == nQPoints())
23 return;
24
25 if (_resize_only_if_smaller && n_qpoints < nQPoints())
26 return;
27
28 for (const auto state : _storage.stateIndexRange())
29 props(state).resizeItems(n_qpoints, {});
30 _n_qpoints = n_qpoints;
31}
