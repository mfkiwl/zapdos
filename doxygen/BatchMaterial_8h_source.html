200template <typename Tuple, typename Output, typename... Input>
201template <int I, typename T, typename... Names>
202void
203BatchMaterial<Tuple, Output, Input...>::construct(T && name, Names &&... names)
204{
205 // couple the variable or material property
206 typedef typename std::tuple_element<I, std::tuple<Input...>>::type CurrentElement;
207 std::get<I>(_input_ptr) = CurrentElement::getPointer(*this, name);
208
209 // recursively couple the next variable or material property
210 if constexpr (sizeof...(Names) > 0)
211 construct<I + 1, Names...>(std::forward<Names>(names)...);
212}

214template <typename Tuple, typename Output, typename... Input>
215template <int I>
216void
217BatchMaterial<Tuple, Output, Input...>::copy(const unsigned int nqp)
218{
219 // copy all qp values for the current item
220 for (const auto qp : make_range(nqp))
221 Tuple::template get<I>(_input_data[_index + qp]) = (*std::get<I>(_input_ptr))[qp];
222
223 // proceed to next item
224 if constexpr (I + 1 < sizeof...(Input))
225 copy<I + 1>(nqp);
226}

246template <typename Tuple, typename Output, typename... Input>
247void
248BatchMaterial<Tuple, Output, Input...>::execute()
249{
250 // update index map
251 _index_map[_current_elem->id()] = _index;
252
253 // make sure the input data is sized sufficiently big
254 const auto nqp = _qrule->n_points();
255 if (_input_data.size() < _index + nqp)
256 _input_data.resize(_index + nqp);
257
258 // copy data
259 copy<0>(nqp);
260 _index += nqp;
261}

263template <typename Tuple, typename Output, typename... Input>
264void
265BatchMaterial<Tuple, Output, Input...>::threadJoin(const UserObject & uo)
266{
267 // join maps (with index shift)
268 const auto & bm = static_cast<const BatchMaterialType &>(uo);
269 for (const auto & [id, index] : bm._index_map)
270 _index_map[id] = index + _index;
271
272 // make sure we have enough space
273 const auto capacity = _input_data.capacity();
274 if (capacity < _index + bm._index)
275 _input_data.reserve(_index + bm._index);
276
277 // concatenate input data
278 _input_data.insert(std::begin(_input_data) + _index,
279 std::begin(bm._input_data),
280 std::begin(bm._input_data) + bm._index);
281 _index += bm._index;
282}
