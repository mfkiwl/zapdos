41void
42ElementStatistics::execute()
43{
44 // Get value to to update statistics
45 Real value = computeValue();
46
47 if (_max < value)
48 _max = value;
49
50 if (_min > value)
51 _min = value;
52
53 _integral += value * _current_elem_volume;
54
55 // Update the total and the number to get the average when "finalizing"
56 _average += value;
57 _number_elements++;
58}

61ElementStatistics::threadJoin(const UserObject & uo)

71void
72ElementStatistics::finalize()
73{
74 _communicator.max(_max);
75 _communicator.min(_min);
76 _communicator.sum(_integral);
77 _communicator.sum(_average);
78 _communicator.sum(_number_elements);

80 // Compute the average;
81 _average /= _number_elements;
