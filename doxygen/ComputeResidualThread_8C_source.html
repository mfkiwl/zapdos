76void
77ComputeResidualThread::determineObjectWarehouses()
78{
79 // If users pass a empty vector or a full size of vector,
80 // we take all kernels
81 if (!_tags.size() || _tags.size() == _fe_problem.numVectorTags(Moose::VECTOR_TAG_RESIDUAL))
82 {
83 _tag_kernels = &_kernels;
84 _dg_warehouse = &_dg_kernels;
85 _ibc_warehouse = &_integrated_bcs;
86 _ik_warehouse = &_interface_kernels;
87 }
88 // If we have one tag only,
89 // We call tag based storage
90 else if (_tags.size() == 1)
91 {
92 _tag_kernels = &(_kernels.getVectorTagObjectWarehouse(*(_tags.begin()), _tid));
93 _dg_warehouse = &(_dg_kernels.getVectorTagObjectWarehouse(*(_tags.begin()), _tid));
94 _ibc_warehouse = &(_integrated_bcs.getVectorTagObjectWarehouse(*(_tags.begin()), _tid));
95 _ik_warehouse = &(_interface_kernels.getVectorTagObjectWarehouse(*(_tags.begin()), _tid));
96 }
97 // This one may be expensive
98 else
99 {
100 _tag_kernels = &(_kernels.getVectorTagsObjectWarehouse(_tags, _tid));
101 _dg_warehouse = &(_dg_kernels.getVectorTagsObjectWarehouse(_tags, _tid));
102 _ibc_warehouse = &(_integrated_bcs.getVectorTagsObjectWarehouse(_tags, _tid));
103 _ik_warehouse = &(_interface_kernels.getVectorTagsObjectWarehouse(_tags, _tid));
104 }
105
106 if (_fe_problem.haveFV())
107 {
108 _fv_kernels.clear();
109 _fe_problem.theWarehouse()
110 .query()
111 .template condition<AttribSysNum>(_nl.number())
112 .template condition<AttribSystem>("FVElementalKernel")
113 .template condition<AttribSubdomains>(_subdomain)
114 .template condition<AttribThread>(_tid)
115 .template condition<AttribVectorTags>(_tags)
116 .queryInto(_fv_kernels);
117 }
118}
