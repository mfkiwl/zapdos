92NonlinearSystem::NonlinearSystem(FEProblemBase & fe_problem, const std::string & name)
93 : NonlinearSystemBase(
94 fe_problem, fe_problem.es().add_system<NonlinearImplicitSystem>(name), name),
95 _nl_implicit_sys(fe_problem.es().get_system<NonlinearImplicitSystem>(name)),
96 _nl_residual_functor(_fe_problem),
97 _fd_residual_functor(_fe_problem),
98 _resid_and_jac_functor(_fe_problem),
99 _use_coloring_finite_difference(false),
100 _solution_is_invalid(false)
101{
102 nonlinearSolver()->residual_object = &_nl_residual_functor;
103 nonlinearSolver()->jacobian = Moose::compute_jacobian;
104 nonlinearSolver()->bounds = Moose::compute_bounds;
105 nonlinearSolver()->nullspace = Moose::compute_nullspace;
106 nonlinearSolver()->transpose_nullspace = Moose::compute_transpose_nullspace;
107 nonlinearSolver()->nearnullspace = Moose::compute_nearnullspace;
108
109 PetscNonlinearSolver<Real> * petsc_solver =
110 static_cast<PetscNonlinearSolver<Real> *>(_nl_implicit_sys.nonlinear_solver.get());
111 if (petsc_solver)
112 {
113 petsc_solver->set_residual_zero_out(false);
114 petsc_solver->set_jacobian_zero_out(false);
115 petsc_solver->use_default_monitor(false);
116 }
117}

131void
132NonlinearSystem::solve()
133{
134 // Only attach the postcheck function to the solver if we actually
135 // have dampers or if the FEProblemBase needs to update the solution,
136 // which is also done during the linesearch postcheck. It doesn't
137 // hurt to do this multiple times, it is just setting a pointer.
138 if (_fe_problem.hasDampers() || _fe_problem.shouldUpdateSolution() ||
139 _fe_problem.needsPreviousNewtonIteration())
140 _nl_implicit_sys.nonlinear_solver->postcheck = Moose::compute_postcheck;
141
142 if (_fe_problem.solverParams()._type != Moose::ST_LINEAR)
143 {
144 TIME_SECTION("nlInitialResidual", 3, "Computing Initial Residual");
145 // Calculate the initial residual for use in the convergence criterion.
146 _computing_initial_residual = true;
147 _fe_problem.computeResidualSys(_nl_implicit_sys, *_current_solution, *_nl_implicit_sys.rhs);
148 _computing_initial_residual = false;
149 _nl_implicit_sys.rhs->close();
150 _initial_residual_before_preset_bcs = _nl_implicit_sys.rhs->l2_norm();
151 if (_compute_initial_residual_before_preset_bcs)
152 _console << "Initial residual before setting preset BCs: "
153 << _initial_residual_before_preset_bcs << std::endl;
154 }
155
156 // Clear the iteration counters
157 _current_l_its.clear();
158 _current_nl_its = 0;
159
160 // Initialize the solution vector using a predictor and known values from nodal bcs
161 setInitialSolution();
162
163 // Now that the initial solution has ben set, potentially perform a residual/Jacobian evaluation
164 // to determine variable scaling factors
165 if (_automatic_scaling)
166 {
167 if (_compute_scaling_once)
168 {
169 if (!_computed_scaling)
170 {
171 computeScaling();
172 _computed_scaling = true;
173 }
174 }
175 else
176 computeScaling();
177 }
178 // We do not know a priori what variable a global degree of freedom corresponds to, so we need a
179 // map from global dof to scaling factor. We just use a ghosted NumericVector for that mapping
180 assembleScalingVector();
181
182 if (_use_finite_differenced_preconditioner)
183 {
184 _nl_implicit_sys.nonlinear_solver->fd_residual_object = &_fd_residual_functor;
185 setupFiniteDifferencedPreconditioner();
186 }
187
188 PetscNonlinearSolver<Real> & solver =
189 static_cast<PetscNonlinearSolver<Real> &>(*_nl_implicit_sys.nonlinear_solver);
190 solver.mffd_residual_object = &_fd_residual_functor;
191
192 solver.set_snesmf_reuse_base(_fe_problem.useSNESMFReuseBase());
193
194 if (_time_integrator)
195 {
196 // reset solution invalid counter for the time step
197 _app.solutionInvalidity().resetSolutionInvalidTimeStep();
198 _time_integrator->solve();
199 _time_integrator->postSolve();
200 _n_iters = _time_integrator->getNumNonlinearIterations();
201 _n_linear_iters = _time_integrator->getNumLinearIterations();
202 // Accumulate only the occurence of solution invalid warnings for the current time step counters
203 _app.solutionInvalidity().solutionInvalidAccumulationTimeStep();
204 }
205 else
206 {
207 system().solve();
208 _n_iters = _nl_implicit_sys.n_nonlinear_iterations();
209 _n_linear_iters = solver.get_total_linear_iterations();
210 }
211
212 // store info about the solve
213 _final_residual = _nl_implicit_sys.final_nonlinear_residual();
214
215 // determine whether solution invalid occurs in the converged solution
216 _solution_is_invalid = _app.solutionInvalidity().solutionInvalid();
217
218 // output the solution invalid summary
219 if (_solution_is_invalid)
220 {
221 // sync all solution invalid counts to rank 0 process
222 _app.solutionInvalidity().sync();
223
224 if (_fe_problem.allowInvalidSolution())
225 mooseWarning("The Solution Invalidity warnings are detected but silenced! "
226 "Use Problem/allow_invalid_solution=false to activate ");
227 else
228 // output the occurrence of solution invalid in a summary table
229 _app.solutionInvalidity().print(_console);
230 }
231
232 if (_use_coloring_finite_difference)
233 MatFDColoringDestroy(&_fdcoloring);
234}

236void
237NonlinearSystem::stopSolve()
238{
239 PetscNonlinearSolver<Real> & solver =
240 static_cast<PetscNonlinearSolver<Real> &>(*sys().nonlinear_solver);
241 SNESSetFunctionDomainError(solver.snes());
242
243 // Insert a NaN into the residual vector. As of PETSc-3.6, this
244 // should make PETSc return DIVERGED_NANORINF the next time it does
245 // a reduction. We'll write to the first local dof on every
246 // processor that has any dofs.
247 _nl_implicit_sys.rhs->close();
248
249 if (_nl_implicit_sys.rhs->local_size())
250 _nl_implicit_sys.rhs->set(_nl_implicit_sys.rhs->first_local_index(),
251 std::numeric_limits<Real>::quiet_NaN());
252 _nl_implicit_sys.rhs->close();
253
254 // Clean up by getting other vectors into a valid state for a
255 // (possible) subsequent solve. There may be more than just
256 // these...
257 if (_Re_time)
258 _Re_time->close();
259 _Re_non_time->close();
260}

262void
263NonlinearSystem::setupFiniteDifferencedPreconditioner()
264{
265 std::shared_ptr<FiniteDifferencePreconditioner> fdp =
266 std::dynamic_pointer_cast<FiniteDifferencePreconditioner>(_preconditioner);
267 if (!fdp)
268 mooseError("Did not setup finite difference preconditioner, and please add a preconditioning "
269 "block with type = fdp");
270
271 if (fdp->finiteDifferenceType() == "coloring")
272 {
273 setupColoringFiniteDifferencedPreconditioner();
274 _use_coloring_finite_difference = true;
275 }
276
277 else if (fdp->finiteDifferenceType() == "standard")
278 {
279 setupStandardFiniteDifferencedPreconditioner();
280 _use_coloring_finite_difference = false;
281 }
282 else
283 mooseError("Unknown finite difference type");
284}

286void
287NonlinearSystem::setupStandardFiniteDifferencedPreconditioner()
288{
289 // Make sure that libMesh isn't going to override our preconditioner
290 _nl_implicit_sys.nonlinear_solver->jacobian = nullptr;
291
292 PetscNonlinearSolver<Number> * petsc_nonlinear_solver =
293 static_cast<PetscNonlinearSolver<Number> *>(_nl_implicit_sys.nonlinear_solver.get());
294
295 PetscMatrix<Number> * petsc_mat =
296 static_cast<PetscMatrix<Number> *>(&_nl_implicit_sys.get_system_matrix());
297
298 SNESSetJacobian(petsc_nonlinear_solver->snes(),
299 petsc_mat->mat(),
300 petsc_mat->mat(),
301 SNESComputeJacobianDefault,
302 nullptr);
303}

305void
306NonlinearSystem::setupColoringFiniteDifferencedPreconditioner()
307{
308 // Make sure that libMesh isn't going to override our preconditioner
309 _nl_implicit_sys.nonlinear_solver->jacobian = nullptr;
310
311 PetscNonlinearSolver<Number> & petsc_nonlinear_solver =
312 dynamic_cast<PetscNonlinearSolver<Number> &>(*_nl_implicit_sys.nonlinear_solver);
313
314 // Pointer to underlying PetscMatrix type
315 PetscMatrix<Number> * petsc_mat =
316 dynamic_cast<PetscMatrix<Number> *>(&_nl_implicit_sys.get_system_matrix());
317
318 Moose::compute_jacobian(*_nl_implicit_sys.current_local_solution, *petsc_mat, _nl_implicit_sys);
319
320 if (!petsc_mat)
321 mooseError("Could not convert to Petsc matrix.");
322
323 petsc_mat->close();
324
325 PetscErrorCode ierr = 0;
326 ISColoring iscoloring;
327
328 // PETSc 3.5.x
329 MatColoring matcoloring;
330 ierr = MatColoringCreate(petsc_mat->mat(), &matcoloring);
331 CHKERRABORT(_communicator.get(), ierr);
332 ierr = MatColoringSetType(matcoloring, MATCOLORINGLF);
333 CHKERRABORT(_communicator.get(), ierr);
334 ierr = MatColoringSetFromOptions(matcoloring);
335 CHKERRABORT(_communicator.get(), ierr);
336 ierr = MatColoringApply(matcoloring, &iscoloring);
337 CHKERRABORT(_communicator.get(), ierr);
338 ierr = MatColoringDestroy(&matcoloring);
339 CHKERRABORT(_communicator.get(), ierr);
340
341 MatFDColoringCreate(petsc_mat->mat(), iscoloring, &_fdcoloring);
342 MatFDColoringSetFromOptions(_fdcoloring);
343 // clang-format off
344 MatFDColoringSetFunction(_fdcoloring,
345 (PetscErrorCode(*)(void))(void (*)(void)) &
346 libMesh::libmesh_petsc_snes_fd_residual,
347 &petsc_nonlinear_solver);
348 // clang-format on
349 MatFDColoringSetUp(petsc_mat->mat(), iscoloring, _fdcoloring);
350 SNESSetJacobian(petsc_nonlinear_solver.snes(),
351 petsc_mat->mat(),
352 petsc_mat->mat(),
353 SNESComputeJacobianDefaultColor,
354 _fdcoloring);
355 // PETSc >=3.3.0
356 ISColoringDestroy(&iscoloring);
357}

359bool
360NonlinearSystem::converged()
361{
362 if (_fe_problem.hasException())
363 return false;
364 if (!_fe_problem.allowInvalidSolution() && _solution_is_invalid)
365 {
366 mooseWarning("The solution is not converged due to the solution being invalid.");
367 return false;
368 }
369 return _nl_implicit_sys.nonlinear_solver->converged;
370}
