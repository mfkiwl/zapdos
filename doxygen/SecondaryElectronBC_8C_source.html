52 // Resize the vectors to store _num_ions values:

60 // Retrieve the values for each ion and store in the relevant vectors.

76ADReal
77SecondaryElectronBC::computeQpResidual()
78{
79 if (_normals[_qp] * -1.0 * -_grad_potential[_qp] > 0.0)
80 {
81 _a = 1.0;
82 }
83 else
84 {
85 _a = 0.0;
86 }
87
88 _ion_flux = 0;
89 for (unsigned int i = 0; i < _num_ions; ++i)
90 {
91 if (_normals[_qp] * (*_sgnip[i])[_qp] * -_grad_potential[_qp] > 0.0)
92 _b = 1.0;
93 else
94 _b = 0.0;
95 _ion_flux += std::exp((*_ip[i])[_qp]) *
96 (0.5 * std::sqrt(8 * _kb[_qp] * (*_Tip[i])[_qp] / (M_PI * (*_massip[i])[_qp])) +
97 (2 * _b - 1) * (*_sgnip[i])[_qp] * (*_muip[i])[_qp] * -_grad_potential[_qp] *
98 _r_units * _normals[_qp]);
99 }
100 _ion_flux *= (1.0 - _r_ion) / (1.0 + _r_ion);
101
102 _n_gamma = (1. - _a) * _se_coeff[_qp] * _ion_flux /
103 (_muem[_qp] * -_grad_potential[_qp] * _r_units * _normals[_qp] +
104 std::numeric_limits<double>::epsilon());
105 _v_thermal =
106 std::sqrt(8 * _e[_qp] * 2.0 / 3 * std::exp(_mean_en[_qp] - _u[_qp]) / (M_PI * _massem[_qp]));
107
108 return _test[_i][_qp] * _r_units *
109 ((1. - _r) / (1. + _r) * (-0.5 * _v_thermal * _n_gamma) -
110 2. / (1. + _r) * (1. - _a) * _se_coeff[_qp] * _ion_flux);
111}
