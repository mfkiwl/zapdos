35Real
36BolsigValueScalar::computeValue()
37{
38 if (_data_type == "diffusivity")
39 {
40 mooseError("No diffusivity function yet! Whoops!");
41 }
42 else if (_data_type == "mobility" && !_sample_value)
43 {
44 return _data.electron_mobility();
45 }
46 else if (_data_type == "mobility" && _sample_value)
47 {
48 return _data.electron_mobility_sample(_sampler_var[0]/1e-21);
49 }
50 else if (_data_type == "electron_temperature" && !_sample_value)
51 {
52 return _data.electron_temperature();
53 }
54 else if (_data_type == "electron_temperature" && _sample_value)
55 {
56 return _data.electron_temperature_sample(_sampler_var[0]/1e-21);
57 }
58 else
59 {
60 mooseError("_type of "+_data_type+" not recognized! Choose \'diffusivity\' or \'mobility\'.");
61 }
62}
