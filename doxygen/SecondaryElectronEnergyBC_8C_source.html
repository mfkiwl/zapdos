78ADReal
79SecondaryElectronEnergyBC::computeQpResidual()
80{
81 if (_normals[_qp] * -1.0 * -_grad_potential[_qp] > 0.0)
82 {
83 _a = 1.0;
84 }
85 else
86 {
87 _a = 0.0;
88 }
89
90 _ion_flux = 0;
91 for (unsigned int i = 0; i < _num_ions; ++i)
92 {
93 if (_normals[_qp] * (*_sgnip[i])[_qp] * -_grad_potential[_qp] > 0.0)
94 _b = 1.0;
95 else
96 _b = 0.0;
97 _ion_flux += std::exp((*_ip[i])[_qp]) *
98 (0.5 * std::sqrt(8 * _kb[_qp] * (*_Tip[i])[_qp] / (M_PI * (*_massip[i])[_qp])) +
99 (2 * _b - 1) * (*_sgnip[i])[_qp] * (*_muip[i])[_qp] * -_grad_potential[_qp] *
100 _r_units * _normals[_qp]);
101 }
102
103 _n_gamma = (1. - _a) * _se_coeff[_qp] * _ion_flux /
104 (_muem[_qp] * -_grad_potential[_qp] * _r_units * _normals[_qp] +
105 std::numeric_limits<double>::epsilon());
106
107 _v_thermal =
108 std::sqrt(8 * _e[_qp] * 2.0 / 3 * std::exp(_u[_qp] - _em[_qp]) / (M_PI * _massem[_qp]));
109
110 return _test[_i][_qp] * _r_units *
111 ((1 - _r) / (1 + _r) * (-5. / 6. * _v_thermal * _n_gamma * _se_energy[_qp]) -
112 5. / 3. * (2. / (1 + _r)) * _se_energy[_qp] * (1 - _a) * _se_coeff[_qp] * _ion_flux);
113}
