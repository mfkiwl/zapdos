112struct RegistryEntryBase : public RegistryEntryData
113{
114 RegistryEntryBase(const RegistryEntryData & data) : RegistryEntryData(data) {}
115 virtual ~RegistryEntryBase() {}
117 virtual std::shared_ptr<MooseObject> build(const InputParameters & parameters) = 0;
118 virtual std::shared_ptr<Action> buildAction(const InputParameters & parameters) = 0;
119 virtual InputParameters buildParameters() = 0;
121 std::string name() const
122 {
123 std::string name = _name;
124 if (name.empty())
125 name = _alias;
126 if (name.empty())
127 name = _classname;

129 }
