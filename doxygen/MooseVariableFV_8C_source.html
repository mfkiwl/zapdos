32template <typename OutputType>
33InputParameters
34MooseVariableFV<OutputType>::validParams()
35{
36 InputParameters params = MooseVariableField<OutputType>::validParams();
37 params.set<bool>("fv") = true;
38 params.set<MooseEnum>("family") = "MONOMIAL";
39 params.set<MooseEnum>("order") = "CONSTANT";
40 params.template addParam<bool>(
41 "two_term_boundary_expansion",
42 true,
43 "Whether to use a two-term Taylor expansion to calculate boundary face values. "
44 "If the two-term expansion is used, then the boundary face value depends on the "
45 "adjoining cell center gradient, which itself depends on the boundary face value. "
46 "Consequently an implicit solve is used to simultaneously solve for the adjoining cell "
47 "center gradient and boundary face value(s).");
48 MooseEnum face_interp_method("average skewness-corrected", "average");
49 params.template addParam<MooseEnum>("face_interp_method",
50 face_interp_method,
51 "Switch that can select between face interpoaltion methods.");
52 params.template addParam<bool>(
53 "cache_cell_gradients", true, "Whether to cache cell gradients or re-compute them.");
54 // Just evaluating finite volume variables at an arbitrary location in a cell requires a layer of
55 // ghosting since we will use two term expansions
56 params.addRelationshipManager("ElementSideNeighborLayers",
57 Moose::RelationshipManagerType::GEOMETRIC |
58 Moose::RelationshipManagerType::ALGEBRAIC);
59 return params;
60}

337template <typename OutputType>
338void
339MooseVariableFV<OutputType>::computeFaceValues(const FaceInfo & fi)
340{
341 _element_data->setGeometry(Moose::Face);
342 _neighbor_data->setGeometry(Moose::Face);
343
344 const auto facetype = fi.faceType(std::make_pair(this->number(), this->sys().number()));
345 if (facetype == FaceInfo::VarFaceNeighbors::NEITHER)
346 return;
347 else if (facetype == FaceInfo::VarFaceNeighbors::BOTH)
348 {
349 _element_data->computeValuesFace(fi);
350 _neighbor_data->computeValuesFace(fi);
351 }
352 else if (facetype == FaceInfo::VarFaceNeighbors::ELEM)
353 _element_data->computeValuesFace(fi);
354 else if (facetype == FaceInfo::VarFaceNeighbors::NEIGHBOR)
355 _neighbor_data->computeValuesFace(fi);
356 else
357 mooseError("robert wrote broken MooseVariableFV code");
358}

412template <typename OutputType>
413std::pair<bool, const FVDirichletBCBase *>
414MooseVariableFV<OutputType>::getDirichletBC(const FaceInfo & fi) const
415{
416 if (!_dirichlet_map_setup)
417 const_cast<MooseVariableFV<OutputType> *>(this)->determineBoundaryToDirichletBCMap();
418
419 for (const auto bnd_id : fi.boundaryIDs())
420 if (auto it = _boundary_id_to_dirichlet_bc.find(bnd_id);
421 it != _boundary_id_to_dirichlet_bc.end())
422 return {true, it->second};
423
424 return {false, nullptr};

427template <typename OutputType>
428std::pair<bool, std::vector<const FVFluxBC *>>
429MooseVariableFV<OutputType>::getFluxBCs(const FaceInfo & fi) const
430{
431 std::vector<const FVFluxBC *> bcs;
432
433 this->_subproblem.getMooseApp()
434 .theWarehouse()
435 .query()
436 .template condition<AttribSystem>("FVFluxBC")
437 .template condition<AttribThread>(_tid)
438 .template condition<AttribBoundaries>(fi.boundaryIDs())
439 .template condition<AttribVar>(_var_num)
440 .template condition<AttribSysNum>(this->_sys.number())
441 .queryInto(bcs);
442
443 bool has_flux_bc = bcs.size() > 0;
444
445 if (has_flux_bc)
446 return std::make_pair(true, bcs);
447 else
448 return std::make_pair(false, std::vector<const FVFluxBC *>());
449}

451template <typename OutputType>
452ADReal
453MooseVariableFV<OutputType>::getElemValue(const Elem * const elem, const StateArg & state) const
454{
455 mooseAssert(elem,
456 "The elem shall exist! This typically occurs when the "
457 "user wants to evaluate non-existing elements (nullptr) at physical boundaries.");
458 mooseAssert(
459 this->hasBlocks(elem->subdomain_id()),
460 "The variable should be defined on the element's subdomain! This typically occurs when the "
461 "user wants to evaluate the elements right next to the boundary of two variables (block "
462 "boundary). The subdomain which is queried: " +
463 Moose::stringify(this->activeSubdomains()) + " the subdomain of the element " +
464 std::to_string(elem->subdomain_id()));
465
466 Moose::initDofIndices(const_cast<MooseVariableFV<OutputType> &>(*this), *elem);
467
468 mooseAssert(
469 this->_dof_indices.size() == 1,
470 "There should only be one dof-index for a constant monomial variable on any given element");
471
472 const dof_id_type index = this->_dof_indices[0];
473
474 // It's not safe to use solutionState(0) because it returns the libMesh System solution member
475 // which is wrong during things like finite difference Jacobian evaluation, e.g. when PETSc
476 // perturbs the solution vector we feed these perturbations into the current_local_solution
477 // while the libMesh solution is frozen in the non-perturbed state
478 const auto & global_soln = (state.state == 0)
479 ? *this->_sys.currentSolution()
480 : this->_sys.solutionState(state.state, state.iteration_type);
481
482 ADReal value = global_soln(index);
483
484 if (ADReal::do_derivatives && state.state == 0 &&
485 this->_sys.number() == this->_subproblem.currentNlSysNum())
486 Moose::derivInsert(value.derivatives(), index, 1.);
487
488 return value;
489}

503template <typename OutputType>
504ADReal
505MooseVariableFV<OutputType>::getDirichletBoundaryFaceValue(
506 const FaceInfo & fi,

538template <typename OutputType>
539ADReal
540MooseVariableFV<OutputType>::getExtrapolatedBoundaryFaceValue(const FaceInfo & fi,
541 const bool two_term_expansion,
542 const bool correct_skewness,
543 const Elem * elem_to_extrapolate_from,
544 const StateArg & state) const
545{
546 mooseAssert(
547 isExtrapolatedBoundaryFace(fi, elem_to_extrapolate_from, state) || !two_term_expansion,
548 "We allow Dirichlet boundary conditions to call this method. However, the only way to "
549 "ensure we don't have infinite recursion, with Green Gauss gradients calling back to the "
550 "Dirichlet boundary condition calling back to this method, is to do a one term expansion");
551
552 ADReal boundary_value;
553 bool elem_to_extrapolate_from_is_fi_elem;
554 std::tie(elem_to_extrapolate_from, elem_to_extrapolate_from_is_fi_elem) =
555 [this, &fi, elem_to_extrapolate_from]() -> std::pair<const Elem *, bool>
556 {
557 if (elem_to_extrapolate_from)
558 // Somebody already specified the element to extropolate from
559 return {elem_to_extrapolate_from, elem_to_extrapolate_from == fi.elemPtr()};
560 else
561 {
562 const auto [elem_guaranteed_to_have_dofs,
563 other_elem,
564 elem_guaranteed_to_have_dofs_is_fi_elem] =
565 Moose::FV::determineElemOneAndTwo(fi, *this);
566 // We only care about the element guaranteed to have degrees of freedom and current C++
567 // doesn't allow us to not assign one of the returned items like python does
568 libmesh_ignore(other_elem);
569 // We will extrapolate from the element guaranteed to have degrees of freedom
570 return {elem_guaranteed_to_have_dofs, elem_guaranteed_to_have_dofs_is_fi_elem};
571 }
572 }();
573
574 if (two_term_expansion)
575 {
576 const Point vector_to_face = elem_to_extrapolate_from_is_fi_elem
577 ? (fi.faceCentroid() - fi.elemCentroid())
578 : (fi.faceCentroid() - fi.neighborCentroid());
579 boundary_value = adGradSln(elem_to_extrapolate_from, state, correct_skewness) * vector_to_face +
580 getElemValue(elem_to_extrapolate_from, state);
581 }
582 else
583 boundary_value = getElemValue(elem_to_extrapolate_from, state);
584
585 return boundary_value;
586}

588template <typename OutputType>
589ADReal
590MooseVariableFV<OutputType>::getBoundaryFaceValue(const FaceInfo & fi,
591 const StateArg & state,
592 const bool correct_skewness) const
593{
594 mooseAssert(!this->isInternalFace(fi),
595 "A boundary face value has been requested on an internal face.");
596
597 if (isDirichletBoundaryFace(fi, nullptr, state))
598 return getDirichletBoundaryFaceValue(fi, nullptr, state);
599 else if (isExtrapolatedBoundaryFace(fi, nullptr, state))
600 return getExtrapolatedBoundaryFaceValue(
601 fi, _two_term_boundary_expansion, correct_skewness, nullptr, state);
602
603 mooseError("Unknown boundary face type!");
604}

606template <typename OutputType>
607const VectorValue<ADReal> &
608MooseVariableFV<OutputType>::adGradSln(const Elem * const elem,
609 const StateArg & state,
610 const bool correct_skewness) const
611{
612 // We ensure that no caching takes place when we compute skewness-corrected
613 // quantities.
614 if (_cache_cell_gradients && !correct_skewness && state.state == 0)
615 {
616 auto it = _elem_to_grad.find(elem);
617
618 if (it != _elem_to_grad.end())
619 return it->second;
620 }
621
622 auto grad = FV::greenGaussGradient(
623 ElemArg({elem, correct_skewness}), state, *this, _two_term_boundary_expansion, this->_mesh);
624
625 if (_cache_cell_gradients && !correct_skewness && state.state == 0)
626 {
627 auto pr = _elem_to_grad.emplace(elem, std::move(grad));
628 mooseAssert(pr.second, "Insertion should have just happened.");
629 return pr.first->second;
630 }
631 else
632 {
633 _temp_cell_gradient = std::move(grad);
634 return _temp_cell_gradient;
635 }
636}

638template <typename OutputType>
639VectorValue<ADReal>
640MooseVariableFV<OutputType>::uncorrectedAdGradSln(const FaceInfo & fi,
641 const StateArg & state,
642 const bool correct_skewness) const
643{
644 const bool var_defined_on_elem = this->hasBlocks(fi.elem().subdomain_id());
645 const Elem * const elem_one = var_defined_on_elem ? &fi.elem() : fi.neighborPtr();
646 const Elem * const elem_two = var_defined_on_elem ? fi.neighborPtr() : &fi.elem();
647
648 const VectorValue<ADReal> elem_one_grad = adGradSln(elem_one, state, correct_skewness);
649
650 // If we have a neighbor then we interpolate between the two to the face. If we do not, then we
651 // apply a zero Hessian assumption and use the element centroid gradient as the uncorrected face
652 // gradient
653 if (elem_two && this->hasBlocks(elem_two->subdomain_id()))
654 {
655 const VectorValue<ADReal> & elem_two_grad = adGradSln(elem_two, state, correct_skewness);
656
657 // Uncorrected gradient value
658 return Moose::FV::linearInterpolation(elem_one_grad, elem_two_grad, fi, var_defined_on_elem);
659 }
660 else
661 return elem_one_grad;
662}

664template <typename OutputType>
665VectorValue<ADReal>
666MooseVariableFV<OutputType>::adGradSln(const FaceInfo & fi,
667 const StateArg & state,
668 const bool correct_skewness) const
669{
670 const bool var_defined_on_elem = this->hasBlocks(fi.elem().subdomain_id());
671 const Elem * const elem = &fi.elem();
672 const Elem * const neighbor = fi.neighborPtr();
673
674 const bool is_internal_face = this->isInternalFace(fi);
675
676 const ADReal side_one_value = (!is_internal_face && !var_defined_on_elem)
677 ? getBoundaryFaceValue(fi, state, correct_skewness)
678 : getElemValue(elem, state);
679 const ADReal side_two_value = (var_defined_on_elem && !is_internal_face)
680 ? getBoundaryFaceValue(fi, state, correct_skewness)
681 : getElemValue(neighbor, state);
682
683 const auto delta =
684 this->isInternalFace(fi)
685 ? fi.dCNMag()
686 : (fi.faceCentroid() - (var_defined_on_elem ? fi.elemCentroid() : fi.neighborCentroid()))
687 .norm();
688
689 // This is the component of the gradient which is parallel to the line connecting
690 // the cell centers. Therefore, we can use our second order, central difference
691 // scheme to approximate it.
692 auto face_grad = ((side_two_value - side_one_value) / delta) * fi.eCN();
693
694 // We only need non-orthogonal correctors in 2+ dimensions
695 if (this->_mesh.dimension() > 1)
696 {
697 // We are using an orthogonal approach for the non-orthogonal correction, for more information
698 // see Hrvoje Jasak's PhD Thesis (Imperial College, 1996)
699 const auto & interpolated_gradient = uncorrectedAdGradSln(fi, state, correct_skewness);
700 face_grad += interpolated_gradient - (interpolated_gradient * fi.eCN()) * fi.eCN();
701 }
702
703 return face_grad;
704}

745template <typename OutputType>
746typename MooseVariableFV<OutputType>::ValueType
747MooseVariableFV<OutputType>::evaluate(const FaceArg & face, const StateArg & state) const
748{
749 const FaceInfo * const fi = face.fi;
750 mooseAssert(fi, "The face information must be non-null");
751 if (isDirichletBoundaryFace(*fi, face.face_side, state))
752 {
753 mooseAssert(state.state == 0,
754 "We have not yet added support for evaluating Dirichlet boundary conditions at "
755 "states other than the current solution state (e.g. current time)");
756 return getDirichletBoundaryFaceValue(*fi, face.face_side, state);
757 }
758 else if (isExtrapolatedBoundaryFace(*fi, face.face_side, state))
759 {
760 bool two_term_boundary_expansion = _two_term_boundary_expansion;
761 if (face.limiter_type == Moose::FV::LimiterType::Upwind)
762 if ((face.elem_is_upwind && face.face_side == fi->elemPtr()) ||
763 (!face.elem_is_upwind && face.face_side == fi->neighborPtr()))
764 two_term_boundary_expansion = false;
765 return getExtrapolatedBoundaryFaceValue(
766 *fi, two_term_boundary_expansion, face.correct_skewness, face.face_side, state);
767 }
768 else
769 {
770 mooseAssert(this->isInternalFace(*fi),
771 "We must be either Dirichlet, extrapolated, or internal");
772 return Moose::FV::interpolate(*this, face, state);
773 }
774}

776template <typename OutputType>
777typename MooseVariableFV<OutputType>::ValueType
778MooseVariableFV<OutputType>::evaluate(const NodeArg & node_arg, const StateArg & state) const
779{
780 const auto & node_to_elem_map = this->_mesh.nodeToElemMap();
781 const auto & elem_ids = libmesh_map_find(node_to_elem_map, node_arg.node->id());
782 ValueType sum = 0;
783 Real total_weight = 0;
784 mooseAssert(elem_ids.size(), "There should always be at least one element connected to a node");
785 for (const auto elem_id : elem_ids)
786 {
787 const Elem * const elem = this->_mesh.queryElemPtr(elem_id);
788 mooseAssert(elem, "We should have this element available");
789 if (!this->hasBlocks(elem->subdomain_id()))
790 continue;
791 const ElemPointArg elem_point{
792 elem, *node_arg.node, _face_interp_method == Moose::FV::InterpMethod::SkewCorrectedAverage};
793 const auto weight = 1 / (*node_arg.node - elem->vertex_average()).norm();
794 sum += weight * (*this)(elem_point, state);
795 total_weight += weight;
796 }
797 return sum / total_weight;
798}

848template <typename OutputType>
849void
850MooseVariableFV<OutputType>::determineBoundaryToDirichletBCMap()
851{
852 _boundary_id_to_dirichlet_bc.clear();
853 std::vector<FVDirichletBCBase *> bcs;
854
855 // I believe because query() returns by value but condition returns by reference that binding to a
856 // const lvalue reference results in the query() getting destructed and us holding onto a dangling
857 // reference. I think that condition returned by value we would be able to bind to a const lvalue
858 // reference here. But as it is we'll bind to a regular lvalue
859 const auto base_query = this->_subproblem.getMooseApp()
860 .theWarehouse()
861 .query()
862 .template condition<AttribSystem>("FVDirichletBC")
863 .template condition<AttribThread>(_tid)
864 .template condition<AttribVar>(_var_num)
865 .template condition<AttribSysNum>(this->_sys.number());
866
867 for (const auto bnd_id : this->_mesh.getBoundaryIDs())
868 {
869 auto base_query_copy = base_query;
870 base_query_copy.template condition<AttribBoundaries>(std::set<BoundaryID>({bnd_id}))
871 .queryInto(bcs);
872 mooseAssert(bcs.size() <= 1, "cannot have multiple dirichlet BCs on the same boundary");
873 if (!bcs.empty())
874 _boundary_id_to_dirichlet_bc.emplace(bnd_id, bcs[0]);
875 }
876
877 _dirichlet_map_setup = true;
878}
