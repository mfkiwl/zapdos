69template <typename T>
70void
71DependencyResolverInterface::sortDFS(typename std::vector<T> & vector)
72{
73 if (vector.size() <= 1)
74 return;
75
79 DependencyResolver<T> graph;
80
81 // Map of suppliers: what is supplied -> by what object
82 std::multimap<std::string, T> suppliers_map;
83 for (auto & v : vector)
84 {
85 for (const auto & supplied_item : v->getSuppliedItems())
86 {
87 suppliers_map.emplace(supplied_item, v);
88 graph.addNode(v);
89 }
90 }
91
92 // build the dependency graph
93 for (auto & v : vector)
94 {
95 for (const auto & requested_item : v->getRequestedItems())
96 {
97 const auto & [begin_it, end_it] = suppliers_map.equal_range(requested_item);
98 if (begin_it == end_it)
99 graph.addNode(v);
100 else
101 for (const auto & [supplier_name, supplier_object] : as_range(begin_it, end_it))
102 {
103 libmesh_ignore(supplier_name);
104 if (supplier_object == v)
105 // We allow an object to have a circular dependency within itself; e.g. we choose to
106 // trust a developer knows what they are doing within a single object
107 continue;
108 graph.addEdge(supplier_object, v);
109 }
110 }
111 }
112
113 vector = graph.dfs();
114}

116template <typename T, typename T2>
117void
118DependencyResolverInterface::cyclicDependencyError(CyclicDependencyException<T2> & e,
119 const std::string & header)
120{
121 std::ostringstream oss;
122
123 oss << header << ":\n";
124 const auto cycle = e.getCyclicDependencies();
125 std::vector<std::string> names(cycle.size());
126 for (const auto i : index_range(cycle))
127 names[i] = static_cast<T>(cycle[i])->name();
128 oss << MooseUtils::join(names, " <- ");
129 mooseError(oss.str());
130}
