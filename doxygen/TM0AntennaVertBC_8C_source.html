35ADReal
36TM0AntennaVertBC::computeQpResidual()
37{
38 if (_time_dependent)
39 return _test[_i][_qp] *
40 (-_u[_qp] / _q_point[_qp](0) + _omega * _eps_r * _eps0 * std::cos(_omega * _t));
41
42 else
43 return _test[_i][_qp] * (-_u[_qp] / _q_point[_qp](0) + _omega * _eps_r * _eps0);
44}
