54template <bool is_ad>
55Real
56ProcRateTempl<is_ad>::computeValue()
57{
58 _em_current = 6.02e23 * (_sgnem[_qp] * raw_value(_muem[_qp]) * -_grad_potential[_qp] * _r_units *
59 std::exp(_em[_qp]) -
60 raw_value(_diffem[_qp]) * std::exp(_em[_qp]) * _grad_em[_qp] * _r_units);
61
62 return raw_value(_alpha[_qp]) * _em_current.norm();
63}
