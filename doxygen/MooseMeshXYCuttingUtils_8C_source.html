24void
25lineRemoverMoveNode(ReplicatedMesh & mesh,
26 const std::vector<Real> & bdry_pars,
27 const subdomain_id_type block_id_to_remove,
28 const std::set<subdomain_id_type> & subdomain_ids_set,
29 const boundary_id_type trimming_section_boundary_id,
30 const boundary_id_type external_boundary_id,
31 const std::vector<boundary_id_type> & other_boundaries_to_conform,
32 const bool assign_ext_to_new,
33 const bool side_to_remove)
34{
35 // Build boundary information of the mesh
36 BoundaryInfo & boundary_info = mesh.get_boundary_info();
37 auto bdry_side_list = boundary_info.build_side_list();
38 // Only select the boundaries_to_conform
39 std::vector<std::tuple<dof_id_type, unsigned short int, boundary_id_type>> slc_bdry_side_list;
40 for (unsigned int i = 0; i < bdry_side_list.size(); i++)
41 if (std::get<2>(bdry_side_list[i]) == external_boundary_id ||
42 std::find(other_boundaries_to_conform.begin(),
43 other_boundaries_to_conform.end(),
44 std::get<2>(bdry_side_list[i])) != other_boundaries_to_conform.end())
45 slc_bdry_side_list.push_back(bdry_side_list[i]);
46
47 // Assign block id for elements to be removed
48 // Also record the elements crossed by the line and with its average vertices on the removal side
49 std::vector<dof_id_type> crossed_elems_to_remove;
50 for (auto elem_it = mesh.active_elements_begin(); elem_it != mesh.active_elements_end();
51 elem_it++)
52 {
53 // Check all the vertices of the element
54 unsigned short removal_side_count = 0;
55 for (unsigned int i = 0; i < (*elem_it)->n_vertices(); i++)
56 {
57 // First check if the vertex is on the XY-Plane
58 if (!MooseUtils::absoluteFuzzyEqual((*elem_it)->point(i)(2), 0.0))
59 mooseError(
60 "MooseMeshXYCuttingUtils::lineRemoverMoveNode() only works for 2D meshes in XY plane.");
61 if (lineSideDeterminator((*elem_it)->point(i)(0),
62 (*elem_it)->point(i)(1),
63 bdry_pars[0],
64 bdry_pars[1],
65 bdry_pars[2],
66 side_to_remove))
67 removal_side_count++;
68 }
69 if (removal_side_count == (*elem_it)->n_vertices())
70 {
71 (*elem_it)->subdomain_id() = block_id_to_remove;
72 continue;
73 }
74 // Check the average of the vertices of the element
75 if (lineSideDeterminator((*elem_it)->vertex_average()(0),
76 (*elem_it)->vertex_average()(1),
77 bdry_pars[0],
78 bdry_pars[1],
79 bdry_pars[2],
80 side_to_remove))
81 crossed_elems_to_remove.push_back((*elem_it)->id());
82 }
83 // Check each crossed element to see if removing it would lead to boundary moving
84 for (const auto & elem_id : crossed_elems_to_remove)
85 {
86 bool remove_flag = true;
87 for (unsigned int i = 0; i < mesh.elem_ptr(elem_id)->n_sides(); i++)
88 {
89 if (mesh.elem_ptr(elem_id)->neighbor_ptr(i) != nullptr)
90 if (mesh.elem_ptr(elem_id)->neighbor_ptr(i)->subdomain_id() != block_id_to_remove &&
91 std::find(crossed_elems_to_remove.begin(),
92 crossed_elems_to_remove.end(),
93 mesh.elem_ptr(elem_id)->neighbor_ptr(i)->id()) ==
94 crossed_elems_to_remove.end())
95 {
96 if (mesh.elem_ptr(elem_id)->subdomain_id() !=
97 mesh.elem_ptr(elem_id)->neighbor_ptr(i)->subdomain_id())
98 {
99 remove_flag = false;
100 break;
101 }
102 }
103 }
104 if (remove_flag)
105 mesh.elem_ptr(elem_id)->subdomain_id() = block_id_to_remove;
106 }
107
108 // Identify all the nodes that are on the interface between block_id_to_remove and other blocks
109 // !!! We need a check here: if a node is on the retaining side, the removed element has a
110 // different subdomain id
111 std::vector<dof_id_type> node_list;
112 for (auto elem_it = mesh.active_subdomain_set_elements_begin(subdomain_ids_set);
113 elem_it != mesh.active_subdomain_set_elements_end(subdomain_ids_set);
114 elem_it++)
115 {
116 for (unsigned int i = 0; i < (*elem_it)->n_sides(); i++)
117 {
118 if ((*elem_it)->neighbor_ptr(i) != nullptr)
119 if ((*elem_it)->neighbor_ptr(i)->subdomain_id() == block_id_to_remove)
120 {
121 node_list.push_back((*elem_it)->side_ptr(i)->node_ptr(0)->id());
122 node_list.push_back((*elem_it)->side_ptr(i)->node_ptr(1)->id());
123 boundary_info.add_side(*elem_it, i, trimming_section_boundary_id);
124 if (assign_ext_to_new && trimming_section_boundary_id != external_boundary_id)
125 boundary_info.add_side(*elem_it, i, external_boundary_id);
126 }
127 }
128 }
129 // Remove duplicate nodes
130 const auto unique_it = std::unique(node_list.begin(), node_list.end());
131 node_list.resize(std::distance(node_list.begin(), unique_it));
132 // Mark those nodes that are on a boundary that requires conformality
133 // If both nodes of a side are involved, we should only move one node
134 std::vector<bool> node_list_flag(node_list.size(), false);
135 std::vector<Point> node_list_point(node_list.size(), Point(0.0, 0.0, 0.0));
136 // Loop over all the selected sides
137 for (unsigned int i = 0; i < slc_bdry_side_list.size(); i++)
138 {
139 // Get the two node ids of the side
140 dof_id_type side_id_0 = mesh.elem_ptr(std::get<0>(slc_bdry_side_list[i]))
141 ->side_ptr(std::get<1>(slc_bdry_side_list[i]))
142 ->node_ptr(0)
143 ->id();
144 dof_id_type side_id_1 = mesh.elem_ptr(std::get<0>(slc_bdry_side_list[i]))
145 ->side_ptr(std::get<1>(slc_bdry_side_list[i]))
146 ->node_ptr(1)
147 ->id();
148 // True means the selected bdry node is in the node list of the trimming interface
149 bool side_id_0_in =
150 !(std::find(node_list.begin(), node_list.end(), side_id_0) == node_list.end());
151 bool side_id_1_in =
152 !(std::find(node_list.begin(), node_list.end(), side_id_1) == node_list.end());
153
154 // True means the selected bdry node is on the removal side of the trimming interface
155 bool side_node_0_remove = lineSideDeterminator((*mesh.node_ptr(side_id_0))(0),
156 (*mesh.node_ptr(side_id_0))(1),
157 bdry_pars[0],
158 bdry_pars[1],
159 bdry_pars[2],
160 side_to_remove);
161 bool side_node_1_remove = lineSideDeterminator((*mesh.node_ptr(side_id_1))(0),
162 (*mesh.node_ptr(side_id_1))(1),
163 bdry_pars[0],
164 bdry_pars[1],
165 bdry_pars[2],
166 side_to_remove);
167 // If both nodes of that side are involved in the trimming interface
168 if (side_id_0_in && side_id_1_in)
169 // The side needs to be removed from the sideset because it is not longer an interface
170 // The other node will be handled by other element's side
171 boundary_info.remove_side(mesh.elem_ptr(std::get<0>(slc_bdry_side_list[i])),
172 std::get<1>(slc_bdry_side_list[i]),
173 std::get<2>(slc_bdry_side_list[i]));
174 // If node 0 is on the trimming interface, and the side is cut by the trimming line
175 else if (side_id_0_in && (side_node_0_remove != side_node_1_remove))
176 {
177 // Use the intersection point as the destination of the node after moving
178 node_list_flag[std::distance(
179 node_list.begin(), std::find(node_list.begin(), node_list.end(), side_id_0))] = true;
180 const Point p0 = *mesh.node_ptr(side_id_0);
181 const Point p1 = *mesh.node_ptr(side_id_1);
182
183 node_list_point[std::distance(node_list.begin(),
184 std::find(node_list.begin(), node_list.end(), side_id_0))] =
185 twoPointandLineIntersection(p0, p1, bdry_pars[0], bdry_pars[1], bdry_pars[2]);
186 }
187 // If node 1 is on the trimming interface, and the side is cut by the trimming line
188 else if (side_id_1_in && (side_node_0_remove != side_node_1_remove))
189 {
190 // Use the intersection point as the destination of the node after moving
191 node_list_flag[std::distance(
192 node_list.begin(), std::find(node_list.begin(), node_list.end(), side_id_1))] = true;
193 const Point p0 = *mesh.node_ptr(side_id_0);
194 const Point p1 = *mesh.node_ptr(side_id_1);
195
196 node_list_point[std::distance(node_list.begin(),
197 std::find(node_list.begin(), node_list.end(), side_id_1))] =
198 twoPointandLineIntersection(p0, p1, bdry_pars[0], bdry_pars[1], bdry_pars[2]);
199 }
200 }
201
202 // move nodes
203 for (unsigned int i = 0; i < node_list.size(); i++)
204 {
205 // This means the node is on both the trimming boundary and the original external
206 // boundary/selected interface boundaries. In order to keep the shape of the original external
207 // boundary, the node is moved along the original external boundary.
208 if (node_list_flag[i])
209 *(mesh.node_ptr(node_list[i])) = node_list_point[i];
210 // This means the node does not need to conform to any boundaries.
211 // Just move it along the normal direction of the trimming line.
212 else
213 {
214 const Real x0 = (*(mesh.node_ptr(node_list[i])))(0);
215 const Real y0 = (*(mesh.node_ptr(node_list[i])))(1);
216 (*(mesh.node_ptr(node_list[i])))(0) =
217 (bdry_pars[1] * (bdry_pars[1] * x0 - bdry_pars[0] * y0) - bdry_pars[0] * bdry_pars[2]) /
218 (bdry_pars[0] * bdry_pars[0] + bdry_pars[1] * bdry_pars[1]);
219 (*(mesh.node_ptr(node_list[i])))(1) =
220 (bdry_pars[0] * (-bdry_pars[1] * x0 + bdry_pars[0] * y0) - bdry_pars[1] * bdry_pars[2]) /
221 (bdry_pars[0] * bdry_pars[0] + bdry_pars[1] * bdry_pars[1]);
222 }
223 }
224
225 // Delete the block
226 for (auto elem_it = mesh.active_subdomain_elements_begin(block_id_to_remove);
227 elem_it != mesh.active_subdomain_elements_end(block_id_to_remove);
228 elem_it++)
229 mesh.delete_elem(*elem_it);
230 mesh.contract();
231 mesh.find_neighbors();
232 // Delete zero volume elements
233 std::vector<dof_id_type> zero_elems;
234 for (auto elem_it = mesh.elements_begin(); elem_it != mesh.elements_end(); elem_it++)
235 {
236 if (MooseUtils::absoluteFuzzyEqual((*elem_it)->volume(), 0.0))
237 {
238 for (unsigned int i = 0; i < (*elem_it)->n_sides(); i++)
239 {
240 if ((*elem_it)->neighbor_ptr(i) != nullptr)
241 {
242 boundary_info.add_side((*elem_it)->neighbor_ptr(i),
243 ((*elem_it)->neighbor_ptr(i))->which_neighbor_am_i(*elem_it),
244 external_boundary_id);
245 boundary_info.add_side((*elem_it)->neighbor_ptr(i),
246 ((*elem_it)->neighbor_ptr(i))->which_neighbor_am_i(*elem_it),
247 trimming_section_boundary_id);
248 }
249 }
250 zero_elems.push_back((*elem_it)->id());
251 }
252 }
253 for (const auto & zero_elem : zero_elems)
254 mesh.delete_elem(mesh.elem_ptr(zero_elem));
255 mesh.contract();
256 // As we modified the side_list, it is safer to clear the node_list
257 boundary_info.clear_boundary_node_ids();
258 mesh.prepare_for_use();
259}

261bool
262lineSideDeterminator(const Real px,
263 const Real py,
264 const Real param_1,
265 const Real param_2,
266 const Real param_3,
267 const bool direction_param,
268 const Real dis_tol)
269{
270 const Real tmp = px * param_1 + py * param_2 + param_3;
271 return direction_param ? tmp >= dis_tol : tmp <= dis_tol;
272}

274Point
275twoLineIntersection(const Real param_11,
276 const Real param_12,
277 const Real param_13,
278 const Real param_21,
279 const Real param_22,
280 const Real param_23)
281{
282 return Point(
283 (param_12 * param_23 - param_22 * param_13) / (param_11 * param_22 - param_21 * param_12),
284 (param_13 * param_21 - param_23 * param_11) / (param_11 * param_22 - param_21 * param_12),
285 0.0);
286}

295 return twoLineIntersection(param_1,

303bool
304quasiTriElementsFixer(ReplicatedMesh & mesh,
305 const std::set<subdomain_id_type> & subdomain_ids_set,
306 const subdomain_id_type tri_elem_subdomain_shift,
307 const SubdomainName tri_elem_subdomain_name_suffix)
308{
309 BoundaryInfo & boundary_info = mesh.get_boundary_info();
310 // Define the subdomain id shift for the new TRI3 element subdomain(s)
311 const subdomain_id_type max_subdomain_id(*subdomain_ids_set.rbegin());
312 const subdomain_id_type tri_subdomain_id_shift =
313 tri_elem_subdomain_shift == Moose::INVALID_BLOCK_ID ? max_subdomain_id
314 : tri_elem_subdomain_shift;
315 mooseAssert(std::numeric_limits<subdomain_id_type>::max() - max_subdomain_id >
316 tri_subdomain_id_shift,
317 "The TRI elements subdomain id to be assigned may exceed the numeric limit.");
318 const unsigned int n_elem_extra_ids = mesh.n_elem_integers();
319 std::vector<dof_id_type> exist_extra_ids(n_elem_extra_ids);
320 std::vector<std::tuple<Elem *, unsigned int, bool, bool>> bad_elems_rec;
321 // Loop over all the active elements to find any degenerate QUAD elements
322 for (auto & elem : as_range(mesh.active_elements_begin(), mesh.active_elements_end()))
323 {
324 // Two types of degenerate QUAD elements are identified:
325 // (1) QUAD elements with three collinear vertices
326 // (2) QUAD elements with two overlapped vertices
327 const auto elem_angles = vertex_angles(*elem);
328 const auto elem_distances = vertex_distances(*elem);
329 // Type 1
330 if (MooseUtils::absoluteFuzzyEqual(elem_angles.front().first, M_PI, 0.001))
331 {
332 bad_elems_rec.push_back(std::make_tuple(elem, elem_angles.front().second, false, true));
333 continue;
334 }
335 // Type 2
336 if (MooseUtils::absoluteFuzzyEqual(elem_distances.front().first, 0.0))
337 {
338 bad_elems_rec.push_back(std::make_tuple(elem, elem_distances.front().second, false, false));
339 }
340 }
341 std::set<subdomain_id_type> new_subdomain_ids;
342 // Loop over all the identified degenerate QUAD elements
343 for (const auto & bad_elem : bad_elems_rec)
344 {
345 std::vector<boundary_id_type> elem_bdry_container_0;
346 std::vector<boundary_id_type> elem_bdry_container_1;
347 std::vector<boundary_id_type> elem_bdry_container_2;
348
349 Elem * elem_0 = std::get<0>(bad_elem);
350 if (std::get<3>(bad_elem))
351 {
352 // elems 1 and 2 are the neighboring elements of the degenerate element corresponding to the
353 // two collinear sides.
354 // For the degenerated element with three colinear vertices, if the elems 2 and 3 do not
355 // exist, the two sides are on the external boundary formed by trimming.
356 Elem * elem_1 = elem_0->neighbor_ptr(std::get<1>(bad_elem));
357 Elem * elem_2 = elem_0->neighbor_ptr((std::get<1>(bad_elem) - 1) % elem_0->n_vertices());
358 if ((elem_1 != nullptr || elem_2 != nullptr))
359 throw MooseException("The input mesh has degenerate quad element before trimming.");
360 }
361 mesh.get_boundary_info().boundary_ids(
362 elem_0, (std::get<1>(bad_elem) + 1) % elem_0->n_vertices(), elem_bdry_container_0);
363 mesh.get_boundary_info().boundary_ids(
364 elem_0, (std::get<1>(bad_elem) + 2) % elem_0->n_vertices(), elem_bdry_container_1);
365 mesh.get_boundary_info().boundary_ids(
366 elem_0, (std::get<1>(bad_elem) + 3) % elem_0->n_vertices(), elem_bdry_container_2);
367
368 // Record subdomain id of the degenerate element
369 auto elem_block_id = elem_0->subdomain_id();
370 // Define the three of four nodes that will be used to generate the TRI element
371 auto pt0 = elem_0->node_ptr((std::get<1>(bad_elem) + 1) % elem_0->n_vertices());
372 auto pt1 = elem_0->node_ptr((std::get<1>(bad_elem) + 2) % elem_0->n_vertices());
373 auto pt2 = elem_0->node_ptr((std::get<1>(bad_elem) + 3) % elem_0->n_vertices());
374 // Record all the element extra integers of the degenerate element
375 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
376 exist_extra_ids[j] = elem_0->get_extra_integer(j);
377 // Delete the degenerate QUAD element
378 mesh.delete_elem(elem_0);
379 // Create the new TRI element
380 Elem * elem_Tri3 = mesh.add_elem(new Tri3);
381 elem_Tri3->set_node(0) = pt0;
382 elem_Tri3->set_node(1) = pt1;
383 elem_Tri3->set_node(2) = pt2;
384 // Retain the boundary information
385 for (auto bdry_id : elem_bdry_container_0)
386 boundary_info.add_side(elem_Tri3, 0, bdry_id);
387 for (auto bdry_id : elem_bdry_container_1)
388 boundary_info.add_side(elem_Tri3, 1, bdry_id);
389 for (auto bdry_id : elem_bdry_container_2)
390 boundary_info.add_side(elem_Tri3, 2, bdry_id);
391 // Assign subdomain id for the TRI element by shifting its original subdomain id
392 elem_Tri3->subdomain_id() = elem_block_id + tri_subdomain_id_shift;
393 new_subdomain_ids.emplace(elem_block_id + tri_subdomain_id_shift);
394 // Retain element extra integers
395 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
396 elem_Tri3->set_extra_integer(j, exist_extra_ids[j]);
397 }
398 // Assign subdomain names for the new TRI elements
399 for (auto & nid : new_subdomain_ids)
400 {
401 const SubdomainName old_name = mesh.subdomain_name(nid - tri_subdomain_id_shift);
402 if (MooseMeshUtils::getSubdomainID(
403 (old_name.empty() ? (SubdomainName)(std::to_string(nid - tri_subdomain_id_shift))
404 : old_name) +
405 "_" + tri_elem_subdomain_name_suffix,
406 mesh) != Moose::INVALID_BLOCK_ID)
407 throw MooseException("The new subdomain name already exists in the mesh.");
408 mesh.subdomain_name(nid) =
409 (old_name.empty() ? (SubdomainName)(std::to_string(nid - tri_subdomain_id_shift))
410 : old_name) +
411 "_" + tri_elem_subdomain_name_suffix;
412 mooseWarning("Degenerate QUAD elements have been converted into TRI elements with a new "
413 "subdomain name: " +
414 mesh.subdomain_name(nid) + ".");
415 }
416 return bad_elems_rec.size();
417}

419std::vector<std::pair<Real, unsigned int>>
420vertex_angles(const Elem & elem)
421{
422 std::vector<std::pair<Real, unsigned int>> angles;
423 const unsigned int n_vertices = elem.n_vertices();
424
425 for (unsigned int i = 0; i < n_vertices; i++)
426 {
427 Point v1 = (*elem.node_ptr((i - 1) % n_vertices) - *elem.node_ptr(i % n_vertices));
428 Point v2 = (*elem.node_ptr((i + 1) % n_vertices) - *elem.node_ptr(i % n_vertices));
429 Real tmp = v1 * v2 / v1.norm() / v2.norm();
430 if (tmp > 1.0)
431 tmp = 1.0;
432 else if (tmp < -1.0)
433 tmp = -1.0;
434 angles.push_back(std::make_pair(acos(tmp), i));
435 }
436 std::sort(angles.begin(), angles.end(), std::greater<>());
437 return angles;
438}

440std::vector<std::pair<Real, unsigned int>>
441vertex_distances(const Elem & elem)
442{
443 std::vector<std::pair<Real, unsigned int>> distances;
444 const unsigned int n_vertices = elem.n_vertices();
445
446 for (unsigned int i = 0; i < n_vertices; i++)
447 {
448 Point v1 = (*elem.node_ptr((i + 1) % n_vertices) - *elem.node_ptr(i % n_vertices));
449 distances.push_back(std::make_pair(v1.norm(), i));
450 }
451 std::sort(distances.begin(), distances.end());
452 return distances;
453}

455void
456triElemSplitter(ReplicatedMesh & mesh,
457 const dof_id_type elem_id,
458 const unsigned short node_shift,
459 const dof_id_type nid_3,
460 const dof_id_type nid_4,
461 const subdomain_id_type single_elem_side_id,
462 const subdomain_id_type double_elem_side_id,
463 const boundary_id_type new_boundary_id)
464{
465 const auto elem_old = mesh.elem_ptr(elem_id);
466 const dof_id_type nid_0 = elem_old->node_ptr(node_shift % 3)->id();
467 const dof_id_type nid_1 = elem_old->node_ptr((1 + node_shift) % 3)->id();
468 const dof_id_type nid_2 = elem_old->node_ptr((2 + node_shift) % 3)->id();
469
470 const bool m1_side_flag =
471 MooseUtils::absoluteFuzzyEqual((*(mesh.node_ptr(nid_3)) - *(mesh.node_ptr(nid_0)))
472 .cross(*(mesh.node_ptr(nid_1)) - *(mesh.node_ptr(nid_0)))
473 .norm(),
474 0.0);
475 const dof_id_type nid_m1 = m1_side_flag ? nid_3 : nid_4;
476 const dof_id_type nid_m2 = m1_side_flag ? nid_4 : nid_3;
477 // Build boundary information of the mesh
478 BoundaryInfo & boundary_info = mesh.get_boundary_info();
479 auto bdry_side_list = boundary_info.build_side_list();
480 // Create a list of sidesets involving the element to be split
481 std::vector<std::vector<boundary_id_type>> elem_side_list;
482 elem_side_list.resize(3);
483 for (unsigned int i = 0; i < bdry_side_list.size(); i++)
484 {
485 if (std::get<0>(bdry_side_list[i]) == elem_id)
486 {
487 elem_side_list[(std::get<1>(bdry_side_list[i]) + 3 - node_shift) % 3].push_back(
488 std::get<2>(bdry_side_list[i]));
489 }
490 }
491
492 const unsigned int n_elem_extra_ids = mesh.n_elem_integers();
493 std::vector<dof_id_type> exist_extra_ids(n_elem_extra_ids);
494 // Record all the element extra integers of the original element
495 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
496 exist_extra_ids[j] = mesh.elem_ptr(elem_id)->get_extra_integer(j);
497
498 Elem * elem_Tri3_0 = mesh.add_elem(new Tri3);
499 elem_Tri3_0->set_node(0) = mesh.node_ptr(nid_0);
500 elem_Tri3_0->set_node(1) = mesh.node_ptr(nid_m1);
501 elem_Tri3_0->set_node(2) = mesh.node_ptr(nid_m2);
502 elem_Tri3_0->subdomain_id() = single_elem_side_id;
503 Elem * elem_Tri3_1 = mesh.add_elem(new Tri3);
504 elem_Tri3_1->set_node(0) = mesh.node_ptr(nid_1);
505 elem_Tri3_1->set_node(1) = mesh.node_ptr(nid_m2);
506 elem_Tri3_1->set_node(2) = mesh.node_ptr(nid_m1);
507 elem_Tri3_1->subdomain_id() = double_elem_side_id;
508 Elem * elem_Tri3_2 = mesh.add_elem(new Tri3);
509 elem_Tri3_2->set_node(0) = mesh.node_ptr(nid_2);
510 elem_Tri3_2->set_node(1) = mesh.node_ptr(nid_m2);
511 elem_Tri3_2->set_node(2) = mesh.node_ptr(nid_1);
512 elem_Tri3_2->subdomain_id() = double_elem_side_id;
513 // Retain element extra integers
514 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
515 {
516 elem_Tri3_0->set_extra_integer(j, exist_extra_ids[j]);
517 elem_Tri3_1->set_extra_integer(j, exist_extra_ids[j]);
518 elem_Tri3_2->set_extra_integer(j, exist_extra_ids[j]);
519 }
520
521 // Add sideset information to the new elements
522 for (const auto & side_info_0 : elem_side_list[0])
523 {
524 boundary_info.add_side(elem_Tri3_0, 0, side_info_0);
525 boundary_info.add_side(elem_Tri3_1, 2, side_info_0);
526 }
527 for (const auto & side_info_1 : elem_side_list[1])
528 boundary_info.add_side(elem_Tri3_2, 2, side_info_1);
529 for (const auto & side_info_2 : elem_side_list[2])
530 {
531 boundary_info.add_side(elem_Tri3_0, 2, side_info_2);
532 boundary_info.add_side(elem_Tri3_2, 0, side_info_2);
533 }
534 // Add cutting boundary to both sides as we are deleting one anyway
535 boundary_info.add_side(elem_Tri3_0, 1, new_boundary_id);
536 boundary_info.add_side(elem_Tri3_1, 1, new_boundary_id);
537}

539void
540triElemSplitter(ReplicatedMesh & mesh,
541 const dof_id_type elem_id,
542 const unsigned short node_shift,
543 const dof_id_type nid_m,
544 const subdomain_id_type first_elem_side_id,
545 const subdomain_id_type second_elem_side_id,
546 const boundary_id_type new_boundary_id)
547{
548 const auto elem_old = mesh.elem_ptr(elem_id);
549 const dof_id_type nid_0 = elem_old->node_ptr(node_shift % 3)->id();
550 const dof_id_type nid_1 = elem_old->node_ptr((1 + node_shift) % 3)->id();
551 const dof_id_type nid_2 = elem_old->node_ptr((2 + node_shift) % 3)->id();
552 // Build boundary information of the mesh
553 BoundaryInfo & boundary_info = mesh.get_boundary_info();
554 auto bdry_side_list = boundary_info.build_side_list();
555 // Create a list of sidesets involving the element to be split
556 std::vector<std::vector<boundary_id_type>> elem_side_list;
557 elem_side_list.resize(3);
558 for (unsigned int i = 0; i < bdry_side_list.size(); i++)
559 {
560 if (std::get<0>(bdry_side_list[i]) == elem_id)
561 {
562 elem_side_list[(std::get<1>(bdry_side_list[i]) + 3 - node_shift) % 3].push_back(
563 std::get<2>(bdry_side_list[i]));
564 }
565 }
566
567 const unsigned int n_elem_extra_ids = mesh.n_elem_integers();
568 std::vector<dof_id_type> exist_extra_ids(n_elem_extra_ids);
569 // Record all the element extra integers of the original element
570 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
571 exist_extra_ids[j] = mesh.elem_ptr(elem_id)->get_extra_integer(j);
572
573 Elem * elem_Tri3_0 = mesh.add_elem(new Tri3);
574 elem_Tri3_0->set_node(0) = mesh.node_ptr(nid_0);
575 elem_Tri3_0->set_node(1) = mesh.node_ptr(nid_1);
576 elem_Tri3_0->set_node(2) = mesh.node_ptr(nid_m);
577 elem_Tri3_0->subdomain_id() = first_elem_side_id;
578 Elem * elem_Tri3_1 = mesh.add_elem(new Tri3);
579 elem_Tri3_1->set_node(0) = mesh.node_ptr(nid_0);
580 elem_Tri3_1->set_node(1) = mesh.node_ptr(nid_m);
581 elem_Tri3_1->set_node(2) = mesh.node_ptr(nid_2);
582 elem_Tri3_1->subdomain_id() = second_elem_side_id;
583 // Retain element extra integers
584 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
585 {
586 elem_Tri3_0->set_extra_integer(j, exist_extra_ids[j]);
587 elem_Tri3_1->set_extra_integer(j, exist_extra_ids[j]);
588 }
589
590 // Add sideset information to the new elements
591 for (const auto & side_info_0 : elem_side_list[0])
592 boundary_info.add_side(elem_Tri3_0, 0, side_info_0);
593 for (const auto & side_info_1 : elem_side_list[1])
594 {
595 boundary_info.add_side(elem_Tri3_0, 1, side_info_1);
596 boundary_info.add_side(elem_Tri3_1, 1, side_info_1);
597 }
598 for (const auto & side_info_2 : elem_side_list[2])
599 boundary_info.add_side(elem_Tri3_0, 2, side_info_2);
600
601 // Add cutting boundary to both sides as we are deleting one anyway
602 boundary_info.add_side(elem_Tri3_0, 2, new_boundary_id);
603 boundary_info.add_side(elem_Tri3_1, 0, new_boundary_id);
604}

606void
607quadElemSplitter(ReplicatedMesh & mesh,
608 const dof_id_type elem_id,
609 const subdomain_id_type tri_elem_subdomain_shift)
610{
611 // Build boundary information of the mesh
612 BoundaryInfo & boundary_info = mesh.get_boundary_info();
613 auto bdry_side_list = boundary_info.build_side_list();
614 // Create a list of sidesets involving the element to be split
615 std::vector<std::vector<boundary_id_type>> elem_side_list;
616 elem_side_list.resize(4);
617 for (unsigned int i = 0; i < bdry_side_list.size(); i++)
618 {
619 if (std::get<0>(bdry_side_list[i]) == elem_id)
620 {
621 elem_side_list[std::get<1>(bdry_side_list[i])].push_back(std::get<2>(bdry_side_list[i]));
622 }
623 }
624
625 auto node_0 = mesh.elem_ptr(elem_id)->node_ptr(0);
626 auto node_1 = mesh.elem_ptr(elem_id)->node_ptr(1);
627 auto node_2 = mesh.elem_ptr(elem_id)->node_ptr(2);
628 auto node_3 = mesh.elem_ptr(elem_id)->node_ptr(3);
629
630 const unsigned int n_elem_extra_ids = mesh.n_elem_integers();
631 std::vector<dof_id_type> exist_extra_ids(n_elem_extra_ids);
632 // Record all the element extra integers of the original quad element
633 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
634 exist_extra_ids[j] = mesh.elem_ptr(elem_id)->get_extra_integer(j);
635
636 // There are two trivial ways to split a quad element
637 // We prefer the way that leads to triangles with similar areas
638 if (std::abs((*node_1 - *node_0).cross(*node_3 - *node_0).norm() -
639 (*node_1 - *node_2).cross(*node_3 - *node_2).norm()) >
640 std::abs((*node_0 - *node_1).cross(*node_2 - *node_1).norm() -
641 (*node_0 - *node_3).cross(*node_2 - *node_3).norm()))
642 {
643 Elem * elem_Tri3_0 = mesh.add_elem(new Tri3);
644 elem_Tri3_0->set_node(0) = node_0;
645 elem_Tri3_0->set_node(1) = node_1;
646 elem_Tri3_0->set_node(2) = node_2;
647 elem_Tri3_0->subdomain_id() = mesh.elem_ptr(elem_id)->subdomain_id() + tri_elem_subdomain_shift;
648 Elem * elem_Tri3_1 = mesh.add_elem(new Tri3);
649 elem_Tri3_1->set_node(0) = node_0;
650 elem_Tri3_1->set_node(1) = node_2;
651 elem_Tri3_1->set_node(2) = node_3;
652 elem_Tri3_1->subdomain_id() = mesh.elem_ptr(elem_id)->subdomain_id() + tri_elem_subdomain_shift;
653 // Retain element extra integers
654 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
655 {
656 elem_Tri3_0->set_extra_integer(j, exist_extra_ids[j]);
657 elem_Tri3_1->set_extra_integer(j, exist_extra_ids[j]);
658 }
659
660 // Add sideset information to the new elements
661 for (const auto & side_info_0 : elem_side_list[0])
662 boundary_info.add_side(elem_Tri3_0, 0, side_info_0);
663 for (const auto & side_info_1 : elem_side_list[1])
664 boundary_info.add_side(elem_Tri3_0, 1, side_info_1);
665 for (const auto & side_info_2 : elem_side_list[2])
666 boundary_info.add_side(elem_Tri3_1, 1, side_info_2);
667 for (const auto & side_info_3 : elem_side_list[3])
668 boundary_info.add_side(elem_Tri3_1, 2, side_info_3);
669 }
670 else
671 {
672 Elem * elem_Tri3_0 = mesh.add_elem(new Tri3);
673 elem_Tri3_0->set_node(0) = node_0;
674 elem_Tri3_0->set_node(1) = node_1;
675 elem_Tri3_0->set_node(2) = node_3;
676 elem_Tri3_0->subdomain_id() = mesh.elem_ptr(elem_id)->subdomain_id() + tri_elem_subdomain_shift;
677 Elem * elem_Tri3_1 = mesh.add_elem(new Tri3);
678 elem_Tri3_1->set_node(0) = node_1;
679 elem_Tri3_1->set_node(1) = node_2;
680 elem_Tri3_1->set_node(2) = node_3;
681 elem_Tri3_1->subdomain_id() = mesh.elem_ptr(elem_id)->subdomain_id() + tri_elem_subdomain_shift;
682 // Retain element extra integers
683 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
684 {
685 elem_Tri3_0->set_extra_integer(j, exist_extra_ids[j]);
686 elem_Tri3_1->set_extra_integer(j, exist_extra_ids[j]);
687 }
688
689 // Add sideset information to the new elements
690 for (const auto & side_info_0 : elem_side_list[0])
691 boundary_info.add_side(elem_Tri3_0, 0, side_info_0);
692 for (const auto & side_info_1 : elem_side_list[1])
693 boundary_info.add_side(elem_Tri3_1, 0, side_info_1);
694 for (const auto & side_info_2 : elem_side_list[2])
695 boundary_info.add_side(elem_Tri3_1, 1, side_info_2);
696 for (const auto & side_info_3 : elem_side_list[3])
697 boundary_info.add_side(elem_Tri3_0, 2, side_info_3);
698 }
699}

701void
702quadToTriOnLine(ReplicatedMesh & mesh,
703 const std::vector<Real> & cut_line_params,
704 const dof_id_type tri_subdomain_id_shift,
705 const SubdomainName tri_elem_subdomain_name_suffix)
706{
707 // Preprocess: find all the quad elements that are across the cutting line
708 std::vector<dof_id_type> cross_elems_quad;
709 std::set<subdomain_id_type> new_subdomain_ids;
710 for (auto elem_it = mesh.active_elements_begin(); elem_it != mesh.active_elements_end();
711 elem_it++)
712 {
713 if ((*elem_it)->n_vertices() == 4)
714 {
715 std::vector<unsigned short> node_side_rec;
716 for (unsigned int i = 0; i < 4; i++)
717 {
718 const Point v_point = (*elem_it)->point(i);
719 node_side_rec.push_back(lineSideDeterminator(v_point(0),
720 v_point(1),
721 cut_line_params[0],
722 cut_line_params[1],
723 cut_line_params[2],
724 true));
725 }
726 if (std::accumulate(node_side_rec.begin(), node_side_rec.end(), 0) != 4 &&
727 std::accumulate(node_side_rec.begin(), node_side_rec.end(), 0) > 0)
728 {
729 cross_elems_quad.push_back((*elem_it)->id());
730 new_subdomain_ids.emplace((*elem_it)->subdomain_id() + tri_subdomain_id_shift);
731 }
732 }
733 }
734 // Then convert these quad elements into tri elements
735 for (const auto & cross_elem_quad : cross_elems_quad)
736 {
737 quadElemSplitter(mesh, cross_elem_quad, tri_subdomain_id_shift);
738 mesh.delete_elem(mesh.elem_ptr(cross_elem_quad));
739 }
740 for (auto & nid : new_subdomain_ids)
741 {
742 const SubdomainName old_name = mesh.subdomain_name(nid - tri_subdomain_id_shift);
743 if (MooseMeshUtils::getSubdomainID(
744 (old_name.empty() ? (SubdomainName)(std::to_string(nid - tri_subdomain_id_shift))
745 : old_name) +
746 "_" + tri_elem_subdomain_name_suffix,
747 mesh) != Moose::INVALID_BLOCK_ID)
748 throw MooseException("The new subdomain name already exists in the mesh.");
749 mesh.subdomain_name(nid) =
750 (old_name.empty() ? (SubdomainName)(std::to_string(nid - tri_subdomain_id_shift))
751 : old_name) +
752 "_" + tri_elem_subdomain_name_suffix;
753 mooseWarning("QUAD elements have been converted into TRI elements with a new "
754 "subdomain name: " +
755 mesh.subdomain_name(nid) + ".");
756 }
757 mesh.contract();
758}

760void
761lineRemoverCutElemTri(ReplicatedMesh & mesh,
762 const std::vector<Real> & cut_line_params,
763 const subdomain_id_type block_id_to_remove,
764 const boundary_id_type new_boundary_id)
765{
766 // Find all the elements that are across the cutting line
767 std::vector<dof_id_type> cross_elems;
768 // A vector for element specific information
769 std::vector<std::vector<std::pair<dof_id_type, dof_id_type>>> node_pairs_vec;
770 // A set for unique pairs
771 std::vector<std::pair<dof_id_type, dof_id_type>> node_pairs_unique_vec;
772 for (auto elem_it = mesh.active_elements_begin(); elem_it != mesh.active_elements_end();
773 elem_it++)
774 {
775 std::vector<unsigned short> node_side_rec;
776 const auto n_vertices = (*elem_it)->n_vertices();
777 node_side_rec.resize(n_vertices);
778 for (unsigned int i = 0; i < n_vertices; i++)
779 {
780 // First check if the vertex is in the XY Plane
781 if (!MooseUtils::absoluteFuzzyEqual((*elem_it)->point(i)(2), 0.0))
782 mooseError("MooseMeshXYCuttingUtils::lineRemoverCutElemTri() only works for 2D meshes in "
783 "XY Plane.");
784 const Point v_point = (*elem_it)->point(i);
785 node_side_rec[i] = lineSideDeterminator(
786 v_point(0), v_point(1), cut_line_params[0], cut_line_params[1], cut_line_params[2], true);
787 }
788 if (std::accumulate(node_side_rec.begin(), node_side_rec.end(), 0) == (int)node_side_rec.size())
789 {
790 (*elem_it)->subdomain_id() = block_id_to_remove;
791 }
792 else if (std::accumulate(node_side_rec.begin(), node_side_rec.end(), 0) > 0)
793 {
794 if ((*elem_it)->n_vertices() != 3 || (*elem_it)->n_nodes() != 3)
795 mooseError("The element across the cutting line is not TRI3, which is not supported.");
796 cross_elems.push_back((*elem_it)->id());
797 // Then we need to check pairs of nodes that are on the different side
798 std::vector<std::pair<dof_id_type, dof_id_type>> node_pairs;
799 for (unsigned int i = 0; i < node_side_rec.size(); i++)
800 {
801 // first node on removal side and second node on retaining side
802 if (node_side_rec[i] > 0 && node_side_rec[(i + 1) % node_side_rec.size()] == 0)
803 {
804 // Removal side first
805 node_pairs.push_back(
806 std::make_pair((*elem_it)->node_ptr(i)->id(),
807 (*elem_it)->node_ptr((i + 1) % node_side_rec.size())->id()));
808 node_pairs_unique_vec.push_back(node_pairs.back());
809 }
810 // first node on retaining side and second node on removal side
811 else if (node_side_rec[i] == 0 && node_side_rec[(i + 1) % node_side_rec.size()] > 0)
812 {
813 // Removal side first
814 node_pairs.push_back(
815 std::make_pair((*elem_it)->node_ptr((i + 1) % node_side_rec.size())->id(),
816 (*elem_it)->node_ptr(i)->id()));
817 node_pairs_unique_vec.push_back(node_pairs.back());
818 }
819 }
820 node_pairs_vec.push_back(node_pairs);
821 }
822 }
823 auto vec_ip = std::unique(node_pairs_unique_vec.begin(), node_pairs_unique_vec.end());
824 node_pairs_unique_vec.resize(std::distance(node_pairs_unique_vec.begin(), vec_ip));
825
826 // Loop over all the node pairs to define new nodes that sit on the cutting line
827 std::vector<Node *> nodes_on_line;
828 // whether the on-line node is overlapped with the node pairs or a brand new node
829 std::vector<unsigned short> nodes_on_line_overlap;
830 for (const auto & node_pair : node_pairs_unique_vec)
831 {
832 const Point pt1 = *mesh.node_ptr(node_pair.first);
833 const Point pt2 = *mesh.node_ptr(node_pair.second);
834 const Point pt_line = twoPointandLineIntersection(
835 pt1, pt2, cut_line_params[0], cut_line_params[1], cut_line_params[2]);
836 if ((pt_line - pt1).norm() < libMesh::TOLERANCE)
837 {
838 nodes_on_line.push_back(mesh.node_ptr(node_pair.first));
839 nodes_on_line_overlap.push_back(1);
840 }
841 else if ((pt_line - pt2).norm() < libMesh::TOLERANCE)
842 {
843 nodes_on_line.push_back(mesh.node_ptr(node_pair.second));
844 nodes_on_line_overlap.push_back(2);
845 }
846 else
847 {
848 nodes_on_line.push_back(mesh.add_point(pt_line));
849 nodes_on_line_overlap.push_back(0);
850 }
851 }
852
853 // make new elements
854 for (unsigned int i = 0; i < cross_elems.size(); i++)
855 {
856 // Only TRI elements are involved after preprocessing
857 auto cross_elem = mesh.elem_ptr(cross_elems[i]);
858 auto node_0 = cross_elem->node_ptr(0);
859 auto node_1 = cross_elem->node_ptr(1);
860 auto node_2 = cross_elem->node_ptr(2);
861 const std::vector<dof_id_type> tri_nodes = {node_0->id(), node_1->id(), node_2->id()};
862
863 const auto online_node_index_1 = std::distance(node_pairs_unique_vec.begin(),
864 std::find(node_pairs_unique_vec.begin(),
865 node_pairs_unique_vec.end(),
866 node_pairs_vec[i][0]));
867 const auto online_node_index_2 = std::distance(node_pairs_unique_vec.begin(),
868 std::find(node_pairs_unique_vec.begin(),
869 node_pairs_unique_vec.end(),
870 node_pairs_vec[i][1]));
871 auto node_3 = nodes_on_line[online_node_index_1];
872 auto node_4 = nodes_on_line[online_node_index_2];
873 const auto node_3_overlap_flag = nodes_on_line_overlap[online_node_index_1];
874 const auto node_4_overlap_flag = nodes_on_line_overlap[online_node_index_2];
875 // Most common case, no overlapped nodes
876 if (node_3_overlap_flag == 0 && node_4_overlap_flag == 0)
877 {
878 // True if the common node is on the removal side; false if on the retaining side
879 const bool common_node_side = node_pairs_vec[i][0].first == node_pairs_vec[i][1].first;
880 const subdomain_id_type block_id_to_assign_1 =
881 common_node_side ? block_id_to_remove : cross_elem->subdomain_id();
882 const subdomain_id_type block_id_to_assign_2 =
883 common_node_side ? cross_elem->subdomain_id() : block_id_to_remove;
884 // The reference node ids need to be adjusted according to the common node of the two cut
885 // sides
886 const dof_id_type common_node_id =
887 common_node_side ? node_pairs_vec[i][0].first : node_pairs_vec[i][0].second;
888
889 triElemSplitter(mesh,
890 cross_elem->id(),
891 std::distance(tri_nodes.begin(),
892 std::find(tri_nodes.begin(), tri_nodes.end(), common_node_id)),
893 node_3->id(),
894 node_4->id(),
895 block_id_to_assign_1,
896 block_id_to_assign_2,
897 new_boundary_id);
898 mesh.delete_elem(cross_elem);
899 }
900 // both node_3 and node_4 are overlapped
901 else if (node_3_overlap_flag > 0 && node_4_overlap_flag > 0)
902 {
903 // In this case, the entire element is on one side of the cutting line
904 // No change needed just check which side the element is on
905 cross_elem->subdomain_id() = lineSideDeterminator(cross_elem->vertex_average()(0),
906 cross_elem->vertex_average()(1),
907 cut_line_params[0],
908 cut_line_params[1],
909 cut_line_params[2],
910 true)
911 ? block_id_to_remove
912 : cross_elem->subdomain_id();
913 }
914 // node_3 or node_4 is overlapped
915 else
916 {
917 const auto node_3_finder = std::distance(
918 tri_nodes.begin(), std::find(tri_nodes.begin(), tri_nodes.end(), node_3->id()));
919 const auto node_4_finder = std::distance(
920 tri_nodes.begin(), std::find(tri_nodes.begin(), tri_nodes.end(), node_4->id()));
921 // As only one of the two above values should be less than the three, the smaller one should
922 // be used
923 const dof_id_type node_id = node_3_finder < node_4_finder ? node_4->id() : node_3->id();
924 const auto node_finder = std::min(node_3_finder, node_4_finder);
925
926 triElemSplitter(
927 mesh,
928 cross_elem->id(),
929 node_finder,
930 node_id,
931 tri_nodes[(node_finder + 1) % 3] == node_pairs_vec[i][node_3_finder > node_4_finder].first
932 ? block_id_to_remove
933 : cross_elem->subdomain_id(),
934 tri_nodes[(node_finder + 1) % 3] == node_pairs_vec[i][node_3_finder > node_4_finder].first
935 ? cross_elem->subdomain_id()
936 : block_id_to_remove,
937 new_boundary_id);
938 mesh.delete_elem(cross_elem);
939 }
940 }
941 mesh.contract();
942
943 // Delete the block to remove
944 for (auto elem_it = mesh.active_subdomain_elements_begin(block_id_to_remove);
945 elem_it != mesh.active_subdomain_elements_end(block_id_to_remove);
946 elem_it++)
947 mesh.delete_elem(*elem_it);
948 mesh.contract();
949}

953 const std::vector<Real> & cut_line_params,
954 const dof_id_type tri_subdomain_id_shift,
955 const SubdomainName tri_elem_subdomain_name_suffix,
956 const subdomain_id_type block_id_to_remove,

967 boundaryTriElemImprover(mesh, new_boundary_id);

970void
971boundaryTriElemImprover(ReplicatedMesh & mesh, const boundary_id_type boundary_to_improve)
972{
973 if (!MooseMeshUtils::hasBoundaryID(mesh, boundary_to_improve))
974 mooseError(
975 "MooseMeshXYCuttingUtils::boundaryTriElemImprover(): The boundary_to_improve provided "
976 "does not exist in the given mesh.");
977 BoundaryInfo & boundary_info = mesh.get_boundary_info();
978 auto side_list = boundary_info.build_side_list();
979 // Here we would like to collect the following information for all the TRI3 elements on the
980 // boundary: Key: node id of the off-boundary node Value: a vector of tuples, each tuple contains
981 // the following information:
982 // 1. The element id of the element that is on the boundary to improve
983 // 2. the one node id of that element that is on the boundary to improve
984 // 3. the other node id of the element that is on the boundary to improve
985 std::map<dof_id_type, std::vector<std::tuple<dof_id_type, dof_id_type, dof_id_type>>>
986 tri3_elem_info;
987 for (const auto & side : side_list)
988 {
989 if (std::get<2>(side) == boundary_to_improve)
990 {
991 Elem * elem = mesh.elem_ptr(std::get<0>(side));
992 if (elem->type() == TRI3)
993 {
994 const auto key_node_id = elem->node_id((std::get<1>(side) + 2) % 3);
995 const auto value_elem_id = elem->id();
996 const auto value_node_id_1 = elem->node_id(std::get<1>(side));
997 const auto value_node_id_2 = elem->node_id((std::get<1>(side) + 1) % 3);
998 tri3_elem_info[key_node_id].push_back(
999 std::make_tuple(value_elem_id, value_node_id_1, value_node_id_2));
1000 }
1001 }
1002 }
1003 // Elements that need to be removed
1004 std::vector<dof_id_type> elems_to_remove;
1005 // Now check if any group of TRI3 sharing an off-boundary node can be improved.
1006 for (const auto & tri_group : tri3_elem_info)
1007 {
1008 // It is possible to improve only when more than one TRI3 elements share the same off-boundary
1009 // node
1010 std::vector<std::pair<dof_id_type, dof_id_type>> node_assm;
1011 std::vector<dof_id_type> elem_id_list;
1012 for (const auto & tri : tri_group.second)
1013 {
1014 node_assm.push_back(std::make_pair(std::get<1>(tri), std::get<2>(tri)));
1015 elem_id_list.push_back(std::get<0>(tri));
1016 }
1017 std::vector<dof_id_type> ordered_node_list;
1018 std::vector<dof_id_type> ordered_elem_list;
1019 MooseMeshUtils::makeOrderedNodeList(
1020 node_assm, elem_id_list, ordered_node_list, ordered_elem_list);
1021
1022 // For all the elements sharing the same off-boundary node, we need to know how many separated
1023 // subdomains are involved
1024 // If there are extra element ids defined on the mesh, they also want to retain their boundaries
1025 // Only triangle elements that share a side can be merged
1026 const unsigned int n_elem_extra_ids = mesh.n_elem_integers();
1027 std::vector<std::tuple<subdomain_id_type, std::vector<dof_id_type>, unsigned int>> blocks_info;
1028 for (const auto & elem_id : ordered_elem_list)
1029 {
1030 std::vector<dof_id_type> exist_extra_ids(n_elem_extra_ids);
1031 // Record all the element extra integers of the original quad element
1032 for (unsigned int j = 0; j < n_elem_extra_ids; j++)
1033 exist_extra_ids[j] = mesh.elem_ptr(elem_id)->get_extra_integer(j);
1034 if (!blocks_info.empty())
1035 {
1036 if (mesh.elem_ptr(elem_id)->subdomain_id() == std::get<0>(blocks_info.back()) &&
1037 exist_extra_ids == std::get<1>(blocks_info.back()))
1038 {
1039 std::get<2>(blocks_info.back())++;
1040 continue;
1041 }
1042 }
1043 blocks_info.push_back(
1044 std::make_tuple(mesh.elem_ptr(elem_id)->subdomain_id(), exist_extra_ids, 1));
1045 }
1046 // For each separated subdomain / set of extra ids, we try to improve the boundary elements
1047 unsigned int side_counter = 0;
1048 for (const auto & block_info : blocks_info)
1049 {
1050 const auto node_1 = mesh.node_ptr(ordered_node_list[side_counter]);
1051 // we do not need to subtract 1 for node_2
1052 const auto node_2 = mesh.node_ptr(ordered_node_list[side_counter + std::get<2>(block_info)]);
1053 const auto node_0 = mesh.node_ptr(tri_group.first);
1054 const Point v1 = *node_1 - *node_0;
1055 const Point v2 = *node_2 - *node_0;
1056 const Real angle = std::acos(v1 * v2 / v1.norm() / v2.norm()) / M_PI * 180.0;
1057 const std::vector<dof_id_type> block_elems(ordered_elem_list.begin() + side_counter,
1058 ordered_elem_list.begin() + side_counter +
1059 std::get<2>(block_info));
1060 // We assume that there are no sidesets defined inside a subdomain
1061 // For the first TRI3 element, we want to check if its side defined by node_0 and node_1 is
1062 // defined in any sidesets
1063 unsigned short side_id_0;
1064 unsigned short side_id_t;
1065 bool is_inverse_0;
1066 bool is_inverse_t;
1067 elemSideLocator(mesh,
1068 block_elems.front(),
1069 tri_group.first,
1070 ordered_node_list[side_counter],
1071 side_id_0,
1072 is_inverse_0);
1073 elemSideLocator(mesh,
1074 block_elems.back(),
1075 ordered_node_list[side_counter + std::get<2>(block_info)],
1076 tri_group.first,
1077 side_id_t,
1078 is_inverse_t);
1079 // Collect boundary information of the identified sides
1080 std::vector<boundary_id_type> side_0_boundary_ids;
1081 boundary_info.boundary_ids(
1082 mesh.elem_ptr(block_elems.front()), side_id_0, side_0_boundary_ids);
1083 std::vector<boundary_id_type> side_t_boundary_ids;
1084 boundary_info.boundary_ids(mesh.elem_ptr(block_elems.back()), side_id_t, side_t_boundary_ids);
1085
1086 // Ideally we want this angle to be 60 degrees
1087 // In reality, we want one TRI3 element if the angle is less than 90 degrees;
1088 // we want two TRI3 elements if the angle is greater than 90 degrees and less than 135
1089 // degrees; we want three TRI3 elements if the angle is greater than 135 degrees and less than
1090 // 180 degrees.
1091 if (angle < 90.0)
1092 {
1093 if (std::get<2>(block_info) > 1)
1094 {
1095 makeImprovedTriElement(mesh,
1096 tri_group.first,
1097 ordered_node_list[side_counter],
1098 ordered_node_list[side_counter + std::get<2>(block_info)],
1099 std::get<0>(block_info),
1100 std::get<1>(block_info),
1101 {boundary_to_improve},
1102 side_0_boundary_ids,
1103 side_t_boundary_ids);
1104 elems_to_remove.insert(elems_to_remove.end(), block_elems.begin(), block_elems.end());
1105 }
1106 }
1107 else if (angle < 135.0)
1108 {
1109 // We can just add the middle node because there's nothing on the other side
1110 const auto node_m = mesh.add_point((*node_1 + *node_2) / 2.0);
1111 makeImprovedTriElement(mesh,
1112 tri_group.first,
1113 ordered_node_list[side_counter],
1114 node_m->id(),
1115 std::get<0>(block_info),
1116 std::get<1>(block_info),
1117 {boundary_to_improve},
1118 side_0_boundary_ids,
1119 std::vector<boundary_id_type>());
1120 makeImprovedTriElement(mesh,
1121 tri_group.first,
1122 node_m->id(),
1123 ordered_node_list[side_counter + std::get<2>(block_info)],
1124 std::get<0>(block_info),
1125 std::get<1>(block_info),
1126 {boundary_to_improve},
1127 std::vector<boundary_id_type>(),
1128 side_t_boundary_ids);
1129 elems_to_remove.insert(elems_to_remove.end(), block_elems.begin(), block_elems.end());
1130 }
1131 else
1132 {
1133 const auto node_m1 = mesh.add_point((*node_1 * 2.0 + *node_2) / 3.0);
1134 const auto node_m2 = mesh.add_point((*node_1 + *node_2 * 2.0) / 3.0);
1135 makeImprovedTriElement(mesh,
1136 tri_group.first,
1137 ordered_node_list[side_counter],
1138 node_m1->id(),
1139 std::get<0>(block_info),
1140 std::get<1>(block_info),
1141 {boundary_to_improve},
1142 side_0_boundary_ids,
1143 std::vector<boundary_id_type>());
1144 makeImprovedTriElement(mesh,
1145 tri_group.first,
1146 node_m1->id(),
1147 node_m2->id(),
1148 std::get<0>(block_info),
1149 std::get<1>(block_info),
1150 {boundary_to_improve},
1151 std::vector<boundary_id_type>(),
1152 std::vector<boundary_id_type>());
1153 makeImprovedTriElement(mesh,
1154 tri_group.first,
1155 node_m2->id(),
1156 ordered_node_list[side_counter + std::get<2>(block_info)],
1157 std::get<0>(block_info),
1158 std::get<1>(block_info),
1159 {boundary_to_improve},
1160 std::vector<boundary_id_type>(),
1161 side_t_boundary_ids);
1162 elems_to_remove.insert(elems_to_remove.end(), block_elems.begin(), block_elems.end());
1163 }
1164 side_counter += std::get<2>(block_info);
1165 }
1166 // TODO: Need to check if the new element is inverted?
1167 }
1168 // Delete the original elements
1169 for (const auto & elem_to_remove : elems_to_remove)
1170 mesh.delete_elem(mesh.elem_ptr(elem_to_remove));
1171 mesh.contract();
1172}
