43template <typename T, bool is_ad>
44GenericReal<is_ad>
45PenaltyInterfaceDiffusionTempl<T, is_ad>::computeQpResidual(Moose::DGResidualType type)
46{
47 GenericReal<is_ad> r = 0;
48
49 Moose::GenericType<T, is_ad> jump_value = 0;
50
51 if (_jump != nullptr)
52 jump_value = (*_jump)[_qp];
53 else
54 jump_value = _u[_qp] - _neighbor_value[_qp];
55
56 switch (type)
57 {
58 case Moose::Element:
59 r = _test[_i][_qp] * _penalty * jump_value;
60 break;
61
62 case Moose::Neighbor:
63 r = _test_neighbor[_i][_qp] * -_penalty * jump_value;
64 break;
65 }
66
67 return r;
68}

70template <typename T, bool is_ad>
71Real
72PenaltyInterfaceDiffusionTempl<T, is_ad>::computeQpJacobian(Moose::DGJacobianType type)
73{
74 Real jac = 0;
75
76 switch (type)
77 {
78 case Moose::ElementElement:
79 jac = _test[_i][_qp] * _penalty * _phi[_j][_qp];
80 break;
81
82 case Moose::ElementNeighbor:
83 jac = _test[_i][_qp] * _penalty * -_phi_neighbor[_j][_qp];
84 break;
85
86 case Moose::NeighborElement:
87 jac = _test_neighbor[_i][_qp] * -_penalty * _phi[_j][_qp];
88 break;
89
90 case Moose::NeighborNeighbor:
91 jac = _test_neighbor[_i][_qp] * -_penalty * -_phi_neighbor[_j][_qp];
92 break;
93 }
94
95 return jac;
96}
