32ADReal
33EqualGradientConstraint::computeQpResidual(Moose::MortarType mortar_type)
34{
35 switch (mortar_type)
36 {
37 case Moose::MortarType::Secondary:
38 return -_lambda[_qp] * _grad_test_secondary[_i][_qp](_component);
39 case Moose::MortarType::Primary:
40 return _lambda[_qp] * _grad_test_primary[_i][_qp](_component);
41 case Moose::MortarType::Lower:
42 return (_grad_u_primary[_qp](_component) - _grad_u_secondary[_qp](_component)) *
43 _test[_i][_qp];
44 default:
45 return 0;
46 }
47}
