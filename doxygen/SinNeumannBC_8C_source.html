36Real
37SinNeumannBC::computeQpResidual()
38{
39 Real value;
40
41 if (_t < _duration)
42 value = _initial + (_final - _initial) * std::sin((0.5 / _duration) * libMesh::pi * _t);
43 else
44 value = _final;
45
46 return -_test[_i][_qp] * value;
47}
