47ADReal
48Function::value(const ADReal & t, const ADPoint & p) const
49{
50 const auto rt = MetaPhysicL::raw_value(t);
51 const auto rp = MetaPhysicL::raw_value(p);
52 const auto grad = gradient(rt, rp);
53 ADReal ret = value(rt, rp);
54 ret.derivatives() = grad(0) * p(0).derivatives()
55#if LIBMESH_DIM > 1
56 + grad(1) * p(1).derivatives()
57#endif
58#if LIBMESH_DIM > 2
59 + grad(2) * p(2).derivatives()
60#endif
61 + timeDerivative(rt, rp) * t.derivatives();
62 return ret;
63}

127typename Function::ValueType
128Function::evaluate(const FaceArg & face, const Moose::StateArg & state) const
129{
130 if (face.face_side && face.fi->neighborPtr() &&
131 (face.fi->elem().subdomain_id() != face.fi->neighbor().subdomain_id()))
132 {
133 // Some users like to put discontinuities in their functions at subdomain changes in which case
134 // in order to always get the proper discontinuous effect we should evaluate ever so slightly
135 // off the face. Consider evaluation of: if(x < 0, -1, 1) if the face centroid is right at x ==
136 // 0 for example. The user likely doesn't want you to return 1 if they've asked for a 0-
137 // evaluation
138 //
139 // I can't quite tell but I think the tolerance for comparing coordinates (x, y, z, t) in
140 // fparser is ~1e-9 so we need to use something larger than that. The comparison is absolute
141 static constexpr Real offset_tolerance = 1e-8;
142 auto offset = offset_tolerance * face.fi->normal();
143 if (face.face_side == face.fi->elemPtr())
144 offset *= -1;
145 return value(_ti_feproblem.getTimeFromStateArg(state), face.getPoint() + offset);
146 }
147 else
148 return value(_ti_feproblem.getTimeFromStateArg(state), face.getPoint());
149}
