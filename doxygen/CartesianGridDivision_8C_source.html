20CartesianGridDivision::validParams()
21{
22 InputParameters params = MeshDivision::validParams();
23 params.addClassDescription("Divide the mesh along a Cartesian grid. Numbering increases from "
24 "bottom to top and from left to right and from back to front. The "
25 "inner ordering is X, then Y, then Z");
26 params.addParam<Point>("bottom_left", "Bottom-back-left corner of the grid");
27 params.addParam<Point>("top_right", "Top-front-right corner of the grid");
28 params.addParam<Point>("center", "Center of the Cartesian grid");
29 params.addParam<PositionsName>("center_positions",
30 "Positions of the centers of divided Cartesian grids");
31 params.addParam<Point>("widths", "Widths in the X, Y and Z directions");
32 params.addRequiredRangeCheckedParam<unsigned int>("nx", "nx>0", "Number of divisions in X");
33 params.addRequiredRangeCheckedParam<unsigned int>("ny", "ny>0", "Number of divisions in Y");
34 params.addRequiredRangeCheckedParam<unsigned int>("nz", "nz>0", "Number of divisions in Z");
35 params.addParam<bool>(
36 "assign_domain_outside_grid_to_border",
37 false,
38 "Whether to map the domain outside the grid back to the border of the grid");
39

43CartesianGridDivision::CartesianGridDivision(const InputParameters & parameters)
44 : MeshDivision(parameters),
45 _bottom_left(isParamValid("bottom_left") ? getParam<Point>("bottom_left") : Point(0, 0, 0)),
46 _top_right(isParamValid("top_right") ? getParam<Point>("top_right") : Point(0, 0, 0)),
47 _center(isParamValid("center") ? &getParam<Point>("center") : nullptr),
48 _center_positions(
49 isParamValid("center_positions")
50 ? &_fe_problem->getPositionsObject(getParam<PositionsName>("center_positions"))
51 : nullptr),
52 _widths(isParamValid("widths") ? getParam<Point>("widths") : Point(_top_right - _bottom_left)),
53 _nx(getParam<unsigned int>("nx")),
54 _ny(getParam<unsigned int>("ny")),
55 _nz(getParam<unsigned int>("nz")),
56 _outside_grid_counts_as_border(getParam<bool>("assign_domain_outside_grid_to_border"))
57{
58 CartesianGridDivision::initialize();
59
60 // Check non-overlapping inputs for the dimensions of the grid
61 if ((_center || _center_positions) && (isParamValid("bottom_left") || isParamValid("top_right")))
62 mooseError("Either the center or the edges of the grids must be specified");
63 if ((isParamValid("top_right") + isParamValid("bottom_left") == 1) && !isParamValid("widths"))
64 paramError("bottom_left",
65 "Both bottom_left and top_right must be passed to be able to determine the width");
66
67 // Pre-determine the bounds if we can
68 if (!_center_positions && _center)
69 {
70 _bottom_left = *_center - _widths / 2;
71 _top_right = *_center + _widths / 2;
72 }
73
74 // Check widths
75 if (_widths(0) < 0)
76 paramError("top_right",
77 "Top-front-right corner must be right (X axis) of bottom-left-back corner");
78 if (_widths(1) < 0)
79 paramError("top_right",
80 "Top-front-right corner must be in front (Y axis) of bottom-left-back corner");
81 if (_widths(2) < 0)
82 paramError("top_right",
83 "Top-front-right corner must be on top (Z axis) of bottom-left-back corner");
84 if ((_nx > 1) && MooseUtils::absoluteFuzzyEqual(_widths(0), 0))
85 paramError("nx", "Subdivision number must be 1 if width is 0 in X direction");
86 if ((_ny > 1) && MooseUtils::absoluteFuzzyEqual(_widths(1), 0))
87 paramError("ny", "Subdivision number must be 1 if width is 0 in Y direction");
88 if ((_nz > 1) && MooseUtils::absoluteFuzzyEqual(_widths(2), 0))
89 paramError("nz", "Subdivision number must be 1 if width is 0 in Z direction");
90}

92void
93CartesianGridDivision::initialize()
94{
95 setNumDivisions(_nx * _ny * _nz);
96
97 // Check that the grid is well-defined
98 if (_center_positions)
99 {
100 Real min_dist = _widths.norm();
101 Real min_center_dist = _center_positions->getMinDistanceBetweenPositions();
102 // Note that if the positions are not co-planar, the distance reported would be bigger but there
103 // could still be an overlap. Looking at min_center_dist is not enough
104 if (min_dist > min_center_dist)
105 mooseError(
106 "Cartesian grids centered on the positions are too close to each other (min distance: ",
107 min_center_dist,
108 "), closer than the extent of each grid. Mesh division is ill-defined");
109 }
110
111 // Examine mesh bounding box to see if the mesh may not be fully contained
112 _mesh_fully_indexed = true;
113 if (!_outside_grid_counts_as_border)
114 for (auto i : make_range(LIBMESH_DIM))
115 {
116 if (_center_positions)
117 {
118 _mesh_fully_indexed = false;
119 break;
120 }
121 else if (_bottom_left(i) > _mesh.getInflatedProcessorBoundingBox(0).first(i) ||
122 _top_right(i) < _mesh.getInflatedProcessorBoundingBox(0).second(i))
123 {
124 _mesh_fully_indexed = false;
125 break;
126 }
127 }
128}

136unsigned int
137CartesianGridDivision::divisionIndex(const Point & pt) const
138{
139 unsigned int offset = 0;
140 // Determine the local grid bounds
141 Point bottom_left, top_right, p;
142 if (_center_positions)
143 {
144 // If dividing using positions, find the closest position and
145 // look at the relative position of the point compared to that position
146 const bool initial = _fe_problem->getCurrentExecuteOnFlag() == EXEC_INITIAL;
147 const auto nearest_grid_center_index = _center_positions->getNearestPositionIndex(pt, initial);
148 offset = nearest_grid_center_index * getNumDivisions();
149 const auto nearest_grid_center =
150 _center_positions->getPosition(nearest_grid_center_index, initial);
151 bottom_left = -_widths / 2;
152 top_right = _widths / 2;
153 p = pt - nearest_grid_center;
154 }
155 else
156 {
157 bottom_left = _bottom_left;
158 top_right = _top_right;
159 p = pt;
160 }
161
162 if (!_outside_grid_counts_as_border)
163 {
164 if (MooseUtils::absoluteFuzzyLessThan(p(0), bottom_left(0)) ||
165 MooseUtils::absoluteFuzzyGreaterThan(p(0), top_right(0)))
166 return MooseMeshDivision::INVALID_DIVISION_INDEX;
167 if (MooseUtils::absoluteFuzzyLessThan(p(1), bottom_left(1)) ||
168 MooseUtils::absoluteFuzzyGreaterThan(p(1), top_right(1)))
169 return MooseMeshDivision::INVALID_DIVISION_INDEX;
170 if (MooseUtils::absoluteFuzzyLessThan(p(2), bottom_left(2)) ||
171 MooseUtils::absoluteFuzzyGreaterThan(p(2), top_right(2)))
172 return MooseMeshDivision::INVALID_DIVISION_INDEX;
173 }
174
175 const auto not_found = MooseMeshDivision::INVALID_DIVISION_INDEX;
176 auto ix = not_found, iy = not_found, iz = not_found;
177
178 // Look inside the grid and on the left / back / bottom
179 for (const auto jx : make_range(_nx + 1))
180 {
181 const auto border_x = bottom_left(0) + _widths(0) * jx / _nx;
182 if (jx > 0 && jx < _nx && MooseUtils::absoluteFuzzyEqual(border_x, p(0)))
183 mooseWarning(
184 "Querying the division index for a point of a boundary between two regions in X: " +
185 Moose::stringify(p));
186 if (border_x >= p(0))
187 {
188 ix = (jx > 0) ? jx - 1 : 0;
189 break;
190 }
191 }
192 for (const auto jy : make_range(_ny + 1))
193 {
194 const auto border_y = bottom_left(1) + _widths(1) * jy / _ny;
195 if (jy > 0 && jy < _ny && MooseUtils::absoluteFuzzyEqual(border_y, p(1)))
196 mooseWarning(
197 "Querying the division index for a point of a boundary between two regions in Y: " +
198 Moose::stringify(p));
199 if (border_y >= p(1))
200 {
201 iy = (jy > 0) ? jy - 1 : 0;
202 break;
203 }
204 }
205 for (const auto jz : make_range(_nz + 1))
206 {
207 const auto border_z = bottom_left(2) + _widths(2) * jz / _nz;
208 if (jz > 0 && jz < _nz && MooseUtils::absoluteFuzzyEqual(border_z, p(2)))
209 mooseWarning(
210 "Querying the division index for a point of a boundary between two regions in Z: " +
211 Moose::stringify(p));
212 if (border_z >= p(2))
213 {
214 iz = (jz > 0) ? jz - 1 : 0;
215 break;
216 }
217 }
218
219 // Look on the right / front / top of the grid
220 if (MooseUtils::absoluteFuzzyGreaterEqual(p(0), top_right(0)))
221 ix = _nx - 1;
222 if (MooseUtils::absoluteFuzzyGreaterEqual(p(1), top_right(1)))
223 iy = _ny - 1;
224 if (MooseUtils::absoluteFuzzyGreaterEqual(p(2), top_right(2)))
225 iz = _nz - 1;
226
227 // Handle edge case on widths
228 if (ix == not_found && MooseUtils::absoluteFuzzyEqual(_widths(0), 0))
229 ix = 0;
230 if (iy == not_found && MooseUtils::absoluteFuzzyEqual(_widths(1), 0))
231 iy = 0;
232 if (iz == not_found && MooseUtils::absoluteFuzzyEqual(_widths(2), 0))
233 iz = 0;
234 mooseAssert(ix != not_found, "We should have found a mesh division bin in X");
235 mooseAssert(iy != not_found, "We should have found a mesh division bin in Y");
236 mooseAssert(iz != not_found, "We should have found a mesh division bin in Z");
237
238 return offset + ix + _nx * iy + iz * _nx * _ny;
239}
