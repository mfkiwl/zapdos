29Real
30WeightedGapAux::computeValue()
31{
32 _weighted_gap = 0;
33 for (_qp = 0; _qp < _qrule_msm->n_points(); _qp++)
34 {
35 computeQpProperties();
36 for (_i = 0; _i < _test_lower.size(); ++_i)
37 computeQpIProperties();
38 }
39
40 return _weighted_gap;
41}

43void
44WeightedGapAux::computeQpProperties()
45{
46 const auto gap_vec = _phys_points_primary[_qp] - _phys_points_secondary[_qp];
47
48 _qp_gap_nodal = gap_vec * (_JxW_msm[_qp] * _coord_msm[_qp]);
49
50 _msm_volume += _JxW_msm[_qp] * _coord_msm[_qp];
51}

53void
54WeightedGapAux::computeQpIProperties()
55{
56 _weighted_gap += _test_lower[_i][_qp] * _qp_gap_nodal * _normals[_i];
