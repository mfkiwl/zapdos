54void
55JSONFileReader::getScalar(const std::vector<std::string> & scalar_keys, Real & scalar) const
56{
57 if (!scalar_keys.size())
58 mooseError("There should be at least one key to retrieve a value from the JSON");
59
60 // traverse the JSON tree
61 auto * current_node = &_root[scalar_keys[0]];
62 for (const auto key_index : index_range(scalar_keys))
63 {
64 if (key_index == scalar_keys.size() - 1)
65 {
66 scalar = getReal(*current_node);
67 break;
68 }
69 current_node = &(*current_node)[scalar_keys[key_index + 1]];
70 }
71}

73void
74JSONFileReader::getVector(const std::string & vector_name, std::vector<Real> & vector_to_fill) const
75{
76 if (!_root.contains(vector_name))
77 mooseError("Attempted to get '",
78 vector_name,
79 "' but the JSON file does not contain this key at the root level");
80 const auto num_items = _root[vector_name].size();
81 vector_to_fill.resize(num_items);
82 for (const auto index : make_range(num_items))
83 vector_to_fill[index] = getReal(_root[vector_name][index]);
84}

86void
87JSONFileReader::getVector(const std::vector<std::string> & vector_keys,
88 std::vector<Real> & vector_to_fill) const
89{
90 if (!vector_keys.size())
91 mooseError("There should be at least one key to retrieve a value from the JSON");
92
93 // traverse the JSON tree
94 auto * current_node = &_root[vector_keys[0]];
95 for (const auto key_index : index_range(vector_keys))
96 {
97 if (key_index == vector_keys.size() - 1)
98 {
99 if (!current_node->is_array())
100 mooseError("Cannot retrieve a vector from JSON node",
101 *current_node,
102 "obtained with last key",
103 vector_keys[key_index]);
104 const auto num_items = current_node->size();
105 vector_to_fill.resize(num_items);
106 for (const auto index : make_range(num_items))
107 vector_to_fill[index] = getReal((*current_node)[index]);
108 return;
109 }
110 if (current_node->is_array())
111 mooseError("Cannot obtain nested JSON item with key",
112 vector_keys[key_index + 1],
113 "because the current item is an array:",
114 *current_node);
115 current_node = &(*current_node)[vector_keys[key_index + 1]];
116 }
117}
