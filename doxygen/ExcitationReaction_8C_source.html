46ADReal
47ExcitationReaction::computeQpResidual()
48{
49 ADReal electron_flux_mag = (-_muem[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_em[_qp]) -
50 _diffem[_qp] * std::exp(_em[_qp]) * _grad_em[_qp] * _r_units)
51 .norm();
52
53 // Determines if the impacted species is the reactant or a product,
54 // defining townsend coefficient accordingly.
55 // Currently only accepts townsend coefficients for electron-impact excitation
56 // reactions with the neutral background gas, and the reverse reaction
57 // ("deexcitation"). Will be generalized into separate class
58 // ("ElectronImpactReaction") to accept any townsend coefficient for any
59 // generic electron-impact reaction.
60 if (_reactant)
61 {
62 ADReal alpha = _alpha_source[_qp] / _n_gas[_qp] * std::exp(_u[_qp]);
63 ADReal iz_term = alpha * electron_flux_mag;
64 return _test[_i][_qp] * iz_term;
65 }
66 else
67 {
68 ADReal alpha = _alpha_source[_qp];
69 ADReal iz_term = alpha * electron_flux_mag;
70 return -_test[_i][_qp] * iz_term;
71 }
72}
