55ADReal
56SakiyamaSecondaryElectronBC::computeQpResidual()
57{
58 _ion_flux.zero();
59 for (unsigned int i = 0; i < _num_ions; ++i)
60 {
61 if (_normals[_qp] * (*_sgnip[i])[_qp] * -_grad_potential[_qp] > 0.0)
62 {
63 _a = 1.0;
64 }
65 else
66 {
67 _a = 0.0;
68 }
69
70 _ion_flux += _a * (*_sgnip[i])[_qp] * (*_muip[i])[_qp] * -_grad_potential[_qp] * _r_units *
71 std::exp((*_ip[i])[_qp]);
72 }
73
74 return -_test[_i][_qp] * _r_units * _a * _user_se_coeff * _ion_flux * _normals[_qp];
75}
