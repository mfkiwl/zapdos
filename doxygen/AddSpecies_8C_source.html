47void
48AddSpecies::act()
49{
50 if (_current_task == "add_variable")
51 {
52 std::string family;
53 Real scale_factor;
54 for (auto i = beginIndex(_vars); i < _vars.size(); ++i)
55 {
56 if (isParamValid("scale_factors"))
57 scale_factor = _scale_factor[i];
58 else
59 scale_factor = 1.0;
60
61 _moose_object_pars.applySpecificParameters(_pars, {"order", "family"});
62 _moose_object_pars.set<std::vector<Real>>("scaling") = {scale_factor};
63 const auto type = variableType(feType(_moose_object_pars));
64
65 _problem->addVariable(type, _vars[i], _moose_object_pars);
66
67 createInitialConditions(_vars[i], _vals[i]);
68 }
69 }
70
71 // Add time derivatives to the system
72 if (_add_time_derivatives)
73 {
74 if (_current_task == "add_kernel" && !_use_scalar)
75 {
76 std::string time_kernel = "TimeDerivative";
77 if (_use_log)
78 time_kernel = time_kernel + "Log";
79 for (auto i = beginIndex(_vars); i < _vars.size(); ++i)
80 {
81 InputParameters params = _factory.getValidParams(time_kernel);
82 params.set<NonlinearVariableName>("variable") = _vars[i];
83 _problem->addKernel(time_kernel, "dvar" + std::to_string(i) + "_dt", params);
84 }
85 }
86 else if (_current_task == "add_scalar_kernel" && _use_scalar)
87 {
88 std::string time_kernel = "ODETimeDerivative";
89 if (_use_log)
90 time_kernel = time_kernel + "Log";
91 for (auto i = beginIndex(_vars); i < _vars.size(); ++i)
92 {
93 InputParameters params = _factory.getValidParams(time_kernel);
94 params.set<NonlinearVariableName>("variable") = _vars[i];
95 _problem->addScalarKernel(time_kernel, "dvar" + std::to_string(i) + "_dt", params);
96 }
97 }
98 }
99}

101void
102AddSpecies::createInitialConditions(const std::string & var_name, const Real & value)
103{
104 // Variable name
105 // std::string var_name = name();
106
107 // Create the object name
108 std::string long_name("");
109 long_name += var_name;
110 long_name += "_moose";
111
112 // Set the parameters for the action
113 InputParameters action_params = _action_factory.getValidParams("AddOutputAction");
114 action_params.set<ActionWarehouse *>("awh") = &_awh;
115
116 if (_use_scalar)
117 action_params.set<std::string>("type") = "ScalarConstantIC";
118 else
119 action_params.set<std::string>("type") = "ConstantIC";
120
121 // Create the action
122 std::shared_ptr<MooseObjectAction> action = std::static_pointer_cast<MooseObjectAction>(
123 _action_factory.create("AddInitialConditionAction", long_name, action_params));
124
125 // Set the required parameters for the object to be created
126 action->getObjectParams().set<VariableName>("variable") = var_name;
127 // action->getObjectParams().set<Real>("value") = getParam<Real>("initial_condition");
128 action->getObjectParams().set<Real>("value") = value;
129
130 // Store the action in the ActionWarehouse
131 _awh.addActionBlock(action);
132}
