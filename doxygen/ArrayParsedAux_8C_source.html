44ArrayParsedAux::ArrayParsedAux(const InputParameters & parameters)
45 : ArrayAuxKernel(parameters),
46 FunctionParserUtils(parameters),
47 _function(getParam<std::string>("expression")),
48 _n_vars(coupledComponents("coupled_variables")),
49 _n_array_vars(coupledComponents("coupled_array_variables")),
50 _vars(coupledValues("coupled_variables")),
51 _array_vars(coupledArrayValues("coupled_array_variables")),
52 _use_xyzt(getParam<bool>("use_xyzt"))
53{
54 // build variables argument
55 std::string variables;
56
57 // coupled field variables
58 for (const auto & i : make_range(_n_vars))
59 variables += (i == 0 ? "" : ",") + getFieldVar("coupled_variables", i)->name();
60 for (const auto & i : make_range(_n_array_vars))
61 {
62 const auto & var = *getArrayVar("coupled_array_variables", i);
63 if (var.count() != _var.count())
64 paramError("coupled_array_variables",
65 "The number of components in '",
66 _var.name(),
67 "' (",
68 _var.count(),
69 ") does not match the number of components in '",
70 var.name(),
71 "' (",
72 var.count(),
73 ").");
74 variables += (i == 0 && _n_vars == 0 ? "" : ",") + var.name();
75 }
76
77 // "system" variables
78 const std::vector<std::string> xyzt = {"x", "y", "z", "t"};
79 if (_use_xyzt)
80 for (auto & v : xyzt)
81 variables += (variables.empty() ? "" : ",") + v;
82
83 // base function object
84 _func_F = std::make_shared<SymFunction>();
85
86 // set FParser internal feature flags
87 setParserFeatureFlags(_func_F);
88
89 // add the constant expressions
90 addFParserConstants(_func_F,
91 getParam<std::vector<std::string>>("constant_names"),
92 getParam<std::vector<std::string>>("constant_expressions"));
93
94 // parse function
95 if (_func_F->Parse(_function, variables) >= 0)
96 mooseError("Invalid function\n",
97 _function,
98 "\nin ArrayParsedAux ",
99 name(),
100 ".\n",
101 _func_F->ErrorMsg());
102
103 // optimize
104 if (!_disable_fpoptimizer)
105 _func_F->Optimize();
106
107 // just-in-time compile
108 if (_enable_jit)
109 {
110 // let rank 0 do the JIT compilation first
111 if (_communicator.rank() != 0)
112 _communicator.barrier();
113
114 _func_F->JITCompile();
115
116 // wait for ranks > 0 to catch up
117 if (_communicator.rank() == 0)
118 _communicator.barrier();
119 }
120
121 // reserve storage for parameter passing buffer
122 _func_params.resize(_n_vars + _n_array_vars + (_use_xyzt ? 4 : 0));
123}

125RealEigenVector
126ArrayParsedAux::computeValue()
127{
128 // Returned value
129 RealEigenVector val(_var.count());
130
131 // Gather regular variable values
132 for (const auto & i : make_range(_n_vars))
133 _func_params[i] = (*_vars[i])[_qp];
134
135 // Gather xyzt values
136 if (_use_xyzt)
137 {
138 for (const auto j : make_range(Moose::dim))
139 _func_params[_n_vars + _n_array_vars + j] =
140 isNodal() ? (*_current_node)(j) : _q_point[_qp](j);
141 _func_params[_n_vars + _n_array_vars + 3] = _t;
142 }
143
144 // Loop through each component
145 for (const auto & c : make_range(_var.count()))
146 {
147 // Gather array variables
148 for (const auto & i : make_range(_n_array_vars))
149 _func_params[_n_vars + i] = (*_array_vars[i])[_qp](c);
150
151 // Evaluate the parsed expression
152 val(c) = evaluate(_func_F);
153 }
154
155 return val;
156}
