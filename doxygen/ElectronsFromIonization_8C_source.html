36ElectronsFromIonization::ElectronsFromIonization(const InputParameters & parameters)
37 : ADKernel(parameters),
38
39 _r_units(1. / getParam<Real>("position_units")),
40
41 _diffem(getParam<bool>("use_material_props") ? getADMaterialProperty<Real>("diffem")
42 : _user_diffem),
43 _muem(getParam<bool>("use_material_props") ? getADMaterialProperty<Real>("muem") : _user_muem),
44 _alpha_iz(getParam<bool>("use_material_props") ? getADMaterialProperty<Real>("alpha_iz")
45 : _user_alpha_iz),
46 _mean_en(adCoupledValue("mean_en")),
47 _grad_potential(isCoupled("potential") ? adCoupledGradient("potential") : _minus_e_field),
48 _em(adCoupledValue("em")),
49 _grad_em(adCoupledGradient("em"))
50{
51 if (!(isCoupled("potential") || parameters.isParamSetByUser("EField")))
52 mooseError("You must either couple in a potential variable or set an EField.");
53
54 if (!(isCoupled("potential")))
55 _minus_e_field.resize(_fe_problem.getMaxQps(), RealGradient(-getParam<Real>("EField")));
56 if (!(getParam<bool>("use_material_props")))
57 {
58 auto max_qps = _fe_problem.getMaxQps();
59 _user_diffem.resize(max_qps);
60 _user_muem.resize(max_qps);
61 _user_alpha_iz.resize(max_qps);
62 for (decltype(max_qps) qp = 0; qp < max_qps; ++qp)
63 {
64 _user_diffem[qp] = getParam<Real>("diffem");
65 _user_muem[qp] = getParam<Real>("muem");
66 _user_alpha_iz[qp] = getParam<Real>("alpha_iz");
67 }
68 }
69}

73ADReal
74ElectronsFromIonization::computeQpResidual()
75{
76 ADReal electron_flux_mag = (-_muem[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_em[_qp]) -
77 _diffem[_qp] * std::exp(_em[_qp]) * _grad_em[_qp] * _r_units)
78 .norm();
79 ADReal iz_term = _alpha_iz[_qp] * electron_flux_mag;
80
81 return -_test[_i][_qp] * iz_term;
82}
