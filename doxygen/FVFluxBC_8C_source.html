44void
45FVFluxBC::computeResidual(const FaceInfo & fi)
46{
47 _face_info = &fi;
48 _normal = fi.normal();
49 _face_type = fi.faceType(std::make_pair(_var.number(), _var.sys().number()));
50
51 // For FV flux kernels, the normal is always oriented outward from the lower-id
52 // element's perspective. But for BCs, there is only a residual
53 // contribution to one element (one side of the face). Because of this, we
54 // make an exception and orient the normal to point outward from whichever
55 // side of the face the BC's variable is defined on; we flip it if this
56 // variable is defined on the neighbor side of the face (instead of elem) since
57 // the FaceInfo normal polarity is always oriented with respect to the lower-id element.
58 if (_face_type == FaceInfo::VarFaceNeighbors::NEIGHBOR)
59 _normal = -_normal;
60
61 if (_face_type == FaceInfo::VarFaceNeighbors::BOTH)
62 mooseError("A FVFluxBC is being triggered on an internal face with centroid: ",
63 fi.faceCentroid());
64 else if (_face_type == FaceInfo::VarFaceNeighbors::NEITHER)
65 mooseError("A FVFluxBC is being triggered on a face which does not connect to a block ",
66 "with the relevant finite volume variable. Its centroid: ",
67 fi.faceCentroid());
68
69 auto r = MetaPhysicL::raw_value(fi.faceArea() * fi.faceCoord() * computeQpResidual());
70
71 // This could be an "internal" boundary - one created by variable block
72 // restriction where the var is only defined on one side of the face. We
73 // need to make sure that we add the residual contribution to the correct
74 // side - the one where the variable is defined.
75 if (_face_type == FaceInfo::VarFaceNeighbors::ELEM)
76 prepareVectorTag(_assembly, _var.number());
77 else if (_face_type == FaceInfo::VarFaceNeighbors::NEIGHBOR)
78 prepareVectorTagNeighbor(_assembly, _var.number());
79
80 _local_re(0) = r;
81 accumulateTaggedLocalResidual();
82}

90void
91FVFluxBC::computeJacobian(const FaceInfo & fi)
92{
93 _face_info = &fi;
94 _normal = fi.normal();
95 _face_type = fi.faceType(std::make_pair(_var.number(), _var.sys().number()));
96
97 // For FV flux kernels, the normal is always oriented outward from the lower-id
98 // element's perspective. But for BCs, there is only a Jacobian
99 // contribution to one element (one side of the face). Because of this, we
100 // make an exception and orient the normal to point outward from whichever
101 // side of the face the BC's variable is defined on; we flip it if this
102 // variable is defined on the neighbor side of the face (instead of elem) since
103 // the FaceInfo normal polarity is always oriented with respect to the lower-id element.
104 if (_face_type == FaceInfo::VarFaceNeighbors::NEIGHBOR)
105 _normal = -_normal;
106
107 ADReal r = fi.faceArea() * fi.faceCoord() * computeQpResidual();
108
109 const auto & dof_indices = (_face_type == FaceInfo::VarFaceNeighbors::ELEM)
110 ? _var.dofIndices()
111 : _var.dofIndicesNeighbor();
112
113 mooseAssert(dof_indices.size() == 1, "We're currently built to use CONSTANT MONOMIALS");
114
115 addResidualsAndJacobian(_assembly, std::array<ADReal, 1>{{r}}, dof_indices, _var.scalingFactor());
116}

118const ADReal &
119FVFluxBC::uOnUSub() const
120{
121 mooseAssert(_face_info, "The face info has not been set");
122 const auto ft = _face_info->faceType(std::make_pair(_var.number(), _var.sys().number()));
123 mooseAssert(
124 ft == FaceInfo::VarFaceNeighbors::ELEM || ft == FaceInfo::VarFaceNeighbors::NEIGHBOR,
125 "The variable " << _var.name()
126 << " should be defined on exactly one adjacent subdomain for FVFluxBC "
127 << this->name());
128 mooseAssert(_qp == 0,
129 "At the time of writing, we only support one quadrature point, which should "
130 "correspond to the location of the cell centroid. If that changes, we should "
131 "probably change the body of FVFluxBC::uOnUSub");
132
133 if (ft == FaceInfo::VarFaceNeighbors::ELEM)
134 return _u[_qp];
135 else
136 return _u_neighbor[_qp];
137}
