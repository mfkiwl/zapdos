41void
42SetupResidualDebugAction::act()
43{
44 if (_problem.get() == NULL)
45 return;
46
47 if (!_show_var_residual.empty())
48 _problem->getNonlinearSystemBase(/*nl_sys=*/0).debuggingResiduals(true);
49
50 // debug variable residuals
51 for (const auto & var_name : _show_var_residual)
52 {
53 // add aux-variable
54 auto order = AddVariableAction::getNonlinearVariableOrders();
55 auto family = AddVariableAction::getNonlinearVariableFamilies();
56
57 MooseVariableFEBase & var = _problem->getVariable(
58 0, var_name, Moose::VarKindType::VAR_NONLINEAR, Moose::VarFieldType::VAR_FIELD_STANDARD);
59 auto fe_type = var.feType();
60 order = Utility::enum_to_string<Order>(fe_type.order);
61 family = Utility::enum_to_string(fe_type.family);
62
63 InputParameters params = _factory.getValidParams("DebugResidualAux");
64 const std::set<SubdomainID> & subdomains = var.activeSubdomains();
65
66 std::stringstream aux_var_ss;
67 aux_var_ss << "residual_" << var.name();
68 std::string aux_var_name = aux_var_ss.str();
69
70 auto var_params = _factory.getValidParams("MooseVariable");
71 var_params.set<MooseEnum>("family") = family;
72 var_params.set<MooseEnum>("order") = order;
73
74 if (subdomains.empty())
75 _problem->addAuxVariable("MooseVariable", aux_var_name, var_params);
76 else
77 {
78 std::vector<SubdomainName> block_names;
79 block_names.reserve(subdomains.size());
80 for (const SubdomainID & id : subdomains)
81 block_names.push_back(Moose::stringify(id));
82 params.set<std::vector<SubdomainName>>("block") = block_names;
83 var_params.set<std::vector<SubdomainName>>("block") = block_names;
84
85 _problem->addAuxVariable("MooseVariable", aux_var_name, var_params);
86 }
87
88 // add aux-kernel
89 std::stringstream kern_ss;
90 kern_ss << "residual_" << var.name() << "_kernel";
91 std::string kern_name = kern_ss.str();
92
93 params.set<AuxVariableName>("variable") = aux_var_name;
94 params.set<NonlinearVariableName>("debug_variable") = var.name();
95 params.set<ExecFlagEnum>("execute_on") = {EXEC_LINEAR, EXEC_TIMESTEP_END};
96 _problem->addAuxKernel("DebugResidualAux", kern_name, params);
97 }
98}
