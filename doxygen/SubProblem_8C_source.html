81TagID
82SubProblem::addVectorTag(const TagName & tag_name,
83 const Moose::VectorTagType type /* = Moose::VECTOR_TAG_RESIDUAL */)
84{
85 if (type == Moose::VECTOR_TAG_ANY)
86 mooseError("Vector tag type cannot be VECTOR_TAG_ANY");
87
88 const auto tag_name_upper = MooseUtils::toUpper(tag_name);
89
90 // First, see if the tag exists already
91 for (const auto & vector_tag : _vector_tags)
92 {
93 mooseAssert(_vector_tags[vector_tag._id] == vector_tag, "Vector tags index mismatch");
94 if (vector_tag._name == tag_name_upper)
95 {
96 if (vector_tag._type != type)
97 mooseError("While attempting to add vector tag with name '",
98 tag_name_upper,
99 "' and type ",
100 type,
101 ",\na tag with the same name but type ",
102 vector_tag._type,
103 " was found.\n\nA tag can only exist with one type.");
104
105 return vector_tag._id;
106 }
107 }
108
109 // Doesn't exist - create it
110 const TagID new_tag_id = _vector_tags.size();
111 const TagTypeID new_tag_type_id = _typed_vector_tags[type].size();
112 // Primary storage for all tags where the index in the vector == the tag ID
113 _vector_tags.emplace_back(new_tag_id, new_tag_type_id, tag_name_upper, type);
114 // Secondary storage for each type so that we can have quick access to all tags of a type
115 _typed_vector_tags[type].emplace_back(new_tag_id, new_tag_type_id, tag_name_upper, type);
116 // Name map storage for quick name access
117 _vector_tags_name_map.emplace(tag_name_upper, new_tag_id);
118
119 // Make sure that _vector_tags, _typed_vector_tags, and _vector_tags_name_map are sane
120 verifyVectorTags();
121
122 return new_tag_id;
123}

218bool
219SubProblem::verifyVectorTags() const
220{
221 for (TagID tag_id = 0; tag_id < _vector_tags.size(); ++tag_id)
222 {
223 const auto & vector_tag = _vector_tags[tag_id];
224
225 if (vector_tag._id != tag_id)
226 mooseError("Vector tag ", vector_tag._id, " id mismatch in _vector_tags");
227 if (vector_tag._type == Moose::VECTOR_TAG_ANY)
228 mooseError("Vector tag '", vector_tag._name, "' has type VECTOR_TAG_ANY");
229
230 const auto search = _vector_tags_name_map.find(vector_tag._name);
231 if (search == _vector_tags_name_map.end())
232 mooseError("Vector tag ", vector_tag._id, " is not in _vector_tags_name_map");
233 else if (search->second != tag_id)
234 mooseError("Vector tag ", vector_tag._id, " has incorrect id in _vector_tags_name_map");
235
236 unsigned int found_in_type = 0;
237 for (TagTypeID tag_type_id = 0; tag_type_id < _typed_vector_tags[vector_tag._type].size();
238 ++tag_type_id)
239 {
240 const auto & vector_tag_type = _typed_vector_tags[vector_tag._type][tag_type_id];
241 if (vector_tag_type == vector_tag)
242 {
243 ++found_in_type;
244 if (vector_tag_type._type_id != tag_type_id)
245 mooseError("Type ID for Vector tag ", tag_id, " is incorrect");
246 }
247 }
248
249 if (found_in_type == 0)
250 mooseError("Vector tag ", tag_id, " not found in _typed_vector_tags");
251 if (found_in_type > 1)
252 mooseError("Vector tag ", tag_id, " found multiple times in _typed_vector_tags");
253 }
254
255 unsigned int num_typed_vector_tags = 0;
256 for (const auto & typed_vector_tags : _typed_vector_tags)
257 num_typed_vector_tags += typed_vector_tags.size();
258 if (num_typed_vector_tags != _vector_tags.size())
259 mooseError("Size mismatch between _vector_tags and _typed_vector_tags");
260 if (_vector_tags_name_map.size() != _vector_tags.size())
261 mooseError("Size mismatch between _vector_tags and _vector_tags_name_map");
262
263 return true;
264}

589void
590SubProblem::checkBlockMatProps()
591{
592 // Variable for storing all available blocks/boundaries from the mesh
593 std::set<SubdomainID> all_ids(mesh().meshSubdomains());
594
595 std::stringstream errors;
596
597 // Loop through the properties to check
598 for (const auto & check_it : _map_block_material_props_check)
599 {
600 // The current id for the property being checked (BoundaryID || BlockID)
601 SubdomainID check_id = check_it.first;
602
603 std::set<SubdomainID> check_ids = {check_id};
604
605 // Loop through all the block/boundary ids
606 for (const auto & id : check_ids)
607 {
608 // Loop through all the stored properties
609 for (const auto & prop_it : check_it.second)
610 {
611 // Produce an error if the material property is not defined on the current block/boundary
612 // and any block/boundary
613 // and not is not a zero material property.
614 if (_map_block_material_props[id].count(prop_it.second) == 0 &&
615 _zero_block_material_props[id].count(prop_it.second) == 0)
616 {
617 std::string check_name = restrictionSubdomainCheckName(id);
618 if (check_name.empty())
619 check_name = std::to_string(id);
620 errors << "Material property '" << prop_it.second << "', requested by '" << prop_it.first
621 << "' is not defined on block " << check_name << "\n";
622 }
623 }
624 }
625 }
626
627 if (!errors.str().empty())
628 mooseError(errors.str());
629}

631void
632SubProblem::checkBoundaryMatProps()
633{
634 // Variable for storing the value for ANY_BOUNDARY_ID
635 BoundaryID any_id = Moose::ANY_BOUNDARY_ID;
636
637 // Variable for storing all available blocks/boundaries from the mesh
638 std::set<BoundaryID> all_ids(mesh().getBoundaryIDs());
639
640 std::stringstream errors;
641
642 // Loop through the properties to check
643 for (const auto & check_it : _map_boundary_material_props_check)
644 {
645 // The current id for the property being checked (BoundaryID || BlockID)
646 BoundaryID check_id = check_it.first;
647
648 // In the case when the material being checked has an ID is set to ANY, then loop through all
649 // the possible ids and verify that the material property is defined.
650 std::set<BoundaryID> check_ids{check_id};
651 if (check_id == any_id)
652 check_ids = all_ids;
653
654 // Loop through all the block/boundary ids
655 for (const auto & id : check_ids)
656 {
657 // Loop through all the stored properties
658 for (const auto & prop_it : check_it.second)
659 {
660 // Produce an error if the material property is not defined on the current block/boundary
661 // and any block/boundary
662 // and not is not a zero material property.
663 if (_map_boundary_material_props[id].count(prop_it.second) == 0 &&
664 _map_boundary_material_props[any_id].count(prop_it.second) == 0 &&
665 _zero_boundary_material_props[id].count(prop_it.second) == 0 &&
666 _zero_boundary_material_props[any_id].count(prop_it.second) == 0)
667 {
668 std::string check_name = restrictionBoundaryCheckName(id);
669 if (check_name.empty())
670 check_name = std::to_string(id);
671 errors << "Material property '" << prop_it.second << "', requested by '" << prop_it.first
672 << "' is not defined on boundary " << check_name << "\n";
673 }
674 }
675 }
676 }
677
678 if (!errors.str().empty())
679 mooseError(errors.str());
680}

766template <typename T>
767MooseVariableFEBase &
768SubProblem::getVariableHelper(const THREAD_ID tid,
769 const std::string & var_name,
770 Moose::VarKindType expected_var_type,
771 Moose::VarFieldType expected_var_field_type,
772 const std::vector<T> & nls,
773 const SystemBase & aux) const
774{
775 // Eventual return value
776 MooseVariableFEBase * var = nullptr;
777
778 const auto [var_in_nl, nl_sys_num] = determineNonlinearSystem(var_name);
779
780 // First check that the variable is found on the expected system.
781 if (expected_var_type == Moose::VarKindType::VAR_ANY)
782 {
783 if (var_in_nl)
784 var = &(nls[nl_sys_num]->getVariable(tid, var_name));
785 else if (aux.hasVariable(var_name))
786 var = &(aux.getVariable(tid, var_name));
787 else
788 mooseError("Unknown variable " + var_name);
789 }
790 else if (expected_var_type == Moose::VarKindType::VAR_NONLINEAR && var_in_nl &&
791 nls[nl_sys_num]->hasVariable(var_name))
792 var = &(nls[nl_sys_num]->getVariable(tid, var_name));
793 else if (expected_var_type == Moose::VarKindType::VAR_AUXILIARY && aux.hasVariable(var_name))
794 var = &(aux.getVariable(tid, var_name));
795 else
796 {
797 std::string expected_var_type_string =
798 (expected_var_type == Moose::VarKindType::VAR_NONLINEAR ? "nonlinear" : "auxiliary");
799 mooseError("No ",
800 expected_var_type_string,
801 " variable named ",
802 var_name,
803 " found. "
804 "Did you specify an auxiliary variable when you meant to specify a nonlinear "
805 "variable (or vice-versa)?");
806 }
807
808 // Now make sure the var found has the expected field type.
809 if ((expected_var_field_type == Moose::VarFieldType::VAR_FIELD_ANY) ||
810 (expected_var_field_type == var->fieldType()))
811 return *var;
812 else
813 {
814 std::string expected_var_field_type_string =
815 MooseUtils::toLower(Moose::stringify(expected_var_field_type));
816 std::string var_field_type_string = MooseUtils::toLower(Moose::stringify(var->fieldType()));
817
818 mooseError("No ",
819 expected_var_field_type_string,
820 " variable named ",
821 var_name,
822 " found. "
823 "Did you specify a ",
824 var_field_type_string,
825 " variable when you meant to specify a ",
826 expected_var_field_type_string,
827 " variable?");
828 }

831void
832SubProblem::reinitElemFaceRef(const Elem * elem,
833 unsigned int side,
834 BoundaryID bnd_id,
835 Real tolerance,
836 const std::vector<Point> * const pts,
837 const std::vector<Real> * const weights,
838 const THREAD_ID tid)
839{
840 for (const auto nl_sys_num : make_range(numNonlinearSystems()))
841 {
842 // - Set our _current_elem for proper dof index getting in the moose variables
843 // - Reinitialize all of our FE objects so we have current phi, dphi, etc. data
844 // Note that our number of shape functions will reflect the number of shapes associated with the
845 // interior element while the number of quadrature points will be determined by the passed pts
846 // parameter (which presumably will have a number of pts reflective of a facial quadrature rule)
847 assembly(tid, nl_sys_num).reinitElemFaceRef(elem, side, tolerance, pts, weights);
848
849 auto & nl = systemBaseNonlinear(nl_sys_num);
850
851 // Actually get the dof indices in the moose variables
852 nl.prepare(tid);
853
854 // Let's finally compute our variable values!
855 nl.reinitElemFace(elem, side, bnd_id, tid);
856 }
857
858 // do same for aux as for nl
859 systemBaseAuxiliary().prepare(tid);
860 systemBaseAuxiliary().reinitElemFace(elem, side, bnd_id, tid);
861
862 // With the dof indices set in the moose variables, now let's properly size
863 // our local residuals/Jacobians
864 auto & current_assembly = assembly(tid, currentNlSysNum());
865 if (currentlyComputingJacobian() || currentlyComputingResidualAndJacobian())
866 current_assembly.prepareJacobianBlock();
867 if (!currentlyComputingJacobian())
868 current_assembly.prepareResidual();
869}

954{
955 EquationSystems & eq = es();
956 const auto n_sys = eq.n_systems();
957
958 auto pr = _root_alg_gf_to_sys_clones.emplace(
959 &algebraic_gf, std::vector<std::shared_ptr<GhostingFunctor>>(n_sys - 1));
960 mooseAssert(pr.second, "We are adding a duplicate algebraic ghosting functor");
961 auto & clones_vec = pr.first->second;
962
963 for (MooseIndex(n_sys) i = 1; i < n_sys; ++i)

966 std::shared_ptr<GhostingFunctor> clone_alg_gf = algebraic_gf.clone();

968 ->init(*algebraic_gf.get_mesh(), &dof_map);
969 dof_map.add_algebraic_ghosting_functor(clone_alg_gf, to_mesh);
970 clones_vec[i - 1] = clone_alg_gf;
971 }

1018void
1019SubProblem::removeAlgebraicGhostingFunctor(GhostingFunctor & algebraic_gf)
1020{
1021 EquationSystems & eq = es();
1022 const auto n_sys = eq.n_systems();
1023
1024#ifndef NDEBUG
1025 const DofMap & nl_dof_map = eq.get_system(0).get_dof_map();
1026 const bool found_in_root_sys =
1027 std::find(nl_dof_map.algebraic_ghosting_functors_begin(),
1028 nl_dof_map.algebraic_ghosting_functors_end(),
1029 &algebraic_gf) != nl_dof_map.algebraic_ghosting_functors_end();
1030 const bool found_in_our_map =
1031 _root_alg_gf_to_sys_clones.find(&algebraic_gf) != _root_alg_gf_to_sys_clones.end();
1032 mooseAssert(found_in_root_sys == found_in_our_map,
1033 "If the ghosting functor exists in the root DofMap, then we need to have a key for "
1034 "it in our gf to clones map");
1035#endif
1036
1037 eq.get_system(0).get_dof_map().remove_algebraic_ghosting_functor(algebraic_gf);
1038
1039 auto it = _root_alg_gf_to_sys_clones.find(&algebraic_gf);
1040 if (it == _root_alg_gf_to_sys_clones.end())
1041 return;
1042
1043 auto & clones_vec = it->second;
1044 mooseAssert((n_sys - 1) == clones_vec.size(),
1045 "The size of the gf clones vector doesn't match the number of systems minus one");
1046 if (clones_vec.empty())
1047 {
1048 mooseAssert(n_sys == 1, "The clones vector should only be empty if there is only one system");
1049 return;
1050 }
1051
1052 for (const auto i : make_range(n_sys))
1053 eq.get_system(i + 1).get_dof_map().remove_algebraic_ghosting_functor(*clones_vec[i]);
1054
1055 _root_alg_gf_to_sys_clones.erase(it->first);
1056}
