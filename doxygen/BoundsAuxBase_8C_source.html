28BoundsAuxBase::BoundsAuxBase(const InputParameters & parameters)
29 : AuxKernel(parameters),
30 _type((BoundType)(int)parameters.get<MooseEnum>("bound_type")),
31 _bounded_vector(_type == 0 ? _nl_sys.getVector("upper_bound")
32 : _nl_sys.getVector("lower_bound")),
33 _bounded_var(_nl_sys.getVariable(_tid, getParam<NonlinearVariableName>("bounded_variable"))),
34 _bounded_var_name(parameters.get<NonlinearVariableName>("bounded_variable")),
35 _fe_var(_bounded_var.isFV() ? nullptr
36 : &_subproblem.getStandardVariable(_tid, _bounded_var_name))
37{
38 if (!Moose::PetscSupport::isSNESVI(*dynamic_cast<FEProblemBase *>(&_subproblem)))
39 mooseDoOnce(mooseWarning(
40 "A variational inequalities solver must be used in conjunction with BoundsAux"));
41
42 // Check that the bounded variable is of a supported type
43 if (!_bounded_var.isNodal() && (_bounded_var.feType().order != CONSTANT))
44 paramError("bounded_variable", "Bounded variable must be nodal or of a CONSTANT order!");
45
46 const auto & dummy =
47 _aux_sys.getActualFieldVariable<Real>(_tid, parameters.get<AuxVariableName>("variable"));
48
49 // Check that the dummy variable matches the bounded variable
50 if (dummy.feType() != _bounded_var.feType())
51 paramError("variable",
52 "Dummy bounds aux variable and bounded variable must use the same finite element "
53 "order and family");
54}

57BoundsAuxBase::computeValue()

70dof_id_type
71BoundsAuxBase::getDoFIndex() const
72{
73 if (isNodal())
74 {
75 if (_current_node->n_dofs(_nl_sys.number(), _bounded_var.number()) > 0)
76 {
77 mooseAssert(_current_node->n_dofs(_nl_sys.number(), _bounded_var.number()) == 1,
78 "Bounds are only set on one DOF value per node currently");
79 // The zero is for the component, this will only work for Lagrange variables
80 return _current_node->dof_number(_nl_sys.number(), _bounded_var.number(), 0);
81 }
82 }
83 else if (_current_elem->n_dofs(_nl_sys.number(), _bounded_var.number()) > 0)
84 {
85 mooseAssert(_current_elem->n_dofs(_nl_sys.number(), _bounded_var.number()) == 1,
86 "Bounds are only set on one DOF value per element currently");
87 // The zero is for the component, this will only work for CONSTANT variables
88 return _current_elem->dof_number(_nl_sys.number(), _bounded_var.number(), 0);
89 }
90 // No local dof for the bounded variable. This can happen for example if:
91 // - block restriction of dummy is different from bounded variable
92 // - we have a first order variable on a second order mesh
93 return std::numeric_limits<dof_id_type>::max();
94}
