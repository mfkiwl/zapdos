40ADReal
41DGEFieldAdvection::computeQpResidual(Moose::DGResidualType type)
42{
43 ADReal r = 0;
44 ADRealVectorValue _velocity = _mu[_qp] * _sgn[_qp] * -_grad_potential[_qp];
45 ADRealVectorValue _velocity_neighbor =
46 _mu_neighbor[_qp] * _sgn_neighbor[_qp] * -_grad_potential_neighbor[_qp];
47 ADRealVectorValue _velocity_average = 0.5 * (_velocity + _velocity_neighbor);
48
49 switch (type)
50 {
51 case Moose::Element:
52 if ((_velocity_average * _normals[_qp]) >= 0.0)
53 r += (_velocity_average * _normals[_qp]) * std::exp(_u[_qp]) * _test[_i][_qp];
54 else
55 r += (_velocity_average * _normals[_qp]) * std::exp(_u_neighbor[_qp]) * _test[_i][_qp];
56 break;
57
58 case Moose::Neighbor:
59 if ((_velocity_average * _normals[_qp]) >= 0.0)
60 r += -(_velocity_average * _normals[_qp]) * std::exp(_u[_qp]) * _test_neighbor[_i][_qp];
61 else
62 r += -(_velocity_average * _normals[_qp]) * std::exp(_u_neighbor[_qp]) *
63 _test_neighbor[_i][_qp];
64 break;
65 }
66 return r;
67}
