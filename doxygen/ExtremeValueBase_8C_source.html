60ExtremeValueBase<T>::computeExtremeValue()

63 switch (_type)
64 {
65 case ExtremeType::MAX:

73 break;

84template <class T>
85void
86ExtremeValueBase<T>::finalize()
87{
88 switch (_type)
89 {
90 case ExtremeType::MAX:
91 if (_use_proxy)
92 this->gatherProxyValueMax(_proxy_value.first, _proxy_value.second);
93 else
94 {
95 this->gatherMax(_proxy_value.first);
96 _proxy_value.second = _proxy_value.first;
97 }
98 break;
99 case ExtremeType::MIN:
100 if (_use_proxy)
101 this->gatherProxyValueMin(_proxy_value.first, _proxy_value.second);
102 else
103 {
104 this->gatherMin(_proxy_value.first);
105 _proxy_value.second = _proxy_value.first;
106 }
107 break;
108 }
109}
