48SideAdvectiveFluxIntegralTempl<is_ad>::SideAdvectiveFluxIntegralTempl(
49 const InputParameters & parameters)
50 : SideIntegralPostprocessor(parameters),
51 _use_normal(getParam<MooseEnum>("component") == "normal"),
52 _component(getParam<MooseEnum>("component")),
53 _advected_variable_supplied(isParamValid("advected_variable")),
54 _advected_variable(_advected_variable_supplied ? coupledValue("advected_variable") : _zero),
55 _advected_mat_prop_supplied(parameters.isParamSetByUser("advected_mat_prop")),

62{
63 // Check that at most one advected quantity has been provided
64 if (_advected_mat_prop_supplied)

68 "SideAdvectiveFluxIntegralPostprocessor should be provided either an advected quantity "
69 "or an advected material property");
70 }
71
72 // Check whether a finite element or finite volume variable is provide
73 _qp_integration = !_adv_quant;

76template <bool is_ad>
77Real
78SideAdvectiveFluxIntegralTempl<is_ad>::computeFaceInfoIntegral(const FaceInfo * const fi)
79{
80 mooseAssert(fi, "We should have a face info in " + name());
81 mooseAssert(_adv_quant, "We should have an advected quantity in " + name());
82
83 const auto state = determineState();
84
85 // Get face value for velocity
86 const auto vel_x =
87 (_vel_x)(Moose::FaceArg(
88 {fi, Moose::FV::LimiterType::CentralDifference, true, false, nullptr}),
89 state);
90 const auto vel_y =
91 _vel_y
92 ? ((*_vel_y)(Moose::FaceArg(
93 {fi, Moose::FV::LimiterType::CentralDifference, true, false, nullptr}),
94 state))
95 : 0;
96 const auto vel_z =
97 _vel_z
98 ? ((*_vel_z)(Moose::FaceArg(
99 {fi, Moose::FV::LimiterType::CentralDifference, true, false, nullptr}),
100 state))
101 : 0;
102
103 auto fi_normal = _current_elem == fi->elemPtr() ? fi->normal() : Point(-fi->normal());
104 const bool elem_is_upwind = RealVectorValue(vel_x, vel_y, vel_z) * fi_normal >= 0;
105
106 const auto adv_quant_face = (*_adv_quant)(
107 Moose::FaceArg(
108 {fi, Moose::FV::LimiterType::CentralDifference, elem_is_upwind, false, nullptr}),
109 state);
110
111 return fi_normal * adv_quant_face * RealVectorValue(vel_x, vel_y, vel_z);
112}

114template <bool is_ad>
115Real
116SideAdvectiveFluxIntegralTempl<is_ad>::computeQpIntegral()
117{
118 using MetaPhysicL::raw_value;
119
120 const auto state = determineState();
121 const Moose::ElemSideQpArg side_arg = {_current_elem, _current_side, _qp, _qrule, _q_point[_qp]};
122 const auto vel_x = raw_value(_vel_x(side_arg, state));
123 const auto vel_y = _vel_y ? raw_value((*_vel_y)(side_arg, state)) : 0;
124 const auto vel_z = _vel_z ? raw_value((*_vel_z)(side_arg, state)) : 0;
125
126 if (_advected_variable_supplied)
127 {
128
129 auto & variable = getCoupledMooseVars();
130 if (std::any_of(variable.begin(), variable.end(), [](auto & var) { return !var->isNodal(); }))
131 mooseError("Trying to use a non-nodal variable 'advected_variable' for side advection "
132 "integral calculation, which is currently not supported.");
133
134 return (_use_normal
135 ? _advected_variable[_qp] * RealVectorValue(vel_x, vel_y, vel_z) * _normals[_qp]
136 : _advected_variable[_qp] * RealVectorValue(vel_x, vel_y, vel_z)(_component));
137 }
138 else if (_advected_mat_prop_supplied)
139 return (_use_normal ? raw_value(_advected_material_property[_qp]) *
140 RealVectorValue(vel_x, vel_y, vel_z) * _normals[_qp]
141 : raw_value(_advected_material_property[_qp]) *
142 RealVectorValue(vel_x, vel_y, vel_z)(_component));
143 else
144 return (_use_normal ? RealVectorValue(vel_x, vel_y, vel_z) * _normals[_qp]
145 : RealVectorValue(vel_x, vel_y, vel_z)(_component));
146}
