32template <typename T>
33MaterialFunctorConverterTempl<T>::MaterialFunctorConverterTempl(const InputParameters & parameters)
34 : Material(parameters),
35 _num_functors_to_convert(getParam<std::vector<MooseFunctorName>>("functors_in").size())
36{
37 const auto & functors_in = getParam<std::vector<MooseFunctorName>>("functors_in");
38 const auto & reg_props_out = getParam<std::vector<MaterialPropertyName>>("reg_props_out");
39 const auto & ad_props_out = getParam<std::vector<MaterialPropertyName>>("ad_props_out");
40
41 if (reg_props_out.size() && ad_props_out.size())
42 paramError("reg_props_out",
43 "We dont support converting functors to both regular and AD "
44 "material properties in "
45 "a single instance of '",
46 type(),
47 "'. Please create two instances, one for regular and one for AD.");
48
49 if (functors_in.size() != reg_props_out.size() && functors_in.size() != ad_props_out.size())
50 paramError(
51 "functors_in",
52 "The number of output properties must match the number of input functors, which is " +
53 std::to_string(functors_in.size()));
54
55 _functors_in.resize(_num_functors_to_convert);
56 _ad_props_out.resize(ad_props_out.size());
57 _reg_props_out.resize(reg_props_out.size());
58
59 for (const auto i : make_range(_num_functors_to_convert))
60 _functors_in[i] = &getFunctor<Moose::GenericType<T, true>>(functors_in[i]);
61
62 for (const auto i : index_range(ad_props_out))
63 _ad_props_out[i] = &declareADProperty<T>(ad_props_out[i]);
64
65 for (const auto i : index_range(reg_props_out))
66 _reg_props_out[i] = &declareProperty<T>(reg_props_out[i]);
67}

76template <typename T>
77void
78MaterialFunctorConverterTempl<T>::computeQpProperties()
79{
80 // Using Qp 0 can leverage the functor caching
81 // TODO: Find a way to effectively use subdomain-constant-ness
82 unsigned int qp_used = (_constant_option == ConstantTypeEnum::NONE) ? _qp : 0;
83
84 const auto state = Moose::currentState();
85 if (_bnd)
86 {
87 const Moose::ElemSideQpArg side_arg = {
88 _current_elem, _current_side, qp_used, _qrule, _q_point[_qp]};
89 for (const auto i : index_range(_ad_props_out))
90 (*_ad_props_out[i])[_qp] = (*_functors_in[i])(side_arg, state);
91
92 for (const auto i : index_range(_reg_props_out))
93 (*_reg_props_out[i])[_qp] = MetaPhysicL::raw_value((*_functors_in[i])(side_arg, state));
94 }
95 else
96 {
97 const Elem * elem = _neighbor ? _current_elem->neighbor_ptr(_current_side) : _current_elem;
98 mooseAssert(elem, "We should have an element");
99 const Moose::ElemQpArg elem_arg = {elem, qp_used, _qrule, _q_point[_qp]};
100 for (const auto i : index_range(_ad_props_out))
101 (*_ad_props_out[i])[_qp] = (*_functors_in[i])(elem_arg, state);
102
103 for (const auto i : index_range(_reg_props_out))
104 (*_reg_props_out[i])[_qp] = MetaPhysicL::raw_value((*_functors_in[i])(elem_arg, state));
105 }
106}
