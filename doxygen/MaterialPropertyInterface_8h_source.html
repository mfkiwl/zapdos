577class OptionalMaterialPropertyProxy : public OptionalMaterialPropertyProxyBase<M>
578{
579public:
580 OptionalMaterialPropertyProxy(const std::string & name, const unsigned int state)
581 : OptionalMaterialPropertyProxyBase<M>(name, state)
582 {

584 const GenericOptionalMaterialProperty<T, is_ad> & value() const { return _value; }
585 void resolve(M & mpi) override
586 {
587 if (mpi.template hasGenericMaterialProperty<T, is_ad>(this->_name))
588 {
589 if constexpr (is_ad)
590 if (this->_state > 0)
591 mooseError("Non-current (state > 0) material properties are not available as AD");
592
593 _value.set(&mpi.template getGenericMaterialProperty<T, is_ad>(this->_name, this->_state));
594 }

601template <typename T, bool is_ad>
602const GenericMaterialProperty<T, is_ad> *
603MaterialPropertyInterface::defaultGenericMaterialProperty(const std::string & name)
604{
605 if constexpr (std::is_same_v<T, Real> || std::is_same_v<T, RealVectorValue>)
606 {
607 std::istringstream ss(name);
608 Real real_value;
609
610 // check if the string parsed cleanly into a Real number
611 if (ss >> real_value && ss.eof())
612 {
613 using prop_type = GenericMaterialProperty<T, is_ad>;
614
615 const auto nqp = Moose::constMaxQpsPerElem;
616 auto & property =
617 _default_properties.emplace_back(std::make_unique<prop_type>(default_property_id));
618 auto & T_property = static_cast<prop_type &>(*property);
619
620 T_property.resize(nqp);
621 for (const auto qp : make_range(nqp))
622 T_property[qp] = real_value;
623
624 return &T_property;
625 }
626 }
627
628 return nullptr;
629}

631template <typename T>
632std::pair<const MaterialProperty<T> *, std::set<SubdomainID>>
633MaterialPropertyInterface::getBlockMaterialProperty(const MaterialPropertyName & name_in)
634{
635 const auto name = _get_suffix.empty()
636 ? static_cast<const std::string &>(name_in)
637 : MooseUtils::join(std::vector<std::string>({name_in, _get_suffix}), "_");
638
639 if (_mi_block_ids.empty())
640 mooseError("getBlockMaterialProperty must be called by a block restrictable object");
641
642 using pair_type = std::pair<const MaterialProperty<T> *, std::set<SubdomainID>>;
643
644 if (!hasMaterialPropertyByName<T>(name))
645 return pair_type(nullptr, {});
646
647 // Call first so that the ID gets registered
648 const auto & prop = _material_data.getProperty<T, false>(name, 0, _mi_moose_object);
649 auto blocks = getMaterialPropertyBlocks(name);
650 auto prop_blocks_pair = pair_type(&prop, std::move(blocks));
651
652 _material_property_dependencies.insert(_material_data.getPropertyId(name));
653
654 // Update consumed properties in MaterialPropertyDebugOutput
655 addConsumedPropertyName(_mi_moose_object_name, name);
656
657 return prop_blocks_pair;
658}

698template <typename T, bool is_ad>
699const GenericMaterialProperty<T, is_ad> &
700MaterialPropertyInterface::getGenericZeroMaterialProperty()
701{
702 // static zero property storage
703 static GenericMaterialProperty<T, is_ad> zero(zero_property_id);
704
705 // resize to accomodate maximum number of qpoints
706 // (in multiapp scenarios getMaxQps can return different values in each app; we need the max)
707 unsigned int nqp = getMaxQps();
708 if (nqp > zero.size())
709 zero.resize(nqp);
710
711 // set values for all qpoints to zero
712 for (unsigned int qp = 0; qp < nqp; ++qp)
713 MathUtils::mooseSetToZero(zero[qp]);
714
715 return zero;
716}

749template <typename T, bool is_ad>
750const GenericMaterialProperty<T, is_ad> &
751MaterialPropertyInterface::getGenericMaterialProperty(const std::string & name,
752 MaterialData & material_data,
753 const unsigned int state)
754{
755 // Check if the supplied parameter is a valid input parameter key
756 const auto prop_name = getMaterialPropertyName(name);
757
758 // Check if it's just a constant
759 if (const auto * default_property = defaultGenericMaterialProperty<T, is_ad>(prop_name))
760 return *default_property;
761
762 if (state > 0 && !_stateful_allowed)
763 mooseError("Stateful material properties not allowed for this object."
764 " State ",
765 state,
766 " property for \"",
767 name,
768 "\" was requested.");
769
770 return this->getGenericMaterialPropertyByName<T, is_ad>(prop_name, material_data, state);
771}

773template <typename T, bool is_ad>
774const GenericMaterialProperty<T, is_ad> &
775MaterialPropertyInterface::getGenericMaterialPropertyByName(const MaterialPropertyName & name_in,
776 MaterialData & material_data,
777 const unsigned int state)
778{
779 if (_use_interpolated_state)
780 {
781 if (state == 1)
782 return getGenericMaterialPropertyByName<T, is_ad>(
783 name_in + _interpolated_old, material_data, 0);
784 if (state == 2)
785 return getGenericMaterialPropertyByName<T, is_ad>(
786 name_in + _interpolated_older, material_data, 0);
787 }
788
789 const auto name = _get_suffix.empty()
790 ? static_cast<const std::string &>(name_in)
791 : MooseUtils::join(std::vector<std::string>({name_in, _get_suffix}), "_");
792
793 checkExecutionStage();
794 checkMaterialProperty(name, state);
795
796 // mark property as requested
797 markMatPropRequested(name);
798
799 // Update the boolean flag.
800 _get_material_property_called = true;
801
802 // Call first so that the ID gets registered
803 auto & prop = material_data.getProperty<T, is_ad>(name, state, _mi_moose_object);
804
805 // Does the material data used here matter?
806 _material_property_dependencies.insert(material_data.getPropertyId(name));
807
808 if (state == 0)
809 addConsumedPropertyName(_mi_moose_object_name, name);
810
811 return prop;
