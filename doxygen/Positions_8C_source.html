14InputParameters
15Positions::validParams()
16{
17 InputParameters params = GeneralReporter::validParams();
18
19 params.addParam<PositionsName>("initial_positions",
20 "Positions at the beginning of the simulation");
21
22 // This parameter should be set by each derived class depending on whether the generation of
23 // positions is replicated or distributed. We want positions to be replicated across all ranks
24 params.addRequiredParam<bool>("auto_broadcast",
25 "Wether Positions should be broadcasted across all ranks");
26 // This parameter should be set by each derived class depending on whether positions should
27 // be sorted after being communicated. Sorting is usually undesirable, as the user input order
28 // should be respected. It may be only desirable for testing reproducibility
29 params.addRequiredParam<bool>("auto_sort", "Whether Positions should be sorted by coordinates");
30
31 // In general, no need to refresh Positions unless the mesh moved
32 params.set<ExecFlagEnum>("execute_on") = EXEC_NONE;
33
34 params.addParamNamesToGroup("auto_broadcast auto_sort", "Advanced");
35 params.registerBase("Positions");
36 return params;
37}

55const Point &
56Positions::getPosition(unsigned int index, bool initial) const
57{
58 mooseAssert(initialized(initial), "Positions vector has not been initialized.");
59 // Check sizes of inital positions
60 if (initial && _initial_positions && (*_initial_positions).size() < index)
61 mooseError("Initial positions is not sized or initialized appropriately");
62
63 // Check sizes of regular positions
64 if (!initial)
65 {
66 if (_initial_positions && _positions.size() != (*_initial_positions).size())
67 mooseError("Initial positions and current positions array length do not match");
68 else if (_positions.size() < index)
69 mooseError("Positions retrieved with an out-of-bound index: '",
70 index,
71 "' when there are only ",
72 _positions.size(),
73 " positions.");
74 }
75
76 if (initial && _initial_positions)
77 return (*_initial_positions)[index];
78 if (_positions.size())
79 return _positions[index];
80 else
81 mooseError("Positions vector has not been initialized.");
82}

92unsigned int
93Positions::getNearestPositionIndex(const Point & target, const bool initial) const
94{
95 mooseAssert(initialized(initial), "Positions vector has not been initialized.");
96 const auto & positions = (initial && _initial_positions) ? *_initial_positions : _positions;
97 // To keep track of indetermination due to equidistant positions
98 std::pair<int, int> conflict_index(-1, -1);
99
100 // TODO Use faster & fancier machinery such as a KNN-partition
101 std::size_t nearest_index = 0;
102 auto nearest_distance_sq = std::numeric_limits<Real>::max();
103 for (const auto i : index_range(positions))
104 {
105 const auto & pt = positions[i];
106 if (MooseUtils::absoluteFuzzyLessThan((pt - target).norm_sq(), nearest_distance_sq))
107 {
108 nearest_index = i;
109 nearest_distance_sq = (pt - target).norm_sq();
110 }
111 // Check that no two positions are equidistant to the target
112 else if (MooseUtils::absoluteFuzzyEqual((positions[i] - target).norm_sq(), nearest_distance_sq))
113 conflict_index = std::make_pair(cast_int<int>(i), cast_int<int>(nearest_index));
114 }
115
116 // If there were multiple "closest" positions, report a warning
117 if (cast_int<int>(nearest_index) == conflict_index.second)
118 {
119 mooseWarning("Search for nearest position found at least two matches: " +
120 Moose::stringify(positions[conflict_index.first]) + " and " +
121 Moose::stringify(positions[nearest_index]),
122 " for point " + Moose::stringify(target) + " at a distance of " +
123 std::to_string(std::sqrt(nearest_distance_sq)));
124 }
125
126 return nearest_index;
127}

176void
177Positions::unrollMultiDPositions()
178{
179 // Unroll in every dimension available
180 std::vector<Point> temp_2d_unrolled;
181 std::vector<Point> temp_3d_unrolled;
182 std::vector<Point> temp_4d_unrolled;
183 for (auto vec : _positions_2d)
184 temp_2d_unrolled.insert(temp_2d_unrolled.end(), vec.begin(), vec.end());
185 for (const auto & vec_vec : _positions_3d)
186 for (const auto & vec : vec_vec)
187 temp_3d_unrolled.insert(temp_3d_unrolled.end(), vec.begin(), vec.end());
188 for (const auto & vec_vec_vec : _positions_4d)
189 for (const auto & vec_vec : vec_vec_vec)
190 for (const auto & vec : vec_vec)
191 temp_4d_unrolled.insert(temp_4d_unrolled.end(), vec.begin(), vec.end());
192
193 // for now we wont even tolerate a different ordering
194 // 2D & 1D: check for conflicts
195 if (temp_2d_unrolled.size() && _positions.size() && temp_2d_unrolled != _positions)
196 mooseError("Inconsistency between the 2D and 1D position vectors detected. "
197 "The 2D positions must unroll into the 1D positions");
198
199 // 3D vs 2D & 1D
200 if (temp_3d_unrolled.size() && temp_2d_unrolled.size() && temp_3d_unrolled != temp_3d_unrolled)
201 mooseError("Inconsistency between the 3D and 2D position vectors detected. "
202 "The 3D positions must unroll the same way as the 2D positions");
203 if (temp_3d_unrolled.size() && _positions.size() && temp_3d_unrolled != _positions)
204 mooseError("Inconsistency between the 3D and 1D position vectors detected. "
205 "The 3D positions must unroll into the 1D positions");
206
207 // 4D vs all lower dimensions
208 if (temp_4d_unrolled.size() && temp_3d_unrolled.size() && temp_4d_unrolled != temp_3d_unrolled)
209 mooseError("Inconsistency between the 4D and 3D position vectors detected. "
210 "The 4D positions must unroll the same way as the 3D positions");
211 if (temp_4d_unrolled.size() && temp_2d_unrolled.size() && temp_4d_unrolled != temp_2d_unrolled)
212 mooseError("Inconsistency between the 4D and 2D position vectors detected. "
213 "The 4D positions must unroll the same way as the 2D positions");
214 if (temp_4d_unrolled.size() && _positions.size() && temp_4d_unrolled != _positions)
215 mooseError("Inconsistency between the 4D and 1D position vectors detected. "
216 "The 4D positions must unroll into the 1D positions");
217
218 // Use any of the higher D to set the one D vector
219 if (!_positions.size())
220 {
221 if (temp_2d_unrolled.size())
222 _positions = temp_2d_unrolled;
223 else if (temp_3d_unrolled.size())
224 _positions = temp_3d_unrolled;
225 else if (temp_4d_unrolled.size())
226 _positions = temp_4d_unrolled;
227 else
228 mooseError("Positions::unrollMultiDPositions() may only be called if there is at least one "
229 "non-empty positions vector.");
230 }
231}

233void
234Positions::finalize()
235{
236 // Gather up the positions vector on all ranks
237 mooseAssert(initialized(false), "Positions vector has not been initialized.");
238 if (_need_broadcast)
239 // The consumer/producer reporter interface can keep track of whether a reduction is needed
240 // (for example if a consumer needs replicated data, but the producer is distributed) however,
241 // we have currently made the decision that positions should ALWAYS be replicated
242 _communicator.allgather(_positions, /* identical buffer lengths = */ false);
243
244 // Sort positions by X then Y then Z
245 if (_need_sort)
246 std::sort(_positions.begin(), _positions.end());
247}

249Real
250Positions::getMinDistanceBetweenPositions() const
251{
252 mooseAssert(initialized(false), "Positions vector has not been initialized.");
253 // Dumb nested loops. We can revisit this once we have a KDTree for nearest position searching
254 // These nested loops are still faster than calling getNearestPositions on each position for now
255 Real min_distance_sq = std::numeric_limits<Real>::max();
256 for (const auto i1 : index_range(_positions))
257 for (auto i2 = i1 + 1; i2 < _positions.size(); i2++)
258 min_distance_sq = std::min(min_distance_sq, (_positions[i1] - _positions[i2]).norm_sq());
259 return std::sqrt(min_distance_sq);
260}
