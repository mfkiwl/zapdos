18void
19VariableWarehouse::add(const std::string & var_name, std::shared_ptr<MooseVariableBase> var)
20{
21 _names.push_back(var_name);
22 auto * raw_var = var.get();
23 _all_objects[var->number()] = var;
24 _var_name[var_name] = raw_var;
25
26 if (auto * tmp_var = dynamic_cast<MooseVariableFieldBase *>(raw_var))
27 {
28 _vars.push_back(tmp_var);
29 if (auto * tmp_var = dynamic_cast<MooseVariable *>(raw_var))
30 {
31 _regular_vars_by_number[tmp_var->number()] = tmp_var;
32 _regular_vars_by_name[var_name] = tmp_var;
33 }
34 else if (auto * tmp_var = dynamic_cast<MooseVariableFVReal *>(raw_var))
35 {
36 _fv_vars_by_number[tmp_var->number()] = tmp_var;
37 _fv_vars_by_name[var_name] = tmp_var;
38 }
39 else if (auto * tmp_var = dynamic_cast<VectorMooseVariable *>(raw_var))
40 {
41 _vector_vars_by_number[tmp_var->number()] = tmp_var;
42 _vector_vars_by_name[var_name] = tmp_var;
43 }
44 else if (auto * tmp_var = dynamic_cast<ArrayMooseVariable *>(raw_var))
45 {
46 _array_vars_by_number[tmp_var->number()] = tmp_var;
47 _array_vars_by_name[var_name] = tmp_var;
48 }
49 else
50 mooseError("Unknown variable class passed into VariableWarehouse. Attempt to hack us?");
51 }
52 else if (auto * tmp_var = dynamic_cast<MooseVariableScalar *>(raw_var))
53 _scalar_vars.push_back(tmp_var);
54 else
55 mooseError("Unknown variable class passed into VariableWarehouse. Attempt to hack us?");
56}
