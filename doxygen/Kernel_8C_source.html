30Kernel::Kernel(const InputParameters & parameters)
31 : KernelBase(parameters),
32 MooseVariableInterface<Real>(this,
33 false,
34 "variable",
35 Moose::VarKindType::VAR_NONLINEAR,
36 Moose::VarFieldType::VAR_FIELD_STANDARD),
37 _var(*mooseVariable()),
38 _test(_var.phi()),
39 _grad_test(_var.gradPhi()),
40 _phi(_assembly.phi(_var)),
41 _grad_phi(_assembly.gradPhi(_var)),
42 _u(_is_implicit ? _var.sln() : _var.slnOld()),
43 _grad_u(_is_implicit ? _var.gradSln() : _var.gradSlnOld())
44{
45 addMooseVariableDependency(mooseVariable());
46 _save_in.resize(_save_in_strings.size());
47 _diag_save_in.resize(_diag_save_in_strings.size());
48
49 for (unsigned int i = 0; i < _save_in_strings.size(); i++)
50 {
51 MooseVariable * var = &_subproblem.getStandardVariable(_tid, _save_in_strings[i]);
52
53 if (_fe_problem.getNonlinearSystemBase(_sys.number()).hasVariable(_save_in_strings[i]))
54 paramError("save_in", "cannot use solution variable as save-in variable");
55
56 if (var->feType() != _var.feType())
57 paramError(
58 "save_in",
59 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
60 moose::internal::incompatVarMsg(*var, _var));
61
62 _save_in[i] = var;
63 var->sys().addVariableToZeroOnResidual(_save_in_strings[i]);
64 addMooseVariableDependency(var);
65 }
66
67 _has_save_in = _save_in.size() > 0;
68
69 for (unsigned int i = 0; i < _diag_save_in_strings.size(); i++)
70 {
71 MooseVariable * var = &_subproblem.getStandardVariable(_tid, _diag_save_in_strings[i]);
72
73 if (_fe_problem.getNonlinearSystemBase(_sys.number()).hasVariable(_diag_save_in_strings[i]))
74 paramError("diag_save_in", "cannot use solution variable as diag save-in variable");
75
76 if (var->feType() != _var.feType())
77 paramError(
78 "diag_save_in",
79 "saved-in auxiliary variable is incompatible with the object's nonlinear variable: ",
80 moose::internal::incompatVarMsg(*var, _var));
81
82 _diag_save_in[i] = var;
83 var->sys().addVariableToZeroOnJacobian(_diag_save_in_strings[i]);
84 addMooseVariableDependency(var);
85 }
86
87 _has_diag_save_in = _diag_save_in.size() > 0;
88}

90void
91Kernel::computeResidual()
92{
93 prepareVectorTag(_assembly, _var.number());
94
95 precalculateResidual();
96 for (_i = 0; _i < _test.size(); _i++)
97 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
98 _local_re(_i) += _JxW[_qp] * _coord[_qp] * computeQpResidual();
99
100 accumulateTaggedLocalResidual();
101
102 if (_has_save_in)
103 {
104 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
105 for (const auto & var : _save_in)
106 var->sys().solution().add_vector(_local_re, var->dofIndices());
107 }
108}

110void
111Kernel::computeJacobian()
112{
113 prepareMatrixTag(_assembly, _var.number(), _var.number());
114
115 precalculateJacobian();
116 for (_i = 0; _i < _test.size(); _i++)
117 for (_j = 0; _j < _phi.size(); _j++)
118 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
119 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeQpJacobian();
120
121 accumulateTaggedLocalMatrix();
122
123 if (_has_diag_save_in && !_sys.computingScalingJacobian())
124 {
125 DenseVector<Number> diag = _assembly.getJacobianDiagonal(_local_ke);
126 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
127 for (const auto & var : _diag_save_in)
128 var->sys().solution().add_vector(diag, var->dofIndices());
129 }
130}

132void
133Kernel::computeOffDiagJacobian(const unsigned int jvar_num)
134{
135 const auto & jvar = getVariable(jvar_num);
136
137 if (jvar_num == _var.number())
138 computeJacobian();
139 else
140 {
141 prepareMatrixTag(_assembly, _var.number(), jvar_num);
142
143 // This (undisplaced) jvar could potentially yield the wrong phi size if this object is acting
144 // on the displaced mesh
145 auto phi_size = jvar.dofIndices().size();
146 mooseAssert(
147 phi_size * jvar.count() == _local_ke.n(),
148 "The size of the phi container does not match the number of local Jacobian columns");
149
150 if (_local_ke.m() != _test.size())
151 return;
152
153 precalculateOffDiagJacobian(jvar_num);
154 if (jvar.count() == 1)
155 {
156 for (_i = 0; _i < _test.size(); _i++)
157 for (_j = 0; _j < phi_size; _j++)
158 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
159 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobian(jvar.number());
160 }
161 else
162 {
163 unsigned int n = phi_size;
164 for (_i = 0; _i < _test.size(); _i++)
165 for (_j = 0; _j < n; _j++)
166 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
167 {
168 RealEigenVector v = _JxW[_qp] * _coord[_qp] *
169 computeQpOffDiagJacobianArray(static_cast<ArrayMooseVariable &>(
170 const_cast<MooseVariableFieldBase &>(jvar)));
171 for (unsigned int k = 0; k < v.size(); ++k)
172 _local_ke(_i, _j + k * n) += v(k);
173 }
174 }
175
176 accumulateTaggedLocalMatrix();
177 }
178}

180void
181Kernel::computeOffDiagJacobianScalar(const unsigned int jvar)
182{
183 MooseVariableScalar & jv = _sys.getScalarVariable(_tid, jvar);
184 prepareMatrixTag(_assembly, _var.number(), jvar);
185
186 for (_i = 0; _i < _test.size(); _i++)
187 for (_j = 0; _j < jv.order(); _j++)
188 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
189 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeQpOffDiagJacobianScalar(jvar);
190
191 accumulateTaggedLocalMatrix();
192}

195Kernel::computeResidualAndJacobian()
196{
197 computeResidual();
198
199 for (const auto & [ivariable, jvariable] : _fe_problem.couplingEntries(_tid, _sys.number()))
200 {
201 const unsigned int ivar = ivariable->number();
202 const unsigned int jvar = jvariable->number();
203
204 if (ivar != _var.number())
205 continue;
206
207 if (_is_implicit)
208 {
209 prepareShapes(jvar);
210 computeOffDiagJacobian(jvar);
211 }
212 }
