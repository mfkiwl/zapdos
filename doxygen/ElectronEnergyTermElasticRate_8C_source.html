42ElectronEnergyTermElasticRate::computeQpResidual()
43{
44 ADReal energy_elastic = -3.0 * (_massIncident[_qp] / _massTarget[_qp]) * 2.0 / 3.0 *
45 std::exp(_u[_qp] - _electron[_qp]);
