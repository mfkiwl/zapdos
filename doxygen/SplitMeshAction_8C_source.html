30void
31SplitMeshAction::act()
32{
33 auto mesh = _app.actionWarehouse().mesh();
34 auto split_file_arg = _app.parameters().get<std::string>("split_file");
35
36 if (mesh->getFileName() == "" && split_file_arg == "")
37 mooseError("Output mesh file name must be specified (with --split-file) when splitting "
38 "non-file-based meshes");
39
40 auto splitstr = _app.parameters().get<std::string>("split_mesh");
41 std::vector<unsigned int> splits;
42 bool success = MooseUtils::tokenizeAndConvert(splitstr, splits, ", ");
43 if (!success)
44 mooseError("invalid argument for --split-mesh: '", splitstr, "'");
45
46 // Decide whether to create ASCII or binary splits based on the split_file_arg. We use the
47 // following rules to decide:
48 // 1.) No file extension -> binary
49 // 2.) .cpr file extension -> binary
50 // 3.) .cpa file extension -> ASCII
51 // 4.) Any other file extension -> mooseError
52
53 // Get the file extension without the dot.
54 // TODO: Maybe this should be in MooseUtils?
55 std::string split_file_arg_ext;
56 auto pos = split_file_arg.rfind(".");
57 if (pos != std::string::npos)
58 split_file_arg_ext = split_file_arg.substr(pos + 1, std::string::npos);
59
60 // If stripExtension() returns the original string, then there is no
61 // file extension or the original string was empty.
62 bool checkpoint_binary_flag = true;
63
64 if (split_file_arg_ext != "")
65 {
66 if (split_file_arg_ext == "cpr")
67 checkpoint_binary_flag = true;
68 else if (split_file_arg_ext == "cpa")
69 checkpoint_binary_flag = false;
70 else
71 mooseError("The argument to --split-file, ",
72 split_file_arg,
73 ", must not end in a file extension other than .cpr or .cpa");
74 }
75
76 // To name the split files, we start with the given mesh filename
77 // (if set) or the argument to --split-file, strip any existing
78 // extension, and then append either .cpr or .cpa depending on the
79 // checkpoint_binary_flag.
80 auto fname = mesh->getFileName();
81 if (fname == "")
82 fname = split_file_arg;
83 fname = MooseUtils::stripExtension(fname) + (checkpoint_binary_flag ? ".cpr" : ".cpa");
84
85 for (std::size_t i = 0; i < splits.size(); i++)
86 {
87 processor_id_type n = splits[i];
88 Moose::out << "Splitting " << n << " ways..." << std::endl;
89
90 auto cp = libMesh::split_mesh(*mesh, n);
91 Moose::out << " - writing " << cp->current_processor_ids().size() << " files per process..."
92 << std::endl;
93 cp->binary() = checkpoint_binary_flag;
94
95 // different splits will be written into subfolders with n being the folder name
96 cp->write(fname);
97 }
98
99 // Write mesh metadata
100 if (processor_id() == 0)
101 {
102 const auto filenames = _app.writeRestartableMetaData(MooseApp::MESH_META_DATA, fname);
103 Moose::out << "Mesh meta data written into "
104 << std::filesystem::absolute(filenames[0].parent_path()) << "." << std::endl;
105 }
106}
