41ComputeMarkerThread::subdomainChanged()
42{

44 _marker_whs.subdomainSetup(_tid);
45

47 _marker_whs.updateVariableDependency(needed_moose_vars, _tid);

56void
57ComputeMarkerThread::onElement(const Elem * elem)
58{
59 _fe_problem.prepare(elem, _tid);
60 _fe_problem.reinitElem(elem, _tid);
61
62 // Set up Sentinel class so that, even if reinitMaterials() throws, we
63 // still remember to swap back during stack unwinding.
64 SwapBackSentinel sentinel(_fe_problem,
65 &FEProblem::swapBackMaterials,
66 _tid,
67 _fe_problem.hasActiveMaterialProperties(_tid));
68
69 _fe_problem.reinitMaterials(_subdomain, _tid);
70
71 if (_marker_whs.hasActiveBlockObjects(_subdomain, _tid))
72 {
73 const std::vector<std::shared_ptr<Marker>> & markers =
74 _marker_whs.getActiveBlockObjects(_subdomain, _tid);
75 for (const auto & marker : markers)
76 marker->computeMarker();
77 }
78
79 {
80 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
81 for (auto * var : _aux_sys._elem_vars[_tid])
82 var->insert(_aux_sys.solution());
83 }
84}

125void
126ComputeMarkerThread::printBlockExecutionInformation() const
127{
128 if (!_fe_problem.shouldPrintExecution(_tid) || _blocks_exec_printed.count(_subdomain) ||
129 !_marker_whs.hasActiveBlockObjects(_subdomain, _tid))
130 return;
131
132 const auto & console = _fe_problem.console();
133 const auto & markers = _marker_whs.getActiveBlockObjects(_subdomain, _tid);

135 printExecutionOrdering<Marker>(markers, false);
136 _blocks_exec_printed.insert(_subdomain);
