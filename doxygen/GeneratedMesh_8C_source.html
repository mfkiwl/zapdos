25InputParameters
26GeneratedMesh::validParams()
27{
28 InputParameters params = MooseMesh::validParams();
29
30 MooseEnum elem_types(LIST_GEOM_ELEM); // no default
31
32 MooseEnum dims("1=1 2 3");
33 params.addRequiredParam<MooseEnum>(
34 "dim", dims, "The dimension of the mesh to be generated"); // Make this parameter required
35
36 params.addParam<unsigned int>("nx", 1, "Number of elements in the X direction");
37 params.addParam<unsigned int>("ny", 1, "Number of elements in the Y direction");
38 params.addParam<unsigned int>("nz", 1, "Number of elements in the Z direction");
39 params.addParam<Real>("xmin", 0.0, "Lower X Coordinate of the generated mesh");
40 params.addParam<Real>("ymin", 0.0, "Lower Y Coordinate of the generated mesh");
41 params.addParam<Real>("zmin", 0.0, "Lower Z Coordinate of the generated mesh");
42 params.addParam<Real>("xmax", 1.0, "Upper X Coordinate of the generated mesh");
43 params.addParam<Real>("ymax", 1.0, "Upper Y Coordinate of the generated mesh");
44 params.addParam<Real>("zmax", 1.0, "Upper Z Coordinate of the generated mesh");
45 params.addParam<MooseEnum>("elem_type",
46 elem_types,
47 "The type of element from libMesh to "
48 "generate (default: linear element for "
49 "requested dimension)");
50 params.addParam<bool>(
51 "gauss_lobatto_grid",
52 false,
53 "Grade mesh into boundaries according to Gauss-Lobatto quadrature spacing.");
54 params.addRangeCheckedParam<Real>(
55 "bias_x",
56 1.,
57 "bias_x>=0.5 & bias_x<=2",
58 "The amount by which to grow (or shrink) the cells in the x-direction.");
59 params.addRangeCheckedParam<Real>(
60 "bias_y",
61 1.,
62 "bias_y>=0.5 & bias_y<=2",
63 "The amount by which to grow (or shrink) the cells in the y-direction.");
64 params.addRangeCheckedParam<Real>(
65 "bias_z",
66 1.,
67 "bias_z>=0.5 & bias_z<=2",
68 "The amount by which to grow (or shrink) the cells in the z-direction.");
69
70 params.addParamNamesToGroup("dim", "Main");
71
72 params.addClassDescription(
73 "Create a line, square, or cube mesh with uniformly spaced or biased elements.");
74 return params;
75}

156void
157GeneratedMesh::buildMesh()
158{
159 MooseEnum elem_type_enum = getParam<MooseEnum>("elem_type");
160
161 if (!isParamValid("elem_type"))
162 {
163 // Switching on MooseEnum
164 switch (_dim)
165 {
166 case 1:
167 elem_type_enum = "EDGE2";
168 break;
169 case 2:
170 elem_type_enum = "QUAD4";
171 break;
172 case 3:
173 elem_type_enum = "HEX8";
174 break;
175 }
176 }
177
178 ElemType elem_type = Utility::string_to_enum<ElemType>(elem_type_enum);
179
180 // Switching on MooseEnum
181 switch (_dim)
182 {
183 // The build_XYZ mesh generation functions take an
184 // UnstructuredMesh& as the first argument, hence the dynamic_cast.
185 case 1:
186 MeshTools::Generation::build_line(dynamic_cast<UnstructuredMesh &>(getMesh()),
187 _nx,
188 _xmin,
189 _xmax,
190 elem_type,
191 _gauss_lobatto_grid);
192 break;
193 case 2:
194 MeshTools::Generation::build_square(dynamic_cast<UnstructuredMesh &>(getMesh()),
195 _nx,
196 _ny,
197 _xmin,
198 _xmax,
199 _ymin,
200 _ymax,
201 elem_type,
202 _gauss_lobatto_grid);
203 break;
204 case 3:
205 MeshTools::Generation::build_cube(dynamic_cast<UnstructuredMesh &>(getMesh()),
206 _nx,
207 _ny,
208 _nz,
209 _xmin,
210 _xmax,
211 _ymin,
212 _ymax,
213 _zmin,
214 _zmax,
215 elem_type,
216 _gauss_lobatto_grid);
217 break;
218 }
219
220 // Apply the bias if any exists
221 if (_bias_x != 1.0 || _bias_y != 1.0 || _bias_z != 1.0)
222 {
223 // Reference to the libmesh mesh
224 MeshBase & mesh = getMesh();
225
226 // Biases
227 std::array<Real, LIBMESH_DIM> bias = {
228 {_bias_x, _dim > 1 ? _bias_y : 1.0, _dim > 2 ? _bias_z : 1.0}};
229
230 // "width" of the mesh in each direction
231 std::array<Real, LIBMESH_DIM> width = {
232 {_xmax - _xmin, _dim > 1 ? _ymax - _ymin : 0, _dim > 2 ? _zmax - _zmin : 0}};
233
234 // Min mesh extent in each direction.
235 std::array<Real, LIBMESH_DIM> mins = {
236 {getMinInDimension(0), getMinInDimension(1), getMinInDimension(2)}};
237
238 // Number of elements in each direction.
239 std::array<unsigned int, LIBMESH_DIM> nelem = {{_nx, _dim > 1 ? _ny : 1, _dim > 2 ? _nz : 1}};
240
241 // We will need the biases raised to integer powers in each
242 // direction, so let's pre-compute those...
243 std::array<std::vector<Real>, LIBMESH_DIM> pows;
244 for (unsigned int dir = 0; dir < LIBMESH_DIM; ++dir)
245 {
246 pows[dir].resize(nelem[dir] + 1);
247 for (unsigned int i = 0; i < pows[dir].size(); ++i)
248 pows[dir][i] = std::pow(bias[dir], static_cast<int>(i));
249 }
250
251 // Loop over the nodes and move them to the desired location
252 for (auto & node_ptr : mesh.node_ptr_range())
253 {
254 Node & node = *node_ptr;
255
256 for (unsigned int dir = 0; dir < LIBMESH_DIM; ++dir)
257 {
258 if (width[dir] != 0. && bias[dir] != 1.)
259 {
260 // Compute the scaled "index" of the current point. This
261 // will either be close to a whole integer or a whole
262 // integer+0.5 for quadratic nodes.
263 Real float_index = (node(dir) - mins[dir]) * nelem[dir] / width[dir];
264
265 Real integer_part = 0;
266 Real fractional_part = std::modf(float_index, &integer_part);
267
268 // Figure out where to move the node...
269 if (std::abs(fractional_part) < TOLERANCE || std::abs(fractional_part - 1.0) < TOLERANCE)
270 {
271 // If the fractional_part ~ 0.0 or 1.0, this is a vertex node, so
272 // we don't need an average.
273 //
274 // Compute the "index" we are at in the current direction. We
275 // round to the nearest integral value to get this instead
276 // of using "integer_part", since that could be off by a
277 // lot (e.g. we want 3.9999 to map to 4.0 instead of 3.0).
278 int index = round(float_index);
279
280 mooseAssert(index >= static_cast<int>(0) && index < static_cast<int>(pows[dir].size()),
281 "Scaled \"index\" out of range");
282
283 // Move node to biased location.
284 node(dir) =
285 mins[dir] + width[dir] * (1. - pows[dir][index]) / (1. - pows[dir][nelem[dir]]);
286 }
287 else if (std::abs(fractional_part - 0.5) < TOLERANCE)
288 {
289 // If the fractional_part ~ 0.5, this is a midedge/face
290 // (i.e. quadratic) node. We don't move those with the same
291 // bias as the vertices, instead we put them midway between
292 // their respective vertices.
293 //
294 // Also, since the fractional part is nearly 0.5, we know that
295 // the integer_part will be the index of the vertex to the
296 // left, and integer_part+1 will be the index of the
297 // vertex to the right.
298 node(dir) = mins[dir] +
299 width[dir] *
300 (1. - 0.5 * (pows[dir][integer_part] + pows[dir][integer_part + 1])) /
301 (1. - pows[dir][nelem[dir]]);
302 }
303 else
304 {
305 // We don't yet handle anything higher order than quadratic...
306 mooseError("Unable to bias node at node(", dir, ")=", node(dir));
307 }
308 }
309 }
310 }
311 }
