36ParsedElementDeletionGenerator::ParsedElementDeletionGenerator(const InputParameters & parameters)
37 : ElementDeletionGeneratorBase(parameters), FunctionParserUtils<false>(parameters)
38{
39 _function = std::make_shared<SymFunction>();
40
41 // set FParser internal feature flags
42 setParserFeatureFlags(_function);
43
44 // add the constant expressions
45 addFParserConstants(_function,
46 getParam<std::vector<std::string>>("constant_names"),
47 getParam<std::vector<std::string>>("constant_expressions"));
48
49 // parse function
50 if (_function->Parse(getParam<ParsedFunctionExpression>("expression"), "x,y,z,volume,id") >= 0)
51 paramError("expression", "Invalid function: ", _function->ErrorMsg());
52
53 _func_params.resize(5);
54}

56bool
57ParsedElementDeletionGenerator::shouldDelete(const Elem * elem)
58{
59 const auto vertex_average = elem->vertex_average();
60 for (const auto i : make_range(3))
61 _func_params[i] = vertex_average(i);
62 _func_params[3] = elem->volume();
63 _func_params[4] = elem->id();
64
65 return evaluate(_function, "expression") > 0;
66}
