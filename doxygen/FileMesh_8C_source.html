58void
59FileMesh::buildMesh()
60{
61 TIME_SECTION("buildMesh", 2, "Reading Mesh");
62
63 getMesh().set_mesh_dimension(getParam<MooseEnum>("dim"));
64
65 if (_is_nemesis)
66 {
67 // Nemesis_IO only takes a reference to DistributedMesh, so we can't be quite so short here.
68 DistributedMesh & pmesh = cast_ref<DistributedMesh &>(getMesh());
69 Nemesis_IO(pmesh).read(_file_name);
70
71 getMesh().allow_renumbering(false);
72
73 // Even if we want repartitioning when load balancing later, we'll
74 // begin with the default partitioning defined by the Nemesis
75 // file.
76 bool skip_partitioning_later = getMesh().skip_partitioning();
77 getMesh().skip_partitioning(true);
78 getMesh().prepare_for_use();
79 getMesh().skip_partitioning(skip_partitioning_later);
80 }
81 else // not reading Nemesis files
82 {
83 // See if the user has requested reading a solution from the file. If so, we'll need to read
84 // the mesh with the exodus reader instead of using mesh.read(). This will read the mesh on
85 // every processor
86
87 if (_app.getExodusFileRestart() && (_file_name.rfind(".exd") < _file_name.size() ||
88 _file_name.rfind(".e") < _file_name.size()))
89 {
90 MooseUtils::checkFileReadable(_file_name);
91
92 auto exreader = std::make_shared<ExodusII_IO>(getMesh());
93 _app.setExReaderForRestart(std::move(exreader));
94 exreader->read(_file_name);
95
96 getMesh().allow_renumbering(false);
97 getMesh().prepare_for_use();
98 }
99 else
100 {
101 // Supports old suffix (xxxx_mesh.cpr -> xxxx-mesh.cpr) and LATEST
102 _file_name = FileMeshGenerator::deduceCheckpointPath(*this, _file_name);
103
104 // If we are reading a mesh while restarting, then we might have
105 // a solution file that relies on that mesh partitioning and/or
106 // numbering. In that case, we need to turn off repartitioning
107 // and renumbering, at least at first.
108 bool restarting = _file_name.rfind(".cpr") < _file_name.size();
109
110 const bool skip_partitioning_later = restarting && getMesh().skip_partitioning();
111 const bool allow_renumbering_later = restarting && getMesh().allow_renumbering();
112
113 if (restarting)
114 {
115 getMesh().skip_partitioning(true);
116 getMesh().allow_renumbering(false);
117 }
118
119 MooseUtils::checkFileReadable(_file_name);
120 getMesh().read(_file_name);
121
122 if (getParam<bool>("clear_spline_nodes"))
123 MeshTools::clear_spline_nodes(getMesh());
124
125 _app.possiblyLoadRestartableMetaData(MooseApp::MESH_META_DATA, _file_name);
126
127 if (restarting)
128 {
129 getMesh().allow_renumbering(allow_renumbering_later);
130 getMesh().skip_partitioning(skip_partitioning_later);
131 }
132 }
133 }
134}

136void
137FileMesh::read(const std::string & file_name)
138{
139 if (dynamic_cast<DistributedMesh *>(&getMesh()) && !_is_nemesis)
140 getMesh().read(file_name, /*mesh_data=*/NULL, /*skip_renumber=*/false);
141 else
142 getMesh().read(file_name, /*mesh_data=*/NULL, /*skip_renumber=*/true);
143}
