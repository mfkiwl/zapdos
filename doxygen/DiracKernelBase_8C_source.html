85void
86DiracKernelBase::addPoint(const Elem * elem, Point p, unsigned /*id*/)
87{
88 if (!elem || !hasBlocks(elem->subdomain_id()))
89 {
90 std::stringstream msg;
91 msg << "Point " << p << " not found in block(s) " << Moose::stringify(blockIDs(), ", ") << ".";
92 switch (_point_not_found_behavior)
93 {
94 case PointNotFoundBehavior::ERROR:
95 mooseError(msg.str());
96 break;
97 case PointNotFoundBehavior::WARNING:
98 mooseDoOnce(mooseWarning(msg.str()));
99 break;
100 case PointNotFoundBehavior::IGNORE:
101 break;
102 default:
103 mooseError("Internal enum error.");
104 }
105 return;
106 }
107
108 if (elem->processor_id() != processor_id())
109 return;
110
111 _dirac_kernel_info.addPoint(elem, p);
112 _local_dirac_kernel_info.addPoint(elem, p);
113}

115const Elem *
116DiracKernelBase::addPoint(Point p, unsigned id)
117{
118 // Make sure that this method was called with the same id on all
119 // processors. It's an extra communication, though, so let's only
120 // do it in DEBUG mode.
121 libmesh_assert(comm().verify(id));
122
123 if (id != libMesh::invalid_uint)
124 return addPointWithValidId(p, id);
125
126 // If id == libMesh::invalid_uint (the default), the user is not
127 // enabling caching when they add Dirac points. So all we can do is
128 // the PointLocator lookup, and call the other addPoint() method.
129 const Elem * elem = _dirac_kernel_info.findPoint(p, _mesh, blockIDs());
130 addPoint(elem, p, id);
131 return elem;
132}

134const Elem *
135DiracKernelBase::addPointWithValidId(Point p, unsigned id)
136{
137 // The Elem we'll eventually return. We can't return early on some
138 // processors, because we may need to call parallel_only() functions in
139 // the remainder of this scope.
140 const Elem * return_elem = NULL;
141
142 // May be set if the Elem is found in our cache, otherwise stays as NULL.
143 const Elem * cached_elem = NULL;
144
145 // OK, the user gave us an ID, let's see if we already have it...
146 point_cache_t::iterator it = _point_cache.find(id);
147
148 // Was the point found in a _point_cache on at least one processor?
149 unsigned int i_found_it = static_cast<unsigned int>(it != _point_cache.end());
150 unsigned int we_found_it = i_found_it;
151 comm().max(we_found_it);
152
153 // If nobody found it in their local caches, it means we need to
154 // do the PointLocator look-up and update the caches. This is
155 // safe, because all processors have the same value of we_found_it.
156 if (!we_found_it)
157 {
158 const Elem * elem = _dirac_kernel_info.findPoint(p, _mesh, blockIDs());
159
160 // Only add the point to the cache on this processor if the Elem is local
161 if (elem && (elem->processor_id() == processor_id()))
162 {
163 // Add the point to the cache...
164 _point_cache[id] = std::make_pair(elem, p);
165
166 // ... and to the reverse cache.
167 std::vector<std::pair<Point, unsigned>> & points = _reverse_point_cache[elem];
168 points.push_back(std::make_pair(p, id));
169 }
170
171 // Call the other addPoint() method. This method ignores non-local
172 // and NULL elements automatically.
173 addPoint(elem, p, id);
174 return_elem = elem;
175 }
176
177 // If the point was found in a cache, but not my cache, I'm not
178 // responsible for it.
179 //
180 // We can't return early here: then we aren't allowed to call any more
181 // parallel_only() functions in the remainder of this function!
182 if (we_found_it && !i_found_it)
183 return_elem = NULL;
184
185 // This flag may be set by the processor that cached the Elem because it
186 // needs to call findPoint() (due to moving mesh, etc.). If so, we will
187 // call it at the end of the while loop below.
188 bool i_need_find_point = false;
189
190 // Now that we only cache local data, some processors may enter
191 // this if statement and some may not. Therefore we can't call
192 // any parallel_only() functions inside this if statement.
193 while (i_found_it)
194 {
195 // We have something cached, now make sure it's actually the same Point.
196 // TODO: we should probably use this same comparison in the DiracKernelInfo code!
197 Point cached_point = (it->second).second;
198
199 if (cached_point.relative_fuzzy_equals(p))
200 {
201 // Find the cached element associated to this point
202 cached_elem = (it->second).first;
203
204 // If the cached element's processor ID doesn't match ours, we
205 // are no longer responsible for caching it. This can happen
206 // due to adaptivity...
207 if (cached_elem->processor_id() != processor_id())
208 {
209 // Update the caches, telling them to drop the cached Elem.
210 // Analogously to the rest of the DiracKernel system, we
211 // also return NULL because the Elem is non-local.
212 updateCaches(cached_elem, NULL, p, id);
213 return_elem = NULL;
214 break; // out of while loop
215 }
216
217 bool active = cached_elem->active();
218 bool contains_point = cached_elem->contains_point(p);
219
220 // If the cached Elem is active and the point is still
221 // contained in it, call the other addPoint() method and
222 // return its result.
223 if (active && contains_point)
224 {
225 addPoint(cached_elem, p, id);
226 return_elem = cached_elem;
227 break; // out of while loop
228 }
229
230 // Is the Elem not active (been refined) but still contains the point?
231 // Then search in its active children and update the caches.
232 else if (!active && contains_point)
233 {
234 // Get the list of active children
235 std::vector<const Elem *> active_children;
236 cached_elem->active_family_tree(active_children);
237
238 // Linear search through active children for the one that contains p
239 for (unsigned c = 0; c < active_children.size(); ++c)
240 if (active_children[c]->contains_point(p))
241 {
242 updateCaches(cached_elem, active_children[c], p, id);
243 addPoint(active_children[c], p, id);
244 return_elem = active_children[c];
245 break; // out of for loop
246 }
247
248 // If we got here without setting return_elem, it means the Point was
249 // found in the parent element, but not in any of the active
250 // children... this is not possible under normal
251 // circumstances, so something must have gone seriously
252 // wrong!
253 if (!return_elem)
254 mooseError("Error, Point not found in any of the active children!");
255
256 break; // out of while loop
257 }
258
259 else if (
260 // Is the Elem active but the point is not contained in it any
261 // longer? (For example, did the Mesh move out from under
262 // it?) Then we fall back to the expensive Point Locator
263 // lookup. TODO: we could try and do something more optimized
264 // like checking if any of the active neighbors contains the
265 // point. Update the caches.
266 (active && !contains_point) ||
267
268 // The Elem has been refined *and* the Mesh has moved out
269 // from under it, we fall back to doing the expensive Point
270 // Locator lookup. TODO: We could try and look in the
271 // active children of this Elem's neighbors for the Point.
272 // Update the caches.
273 (!active && !contains_point))
274 {
275 i_need_find_point = true;
276 break; // out of while loop
277 }
278
279 else
280 mooseError("We'll never get here!");
281 } // if (cached_point.relative_fuzzy_equals(p))
282 else
283 mooseError("Cached Dirac point ",
284 cached_point,
285 " already exists with ID: ",
286 id,
287 " and does not match point ",
288 p,
289 "If Dirac sources are moving, please set 'allow_moving_sources' to true");
290
291 // We only want one iteration of this while loop at maximum.
292 i_found_it = false;
293 } // while (i_found_it)
294
295 // We are back to all processors here because we do not return
296 // early in the code above...
297
298 // Does we need to call findPoint() on all processors.
299 unsigned int we_need_find_point = static_cast<unsigned int>(i_need_find_point);
300 comm().max(we_need_find_point);
301
302 if (we_need_find_point)
303 {
304 // findPoint() is a parallel-only function
305 const Elem * elem = _dirac_kernel_info.findPoint(p, _mesh, blockIDs());
306
307 updateCaches(cached_elem, elem, p, id);
308 addPoint(elem, p, id);
309 return_elem = elem;
310 }
311
312 return return_elem;
313}

344void
345DiracKernelBase::updateCaches(const Elem * old_elem, const Elem * new_elem, Point p, unsigned id)
346{
347 // Update the point cache. Remove old cached data, only cache
348 // new_elem if it is non-NULL and local.
349 _point_cache.erase(id);
350 if (new_elem && (new_elem->processor_id() == processor_id()))
351 _point_cache[id] = std::make_pair(new_elem, p);
352
353 // Update the reverse cache
354 //
355 // First, remove the Point from the old_elem's vector
356 reverse_cache_t::iterator it = _reverse_point_cache.find(old_elem);
357 if (it != _reverse_point_cache.end())
358 {
359 reverse_cache_t::mapped_type & points = it->second;
360 {
361 reverse_cache_t::mapped_type::iterator points_it = points.begin(), points_end = points.end();
362
363 for (; points_it != points_end; ++points_it)
364 {
365 // If the point matches, remove it from the vector of points
366 if (p.relative_fuzzy_equals(points_it->first))
367 {
368 // Vector erasure. It can be slow but these vectors are
369 // generally very short. It also invalidates existing
370 // iterators, so we need to break out of the loop and
371 // not use them any more.
372 points.erase(points_it);
373 break;
374 }
375 }
376
377 // If the points vector is now empty, remove old_elem from the reverse cache entirely
378 if (points.empty())
379 _reverse_point_cache.erase(old_elem);
380 }
381 }
382
383 // Next, if new_elem is not NULL and local, add the point to the new_elem's vector
384 if (new_elem && (new_elem->processor_id() == processor_id()))
385 {
386 reverse_cache_t::mapped_type & points = _reverse_point_cache[new_elem];
387 points.push_back(std::make_pair(p, id));
388 }
389}

391unsigned
392DiracKernelBase::currentPointCachedID()
393{
394 reverse_cache_t::iterator it = _reverse_point_cache.find(_current_elem);
395
396 // If the current Elem is not in the cache, return invalid_uint
397 if (it == _reverse_point_cache.end())
398 return libMesh::invalid_uint;
399
400 // Do a linear search in the (hopefully small) vector of Points for this Elem
401 reverse_cache_t::mapped_type & points = it->second;
402
403 for (const auto & points_it : points)
404 {
405 // If the current_point equals the cached point, return the associated id
406 if (_current_point.relative_fuzzy_equals(points_it.first))
407 return points_it.second;
408 }
409
410 // If we made it here, we didn't find the cached point, so return invalid_uint
411 return libMesh::invalid_uint;
