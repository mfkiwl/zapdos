30DiffusionRateTemp::DiffusionRateTemp(const InputParameters & parameters)
31 : Material(parameters),
32 _diff_rate(declareProperty<Real>("diffusion_rate")),
33 _gap_length(getMaterialProperty<Real>("gap_length")),
34 _radius(getMaterialProperty<Real>("radius"))
35{
36 std::string file_name = getParam<std::string>("file_location") + "/" + "electron_temperature.txt";
37 MooseUtils::checkFileReadable(file_name);
38 const char * charPath = file_name.c_str();
39 std::ifstream myfile(charPath);
40 Real value;
41
42 std::vector<Real> reduced_field;
43 std::vector<Real> temperature;
44 if (myfile.is_open())
45 {
46 while (myfile >> value)
47 {
48 reduced_field.push_back(value);
49 myfile >> value;
50 temperature.push_back(value);
51 }
52 myfile.close();
53 }
54 else
55 mooseError("Unable to open file");
56
57 _elec_temp.setData(reduced_field, temperature);
58}
