57template <bool is_ad>
58Real
59ReactionRateEEDFTownsendLogTempl<is_ad>::computeValue()
60{
61
62 Real electron_flux_mag =
63 raw_value((-_muem[_qp] * -_grad_potential[_qp] * _r_units * std::exp(_em[_qp]) -
64 _diffem[_qp] * std::exp(_em[_qp]) * _grad_em[_qp] * _r_units)
65 .norm());
66 return raw_value(_alpha[_qp] * std::exp(_target[_qp]) * electron_flux_mag * 6.022e23);
67}
