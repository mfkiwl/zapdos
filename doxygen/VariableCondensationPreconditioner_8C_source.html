40InputParameters
41VariableCondensationPreconditioner::validParams()
42{
43 InputParameters params = MoosePreconditioner::validParams();
44
45 params.addClassDescription(
46 "Varialble condensation preconditioner (VCP) condenses out specified variable(s) "
47 "from the Jacobian matrix and produces a system of equations with less unkowns to "
48 "be solved by the underlying preconditioners.");
49
50 params.addParam<std::vector<NonlinearVariableName>>(
51 "coupled_groups",
52 {},
53 "List multiple space separated groups of comma separated variables. "
54 "Off-diagonal jacobians will be generated for all pairs within a group.");
55
56 params.addParam<bool>(
57 "is_lm_coupling_diagonal",
58 false,
59 "Set to true if you are sure the coupling matrix between Lagrange multiplier variable and "
60 "the coupled primal variable is strict diagonal. This will speedup the linear solve. "
61 "Otherwise set to false to ensure linear solve accuracy.");
62 params.addParam<bool>(
63 "adaptive_condensation",
64 true,
65 "By default VCP will check the Jacobian and only condense the rows with zero diagonals. Set "
66 "to false if you want to condense out all the specified variable dofs.");
67 params.addRequiredParam<std::vector<std::string>>("preconditioner", "Preconditioner type.");
68 params.addRequiredParam<std::vector<std::string>>(
69 "lm_variable",
70 "Name of the variable(s) that is to be condensed out. Usually "
71 "this will be the Lagrange multiplier variable(s).");
72 params.addRequiredParam<std::vector<std::string>>(
73 "primary_variable",
74 "Name of the variable(s) that couples with the variable(s) specified in the `variable` "
75 "block. Usually this is the primary variable that the Lagrange multiplier correspond to.");
76 return params;
77}

79VariableCondensationPreconditioner::VariableCondensationPreconditioner(
80 const InputParameters & params)
81 : MoosePreconditioner(params),
82 Preconditioner<Number>(MoosePreconditioner::_communicator),
83 _nl(_fe_problem.getNonlinearSystemBase(_nl_sys_num)),
84 _mesh(_fe_problem.mesh()),
85 _dofmap(_nl.system().get_dof_map()),
86 _is_lm_coupling_diagonal(getParam<bool>("is_lm_coupling_diagonal")),
87 _adaptive_condensation(getParam<bool>("adaptive_condensation")),
88 _n_vars(_nl.nVariables()),
89 _lm_var_names(getParam<std::vector<std::string>>("lm_variable")),
90 _primary_var_names(getParam<std::vector<std::string>>("primary_variable")),
91 _D(std::make_unique<PetscMatrix<Number>>(MoosePreconditioner::_communicator)),
92 _M(std::make_unique<PetscMatrix<Number>>(MoosePreconditioner::_communicator)),
93 _K(std::make_unique<PetscMatrix<Number>>(MoosePreconditioner::_communicator)),
94 _dinv(nullptr),
95 _J_condensed(std::make_unique<PetscMatrix<Number>>(MoosePreconditioner::_communicator)),
96 _x_hat(NumericVector<Number>::build(MoosePreconditioner::_communicator)),
97 _y_hat(NumericVector<Number>::build(MoosePreconditioner::_communicator)),
98 _primary_rhs_vec(NumericVector<Number>::build(MoosePreconditioner::_communicator)),
99 _lm_sol_vec(NumericVector<Number>::build(MoosePreconditioner::_communicator)),
100 _need_condense(true),
101 _init_timer(registerTimedSection("init", 2)),
102 _apply_timer(registerTimedSection("apply", 1))
103{
104 if (_lm_var_names.size() != _primary_var_names.size())
105 paramError("coupled_variable", "coupled_variable should have the same size as the variable.");
106
107 if (!_mesh.getMesh().is_replicated())
108 mooseError("The VariableCondensationPreconditioner cannot be used with DistributedMesh");
109
110 // get variable ids from the variable names
111 for (const auto & var_name : _lm_var_names)
112 {
113 if (!_nl.system().has_variable(var_name))
114 paramError("variable ", var_name, " does not exist in the system");
115 const unsigned int id = _nl.system().variable_number(var_name);
116 _lm_var_ids.push_back(id);
117 }
118
119 // get coupled variable ids from the coupled variable names
120 for (const auto & var_name : _primary_var_names)
121 {
122 if (!_nl.system().has_variable(var_name))
123 paramError("coupled_variable ", var_name, " does not exist in the system");
124 const unsigned int id = _nl.system().variable_number(var_name);
125 _primary_var_ids.push_back(id);
126 }
127
128 // PC type
129 const std::vector<std::string> & pc_type = getParam<std::vector<std::string>>("preconditioner");
130 if (pc_type.size() > 1)
131 mooseWarning("We only use one preconditioner type in VCP, the ",
132 pc_type[0],
133 " preconditioner is utilized.");
134 _pre_type = Utility::string_to_enum<PreconditionerType>(pc_type[0]);
135
136 // The following obtains and sets the coupling matrix.
137 // TODO: This part can be refactored together with what are in other classes, e.g.,
138 // PhysicsBasedPreconditioner
139 std::unique_ptr<CouplingMatrix> cm = std::make_unique<CouplingMatrix>(_n_vars);
140 const bool full = getParam<bool>("full");
141
142 if (!full)
143 {
144 // put 1s on diagonal
145 for (const auto i : make_range(_n_vars))
146 (*cm)(i, i) = 1;
147
148 // off-diagonal entries from the off_diag_row and off_diag_column parameters
149 std::vector<std::vector<unsigned int>> off_diag(_n_vars);
150 if (isParamValid("off_diag_row") && isParamValid("off_diag_column"))
151
152 for (const auto i : index_range(getParam<std::vector<NonlinearVariableName>>("off_diag_row")))
153 {
154 const unsigned int row =
155 _nl.getVariable(0, getParam<std::vector<NonlinearVariableName>>("off_diag_row")[i])
156 .number();
157 const unsigned int column =
158 _nl.getVariable(0, getParam<std::vector<NonlinearVariableName>>("off_diag_column")[i])
159 .number();
160 (*cm)(row, column) = 1;
161 }
162
163 // off-diagonal entries from the coupled_groups parameters
164 for (const auto & coupled_group :
165 getParam<std::vector<NonlinearVariableName>>("coupled_groups"))
166 {
167 std::vector<NonlinearVariableName> vars;
168 MooseUtils::tokenize<NonlinearVariableName>(coupled_group, vars, 1, ",");
169 for (unsigned int j : index_range(vars))
170 for (unsigned int k = j + 1; k < vars.size(); ++k)
171 {
172 const unsigned int row = _nl.getVariable(0, vars[j]).number();
173 const unsigned int column = _nl.getVariable(0, vars[k]).number();
174 (*cm)(row, column) = 1;
175 (*cm)(column, row) = 1;
176 }
177 }
178 }
179 else
180 {
181 for (unsigned int i = 0; i < _n_vars; i++)
182 for (unsigned int j = 0; j < _n_vars; j++)
183 (*cm)(i, j) = 1;
184 }
185
186 setCouplingMatrix(std::move(cm));
187
188 _nl.attachPreconditioner(this);
189}

193void
194VariableCondensationPreconditioner::getDofToCondense()
195{
196 // clean the containers if we want to update the dofs
197 _global_lm_dofs.clear();
198 _lm_dofs.clear();
199 _global_primary_dofs.clear();
200 _primary_dofs.clear();
201 _map_global_lm_primary.clear();
202 _map_global_primary_order.clear();
203
204 // TODO: this might not work for distributed mesh and needs to be improved
205 NodeRange * active_nodes = _mesh.getActiveNodeRange();
206
207 // loop through the variable ids
208 std::vector<dof_id_type> di, cp_di;
209 for (const auto & vn : index_range(_lm_var_ids))
210 for (const auto & node : *active_nodes)
211 {
212 di.clear();
213 cp_di.clear();
214 const auto var_id = _lm_var_ids[vn];
215 // get coupled variable id
216 const auto cp_var_id = _primary_var_ids[vn];
217 // get var and cp_var dofs associated with this node
218 _dofmap.dof_indices(node, di, var_id);
219 // skip when di is empty
220 if (di.empty())
221 continue;
222 _dofmap.dof_indices(node, cp_di, cp_var_id);
223 if (cp_di.size() != di.size())
224 mooseError("variable and coupled variable do not have the same number of dof on node ",
225 node->id(),
226 ".");
227 for (const auto & i : index_range(di))
228 {
229 // when we have adaptive condensation, skip when di does not contain any indices in
230 // _zero_rows
231 if (std::find(_zero_rows.begin(), _zero_rows.end(), di[i]) == _zero_rows.end() &&
232 _adaptive_condensation)
233 break;
234 _global_lm_dofs.push_back(di[i]);
235 if (_dofmap.local_index(di[i]))
236 _lm_dofs.push_back(di[i]);
237
238 // save the corresponding coupled dof indices
239 _global_primary_dofs.push_back(cp_di[i]);
240 if (_dofmap.local_index(cp_di[i]))
241 _primary_dofs.push_back(cp_di[i]);
242 _map_global_lm_primary.insert(std::make_pair(di[i], cp_di[i]));
243 }
244 }
245
246 // check if we endup with none dof to condense
247 if (_global_lm_dofs.empty())
248 {
249 _need_condense = false;
250 _console << std::endl
251 << "The variable(s) provided do not have a saddle-point character at this step. VCP "
252 "will continue without condensing the dofs."
253 << std::endl
254 << std::endl;
255 }
256 else
257 _need_condense = true;
258
259 std::sort(_global_lm_dofs.begin(), _global_lm_dofs.end());
260 std::sort(_lm_dofs.begin(), _lm_dofs.end());
261
262 std::sort(_global_primary_dofs.begin(), _global_primary_dofs.end());
263 std::sort(_primary_dofs.begin(), _primary_dofs.end());
264
265 for (const auto & i : index_range(_global_lm_dofs))
266 {
267 auto it = _map_global_lm_primary.find(_global_lm_dofs[i]);
268 mooseAssert(it != _map_global_lm_primary.end(), "Index does not exist in the map.");
269 _map_global_primary_order.insert(std::make_pair(it->second, i));
270 }
271}

273void
274VariableCondensationPreconditioner::getDofColRow()
275{
276 // clean the containers if we want to update the dofs
277 _global_rows.clear();
278 _rows.clear();
279 _global_cols.clear();
280 _cols.clear();
281 _global_rows_to_idx.clear();
282 _rows_to_idx.clear();
283 _global_cols_to_idx.clear();
284 _cols_to_idx.clear();
285 // row: all without primary variable dofs
286 // col: all without lm variable dofs
287 for (dof_id_type i = 0; i < _dofmap.n_dofs(); ++i)
288 {
289 if (std::find(_global_primary_dofs.begin(), _global_primary_dofs.end(), i) !=
290 _global_primary_dofs.end())
291 continue;
292
293 _global_rows.push_back(i);
294 _global_rows_to_idx.insert(std::make_pair(i, _global_rows.size() - 1));
295 if (_dofmap.local_index(i))
296 {
297 _rows.push_back(i);
298 _rows_to_idx.insert(std::make_pair(i, _global_rows_to_idx[i]));
299 }
300
301 // ensure the lm and primary correspondance, so that the condensed Jacobian has non-zero
302 // diagonal if the dof corresponds to the lm variable, then find the corresponding primary
303 // variable dof and add to _global_cols
304 if (_map_global_lm_primary.find(i) != _map_global_lm_primary.end())
305 {
306 auto primary_idx = _map_global_lm_primary[i];
307 _global_cols.push_back(primary_idx);
308 _global_cols_to_idx.insert(std::make_pair(primary_idx, _global_cols.size() - 1));
309
310 if (_dofmap.local_index(primary_idx))
311 {
312 _cols.push_back(primary_idx);
313 _cols_to_idx.insert(std::make_pair(primary_idx, _global_cols_to_idx[primary_idx]));
314 }
315 }
316 else // if the dof does not correspond to the lm nor primary varialble, just add to _global_cols
317 {
318 _global_cols.push_back(i);
319 _global_cols_to_idx.insert(std::make_pair(i, _global_cols.size() - 1));
320
321 if (_dofmap.local_index(i))
322 {
323 _cols.push_back(i);
324 _cols_to_idx.insert(std::make_pair(i, _global_cols_to_idx[i]));
325 }
326 }
327 }

342void
343VariableCondensationPreconditioner::condenseSystem()
344{
345 PetscErrorCode ierr = 0;
346
347 // extract _M from the original matrix
348 _matrix->create_submatrix(*_M, _rows, _lm_dofs);
349
350 // get the row associated with the coupled primary variable
351 _K->init(_global_primary_dofs.size(), _global_cols.size(), _primary_dofs.size(), _cols.size());
352 // Note: enabling nonzero allocation may be expensive. Improved memeory pre-allocation will be
353 // investigated in the future
354 ierr = MatSetOption(_K->mat(), MAT_NEW_NONZERO_ALLOCATION_ERR, PETSC_FALSE);
355 LIBMESH_CHKERR(ierr);
356 // here the _global_cols may not be sorted
357 _matrix->create_submatrix_nosort(*_K, _global_primary_dofs, _global_cols);
358
359 _matrix->create_submatrix(*_D, _primary_dofs, _lm_dofs);
360
361 // clean dinv
362 if (_dinv)
363 {
364 ierr = MatDestroy(&_dinv);
365 LIBMESH_CHKERR(ierr);
366 _dinv = nullptr;
367 }
368
369 // Compute inverse of D
370 if (_is_lm_coupling_diagonal)
371 // when _D is strictly diagonal, we only need to compute the reciprocal number of the diagonal
372 // entries
373 computeDInverseDiag(_dinv);
374 else
375 // for general cases when _D is not necessarily strict diagonal, we compute the inverse of _D
376 // using LU
377 computeDInverse(_dinv);
378
379 Mat MdinvK;
380 // calculate MdinvK
381 ierr = MatMatMatMult(_M->mat(), _dinv, _K->mat(), MAT_INITIAL_MATRIX, PETSC_DEFAULT, &MdinvK);
382 LIBMESH_CHKERR(ierr);
383 PetscMatrix<Number> MDinv_K(MdinvK, MoosePreconditioner::_communicator);
384
385 // Preallocate memory for _J_condensed
386 // memory info is obtained from _matrix and MDinv_K
387 // indices are from _global_rows, _global_cols
388 auto pc_original_mat = cast_ptr<PetscMatrix<Number> *>(_matrix);
389 preallocateCondensedJacobian(
390 *_J_condensed, *pc_original_mat, _rows, _cols, _global_rows, _global_cols, MDinv_K);
391
392 // Extract unchanged parts from _matrix and add changed parts (MDinv_K) to _J_condensed
393 computeCondensedJacobian(*_J_condensed, *pc_original_mat, _global_rows, MDinv_K);
394
395 // Destroy MdinvK here otherwise we will have memory leak
396 ierr = MatDestroy(&MdinvK);
397 LIBMESH_CHKERR(ierr);
398}

400void
401VariableCondensationPreconditioner::computeCondensedJacobian(PetscMatrix<Number> & condensed_mat,
402 PetscMatrix<Number> & original_mat,
403 const std::vector<dof_id_type> & grows,
404 PetscMatrix<Number> & block_mat)
405{
406 PetscErrorCode ierr = 0;
407
408 // obtain entries from the original matrix
409 PetscInt pc_ncols = 0, block_ncols = 0;
410 const PetscInt *pc_cols, *block_cols;
411 const PetscScalar *pc_vals, *block_vals;
412
413 // containers for the data
414 std::vector<PetscInt> sub_cols;
415 std::vector<PetscScalar> sub_vals;
416
417 for (const auto & i : index_range(grows))
418 {
419 PetscInt sub_rid[] = {static_cast<PetscInt>(i)};
420 PetscInt rid = grows[i];
421 if (grows[i] >= original_mat.row_start() && grows[i] < original_mat.row_stop())
422 {
423 // get one row of data from the original matrix
424 ierr = MatGetRow(original_mat.mat(), rid, &pc_ncols, &pc_cols, &pc_vals);
425 LIBMESH_CHKERR(ierr);
426 // get corresponding row of data from the block matrix
427 ierr = MatGetRow(block_mat.mat(), i, &block_ncols, &block_cols, &block_vals);
428 LIBMESH_CHKERR(ierr);
429 // extract data from certain cols, subtract the value from the block mat, and save the indices
430 // and entries sub_cols and sub_vals
431 // First, save the submatrix col index and value as a map
432 std::map<PetscInt, PetscScalar> pc_col_map;
433 for (PetscInt pc_idx = 0; pc_idx < pc_ncols; pc_idx++)
434 {
435 // save only if the col exists in the condensed matrix
436 if (_global_cols_to_idx.find(pc_cols[pc_idx]) != _global_cols_to_idx.end())
437 pc_col_map.insert(std::make_pair(_global_cols_to_idx[pc_cols[pc_idx]], pc_vals[pc_idx]));
438 }
439 // Second, check the block cols and calculate new entries for the condensed system
440 for (PetscInt block_idx = 0; block_idx < block_ncols; block_idx++)
441 {
442 PetscInt block_col = block_cols[block_idx];
443 PetscScalar block_val = block_vals[block_idx];
444 // if the block mat has nonzero at the same column, subtract value
445 // otherwise, create a new key and save the negative value from the block matrix
446 if (pc_col_map.find(block_col) != pc_col_map.end())
447 pc_col_map[block_col] -= block_val;
448 else
449 pc_col_map[block_col] = -block_val;
450 }
451
452 // Third, save keys in the sub_cols and values in the sub_vals
453 for (std::map<PetscInt, PetscScalar>::iterator it = pc_col_map.begin();
454 it != pc_col_map.end();
455 ++it)
456 {
457 sub_cols.push_back(it->first);
458 sub_vals.push_back(it->second);
459 }
460
461 // Then, set values
462 ierr = MatSetValues(condensed_mat.mat(),
463 1,
464 sub_rid,
465 sub_vals.size(),
466 sub_cols.data(),
467 sub_vals.data(),
468 INSERT_VALUES);
469 LIBMESH_CHKERR(ierr);
470 ierr = MatRestoreRow(original_mat.mat(), rid, &pc_ncols, &pc_cols, &pc_vals);
471 LIBMESH_CHKERR(ierr);
472 ierr = MatRestoreRow(block_mat.mat(), i, &block_ncols, &block_cols, &block_vals);
473 LIBMESH_CHKERR(ierr);
474 // clear data for this row
475 sub_cols.clear();
476 sub_vals.clear();
477 }
478 }
479 condensed_mat.close();
480}

482void
483VariableCondensationPreconditioner::preallocateCondensedJacobian(
484 PetscMatrix<Number> & condensed_mat,
485 PetscMatrix<Number> & original_mat,
486 const std::vector<dof_id_type> & rows,
487 const std::vector<dof_id_type> & cols,
488 const std::vector<dof_id_type> & grows,
489 const std::vector<dof_id_type> & gcols,
490 PetscMatrix<Number> & block_mat)
491{
492 // quantities from the original matrix and the block matrix
493 PetscInt ncols = 0, block_ncols = 0;
494 const PetscInt * col_vals;
495 const PetscInt * block_col_vals;
496 const PetscScalar * vals;
497 const PetscScalar * block_vals;
498
499 std::vector<PetscInt> block_cols_to_org; // stores the nonzero column indices of the block
500 // matrix w.r.t original matrix
501 std::vector<PetscInt>
502 merged_cols; // stores the nonzero column indices estimate of the condensed matrix
503
504 // number of nonzeros in each row of the DIAGONAL and OFF-DIAGONAL portion of the local
505 // condensed matrix
506 std::vector<dof_id_type> n_nz, n_oz;
507
508 PetscErrorCode ierr = 0;
509
510 // Get number of nonzeros from original_mat and block_mat for each row
511 for (const auto & row_id : _rows)
512 {
513 // get number of non-zeros in the original matrix
514 ierr = MatGetRow(original_mat.mat(), row_id, &ncols, &col_vals, &vals);
515 LIBMESH_CHKERR(ierr);
516
517 // get number of non-zeros in the block matrix
518 dof_id_type block_row_id; // row id in the block matrix
519
520 if (_global_rows_to_idx.find(row_id) != _global_rows_to_idx.end())
521 block_row_id = _global_rows_to_idx[row_id];
522 else
523 mooseError("DoF ", row_id, " does not exist in the rows of condensed_mat");
524
525 ierr = MatGetRow(block_mat.mat(), block_row_id, &block_ncols, &block_col_vals, &block_vals);
526 LIBMESH_CHKERR(ierr);
527
528 // make sure the block index is transformed in terms of the original mat
529 block_cols_to_org.clear();
530 for (PetscInt i = 0; i < block_ncols; ++i)
531 {
532 auto idx = gcols[block_col_vals[i]];
533 block_cols_to_org.push_back(idx);
534 }
535
536 // Now store nonzero column indices for the condensed Jacobian
537 // merge `col_vals` and `block_cols_to_org` and save the common indices in `merged_cols`.
538 mergeArrays(col_vals, block_cols_to_org.data(), ncols, block_ncols, merged_cols);
539
540 ierr = MatRestoreRow(block_mat.mat(), block_row_id, &block_ncols, &block_col_vals, &block_vals);
541 LIBMESH_CHKERR(ierr);
542
543 ierr = MatRestoreRow(original_mat.mat(), row_id, &ncols, &col_vals, &vals);
544 LIBMESH_CHKERR(ierr);
545
546 // Count the nnz for DIAGONAL and OFF-DIAGONAL parts
547 PetscInt row_n_nz = 0, row_n_oz = 0;
548 for (const auto & merged_col : merged_cols)
549 {
550 // find corresponding index in the block mat and skip the cols that do not exist in the
551 // condensed system
552 if (_global_cols_to_idx.find(merged_col) == _global_cols_to_idx.end())
553 continue;
554
555 dof_id_type col_idx = _global_cols_to_idx[merged_col];
556 // find the corresponding row index
557 dof_id_type row_idx = grows[col_idx];
558 // check whether the index is local;
559 // yes - DIAGONAL, no - OFF-DIAGONAL
560 if (_rows_to_idx.find(row_idx) != _rows_to_idx.end())
561 row_n_nz++;
562 else
563 row_n_oz++;
564 }
565
566 n_nz.push_back(cast_int<dof_id_type>(row_n_nz));
567 n_oz.push_back(cast_int<dof_id_type>(row_n_oz));
568 }
569 // Then initialize and allocate memory for the condensed system matrix
570 condensed_mat.init(grows.size(), gcols.size(), rows.size(), cols.size(), n_nz, n_oz);
571}

573void
574VariableCondensationPreconditioner::mergeArrays(const PetscInt * a,
575 const PetscInt * b,
576 const PetscInt & na,
577 const PetscInt & nb,
578 std::vector<PetscInt> & c)
579{
580 c.clear();
581
582 // use map to store unique elements.
583 std::map<PetscInt, bool> mp;
584
585 // Inserting values to a map.
586 for (const auto & i : make_range(na))
587 mp[a[i]] = true;
588
589 for (const auto & i : make_range(nb))
590 mp[b[i]] = true;
591
592 // Save the merged values to c, if only the value also exist in gcols
593 for (const auto & i : mp)
594 c.push_back(i.first);
595}

598VariableCondensationPreconditioner::setup()
599{
600 if (_adaptive_condensation)
601 findZeroDiagonals(*_matrix, _zero_rows);
602
603 // save dofs that are to be condensed out
604 getDofToCondense();
605
606 // solve the condensed system only when needed, otherwise solve the original system
607 if (_need_condense)
608 {
609 // get condensed dofs for rows and cols
610 getDofColRow();
611
612 condenseSystem();
613
614 // make sure diagonal entries are not empty
615 for (const auto & i : make_range(_J_condensed->row_start(), _J_condensed->row_stop()))
616 _J_condensed->add(i, i, 0.0);
617 _J_condensed->close();
618
619 _preconditioner->set_matrix(*_J_condensed);
620 }
621 else
622 _preconditioner->set_matrix(*_matrix);
623
624 _preconditioner->set_type(_pre_type);
625 _preconditioner->init();
626}

652 NumericVector<Number> & x)
653{
654 Mat mdinv;
655 PetscErrorCode ierr = 0;
656 // calculate mdinv
657 ierr = MatMatMult(_M->mat(), _dinv, MAT_INITIAL_MATRIX, PETSC_DEFAULT, &mdinv);
658 LIBMESH_CHKERR(ierr);
659 PetscMatrix<Number> MDinv(mdinv, MoosePreconditioner::_communicator);
660
661 _x_hat->init(_J_condensed->n(), _J_condensed->local_n(), false, PARALLEL);
662 _y_hat->init(_J_condensed->m(), _J_condensed->local_m(), false, PARALLEL);
663
664 x.create_subvector(*_x_hat, _global_cols);
665 y.create_subvector(*_y_hat, _global_rows);
666
667 _primary_rhs_vec->init(MDinv.n(), MDinv.local_n(), false, PARALLEL);
668
669 std::unique_ptr<NumericVector<Number>> mdinv_primary_rhs(
670 NumericVector<Number>::build(MoosePreconditioner::_communicator));
671 mdinv_primary_rhs->init(MDinv.m(), MDinv.local_m(), false, PARALLEL);
672
673 // get _primary_rhs_vec from the original y
674 y.create_subvector(*_primary_rhs_vec, _global_primary_dofs);
675
676 MDinv.vector_mult(*mdinv_primary_rhs, *_primary_rhs_vec);
677 mdinv_primary_rhs->close();
678
679 (*_y_hat) -= (*mdinv_primary_rhs);
680
681 _y_hat->close();
682 _x_hat->close();
683
684 ierr = MatDestroy(&mdinv);
685 LIBMESH_CHKERR(ierr);
686}

736 std::vector<dof_id_type> & indices)
737{
738 indices.clear();
739 IS zerodiags, zerodiags_all;
740 PetscErrorCode ierr;
741 const PetscInt * petsc_idx;
742 PetscInt nrows;
743 // make sure we have a PETSc matrix

746 LIBMESH_CHKERR(ierr);

748 ierr = ISAllGather(zerodiags, &zerodiags_all);

753 LIBMESH_CHKERR(ierr);
754
755 for (PetscInt i = 0; i < nrows; ++i)
756 indices.push_back(petsc_idx[i]);
757
758 ISRestoreIndices(zerodiags_all, &petsc_idx);
759 LIBMESH_CHKERR(ierr);
760 ierr = ISDestroy(&zerodiags);
761 LIBMESH_CHKERR(ierr);
762 ierr = ISDestroy(&zerodiags_all);
763 LIBMESH_CHKERR(ierr);

777void
778VariableCondensationPreconditioner::computeDInverse(Mat & dinv)
779{
780 PetscErrorCode ierr;
781 Mat F, I, dinv_dense;
782 IS perm, iperm;
783 MatFactorInfo info;
784
785 ierr = MatCreateDense(
786 PETSC_COMM_WORLD, _D->local_n(), _D->local_m(), _D->n(), _D->m(), NULL, &dinv_dense);
787 LIBMESH_CHKERR(ierr);
788
789 // Create an identity matrix as the right-hand-side
790 ierr = MatCreateDense(PETSC_COMM_WORLD, _D->local_m(), _D->local_m(), _D->m(), _D->m(), NULL, &I);
791 LIBMESH_CHKERR(ierr);
792
793 for (unsigned int i = 0; i < _D->m(); ++i)
794 {
795 ierr = MatSetValue(I, i, i, 1.0, INSERT_VALUES);
796 LIBMESH_CHKERR(ierr);
797 }
798
799 ierr = MatAssemblyBegin(I, MAT_FINAL_ASSEMBLY);
800 LIBMESH_CHKERR(ierr);
801 ierr = MatAssemblyEnd(I, MAT_FINAL_ASSEMBLY);
802 LIBMESH_CHKERR(ierr);
803
804 // Factorize D
805 ierr = MatGetOrdering(_D->mat(), MATORDERINGND, &perm, &iperm);
806 LIBMESH_CHKERR(ierr);
807
808 ierr = MatFactorInfoInitialize(&info);
809 LIBMESH_CHKERR(ierr);
810
811 ierr = MatGetFactor(_D->mat(), MATSOLVERSUPERLU_DIST, MAT_FACTOR_LU, &F);
812 LIBMESH_CHKERR(ierr);
813
814 ierr = MatLUFactorSymbolic(F, _D->mat(), perm, iperm, &info);
815 LIBMESH_CHKERR(ierr);
816
817 ierr = MatLUFactorNumeric(F, _D->mat(), &info);
818 LIBMESH_CHKERR(ierr);
819
820 // Solve for Dinv
821 ierr = MatMatSolve(F, I, dinv_dense);
822 LIBMESH_CHKERR(ierr);
823
824 ierr = MatAssemblyBegin(dinv_dense, MAT_FINAL_ASSEMBLY);
825 LIBMESH_CHKERR(ierr);
826 ierr = MatAssemblyEnd(dinv_dense, MAT_FINAL_ASSEMBLY);
827 LIBMESH_CHKERR(ierr);
828
829 // copy value to dinv
830 ierr = MatConvert(dinv_dense, MATAIJ, MAT_INITIAL_MATRIX, &dinv);
831 LIBMESH_CHKERR(ierr);
832
833 ierr = MatDestroy(&dinv_dense);
834 LIBMESH_CHKERR(ierr);
835
836 ierr = MatDestroy(&I);
837 LIBMESH_CHKERR(ierr);
838 ierr = MatDestroy(&F);
839 LIBMESH_CHKERR(ierr);
840 ierr = ISDestroy(&perm);
841 LIBMESH_CHKERR(ierr);
842 ierr = ISDestroy(&iperm);
843 LIBMESH_CHKERR(ierr);
844}

846void
847VariableCondensationPreconditioner::computeDInverseDiag(Mat & dinv)
848{
849 PetscErrorCode ierr;
850 auto diag_D = NumericVector<Number>::build(MoosePreconditioner::_communicator);
851 // Initialize dinv
852 ierr = MatCreateAIJ(
853 PETSC_COMM_WORLD, _D->local_n(), _D->local_m(), _D->n(), _D->m(), 1, NULL, 0, NULL, &dinv);
854 LIBMESH_CHKERR(ierr);
855 // Allocate storage
856 diag_D->init(_D->m(), _D->local_m(), false, PARALLEL);
857 // Fill entries
858 for (const auto & i : make_range(_D->row_start(), _D->row_stop()))
859 {
860 auto it = _map_global_primary_order.find(_global_primary_dofs[i]);
861 mooseAssert(it != _map_global_primary_order.end(), "Index does not exist in the map.");
862 diag_D->set(it->second, (*_D)(i, it->second));
863 }
864
865 for (const auto & i : make_range(_D->row_start(), _D->row_stop()))
866 {
867 if (MooseUtils::absoluteFuzzyEqual((*diag_D)(i), 0.0))
868 mooseError("Trying to compute reciprocal of 0.");
869 ierr = MatSetValue(dinv,
870 i,
871 _map_global_primary_order.at(_global_primary_dofs[i]),
872 1.0 / (*diag_D)(i),
873 INSERT_VALUES);
874 LIBMESH_CHKERR(ierr);
875 }
876
877 ierr = MatAssemblyBegin(dinv, MAT_FINAL_ASSEMBLY);
878 LIBMESH_CHKERR(ierr);
879 ierr = MatAssemblyEnd(dinv, MAT_FINAL_ASSEMBLY);
880 LIBMESH_CHKERR(ierr);
881}
