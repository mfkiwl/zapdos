137{
138 if (_potential_units.compare("V") == 0)
139 _voltage_scaling = 1.;
140 else if (_potential_units.compare("kV") == 0)
141 _voltage_scaling = 1000;
142
143 std::vector<Real> actual_mean_energy;
144 std::vector<Real> alpha;
145 std::vector<Real> alphaEx;
146 std::vector<Real> alphaEl;
147 std::vector<Real> mu;
148 std::vector<Real> diff;
149
150 std::string file_name = getParam<FileName>("property_tables_file");
151 MooseUtils::checkFileReadable(file_name);
152 const char * charPath = file_name.c_str();
153 std::ifstream myfile(charPath);
154 Real value;
155
156 if (myfile.is_open())
157 {
158 while (myfile >> value)
159 {
160 actual_mean_energy.push_back(value);
161 myfile >> value;
162 alpha.push_back(value);
163 myfile >> value;
164 alphaEx.push_back(value);
165 myfile >> value;
166 alphaEl.push_back(value);
167 myfile >> value;
168 mu.push_back(value);
169 myfile >> value;
170 diff.push_back(value);
171 }
172 myfile.close();
173 }
174
175 else
176 mooseError("Unable to open file");
177
178 _alpha_interpolation.setData(actual_mean_energy, alpha);
179 _alphaEx_interpolation.setData(actual_mean_energy, alphaEx);
180 _alphaEl_interpolation.setData(actual_mean_energy, alphaEl);
181 _mu_interpolation.setData(actual_mean_energy, mu);
182 _diff_interpolation.setData(actual_mean_energy, diff);
183}

185void
186Gas::computeQpProperties()
187{
188 _massem[_qp] = 9.11e-31;
189 _massGas[_qp] = 40.0 * 1.66e-27;
190 _massArp[_qp] = 40.0 * 1.66e-27;
191 _T_gas[_qp] = _user_T_gas;
192 _p_gas[_qp] = _user_p_gas;
193 _k_boltz[_qp] = 1.38e-23;
194 _Avogadro[_qp] = 6.0221409E23;
195 if (_use_moles)
196 _n_gas[_qp] = _p_gas[_qp] / (8.3145 * _T_gas[_qp]);
197 else
198 _n_gas[_qp] = _p_gas[_qp] / (_k_boltz[_qp] * _T_gas[_qp]);
199
200 _se_coeff[_qp] = _user_se_coeff;
201 _work_function[_qp] = _user_work_function;
202 _field_enhancement[_qp] = _user_field_enhancement;
203
204 _Richardson_coefficient[_qp] = _user_Richardson_coefficient;
205 _cathode_temperature[_qp] = _user_cathode_temperature;
206
207 _se_energy[_qp] = 2. * 3. / 2.; // Emi uses 2 Volts coming off the wall (presumably for Te).
208 // Multiply by 3/2 to get mean_en
209 _e[_qp] = 1.6e-19;
210 _eps[_qp] = 8.85e-12;
211 _sgnem[_qp] = -1.;
212 _sgnmean_en[_qp] = -1.;
213 _sgnArp[_qp] = 1.;
214 _diffpotential[_qp] = _eps[_qp];
215
216 _TArp[_qp] = 300;
217
218 // With the exception of temperature/energy (generally in eV), all properties are in standard SI
219 // units unless otherwise indicated
220
221 if (_interp_trans_coeffs)
222 {
223 if (_ramp_trans_coeffs)
224 {
225 _muem[_qp].value() =
226 (std::tanh(_t / 1e-6) *
227 _mu_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) +
228 (1. - std::tanh(_t / 1e-6)) * .0352) *
229 _voltage_scaling;
230 _muem[_qp].derivatives() =
231 std::tanh(_t / 1e-6) *
232 _mu_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
233 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
234 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _voltage_scaling;
235 _diffem[_qp].value() =
236 std::tanh(_t / 1e-6) *
237 _diff_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) +
238 (1. - std::tanh(_t / 1e-6)) * .30;
239 _diffem[_qp].derivatives() =
240 std::tanh(_t / 1e-6) *
241 _diff_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
242 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
243 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
244 }
245 else
246 {
247 _muem[_qp].value() =
248 _mu_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
249 _voltage_scaling;
250 _muem[_qp].derivatives() =
251 _mu_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
252 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
253 (_mean_en[_qp].derivatives() - _em[_qp].derivatives()) * _voltage_scaling;
254 _diffem[_qp].value() =
255 _diff_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value()));
256 _diffem[_qp].derivatives() =
257 _diff_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
258 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
259 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
260 }
261 }
262 else
263 {
264 // From bolos at atmospheric pressure and an EField of 2e5 V/m
265 _muem[_qp].value() =
266 0.0352103411399 * _voltage_scaling; // units of m^2/(kV*s) if _voltage_scaling = 1000
267 _muem[_qp].derivatives() = 0.0;
268 _diffem[_qp].value() = 0.297951680159;
269 _diffem[_qp].derivatives() = 0.0;
270 }
271
272 // From Richards and Sawin, muArp*pressure = 1444 cm^2*Torr/(V*s) and diffArp*pressure = 40
273 // cm^2*Torr/s. Use pressure = 760 torr.
274 _muArp[_qp] =
275 1444. * _voltage_scaling /
276 (10000. * 760. * _p_gas[_qp] / 1.01E5); // units of m^2/(kV*s) if _voltage_scaling = 1000
277 _diffArp[_qp] = 0.004 / (760. * _p_gas[_qp] / 1.01E5); // covert to m^2 and include press
278
279 // 100 times less than electrons
280 // _muArp[_qp] = 3.52e-4;
281 // _diffArp[_qp] = 2.98e-3;
282
283 // From curve fitting with bolos
284 _iz_coeff_efield_a[_qp] = 1.43171672e-1;
285 _iz_coeff_efield_b[_qp] = 9.05925536e-1;
286 _iz_coeff_efield_c[_qp] = 3.04958892e+6;
287
288 // if (_townsend) {
289 _iz_coeff_energy_a[_qp] = 1.52165930e+8;
290 _iz_coeff_energy_b[_qp] = -2.87277596e-1;
291 _iz_coeff_energy_c[_qp] = 5.51972192e+1;
292 // }
293 // else {
294 // _iz_coeff_energy_a[_qp] = 1.43878529e-11;
295 // _iz_coeff_energy_b[_qp] = -2.70610234e-1;
296 // _iz_coeff_energy_c[_qp] = 7.64727794e+1;
297 // }
298
299 _actual_mean_energy[_qp] = std::exp(_mean_en[_qp] - _em[_qp]);
300 _alpha_iz[_qp].value() =
301 _alpha_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value()));
302 _alpha_iz[_qp].derivatives() =
303 _alpha_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
304 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
305 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
306 _alpha_ex[_qp].value() =
307 _alphaEx_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value()));
308 _alpha_ex[_qp].derivatives() =
309 _alphaEx_interpolation.sampleDerivative(std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
310 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
311 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
312 if (_interp_elastic_coeff)
313 {
314 _alpha_el[_qp].value() =
315 _alphaEl_interpolation.sample(std::exp(_mean_en[_qp].value() - _em[_qp].value()));
316 _alpha_el[_qp].derivatives() = _alphaEl_interpolation.sampleDerivative(
317 std::exp(_mean_en[_qp].value() - _em[_qp].value())) *
318 std::exp(_mean_en[_qp].value() - _em[_qp].value()) *
319 (_mean_en[_qp].derivatives() - _em[_qp].derivatives());
320 }
321 else
322 {
323 _alpha_el[_qp].value() = 5e8;
324 _alpha_el[_qp].derivatives() = 0.0;
325 }
326
327 _el_coeff_energy_a[_qp] = 1.60638169e-13;
328 _el_coeff_energy_b[_qp] = 3.17917979e-1;
329 _el_coeff_energy_c[_qp] = 4.66301096;
330
331 _N_A[_qp] = 6.02e23;
332 _Ar[_qp] = 1.01e5 / (300 * 1.38e-23);
333 _Eiz[_qp] = 15.76;
334 _Eex[_qp] = 11.5;
335
336 // From Hagelaar: The below approximations can be derived assumption Maxwell EEDF, const
337 // momentum-transfer frequency, and constant kinetic pressure.
338 _mumean_en[_qp].value() = 5.0 / 3.0 * _muem[_qp].value();
339 _diffmean_en[_qp].value() = 5.0 / 3.0 * _diffem[_qp].value();
340 if (_interp_trans_coeffs)
341 {
342 _mumean_en[_qp].derivatives() = 5.0 / 3.0 * _muem[_qp].derivatives();
343 _diffmean_en[_qp].derivatives() = 5.0 / 3.0 * _diffem[_qp].derivatives();
344 }
345 else
346 {
347 _mumean_en[_qp].derivatives() = 0.0;
348 _diffmean_en[_qp].derivatives() = 0.0;
349 }
350
351 _rate_coeff_elastic[_qp] = 1e-13;
352
353 _TemVolts[_qp] = 2. / 3. * std::exp(_mean_en[_qp] - _em[_qp]);
354 _Tem[_qp] = _e[_qp] * _TemVolts[_qp] / _k_boltz[_qp];
355
356 _kiz[_qp] = 2.34e-14 * std::pow(_TemVolts[_qp], .59) * std::exp(-17.44 / _TemVolts[_qp]);
357 _kex[_qp] = 2.48e-14 * std::pow(_TemVolts[_qp], .33) * std::exp(-12.78 / _TemVolts[_qp]);
358 _kel[_qp].value() = 1e-13; // Approximate elastic rate coefficient
359 _kel[_qp].derivatives() = 0.;
360 if (_use_moles)
361 {
362 _kiz[_qp] = _kiz[_qp] * _N_A[_qp];
363 _kex[_qp] = _kex[_qp] * _N_A[_qp];
364 _kel[_qp] = _kel[_qp] * _N_A[_qp];
365 }
366}
