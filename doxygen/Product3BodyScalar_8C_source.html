55Real
56Product3BodyScalar::computeQpJacobian()
57{
58 Real power, eq_u_mult, gas_mult;
59 power = 0.0;
60 eq_u_mult = 1.0;
61 gas_mult = 1.0;
62
63 if (isCoupledScalar("v") && _v_eq_u)
64 {
65 power += 1.0;
66 eq_u_mult = _v[_i];
67 }
68 else
69 gas_mult *= _v[_i];
70
71 if (isCoupledScalar("w") && _w_eq_u)
72 {
73 power += 1.0;
74 eq_u_mult = _w[_i];
75 }
76 else
77 gas_mult *= _w[_i];
78
79 if (isCoupledScalar("x") && _x_eq_u)
80 {
81 power += 1.0;
82 eq_u_mult = _x[_i];
83 }
84 else
85 gas_mult *= _x[_i];
86
87 return -_stoichiometric_coeff * _rate_coefficient[_i] * gas_mult * power *
88 std::pow(eq_u_mult, power - 1);
89}

91Real
92Product3BodyScalar::computeQpOffDiagJacobian(unsigned int jvar)
93{
94 Real power, deriv_factor, other_factor;
95 power = 0;
96 other_factor = 1;
97 deriv_factor = 1;
98
99 if (jvar == _v_var && !_v_eq_u)
100 {
101 power += 1;
102 deriv_factor = _v[_i];
103 }
104 else
105 other_factor *= _v[_i];
106
107 if (jvar == _w_var && !_w_eq_u)
108 {
109 power += 1;
110 deriv_factor = _w[_i];
111 }
112 else
113 other_factor *= _w[_i];
114
115 if (jvar == _x_var && !_x_eq_u)
116 {
117 power += 1;
118 deriv_factor = _x[_i];
119 }
120 else
121 other_factor *= _x[_i];
122
123 return -_stoichiometric_coeff * _rate_coefficient[_i] * other_factor * power *
124 std::pow(deriv_factor, power - 1);
125}
