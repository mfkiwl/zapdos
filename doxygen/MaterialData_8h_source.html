238template <typename T, bool is_ad>
239inline bool
240MaterialData::haveGenericProperty(const std::string & prop_name) const
241{
242 if (!hasProperty(prop_name))
243 return false;
244
245 const auto prop_id = getPropertyId(prop_name);
246 // the property id exists, but the property was not created in this instance of the material type
247 if (prop_id >= props(0).size())
248 return false;
249
250 const PropertyValue * const base_prop = props(0).queryValue(prop_id);
251 return dynamic_cast<const GenericMaterialProperty<T, is_ad> *>(base_prop) != nullptr;
252}

254template <typename T, bool is_ad, bool declare>
255GenericMaterialProperty<T, is_ad> &
256MaterialData::getPropertyHelper(const std::string & prop_name,
257 const unsigned int state,
258 const MooseObject & requestor)
259{
260 if constexpr (is_ad)
261 mooseAssert(state == 0, "Cannot request/declare AD properties for states other than zero");
262 if constexpr (declare)
263 mooseAssert(state == 0, "Cannot declare properties for states other than zero");
264
265 // Register/get the ID of the property
266 const auto prop_id = addPropertyHelper(prop_name, state);
267
268 // Initialize the states that we need
269 const auto size = prop_id + 1;
270 for (const auto state_i : make_range(state + 1))
271 {
272 auto & entry = props(state_i);
273 if (entry.size() < size)
274 entry.resize(size, {});
275 if (!entry.hasValue(prop_id))
276 {
277 std::unique_ptr<PropertyValue> value =
278 state_i == 0 ? std::make_unique<GenericMaterialProperty<T, is_ad>>(prop_id)
279 : _props[0][prop_id].clone(0);
280 entry.setPointer(prop_id, std::move(value), {});
281 }
282 }
283
284 // Should be available now
285 auto & base_prop = props(state)[prop_id];
286
287 // In the event that this property was already declared/requested, make sure
288 // that the types are consistent
289 auto prop = dynamic_cast<GenericMaterialProperty<T, is_ad> *>(&base_prop);
290 if (!prop)
291 {
292 constexpr std::string_view action = declare ? "declared" : "requested";
293 constexpr auto is_ad_to_str = [](const bool is_ad_bool)
294 { return std::string_view(is_ad_bool ? "AD" : "non-AD"); };
295 constexpr std::string_view ad_type = is_ad_to_str(is_ad);
296
297 std::stringstream error;
298 error << "The " << action << " " << ad_type << " "
299 << "material property '" + prop_name + "' of type '" << MooseUtils::prettyCppType<T>()
300 << "'\nis already retrieved or declared as a " << is_ad_to_str(base_prop.isAD())
301 << " property of type '" << base_prop.type() << "'.";
302 mooseErrorHelper(requestor, error.str());
303 }
304
305 return *prop;
306}
