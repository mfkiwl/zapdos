40ZapdosApp::registerAll(Factory & f, ActionFactory & af, Syntax & s)
41{
42 ModulesApp::registerAllObjects<ZapdosApp>(f, af, s);
43 Registry::registerObjectsTo(f, {"ZapdosApp"});
44 Registry::registerActionsTo(af, {"ZapdosApp"});

46 /* register objects from dependent applications */
