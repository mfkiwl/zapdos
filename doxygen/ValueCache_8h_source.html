82template <typename T>
83void
84ValueCache<T>::insert(const std::vector<Real> & in_val, const T & out_val)
85{
86 mooseAssert(in_val.size() == _in_dim, "Key dimensions do not match cache dimensions");
87
88 auto id = _point_cloud._pts.size();
89 mooseAssert(size() == id, "Inconsistent cache data size.");
90
91 _point_cloud._pts.push_back(in_val);
92 _data.push_back(out_val);
93
94 // do we have too many subtrees?
95 if (_kd_tree && _kd_tree->getAllIndices().size() > _max_subtrees)
96 _kd_tree = nullptr;
97
98 // rebuild tree or add point
99 if (!_kd_tree)
100 {
101 _kd_tree = std::make_unique<KdTreeT>(_in_dim, _point_cloud, _max_leaf_size);
102 mooseAssert(_kd_tree != nullptr, "KDTree was not properly initialized.");
103 }
104 else
105 _kd_tree->addPoints(id, id);
106}

115template <typename T>
116bool
117ValueCache<T>::guess(const std::vector<Real> & in_val, T & out_val, Real & distance_sqr)
118{
119 // cache is empty
120 if (_data.empty())
121 return false;
122
123 nanoflann::KNNResultSet<Real> result_set(1);
124 std::size_t return_index;
125 result_set.init(&return_index, &distance_sqr);
126
127 // perform search
128 _kd_tree->findNeighbors(result_set, in_val.data());
129
130 // no result found
131 if (result_set.size() != 1)
132 return false;
133
134 out_val = _data[return_index];
135 return true;
136}

142template <typename T>
143std::vector<std::tuple<std::vector<Real> &, T &, Real>>
144ValueCache<T>::kNearestNeighbors(const std::vector<Real> & in_val, const std::size_t k)
145{
146 std::vector<std::tuple<std::vector<Real> &, T &, Real>> nearest_neighbors;
147
148 nanoflann::KNNResultSet<Real> result_set(std::min(k, size()));
149 std::vector<std::size_t> return_indices(std::min(k, size()));
150 std::vector<Real> distances(std::min(k, size()));
151
152 result_set.init(return_indices.data(), distances.data());
153
154 // kNN search
155 _kd_tree->findNeighbors(result_set, in_val.data());
156
157 for (std::size_t i = 0; i < result_set.size(); ++i)
158 nearest_neighbors.push_back(
159 std::tie((_point_cloud._pts[return_indices[i]]), (_data[return_indices[i]]), distances[i]));
160
161 return nearest_neighbors;
162}
