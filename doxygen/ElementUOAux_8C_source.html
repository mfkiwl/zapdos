47Real
48ElementUOAux::computeValue()
49{
50 if (_field_type == "long")
51 {
52 auto value = _elem_uo.getElementalValueLong(_current_elem->id(), _field_name);
53 if (value == std::numeric_limits<unsigned long>::max())
54 return -1.0;
55 else
56 return value;
57 }
58 else
59 return _elem_uo.getElementalValueReal(_current_elem->id(), _field_name);
60}
