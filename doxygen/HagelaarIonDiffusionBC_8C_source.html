42HagelaarIonDiffusionBC::computeQpResidual()
43{
44 if (_user_velocity > 0.)
45 _v_thermal = _user_velocity;
46 else
47 _v_thermal = std::sqrt(8 * _kb[_qp] * _T[_qp] / (M_PI * _mass[_qp]));
48
49 return _test[_i][_qp] * _r_units * (1. - _r) / (1. + _r) * 0.5 * _v_thermal * std::exp(_u[_qp]);
