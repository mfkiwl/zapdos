85void
86SideSetsGeneratorBase::flood(const Elem * elem,
87 Point normal,
88 boundary_id_type side_id,
89 MeshBase & mesh)
90{
91 if (elem == nullptr || (_visited[side_id].find(elem) != _visited[side_id].end()))
92 return;
93
94 _visited[side_id].insert(elem);
95 for (unsigned int side = 0; side < elem->n_sides(); ++side)
96 {
97 if (elem->neighbor_ptr(side))
98 continue;
99
100 _fe_face->reinit(elem, side);
101 const std::vector<Point> normals = _fe_face->get_normals();
102
103 // We'll just use the normal of the first qp
104 if (std::abs(1.0 - normal * normals[0]) <= _variance)
105 {
106 if (_replace)
107 mesh.get_boundary_info().remove_side(elem, side);
108
109 mesh.get_boundary_info().add_side(elem, side, side_id);
110 for (unsigned int neighbor = 0; neighbor < elem->n_sides(); ++neighbor)
111 {
112 // Flood to the neighboring elements using the current matching side normal from this
113 // element.
114 // This will allow us to tolerate small changes in the normals so we can "paint" around a
115 // curve.
116 flood(elem->neighbor_ptr(neighbor), _fixed_normal ? normal : normals[0], side_id, mesh);
117 }
118 }
119 }
