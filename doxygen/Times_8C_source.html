13InputParameters
14Times::validParams()
15{
16 InputParameters params = GeneralReporter::validParams();
17 params.addParam<bool>(
18 "unique_times", true, "Whether duplicate values should be removed from the Times vector");
19 params.addParam<Real>(
20 "unique_tolerance",
21 1e-12,
22 "Absolute tolerance for removing duplicated times when getting a set of unique times");
23 params.addParam<bool>("auto_sort", true, "Whether Times should be sorted");
24 // This parameter should be set by each derived class depending on whether the generation of
25 // times is replicated or distributed. We want times to be replicated across all ranks
26 params.addRequiredParam<bool>("auto_broadcast",
27 "Wether Times should be broadcasted across all ranks");
28 params.addParamNamesToGroup("auto_broadcast auto_sort unique_times unique_tolerance", "Advanced");
29 params.registerBase("Times");
30 return params;
31}

57Real
58Times::getPreviousTime(const Real current_time) const
59{
60 if (!_times.size())
61 mooseError("Times vector has not been initialized.");
62
63 Real previous_time = _times[0];
64 for (const auto i : make_range(std::size_t(1), _times.size()))
65 {
66 const auto & time = _times[i];
67 if (MooseUtils::absoluteFuzzyGreaterThan(time, current_time))
68 return previous_time;
69 previous_time = time;
70 }
71 // entire Times vector is prior to current_time
72 return previous_time;
73}

75Real
76Times::getNextTime(const Real current_time) const
77{
78 for (const auto i : index_range(_times))
79 {
80 const auto & time = _times[i];
81 if (MooseUtils::absoluteFuzzyGreaterThan(time, current_time))
82 return time;
83 }
84 if (_times.size())
85 mooseError("No next time in Times vector for time ",
86 current_time,
87 ". Maximum time in vector is ",
88 *_times.end());
89 else
90 mooseError("Times vector has not been initialized.");
91}

108void
109Times::finalize()
110{
111 if (_need_broadcast)
112 // The consumer/producer reporter interface can keep track of whether a reduction is needed
113 // (for example if a consumer needs replicated data, but the producer is distributed) however,
114 // we have currently made the decision that times should ALWAYS be replicated
115 _communicator.allgather(_times, /* identical buffer lengths = */ false);
116
117 if (_need_sort)
118 std::sort(_times.begin(), _times.end());
119 if (_need_unique)
120 _times.erase(unique(_times.begin(),
121 _times.end(),
122 [this](Real l, Real r) { return std::abs(l - r) < _unique_tol; }),
123 _times.end());
124}
