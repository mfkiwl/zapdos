40Real
41FunctorAux::computeValue()
42{
43 const auto state = determineState();
44 if (isNodal())
45 {
46 const Moose::NodeArg node_arg = {_current_node, Moose::INVALID_BLOCK_ID};
47 return _factor(node_arg, state) * _functor(node_arg, state);
48 }
49 else if (_is_fe)
50 {
51 const Moose::ElemQpArg qp_arg = {_current_elem, _qp, _qrule, _q_point[_qp]};
52 return _factor(qp_arg, state) * _functor(qp_arg, state);
53 }
54 else
55 {
56 const auto elem_arg = makeElemArg(_current_elem);
57 return _factor(elem_arg, state) * _functor(elem_arg, state);
58 }
59}
