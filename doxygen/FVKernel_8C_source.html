25InputParameters
26FVKernel::validParams()
27{
28 InputParameters params = ResidualObject::validParams();
29 params += BlockRestrictable::validParams();
30 params += ADFunctorInterface::validParams();
31 params.addParam<bool>("use_displaced_mesh",
32 false,
33 "Whether or not this object should use the "
34 "displaced mesh for computation. Note that in "
35 "the case this is true but no displacements "
36 "are provided in the Mesh block the "
37 "undisplaced mesh will still be used.");
38 params.addParamNamesToGroup("use_displaced_mesh", "Advanced");
39
40 params.addParam<unsigned short>("ghost_layers", 1, "The number of layers of elements to ghost.");
41 params.addParam<bool>("use_point_neighbors",
42 false,
43 "Whether to use point neighbors, which introduces additional ghosting to "
44 "that used for simple face neighbors.");
45 params.addParamNamesToGroup("ghost_layers use_point_neighbors", "Parallel ghosting");
46
47 // FV Kernels always need one layer of ghosting because when looping over
48 // faces to compute fluxes, the elements on each side of the face may be on
49 // different MPI ranks, but we still need to access them as a pair to
50 // compute the numerical face flux.
51 params.addRelationshipManager(
52 "ElementSideNeighborLayers",
53 Moose::RelationshipManagerType::GEOMETRIC | Moose::RelationshipManagerType::ALGEBRAIC |
54 Moose::RelationshipManagerType::COUPLING,
55 [](const InputParameters & obj_params, InputParameters & rm_params) {
56 FVKernel::setRMParams(
57 obj_params, rm_params, obj_params.get<unsigned short>("ghost_layers"));
58 });
59
60 params.registerBase("FVKernel");
61 return params;
62}
