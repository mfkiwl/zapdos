51Real
52ElementDamper::computeDamping()
53{
54 Real damping = 1.0;
55 Real cur_damping = 1.0;
56
57 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
58 {
59 cur_damping = computeQpDamping();
60 if (cur_damping < damping)
61 damping = cur_damping;
62 }
63
64 return damping;
65}
