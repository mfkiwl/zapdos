29ExtraIDIntegralVectorPostprocessor::ExtraIDIntegralVectorPostprocessor(
30 const InputParameters & parameters)
31 : ElementVariableVectorPostprocessor(parameters),
32 _nvar(coupledComponents("variable")),
33 _extra_id(getParam<std::vector<ExtraElementIDName>>("id_name")),
34 _n_extra_id(_extra_id.size())
35{
36 // create map of element ids to parsed vpp ids
37 _unique_vpp_ids =
38 MooseMeshUtils::getExtraIDUniqueCombinationMap(_mesh.getMesh(), blockIDs(), _extra_id);
39
40 // set up variable vector containing parsed extra ids
41 for (unsigned int i = 0; i < _n_extra_id; ++i)
42 {
43 const auto id_type = _extra_id[i];
44 auto & p = declareVector("Level-" + std::to_string(i) + "-" + id_type);
45 // collect local map of local vpp id to extra id
46 std::map<dof_id_type, dof_id_type> extra_ids;
47 for (const auto & elem : _mesh.getMesh().active_local_element_ptr_range())
48 {
49 if (!hasBlocks(elem->subdomain_id()))
50 continue;
51 auto vpp_id = _unique_vpp_ids[elem->id()];
52 if (extra_ids.find(vpp_id) == extra_ids.end())
53 extra_ids[vpp_id] = elem->get_extra_integer(getElementIDIndexByName(id_type));
54 }
55 // create global map of local vpp id to extra id
56 comm().set_union(extra_ids);
57 p.resize(extra_ids.size());
58 for (auto it : extra_ids)
59 p[it.first] = it.second;
60 _var_extra_ids.push_back(&p);
61 }
62
63 // declare vectors containing integral values
64 for (unsigned int i = 0; i < _nvar; ++i)
65 {
66 _vars.push_back(getVar("variable", i));
67 _var_values.push_back(&coupledValue("variable", i));
68 auto & p = declareVector(_vars[i]->name());
69 p.resize((*_var_extra_ids[0]).size());
70 _var_integrals.push_back(&p);
71 }
72}

81void
82ExtraIDIntegralVectorPostprocessor::execute()
83{
84 if (hasBlocks(_current_elem->subdomain_id()))
85 {
86 auto ipos = _unique_vpp_ids[_current_elem->id()];
87 for (unsigned int ivar = 0; ivar < _nvar; ++ivar)
88 if (_vars[ivar]->hasBlocks(_current_elem->subdomain_id()))
89 for (unsigned int qp = 0; qp < _qrule->n_points(); qp++)
90 (*_var_integrals[ivar])[ipos] += _JxW[qp] * _coord[qp] * (*_var_values[ivar])[qp];
91 }
92}

102ExtraIDIntegralVectorPostprocessor::threadJoin(const UserObject & s)
103{
104 const auto & sibling = static_cast<const ExtraIDIntegralVectorPostprocessor &>(s);
105
106 for (unsigned int ivar = 0; ivar < _nvar; ++ivar)
107 for (size_t i = 0; i < (*_var_integrals[ivar]).size(); ++i)
108 (*_var_integrals[ivar])[i] += (*sibling._var_integrals[ivar])[i];
