36EEDFRateCoefficientScalar::computeValue()
37{
38 Real val;
39 if (_sample_value)
40 {
41 val = _data.coefficient_sample(_reaction_number, _sampler_var[0]/1e-21);
42 if (val < 0.0)
43 {
44 return 0.0;
45 }
46 else
47 {
48 return val;
49 }
50 }
51 else
52 {
53 return _data.test(_reaction_number);
54 }
