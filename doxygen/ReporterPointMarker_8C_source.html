42void
43ReporterPointMarker::markerSetup()
44{
45 _pl = _fe_problem.mesh().getPointLocator();
46 _pl->enable_out_of_mesh_mode();
47 const auto npoints = _x_coord.size();
48 if (npoints != _y_coord.size() || npoints != _z_coord.size())
49 mooseError("The coordinate vectors are a different size. \n",
50 " x_coord size = ",
51 npoints,
52 "; y_coord size = ",
53 _y_coord.size(),
54 "; z_coord size = ",
55 _z_coord.size());
56
57 _point_elems.clear();
58 for (std::size_t i = 0; i < npoints; ++i)
59 {
60 Point pt(_x_coord[i], _y_coord[i], _z_coord[i]);
61 const auto elem = (*_pl)(pt);
62 if (elem)
63 _point_elems.insert(elem->id());
64 }
65}
