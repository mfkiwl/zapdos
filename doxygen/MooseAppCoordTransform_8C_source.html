25void
26MooseAppCoordTransform::setUpDirection(const Direction up_direction)
27{
28 Real alpha = 0, beta = 0, gamma = 0;
29
30 const bool must_rotate_axes =
31 _coord_type == Moose::COORD_RZ || _coord_type == Moose::COORD_RSPHERICAL;
32 // Don't error immediately for unit testing purposes
33 bool negative_radii = false;
34
35 if (up_direction == X)
36 {
37 alpha = 90, beta = 0, gamma = 0;
38 if (must_rotate_axes)
39 {
40 if (_r_axis == X)
41 {
42 _r_axis = Y;
43 _z_axis = processZAxis(X);
44 }
45 else if (_r_axis == Y)
46 {
47 negative_radii = true;
48 _r_axis = X;
49 _z_axis = processZAxis(Y);
50 }
51 else
52 mooseError("Bad r-axis value");
53 }
54 }
55 else if (up_direction == Y)
56 alpha = 0, beta = 0, gamma = 0;
57 else if (up_direction == Z)
58 {
59 alpha = 0, beta = -90, gamma = 0;
60 if (must_rotate_axes)
61 {
62 if (_r_axis == X)
63 {
64 _r_axis = X;
65 _z_axis = processZAxis(Z);
66 }
67 else if (_r_axis == Y)
68 {
69 negative_radii = true;
70 _r_axis = Z;
71 _z_axis = processZAxis(X);
72 }
73 else
74 mooseError("Bad r-axis value");
75 }
76 }
77 else
78 mooseError("Bad up direction value");
79
80 _euler_angles = {{alpha, beta, gamma}};
81
82 _rotate = std::make_unique<RealTensorValue>(
83 RealTensorValue::extrinsic_rotation_matrix(alpha, beta, gamma));
84 computeRS();
85
86 if (negative_radii)
87 mooseError("Rotation yields negative radial values");
88}

90void
91MooseAppCoordTransform::setRotation(const Real alpha, const Real beta, const Real gamma)
92{
93 const bool must_rotate_axes =
94 _coord_type == Moose::COORD_RZ || _coord_type == Moose::COORD_RSPHERICAL;
95 bool axes_rotated = false;
96 if (must_rotate_axes)
97 {
98 const auto angles = std::make_tuple(alpha, beta, gamma);
99 if (angles == std::make_tuple(0, 90, 0))
100 {
101 if (_r_axis == X)
102 {
103 mooseAssert((_coord_type == Moose::COORD_RZ && _z_axis == Y) ||
104 (_coord_type == Moose::COORD_RSPHERICAL && _z_axis == INVALID),
105 "'_z_axis' is not an expected value");
106 _r_axis = X;
107 _z_axis = _coord_type == Moose::COORD_RZ ? Z : INVALID;
108 }
109 else if (_r_axis == Y)
110 {
111 mooseAssert((_coord_type == Moose::COORD_RZ && _z_axis == X) ||
112 (_coord_type == Moose::COORD_RSPHERICAL && _z_axis == INVALID),
113 "'_z_axis' is not an expected value");
114 _r_axis = Z;
115 _z_axis = _coord_type == Moose::COORD_RZ ? X : INVALID;
116 }
117 axes_rotated = true;
118 }
119 }
120
121 _euler_angles = {{alpha, beta, gamma}};
122 _rotate = std::make_unique<RealTensorValue>(
123 RealTensorValue::extrinsic_rotation_matrix(alpha, beta, gamma));
124 computeRS();
125
126 if (must_rotate_axes && !axes_rotated)
127 mooseError("Unsupported manual angle prescription in 'MooseAppCoordTransform::setRotation'. "
128 "For non-Cartesian coordinate systems, the only currently supported rotation is "
129 "(alpha, beta, gamma) = (0, 90, 0)");
130}

161void
162MooseAppCoordTransform::setCoordinateSystem(const MooseMesh & mesh)
163{
164 const auto & params = mesh.parameters();
165
166 // If we have multiple different coordinate system types in our problem, we
167 // take note of it because that can cause issues if there is a non-Cartesian
168 // destination coordinate system
169 const auto & coord_sys = mesh.getCoordSystem();
170 std::unordered_set<Moose::CoordinateSystemType> coord_types;
171 auto map_it = coord_sys.begin();
172 // It's possible that the mesh is not in a complete state
173 if (map_it == coord_sys.end())
174 setCoordinateSystem(Moose::COORD_XYZ);
175 else
176 setCoordinateSystem(
177 map_it->second,
178 Direction(static_cast<unsigned int>(int(params.get<MooseEnum>("rz_coord_axis")))));
179 for (; map_it != coord_sys.end(); ++map_it)
180 coord_types.insert(map_it->second);
181
182 _has_different_coord_sys = coord_types.size() > 1;
183
184 if (mesh.usingGeneralAxisymmetricCoordAxes())
185 _using_general_rz_coord_axes = true;
186}

258MooseAppCoordTransform::MooseAppCoordTransform(const MooseMesh & mesh)
259 : _coord_type(Moose::COORD_XYZ),
260 _r_axis(INVALID),
261 _z_axis(INVALID),
262 _has_different_coord_sys(false),
263 _using_general_rz_coord_axes(false),
264 _length_unit(std::string("1*m")),
265 _euler_angles(),
266 _mesh_transformed(false)
267{
268 //
269 // Coordinate system transformation
270 //
271 setCoordinateSystem(mesh);
272
273 const auto & params = mesh.parameters();
274
275 //
276 // Rotation
277 //
278 const bool has_alpha = params.isParamValid("alpha_rotation");
279 const bool has_beta = params.isParamValid("beta_rotation");
280 const bool has_gamma = params.isParamValid("gamma_rotation");
281 const auto & up_direction = params.get<MooseEnum>("up_direction");
282
283 if (has_alpha || has_beta || has_gamma)
284 {
285 if (up_direction.isValid())
286 mooseError("Cannot simultaneously set rotation angles as well as an up direction");
287
288 const auto alpha = (has_alpha ? params.get<Real>("alpha_rotation") : Real(0));
289 const auto beta = (has_beta ? params.get<Real>("beta_rotation") : Real(0));
290 const auto gamma = (has_gamma ? params.get<Real>("gamma_rotation") : Real(0));
291
292 setRotation(alpha, beta, gamma);
293 }
294 else if (up_direction.isValid())
295 setUpDirection(Direction(static_cast<unsigned int>(int(up_direction))));
296
297 //
298 // Scaling
299 //
300 if (params.isParamValid("length_unit"))
301 setLengthUnit(MooseUnits(params.get<std::string>("length_unit")));
302}

393void
394MooseAppCoordTransform::computeRS()
395{
396 if (_scale || _rotate)
397 {
398 _rs = std::make_unique<RealTensorValue>(RealTensorValue(1, 0, 0, 0, 1, 0, 0, 0, 1));
399
400 if (_scale)
401 *_rs = *_scale * *_rs;
402 if (_rotate)
403 *_rs = *_rotate * *_rs;
404
405 _rs_inverse = std::make_unique<RealTensorValue>(_rs->inverse());
406 }
407 else
408 {
409 _rs.reset();
410 _rs_inverse.reset();
411 }
412}

414void
415MooseAppCoordTransform::transformMesh(MooseMesh & mesh, const Point & translation)
416{
417 // Transforming a RZ or R-spherical mesh doesnt always make sense, disallow it
418 if (_coord_type != Moose::COORD_XYZ)
419 mooseError("Running MultiApps 'in position' is only supported for XYZ coordinate systems");
420
421 if (_using_general_rz_coord_axes && hasScalingOrRotationTransformation())
422 mooseError("Scaling and rotation are currently not supported for general axisymmetric "
423 "coordinate systems.");
424
425 if (_mesh_transformed)
426 mooseError("App mesh is being transformed twice");
427
428 // Apply all the transformation to the mesh
429 if (_scale)
430 MeshTools::Modification::scale(mesh, (*_scale)(0, 0), (*_scale)(1, 1), (*_scale)(2, 2));
431 if (_rotate)
432 MeshTools::Modification::rotate(mesh, _euler_angles[0], _euler_angles[1], _euler_angles[2]);
433 if (translation != Point(0, 0, 0))
434 MeshTools::Modification::translate(mesh, translation(0), translation(1), translation(2));
435
436 // Translation, scaling and rotation need not be applied anymore when performing coordinate
437 // transforms
438 _mesh_transformed = true;
439}

441bool
442MooseAppCoordTransform::hasScalingOrRotationTransformation() const
443{
444 if (_rs)
445 for (const auto i : make_range(Moose::dim))
446 for (const auto j : make_range(Moose::dim))
447 {
448 const auto matrix_elem = (*_rs)(i, j);
449 if (i == j)
450 {
451 if (!MooseUtils::absoluteFuzzyEqual(matrix_elem, 1))
452 return true;
453 }
454 else if (!MooseUtils::absoluteFuzzyEqual(matrix_elem, 0))
455 return true;
456 }
457
458 return false;
459}

468Point
469MultiAppCoordTransform::operator()(const Point & point) const
470{
471 mooseAssert(_destination_app_transform, "The destination application transform must be set");
472
473 Point ret(point);
474
475 // Translation, rotation and scaling already applied, coordinate system conversion not supported
476 if (_our_app_transform._mesh_transformed)
477 return ret;
478
479 // Apply scaling and then rotation
480 if (_our_app_transform._rs)
481 ret = (*_our_app_transform._rs) * ret;
482
483 // If this shows up in profiling we can make _translation a pointer
484 ret += _translation;
485
486 if (_skip_coordinate_collapsing)
487 return ret;
488
489 // Finally, coordinate system conversions
490 if (_our_app_transform._coord_type == Moose::COORD_XYZ &&
491 _destination_app_transform->_coord_type == Moose::COORD_RZ)
492 {
493 Real r_squared = 0;
494 for (unsigned int i = 0; i < LIBMESH_DIM; ++i)
495 if (i != _destination_app_transform->_z_axis)
496 r_squared += ret(i) * ret(i);
497
498 const auto r = std::sqrt(r_squared);
499 const auto z = ret(_destination_app_transform->_z_axis);
500 ret = 0;
501 ret(_destination_app_transform->_r_axis) = r;
502 ret(_destination_app_transform->_z_axis) = z;
503 }
504 else if (_our_app_transform._coord_type == Moose::COORD_XYZ &&
505 _destination_app_transform->_coord_type == Moose::COORD_RSPHERICAL)
506 {
507 Real r_squared = 0;
508 for (unsigned int i = 0; i < LIBMESH_DIM; ++i)
509 r_squared += ret(i) * ret(i);
510
511 const auto r = std::sqrt(r_squared);
512 ret = 0;
513 ret(_destination_app_transform->_r_axis) = r;
514 }
515 else if (_our_app_transform._coord_type == Moose::COORD_RZ &&
516 _destination_app_transform->_coord_type == Moose::COORD_RSPHERICAL)
517 {
518 Real r_squared = 0;
519 for (unsigned int i = 0; i < LIBMESH_DIM; ++i)
520 {
521 mooseAssert(
522 i == _our_app_transform._r_axis || i == _our_app_transform._z_axis ||
523 MooseUtils::absoluteFuzzyEqual(ret(i), 0),
524 "Our point should be 0 if we are evaluating at an index that is neither our r or z-axis");
525 r_squared += ret(i) * ret(i);
526 }
527
528 const auto r = std::sqrt(r_squared);
529 ret = 0;
530 ret(_destination_app_transform->_r_axis) = r;
531 }
532
533 return ret;
534}

536Point
537MultiAppCoordTransform::mapBack(const Point & point) const
538{
539 Point ret(point);
540
541 // Translation, rotation and scaling already applied, coordinate system conversion not supported
542 if (_our_app_transform._mesh_transformed)
543 return ret;
544
545 // inverse translate
546 ret -= _translation;
547
548 // inverse rotate and then inverse scale
549 if (_our_app_transform._rs_inverse)
550 ret = (*_our_app_transform._rs_inverse) * ret;
551
552 if (_skip_coordinate_collapsing)
553 return ret;
554
555 // Finally, coordinate system conversions
556 if ((_our_app_transform._coord_type == Moose::COORD_XYZ &&
557 (_destination_app_transform->_coord_type == Moose::COORD_RZ ||
558 _destination_app_transform->_coord_type == Moose::COORD_RSPHERICAL)) ||
559 (_our_app_transform._coord_type == Moose::COORD_RZ &&
560 _destination_app_transform->_coord_type == Moose::COORD_RSPHERICAL))
561 mooseError("Coordinate collapsing occurred in going to the reference space. There is no unique "
562 "return mapping");
563
564 return ret;
565}

567void
568MultiAppCoordTransform::setDestinationCoordTransform(
569 const MooseAppCoordTransform & destination_app_transform)
570{
571 _destination_app_transform = &destination_app_transform;
572
573 if (_our_app_transform._using_general_rz_coord_axes &&
574 _our_app_transform.hasScalingOrRotationTransformation())
575 mooseError("Scaling and rotation are currently not supported for general axisymmetric "
576 "coordinate systems.");
577
578 // Don't error check mismatching coordinate system types if we've been asked to skip coordinate
579 // collapsing since in that case the mismatch doesn't matter
580 if (_skip_coordinate_collapsing)
581 return;
582
583 if (_destination_app_transform->_has_different_coord_sys &&
584 (_our_app_transform._has_different_coord_sys ||
585 _our_app_transform._coord_type != Moose::COORD_RSPHERICAL))
586 mooseError(
587 "The destination coordinate system has different coordinate systems and we have coordinate "
588 "system(s) that could require coordinate collapsing when transforming from our coordinate "
589 "system to the destination coordinate system. Because our transform method only takes a "
590 "point argument, and not subdomain arguments, the transform is ambiguous");
591
592 if ((_destination_app_transform->_coord_type == Moose::COORD_RZ ||
593 _destination_app_transform->_coord_type == Moose::COORD_RSPHERICAL) &&
594 _our_app_transform._has_different_coord_sys)
595 mooseError("When the destination coordinate system is RZ or RSPHERICAL, we have to perform "
596 "coordinate collapsing based on *our* coordinate system. However, we have multiple "
597 "coordinate systems, and since when evaluating transformations, we are only "
598 "called with a Point argument, we do not know what subdomain we are on and "
599 "consequently we do not know what coordinate collapse to apply.");
600
601 if (_our_app_transform._using_general_rz_coord_axes ||
602 _destination_app_transform->_using_general_rz_coord_axes)
603 mooseError("If either this app or the destination app uses general axisymmetric axes, "
604 "coordinate collapsing must be skipped.");
605}
