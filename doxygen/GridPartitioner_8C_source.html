54void
55GridPartitioner::_do_partition(MeshBase & mesh, const unsigned int /*n*/)
56{
57 // By default, there are one processor along each direction
58 // nx: the number of processors along x direction
59 // ny: the number of processors along y direction
60 // nz: the number of processors along z direction
61 unsigned int nx = 1, ny = 1, nz = 1;
62
63 // Figure out the physical bounds of the given mesh
64 auto bounding_box = MeshTools::create_bounding_box(mesh);
65 const auto & min = bounding_box.min();
66 const auto & max = bounding_box.max();
67
68 auto dim = mesh.mesh_dimension();
69 // Need to make sure the number of cells in the grid matches the number of procs to partition for
70 nx = getParam<unsigned int>("nx");
71
72 if (dim >= 2)
73 ny = getParam<unsigned int>("ny");
74
75 if (dim == 3)
76 nz = getParam<unsigned int>("nz");
77
78 // We should compute a balanced factorization so
79 // that we can assign proper processors to each direction.
80 // I just want to make grid partitioner smarter.
81 if ((nx * ny * nz) != mesh.n_partitions())
82 {
83 // Anybody knows we are living in a 3D space.
84 int dims[] = {0, 0, 0};
85 MPI_Dims_create(mesh.n_partitions(), dim, dims);
86
87 nx = dims[0];
88 if (dim >= 2)
89 ny = dims[1];
90 if (dim == 3)
91 nz = dims[2];
92 }
93
94 // hx: grid interval along x direction
95 // hy: grid interval along y direction
96 // hz: grid interval along z direction
97 // Lx: domain length along x direction
98 // Ly: domain length along y direction
99 // Lz: domain length along z direction
100 Real hx = 1., hy = 1., hz = 1., Lx = 1., Ly = 1., Lz = 1.;
101 Lx = max(0) - min(0);
102 hx = Lx / nx;
103 if (dim >= 2)
104 {
105 Ly = max(1) - min(1);
106 hy = Ly / ny;
107 }
108
109 if (dim == 3)
110 {
111 Lz = max(2) - min(2);
112 hz = Lz / nz;
113 }
114
115 // Processor coordinates along x, y, z directions
116 unsigned int k = 0, j = 0, i = 0;
117 // Coordinates for current element centroid
118 Real coordx = 0, coordy = 0, coordz = 0;
119
120 // Loop over all of the elements in the given mesh
121 for (auto & elem_ptr : mesh.active_element_ptr_range())
122 {
123 // Find the element it lands in in the GeneratedMesh
124 auto centroid = elem_ptr->vertex_average();
125
126 coordx = centroid(0);
127 mooseAssert(coordx >= min(0) && coordy <= max(0),
128 "element is outside of bounding box along x direction");
129 if (dim >= 2)
130 {
131 coordy = centroid(1);
132 mooseAssert(coordy >= min(1) && coordy <= max(1),
133 "element is outside of bounding box along y direction");
134 }
135 if (dim == 3)
136 {
137 coordz = centroid(2);
138 mooseAssert(coordz >= min(2) && coordz <= max(2),
139 "element is outside of bounding box along z direction");
140 }
141
142 // Compute processor coordinates
143 j = k = 0;
144 i = (coordx - min(0)) / hx;
145 if (dim >= 2)
146 j = (coordy - min(1)) / hy;
147 if (dim == 3)
148 k = (coordz - min(2)) / hz;
149
150 mooseAssert(i < nx, "Index caculation is wrong along x direction");
151 mooseAssert(j < ny, "Index caculation is wrong along y direction");
152 mooseAssert(k < nz, "Index caculation is wrong along z direction");
153 // Assign processor ID to current element
154 elem_ptr->processor_id() = k * nx * ny + j * nx + i;
155 }
156}
