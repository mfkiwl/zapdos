32SuperelasticReactionRate::SuperelasticReactionRate(const InputParameters & parameters)
33 : Material(parameters),
34 _reaction_rate(declareProperty<Real>("k_" + getParam<std::string>("reaction"))),
35 _enthalpy_reaction(declareProperty<Real>("delta_H")),
36 _reversible_rate(getMaterialProperty<Real>("k_" + getParam<std::string>("original_reaction"))),
37 _coefficients(getParam<std::vector<Real>>("stoichiometric_coeff")),
38 _participants(getParam<std::vector<std::string>>("participants")),
39 _Tgas(isCoupled("gas_temperature") ? coupledValue("gas_temperature") : _zero)
40{
41 // This material follows the method outlined in:
42 // Mikhail S Finko et al 2017 J. Phys. D: Appl. Phys. 50 485201
43 // Section 2.2, Equation 13
44
45 // Read the participant species' coefficients from files
46 std::string file_name;
47 std::vector<std::vector<Real>> polynomial_coefficients;
48 polynomial_coefficients.resize(_participants.size());
49 _power_coefficient = 0.0;
50 for (unsigned int i = 0; i < _participants.size(); ++i)
51 {
52 _power_coefficient += _coefficients[i]; // Finko, equation 7
53 file_name = getParam<std::string>("file_location") + "/" + _participants[i] + ".txt";
54 MooseUtils::checkFileReadable(file_name);
55 const char * charPath = file_name.c_str();
56 std::ifstream myfile(charPath);
57 Real value;
58
59 if (myfile.is_open())
60 {
61 while (myfile >> value)
62 {
63 polynomial_coefficients[i].push_back(value);
64 }
65 myfile.close();
66 }
67 else
68 {
69 mooseError("Unable to open file: " + file_name);
70 }
71 }
72
73 // Sum the coefficients
74 delta_a.resize(polynomial_coefficients[0].size(), 0.0);
75 for (unsigned int i = 0; i < polynomial_coefficients.size(); ++i)
76 {
77 for (unsigned int j = 0; j < polynomial_coefficients[i].size(); ++j)
78 {
79 delta_a[j] += _coefficients[i] * polynomial_coefficients[i][j];
80 }
81 }
82}

84void
85SuperelasticReactionRate::computeQpProperties()
86{
87 // Finko, equation 8
88 _enthalpy_reaction[_qp] = delta_a[0] + (delta_a[1]/2.0)*_Tgas[_qp] + (delta_a[2]/3.0)*std::pow(_Tgas[_qp],2.0) +
89 (delta_a[3]/4.0)*std::pow(_Tgas[_qp],3.0) + (delta_a[4]/5.0)*std::pow(_Tgas[_qp],4.0) +
90 (delta_a[5]/_Tgas[_qp]);
91
92 // Finko, equation 13 (work in progress...)
93 _equilibrium_constant = std::pow(1.0, _power_coefficient) * std::exp(delta_a[0]*(std::log(_Tgas[_qp] - 1)) +
94 (delta_a[1]/2.0)*_Tgas[_qp] + (delta_a[2]/6.0)*std::pow(_Tgas[_qp], 2.0) + (delta_a[3]/12.0)*std::pow(_Tgas[_qp], 3.0) +
95 (delta_a[4]/20.0)*std::pow(_Tgas[_qp], 4.0) - delta_a[5]*std::pow(_Tgas[_qp], -1.0) + delta_a[6]);
96
97 _reaction_rate[_qp] = _reversible_rate[_qp] / _equilibrium_constant;
98}
