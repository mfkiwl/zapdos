28 "Transfers postprocessor data between the master application and sub-application(s).");

32 params.addRequiredParam<PostprocessorName>(
33 "to_postprocessor",

63void
64MultiAppPostprocessorTransfer::execute()
65{
66 TIME_SECTION("MultiAppPostprocessorTransfer::execute()", 5, "Transferring a postprocessor");
67
68 switch (_current_direction)
69 {
70 case BETWEEN_MULTIAPP:
71 for (unsigned int i = 0; i < getFromMultiApp()->numGlobalApps(); i++)
72 {
73 if (getFromMultiApp()->hasLocalApp(i))
74 {
75 // Get source postprocessor value
76 FEProblemBase & from_problem = getFromMultiApp()->appProblemBase(i);
77 const Real & pp_value = from_problem.getPostprocessorValueByName(_from_pp_name);
78
79 if (getToMultiApp()->hasLocalApp(i))
80 getToMultiApp()->appProblemBase(i).setPostprocessorValueByName(_to_pp_name, pp_value);
81 }
82 }
83 break;
84 case TO_MULTIAPP:
85 {
86 FEProblemBase & from_problem = getToMultiApp()->problemBase();
87
88 const Real & pp_value = from_problem.getPostprocessorValueByName(_from_pp_name);
89
90 for (unsigned int i = 0; i < getToMultiApp()->numGlobalApps(); i++)
91 if (getToMultiApp()->hasLocalApp(i))
92 getToMultiApp()->appProblemBase(i).setPostprocessorValueByName(_to_pp_name, pp_value);
93 break;
94 }
95 case FROM_MULTIAPP:
96 {
97 FEProblemBase & to_problem = getFromMultiApp()->problemBase();
98
99 Real reduced_pp_value;
100 switch (_reduction_type)
101 {
102 case AVERAGE:
103 case SUM:
104 reduced_pp_value = 0;
105 break;
106 case MAXIMUM:
107 reduced_pp_value = -std::numeric_limits<Real>::max();
108 break;
109 case MINIMUM:
110 reduced_pp_value = std::numeric_limits<Real>::max();
111 break;
112 default:
113 mooseError(
114 "Can't get here unless someone adds a new enum and fails to add it to this switch");
115 }
116
117 const auto multi_app = getFromMultiApp() ? getFromMultiApp() : getToMultiApp();
118
119 for (unsigned int i = 0; i < multi_app->numGlobalApps(); i++)
120 {
121 if (multi_app->hasLocalApp(i) && multi_app->isRootProcessor())
122 {
123 const Real & curr_pp_value =
124 multi_app->appProblemBase(i).getPostprocessorValueByName(_from_pp_name);
125 switch (_reduction_type)
126 {
127 case AVERAGE:
128 case SUM:
129 reduced_pp_value += curr_pp_value;
130 break;
131 case MAXIMUM:
132 reduced_pp_value = std::max(curr_pp_value, reduced_pp_value);
133 break;
134 case MINIMUM:
135 reduced_pp_value = std::min(curr_pp_value, reduced_pp_value);
136 break;
137 default:
138 mooseError("Can't get here unless someone adds a new enum and fails to add it to "
139 "this switch");
140 }
141 }
142 }
143
144 switch (_reduction_type)
145 {
146 case AVERAGE:
147 _communicator.sum(reduced_pp_value);
148 reduced_pp_value /= static_cast<Real>(multi_app->numGlobalApps());
149 break;
150 case SUM:
151 _communicator.sum(reduced_pp_value);
152 break;
153 case MAXIMUM:
154 _communicator.max(reduced_pp_value);
155 break;
156 case MINIMUM:
157 _communicator.min(reduced_pp_value);
158 break;
159 default:
160 mooseError(
161 "Can't get here unless someone adds a new enum and fails to add it to this switch");
162 }
163
164 to_problem.setPostprocessorValueByName(_to_pp_name, reduced_pp_value);
165 break;
166 }
167 }
168}
