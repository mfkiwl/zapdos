443template <typename T, typename T2, typename T3 = Real>
444bool
445relativeFuzzyEqual(const T & var1,
446 const T2 & var2,
447 const T3 & tol = libMesh::TOLERANCE * libMesh::TOLERANCE)
448{
449 if constexpr (libMesh::ScalarTraits<T>::value ||
450 libMesh::TensorTools::MathWrapperTraits<T>::value)
451 {
452 static_assert(libMesh::TensorTools::TensorTraits<T>::rank ==
453 libMesh::TensorTools::TensorTraits<T2>::rank,
454 "Mathematical types must be same for arguments to relativelyFuzzEqual");
455 if constexpr (libMesh::TensorTools::TensorTraits<T>::rank == 0)
456 return absoluteFuzzyEqual(
457 var1,
458 var2,
459 tol * (std::abs(MetaPhysicL::raw_value(var1)) + std::abs(MetaPhysicL::raw_value(var2))));
460 else if constexpr (libMesh::TensorTools::TensorTraits<T>::rank == 1)
461 {
462 for (const auto i : make_range(Moose::dim))
463 if (!relativeFuzzyEqual(var1(i), var2(i), tol))
464 return false;
465
466 return true;
467 }
468 else if constexpr (libMesh::TensorTools::TensorTraits<T>::rank == 2)
469 {
470 for (const auto i : make_range(Moose::dim))
471 for (const auto j : make_range(Moose::dim))
472 if (!relativeFuzzyEqual(var1(i, j), var2(i, j), tol))
473 return false;
474
475 return true;
476 }
477 }
478 else
479 {
480 // We dare to dream
481 mooseAssert(var1.size() == var2.size(), "These must be the same size");
482 for (const auto i : index_range(var1))
483 if (!relativeFuzzyEqual(var1(i), var2(i), tol))
484 return false;
485
486 return true;
487 }
488}

620{

742template <typename T>
743void
744expandAllMatches(const std::vector<T> & candidates, std::vector<T> & patterns)
745{
746 std::set<T> expanded;
747 for (const auto & p : patterns)
748 {
749 unsigned int found = 0;
750 for (const auto & c : candidates)
751 if (globCompare(c, p))
752 {
753 expanded.insert(c);
754 found++;
755 }
756 if (!found)
757 throw std::invalid_argument(p);
758 }
759 patterns.assign(expanded.begin(), expanded.end());
760}

768template <typename T>
769void
770tokenize(const std::string & str,
771 std::vector<T> & elements,
772 unsigned int min_len = 1,
773 const std::string & delims = "/")
774{
775 elements.clear();
776
777 std::string::size_type last_pos = str.find_first_not_of(delims, 0);
778 std::string::size_type pos = str.find_first_of(delims, std::min(last_pos + min_len, str.size()));
779
780 while (last_pos != std::string::npos)
781 {
782 elements.push_back(str.substr(last_pos, pos - last_pos));
783 // skip delims between tokens
784 last_pos = str.find_first_not_of(delims, pos);
785 if (last_pos == std::string::npos)
786 break;
787 pos = str.find_first_of(delims, std::min(last_pos + min_len, str.size()));
788 }
789}

795template <typename T>
796bool
797tokenizeAndConvert(const std::string & str,
798 std::vector<T> & tokenized_vector,
799 const std::string & delimiter = " \t\n\v\f\r")
800{
801 std::vector<std::string> tokens;
802 MooseUtils::tokenize(str, tokens, 1, delimiter);
803 tokenized_vector.resize(tokens.size());
804 for (unsigned int j = 0; j < tokens.size(); ++j)
805 {
806 std::stringstream ss(trim(tokens[j]));
807 // we have to make sure that the conversion succeeded _and_ that the string
808 // was fully read to avoid situations like [conversion to Real] 3.0abc to work
809 if ((ss >> tokenized_vector[j]).fail() || !ss.eof())
810 return false;
811 }
812 return true;
813}
