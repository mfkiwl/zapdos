412template <typename Residuals, typename Indices>
413void
414TaggingInterface::addResiduals(Assembly & assembly,
415 const Residuals & residuals,
416 const Indices & dof_indices,
417 const Real scaling_factor)
418{
419 assembly.cacheResiduals(
420 residuals, dof_indices, scaling_factor, Assembly::LocalDataKey{}, _vector_tags);
421 if (!_abs_vector_tags.empty())
422 {
423 _absolute_residuals.resize(residuals.size());
424 for (const auto i : index_range(residuals))
425 _absolute_residuals[i] = std::abs(MetaPhysicL::raw_value(residuals[i]));
426
427 assembly.cacheResiduals(_absolute_residuals,
428 dof_indices,
429 scaling_factor,
430 Assembly::LocalDataKey{},
431 _abs_vector_tags);
432 }
433}
