48ADReal
49InterfaceLogDiffusionElectrons::computeQpResidual(Moose::DGResidualType type)
50{
51 ADReal r = 0;
52
53 switch (type)
54 {
55 case Moose::Element:
56 r = -_diffem[_qp] * std::exp(_neighbor_value[_qp]) * _grad_neighbor_value[_qp] *
57 _r_neighbor_units * _normals[_qp] * _test[_i][_qp] * _r_units;
58 break;
59
60 case Moose::Neighbor:
61 r = 0.;
62 break;
63 }
64
65 return r;
66}
