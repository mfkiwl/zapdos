40void
41CommandLine::addArgument(std::string arg)
42{
43 _argv.push_back(arg);
44
45 auto arg_value = std::string(arg);
46
47 // Handle using a "="
48 if (arg_value.find("=") != std::string::npos)
49 {
50 std::vector<std::string> arg_split;
51
52 MooseUtils::tokenize(arg_value, arg_split, 1, "=");
53
54 for (auto & arg_piece : arg_split)
55 _args.push_back(MooseUtils::trim(arg_piece));
56 }
57 else
58 _args.push_back(arg_value);
59}

63void
64CommandLine::initForMultiApp(const std::string & subapp_full_name)
65{
66 // Get the name and number for the current sub-application
67 std::string sub_name;
68 int sub_num = std::numeric_limits<int>::min(); // this initial value should never be used
69 pcrecpp::RE("(\\S*?)(\\d*)").FullMatch(subapp_full_name, &sub_name, &sub_num);
70
71 if (sub_num == std::numeric_limits<int>::min())
72 mooseError("The sub-application name '", subapp_full_name, "' must contain a number.");
73
74 // "remove" CLI args for other sub-applications; remove_if just moves items to the end, so
75 // an erase is needed to actually remove the items
76 auto new_end = std::remove_if(
77 _argv.begin(),
78 _argv.end(),
79 [&sub_name, sub_num](const std::string & arg)
80 {
81 // Determine if the current command line argument ('arg') and extract the sub-application
82 // name and number. If 'arg' is not command line argument for sub-application then the regex
83 // match fails and the argument is retained.
84 std::string arg_sub_name;
85 int arg_sub_num = -1;
86 pcrecpp::RE("(\\S*?)(\\d*):").PartialMatch(arg, &arg_sub_name, &arg_sub_num);
87 if (!arg_sub_name.empty())
88 {
89 // The argument should be retained if names match and the current argument doesn't have
90 // a number or the supplied sub-application number and the current argument number match
91 bool keep = (sub_name == arg_sub_name && (arg_sub_num == -1 || arg_sub_num == sub_num));
92 return !keep;
93 }
94 return false;
95 });
96 _argv.erase(new_end, _argv.end());
97
98 // If there is an argument meant for a nested subapp, we will want to remove
99 // the leading app name
100 for (auto & arg : _argv)
101 {
102 auto pos = arg.find(":", 0);
103 if (pos != std::string::npos && arg.find(":", pos + 1) != std::string::npos)
104 arg = subapp_full_name + "_" + arg.substr(pos + 1, arg.length() - pos - 1);
105 }
106
107 // Clear hit CLI arguments, these will be populated after the sub-application is created
108 _hiti.clear();
109 _used_hiti.clear();
110}

112void
113CommandLine::addCommandLineOptionsFromParams(InputParameters & params)
114{
115 for (const auto & it : params)
116 {
117 Option cli_opt;
118 std::vector<std::string> syntax;
119 std::string orig_name = it.first;
120
121 cli_opt.description = params.getDocString(orig_name);
122 if (!params.isPrivate(orig_name))
123 // If a param is private then it shouldn't have any command line syntax.
124 syntax = params.getSyntax(orig_name);
125 cli_opt.cli_syntax = syntax;
126 cli_opt.required = false;
127
128 if (params.have_parameter<bool>(orig_name))
129 cli_opt.argument_type = CommandLine::NONE;
130 else
131 cli_opt.argument_type = CommandLine::REQUIRED;
132
133 addOption(orig_name, cli_opt);
134 }
135}

137void
138CommandLine::populateInputParams(InputParameters & params)
139{
140 for (const auto & it : params)
141 {
142 std::string orig_name = it.first;
143
144 if (search(orig_name))
145 {
146 if (params.have_parameter<std::string>(orig_name))
147 {
148 search(orig_name, params.set<std::string>(orig_name));
149 continue;
150 }
151
152 if (params.have_parameter<std::vector<std::string>>(orig_name))
153 {
154 search(orig_name, params.set<std::vector<std::string>>(orig_name));
155 continue;
156 }
157
158 if (params.have_parameter<Real>(orig_name))
159 {
160 search(orig_name, params.set<Real>(orig_name));
161 continue;
162 }
163
164 if (params.have_parameter<unsigned int>(orig_name))
165 {
166 search(orig_name, params.set<unsigned int>(orig_name));
167 continue;
168 }
169
170 if (params.have_parameter<int>(orig_name))
171 {
172 search(orig_name, params.set<int>(orig_name));
173 continue;
174 }
175
176 if (params.have_parameter<bool>(orig_name))
177 {
178 search(orig_name, params.set<bool>(orig_name));
179 continue;
180 }
181 }
182 else if (params.isParamRequired(orig_name))
183 mooseError("Missing required command-line parameter: ",
184 orig_name,
185 "\nDoc String: ",
186 params.getDocString(orig_name));
187 }
188}

199std::vector<std::string>::const_iterator
200CommandLine::find(const std::string & option_name) const
201{
202 auto pos = _cli_options.find(option_name);
203 auto it = _args.end();
204
205 if (pos != _cli_options.end())
206 {
207 for (const auto & search_string : pos->second.cli_switch)
208 {
209 auto it = std::find(_args.begin(), _args.end(), search_string);
210 if (it != _args.end())
211 return it;
212 }
213 }
214
215 return it;
216}

230bool
231CommandLine::search(const std::string & option_name)
232{
233 auto pos = _cli_options.find(option_name);
234 if (pos != _cli_options.end())
235 {
236 for (const auto & search_string : pos->second.cli_switch)
237 for (auto & arg : _args)
238 if (arg == search_string)
239 return true;
240
241 if (pos->second.required)
242 {
243 printUsage();
244 mooseError("Required parameter: ", option_name, " missing");
245 }
246 return false;
247 }
248 mooseError("Unrecognized option name: ", option_name);
249}

259void
260CommandLine::printUsage() const
261{
262 Moose::out << "Usage: " << getExecutableName() << " [<options>]\n\n"
263 << "Options:\n"
264 << std::left;
265
266 for (const auto & i : _cli_options)
267 {
268 if (i.second.cli_syntax.empty())
269 continue;
270
271 std::stringstream oss;
272 for (unsigned int j = 0; j < i.second.cli_syntax.size(); ++j)
273 {
274 if (j)
275 oss << " ";
276 oss << i.second.cli_syntax[j];
277 }
278 Moose::out << " " << std::setw(50) << oss.str() << i.second.description << "\n";
279 }
280
281 Moose::out << "\nSolver Options:\n"
282 << " See solver manual for details (Petsc or Trilinos)" << std::endl;
283}
