67void
68PerfGraphOutput::output()
69{
70 if (!_app.getParam<bool>("no_timing"))
71 {
72 _console << '\n';
73
74 perfGraph().print(_console, _level);
75
76 if (_heaviest_branch)
77 perfGraph().printHeaviestBranch(_console);
78
79 if (_heaviest_sections)
80 perfGraph().printHeaviestSections(_console, _heaviest_sections);
81
82 _console << std::flush;
83 }
84}
