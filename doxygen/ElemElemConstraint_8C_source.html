84void
85ElemElemConstraint::computeElemNeighResidual(Moose::DGResidualType type)
86{
87 bool is_elem;
88 if (type == Moose::Element)
89 is_elem = true;
90 else
91 is_elem = false;
92
93 const VariableTestValue & test_space = is_elem ? _test : _test_neighbor;
94 if (is_elem)
95 prepareVectorTag(_assembly, _var.number());
96 else
97 prepareVectorTagNeighbor(_assembly, _var.number());
98 for (_qp = 0; _qp < _constraint_q_point.size(); _qp++)
99 for (_i = 0; _i < test_space.size(); _i++)
100 _local_re(_i) += _constraint_weight[_qp] * computeQpResidual(type);
101 accumulateTaggedLocalResidual();
102}

114void
115ElemElemConstraint::computeElemNeighJacobian(Moose::DGJacobianType type)
116{
117 const VariableTestValue & test_space =
118 (type == Moose::ElementElement || type == Moose::ElementNeighbor) ? _test : _test_neighbor;
119 const VariableTestValue & loc_phi =
120 (type == Moose::ElementElement || type == Moose::NeighborElement) ? _phi : _phi_neighbor;
121 prepareMatrixTagNeighbor(_assembly, _var.number(), _var.number(), type);
122
123 for (_qp = 0; _qp < _constraint_q_point.size(); _qp++)
124 for (_i = 0; _i < test_space.size(); _i++)
125 for (_j = 0; _j < loc_phi.size(); _j++)
126 _local_ke(_i, _j) += _constraint_weight[_qp] * computeQpJacobian(type);
127
128 accumulateTaggedLocalMatrix();
129}
