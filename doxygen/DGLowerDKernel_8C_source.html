36 : DGKernel(parameters),
37 _lowerd_var(*getVar("lowerd_variable", 0)),
38 _lambda(_is_implicit ? _lowerd_var.slnLower() : _lowerd_var.slnLowerOld()),
39
40 _phi_lambda(_lowerd_var.phiLower()),
41 _test_lambda(_lowerd_var.phiLower())
42{

53
54 // Note: the above two conditions also ensure that the variable and lower-d variable are
55 // different.

76DGLowerDKernel::computeLowerDResidual()
77{
78 prepareVectorTagLower(_assembly, _lowerd_var.number());
79
80 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
81 {
82 initLowerDQpResidual();
83 for (_i = 0; _i < _test_lambda.size(); _i++)
84 _local_re(_i) += _JxW[_qp] * _coord[_qp] * computeLowerDQpResidual();
85 }

114void
115DGLowerDKernel::computeLowerDJacobian(Moose::ConstraintJacobianType type)
116{
117 mooseAssert(type != Moose::PrimaryPrimary && type != Moose::PrimarySecondary &&
118 type != Moose::SecondaryPrimary && type != Moose::SecondarySecondary,
119 "Jacobian types without lower should be handled in computeElemNeighJacobian");
120
121 const auto & test_space =
122 (type == Moose::LowerLower || type == Moose::LowerSecondary || type == Moose::LowerPrimary)
123 ? _test_lambda
124 : (type == Moose::PrimaryLower ? _test : _test_neighbor);
125 const auto ivar =
126 (type == Moose::LowerLower || type == Moose::LowerSecondary || type == Moose::LowerPrimary)
127 ? _lowerd_var.number()
128 : _var.number();
129
130 const auto & loc_phi =
131 (type == Moose::LowerLower || type == Moose::SecondaryLower || type == Moose::PrimaryLower)
132 ? _phi_lambda
133 : (type == Moose::LowerPrimary ? _phi : _phi_neighbor);
134 const auto jvar =
135 (type == Moose::LowerLower || type == Moose::SecondaryLower || type == Moose::PrimaryLower)
136 ? _lowerd_var.number()
137 : _var.number();
138
139 prepareMatrixTagLower(_assembly, ivar, jvar, type);
140
141 if (_local_ke.n() == 0 || _local_ke.m() == 0)
142 return;
143
144 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
145 {
146 initLowerDQpJacobian(type);
147 for (_i = 0; _i < test_space.size(); _i++)
148 for (_j = 0; _j < loc_phi.size(); _j++)
149 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeLowerDQpJacobian(type);
150 }
151
152 accumulateTaggedLocalMatrix();
153}

155void
156DGLowerDKernel::computeOffDiagJacobian(const unsigned int jvar_num)
157{
158 if (!excludeBoundary())
159 {
160 precalculateOffDiagJacobian(jvar_num);
161
162 if (jvar_num == variable().number())
163 {
164 computeElemNeighJacobian(Moose::ElementElement);
165 computeElemNeighJacobian(Moose::ElementNeighbor);
166 computeElemNeighJacobian(Moose::NeighborElement);
167 computeElemNeighJacobian(Moose::NeighborNeighbor);
168 computeLowerDJacobian(Moose::LowerSecondary);
169 computeLowerDJacobian(Moose::LowerPrimary);
170 }
171 else if (jvar_num == _lowerd_var.number())
172 {
173 computeLowerDJacobian(Moose::LowerLower);
174 computeLowerDJacobian(Moose::SecondaryLower);
175 computeLowerDJacobian(Moose::PrimaryLower);
176 }
177 else
178 {
179 const auto & jvar = getVariable(jvar_num);
180
181 // Compute element-element Jacobian
182 computeOffDiagElemNeighJacobian(Moose::ElementElement, jvar);
183
184 // Compute element-neighbor Jacobian
185 computeOffDiagElemNeighJacobian(Moose::ElementNeighbor, jvar);
186
187 // Compute neighbor-element Jacobian
188 computeOffDiagElemNeighJacobian(Moose::NeighborElement, jvar);
189
190 // Compute neighbor-neighbor Jacobian
191 computeOffDiagElemNeighJacobian(Moose::NeighborNeighbor, jvar);
192
193 // Compute the other five pieces of Jacobian related with lower-d variable
194 computeOffDiagLowerDJacobian(Moose::LowerLower, jvar);
195 computeOffDiagLowerDJacobian(Moose::LowerSecondary, jvar);
196 computeOffDiagLowerDJacobian(Moose::LowerPrimary, jvar);
197 computeOffDiagLowerDJacobian(Moose::SecondaryLower, jvar);
198 computeOffDiagLowerDJacobian(Moose::PrimaryLower, jvar);
199 }
200 }
201}

203void
204DGLowerDKernel::computeOffDiagLowerDJacobian(Moose::ConstraintJacobianType type,
205 const MooseVariableFEBase & jvar)
206{
207 mooseAssert(type != Moose::PrimaryPrimary && type != Moose::PrimarySecondary &&
208 type != Moose::SecondaryPrimary && type != Moose::SecondarySecondary,
209 "Jacobian types without lower should be handled in computeOffDiagElemNeighJacobian");
210
211 const auto & test_space =
212 (type == Moose::LowerLower || type == Moose::LowerSecondary || type == Moose::LowerPrimary)
213 ? _test_lambda
214 : (type == Moose::PrimaryLower ? _test : _test_neighbor);
215 const auto ivar =
216 (type == Moose::LowerLower || type == Moose::LowerSecondary || type == Moose::LowerPrimary)
217 ? _lowerd_var.number()
218 : _var.number();
219
220 prepareMatrixTagLower(_assembly, ivar, jvar.number(), type);
221 if (_local_ke.n() == 0 || _local_ke.m() == 0)
222 return;
223
224 if (jvar.fieldType() == Moose::VarFieldType::VAR_FIELD_STANDARD)
225 {
226 const auto & jv0 = static_cast<const MooseVariable &>(jvar);
227 const VariableTestValue & loc_phi =
228 (type == Moose::LowerLower || type == Moose::SecondaryLower || type == Moose::PrimaryLower)
229 ? jv0.phiLower()
230 : (type == Moose::LowerPrimary ? jv0.phiFace() : jv0.phiFaceNeighbor());
231
232 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
233 {
234 initLowerDQpOffDiagJacobian(type, jvar);
235 for (_i = 0; _i < test_space.size(); _i++)
236 for (_j = 0; _j < loc_phi.size(); _j++)
237 _local_ke(_i, _j) += _JxW[_qp] * _coord[_qp] * computeLowerDQpOffDiagJacobian(type, jvar);
238 }
239 }
240 else if (jvar.fieldType() == Moose::VarFieldType::VAR_FIELD_ARRAY)
241 mooseError("Array variable cannot be coupled into DG kernel currently");
242 else
243 mooseError("Vector variable cannot be coupled into DG kernel currently");
244
245 accumulateTaggedLocalMatrix();
246}
