39void
40assemble_matrix(EquationSystems & es, const std::string & system_name)
41{
42 EigenProblem * p = es.parameters.get<EigenProblem *>("_eigen_problem");
43 EigenSystem & eigen_system = es.get_system<EigenSystem>(system_name);
44 NonlinearEigenSystem & eigen_nl = p->getNonlinearEigenSystem(/*nl_sys_num=*/0);
45
46 // If this is a nonlinear eigenvalue problem,
47 // we do not need to assemble anything
48 if (p->isNonlinearEigenvalueSolver())
49 {
50 // If you want an efficient eigensolver,
51 // please use PETSc 3.13 or newer.
52 // We need to do an unnecessary assembly,
53 // if you use PETSc that is older than 3.13.
54#if PETSC_RELEASE_LESS_THAN(3, 13, 0)
55 if (eigen_system.has_matrix_B())
56 p->computeJacobianTag(*eigen_system.current_local_solution,
57 eigen_system.get_matrix_B(),
58 eigen_nl.eigenMatrixTag());
59#endif
60 return;
61 }
62
63#if !PETSC_RELEASE_LESS_THAN(3, 13, 0)
64 // If we use shell matrices and do not use a shell preconditioning matrix,
65 // we only need to form a preconditioning matrix
66 if (eigen_system.use_shell_matrices() && !eigen_system.use_shell_precond_matrix())
67 {
68 p->computeJacobianTag(*eigen_system.current_local_solution,
69 eigen_system.get_precond_matrix(),
70 eigen_nl.precondMatrixTag());
71 return;
72 }
73#endif
74 // If it is a linear generalized eigenvalue problem,
75 // we assemble A and B together
76 if (eigen_system.generalized())
77 {
78 p->computeJacobianAB(*eigen_system.current_local_solution,
79 eigen_system.get_matrix_A(),
80 eigen_system.get_matrix_B(),
81 eigen_nl.nonEigenMatrixTag(),
82 eigen_nl.eigenMatrixTag());
83#if LIBMESH_HAVE_SLEPC
84 if (p->negativeSignEigenKernel())
85 MatScale(static_cast<PetscMatrix<Number> &>(eigen_system.get_matrix_B()).mat(), -1.0);
86#endif
87 return;
88 }
89
90 // If it is a linear eigenvalue problem, we assemble matrix A
91 {
92 p->computeJacobianTag(*eigen_system.current_local_solution,
93 eigen_system.get_matrix_A(),
94 eigen_nl.nonEigenMatrixTag());
95
96 return;
97 }
98}
99}

101NonlinearEigenSystem::NonlinearEigenSystem(EigenProblem & eigen_problem, const std::string & name)
102 : NonlinearSystemBase(eigen_problem, eigen_problem.es().add_system<EigenSystem>(name), name),
103 _eigen_sys(eigen_problem.es().get_system<EigenSystem>(name)),
104 _eigen_problem(eigen_problem),
105 _solver_configuration(nullptr),
106 _n_eigen_pairs_required(eigen_problem.getNEigenPairsRequired()),
107 _work_rhs_vector_AX(addVector("work_rhs_vector_Ax", false, PARALLEL)),
108 _work_rhs_vector_BX(addVector("work_rhs_vector_Bx", false, PARALLEL)),
109 _precond_matrix_includes_eigen(false),
110 _preconditioner(nullptr)
111{
112 sys().attach_assemble_function(Moose::assemble_matrix);
113
114 SlepcEigenSolver<Number> * solver =
115 cast_ptr<SlepcEigenSolver<Number> *>(_eigen_sys.eigen_solver.get());
116
117 if (!solver)
118 mooseError("A slepc eigen solver is required");
119
120 // setup of our class @SlepcSolverConfiguration
121 _solver_configuration = std::make_unique<SlepcEigenSolverConfiguration>(eigen_problem, *solver);
122
123 solver->set_solver_configuration(*_solver_configuration);
124
125 _Ax_tag = eigen_problem.addVectorTag("Ax_tag");
126
127 _Bx_tag = eigen_problem.addVectorTag("Eigen");
128
129 _A_tag = eigen_problem.addMatrixTag("A_tag");
130
131 _B_tag = eigen_problem.addMatrixTag("Eigen");
132
133 // By default, _precond_tag and _A_tag will share the same
134 // objects. If we want to include eigen contributions to
135 // the preconditioning matrix, and then _precond_tag will
136 // point to part of "B" objects
137 _precond_tag = eigen_problem.addMatrixTag("Eigen_precond");
138}

140void
141NonlinearEigenSystem::postAddResidualObject(ResidualObject & object)
142{
143 // If it is an eigen dirichlet boundary condition, we should skip it because their
144 // contributions should be zero. If we do not skip it, preconditioning matrix will
145 // be singular because boundary elements are zero.
146 if (_precond_matrix_includes_eigen && !dynamic_cast<EigenDirichletBC *>(&object) &&
147 !dynamic_cast<EigenArrayDirichletBC *>(&object))
148 object.useMatrixTag(_precond_tag, {});
149
150 auto & vtags = object.getVectorTags({});
151 auto & mtags = object.getMatrixTags({});
152 // If it is an eigen kernel, mark its variable as eigen
153 if (vtags.find(_Bx_tag) != vtags.end() || mtags.find(_B_tag) != mtags.end())
154 {
155 // Note: the object may be on the displaced system
156 auto sys = object.parameters().get<SystemBase *>("_sys");
157 auto vname = object.variable().name();
158 if (hasScalarVariable(vname))
159 sys->getScalarVariable(0, vname).eigen(true);
160 else
161 sys->getVariable(0, vname).eigen(true);
162 }
163
164 // If this is not an eigen kernel
165 // If there is no vector eigen tag and no matrix eigen tag,
166 // then we consider this as noneigen kernel
167 if (vtags.find(_Bx_tag) == vtags.end() && mtags.find(_B_tag) == mtags.end())
168 {
169 // Noneigen Vector tag
170 object.useVectorTag(_Ax_tag, {});
171 // Noneigen Matrix tag
172 object.useMatrixTag(_A_tag, {});
173 // Noneigen Kernels
174 object.useMatrixTag(_precond_tag, {});
175 }
176 else
177 {
178 // Associate the eigen matrix tag and the vector tag
179 // if this is a eigen kernel
180 object.useMatrixTag(_B_tag, {});
181 object.useVectorTag(_Bx_tag, {});
182 }
183}

185void
186NonlinearEigenSystem::solve()
187{
188 // Clear the iteration counters
189 _current_l_its.clear();
190 _current_nl_its = 0;
191 // Initialize the solution vector using a predictor and known values from nodal bcs
192 setInitialSolution();
193
194 // Now that the initial solution has ben set, potentially perform a residual/Jacobian evaluation
195 // to determine variable scaling factors
196 if (_automatic_scaling)
197 {
198 if (_compute_scaling_once)
199 {
200 if (!_computed_scaling)
201 {
202 computeScaling();
203 _computed_scaling = true;
204 }
205 }
206 else
207 computeScaling();
208 }
209 // We do not know a priori what variable a global degree of freedom corresponds to, so we need a
210 // map from global dof to scaling factor. We just use a ghosted NumericVector for that mapping
211 assembleScalingVector();
212
213// In DEBUG mode, Libmesh will check the residual automatically. This may cause
214// an error because B does not need to assembly by default.
215// When PETSc is older than 3.13, we always need to do an extra assembly,
216// so we do not do "close" here
217#if DEBUG && !PETSC_RELEASE_LESS_THAN(3, 13, 0)
218 if (sys().has_matrix_B())
219 sys().get_matrix_B().close();
220#endif
221
222 // We apply initial guess for only nonlinear solver
223 if (_eigen_problem.isNonlinearEigenvalueSolver())
224 _eigen_sys.set_initial_space(solution());
225
226 // Solve the transient problem if we have a time integrator; the
227 // steady problem if not.
228 if (_time_integrator)
229 {
230 _time_integrator->solve();
231 _time_integrator->postSolve();
232 }
233 else
234 system().solve();
235
236 // store eigenvalues
237 unsigned int n_converged_eigenvalues = getNumConvergedEigenvalues();
238
239 _n_eigen_pairs_required = _eigen_problem.getNEigenPairsRequired();
240
241 if (_n_eigen_pairs_required < n_converged_eigenvalues)
242 n_converged_eigenvalues = _n_eigen_pairs_required;
243
244 _eigen_values.resize(n_converged_eigenvalues);
245 for (unsigned int n = 0; n < n_converged_eigenvalues; n++)
246 _eigen_values[n] = getConvergedEigenvalue(n);
247
248 // Update the solution vector to the active eigenvector
249 if (n_converged_eigenvalues)
250 getConvergedEigenpair(_eigen_problem.activeEigenvalueIndex());
251}

253void
254NonlinearEigenSystem::attachSLEPcCallbacks()
255{
256 // Tell libmesh not close matrices before solve
257 _eigen_sys.get_eigen_solver().set_close_matrix_before_solve(false);
258
259 // Matrix A
260 if (_eigen_sys.has_matrix_A())
261 {
262 Mat mat = static_cast<PetscMatrix<Number> &>(_eigen_sys.get_matrix_A()).mat();
263
264 Moose::SlepcSupport::attachCallbacksToMat(_eigen_problem, mat, false);
265 }
266
267 // Matrix B
268 if (_eigen_sys.has_matrix_B())
269 {
270 Mat mat = static_cast<PetscMatrix<Number> &>(_eigen_sys.get_matrix_B()).mat();
271
272 Moose::SlepcSupport::attachCallbacksToMat(_eigen_problem, mat, true);
273 }
274
275 // Shell matrix A
276 if (_eigen_sys.has_shell_matrix_A())
277 {
278 Mat mat = static_cast<PetscShellMatrix<Number> &>(_eigen_sys.get_shell_matrix_A()).mat();
279
280 // Attach callbacks for nonlinear eigenvalue solver
281 Moose::SlepcSupport::attachCallbacksToMat(_eigen_problem, mat, false);
282
283 // Set MatMult operations for shell
284 Moose::SlepcSupport::setOperationsForShellMat(_eigen_problem, mat, false);
285 }
286
287 // Shell matrix B
288 if (_eigen_sys.has_shell_matrix_B())
289 {
290 Mat mat = static_cast<PetscShellMatrix<Number> &>(_eigen_sys.get_shell_matrix_B()).mat();
291
292 Moose::SlepcSupport::attachCallbacksToMat(_eigen_problem, mat, true);
293
294 // Set MatMult operations for shell
295 Moose::SlepcSupport::setOperationsForShellMat(_eigen_problem, mat, true);
296 }
297
298 // Preconditioning matrix
299 if (_eigen_sys.has_precond_matrix())
300 {
301 Mat mat = static_cast<PetscMatrix<Number> &>(_eigen_sys.get_precond_matrix()).mat();
302
303 Moose::SlepcSupport::attachCallbacksToMat(_eigen_problem, mat, true);
304 }
305
306 // Shell preconditioning matrix
307 if (_eigen_sys.has_shell_precond_matrix())
308 {
309 Mat mat = static_cast<PetscShellMatrix<Number> &>(_eigen_sys.get_shell_precond_matrix()).mat();
310
311 Moose::SlepcSupport::attachCallbacksToMat(_eigen_problem, mat, true);
312 }

392void
393NonlinearEigenSystem::checkIntegrity()
394{
395 if (_nodal_bcs.hasActiveObjects())
396 {
397 const auto & nodal_bcs = _nodal_bcs.getActiveObjects();
398 for (const auto & nodal_bc : nodal_bcs)
399 {
400 // If this is a dirichlet boundary condition
401 auto nbc = std::dynamic_pointer_cast<DirichletBC>(nodal_bc);
402 // If this is a eigen Dirichlet boundary condition
403 auto eigen_nbc = std::dynamic_pointer_cast<EigenDirichletBC>(nodal_bc);
404 // ArrayDirichletBC
405 auto anbc = std::dynamic_pointer_cast<ArrayDirichletBC>(nodal_bc);
406 // EigenArrayDirichletBC
407 auto aeigen_nbc = std::dynamic_pointer_cast<EigenArrayDirichletBC>(nodal_bc);
408 // If it is a Dirichlet boundary condition, then value has to be zero
409 if (nbc && nbc->variable().eigen() && nbc->getParam<Real>("value"))
410 mooseError(
411 "Can't set an inhomogeneous Dirichlet boundary condition for eigenvalue problems.");
412 // If it is an array Dirichlet boundary condition, all values should be zero
413 else if (anbc)
414 {
415 auto & values = anbc->getParam<RealEigenVector>("values");
416 for (MooseIndex(values) i = 0; i < values.size(); i++)
417 {
418 if (values(i))
419 mooseError("Can't set an inhomogeneous array Dirichlet boundary condition for "
420 "eigenvalue problems.");
421 }
422 }
423 else if (!nbc && !eigen_nbc && !anbc && !aeigen_nbc)
424 mooseError(
425 "Invalid NodalBC for eigenvalue problems, please use homogeneous (array) Dirichlet.");
426 }
427 }
