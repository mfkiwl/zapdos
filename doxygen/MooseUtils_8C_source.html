161// this implementation is copied from
162// https://en.wikibooks.org/wiki/Algorithm_Implementation/Strings/Levenshtein_distance#C.2B.2B
163int
164levenshteinDist(const std::string & s1, const std::string & s2)
165{
166 // To change the type this function manipulates and returns, change
167 // the return type and the types of the two variables below.
168 auto s1len = s1.size();
169 auto s2len = s2.size();
170
171 auto column_start = (decltype(s1len))1;
172
173 auto column = new decltype(s1len)[s1len + 1];
174 std::iota(column + column_start, column + s1len + 1, column_start);
175
176 for (auto x = column_start; x <= s2len; x++)
177 {
178 column[0] = x;
179 auto last_diagonal = x - column_start;
180 for (auto y = column_start; y <= s1len; y++)
181 {
182 auto old_diagonal = column[y];
183 auto possibilities = {
184 column[y] + 1, column[y - 1] + 1, last_diagonal + (s1[y - 1] == s2[x - 1] ? 0 : 1)};
185 column[y] = std::min(possibilities);
186 last_diagonal = old_diagonal;
187 }
188 }
189 auto result = column[s1len];
190 delete[] column;
191 return result;
192}

194void
195escape(std::string & str)
196{
197 std::map<char, std::string> escapes;
198 escapes['\a'] = "\\a";
199 escapes['\b'] = "\\b";
200 escapes['\f'] = "\\f";
201 escapes['\n'] = "\\n";
202 escapes['\t'] = "\\t";
203 escapes['\v'] = "\\v";
204 escapes['\r'] = "\\r";
205
206 for (const auto & it : escapes)
207 for (size_t pos = 0; (pos = str.find(it.first, pos)) != std::string::npos;
208 pos += it.second.size())
209 str.replace(pos, 1, it.second);
210}

255bool
256checkFileReadable(const std::string & filename,
257 bool check_line_endings,
258 bool throw_on_unreadable,
259 bool check_for_git_lfs_pointer)
260{
261 std::ifstream in(filename.c_str(), std::ifstream::in);
262 if (in.fail())
263 {
264 if (throw_on_unreadable)
265 mooseError(
266 (std::string("Unable to open file \"") + filename +
267 std::string("\". Check to make sure that it exists and that you have read permission."))
268 .c_str());
269 else
270 return false;
271 }
272
273 if (check_line_endings)
274 {
275 std::istream_iterator<char> iter(in);
276 std::istream_iterator<char> eos;
277 in >> std::noskipws;
278 while (iter != eos)
279 if (*iter++ == '\r')
280 mooseError(filename + " contains Windows(DOS) line endings which are not supported.");
281 }
282
283 if (check_for_git_lfs_pointer && checkForGitLFSPointer(in))
284 mooseError(filename + " appears to be a Git-LFS pointer. Make sure you have \"git-lfs\" "
285 "installed so that you may pull this file.");
286 in.close();
287
288 return true;
289}

291bool
292checkForGitLFSPointer(std::ifstream & file)
293{
294 mooseAssert(file.is_open(), "Passed in file handle is not open");
295
296 std::string line;
297
298 // git-lfs pointer files contain several name value pairs. The specification states that the
299 // first name/value pair must be "version {url}". We'll do a simplified check for that.
300 file.seekg(0);
301 std::getline(file, line);
302 if (line.find("version https://") != std::string::npos)
303 return true;
304 else
305 return false;
306}

328void
329parallelBarrierNotify(const Parallel::Communicator & comm, bool messaging)
330{
331 processor_id_type secondary_processor_id;
332
333 if (messaging)
334 Moose::out << "Waiting For Other Processors To Finish" << std::endl;
335 if (comm.rank() == 0)
336 {
337 // The primary process is already through, so report it
338 if (messaging)
339 Moose::out << "Jobs complete: 1/" << comm.size() << (1 == comm.size() ? "\n" : "\r")
340 << std::flush;
341 for (unsigned int i = 2; i <= comm.size(); ++i)
342 {
343 comm.receive(MPI_ANY_SOURCE, secondary_processor_id);
344 if (messaging)
345 Moose::out << "Jobs complete: " << i << "/" << comm.size()
346 << (i == comm.size() ? "\n" : "\r") << std::flush;
347 }
348 }
349 else
350 {
351 secondary_processor_id = comm.rank();
352 comm.send(0, secondary_processor_id);
353 }
354
355 comm.barrier();
356}

358void
359serialBegin(const libMesh::Parallel::Communicator & comm, bool warn)
360{
361 // unless we are the first processor...
362 if (comm.rank() > 0)
363 {
364 // ...wait for the previous processor to finish
365 int dummy = 0;
366 comm.receive(comm.rank() - 1, dummy);
367 }
368 else if (warn)
369 mooseWarning("Entering serial execution block (use only for debugging)");
370}
371
372void
373serialEnd(const libMesh::Parallel::Communicator & comm, bool warn)
374{
375 // unless we are the last processor...
376 if (comm.rank() + 1 < comm.size())
377 {
378 // ...notify the next processor of its turn
379 int dummy = 0;
380 comm.send(comm.rank() + 1, dummy);
381 }
382
383 comm.barrier();
384 if (comm.rank() == 0 && warn)
385 mooseWarning("Leaving serial execution block (use only for debugging)");
386}

388bool
389hasExtension(const std::string & filename, std::string ext, bool strip_exodus_ext)
390{
391 // Extract the extension, w/o the '.'
392 std::string file_ext;
393 if (strip_exodus_ext)
394 {
395 pcrecpp::RE re(
396 ".*\\.([^\\.]*?)(?:-s\\d+)?\\s*$"); // capture the complete extension, ignoring -s*
397 re.FullMatch(filename, &file_ext);
398 }
399 else
400 {
401 pcrecpp::RE re(".*\\.([^\\.]*?)\\s*$"); // capture the complete extension
402 re.FullMatch(filename, &file_ext);
403 }
404
405 // Perform the comparision
406 if (file_ext == ext)
407 return true;
408 else
409 return false;
410}

421std::pair<std::string, std::string>
422splitFileName(std::string full_file)
423{
424 // Error if path ends with /
425 if (full_file.empty() || *full_file.rbegin() == '/')
426 mooseError("Invalid full file name: ", full_file);
427
428 // Define the variables to output
429 std::string path;
430 std::string file;
431
432 // Locate the / sepearting the file from path
433 std::size_t found = full_file.find_last_of("/");
434
435 // If no / is found used "." for the path, otherwise seperate the two
436 if (found == std::string::npos)
437 {
438 path = ".";
439 file = full_file;
440 }
441 else
442 {
443 path = full_file.substr(0, found);
444 file = full_file.substr(found + 1);
445 }
446
447 // Return the path and file as a pair
448 return std::pair<std::string, std::string>(path, file);
449}

464void
465makedirs(const std::string & dir_name, bool throw_on_failure)
466{
467 // split path into directories with delimiter '/'
468 std::vector<std::string> split_dir_names;
469 MooseUtils::tokenize(dir_name, split_dir_names);
470
471 auto n = split_dir_names.size();
472
473 // remove '.' and '..' when possible
474 auto i = n;
475 i = 0;
476 while (i != n)
477 {
478 if (split_dir_names[i] == ".")
479 {
480 for (auto j = i + 1; j < n; ++j)
481 split_dir_names[j - 1] = split_dir_names[j];
482 --n;
483 }
484 else if (i > 0 && split_dir_names[i] == ".." && split_dir_names[i - 1] != "..")
485 {
486 for (auto j = i + 1; j < n; ++j)
487 split_dir_names[j - 2] = split_dir_names[j];
488 n -= 2;
489 --i;
490 }
491 else
492 ++i;
493 }
494 if (n == 0)
495 return;
496
497 split_dir_names.resize(n);
498
499 // start creating directories recursively
500 std::string cur_dir = dir_name[0] == '/' ? "" : ".";
501 for (auto & dir : split_dir_names)
502 {
503 cur_dir += "/" + dir;
504
505 if (!pathExists(cur_dir))
506 {
507 auto code = Utility::mkdir(cur_dir.c_str());
508 if (code != 0)
509 {
510 std::string msg = "Failed creating directory " + dir_name;
511 if (throw_on_failure)
512 throw std::invalid_argument(msg);
513 else
514 mooseError(msg);
515 }
516 }
517 }
518}

521removedirs(const std::string & dir_name, bool throw_on_failure)
522{
523 // split path into directories with delimiter '/'
524 std::vector<std::string> split_dir_names;
525 MooseUtils::tokenize(dir_name, split_dir_names);
526
527 auto n = split_dir_names.size();
528
529 // remove '.' and '..' when possible
530 auto i = n;
531 i = 0;
532 while (i != n)
533 {
534 if (split_dir_names[i] == ".")
535 {
536 for (auto j = i + 1; j < n; ++j)
537 split_dir_names[j - 1] = split_dir_names[j];
538 --n;
539 }
540 else if (i > 0 && split_dir_names[i] == ".." && split_dir_names[i - 1] != "..")
541 {
542 for (auto j = i + 1; j < n; ++j)
543 split_dir_names[j - 2] = split_dir_names[j];
544 n -= 2;
545 --i;
546 }
547 else
548 ++i;
549 }
550 if (n == 0)
551 return;
552
553 split_dir_names.resize(n);
554
555 // start removing directories recursively
556 std::string base_dir = dir_name[0] == '/' ? "" : ".";
557 for (i = n; i > 0; --i)
558 {
559 std::string cur_dir = base_dir;
560 auto j = i;
561 for (j = 0; j < i; ++j)
562 cur_dir += "/" + split_dir_names[j];
563
564 // listDir should return at least '.' and '..'
565 if (pathExists(cur_dir) && listDir(cur_dir).size() == 2)
566 {
567 auto code = rmdir(cur_dir.c_str());
568 if (code != 0)
569 {
570 std::string msg = "Failed removing directory " + dir_name;
571 if (throw_on_failure)
572 throw std::invalid_argument(msg);
573 else
574 mooseError(msg);
575 }
576 }
577 else
578 // stop removing
579 break;
580 }
581}

584camelCaseToUnderscore(const std::string & camel_case_name)
585{
586 std::string replaced = camel_case_name;

595std::string
596underscoreToCamelCase(const std::string & underscore_name, bool leading_upper_case)
597{
598 pcrecpp::StringPiece input(underscore_name);
599 pcrecpp::RE re("([^_]*)(_|$)");
600
601 std::string result;
602 std::string us, not_us;
603 bool make_upper = leading_upper_case;
604 while (re.Consume(&input, &not_us, &us))
605 {
606 if (not_us.length() > 0)
607 {
608 if (make_upper)
609 {
610 result += std::toupper(not_us[0]);
611 if (not_us.length() > 1)
612 result += not_us.substr(1);
613 }
614 else
615 result += not_us;
616 }
617 if (us == "")
618 break;
619
620 // Toggle flag so next match is upper cased
621 make_upper = true;
622 }
623
624 return result;
625}

655unsigned short
656getTermWidth(bool use_environment)
657{
658#ifndef __WIN32__
659 struct winsize w;
660#else
661 struct
662 {
663 unsigned short ws_col;
664 } w;
665#endif
670 w.ws_col = std::numeric_limits<unsigned short>::max();
671
672 if (use_environment)
673 {
674 char * pps_width = std::getenv("MOOSE_PPS_WIDTH");
675 if (pps_width != NULL)
676 {
677 std::stringstream ss(pps_width);
678 ss >> w.ws_col;
679 }
680 }
681 // Default to AUTO if no environment variable was set
682 if (w.ws_col == std::numeric_limits<unsigned short>::max())
683 {
684#ifndef __WIN32__
685 try
686 {
687 ioctl(0, TIOCGWINSZ, &w);
688 }
689 catch (...)
690#endif
691 {
692 }
693 }
694
695 // Something bad happened, make sure we have a sane value
696 // 132 seems good for medium sized screens, and is available as a GNOME preset
697 if (w.ws_col == std::numeric_limits<unsigned short>::max())
698 w.ws_col = 132;
699
700 return w.ws_col;
701}

753void
754indentMessage(const std::string & prefix,
755 std::string & message,
756 const char * color /*= COLOR_CYAN*/,
757 bool indent_first_line,
758 const std::string & post_prefix)
759{
760 // First we need to see if the message we need to indent (with color) also contains color codes
761 // that span lines.
762 // The code matches all of the XTERM constants (see XTermConstants.h). If it does, then we'll work
763 // on formatting
764 // each colored multiline chunk one at a time with the right codes.
765 std::string colored_message;
766 std::string curr_color = COLOR_DEFAULT; // tracks last color code before newline
767 std::string line, color_code;
768
769 bool ends_in_newline = message.empty() ? true : message.back() == '\n';
770
771 bool first = true;
772
773 std::istringstream iss(message);
774 for (std::string line; std::getline(iss, line);) // loop over each line
775 {
776 const static pcrecpp::RE match_color(".*(\\33\\[3\\dm)((?!\\33\\[3\\d)[^\n])*");
777 pcrecpp::StringPiece line_piece(line);
778 match_color.FindAndConsume(&line_piece, &color_code);
779
780 if (!first || indent_first_line)
781 colored_message += color + prefix + post_prefix + curr_color;
782
783 colored_message += line;
784
785 // Only add a newline to the last line if it had one to begin with!
786 if (!iss.eof() || ends_in_newline)
787 colored_message += "\n";
788
789 if (!color_code.empty())
790 curr_color = color_code; // remember last color of this line
791
792 first = false;
793 }
794 message = colored_message;
795}

834std::string
835getLatestCheckpointFilePrefix(const std::list<std::string> & checkpoint_files)
836{
837 // Create storage for newest restart files
838 // Note that these might have the same modification time if the simulation was fast.
839 // In that case we're going to save all of the "newest" files and sort it out momentarily
840 std::time_t newest_time = 0;
841 std::list<std::string> newest_restart_files;
842
843 // Loop through all possible files and store the newest
844 for (const auto & cp_file : checkpoint_files)
845 {
846 if (MooseUtils::hasExtension(cp_file, "rd"))
847 {
848 struct stat stats;
849 stat(cp_file.c_str(), &stats);
850
851 std::time_t mod_time = stats.st_mtime;
852 if (mod_time > newest_time)
853 {
854 newest_restart_files.clear(); // If the modification time is greater, clear the list
855 newest_time = mod_time;
856 }
857
858 if (mod_time == newest_time)
859 newest_restart_files.push_back(cp_file);
860 }
861 }
862
863 // Loop through all of the newest files according the number in the file name
864 int max_file_num = -1;
865 std::string max_file;
866 std::string max_prefix;
867
868 // Pull out the path including the number and the number itself
869 // This takes something_blah_out_cp/0024-restart-1.rd
870 // and returns "something_blah_out_cp/0024" as the "prefix"
871 // and then "24" as the number itself
872 pcrecpp::RE re_file_num("(.*?(\\d+))-restart-\\d+.rd$");
873
874 // Now, out of the newest files find the one with the largest number in it
875 for (const auto & res_file : newest_restart_files)
876 {
877 int file_num = 0;
878
879 // All of the file up to and including the digits
880 std::string file_prefix;
881
882 re_file_num.FullMatch(res_file, &file_prefix, &file_num);
883
884 if (file_num > max_file_num)
885 {
886 // Need both the header and the data
887 if (!RestartableDataReader::isAvailable(res_file))
888 continue;
889
890 max_file_num = file_num;
891 max_file = res_file;
892 max_prefix = file_prefix;
893 }
894 }
895
896 // Error if nothing was located
897 if (max_file_num == -1)
898 mooseError("No checkpoint file found!");
899
900 return max_prefix;
901}

903bool
904wildCardMatch(std::string name, std::string search_string)
905{
906 // Assume that an empty string matches anything
907 if (search_string == "")
908 return true;
909
910 // transform to lower for case insenstive matching
911 std::transform(name.begin(), name.end(), name.begin(), (int (*)(int))std::toupper);
912 std::transform(search_string.begin(),
913 search_string.end(),
914 search_string.begin(),
915 (int (*)(int))std::toupper);
916
917 // exact match!
918 if (search_string.find("*") == std::string::npos)
919 return search_string == name;
920
921 // wildcard
922 std::vector<std::string> tokens;
923 MooseUtils::tokenize(search_string, tokens, 1, "*");
924
925 size_t pos = 0;
926 for (unsigned int i = 0; i < tokens.size() && pos != std::string::npos; ++i)
927 {
928 pos = name.find(tokens[i], pos);
929 // See if we have a leading wildcard
930 if (search_string[0] != '*' && i == 0 && pos != 0)
931 return false;
932 }
933
934 if (pos != std::string::npos && tokens.size() > 0)
935 {
936 // Now see if we have a trailing wildcard
937 size_t last_token_length = tokens.back().length();
938 if (*search_string.rbegin() == '*' || pos == name.size() - last_token_length)
939 return true;
940 else
941 return false;
942 }
943 else
944 return false;
945}

947bool
948globCompare(const std::string & candidate,
949 const std::string & pattern,
950 std::size_t c,
951 std::size_t p)
952{
953 if (p == pattern.size())
954 return c == candidate.size();
955
956 if (pattern[p] == '*')
957 {
958 for (; c < candidate.size(); ++c)
959 if (globCompare(candidate, pattern, c, p + 1))
960 return true;
961 return globCompare(candidate, pattern, c, p + 1);
962 }
963
964 if (pattern[p] != '?' && pattern[p] != candidate[c])
965 return false;
966
967 return globCompare(candidate, pattern, c + 1, p + 1);
968}

970template <typename T>
971T
972convertStringToInt(const std::string & str, bool throw_on_failure)
973{
974 T val;
975
976 // Let's try to read a double and see if we can cast it to an int
977 // This would be the case for scientific notation
978 long double double_val;
979 std::stringstream double_ss(str);
980 double_ss >> double_val;
981
982 // on arm64 the long double does not have sufficient precission
983 bool use_int = false;
984 std::stringstream int_ss(str);
985 if (!(int_ss >> val).fail() && int_ss.eof())
986 use_int = true;
987
988 if (double_ss.fail() || !double_ss.eof())
989 {
990 std::string msg =
991 std::string("Unable to convert '") + str + "' to type " + demangle(typeid(T).name());
992
993 if (throw_on_failure)
994 throw std::invalid_argument(msg);
995 else
996 mooseError(msg);
997 }
998
999 // Check to see if it's an integer (and within range of an integer)
1000 if (double_val == static_cast<T>(double_val))
1001 return use_int ? val : static_cast<T>(double_val);
1002
1003 // Still failure
1004 std::string msg =
1005 std::string("Unable to convert '") + str + "' to type " + demangle(typeid(T).name());
1006
1007 if (throw_on_failure)
1008 throw std::invalid_argument(msg);
1009 else
1010 mooseError(msg);
1011}

1097void
1098linearPartitionItems(dof_id_type num_items,
1099 dof_id_type num_chunks,
1100 dof_id_type chunk_id,
1101 dof_id_type & num_local_items,
1102 dof_id_type & local_items_begin,
1103 dof_id_type & local_items_end)
1104{
1105 auto global_num_local_items = num_items / num_chunks;
1106
1107 num_local_items = global_num_local_items;
1108
1109 auto leftovers = num_items % num_chunks;
1110
1111 if (chunk_id < leftovers)
1112 {
1113 num_local_items++;
1114 local_items_begin = num_local_items * chunk_id;
1115 }
1116 else
1117 local_items_begin =
1118 (global_num_local_items + 1) * leftovers + global_num_local_items * (chunk_id - leftovers);
1119
1120 local_items_end = local_items_begin + num_local_items;
1121}
1122
1123processor_id_type
1124linearPartitionChunk(dof_id_type num_items, dof_id_type num_chunks, dof_id_type item_id)
1125{
1126 auto global_num_local_items = num_items / num_chunks;
1127
1128 auto leftovers = num_items % num_chunks;
1129
1130 auto first_item_past_first_part = leftovers * (global_num_local_items + 1);
1131
1132 // Is it in the first section (that gets an extra item)
1133 if (item_id < first_item_past_first_part)
1134 return item_id / (global_num_local_items + 1);
1135 else
1136 {
1137 auto new_item_id = item_id - first_item_past_first_part;
1138
1139 // First chunk after the first section + the number of chunks after that
1140 return leftovers + (new_item_id / global_num_local_items);
1141 }
1142}

1146{

1152 pos = str.find(delimiter, prev);
1153 output.push_back(str.substr(prev, pos - prev));
1154 prev = pos + delimiter.length();
1155 count += 1;
1156 } while (pos != std::string::npos && count < max_count);
1157
1158 if (pos != std::string::npos)
1159 output.push_back(str.substr(prev));
1160
