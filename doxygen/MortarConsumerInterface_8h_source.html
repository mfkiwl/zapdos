192template <typename Variables, typename DualNumbers>
193void
194MortarConsumerInterface::trimInteriorNodeDerivatives(
195 const std::map<unsigned int, unsigned int> & domain_ip_lowerd_map,
196 const Variables & moose_vars,
197 DualNumbers & dual_numbers,
198 const bool is_secondary)
199{
200 //
201 // Remove interior node variable's derivatives from AD objects.
202 //
203
204 mooseAssert(moose_vars.size(), "Should have passed at least one variable");
205 const auto num_indices = is_secondary ? moose_vars[0]->dofIndices().size()
206 : moose_vars[0]->dofIndicesNeighbor().size();
207#ifdef DEBUG
208 for (const auto i : make_range(std::size_t(1), moose_vars.size()))
209 if (auto * moose_var = moose_vars[i])
210 mooseAssert(is_secondary ? moose_var->dofIndices().size()
211 : moose_var->dofIndicesNeighbor().size() == num_indices,
212 "These must be the same for all passed in variables");
213#endif
214
215 for (const auto dof_index : make_range(num_indices))
216 if (!domain_ip_lowerd_map.count(dof_index))
217 {
218 for (const auto * const moose_var : moose_vars)
219 {
220 // It's valid for a user to pass a container that represents a LIBMESH_DIM vector of
221 // component variables for which one or two of the variables may be null depending on the
222 // mesh dimension in the simulation
223 if (!moose_var)
224 continue;
225
226 mooseAssert(moose_var->isNodal(),
227 "Trimming of interior node's derivatives is only supported for Lagrange "
228 "elements in mortar objects");
229
230 const auto remove_derivative_index = is_secondary
231 ? moose_var->dofIndices()[dof_index]
232 : moose_var->dofIndicesNeighbor()[dof_index];
233 for (auto & dual_number : dual_numbers)
234 trimDerivative(remove_derivative_index, dual_number);
235 }
236 }
237}
