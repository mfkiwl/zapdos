78void
79ArrayDGLowerDKernel::computeLowerDResidual()
80{
81 prepareVectorTagLower(_assembly, _lowerd_var.number());
82
83 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
84 {
85 initLowerDQpResidual();
86 for (_i = 0; _i < _test_lambda.size(); _i++)
87 {
88 _work_vector.setZero();
89 computeLowerDQpResidual(_work_vector);
90 mooseAssert(_work_vector.size() == _count,
91 "Size of local residual is not equal to the number of array variable compoments");
92 _work_vector *= _JxW[_qp] * _coord[_qp];
93 _assembly.saveLocalArrayResidual(_local_re, _i, _test_lambda.size(), _work_vector);
94 }
95 }
96
97 accumulateTaggedLocalResidual();
98}

124void
125ArrayDGLowerDKernel::computeLowerDJacobian(Moose::ConstraintJacobianType type)
126{
127 mooseAssert(type != Moose::PrimaryPrimary && type != Moose::PrimarySecondary &&
128 type != Moose::SecondaryPrimary && type != Moose::SecondarySecondary,
129 "Jacobian types without lower should be handled in computeElemNeighJacobian");
130
131 const ArrayVariableTestValue & test_space =
132 (type == Moose::LowerLower || type == Moose::LowerSecondary || type == Moose::LowerPrimary)
133 ? _test_lambda
134 : (type == Moose::PrimaryLower ? _test : _test_neighbor);
135 unsigned int ivar =
136 (type == Moose::LowerLower || type == Moose::LowerSecondary || type == Moose::LowerPrimary)
137 ? _lowerd_var.number()
138 : _var.number();
139
140 const ArrayVariableTestValue & loc_phi =
141 (type == Moose::LowerLower || type == Moose::SecondaryLower || type == Moose::PrimaryLower)
142 ? _phi_lambda
143 : (type == Moose::LowerPrimary ? _phi : _phi_neighbor);
144 unsigned int jvar =
145 (type == Moose::LowerLower || type == Moose::SecondaryLower || type == Moose::PrimaryLower)
146 ? _lowerd_var.number()
147 : _var.number();
148
149 prepareMatrixTagLower(_assembly, ivar, jvar, type);
150
151 if (_local_ke.n() == 0 || _local_ke.m() == 0)
152 return;
153
154 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
155 {
156 initLowerDQpJacobian(type);
157 for (_i = 0; _i < test_space.size(); _i++)
158 for (_j = 0; _j < loc_phi.size(); _j++)
159 {
160 // vector or matrix?
161 RealEigenVector v = _JxW[_qp] * _coord[_qp] * computeLowerDQpJacobian(type);
162 _assembly.saveDiagLocalArrayJacobian(
163 _local_ke, _i, test_space.size(), _j, loc_phi.size(), ivar, v);
164 }
165 }
166
167 accumulateTaggedLocalMatrix();
168}

170void
171ArrayDGLowerDKernel::computeOffDiagJacobian(const unsigned int jvar_num)
172{
173 if (!excludeBoundary())
174 {
175 precalculateOffDiagJacobian(jvar_num);
176
177 /*
178 * Note: we cannot call compute jacobian functions like in DGLowerDKernel
179 * because we could have cross component couplings for the array variables
180 */
181
182 const auto & jvar = getVariable(jvar_num);
183
184 // Compute element-element Jacobian
185 computeOffDiagElemNeighJacobian(Moose::ElementElement, jvar);
186
187 // Compute element-neighbor Jacobian
188 computeOffDiagElemNeighJacobian(Moose::ElementNeighbor, jvar);
189
190 // Compute neighbor-element Jacobian
191 computeOffDiagElemNeighJacobian(Moose::NeighborElement, jvar);
192
193 // Compute neighbor-neighbor Jacobian
194 computeOffDiagElemNeighJacobian(Moose::NeighborNeighbor, jvar);
195
196 // Compute the other five pieces of Jacobian related with lower-d variable
197 computeOffDiagLowerDJacobian(Moose::LowerLower, jvar);
198 computeOffDiagLowerDJacobian(Moose::LowerSecondary, jvar);
199 computeOffDiagLowerDJacobian(Moose::LowerPrimary, jvar);
200 computeOffDiagLowerDJacobian(Moose::SecondaryLower, jvar);
201 computeOffDiagLowerDJacobian(Moose::PrimaryLower, jvar);
202 }

205void
206ArrayDGLowerDKernel::computeOffDiagLowerDJacobian(Moose::ConstraintJacobianType type,
207 const MooseVariableFEBase & jvar)
208{
209 mooseAssert(type != Moose::PrimaryPrimary && type != Moose::PrimarySecondary &&
210 type != Moose::SecondaryPrimary && type != Moose::SecondarySecondary,
211 "Jacobian types without lower should be handled in computeOffDiagElemNeighJacobian");
212
213 const ArrayVariableTestValue & test_space =
214 (type == Moose::LowerLower || type == Moose::LowerSecondary || type == Moose::LowerPrimary)
215 ? _test_lambda
216 : (type == Moose::PrimaryLower ? _test : _test_neighbor);
217 unsigned int ivar =
218 (type == Moose::LowerLower || type == Moose::LowerSecondary || type == Moose::LowerPrimary)
219 ? _lowerd_var.number()
220 : _var.number();
221
222 prepareMatrixTagLower(_assembly, ivar, jvar.number(), type);
223 if (_local_ke.n() == 0 || _local_ke.m() == 0)
224 return;
225
226 if (jvar.fieldType() == Moose::VarFieldType::VAR_FIELD_STANDARD)
227 {
228 const auto & jv0 = static_cast<const MooseVariable &>(jvar);
229 const VariableTestValue & loc_phi =
230 (type == Moose::LowerLower || type == Moose::SecondaryLower || type == Moose::PrimaryLower)
231 ? jv0.phiLower()
232 : (type == Moose::LowerPrimary ? jv0.phiFace() : jv0.phiFaceNeighbor());
233
234 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
235 {
236 initLowerDQpOffDiagJacobian(type, jvar);
237 for (_i = 0; _i < test_space.size(); _i++)
238 for (_j = 0; _j < loc_phi.size(); _j++)
239 {
240 RealEigenMatrix v = _JxW[_qp] * _coord[_qp] * computeLowerDQpOffDiagJacobian(type, jvar);
241 _assembly.saveFullLocalArrayJacobian(
242 _local_ke, _i, test_space.size(), _j, loc_phi.size(), ivar, jvar.number(), v);
243 }
244 }
245 }
246 else if (jvar.fieldType() == Moose::VarFieldType::VAR_FIELD_ARRAY)
247 {
248 const auto & jv1 = static_cast<const ArrayMooseVariable &>(jvar);
249 const ArrayVariableTestValue & loc_phi =
250 (type == Moose::LowerLower || type == Moose::SecondaryLower || type == Moose::PrimaryLower)
251 ? jv1.phiLower()
252 : (type == Moose::LowerPrimary ? jv1.phiFace() : jv1.phiFaceNeighbor());
253
254 for (_qp = 0; _qp < _qrule->n_points(); _qp++)
255 {
256 initLowerDQpOffDiagJacobian(type, jvar);
257 for (_i = 0; _i < test_space.size(); _i++)
258 for (_j = 0; _j < loc_phi.size(); _j++)
259 {
260 RealEigenMatrix v = _JxW[_qp] * _coord[_qp] * computeLowerDQpOffDiagJacobian(type, jvar);
261 _assembly.saveFullLocalArrayJacobian(
262 _local_ke, _i, test_space.size(), _j, loc_phi.size(), ivar, jvar.number(), v);
263 }
264 }
265 }
266 else
267 mooseError("Vector variable cannot be coupled into array DG kernel currently");
268
269 accumulateTaggedLocalMatrix();
270}

272RealEigenMatrix
273ArrayDGLowerDKernel::computeLowerDQpOffDiagJacobian(Moose::ConstraintJacobianType type,
274 const MooseVariableFEBase & jvar)
275{
276 if (jvar.number() == _var.number())
277 {
278 if (type == Moose::LowerSecondary || type == Moose::LowerPrimary)
279 return computeLowerDQpJacobian(type).asDiagonal();
280 }
281 else if (jvar.number() == _lowerd_var.number())
282 {
283 if (type == Moose::SecondaryLower || type == Moose::PrimaryLower || type == Moose::LowerLower)
284 return computeLowerDQpJacobian(type).asDiagonal();
285 }
286
287 return RealEigenMatrix::Zero(_var.count(), jvar.count());
288}
