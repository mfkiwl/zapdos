36void
37PerfGraphLivePrint::printLiveMessage(PerfGraph::SectionIncrement & section_increment)
38{
39 auto & section_info = _perf_graph_registry.sectionInfo(section_increment._id);
40
41 // If we're not printing dots - we shouldn't be printing the message at all
42 if (!section_info._print_dots || !_stack_top_print_dots)
43 {
44 section_increment._state = PerfGraph::IncrementState::PRINTED;
45 _last_printed_increment = &section_increment;
46 return;
47 }
48
49 // If the live_message is empty - just print the name
50 auto message =
51 !section_info._live_message.empty() ? section_info._live_message : section_info._name;
52
53 // If this section is just started - but other stuff has printed before we got to print its
54 // message, we need to print it anyway because it could be lengthy and print unexplained
55 // dots to the console (until the Finished message). "Currently" conveys the message
56 // that we did not just start it, but we are doing that right now
57 if (_last_num_printed <= _console_num_printed &&
58 section_increment._state == PerfGraph::IncrementState::STARTED &&
59 section_increment._beginning_num_printed != _console_num_printed)
60 {
61 message = "Currently " + message;
62 // The 10 is for "Currently "
63 section_increment._num_dots = 10;
64 }
65
66 // This line is different - need to finish the last line
67 if (_last_printed_increment && _last_printed_increment != &section_increment &&
68 (_last_printed_increment->_state == PerfGraph::IncrementState::PRINTED ||
69 _last_printed_increment->_state == PerfGraph::IncrementState::CONTINUED) &&
70 section_info._print_dots)
71 _console << '\n';
72
73 // Do we need to print dots?
74 if (_last_printed_increment && _last_printed_increment == &section_increment &&
75 (section_increment._state != PerfGraph::IncrementState::FINISHED &&
76 (section_increment._state == PerfGraph::IncrementState::PRINTED ||
77 section_increment._state == PerfGraph::IncrementState::CONTINUED)))
78 {
79 if (section_info._print_dots)
80 {
81 _console << ".";
82 section_increment._num_dots++;
83 }
84 }
85 else if (section_increment._state == PerfGraph::IncrementState::PRINTED ||
86 section_increment._state ==
87 PerfGraph::IncrementState::CONTINUED) // Printed before so print "Still"
88 {
89 _console << std::string(2 * section_increment._print_stack_level, ' ') << "Still " << message;
90
91 // If we're not printing dots - just finish the line
92 if (!section_info._print_dots)
93 _console << '\n';
94
95 // The 6 is for "Still "
96 section_increment._num_dots = 6;
97 }
98 else // Just print the message
99 {
100 _console << std::string(2 * section_increment._print_stack_level, ' ') << message;
101
102 // If we're not printing dots - just finish the line
103 if (!section_info._print_dots)
104 _console << '\n';
105
106 // Reset the dots since we're printing, except in the "Currently" case
107 if (section_increment._state != PerfGraph::IncrementState::STARTED ||
108 _last_num_printed > _console_num_printed ||
109 section_increment._beginning_num_printed == _console_num_printed)
110 section_increment._num_dots = 0;
111 }
112
113 section_increment._state = PerfGraph::IncrementState::PRINTED;
114
115 // Get the message to the screen
116 _console << std::flush;
117
118 // Keep track of where we printed in the console
119 _last_num_printed = section_increment._beginning_num_printed = _console.numPrinted();
120
121 _last_printed_increment = &section_increment;
122
123 _printed = true;
124}

126void
127PerfGraphLivePrint::printStats(PerfGraph::SectionIncrement & section_increment_start,
128 PerfGraph::SectionIncrement & section_increment_finish)
129{
130 if (_stack_level < 1)
131 return;
132
133 mooseAssert(_perf_graph_registry.sectionExists(section_increment_start._id),
134 "Not found in map: " << section_increment_start._id);
135
136 auto & section_info_start = _perf_graph_registry.sectionInfo(section_increment_start._id);
137
138 // If the live_message is empty - just print the name
139 auto message = !section_info_start._live_message.empty() ? section_info_start._live_message
140 : section_info_start._name;
141
142 auto time_increment =
143 std::chrono::duration<double>(section_increment_finish._time - section_increment_start._time)
144 .count();
145
146 auto memory_total = section_increment_finish._memory;
147
148 auto num_horizontal_chars = message.size() + (2 * section_increment_start._print_stack_level);
149
150 // Add multiapps prefix size
151 if (!_app.isUltimateMaster())
152 num_horizontal_chars += _app.name().size() + 2;
153
154 // Do we need to print "Finished"?
155 // This happens after something else printed in-between when this increment started and finished
156 if (!section_info_start._print_dots ||
157 (_last_printed_increment && _last_printed_increment != &section_increment_start) ||
158 (section_increment_start._beginning_num_printed !=
159 _console_num_printed)) // This means someone _else_ printed
160 {
161 // If we had printed some dots - we need to finish the line
162 if (_last_printed_increment &&
163 _last_printed_increment->_state == PerfGraph::IncrementState::PRINTED &&
164 _perf_graph_registry.sectionInfo(_last_printed_increment->_id)._print_dots)
165 _console << '\n';
166
167 _console << std::string(2 * section_increment_start._print_stack_level, ' ') << "Finished "
168 << message;
169
170 // 9 is for "Finished "
171 num_horizontal_chars += 9;
172 }
173 else
174 num_horizontal_chars += section_increment_start._num_dots;
175
176 // Actually do the printing
177 _console << std::setw(WRAP_LENGTH - num_horizontal_chars);
178
179 _console << " [";
180 _console << COLOR_YELLOW;
181 _console << std::setw(6) << std::fixed << std::setprecision(2) << time_increment << " s";
182 _console << COLOR_DEFAULT;
183 _console << ']' << " [";
184 _console << COLOR_YELLOW << std::setw(5) << std::fixed;
185 _console << memory_total;
186 _console << " MB";
187 _console << COLOR_DEFAULT;
188 _console << ']';
189
190 // If we're not printing dots - just finish the line
191 _console << std::endl;
192
193 _last_num_printed = _console.numPrinted();
194
195 _last_printed_increment = &section_increment_finish;
196
197 _printed = true;
198}

200void
201PerfGraphLivePrint::printStackUpToLast()
202{
203 if (_stack_level < 1)
204 return;
205
206 // We need to print out everything on the stack before this that hasn't already been printed...
207 for (unsigned int s = 0; s < _stack_level - 1; s++)
208 {
209 auto & section = _print_thread_stack[s];
210
211 // Hasn't been printed at all and nothing else has been printed since this started
212 if (section._state == PerfGraph::IncrementState::STARTED)
213 printLiveMessage(section);
214
215 // Note: this will reset the state of a "continued" section to "printed" - so that it can be
216 // continued again
217 section._state = PerfGraph::IncrementState::PRINTED;
218 }
219}

221void
222PerfGraphLivePrint::inSamePlace()
223{
224 // If someone else printed since, then we need to start over, and set everything on the stack to
225 // printed Everything is set to printed because if something printed and we're still in the same
226 // place then we need to NOT print out the beginning message
227 if (_last_num_printed != _console_num_printed)
228 {
229 _last_printed_increment = nullptr;
230
231 for (unsigned int s = 0; s < _stack_level; s++)
232 _print_thread_stack[s]._state = PerfGraph::IncrementState::PRINTED;
233
234 return;
235 }
236
237 // Only print if there is something to print!
238 if (_stack_level > 0)
239 {
240 _stack_top_print_dots =
241 _perf_graph_registry.sectionInfo(_print_thread_stack[_stack_level - 1]._id)._print_dots;
242
243 printStackUpToLast();
244
245 printLiveMessage(_print_thread_stack[_stack_level - 1]);
246
247 // Reset this each time
248 _stack_top_print_dots = true;
249 }

252void
253PerfGraphLivePrint::iterateThroughExecutionList()
254{
255 // Current position in the execution list
256 auto p = _last_execution_list_end;
257
258 while (p != _current_execution_list_end)
259 {
260 // The ternary here is for wrapping around
261 auto next_p = p + 1 < MAX_EXECUTION_LIST_SIZE ? p + 1 : 0;
262
263 auto & section_increment = _execution_list[p];
264
265 // New section, add to the stack
266 if (section_increment._state == PerfGraph::IncrementState::STARTED)
267 {
268 section_increment._print_stack_level = _stack_level;
269
270 // Store this increment in the stack
271 _print_thread_stack[_stack_level] = section_increment;
272
273 _stack_level++;
274 }
275 else // This means it's finished need to see if we need to print it
276 {
277 mooseAssert(_stack_level, "Popping beyond the beginning of the stack!");
278
279 // Get the beginning information for this section... it is the thing currently on the top of
280 // the stack
281 auto & section_increment_start = _print_thread_stack[_stack_level - 1];
282
283 auto time_increment =
284 std::chrono::duration<double>(section_increment._time - section_increment_start._time)
285 .count();
286 auto memory_increment = section_increment._memory - section_increment_start._memory;
287
288 // If it has already been printed or meets our criteria then print it and finish it
289 if (_perf_graph._live_print_all ||
290 section_increment_start._state == PerfGraph::IncrementState::PRINTED ||
291 section_increment_start._state == PerfGraph::IncrementState::CONTINUED ||
292 time_increment > _time_limit.load(std::memory_order_relaxed) ||
293 memory_increment > _mem_limit.load(std::memory_order_relaxed))
294 {
295 printStackUpToLast();
296
297 printStats(section_increment_start, section_increment);
298 }
299
300 _stack_level--;
301 }
302
303 p = next_p;
304 }
305}

307void
308PerfGraphLivePrint::start()
309{
310 // Keep going until we're signaled to end
311 // Note that _currently_destructing can only be set to true in this thread
312 // Which means that by the time we make it back to the top of the loop, either
313 // there was nothing to process or everything has been processed.
314 while (!_currently_destructing)
315 {
316 std::unique_lock<std::mutex> lock(_perf_graph._destructing_mutex);
317
318 // Wait for one second, or until notified that a section is finished
319 // For a section to have finished the execution list has to have been appended to
320 // This keeps spurious wakeups from happening
321 // Note that the `lock` is only protecting _destructing since the execution list uses atomics.
322 // It must be atomic in order to keep the main thread from having to lock as it
323 // executes. The only downside to this is that it is possible for this thread to wake,
324 // check the condition, miss the notification, then wait. In our case this is not detrimental,
325 // as the only thing that will happen is we will wait 1 more second. This is also very
326 // unlikely.
327 // One other thing: wait_for() is not guaranteed to wait for 1 second. "Spurious" wakeups
328 // can occur - but the predicate here keeps us from doing anything in that case.
329 // This will either wait until 1 second has passed, the signal is sent, _or_ a spurious
330 // wakeup happens to find that there is work to do.
331 _perf_graph._finished_section.wait_for(
332 lock,
333 std::chrono::duration<Real>(_time_limit.load(std::memory_order_relaxed)),
334 [this]
335 {
336 // Get destructing first so that the execution_list will be in sync
337 this->_currently_destructing = _perf_graph._destructing;
338
339 // The end will be one past the last
340 // This "acquire" synchronizes with the "release" in the PerfGraph
341 // to ensure that all of the writes to the execution list have been
342 // published to this thread for the "end" we're reading
343 this->_current_execution_list_end =
344 _perf_graph._execution_list_end.load(std::memory_order_acquire);
345
346 // Save off the number of things currently printed to the console
347 this->_console_num_printed = _console.numPrinted();
348
349 // If we are destructing or there is new work to do... allow moving on
350 return this->_currently_destructing ||
351 this->_last_execution_list_end != this->_current_execution_list_end;
352 });
353
354 // If the PerfGraph is destructing and we don't have anything left to print - we need to quit
355 // Otherwise, if there are still things to print - do it... afterwards, the loop above
356 // will end because _done_future has been set in PerfGraph.
357 if (this->_currently_destructing &&
358 this->_last_execution_list_end == this->_current_execution_list_end)
359 return;
360
361 // We store this off for one execution of this loop so that it's consistent all for the whole
362 // iteration
363 _should_print = _perf_graph._live_print_active;
364
365 if (!_should_print)
366 continue;
367
368 // The last entry in the current execution list for convenience
369 _current_execution_list_last = static_cast<long int>(_current_execution_list_end) - 1 >= 0
370 ? _current_execution_list_end - 1
371 : MAX_EXECUTION_LIST_SIZE;
372
373 // Only happens if nothing has been added
374 if (_current_execution_list_end == 0 && _last_execution_list_end == _current_execution_list_end)
375 continue;
376
377 _printed = false;
378
379 // Iterate from the last thing printed (begin) to the last thing in the list (end)
380 // If the time or memory of any section is above the threshold, print everything in between and
381 // update begin
382
383 // Are we still sitting in the same place as the last iteration? If so, we need to print
384 // progress and exit
385 if (_last_execution_list_end == _current_execution_list_end)
386 inSamePlace();
387
388 // This means that new stuff has been added to the execution list. We need to iterate through
389 // it, modifying the stack and printing anything that needs printing
390 iterateThroughExecutionList();
391
392 _last_num_printed = _console.numPrinted();
393
394 _last_execution_list_end = _current_execution_list_end;
395 }
