144template <class Key, class Item, class KeyHash>
145template <typename CreateItem>
146std::size_t
147GeneralRegistry<Key, Item, KeyHash>::registerItem(const Key & key, CreateItem & create_item)
148{
149 std::lock_guard<std::mutex> lock_key(_key_to_id_mutex);
150
151 // Is it already registered?
152 const auto it = _key_to_id.find(key);
153 if (it != _key_to_id.end())
154 return it->second;
155
156 // It's not registered
157 std::lock_guard<std::mutex> lock_id(_id_to_item_mutex);
158 const auto id = _id_to_item.size();
159 _key_to_id.emplace(key, id);
160 _id_to_item.emplace_back(std::move(create_item(id)));
161 return id;
162}
