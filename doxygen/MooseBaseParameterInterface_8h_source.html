170 template <typename... Args>
171 std::string paramErrorMsg(const std::string & param, Args... args) const
172 {
173 auto param_prefix = paramErrorPrefix(_pars, param);
174
175 // With no input location information, append object info (name + type)
176 const std::string object_prefix =
177 _pars.inputLocation(param).empty() ? objectErrorPrefix("parameter error") : "";
178
179 std::ostringstream oss;
180 moose::internal::mooseStreamAll(oss, std::forward<Args>(args)...);
181 std::string msg = oss.str();
182
183 // Wrap error message to a separate line from param_prefix if it is about to
184 // blow past 100 chars. But only wrap if the param_prefix is long enough (12
185 // chars) for the wrap to buy us much extra length. We don't check object_prefix
186 // here because it will go before the parameter error
187 if ((param_prefix.size() > 12 && msg.size() + param_prefix.size() > 99) ||
188 msg.find("\n") != std::string::npos)
189 {
190 if (param_prefix.size() > 0 && param_prefix[param_prefix.size() - 1] != ':')
191 param_prefix += ":";
192 return object_prefix + param_prefix + "\n " + MooseUtils::replaceAll(msg, "\n", "\n ");
193 }
194 return object_prefix + param_prefix + " " + msg;
195 }

205template <typename T>
206const T &
207MooseBaseParameterInterface::getRenamedParam(const std::string & old_name,
208 const std::string & new_name) const
209{
210 // this enables having a default on the new parameter but bypassing it with the old one
211 // Most important: accept new parameter
212 if (isParamSetByUser(new_name) && !isParamValid(old_name))
213 return InputParameters::getParamHelper(new_name, _pars, static_cast<T *>(0));
214 // Second most: accept old parameter
215 else if (isParamValid(old_name) && !isParamSetByUser(new_name))
216 return InputParameters::getParamHelper(old_name, _pars, static_cast<T *>(0));
217 // Third most: accept default for new parameter
218 else if (isParamValid(new_name) && !isParamValid(old_name))
219 return InputParameters::getParamHelper(new_name, _pars, static_cast<T *>(0));
220 // Refuse: no default, no value passed
221 else if (!isParamValid(old_name) && !isParamValid(new_name))
222 mooseError(_pars.blockFullpath() + ": parameter '" + new_name +
223 "' is being retrieved without being set.\n"
224 "Did you misspell it?");
225 // Refuse: both old and new parameters set by user
226 else
227 mooseError(_pars.blockFullpath() + ": parameter '" + new_name +
228 "' may not be provided alongside former parameter '" + old_name + "'");
229}
