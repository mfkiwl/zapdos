54ADReal
55ReactionThirdOrderLogForShootMethod::computeQpResidual()
56{
57 Real power;
58
59 power = 1.;
60 if (_v_id == _density_id)
61 power += 1.;
62
63 if (_w_id == _density_id)
64 power += 1.;
65
66 return -_test[_i][_qp] * _stoichiometric_coeff * _reaction_coeff[_qp] * power *
67 std::exp(_v[_qp]) * std::exp(_w[_qp]) * _u[_qp];
68}
