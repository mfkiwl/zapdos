34EconomouDielectricBC::EconomouDielectricBC(const InputParameters & parameters)
35 : ADIntegratedBC(parameters),
36 _r_units(1. / getParam<Real>("position_units")),
37
38 _mean_en(adCoupledValue("mean_en")),
39 _em(adCoupledValue("em")),
40 _grad_u_dot(_var.gradSlnDot()), // TODO: make an AD version of this in MOOSE
41 _u_dot(_var.adUDot()),
42
43 _e(getMaterialProperty<Real>("e")),
44 _massem(getMaterialProperty<Real>("massem")),
45 _user_se_coeff(getParam<Real>("users_gamma")),
46
47 _epsilon_d(getParam<Real>("dielectric_constant")),
48 _thickness(getParam<Real>("thickness")),
49 _a(0.5),
50 _ion_flux(0, 0, 0),
51 _v_thermal(0),
52 _em_flux(0, 0, 0),
53 _potential_units(getParam<std::string>("potential_units"))
54
55{
56 if (_potential_units.compare("V") == 0)
57 _voltage_scaling = 1.;
58 else if (_potential_units.compare("kV") == 0)
59 _voltage_scaling = 1000;
60
61 _num_ions = coupledComponents("ip");
62
63 _ip.resize(_num_ions);
64 _ip_var.resize(_num_ions);
65 _muip.resize(_num_ions);
66 _sgnip.resize(_num_ions);
67 _potential_ion.resize(_num_ions);
68 _grad_potential_ion.resize(_num_ions);
69
70 bool fill_potential_vector = false;
71 // There are two checks in the next loop:
72 //
73 // 1. There are effective potentials, but potential_ion and ip vectors are not the same length.
74 // If so, throw error.
75 //
76 // 2. There are no effective potentials, and potential_ion = potential. If so, ensure
77 // potential_ion is still same size as _ip.
78 if (coupledComponents("potential_ion") != _num_ions)
79 {
80 if ((coupledComponents("potential_ion") == 1) &&
81 (getVar("potential_ion", 0)->number() == _var.number()))
82 {
83 fill_potential_vector = true;
84 }
85 else
86 {
87 mooseError(
88 "EconomouDielectricBC: ip and potential_ion vectors are not same length. There are two "
89 "possible options: \n 1) Ions respond to an effective potential. If so, make sure each "
90 "ion has an associated effective potential. \n 2) Ions and electrons respond to the same "
91 "potential. If so, set potential_ion equal to this potential.\n");
92 }
93 }
94
95 for (unsigned int i = 0; i < _num_ions; ++i)
96 {
97 _ip_var[i] = getVar("ip", i);
98 _ip[i] = &adCoupledValue("ip", i);
99 _muip[i] = &getADMaterialProperty<Real>("mu" + (*getVar("ip", i)).name());
100 _sgnip[i] = &getMaterialProperty<Real>("sgn" + (*getVar("ip", i)).name());
101
102 if (fill_potential_vector)
103 {
104 _potential_ion[i] = &adCoupledValue("potential_ion", 0);
105 _grad_potential_ion[i] = &adCoupledGradient("potential_ion", 0);
106 }
107 else
108 {
109 _potential_ion[i] = &adCoupledValue("potential_ion", i);
110 _grad_potential_ion[i] = &adCoupledGradient("potential_ion", i);
111 }
112 }
113}

115ADReal
116EconomouDielectricBC::computeQpResidual()
117{
118 _ion_flux.zero();
119 for (unsigned int i = 0; i < _num_ions; ++i)
120 {
121 if (_normals[_qp] * (*_sgnip[i])[_qp] * -(*_grad_potential_ion[i])[_qp] > 0.0)
122 {
123 _a = 1.0;
124 }
125 else
126 {
127 _a = 0.0;
128 }
129
130 _ion_flux += _a * (*_sgnip[i])[_qp] * (*_muip[i])[_qp] * -(*_grad_potential_ion[i])[_qp] *
131 _r_units * std::exp((*_ip[i])[_qp]);
132 }
133
134 _v_thermal =
135 std::sqrt(8 * _e[_qp] * 2.0 / 3 * std::exp(_mean_en[_qp] - _em[_qp]) / (M_PI * _massem[_qp]));
136
137 _em_flux =
138 (0.25 * _v_thermal * std::exp(_em[_qp]) * _normals[_qp]) - (_user_se_coeff * _ion_flux);
139
140 return _test[_i][_qp] * _r_units *
141 ((_thickness / _epsilon_d) * _e[_qp] * 6.022e23 * (_ion_flux - _em_flux) * _normals[_qp] /
142 _voltage_scaling +
143 (_thickness / _epsilon_d) * 8.8542e-12 * -_grad_u_dot[_qp] * _r_units * _normals[_qp] -
144 _u_dot[_qp]);
145}
