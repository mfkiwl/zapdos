360void
361MultiApp::createApps()
362{
363 if (!_has_an_app)
364 return;
365
366 TIME_SECTION("createApps", 2, "Instantiating Sub-Apps", false);
367
368 // Read commandLine arguments that will be used when creating apps
369 readCommandLineArguments();
370
371 Moose::ScopedCommSwapper swapper(_my_comm);
372
373 _apps.resize(_my_num_apps);
374
375 // If the user provided an unregistered app type, see if we can load it dynamically
376 if (!AppFactory::instance().isRegistered(_app_type))
377 _app.dynamicAppRegistration(_app_type,
378 getParam<std::string>("library_path"),
379 getParam<std::string>("library_name"),
380 getParam<bool>("library_load_dependencies"));
381
382 bool rank_did_quiet_init = false;
383 unsigned int local_app = libMesh::invalid_uint;
384 if (_wait_for_first_app_init)
385 {
386 if (hasLocalApp(0))
387 {
388 rank_did_quiet_init = true;
389 local_app = globalAppToLocal(0);
390 createLocalApp(local_app);
391 }
392
393 MPI_Barrier(_orig_comm);
394 }
395
396 for (unsigned int i = 0; i < _my_num_apps; i++)
397 {
398 if (rank_did_quiet_init && i == local_app)
399 continue;
400 createLocalApp(i);
401 }

420void
421MultiApp::readCommandLineArguments()
422{
423 if (isParamValid("cli_args_files"))
424 {
425 _cli_args_from_file.clear();
426
427 std::vector<FileName> cli_args_files = getParam<std::vector<FileName>>("cli_args_files");
428 std::vector<FileName> input_files = getParam<std::vector<FileName>>("input_files");
429
430 // If we use parameter "cli_args_files", at least one file should be provided
431 if (!cli_args_files.size())
432 paramError("cli_args_files", "You need to provide at least one commandLine argument file ");
433
434 // If we multiple input files, then we need to check if the number of input files
435 // match with the number of argument files
436 if (cli_args_files.size() != 1 && cli_args_files.size() != input_files.size())
437 paramError("cli_args_files",
438 "The number of commandLine argument files ",
439 cli_args_files.size(),
440 " for MultiApp ",
441 name(),
442 " must either be only one or match the number of input files ",
443 input_files.size());
444
445 // Go through all argument files
446 std::vector<std::string> cli_args;
447 for (unsigned int p_file_it = 0; p_file_it < cli_args_files.size(); p_file_it++)
448 {
449 std::string cli_args_file = cli_args_files[p_file_it];
450 // Clear up
451 cli_args.clear();
452 // Read the file on the root processor then broadcast it
453 if (processor_id() == 0)
454 {
455 MooseUtils::checkFileReadable(cli_args_file);
456
457 std::ifstream is(cli_args_file.c_str());
458 std::copy(std::istream_iterator<std::string>(is),
459 std::istream_iterator<std::string>(),
460 std::back_inserter(cli_args));
461
462 // We do not allow empty files
463 if (!cli_args.size())
464 paramError("cli_args_files",
465 "There is no commandLine argument in the commandLine argument file ",
466 cli_args_file);
467
468 // If we have position files, we need to
469 // make sure the number of commandLine argument strings
470 // match with the number of positions
471 if (_npositions_inputfile.size())
472 {
473 auto num_positions = _npositions_inputfile[p_file_it];
474 // Check if the number of commandLine argument strings equal to
475 // the number of positions
476 if (cli_args.size() == 1)
477 for (MooseIndex(num_positions) num = 0; num < num_positions; num++)
478 _cli_args_from_file.push_back(cli_args.front());
479 else if (cli_args.size() == num_positions)
480 for (auto && cli_arg : cli_args)
481 _cli_args_from_file.push_back(cli_arg);
482 else if (cli_args.size() != num_positions)
483 paramError("cli_args_files",
484 "The number of commandLine argument strings ",
485 cli_args.size(),
486 " in the file ",
487 cli_args_file,
488 " must either be only one or match the number of positions ",
489 num_positions);
490 }
491 else
492 {
493 // If we do not have position files, we will check if the number of
494 // commandLine argument strings match with the total number of subapps
495 for (auto && cli_arg : cli_args)
496 _cli_args_from_file.push_back(cli_arg);
497 }
498 }
499 }
500
501 // Broad cast all arguments to everyone
502 _communicator.broadcast(_cli_args_from_file);
503 }
504
505 if (_cli_args_from_file.size() && _cli_args_from_file.size() != 1 &&
506 _cli_args_from_file.size() != _total_num_apps)
507 mooseError(" The number of commandLine argument strings ",
508 _cli_args_from_file.size(),
509 " must either be only one or match the total "
510 "number of sub apps ",
511 _total_num_apps);
512
513 if (_cli_args_from_file.size() && cliArgs().size())
514 mooseError("Cannot set commandLine arguments from both input_file and external files");
515}

517void
518MultiApp::fillPositions()
519{
520 if (_move_apps.size() != _move_positions.size())
521 mooseError("The number of apps to move and the positions to move them to must be the same for "
522 "MultiApp ",
523 _name);
524
525 if (isParamValid("positions") + isParamValid("positions_file") +
526 isParamValid("positions_objects") >
527 1)
528 mooseError("Only one 'positions' parameter may be specified");
529
530 if (isParamValid("positions"))
531 {
532 _positions = getParam<std::vector<Point>>("positions");
533
534 if (_positions.size() < _input_files.size())
535 mooseError("Not enough positions for the number of input files provided in MultiApp ",
536 name());
537 }
538 else if (isParamValid("positions_file"))
539 {
540 std::vector<FileName> positions_files = getParam<std::vector<FileName>>("positions_file");
541 std::vector<FileName> input_files = getParam<std::vector<FileName>>("input_files");
542
543 if (input_files.size() != 1 && positions_files.size() != input_files.size())
544 mooseError("Number of input_files for MultiApp ",
545 name(),
546 " must either be only one or match the number of positions_file files");
547
548 // Clear out the _input_files because we're going to rebuild it
549 if (input_files.size() != 1)
550 _input_files.clear();
551
552 for (unsigned int p_file_it = 0; p_file_it < positions_files.size(); p_file_it++)
553 {
554 std::string positions_file = positions_files[p_file_it];
555 MooseUtils::DelimitedFileReader file(positions_file, &_communicator);
556 file.setFormatFlag(MooseUtils::DelimitedFileReader::FormatFlag::ROWS);
557 file.read();
558
559 const std::vector<Point> & data = file.getDataAsPoints();
560 for (const auto & d : data)
561 _positions.push_back(d);
562
563 // Save the number of positions for this input file
564 _npositions_inputfile.push_back(data.size());
565
566 for (unsigned int i = 0; i < data.size(); ++i)
567 if (input_files.size() != 1)
568 _input_files.push_back(input_files[p_file_it]);
569 }
570 }
571 else if (isParamValid("positions_objects"))
572 {
573 const auto & positions_param_objs = getParam<std::vector<PositionsName>>("positions_objects");
574 const auto & input_files = getParam<std::vector<FileName>>("input_files");
575
576 if (input_files.size() != 1 && positions_param_objs.size() != input_files.size())
577 mooseError("Number of input_files for MultiApp ",
578 name(),
579 " must either be only one or match the number of positions_objects specified");
580
581 // Clear out the _input_files because we're going to rebuild it
582 if (input_files.size() != 1)
583 _input_files.clear();
584
585 // Keeps track of where each positions object start in terms of subapp numbers
586 unsigned int offset = 0;
587
588 for (const auto p_obj_it : index_range(positions_param_objs))
589 {
590 const std::string & positions_name = positions_param_objs[p_obj_it];
591 auto positions_obj = &_fe_problem.getPositionsObject(positions_name);
592
593 const auto & data = positions_obj->getPositions(true);
594
595 // Append all positions from this object
596 for (const auto & d : data)
597 _positions.push_back(d);
598
599 // Save the number of positions for this input file
600 _npositions_inputfile.push_back(data.size());
601
602 if (!positions_obj)
603 paramError("positions_objects",
604 "'" + positions_name + "' is not of the expected type. Should be a Positions");
605
606 // Keep track of which positions is tied to what subapp
607 for (unsigned int i = 0; i < data.size(); ++i)
608 {
609 if (input_files.size() != 1)
610 _input_files.push_back(input_files[p_obj_it]);
611 _positions_objs.push_back(positions_obj);
612 _positions_index_offsets.push_back(offset);
613 }
614 offset += data.size();
615 }
616 }
617 else
618 {
619 _positions = {Point()};
620
621 if (_positions.size() < _input_files.size())
622 mooseError("Not enough positions for the number of input files provided in MultiApp ",
623 name());
624 }
625
626 mooseAssert(_input_files.size() == 1 || _positions.size() == _input_files.size(),

630void
631MultiApp::preTransfer(Real /*dt*/, Real target_time)
632{
633 // Get a transient executioner to get a user-set tolerance
634 Real timestep_tol = 1e-13;
635 if (dynamic_cast<Transient *>(_fe_problem.getMooseApp().getExecutioner()))
636 timestep_tol =
637 dynamic_cast<Transient *>(_fe_problem.getMooseApp().getExecutioner())->timestepTol();
638
639 // First, see if any Apps need to be reset
640 for (unsigned int i = 0; i < _reset_times.size(); i++)
641 {
642 if (!_reset_happened[i] && (target_time + timestep_tol >= _reset_times[i]))
643 {
644 _reset_happened[i] = true;
645 if (_reset_apps.size() > 0)
646 for (auto & app : _reset_apps)
647 resetApp(app);
648
649 // If we reset an application, then we delete the old objects, including the coordinate
650 // transformation classes. Consequently we need to reset the coordinate transformation classes
651 // in the associated transfer classes
652 for (auto * const transfer : _associated_transfers)
653 transfer->getAppInfo();
654
655 // Similarly we need to transform the mesh again
656 if (_run_in_position)
657 for (const auto i : make_range(_my_num_apps))
658 {
659 auto app_ptr = _apps[i];
660 if (usingPositions())
661 app_ptr->getExecutioner()->feProblem().coordTransform().transformMesh(
662 app_ptr->getExecutioner()->feProblem().mesh(), _positions[_first_local_app + i]);
663 else
664 app_ptr->getExecutioner()->feProblem().coordTransform().transformMesh(
665 app_ptr->getExecutioner()->feProblem().mesh(), Point(0, 0, 0));
666 }
667
668 // If the time step covers multiple reset times, set them all as having 'happened'
669 for (unsigned int j = i; j < _reset_times.size(); j++)
670 if (target_time + timestep_tol >= _reset_times[j])
671 _reset_happened[j] = true;
672
673 break;
674 }
675 }
676
677 // Now move any apps that should be moved
678 if (_use_positions && !_move_happened && target_time + timestep_tol >= _move_time)
679 {
680 _move_happened = true;
681 for (unsigned int i = 0; i < _move_apps.size(); i++)
682 moveApp(_move_apps[i], _move_positions[i]);
683 }
684}

735void
736MultiApp::restore(bool force)
737{
738 TIME_SECTION(_restore_timer);
739
740 if (force || needsRestoration())
741 {
742 // Must be restarting / recovering from main app so hold off on restoring
743 // Instead - the restore will happen in sub-apps' initialSetup()
744 // Note that _backups was already populated by dataLoad() in the main app
745 if (_fe_problem.getCurrentExecuteOnFlag() == EXEC_INITIAL)
746 return;
747
748 // We temporarily copy and store solutions for all subapps
749 if (_keep_solution_during_restore)
750 {
751 _end_solutions.resize(_my_num_apps);
752
753 for (unsigned int i = 0; i < _my_num_apps; i++)
754 {
755 _end_solutions[i] = _apps[i]
756 ->getExecutioner()
757 ->feProblem()
758 .getNonlinearSystemBase(/*nl_sys=*/0)
759 .solution()
760 .clone();
761 auto & sub_multiapps =
762 _apps[i]->getExecutioner()->feProblem().getMultiAppWarehouse().getObjects();
763
764 // multiapps of each subapp should do the same things
765 // It is implemented recursively
766 for (auto & multi_app : sub_multiapps)
767 multi_app->keepSolutionDuringRestore(_keep_solution_during_restore);
768 }
769 }
770
771 if (_fe_problem.verboseMultiApps())
772 _console << "Restoring MultiApp ... ";
773
774 for (unsigned int i = 0; i < _my_num_apps; i++)
775 {
776 _apps[i]->restore(std::move(_sub_app_backups[i]), false);
777 _sub_app_backups[i] = _apps[i]->finalizeRestore();
778 mooseAssert(_sub_app_backups[i], "Should have a backup");
779 }
780
781 if (_fe_problem.verboseMultiApps())
782 _console << name() << std::endl;
783
784 // Now copy the latest solutions back for each subapp
785 if (_keep_solution_during_restore)
786 {
787 for (unsigned int i = 0; i < _my_num_apps; i++)
788 {
789 _apps[i]->getExecutioner()->feProblem().getNonlinearSystemBase(/*nl_sys=*/0).solution() =
790 *_end_solutions[i];
791
792 // We need to synchronize solution so that local_solution has the right values
793 _apps[i]->getExecutioner()->feProblem().getNonlinearSystemBase(/*nl_sys=*/0).update();
794 }
795
796 _end_solutions.clear();
797 }
798 }
799 else
800 {
801 for (unsigned int i = 0; i < _my_num_apps; i++)
802 {
803 for (auto & sub_app :
804 _apps[i]->getExecutioner()->feProblem().getMultiAppWarehouse().getObjects())
805 sub_app->restore(false);
806 }
807 }

820void
821MultiApp::transformBoundingBox(BoundingBox & box, const MultiAppCoordTransform & transform)
822{
823 const Real min_x = box.first(0);
824 const Real max_x = box.second(0);
825 const Real min_y = box.first(1);
826 const Real max_y = box.second(1);
827 const Real min_z = box.first(2);
828 const Real max_z = box.second(2);
829
830 std::array<Point, 8> box_corners = {{Point(min_x, min_y, min_z),
831 Point(max_x, min_y, min_z),
832 Point(min_x, max_y, min_z),
833 Point(max_x, max_y, min_z),
834 Point(min_x, min_y, max_z),
835 Point(max_x, min_y, max_z),
836 Point(min_x, max_y, max_z),
837 Point(max_x, max_y, max_z)}};
838
839 // transform each corner
840 for (auto & corner : box_corners)
841 corner = transform(corner);
842
843 // Create new bounding box
844 Point new_box_min = box_corners[0];
845 Point new_box_max = new_box_min;
846 for (const auto p : make_range(1, 8))
847 for (const auto d : make_range(Moose::dim))
848 {
849 const Point & pt = box_corners[p];
850 if (new_box_min(d) > pt(d))
851 new_box_min(d) = pt(d);
852
853 if (new_box_max(d) < pt(d))
854 new_box_max(d) = pt(d);
855 }
856 box.first = new_box_min;
857 box.second = new_box_max;
858}

860BoundingBox
861MultiApp::getBoundingBox(unsigned int app,
862 bool displaced_mesh,
863 const MultiAppCoordTransform * const coord_transform)
864{
865 if (!_has_an_app)
866 mooseError("No app for ", name(), " on processor ", _orig_rank);
867
868 unsigned int local_app = globalAppToLocal(app);
869 FEProblemBase & fe_problem_base = _apps[local_app]->getExecutioner()->feProblem();
870 MooseMesh & mesh = (displaced_mesh && fe_problem_base.getDisplacedProblem().get() != NULL)
871 ? fe_problem_base.getDisplacedProblem()->mesh()
872 : fe_problem_base.mesh();
873
874 {
875 Moose::ScopedCommSwapper swapper(_my_comm);
876 if (displaced_mesh)
877 _bounding_box[local_app] = MeshTools::create_bounding_box(mesh);
878 else
879 {
880 if (!_has_bounding_box[local_app])
881 {
882 _bounding_box[local_app] = MeshTools::create_bounding_box(mesh);
883 _has_bounding_box[local_app] = true;
884 }
885 }
886 }
887 BoundingBox bbox = _bounding_box[local_app];
888
889 Point min = bbox.min();
890 min -= _bounding_box_padding;
891 Point max = bbox.max();
892 max += _bounding_box_padding;
893
894 Point inflation_amount = (max - min) * _inflation;
895
896 Point inflated_min = min - inflation_amount;
897 Point inflated_max = max + inflation_amount;
898
899 Point shifted_min = inflated_min;
900 Point shifted_max = inflated_max;
901
902 if ((!coord_transform || coord_transform->skipCoordinateCollapsing()) &&
903 fe_problem_base.getCoordSystem(*(mesh.meshSubdomains().begin())) == Moose::COORD_RZ)
904 {
905 // If the problem is RZ then we're going to invent a box that would cover the whole "3D" app
906 // FIXME: Assuming all subdomains are the same coordinate system type!
907 shifted_min(0) = -inflated_max(0);
908 shifted_min(1) = inflated_min(1);
909 shifted_min(2) = -inflated_max(0);
910
911 shifted_max(0) = inflated_max(0);
912 shifted_max(1) = inflated_max(1);
913 shifted_max(2) = inflated_max(0);
914 }
915
916 if (coord_transform)
917 {
918 BoundingBox transformed_bbox(shifted_min, shifted_max);
919 transformBoundingBox(transformed_bbox, *coord_transform);
920 return transformed_bbox;
921 }
922 else
923 {
924 // This is where the app is located. We need to shift by this amount.
925 Point p = position(app);
926
927 // Shift them to the position they're supposed to be
928 shifted_min += p;
929 shifted_max += p;
930 return BoundingBox(shifted_min, shifted_max);
931 }

1053void
1054MultiApp::createApp(unsigned int i, Real start_time)
1055{
1056 // Define the app name
1057 const std::string multiapp_name = getMultiAppName(name(), _first_local_app + i, _total_num_apps);
1058 std::string full_name;
1059
1060 // Only add parent name if the parent is not the main app
1061 if (_app.multiAppLevel() > 0)
1062 full_name = _app.name() + "_" + multiapp_name;
1063 else
1064 full_name = multiapp_name;
1065
1066 InputParameters app_params = AppFactory::instance().getValidParams(_app_type);
1067 app_params.set<FEProblemBase *>("_parent_fep") = &_fe_problem;
1068 app_params.set<std::unique_ptr<Backup> *>("_initial_backup") = &_sub_app_backups[i];
1069
1070 // Set the command line parameters with a copy of the main application command line parameters,
1071 // the copy is required so that the addArgument command below doesn't accumulate more and more
1072 // of the same cli_args, which is important when running in batch mode.
1073 std::shared_ptr<CommandLine> app_cli = std::make_shared<CommandLine>(*_app.commandLine());
1074
1075 if (cliArgs().size() > 0 || _cli_args_from_file.size() > 0)
1076 {
1077 for (const std::string & str : MooseUtils::split(getCommandLineArgsParamHelper(i), ";"))
1078 {
1079 std::ostringstream oss;
1080 oss << full_name << ":" << str;
1081 app_cli->addArgument(oss.str());
1082 }
1083 }
1084 app_cli->initForMultiApp(full_name);
1085 app_params.set<std::shared_ptr<CommandLine>>("_command_line") = app_cli;
1086
1087 if (_fe_problem.verboseMultiApps())
1088 _console << COLOR_CYAN << "Creating MultiApp " << name() << " of type " << _app_type
1089 << " of level " << _app.multiAppLevel() + 1 << " and number " << _first_local_app + i
1090 << " on processor " << processor_id() << " with full name " << full_name
1091 << COLOR_DEFAULT << std::endl;
1092 app_params.set<unsigned int>("_multiapp_level") = _app.multiAppLevel() + 1;
1093 app_params.set<unsigned int>("_multiapp_number") = _first_local_app + i;
1094 if (getParam<bool>("clone_master_mesh") || getParam<bool>("clone_parent_mesh"))
1095 {
1096 if (_fe_problem.verboseMultiApps())
1097 _console << COLOR_CYAN << "Cloned parent app mesh will be used for MultiApp " << name()
1098 << COLOR_DEFAULT << std::endl;
1099 app_params.set<const MooseMesh *>("_master_mesh") = &_fe_problem.mesh();
1100 auto displaced_problem = _fe_problem.getDisplacedProblem();
1101 if (displaced_problem)
1102 app_params.set<const MooseMesh *>("_master_displaced_mesh") = &displaced_problem->mesh();
1103 }
1104 _apps[i] = AppFactory::instance().createShared(_app_type, full_name, app_params, _my_comm);
1105 auto & app = _apps[i];
1106
1107 std::string input_file = "";
1108 if (_input_files.size() == 1) // If only one input file was provided, use it for all the solves
1109 input_file = _input_files[0];
1110 else
1111 input_file = _input_files[_first_local_app + i];
1112
1113 app->setGlobalTimeOffset(start_time);
1114 app->setInputFileName(input_file);
1115 app->setOutputFileNumbers(_app.getOutputWarehouse().getFileNumbers());
1116 app->setRestart(_app.isRestarting());
1117 app->setRecover(_app.isRecovering());
1118
1119 if (_use_positions && getParam<bool>("output_in_position"))
1120 app->setOutputPosition(_app.getOutputPosition() + _positions[_first_local_app + i]);
1121 if (_output_in_position && _run_in_position)
1122 paramError("run_in_position",
1123 "Sub-apps are already displaced, so they are already output in position");
1124
1125 // Update the MultiApp level for the app that was just created
1126 app->setupOptions();
1127 // if multiapp does not have file base in Outputs input block, output file base will
1128 // be empty here since setupOptions() does not set the default file base with the multiapp
1129 // input file name. Parent app will create the default file base for multiapp by taking the
1130 // output base of the parent app problem and appending the name of the multiapp plus a number to
1131 // it
1132 if (app->getOutputFileBase().empty())
1133 setAppOutputFileBase(i);
1134 preRunInputFile();
1135
1136 // Transfer coupling relaxation information to the subapps
1137 _apps[i]->fixedPointConfig().sub_relaxation_factor = getParam<Real>("relaxation_factor");
1138 _apps[i]->fixedPointConfig().sub_transformed_vars =
1139 getParam<std::vector<std::string>>("transformed_variables");
1140 // Handle deprecated parameter
1141 if (!parameters().isParamSetByAddParam("relaxed_variables"))
1142 _apps[i]->fixedPointConfig().sub_transformed_vars =
1143 getParam<std::vector<std::string>>("relaxed_variables");
1144 _apps[i]->fixedPointConfig().sub_transformed_pps =
1145 getParam<std::vector<PostprocessorName>>("transformed_postprocessors");
1146
1147 app->runInputFile();
1148 auto fixed_point_solve = &(_apps[i]->getExecutioner()->fixedPointSolve());
1149 if (fixed_point_solve)
1150 fixed_point_solve->allocateStorage(false);
1151
1152 // Transform the app mesh if requested
1153 if (_run_in_position)
1154 {
1155 if (usingPositions())
1156 app->getExecutioner()->feProblem().coordTransform().transformMesh(
1157 app->getExecutioner()->feProblem().mesh(), _positions[_first_local_app + i]);
1158 else
1159 app->getExecutioner()->feProblem().coordTransform().transformMesh(
1160 app->getExecutioner()->feProblem().mesh(), Point(0, 0, 0));
1161 }

1182
1183LocalRankConfig
1184rankConfig(processor_id_type rank,
1185 processor_id_type nprocs,
1186 dof_id_type napps,
1187 processor_id_type min_app_procs,
1188 processor_id_type max_app_procs,
1189 bool batch_mode)
1190{
1191 if (min_app_procs > nprocs)
1192 mooseError("minimum number of procs per app is higher than the available number of procs");
1193 else if (min_app_procs > max_app_procs)
1194 mooseError("minimum number of procs per app must be lower than the max procs per app");
1195
1196 mooseAssert(rank < nprocs, "rank must be smaller than the number of procs");
1197
1198 // A "slot" is a group of procs/ranks that are grouped together to run a
1199 // single (sub)app/sim in parallel.
1200
1201 const processor_id_type slot_size =
1202 std::max(std::min(cast_int<processor_id_type>(nprocs / napps), max_app_procs), min_app_procs);
1203 const processor_id_type nslots = std::min(
1204 nprocs / slot_size,
1205 cast_int<processor_id_type>(std::min(
1206 static_cast<dof_id_type>(std::numeric_limits<processor_id_type>::max()), napps)));
1207 mooseAssert(nprocs >= (nslots * slot_size),
1208 "Ensure that leftover procs is represented by an unsigned type");
1209 const processor_id_type leftover_procs = nprocs - nslots * slot_size;
1210 const dof_id_type apps_per_slot = napps / nslots;
1211 const dof_id_type leftover_apps = napps % nslots;
1212
1213 std::vector<int> slot_for_rank(nprocs);
1214 processor_id_type slot = 0;
1215 processor_id_type procs_in_slot = 0;
1216 for (processor_id_type rankiter = 0; rankiter <= rank; rankiter++)
1217 {
1218 if (slot < nslots)
1219 slot_for_rank[rankiter] = cast_int<int>(slot);
1220 else
1221 slot_for_rank[rankiter] = -1;
1222 procs_in_slot++;
1223 // this slot keeps growing until we reach slot size plus possibly an extra
1224 // proc if there were any leftover from the slotization of nprocs - this
1225 // must also make sure we don't go over max app procs.
1226 if (procs_in_slot == slot_size + 1 * (slot < leftover_procs && slot_size < max_app_procs))
1227 {
1228 procs_in_slot = 0;
1229 slot++;
1230 }
1231 }
1232
1233 if (slot_for_rank[rank] < 0)
1234 // ranks assigned a negative slot don't have any apps running on them.
1235 return {0, 0, 0, 0, false, 0};
1236 const processor_id_type slot_num = cast_int<processor_id_type>(slot_for_rank[rank]);
1237
1238 const bool is_first_local_rank = rank == 0 || (slot_for_rank[rank - 1] != slot_for_rank[rank]);
1239 const dof_id_type n_local_apps = apps_per_slot + 1 * (slot_num < leftover_apps);
1240
1241 processor_id_type my_first_rank = 0;
1242 for (processor_id_type rankiter = rank; rankiter > 0; rankiter--)
1243 if (slot_for_rank[rank] != slot_for_rank[rankiter])
1244 {
1245 my_first_rank = cast_int<processor_id_type>(slot_for_rank[rankiter + 1]);
1246 break;
1247 }
1248
1249 dof_id_type app_index = 0;
1250 for (processor_id_type slot = 0; slot < slot_num; slot++)
1251 {
1252 const dof_id_type num_slot_apps = apps_per_slot + 1 * (slot < leftover_apps);
1253 app_index += num_slot_apps;
1254 }
1255
1256 if (batch_mode)
1257 return {n_local_apps, app_index, 1, slot_num, is_first_local_rank, my_first_rank};
