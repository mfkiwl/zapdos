214template <class T>
215void
216AddMetaDataGenerator::inputChecker(const std::vector<std::string> data_names,
217 const std::vector<T> data_values,
218 const std::string param_name)
219{
220 std::vector<std::string> data_name_tmp(data_names);
221 if (std::unique(data_name_tmp.begin(), data_name_tmp.end()) != data_name_tmp.end())
222 paramError(param_name + "_metadata_names", "Elements of this parameter must be unique.");
223 if (data_names.size() != data_values.size())
224 paramError(param_name + "_metadata_values",
225 "Length of this parameter must be the same as that of " + param_name +
226 "_metadata_names");
227}
