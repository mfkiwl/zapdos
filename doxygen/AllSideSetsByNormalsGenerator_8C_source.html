46std::unique_ptr<MeshBase>
47AllSideSetsByNormalsGenerator::generate()
48{
49 std::unique_ptr<MeshBase> mesh = std::move(_input);
50 if (!mesh->is_replicated())
51 mooseError("AllSideSetsByNormalsGenerator is not implemented for distributed meshes");
52 setup(*mesh);
53
54 // Get the current list of boundaries so we can generate new ones that won't conflict
55 _mesh_boundary_ids = mesh->get_boundary_info().get_boundary_ids();
56
57 _visited.clear();
58
59 // We'll need to loop over all of the elements to find ones that match this normal.
60 // We can't rely on flood catching them all here...
61 for (const auto & elem : mesh->element_ptr_range())
62 for (unsigned int side = 0; side < elem->n_sides(); ++side)
63 {
64 if (elem->neighbor_ptr(side))
65 continue;
66
67 const std::vector<Point> & normals = _fe_face->get_normals();
68 _fe_face->reinit(elem, side);
69
70 {
71 // See if we've seen this normal before (linear search)
72 const std::map<BoundaryID, RealVectorValue>::value_type * item = nullptr;
73 for (const auto & id_pair : _boundary_to_normal_map)
74 if (std::abs(1.0 - id_pair.second * normals[0]) < 1e-5)
75 {
76 item = &id_pair;
77 break;
78 }
79
80 if (item)
81 flood(elem, normals[0], item->first, *mesh);
82 else
83 {
84 boundary_id_type id = getNextBoundaryID();
85 _boundary_to_normal_map[id] = normals[0];
86 flood(elem, normals[0], id, *mesh);
87 }
88 }
89 }
90
91 finalize();
92
93 mesh->set_isnt_prepared();
94 return dynamic_pointer_cast<MeshBase>(mesh);
95}

97boundary_id_type
98AllSideSetsByNormalsGenerator::getNextBoundaryID()
99{
100 std::set<boundary_id_type>::iterator it;
101 boundary_id_type next_id = 1;
102
103 while ((it = _mesh_boundary_ids.find(next_id)) != _mesh_boundary_ids.end())
104 ++next_id;
105
106 _mesh_boundary_ids.insert(next_id);
107
108 return next_id;
109}
