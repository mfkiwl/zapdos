78 params.addRequiredParam<Real>("num_controller_set",
79 "The number of controller sets"

109void
110AddPeriodicControllers::act()
111{
112 _enable_start_start_time_index.resize(_enable_start.size());
113 _enable_start_end_time_index.resize(_enable_start.size());
114
115 _enable_during_start_time_index.resize(_enable_during.size());
116 _enable_during_end_time_index.resize(_enable_during.size());
117
118 _enable_end_start_time_index.resize(_enable_end.size());
119 _enable_end_end_time_index.resize(_enable_end.size());
120
121 _disable_start_start_time_index.resize(_disable_start.size());
122 _disable_start_end_time_index.resize(_disable_start.size());
123
124 _disable_during_start_time_index.resize(_disable_during.size());
125 _disable_during_end_time_index.resize(_disable_during.size());
126
127 _disable_end_start_time_index.resize(_disable_end.size());
128 _disable_end_end_time_index.resize(_disable_end.size());
129
130 std::string name_num;
131 bool first_controller;
132
133 if (_current_task == "add_control")
134 {
135 for (unsigned int i = 0; i < _num_controller_set; ++i)
136 {
137 // Setting up the start and end times for the TimePeriod Controller.
138 // The end time is start time plus 0.0001 times the period
139 for (MooseIndex(_enable_start) j = 0; j < _enable_start.size(); ++j)
140 {
141 _enable_start_start_time_index[j] = _start_time + _cycles_per_controls * _period * i;
142 _enable_start_end_time_index[j] =
143 _start_time + _cycles_per_controls * _period * i + (_period * 0.0001);
144 }
145
146 for (MooseIndex(_enable_during) j = 0; j < _enable_during.size(); ++j)
147 {
148 _enable_during_start_time_index[j] = _start_time + _cycles_per_controls * _period * i;
149 _enable_during_end_time_index[j] =
150 (_start_time + _period) + _cycles_per_controls * _period * i + (_period * 0.0001);
151 }
152
153 // If control cycles are back to back, then enable_start and enable_end times would be the
154 // same, To avoid this, if cycles are back to back, then the enable_end starts at +
155 // 0.0001*period.
156 if (_cycles_per_controls == 1.0)
157 {
158 for (MooseIndex(_enable_end) j = 0; j < _enable_end.size(); ++j)
159 {
160 _enable_end_start_time_index[j] =
161 (_start_time + _period) + _cycles_per_controls * _period * i + (_period * 0.0001);
162 _enable_end_end_time_index[j] =
163 (_start_time + _period) + _cycles_per_controls * _period * i + (_period * 0.0002);
164 }
165 }
166 else
167 {
168 for (MooseIndex(_enable_end) j = 0; j < _enable_end.size(); ++j)
169 {
170 _enable_end_start_time_index[j] =
171 (_start_time + _period) + _cycles_per_controls * _period * i;
172 _enable_end_end_time_index[j] =
173 (_start_time + _period) + _cycles_per_controls * _period * i + (_period * 0.0001);
174 }
175 }
176
177 for (MooseIndex(_disable_start) j = 0; j < _disable_start.size(); ++j)
178 {
179 _disable_start_start_time_index[j] = _start_time + _cycles_per_controls * _period * i;
180 _disable_start_end_time_index[j] =
181 _start_time + _cycles_per_controls * _period * i + (_period * 0.0001);
182 }
183
184 for (MooseIndex(_disable_during) j = 0; j < _disable_during.size(); ++j)
185 {
186 _disable_during_start_time_index[j] = _start_time + _cycles_per_controls * _period * i;
187 _disable_during_end_time_index[j] =
188 (_start_time + _period) + _cycles_per_controls * _period * i + (_period * 0.0001);
189 }
190
191 // If control cycles are back to back, then disable_start and disable_end times would be the
192 // same, To avoid this, if cycles are back to back, then the disable_end starts at +
193 // 0.0001*period
194 if (_cycles_per_controls == 1.0)
195 {
196 for (MooseIndex(_disable_end) j = 0; j < _disable_end.size(); ++j)
197 {
198 _disable_end_start_time_index[j] =
199 (_start_time + _period) + _cycles_per_controls * _period * i + (_period * 0.0001);
200 _disable_end_end_time_index[j] =
201 (_start_time + _period) + _cycles_per_controls * _period * i + (_period * 0.0002);
202 }
203 }
204 else
205 {
206 for (MooseIndex(_disable_end) j = 0; j < _disable_end.size(); ++j)
207 {
208 _disable_end_start_time_index[j] =
209 (_start_time + _period) + _cycles_per_controls * _period * i;
210 _disable_end_end_time_index[j] =
211 (_start_time + _period) + _cycles_per_controls * _period * i + (_period * 0.0001);
212 }
213 }
214
215 // Adding the multiple TimePeriod controllers
216 if (_enable_start.size() > 0)
217 {
218 name_num = "Enable_Begin_" + std::to_string(i);
219
220 if (i == 0)
221 {
222 first_controller = true;
223 }
224 else
225 {
226 first_controller = false;
227 }
228
229 AddTimePeriod("enable_objects",
230 _enable_start,
231 _enable_start_start_time_index,
232 _enable_start_end_time_index,
233 name_num,
234 first_controller);
235 }
236
237 if (_enable_during.size() > 0)
238 {
239 name_num = "Enable_During_" + std::to_string(i);
240
241 if (i == 0)
242 {
243 first_controller = true;
244 }
245 else
246 {
247 first_controller = false;
248 }
249
250 AddTimePeriod("enable_objects",
251 _enable_during,
252 _enable_during_start_time_index,
253 _enable_during_end_time_index,
254 name_num,
255 first_controller);
256 }
257
258 if (_enable_end.size() > 0)
259 {
260 name_num = "Enable_End_" + std::to_string(i);
261
262 if (i == 0)
263 {
264 first_controller = true;
265 }
266 else
267 {
268 first_controller = false;
269 }
270
271 AddTimePeriod("enable_objects",
272 _enable_end,
273 _enable_end_start_time_index,
274 _enable_end_end_time_index,
275 name_num,
276 first_controller);
277 }
278
279 if (_disable_start.size() > 0)
280 {
281 name_num = "Disable_Begin_" + std::to_string(i);
282
283 if (i == 0)
284 {
285 first_controller = true;
286 }
287 else
288 {
289 first_controller = false;
290 }
291
292 AddTimePeriod("disable_objects",
293 _disable_start,
294 _disable_start_start_time_index,
295 _disable_start_end_time_index,
296 name_num,
297 first_controller);
298 }
299
300 if (_disable_during.size() > 0)
301 {
302 name_num = "Disable_During_" + std::to_string(i);
303
304 if (i == 0)
305 {
306 first_controller = true;
307 }
308 else
309 {
310 first_controller = false;
311 }
312
313 AddTimePeriod("disable_objects",
314 _disable_during,
315 _disable_during_start_time_index,
316 _disable_during_end_time_index,
317 name_num,
318 first_controller);
319 }
320
321 if (_disable_end.size() > 0)
322 {
323 name_num = "Disable_End_" + std::to_string(i);
324
325 if (i == 0)
326 {
327 first_controller = true;
328 }
329 else
330 {
331 first_controller = false;
332 }
333
334 AddTimePeriod("disable_objects",
335 _disable_end,
336 _disable_end_start_time_index,
337 _disable_end_end_time_index,
338 name_num,
339 first_controller);
340 }
341 }
342 }

345void
346AddPeriodicControllers::AddTimePeriod(const std::string & enableORdisable,
347 const std::vector<std::string> & objects,
348 const std::vector<Real> & start_times,
349 const std::vector<Real> & end_times,
350 const std::string & name_num,
351 const bool & first_controller)
352{
353 InputParameters params = _factory.getValidParams("TimePeriod");
354 params.set<std::vector<std::string>>(enableORdisable) = objects;
355 params.set<std::vector<Real>>("start_time") = start_times;
356 params.set<std::vector<Real>>("end_time") = end_times;
357 params.set<ExecFlagEnum>("execute_on", true) = {EXEC_INITIAL, EXEC_TIMESTEP_BEGIN};
358 params.set<bool>("reverse_on_false") = first_controller;
359 params.set<bool>("set_sync_times") = true;
360 std::shared_ptr<Control> control =
361 _factory.create<Control>("TimePeriod", _name + name_num, params);
362 _problem->getControlWarehouse().addObject(control);
363}
