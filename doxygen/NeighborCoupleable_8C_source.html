44const ADVariableValue &
45NeighborCoupleable::adCoupledNeighborValue(const std::string & var_name, unsigned int comp) const
46{
47 const auto * var = getVarHelper<MooseVariableField<Real>>(var_name, comp);
48
49 if (!var)
50 return *getADDefaultValue(var_name);
51
52 if (_neighbor_nodal)
53 mooseError("adCoupledNeighborValue cannot be used for nodal compute objects at this time");
54 if (!_c_is_implicit)
55 mooseError("adCoupledNeighborValue returns a data structure with derivatives. Explicit schemes "
56 "use old solution data which do not have derivatives so adCoupledNeighborValue is "
57 "not appropriate. Please use coupledNeighborValue instead");
58
59 return var->adSlnNeighbor();
60}
