84void
85Split::setup(NonlinearSystemBase & nl, const std::string & prefix)
86{
87 // The Split::setup() implementation does not actually depend on any
88 // specific version of PETSc, so there's no need to wrap the entire
89 // function.
90
91 // A reference to the PetscOptions
92 Moose::PetscSupport::PetscOptions & po = _fe_problem.getPetscOptions();
93 // prefix
94 std::string dmprefix = prefix + "dm_moose_";
95
96 // var options
97 if (!_vars.empty())
98 {
99 po.pairs.emplace_back(dmprefix + "vars", Moose::stringify(_vars, ","));
100
101 // check that variables are either field or scalars
102 for (const auto & var : _vars)
103 if (!_fe_problem.hasVariable(var) && !_fe_problem.hasScalarVariable(var))
104 mooseError("Variable '", var, "' specified in split '", name(), "' does not exist");
105 }
106
107 // block options
108 if (!_blocks.empty())
109 po.pairs.emplace_back(dmprefix + "blocks", Moose::stringify(_blocks, ","));
110
111 // side options
112 if (!_sides.empty())
113 po.pairs.emplace_back(dmprefix + "sides", Moose::stringify(_sides, ","));
114
115 // unside options
116 if (!_unsides.empty())
117 po.pairs.emplace_back(dmprefix + "unsides", Moose::stringify(_unsides, ","));
118
119 if (!_splitting.empty())
120 {
121 // If this split has subsplits, it is presumed that the pc_type used to solve this split's
122 // subsystem is fieldsplit
123 // with the following parameters (unless overridden by the user-specified petsc_options below).
124 po.pairs.emplace_back(prefix + "pc_type", "fieldsplit");
125
126 // set Splitting Type
127 const std::string petsc_splitting_type[] = {
128 "additive", "multiplicative", "symmetric_multiplicative", "schur"};
129 po.pairs.emplace_back(prefix + "pc_fieldsplit_type", petsc_splitting_type[_splitting_type]);
130
131 if (_splitting_type == SplittingTypeSchur)
132 {
133 // set Schur Type
134 const std::string petsc_schur_type[] = {"diag", "upper", "lower", "full"};
135 po.pairs.emplace_back(prefix + "pc_fieldsplit_schur_fact_type",
136 petsc_schur_type[_splitting_type]);
137
138 // set Schur Preconditioner
139 const std::string petsc_schur_pre[] = {"self", "selfp", "a11"};
140 po.pairs.emplace_back(prefix + "pc_fieldsplit_schur_precondition",
141 petsc_schur_pre[_schur_pre]);
142 }
143
144 // The DM associated with this split defines the subsplits' geometry.
145 po.pairs.emplace_back(dmprefix + "nfieldsplits", Moose::stringify(_splitting.size()));
146 po.pairs.emplace_back(dmprefix + "fieldsplit_names", Moose::stringify(_splitting, ","));
147
148 // Finally, recursively configure the splits contained within this split.
149 for (const auto & split_name : _splitting)
150 {
151 std::shared_ptr<Split> split = nl.getSplit(split_name);
152 std::string sprefix = prefix + "fieldsplit_" + split_name + "_";
153 split->setup(nl, sprefix);
154 }
155 }
156
157 // Now we set the user-specified petsc options for this split, possibly overriding the above
158 // settings.
159 for (const auto & item : _petsc_options.flags)
160 {
161 // Need to prepend the prefix and strip off the leading '-' on the option name.
162 std::string op(item);
163 if (op[0] != '-')
164 mooseError("Invalid PETSc option name ", op, " for Split ", _name);
165
166 // push back PETSc options
167 po.flags.push_back(prefix + op.substr(1));
168 }
169
170 for (auto & option : _petsc_options.pairs)
171 {
172 // Need to prepend the prefix and strip off the leading '-' on the option name.
173 const std::string & op = option.first;
174 if (op[0] != '-')
175 mooseError("Invalid PETSc option name ", op, " for Split ", _name);
176
177 po.pairs.emplace_back(prefix + op.substr(1), option.second);
178 }
179}
