48Real
49HFEMDirichletBC::computeLowerDQpResidual()
50{
51 if (_uhat)
52 return (_u[_qp] - (*_uhat)[_qp]) * _test_lambda[_i][_qp];
53 else
54 return (_u[_qp] - _value) * _test_lambda[_i][_qp];

63Real
64HFEMDirichletBC::computeLowerDQpJacobian(Moose::ConstraintJacobianType type)
65{
66 switch (type)
67 {
68 case Moose::LowerPrimary:
69 return _test_lambda[_i][_qp] * _phi[_j][_qp];
70
71 case Moose::PrimaryLower:
72 return _phi_lambda[_j][_qp] * _test[_i][_qp];
73
74 default:
75 break;
76 }
77
78 return 0;
79}

81Real
82HFEMDirichletBC::computeLowerDQpOffDiagJacobian(Moose::ConstraintJacobianType type,
83 const MooseVariableFEBase & jvar)
84{
85 if (_uhat_var && jvar.number() == _uhat_var->number() && type == Moose::LowerLower)
86 return -_test_lambda[_i][_qp] * _phi[_j][_qp];

88 return 0;
