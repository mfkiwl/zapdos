1206template <typename T, typename UP_T>
1207void
1208InputParameters::rangeCheck(const std::string & full_name,
1209 const std::string & short_name,
1210 InputParameters::Parameter<std::vector<T>> * param,
1211 std::ostream & oss)
1212{
1213 mooseAssert(param, "Parameter is NULL");
1214
1215 if (!isParamValid(short_name) || _params[short_name]._range_function.empty())
1216 return;
1217
1229 FunctionParserBase<UP_T> fp;
1230 std::vector<std::string> vars;
1231 if (fp.ParseAndDeduceVariables(_params[short_name]._range_function, vars) != -1) // -1 for success
1232 {
1233 oss << "Error parsing expression: " << _params[short_name]._range_function << '\n';
1234 return;
1235 }
1236
1237 // Fparser parameter buffer
1238 std::vector<UP_T> parbuf(vars.size());
1239
1240 // parameter vector
1241 const std::vector<T> & value = param->set();
1242
1243 // iterate over all vector values (maybe ;)
1244 bool need_to_iterate = false;
1245 unsigned int i = 0;
1246 do
1247 {
1248 // set parameters
1249 for (unsigned int j = 0; j < vars.size(); j++)
1250 {
1251 if (vars[j] == short_name)
1252 {
1253 if (value.size() == 0)
1254 {
1255 oss << "Range checking empty vector: " << _params[short_name]._range_function << '\n';
1256 return;
1257 }
1258
1259 parbuf[j] = value[i];
1260 need_to_iterate = true;
1261 }
1262 else if (vars[j] == short_name + "_size")
1263 parbuf[j] = value.size();
1264 else
1265 {
1266 if (vars[j].substr(0, short_name.size() + 1) != short_name + "_")
1267 {
1268 oss << "Error parsing expression: " << _params[short_name]._range_function << '\n';
1269 return;
1270 }
1271 std::istringstream iss(vars[j]);
1272 iss.seekg(short_name.size() + 1);
1273
1274 size_t index;
1275 if (iss >> index && iss.eof())
1276 {
1277 if (index >= value.size())
1278 {
1279 oss << "Error parsing expression: " << _params[short_name]._range_function
1280 << "\nOut of range variable " << vars[j] << '\n';
1281 return;
1282 }
1283 parbuf[j] = value[index];
1284 }
1285 else
1286 {
1287 oss << "Error parsing expression: " << _params[short_name]._range_function
1288 << "\nInvalid variable " << vars[j] << '\n';
1289 return;
1290 }
1291 }
1292 }
1293
1294 // ensure range-checked input file parameter comparison functions
1295 // do absolute floating point comparisons instead of using a default epsilon.
1296 auto tmp_eps = fp.epsilon();
1297 fp.setEpsilon(0);
1298 UP_T result = fp.Eval(&parbuf[0]);
1299 fp.setEpsilon(tmp_eps);
1300
1301 // test function using the parameters determined above
1302 if (fp.EvalError())
1303 {
1304 oss << "Error evaluating expression: " << _params[short_name]._range_function << '\n';
1305 return;
1306 }
1307
1308 if (!result)
1309 {
1310 oss << "Range check failed for parameter " << full_name
1311 << "\n\tExpression: " << _params[short_name]._range_function << "\n";
1312 if (need_to_iterate)
1313 oss << "\t Component: " << i << '\n';
1314 }
1315
1316 } while (need_to_iterate && ++i < value.size());
1317}

1867void
1868InputParameters::transferParam(const InputParameters & source_params,
1869 const std::string & name_in,
1870 const std::string & new_description)
1871{
1872 const auto name = source_params.checkForRename(std::string(name_in));
1873 if (!source_params.have_parameter<T>(name) && !source_params.hasCoupledValue(name))
1874 mooseError("The '",
1875 name_in,
1876 "' parameter could not be transferred because it does not exist with type '",
1877 MooseUtils::prettyCppType<T>(),
1878 "' in the source parameters");
1879 if (name != name_in)
1880 mooseWarning("The transferred parameter " + name_in + " is deprecated in favor of " + name +
1881 " in the source parameters. The new name should likely be used for the parameter "
1882 "transfer instead.");
1883 const std::string description =
1884 new_description.empty() ? source_params.getDescription(name) : new_description;
1885
1886 if (source_params.isParamRequired(name))
1887 {
1888 // Use the name_in that was specified in the transfer
1889 // Check for a variable parameter
1890 if (source_params.hasCoupledValue(name))
1891 addRequiredCoupledVar(name_in, description);
1892 // Enums parameters have a default list of options
1893 else if constexpr (std::is_same_v<MooseEnum, T> || std::is_same_v<MultiMooseEnum, T>)
1894 addRequiredParam<T>(name_in, source_params.get<T>(name), description);
1895 else if (source_params.isRangeChecked(name))
1896 addRequiredRangeCheckedParam<T>(
1897 name_in, source_params.rangeCheckedFunction(name), description);
1898 else
1899 addRequiredParam<T>(name_in, description);
1900 }
1901 else
1902 {
1903 // Check for a variable parameter
1904 if (source_params.hasCoupledValue(name))
1905 {
1906 if (!source_params.hasDefaultCoupledValue(name))
1907 addCoupledVar(name_in, description);
1908 else if (source_params.numberDefaultCoupledValues(name) == 1)
1909 addCoupledVar(name_in, source_params.defaultCoupledValue(name), description);
1910 else
1911 {
1912 std::vector<Real> coupled_values;
1913 for (const auto i : make_range(source_params.numberDefaultCoupledValues(name)))
1914 coupled_values.push_back(source_params.defaultCoupledValue(name, i));
1915 addCoupledVar(name_in, coupled_values, description);
1916 }
1917 }
1918 else if (source_params.isRangeChecked(name))
1919 {
1920 if (source_params.hasDefault(name))
1921 addRangeCheckedParam<T>(name_in,
1922 source_params.get<T>(name),
1923 source_params.rangeCheckedFunction(name),
1924 description);
1925 else
1926 addRangeCheckedParam<T>(name_in, source_params.rangeCheckedFunction(name), description);
1927 }
1928 else if constexpr (std::is_same_v<MooseEnum, T> || std::is_same_v<MultiMooseEnum, T>)
1929 addParam<T>(name_in, source_params.get<T>(name_in), description);
1930 else
1931 {
1932 if (source_params.hasDefault(name))
1933 addParam<T>(name_in, source_params.get<T>(name), description);
1934 else
1935 addParam<T>(name_in, description);
1936 }
1937 }
1938
1939 // Copy other attributes
1940 if (source_params.isPrivate(name))
1941 _params[name_in]._is_private = true;
1942 if (source_params.isControllable(name))
1943 _params[name_in]._controllable = true;
