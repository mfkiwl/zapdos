50template <typename AuxKernelType>
51void
52ComputeElemAuxVarsThread<AuxKernelType>::subdomainChanged()
53{
54 _fe_problem.subdomainSetup(_subdomain, _tid);
55
56 std::set<MooseVariableFEBase *> needed_moose_vars;
57 std::set<unsigned int> needed_mat_props;
58 std::set<TagID> needed_fe_var_matrix_tags;
59 std::set<TagID> needed_fe_var_vector_tags;
60
61 _fe_problem.getMaterialWarehouse().updateBlockFEVariableCoupledVectorTagDependency(
62 _subdomain, needed_fe_var_vector_tags, _tid);
63 if (_aux_kernels.hasActiveBlockObjects(_subdomain, _tid))
64 {
65 const std::vector<std::shared_ptr<AuxKernelType>> & kernels =
66 _aux_kernels.getActiveBlockObjects(_subdomain, _tid);
67 for (const auto & aux : kernels)
68 {
69 aux->subdomainSetup();
70 const std::set<MooseVariableFEBase *> & mv_deps = aux->getMooseVariableDependencies();
71 const std::set<unsigned int> & mp_deps = aux->getMatPropDependencies();
72 needed_moose_vars.insert(mv_deps.begin(), mv_deps.end());
73 needed_mat_props.insert(mp_deps.begin(), mp_deps.end());
74
75 auto & fe_var_coup_vtags = aux->getFEVariableCoupleableVectorTags();
76 needed_fe_var_vector_tags.insert(fe_var_coup_vtags.begin(), fe_var_coup_vtags.end());
77
78 auto & fe_var_coup_mtags = aux->getFEVariableCoupleableMatrixTags();
79 needed_fe_var_matrix_tags.insert(fe_var_coup_mtags.begin(), fe_var_coup_mtags.end());
80 }
81 }
82
83 _fe_problem.setActiveElementalMooseVariables(needed_moose_vars, _tid);
84 _fe_problem.setActiveMaterialProperties(needed_mat_props, _tid);
85 _fe_problem.prepareMaterials(_subdomain, _tid);
86 _fe_problem.setActiveFEVariableCoupleableMatrixTags(needed_fe_var_matrix_tags, _tid);
87 _fe_problem.setActiveFEVariableCoupleableVectorTags(needed_fe_var_vector_tags, _tid);
88}

90template <typename AuxKernelType>
91void
92ComputeElemAuxVarsThread<AuxKernelType>::onElement(const Elem * elem)
93{
94 if (_aux_kernels.hasActiveBlockObjects(_subdomain, _tid))
95 {
96 const std::vector<std::shared_ptr<AuxKernelType>> & kernels =
97 _aux_kernels.getActiveBlockObjects(_subdomain, _tid);
98 _fe_problem.prepare(elem, _tid);
99 _fe_problem.reinitElem(elem, _tid);
100
101 // Set up the sentinel so that, even if reinitMaterials() throws, we
102 // still remember to swap back.
103 SwapBackSentinel sentinel(_fe_problem, &FEProblem::swapBackMaterials, _tid, _need_materials);
104
105 if (_need_materials)
106 _fe_problem.reinitMaterials(elem->subdomain_id(), _tid);
107
108 for (const auto & aux : kernels)
109 {
110 aux->compute();
111 aux->variable().insert(_aux_sys.solution());
112
113 // update the aux solution vector if writable coupled variables are used
114 if (aux->hasWritableCoupledVariables())
115 {
116 for (auto * var : aux->getWritableCoupledVariables())
117 var->insert(_aux_sys.solution());
118
119 _fe_problem.reinitElem(elem, _tid);
120 }
121 }
122 }
123}

155void
156ComputeElemAuxVarsThread<AuxKernelType>::printBlockExecutionInformation() const
157{
158 if (!_fe_problem.shouldPrintExecution(_tid) || _blocks_exec_printed.count(_subdomain) ||

160 return;
161
162 const auto & console = _fe_problem.console();
163 const auto & kernels = _aux_kernels.getActiveBlockObjects(_subdomain, _tid);

165 printExecutionOrdering<AuxKernelType>(kernels, false);
