25FVAdvection::FVAdvection(const InputParameters & params)
26 : FVFluxKernel(params), _velocity(getParam<RealVectorValue>("velocity"))
27{
28 const bool need_more_ghosting =
29 Moose::FV::setInterpolationMethod(*this, _advected_interp_method, "advected_interp_method");
30 if (need_more_ghosting && _tid == 0)
31 {
32 adjustRMGhostLayers(std::max((unsigned short)(2), _pars.get<unsigned short>("ghost_layers")));
33
34 // If we need more ghosting, then we are a second-order nonlinear limiting scheme whose stencil
35 // is liable to change upon wind-direction change. Consequently we need to tell our problem that
36 // it's ok to have new nonzeros which may crop-up after PETSc has shrunk the matrix memory
37 getCheckedPointerParam<FEProblemBase *>("_fe_problem_base")
38 ->setErrorOnJacobianNonzeroReallocation(false);
39 }
40}

42ADReal
43FVAdvection::computeQpResidual()
44{
45 const bool elem_is_upwind = _velocity * _normal >= 0;
46 const auto face =
47 makeFace(*_face_info, Moose::FV::limiterType(_advected_interp_method), elem_is_upwind);
48 ADReal u_interface = _var(face, determineState());
49
50 return _normal * _velocity * u_interface;
51}
