81void
82ComputeMaterialsObjectThread::onElement(const Elem * elem)
83{
84 if (_materials.hasActiveBlockObjects(_subdomain, _tid) ||
85 _discrete_materials.hasActiveBlockObjects(_subdomain, _tid))
86 {
87 _fe_problem.prepare(elem, _tid);
88 _fe_problem.reinitElem(elem, _tid);
89
90 auto & material_data = _material_props.getMaterialData(_tid);
91
92 unsigned int n_points = _assembly[_tid][0]->qRule()->n_points();
93 material_data.resize(n_points);
94
95 if (_has_stateful_props)
96 {
97 if (_discrete_materials.hasActiveBlockObjects(_subdomain, _tid))
98 _material_props.initStatefulProps(
99 _tid, _discrete_materials.getActiveBlockObjects(_subdomain, _tid), n_points, *elem);
100 if (_materials.hasActiveBlockObjects(_subdomain, _tid))
101 _material_props.initStatefulProps(
102 _tid, _materials.getActiveBlockObjects(_subdomain, _tid), n_points, *elem);
103 }
104 }

107void
108ComputeMaterialsObjectThread::onBoundary(const Elem * elem,
109 unsigned int side,
110 BoundaryID bnd_id,
111 const Elem * /*lower_d_elem = nullptr*/)
112{
113 if (_fe_problem.needBoundaryMaterialOnSide(bnd_id, _tid))
114 {
115 _fe_problem.reinitElemFace(elem, side, bnd_id, _tid);
116 unsigned int face_n_points = _assembly[_tid][0]->qRuleFace()->n_points();
117
118 _bnd_material_props.getMaterialData(_tid).resize(face_n_points);
119
120 if (_has_bnd_stateful_props)
121 {
122 // Face Materials
123 if (_discrete_materials[Moose::FACE_MATERIAL_DATA].hasActiveBlockObjects(_subdomain, _tid))
124 _bnd_material_props.initStatefulProps(
125 _tid,
126 _discrete_materials[Moose::FACE_MATERIAL_DATA].getActiveBlockObjects(_subdomain, _tid),
127 face_n_points,
128 *elem,
129 side);
130 if (_materials[Moose::FACE_MATERIAL_DATA].hasActiveBlockObjects(_subdomain, _tid))
131 _bnd_material_props.initStatefulProps(
132 _tid,
133 _materials[Moose::FACE_MATERIAL_DATA].getActiveBlockObjects(_subdomain, _tid),
134 face_n_points,
135 *elem,
136 side);
137
138 // Boundary Materials
139 if (_discrete_materials.hasActiveBoundaryObjects(bnd_id, _tid))
140 _bnd_material_props.initStatefulProps(
141 _tid, _materials.getActiveBoundaryObjects(bnd_id, _tid), face_n_points, *elem, side);
142 if (_materials.hasActiveBoundaryObjects(bnd_id, _tid))
143 _bnd_material_props.initStatefulProps(
144 _tid, _materials.getActiveBoundaryObjects(bnd_id, _tid), face_n_points, *elem, side);
145 }

158 _bnd_material_props.getMaterialData(_tid).resize(face_n_points);
159 _neighbor_material_props.getMaterialData(_tid).resize(face_n_points);

162 {

164 _bnd_material_props.initStatefulProps(

206void
207ComputeMaterialsObjectThread::onInterface(const Elem * elem, unsigned int side, BoundaryID bnd_id)
208{
209 if (!_fe_problem.needInterfaceMaterialOnSide(bnd_id, _tid))
210 return;
211
212 _fe_problem.reinitElemFace(elem, side, bnd_id, _tid);
213 unsigned int face_n_points = _assembly[_tid][0]->qRuleFace()->n_points();
214
215 _bnd_material_props.getMaterialData(_tid).resize(face_n_points);
216 _neighbor_material_props.getMaterialData(_tid).resize(face_n_points);
217
218 if (_has_bnd_stateful_props)
219 {
220 // Face Materials
221 if (_discrete_materials[Moose::FACE_MATERIAL_DATA].hasActiveBlockObjects(_subdomain, _tid))
222 _bnd_material_props.initStatefulProps(
223 _tid,
224 _discrete_materials[Moose::FACE_MATERIAL_DATA].getActiveBlockObjects(_subdomain, _tid),
225 face_n_points,
226 *elem,
227 side);
228
229 if (_materials[Moose::FACE_MATERIAL_DATA].hasActiveBlockObjects(_subdomain, _tid))
230 _bnd_material_props.initStatefulProps(
231 _tid,
232 _materials[Moose::FACE_MATERIAL_DATA].getActiveBlockObjects(_subdomain, _tid),
233 face_n_points,
234 *elem,
235 side);
236
237 // Boundary Materials
238 if (_discrete_materials.hasActiveBoundaryObjects(bnd_id, _tid))
239 _bnd_material_props.initStatefulProps(
240 _tid, _materials.getActiveBoundaryObjects(bnd_id, _tid), face_n_points, *elem, side);
241
242 if (_materials.hasActiveBoundaryObjects(bnd_id, _tid))
243 _bnd_material_props.initStatefulProps(
244 _tid, _materials.getActiveBoundaryObjects(bnd_id, _tid), face_n_points, *elem, side);
245 }
246
247 const Elem * neighbor = elem->neighbor_ptr(side);
248 unsigned int neighbor_side = neighbor->which_neighbor_am_i(_assembly[_tid][0]->elem());
249
250 // Do we have neighbor stateful properties or do we have stateful interface material properties?
251 // If either then we need to reinit the neighbor, so at least at a minimum _neighbor_elem isn't
252 // NULL!
253 if (neighbor->active() &&
254 (_has_neighbor_stateful_props ||
255 (_has_bnd_stateful_props && _interface_materials.hasActiveBoundaryObjects(bnd_id, _tid))))
256 _fe_problem.reinitNeighbor(elem, side, _tid);
257
258 if (_has_neighbor_stateful_props && neighbor->active())
259 {
260 // Neighbor Materials
261 if (_discrete_materials[Moose::NEIGHBOR_MATERIAL_DATA].hasActiveBlockObjects(
262 neighbor->subdomain_id(), _tid))
263 _neighbor_material_props.initStatefulProps(
264 _tid,
265 _discrete_materials[Moose::NEIGHBOR_MATERIAL_DATA].getActiveBlockObjects(
266 neighbor->subdomain_id(), _tid),
267 face_n_points,
268 *elem,
269 side);
270
271 if (_materials[Moose::NEIGHBOR_MATERIAL_DATA].hasActiveBlockObjects(neighbor->subdomain_id(),
272 _tid))
273 _neighbor_material_props.initStatefulProps(
274 _tid,
275 _materials[Moose::NEIGHBOR_MATERIAL_DATA].getActiveBlockObjects(neighbor->subdomain_id(),
276 _tid),
277 face_n_points,
278 *neighbor,
279 neighbor_side);
280 }
281
282 // Interface Materials. Make sure we do these after neighbors
283 if (_has_bnd_stateful_props && _interface_materials.hasActiveBoundaryObjects(bnd_id, _tid))
284 _bnd_material_props.initStatefulProps(
285 _tid,
286 _interface_materials.getActiveBoundaryObjects(bnd_id, _tid),
287 face_n_points,
288 *elem,
289 side);
290}
