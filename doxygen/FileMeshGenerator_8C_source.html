60std::unique_ptr<MeshBase>
61FileMeshGenerator::generate()
62{
63 auto mesh = buildMeshBaseObject();
64
65 // Figure out if we are reading an Exodus file, but not Tetgen (*.ele)
66 bool exodus = (_file_name.rfind(".exd") < _file_name.size() ||
67 _file_name.rfind(".e") < _file_name.size()) &&
68 _file_name.rfind(".ele") == std::string::npos;
69 bool has_exodus_integers = isParamValid("exodus_extra_element_integers");
70 bool restart_exodus = (getParam<bool>("use_for_exodus_restart") && _app.getExodusFileRestart());
71 if (exodus)
72 {
73 auto exreader = std::make_shared<ExodusII_IO>(*mesh);
74
75 if (has_exodus_integers)
76 exreader->set_extra_integer_vars(
77 getParam<std::vector<std::string>>("exodus_extra_element_integers"));
78
79 if (restart_exodus)
80 {
81 _app.setExReaderForRestart(std::move(exreader));
82 exreader->read(_file_name);
83 mesh->allow_renumbering(false);
84 }
85 else
86 {
87 if (mesh->processor_id() == 0)
88 {
89 exreader->read(_file_name);
90
91 if (getParam<bool>("clear_spline_nodes"))
92 MeshTools::clear_spline_nodes(*mesh);
93 }
94 MeshCommunication().broadcast(*mesh);
95 }
96 // Skip partitioning if the user requested it
97 if (_skip_partitioning)
98 mesh->skip_partitioning(true);
99 mesh->prepare_for_use();
100 mesh->skip_partitioning(false);
101 }
102 else
103 {
104 if (_pars.isParamSetByUser("exodus_extra_element_integers"))
105 mooseError("\"exodus_extra_element_integers\" should be given only for Exodus mesh files");
106 if (_pars.isParamSetByUser("use_for_exodus_restart"))
107 mooseError("\"use_for_exodus_restart\" should be given only for Exodus mesh files");
108
109 // Supports old suffix (xxxx_mesh.cpr -> xxxx-mesh.cpr) and LATEST
110 const auto file_name = deduceCheckpointPath(*this, _file_name);
111
112 mesh->skip_partitioning(_skip_partitioning);
113 mesh->allow_renumbering(_allow_renumbering);
114 mesh->read(file_name);
115
116 // Load the meta data if it is available
117 _app.possiblyLoadRestartableMetaData(MooseApp::MESH_META_DATA, (std::string)file_name);
118 }
119
120 return dynamic_pointer_cast<MeshBase>(mesh);
121}

123std::string
124FileMeshGenerator::deduceCheckpointPath(const MooseObject & object, const std::string & file_name)
125{
126 // Just exists, use it
127 if (MooseUtils::pathExists(file_name))
128 return file_name;
129
130 // xxxx_mesh.cpr -> xxxx-mesh.cpr
131 const std::string old_ending = "_mesh.cpr";
132 if (std::equal(old_ending.rbegin(), old_ending.rend(), file_name.rbegin()))
133 {
134 const std::string new_ending = "-mesh.cpr";
135 auto new_path = file_name;
136 new_path.replace(new_path.size() - old_ending.size(), old_ending.size(), new_ending, 0);
137 if (MooseUtils::pathExists(new_path))
138 {
139 std::stringstream warning;
140 warning
141 << "The supplied checkpoint " << std::filesystem::path(file_name).filename()
142 << " uses the previous default checkpoint suffix of \"" << old_ending
143 << "\".\nThe new default checkpoint suffix is \"" << new_ending << "\".\n\n"
144 << "Your supplied checkpoint was not found, but one with the new ending was found in\n\""
145 << new_path << "\".\n\n"
146 << "The above checkpoint is being used instead.\nYou should modify your input "
147 "accordingly.";
148 object.paramWarning("file", warning.str());
149 return new_path;
150 }
151 }
152
153 // LATEST
154 return MooseUtils::convertLatestCheckpoint(file_name) + object.getMooseApp().checkpointSuffix();
155}
