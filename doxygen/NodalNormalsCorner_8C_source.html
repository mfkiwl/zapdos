39void
40NodalNormalsCorner::execute()
41{
42 std::scoped_lock lock(NodalNormalsPreprocessor::_nodal_normals_mutex);
43 NumericVector<Number> & sln = _aux.solution();
44
45 // Get a reference to our BoundaryInfo object
46 BoundaryInfo & boundary_info = _mesh.getMesh().get_boundary_info();
47
48 for (unsigned int nd = 0; nd < _current_side_elem->n_nodes(); nd++)
49 {
50 const Node * node = _current_side_elem->node_ptr(nd);
51 if (boundary_info.has_boundary_id(node, _corner_boundary_id) &&
52 node->n_dofs(_aux.number(),
53 _fe_problem
54 .getVariable(_tid,
55 "nodal_normal_x",
56 Moose::VarKindType::VAR_AUXILIARY,
57 Moose::VarFieldType::VAR_FIELD_STANDARD)
58 .number()) > 0)
59 {
60 dof_id_type dof_x = node->dof_number(_aux.number(),
61 _fe_problem
62 .getVariable(_tid,
63 "nodal_normal_x",
64 Moose::VarKindType::VAR_AUXILIARY,
65 Moose::VarFieldType::VAR_FIELD_STANDARD)
66 .number(),
67 0);
68 dof_id_type dof_y = node->dof_number(_aux.number(),
69 _fe_problem
70 .getVariable(_tid,
71 "nodal_normal_y",
72 Moose::VarKindType::VAR_AUXILIARY,
73 Moose::VarFieldType::VAR_FIELD_STANDARD)
74 .number(),
75 0);
76 dof_id_type dof_z = node->dof_number(_aux.number(),
77 _fe_problem
78 .getVariable(_tid,
79 "nodal_normal_z",
80 Moose::VarKindType::VAR_AUXILIARY,
81 Moose::VarFieldType::VAR_FIELD_STANDARD)
82 .number(),
83 0);
84
85 // substitute the normal from the face, we are going to have at least one normal every time
86 sln.add(dof_x, _normals[0](0));
87 sln.add(dof_y, _normals[0](1));
88 sln.add(dof_z, _normals[0](2));
89 }
90 }
91}
