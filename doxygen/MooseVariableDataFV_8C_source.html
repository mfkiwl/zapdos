30template <typename OutputType>
31MooseVariableDataFV<OutputType>::MooseVariableDataFV(const MooseVariableFV<OutputType> & var,
32 SystemBase & sys,
33 THREAD_ID tid,
34 Moose::ElementType element_type,
35 const Elem * const & elem)
36
37 : MooseVariableDataBase<OutputType>(var, sys, tid),
38 MeshChangedInterface(var.parameters()),
39 _var(var),
40 _fe_type(_var.feType()),
41 _var_num(_var.number()),
42 _assembly(_subproblem.assembly(_tid, var.kind() == Moose::VAR_NONLINEAR ? sys.number() : 0)),
43 _element_type(element_type),
44 _ad_zero(0),
45 _need_second(false),
46 _need_second_old(false),
47 _need_second_older(false),
48 _need_second_previous_nl(false),
49 _need_curl(false),
50 _need_curl_old(false),
51 _need_curl_older(false),
52 // for FV variables, they use each other's ad_u values to compute ghost cell
53 // values - we don't have any way to propogate these inter-variable-data
54 // dependencies. So if something needs an ad_u value, that need isn't
55 // propogated through to both the element and the neighbor
56 // data structures. So instead just set _need_ad+_need_ad_u values to true always.
57 _need_ad(true),
58 _need_ad_u(true),
59 _need_ad_u_dot(false),
60 _need_ad_u_dotdot(false),
61 _need_ad_grad_u(false),
62 _need_ad_grad_u_dot(false),
63 _need_ad_second_u(false),
64 _time_integrator(_sys.getTimeIntegrator()),
65 _elem(elem),
66 _displaced(dynamic_cast<const DisplacedSystem *>(&_sys) ? true : false),
67 _qrule(nullptr)
68{
69 _fv_elemental_kernel_query_cache =
70 _subproblem.getMooseApp().theWarehouse().query().template condition<AttribSystem>(
71 "FVElementalKernel");
72 _fv_flux_kernel_query_cache =
73 _subproblem.getMooseApp().theWarehouse().query().template condition<AttribSystem>(
74 "FVFluxKernel");
75}

429template <typename OutputType>
430void
431MooseVariableDataFV<OutputType>::computeValues()
432{
433 initDofIndices();
434 initializeSolnVars();
435
436 unsigned int num_dofs = _dof_indices.size();
437
438 if (num_dofs > 0)
439 fetchDoFValues();
440 else
441 // We don't have any dofs. There's nothing to do
442 return;
443
444 mooseAssert(num_dofs == 1 && _vector_tags_dof_u[_solution_tag].size() == 1,
445 "There should only be one dof per elem for FV variables");
446
447 bool is_transient = _subproblem.isTransient();
448 const auto nqp = _qrule->n_points();
449 auto && active_coupleable_matrix_tags =
450 _sys.subproblem().getActiveFEVariableCoupleableMatrixTags(_tid);
451
452 bool second_required =
453 _need_second || _need_second_old || _need_second_older || _need_second_previous_nl;
454 bool curl_required = _need_curl || _need_curl_old;
455
456 for (const auto qp : make_range(nqp))
457 {
458 if (is_transient)
459 {
460 if (_need_u_dot)
461 _u_dot[qp] = _dof_values_dot[0];
462
463 if (_need_u_dotdot)
464 _u_dotdot[qp] = _dof_values_dotdot[0];
465
466 if (_need_u_dot_old)
467 _u_dot_old[qp] = _dof_values_dot_old[0];
468
469 if (_need_u_dotdot_old)
470 _u_dotdot_old[qp] = _dof_values_dotdot_old[0];
471
472 if (_need_du_dot_du)
473 _du_dot_du[qp] = _dof_du_dot_du[0];
474
475 if (_need_du_dotdot_du)
476 _du_dotdot_du[qp] = _dof_du_dotdot_du[0];
477 }
478
479 if (second_required)
480 {
481 if (_need_second)
482 _second_u[qp] = 0;
483
484 if (_need_second_previous_nl)
485 _second_u_previous_nl[qp] = 0;
486
487 if (is_transient)
488 {
489 if (_need_second_old)
490 _second_u_old[qp] = 0;
491
492 if (_need_second_older)
493 _second_u_older[qp] = 0;
494 }
495 }
496
497 if (curl_required)
498 {
499 if (_need_curl)
500 _curl_u[qp] = 0;
501
502 if (is_transient && _need_curl_old)
503 _curl_u_old[qp] = 0;
504 }
505
506 for (auto tag : _required_vector_tags)
507 if (_need_vector_tag_u[tag])
508 _vector_tag_u[tag][qp] = _vector_tags_dof_u[tag][0];
509
510 for (auto tag : active_coupleable_matrix_tags)
511 if (_need_matrix_tag_u[tag])
512 _matrix_tag_u[tag][qp] = _matrix_tags_dof_u[tag][0];
513 }
514
515 // Automatic differentiation
516 if (_need_ad)
517 computeAD(num_dofs, nqp);
518}

520template <typename OutputType>
521void
522MooseVariableDataFV<OutputType>::computeAD(const unsigned int num_dofs, const unsigned int nqp)
523{
524 // This query and if check prevent running this code when we have FV
525 // variables, but no kernels. When this happens, maxVarNDofsPerElem is
526 // not computed (because no kernels) and is zero giving nonsense offsets for
527 // AD stuff. So we just skip all this when that is the case. Maybe there
528 // is a better way to do this - like just checking if getMaxVarNDofsPerElem
529 // returns zero?
530 std::vector<FVKernel *> ks;
531 _fv_elemental_kernel_query_cache.queryInto(ks);
532 if (ks.size() == 0)
533 {
534 _fv_flux_kernel_query_cache.queryInto(ks);
535 if (ks.size() == 0)
536 return;
537 }
538
539 _ad_dof_values.resize(num_dofs);
540 if (_need_ad_u)
541 _ad_u.resize(nqp);
542
543 if (_need_ad_grad_u)
544 _ad_grad_u.resize(nqp);
545
546 if (_need_ad_second_u)
547 _ad_second_u.resize(nqp);
548
549 if (_need_ad_u_dot)
550 {
551 _ad_dofs_dot.resize(num_dofs);
552 _ad_u_dot.resize(nqp);
553 }
554
555 if (_need_ad_u_dotdot)
556 {
557 _ad_dofs_dotdot.resize(num_dofs);
558 _ad_u_dotdot.resize(nqp);
559 }
560
561 if (_need_ad_second_u)
562 assignForAllQps(0, _ad_second_u, nqp);
563
564 if (_need_ad_u_dot)
565 assignForAllQps(_ad_zero, _ad_u_dot, nqp);
566
567 if (_need_ad_u_dotdot)
568 assignForAllQps(_ad_zero, _ad_u_dotdot, nqp);
569
570 const bool do_derivatives =
571 ADReal::do_derivatives && _sys.number() == _subproblem.currentNlSysNum();
572
573 for (unsigned int i = 0; i < num_dofs; i++)
574 {
575 _ad_dof_values[i] = (*_sys.currentSolution())(_dof_indices[i]);
576
577 // NOTE! You have to do this AFTER setting the value!
578 if (do_derivatives)
579 Moose::derivInsert(_ad_dof_values[i].derivatives(), _dof_indices[i], 1.);
580
581 if (_need_ad_u_dot && safeToComputeADUDot() && _time_integrator->dt())
582 {
583 _ad_dofs_dot[i] = _ad_dof_values[i];
584 _time_integrator->computeADTimeDerivatives(_ad_dofs_dot[i],
585 _dof_indices[i],
586 _need_ad_u_dotdot ? _ad_dofs_dotdot[i]
587 : _ad_real_dummy);
588 }
589 }
590
591 if (_need_ad_u)
592 assignForAllQps(_ad_dof_values[0], _ad_u, nqp);
593
594 if (_need_ad_grad_u)
595 assignForAllQps(static_cast<const MooseVariableFV<OutputType> &>(_var).adGradSln(
596 _elem, Moose::currentState()),
597 _ad_grad_u,
598 nqp);
599
600 if (_need_ad_u_dot)
601 {
602 if (safeToComputeADUDot())
603 {
604 assignForAllQps(_ad_dofs_dot[0], _ad_u_dot, nqp);
605 if (_need_ad_u_dotdot)
606 assignForAllQps(_ad_dofs_dotdot[0], _ad_u_dotdot, nqp);
607 }
608 else
609 {
610 assignForAllQps(_u_dot[0], _ad_u_dot, nqp);
611 if (_need_ad_u_dotdot)
612 assignForAllQps(_u_dotdot[0], _ad_u_dotdot, nqp);
613 }
614 }
615}
