49void
50ComputeIndicatorThread::subdomainChanged()
51{
52 _fe_problem.subdomainSetup(_subdomain, _tid);
53
54 _indicator_whs.subdomainSetup(_tid);
55 _internal_side_indicators.subdomainSetup(_tid);
56
57 std::set<MooseVariableFEBase *> needed_moose_vars;
58 _indicator_whs.updateVariableDependency(needed_moose_vars, _tid);
59 _internal_side_indicators.updateVariableDependency(needed_moose_vars, _tid);
60 _fe_problem.setActiveElementalMooseVariables(needed_moose_vars, _tid);
61
62 // Update variable coupleable vector tags
63 std::set<TagID> needed_var_vector_tags;
64 _indicator_whs.updateBlockFEVariableCoupledVectorTagDependency(
65 _subdomain, needed_var_vector_tags, _tid);
66 _internal_side_indicators.updateBlockFEVariableCoupledVectorTagDependency(
67 _subdomain, needed_var_vector_tags, _tid);
68 _fe_problem.getMaterialWarehouse().updateBlockFEVariableCoupledVectorTagDependency(
69 _subdomain, needed_var_vector_tags, _tid);
70 _fe_problem.setActiveFEVariableCoupleableVectorTags(needed_var_vector_tags, _tid);
71
72 std::set<unsigned int> needed_mat_props;
73 _indicator_whs.updateMatPropDependency(needed_mat_props, _tid);
74 _internal_side_indicators.updateMatPropDependency(needed_mat_props, _tid);
75 _fe_problem.setActiveMaterialProperties(needed_mat_props, _tid);
76
77 _fe_problem.prepareMaterials(_subdomain, _tid);
78}

80void
81ComputeIndicatorThread::onElement(const Elem * elem)
82{
83 for (auto * var : _aux_sys._elem_vars[_tid])
84 var->prepareAux();
85
86 _fe_problem.prepare(elem, _tid);
87 _fe_problem.reinitElem(elem, _tid);
88
89 // Set up Sentinel class so that, even if reinitMaterials() throws, we
90 // still remember to swap back during stack unwinding.
91 SwapBackSentinel sentinel(_fe_problem, &FEProblemBase::swapBackMaterials, _tid);
92
93 _fe_problem.reinitMaterials(_subdomain, _tid);
94
95 // Compute
96 if (!_finalize)
97 {
98 if (_indicator_whs.hasActiveBlockObjects(_subdomain, _tid))
99 {
100 const std::vector<std::shared_ptr<Indicator>> & indicators =
101 _indicator_whs.getActiveBlockObjects(_subdomain, _tid);
102 for (const auto & indicator : indicators)
103 indicator->computeIndicator();
104 }
105 }
106
107 // Finalize
108 else
109 {
110 if (_indicator_whs.hasActiveBlockObjects(_subdomain, _tid))
111 {
112 const std::vector<std::shared_ptr<Indicator>> & indicators =
113 _indicator_whs.getActiveBlockObjects(_subdomain, _tid);
114 for (const auto & indicator : indicators)
115 indicator->finalize();
116 }
117
118 if (_internal_side_indicators.hasActiveBlockObjects(_subdomain, _tid))
119 {
120 const std::vector<std::shared_ptr<InternalSideIndicator>> & internal_indicators =
121 _internal_side_indicators.getActiveBlockObjects(_subdomain, _tid);
122 for (const auto & internal_indicator : internal_indicators)
123 internal_indicator->finalize();
124 }
125 }
126
127 if (!_finalize) // During finalize the Indicators should be setting values in the vectors manually
128 {
129 Threads::spin_mutex::scoped_lock lock(Threads::spin_mtx);
130 for (auto * var : _aux_sys._elem_vars[_tid])
131 var->add(_aux_sys.solution());
132 }
133}

143void
144ComputeIndicatorThread::onInternalSide(const Elem * elem, unsigned int side)
145{
146 if (_finalize) // If finalizing we only do something on the elements
147 return;
148
149 // Pointer to the neighbor we are currently working on.
150 const Elem * neighbor = elem->neighbor_ptr(side);
151
152 for (auto * var : _aux_sys._elem_vars[_tid])
153 var->prepareAux();
154
155 SubdomainID block_id = elem->subdomain_id();
156 if (_internal_side_indicators.hasActiveBlockObjects(block_id, _tid))
157 {
158 _fe_problem.reinitNeighbor(elem, side, _tid);
159
160 // Set up Sentinels so that, even if one of the reinitMaterialsXXX() calls throws, we
161 // still remember to swap back during stack unwinding.
162 SwapBackSentinel face_sentinel(_fe_problem, &FEProblemBase::swapBackMaterialsFace, _tid);
163 _fe_problem.reinitMaterialsFace(block_id, _tid);
164
165 SwapBackSentinel neighbor_sentinel(
166 _fe_problem, &FEProblemBase::swapBackMaterialsNeighbor, _tid);
167 _fe_problem.reinitMaterialsNeighbor(neighbor->subdomain_id(), _tid);
168
169 const std::vector<std::shared_ptr<InternalSideIndicator>> & indicators =
170 _internal_side_indicators.getActiveBlockObjects(block_id, _tid);
171 for (const auto & indicator : indicators)
172 indicator->computeIndicator();
173 }
174}

193void
194ComputeIndicatorThread::printGeneralExecutionInformation() const
195{
196 if (!_fe_problem.shouldPrintExecution(_tid))
197 return;
198
199 const auto & console = _fe_problem.console();
200 const auto & execute_on = _fe_problem.getCurrentExecuteOnFlag();
201 if (!_finalize)
202 console << "[DBG] Executing indicators on elements then on internal sides on " << execute_on
203 << std::endl;
204 else
205 console << "[DBG] Finalizing indicator loop" << std::endl;
206}

208void
209ComputeIndicatorThread::printBlockExecutionInformation() const
210{
211 if (!_fe_problem.shouldPrintExecution(_tid) || _blocks_exec_printed.count(_subdomain))
212 return;
213
214 const auto & console = _fe_problem.console();
215 if (_indicator_whs.hasActiveBlockObjects(_subdomain, _tid))
216 {
217 const auto & indicators = _indicator_whs.getActiveBlockObjects(_subdomain, _tid);
218 console << "[DBG] Ordering of element indicators on block " << _subdomain << std::endl;
219 printExecutionOrdering<Indicator>(indicators, false);
220 }
221 if (_internal_side_indicators.hasActiveBlockObjects(_subdomain, _tid))
222 {
223 const auto & indicators = _internal_side_indicators.getActiveBlockObjects(_subdomain, _tid);
224 console << "[DBG] Ordering of element internal sides indicators on block " << _subdomain
225 << std::endl;
226 printExecutionOrdering<InternalSideIndicator>(indicators, false);
227 }
228 _blocks_exec_printed.insert(_subdomain);
229}
