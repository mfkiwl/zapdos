385template <typename T, bool is_ad>
386const GenericMaterialProperty<T, is_ad> &
387MaterialBase::getGenericZeroMaterialPropertyByName(const std::string & prop_name)
388{
389 checkExecutionStage();
390 auto & preload_with_zero = materialData().getProperty<T, is_ad>(prop_name, 0, *this);
391
392 _requested_props.insert(prop_name);
393 registerPropName(prop_name, true, 0);
394 _fe_problem.markMatPropRequested(prop_name);
395
396 // Register this material on these blocks and boundaries as a zero property with relaxed
397 // consistency checking
398 for (std::set<SubdomainID>::const_iterator it = blockIDs().begin(); it != blockIDs().end(); ++it)
399 _fe_problem.storeSubdomainZeroMatProp(*it, prop_name);
400 for (std::set<BoundaryID>::const_iterator it = boundaryIDs().begin(); it != boundaryIDs().end();
401 ++it)
402 _fe_problem.storeBoundaryZeroMatProp(*it, prop_name);
403
404 // set values for all qpoints to zero
405 // (in multiapp scenarios getMaxQps can return different values in each app; we need the max)
406 unsigned int nqp = _fe_problem.getMaxQps();
407 if (nqp > preload_with_zero.size())
408 preload_with_zero.resize(nqp);
409 for (unsigned int qp = 0; qp < nqp; ++qp)
410 MathUtils::mooseSetToZero(preload_with_zero[qp]);
411
412 return preload_with_zero;
413}

447template <typename Consumers>
448std::deque<MaterialBase *>
449MaterialBase::buildRequiredMaterials(const Consumers & mat_consumers,
450 const std::vector<std::shared_ptr<MaterialBase>> & mats,
451 const bool allow_stateful)
452{
453 std::deque<MaterialBase *> required_mats;
454
455 std::unordered_set<unsigned int> needed_mat_props;
456 for (const auto & consumer : mat_consumers)
457 {
458 const auto & mp_deps = consumer->getMatPropDependencies();
459 needed_mat_props.insert(mp_deps.begin(), mp_deps.end());
460 }
461
462 // A predicate of calling this function is that these materials come in already sorted by
463 // dependency with the front of the container having no other material dependencies and following
464 // materials potentially depending on the ones in front of them. So we can start at the back and
465 // iterate forward checking whether the current material supplies anything that is needed, and if
466 // not we discard it
467 for (auto it = mats.rbegin(); it != mats.rend(); ++it)
468 {
469 auto * const mat = it->get();
470 bool supplies_needed = false;
471
472 const auto & supplied_props = mat->getSuppliedPropIDs();
473
474 // Do O(N) with the small container
475 for (const auto supplied_prop : supplied_props)
476 {
477 if (needed_mat_props.count(supplied_prop))
478 {
479 supplies_needed = true;
480 break;
481 }
482 }
483
484 if (!supplies_needed)
485 continue;
486
487 if (!allow_stateful && mat->hasStatefulProperties())
488 ::mooseError(
489 "Someone called buildRequiredMaterials with allow_stateful = false but a material "
490 "dependency ",
491 mat->name(),
492 " computes stateful properties.");
493
494 const auto & mp_deps = mat->getMatPropDependencies();
495 needed_mat_props.insert(mp_deps.begin(), mp_deps.end());
496 required_mats.push_front(mat);
497 }
498
499 return required_mats;
500}
