571template <typename T>
572SymmetricRankFourTensorTempl<T>
573SymmetricRankTwoTensorTempl<T>::positiveProjectionEigenDecomposition(
574 std::vector<T> & eigval, RankTwoTensorTempl<T> & eigvec) const
575{
576 // The calculate of projection tensor follows
577 // C. Miehe and M. Lambrecht, Commun. Numer. Meth. Engng 2001; 17:337~353
578
579 // Compute eigenvectors and eigenvalues of this tensor
580 if (MooseUtils::IsLikeReal<T>::value)
581 {
582 this->symmetricEigenvaluesEigenvectors(eigval, eigvec);
583
584 // Separate out positive and negative eigen values
585 std::array<T, N> epos;
586 std::array<T, N> d;
587 for (unsigned int i = 0; i < N; ++i)
588 {
589 epos[i] = (std::abs(eigval[i]) + eigval[i]) / 2.0;
590 d[i] = 0 < eigval[i] ? 1.0 : 0.0;
591 }
592
593 // projection tensor
594 SymmetricRankFourTensorTempl<T> proj_pos;
595
596 for (unsigned int a = 0; a < Ndim; ++a)
597 {
598 const auto Ma = SymmetricRankTwoTensorTempl<T>::selfOuterProduct(eigvec.column(a));
599 proj_pos += d[a] * Ma.outerProduct(Ma);
600 }
601
602 for (const auto a : make_range(Ndim))
603 for (const auto b : make_range(a))
604 {
605 const auto Ma = SymmetricRankTwoTensorTempl<T>::selfOuterProduct(eigvec.column(a));
606 const auto Mb = SymmetricRankTwoTensorTempl<T>::selfOuterProduct(eigvec.column(b));
607
608 SymmetricRankFourTensorTempl<T> Gabba(SymmetricRankFourTensorTempl<T>::initNone);
609 for (const auto aa : make_range(N))
610 for (const auto bb : make_range(N))
611 {
612 const auto i = SymmetricRankFourTensorTempl<T>::full_index[aa][bb][0];
613 const auto j = SymmetricRankFourTensorTempl<T>::full_index[aa][bb][1];
614 const auto k = SymmetricRankFourTensorTempl<T>::full_index[aa][bb][2];
615 const auto l = SymmetricRankFourTensorTempl<T>::full_index[aa][bb][3];
616
617 Gabba(aa, bb) = (Ma(i, k) * Mb(j, l) + Ma(i, l) * Mb(j, k) + Ma(j, l) * Mb(i, k) +
618 Ma(j, k) * Mb(i, l)) *
619 SymmetricRankFourTensorTempl<T>::mandelFactor(aa, bb);
620 }
621
622 T theta_ab;
623 if (!MooseUtils::relativeFuzzyEqual(eigval[a], eigval[b]))
624 theta_ab = 0.5 * (epos[a] - epos[b]) / (eigval[a] - eigval[b]);
625 else
626 theta_ab = 0.25 * (d[a] + d[b]);
627
628 proj_pos += theta_ab * Gabba;
629 }
630 return proj_pos;
631 }
632 else
633 mooseError("positiveProjectionEigenDecomposition is only available for ordered tensor "
634 "component types");
635}

637template <typename T>
638T
639SymmetricRankTwoTensorTempl<T>::sin3Lode(const T & r0, const T & r0_value) const
640{
641 if (MooseUtils::IsLikeReal<T>::value)
642 {
643 T bar = secondInvariant();
644 if (bar <= r0)
645 // in this case the Lode angle is not defined
646 return r0_value;
647 else
648 // the min and max here gaurd against precision-loss when bar is tiny but nonzero.
649 return std::max(std::min(thirdInvariant() * -1.5 * std::sqrt(3.0) / std::pow(bar, 1.5), 1.0),
650 -1.0);
651 }
652 else
653 mooseError("sin3Lode is only available for ordered tensor component types");
654}

656template <typename T>
657SymmetricRankTwoTensorTempl<T>
658SymmetricRankTwoTensorTempl<T>::dsin3Lode(const T & r0) const
659{
660 if (MooseUtils::IsLikeReal<T>::value)
661 {
662 T bar = secondInvariant();
663 if (bar <= r0)
664 return SymmetricRankTwoTensorTempl<T>();
665 else
666 return -1.5 * std::sqrt(3.0) *
667 (dthirdInvariant() / std::pow(bar, 1.5) -
668 1.5 * dsecondInvariant() * thirdInvariant() / std::pow(bar, 2.5));
669 }
670 else
671 mooseError("dsin3Lode is only available for ordered tensor component types");
672}
