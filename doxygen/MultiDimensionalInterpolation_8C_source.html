32template <typename T>
33void
34MultiDimensionalInterpolationTempl<T>::setData(const std::vector<std::vector<Real>> & base_points,
35 const MultiIndex<Real> & data)
36{
37 _degenerate_interpolation = false;
38 _setup_complete = false;
39
40 // stash away original dimension of data because we might modify it
41 _original_dim = data.dim();
42 _degenerate_index.resize(_original_dim, false);
43
44 // need to check this here to make sure the next loop is safe
45 if (data.dim() != base_points.size())
46 {
47 std::ostringstream oss;
48 oss << "Leading dimension of base_points " << base_points.size() << " and data dimensionality "
49 << data.dim() << " are inconsistent.";
50 throw std::domain_error(oss.str());
51 }
52
53 // first a corner case is addressed; if the size of the MultiIndex in one
54 // dimension is 1, then finding the interpolation "stencil" becomes much
55 // harder. It is easier to slice the data array and keep track of which values
56 // to discard in the interpolation routines
57 MultiIndex<Real>::size_type slice_dimensions;
58 MultiIndex<Real>::size_type slice_indices;
59 MultiIndex<Real>::size_type size = data.size();
60 std::vector<std::vector<Real>> modified_base_points;
61 for (unsigned int j = 0; j < data.dim(); ++j)
62 if (size[j] == 1)
63 {
64 slice_dimensions.push_back(j);
65 slice_indices.push_back(0);
66 _degenerate_index[j] = true;
67 }
68 else
69 modified_base_points.push_back(base_points[j]);
70
71 // if all dimensions have size 1, then there is only one value to return
72 // this is a weird corner case that we should support to make life easier
73 // for whoever uses this object
74 // In this case, we can just use the passed base_points and data objects
75 // and short-cut the computation in the interpolation function
76 if (modified_base_points.size() == 0)
77 {
78 _degenerate_interpolation = true;
79 _base_points = base_points;
80 _data = data;
81 }
82 else
83 {
84 // set the data
85 _base_points = modified_base_points;
86 _data = data.slice(slice_dimensions, slice_indices);
87 }
88 errorCheck();
89}

91template <typename T>
92void
93MultiDimensionalInterpolationTempl<T>::errorCheck()
94{
95 if (_base_points.size() != _data.dim())
96 throw std::domain_error("Size of the leading dimension of base points must be equal to "
97 "dimensionality of data array");
98
99 MultiIndex<Real>::size_type shape = _data.size();
100 for (unsigned int j = 0; j < _data.dim(); ++j)
101 {
102 if (shape[j] != _base_points[j].size())
103 {
104 std::ostringstream oss;
105 oss << "Dimension " << j << " has " << _base_points[j].size()
106 << " base points but dimension of data array is " << shape[j];
107 throw std::domain_error(oss.str());
108 }
109
110 for (unsigned int i = 1; i < _base_points[j].size(); ++i)
111 {
112 if (_base_points[j][i - 1] >= _base_points[j][i])
113 {
114 std::ostringstream oss;
115 oss << "Base point values for dimension " << j << " are not strictly increasing: bp["
116 << i - 1 << "]: " << _base_points[j][i - 1] << " bc[" << i
117 << "]: " << _base_points[j][i];
118 throw std::domain_error(oss.str());
119 }
120 }
121 }
122
123 // now the setup is complete
124 _setup_complete = true;
125}

127template <typename T>
128void
129MultiDimensionalInterpolationTempl<T>::linearSearch(std::vector<T> & values,
130 MultiIndex<Real>::size_type & indices) const
131{
132 assert(values.size() == _data.dim());
133
134 indices.resize(_data.dim());
135 for (unsigned int d = 0; d < _data.dim(); ++d)
136 indices[d] = linearSearchHelper(values[d], _base_points[d]);
137}

139template <typename T>
140unsigned int
141MultiDimensionalInterpolationTempl<T>::linearSearchHelper(T & x,
142 const std::vector<Real> & vector) const
143{
144// check order of vector only in debug
145#ifndef NDEBUG
146 for (unsigned int i = 1; i < vector.size(); ++i)
147 {
148 if (vector[i - 1] >= vector[i])
149 {
150 std::ostringstream oss;
151 oss << "In linearSearchHelper vector is not strictly increasing: vector[" << i - 1
152 << "]: " << vector[i - 1] << " vector[" << i << "]: " << vector[i];
153 throw std::domain_error(oss.str());
154 }
155 }
156#endif
157
158 // smaller/larger cases for tabulation
159 if (x < vector[0])
160 {
161 x = vector[0];
162 return 0;
163 }
164 else if (x >= vector.back())
165 {
166 // this requires explanation because it looks like a bug (I swear it isn't, it's a feature)
167 // if x >= largest base point value and the index i = vector.size() - 1 is returned, then during
168 // the interpolation, i + 1 will be accessed, causing a segementation fault. Logic in the
169 // interpolation is also bad because it may slow it down. By returning i = vector.size() - 2,
170 // the last two values in the base point array are used, but x is reset to be equal to the last
171 // value. This will give the right interpolation behavior.
172 x = vector.back();
173 return vector.size() - 2;
174 }
175
176 for (unsigned int j = 0; j < vector.size() - 2; ++j)
177 if (x >= vector[j] && x < vector[j + 1])
178 return j;
179 return vector.size() - 2;
180}

182template <typename T>
183T
184MultiDimensionalInterpolationTempl<T>::multiLinearInterpolation(const std::vector<T> & x) const
185{
186 // ensure that object has been set up properly
187 if (!_setup_complete)
188 throw std::domain_error(
189 "Object has been constructed properly. This happens when the empty constructor is called "
190 "and setData is not called before interpolation.");
191
192 // assert does not seem to be enough to check data consistency here
193 // because this function is exposed to user and will be frequently used
194 if (x.size() != _original_dim)
195 {
196 std::ostringstream oss;
197 oss << "In sample the parameter x has size " << x.size() << " but data has dimension "
198 << _data.dim();
199 throw std::domain_error(oss.str());
200 }
201
202 // in this case there is only one entry that resides at flat index 0
203 if (_degenerate_interpolation)
204 return _data[0];
205
206 // make a copy of x because linearSearch needs to be able to modify it
207 // also adjust for degenerate indices that were removed
208 std::vector<T> y(_data.dim());
209 unsigned int l = 0;
210 for (unsigned int j = 0; j < _original_dim; ++j)
211 if (!_degenerate_index[j])
212 {
213 y[l] = x[j];
214 ++l;
215 }
216
217 // obtain the indices using linearSearch & get flat index in _data array
218 MultiIndex<Real>::size_type indices;
219 linearSearch(y, indices);
220
221 // pre-compute the volume of the hypercube and weights used for the interpolation
222 Real volume = 1;
223 std::vector<std::vector<T>> weights(_data.dim());
224 for (unsigned int j = 0; j < _data.dim(); ++j)
225 {
226 volume *= _base_points[j][indices[j] + 1] - _base_points[j][indices[j]];
227
228 // now compute weight for each dimension, note that
229 // weight[j][0] = bp[j][high] - y[j]
230 // weight[j][0] = y[j] - bp[j][low]
231 // because in the interpolation the value on the "left" is weighted with the
232 // distance of y to the "right"
233 weights[j].resize(2);
234 weights[j][0] = _base_points[j][indices[j] + 1] - y[j];
235 weights[j][1] = y[j] - _base_points[j][indices[j]];
236 }
237
238 // get flat index and stride
239 unsigned int flat_index = _data.flatIndex(indices);
240 MultiIndex<Real>::size_type stride = _data.stride();
241
242 // this is the interpolation routine, evaluation can be expensive if dim is large
243
244 // first we retrieve the data around the point y
245 // we use a 2^dim MultiIndex hypercube for generating the permulations
246 MultiIndex<Real>::size_type shape(_data.dim(), 2);
247 MultiIndex<Real> hypercube = MultiIndex<Real>(shape);
248 T interpolated_value = 0;
249 for (const auto & p : hypercube)
250 {
251 // this loop computes the flat index for the point in the
252 // hypercube and accumulates the total weight for this point
253 // the selection of the dimensional weight is based on the swapping
254 // done in the pre-computing loop before
255 unsigned int index = flat_index;
256 T w = 1;
257 for (unsigned int j = 0; j < p.first.size(); ++j)
258 {
259 index += stride[j] * p.first[j];
260 w *= weights[j][p.first[j]];
261 }
262
263 // add the contribution of this base point to the interpolated value
264 interpolated_value += w * _data[index];
265 }
266
267 return interpolated_value / volume;
268}
